36// All ending screens will be drawn in a table/list format, with each player
37// on a row. This is to support any number of players, since there could be
38// many players especially from network multiplayer.
39
40#define PLAYER_LIST_ROW_HEIGHT 16
41typedef struct
42{
43 MenuSystem ms;
44 Vec2i pos;
45 Vec2i size;
46 int scroll;
47 GameLoopResult (*updateFunc)(GameLoopData *, LoopRunner *);
48 void (*drawFunc)(void *);
49 void *data;
50 // Whether to use a confirmation "Finish" menu at the end
51 // Useful for final screens where we want to view the scores without
52 // accidentally quitting
53 bool hasMenu;
54 bool showWinners;
55 bool showLastMan;
56 // Store player UIDs so we can display the list in a certain order
57 CArray playerUIDs; // of int
58} PlayerList;
59static int ComparePlayerScores(const void *v1, const void *v2);
60static PlayerList *PlayerListNew(
61 GameLoopResult (*updateFunc)(GameLoopData *, LoopRunner *),
62 void (*drawFunc)(void *), void *data,
63 const bool hasMenu, const bool showWinners)
64{
65 PlayerList *pl;
66 CMALLOC(pl, sizeof *pl);
67 pl->pos = Vec2iZero();
68 pl->size = gGraphicsDevice.cachedConfig.Res;
69 pl->scroll = 0;
70 pl->updateFunc = updateFunc;
71 pl->drawFunc = drawFunc;
72 pl->data = data;
73 pl->hasMenu = hasMenu;
74 pl->showWinners = showWinners;
75 CArrayInit(&pl->playerUIDs, sizeof(int));
76 // Collect all players, then order by score descending
77 int playersAlive = 0;
78 CA_FOREACH(const PlayerData, p, gPlayerDatas)
79 CArrayPushBack(&pl->playerUIDs, &p->UID);
80 if (p->Lives > 0)
81 {
82 playersAlive++;
83 }
84 CA_FOREACH_END()
85 qsort(
86 pl->playerUIDs.data,
87 pl->playerUIDs.size,
88 pl->playerUIDs.elemSize,
89 ComparePlayerScores);
90 pl->showLastMan = playersAlive == 1;
91 return pl;
92}
93static int GetModeScore(const PlayerData *p)
94{
95 // For deathmatch, we count kills instead of score
96 if (gCampaign.Entry.Mode == GAME_MODE_DEATHMATCH)
97 {
98 return p->Totals.Kills;
99 }
100 return p->Totals.Score;
101}
102static int ComparePlayerScores(const void *v1, const void *v2)
103{
104 const PlayerData *p1 = PlayerDataGetByUID(*(const int *)v1);
105 const PlayerData *p2 = PlayerDataGetByUID(*(const int *)v2);
106 int p1s = GetModeScore(p1);
107 int p2s = GetModeScore(p2);
108 if (p1s > p2s)
109 {
110 return -1;
111 }
112 else if (p1s < p2s)
113 {
114 return 1;
115 }
116 return 0;
117}

126static GameLoopData *PlayerListLoop(PlayerList *pl)
127{
128 MenuSystemInit(&pl->ms, &gEventHandlers, &gGraphicsDevice, pl->pos, pl->size);
129 menu_t *menuScores = MenuCreateCustom(
130 "View Scores", PlayerListCustomDraw, PlayerListInput, pl);
131 if (pl->hasMenu)
132 {
133 pl->ms.root = MenuCreateNormal("", "", MENU_TYPE_NORMAL, 0);
134 MenuAddSubmenu(pl->ms.root, menuScores);
135 MenuAddSubmenu(pl->ms.root, MenuCreateReturn("Finish", 0));
136 }
137 else
138 {
139 pl->ms.root = menuScores;
140 }
141 pl->ms.allowAborts = true;
142 MenuAddExitType(&pl->ms, MENU_TYPE_RETURN);
143 return GameLoopDataNew(
144 pl, PlayerListTerminate, PlayerListOnEnter, PlayerListOnExit,
145 NULL, pl->updateFunc, PlayerListDraw);
146}
147static void PlayerListTerminate(GameLoopData *data)
148{
149 PlayerList *pl = data->Data;
150
151 CArrayTerminate(&pl->playerUIDs);
152 MenuSystemTerminate(&pl->ms);
153 CFREE(pl->data);
154 CFREE(pl);
155}

168}

174static GameLoopResult PlayerListUpdate(GameLoopData *data, LoopRunner *l)
175{
176 PlayerList *pl = data->Data;
177
178 const GameLoopResult result = MenuUpdate(&pl->ms);
179 if (result == UPDATE_RESULT_OK)
180 {
181 LoopRunnerChange(
182 l, HighScoresScreen(&gCampaign, &gGraphicsDevice));
183 }
184 return result;
185}

192static int PlayerListMaxScroll(const PlayerList *pl);
193static int PlayerListMaxRows(const PlayerList *pl);
194static void PlayerListCustomDraw(
195 const menu_t *menu, GraphicsDevice *g, const Vec2i pos, const Vec2i size,
196 const void *data)
197{
198 UNUSED(menu);
199 UNUSED(g);
200 // Draw players starting from the index
201 // TODO: custom columns
202 const PlayerList *pl = data;
203
204 // First draw the headers
205 const int xStart = pos.x + 80 + (size.x - 320) / 2;
206 int x = xStart;
207 int y = pos.y;
208 FontStrMask("Player", Vec2iNew(x, y), colorPurple);
209 x += 100;
210 FontStrMask("Score", Vec2iNew(x, y), colorPurple);
211 x += 32;
212 FontStrMask("Kills", Vec2iNew(x, y), colorPurple);
213 y += FontH() * 2 + PLAYER_LIST_ROW_HEIGHT + 4;
214 // Then draw the player list
215 int maxScore = -1;
216 for (int i = pl->scroll;
217 i < MIN((int)pl->playerUIDs.size, pl->scroll + PlayerListMaxRows(pl));
218 i++)
219 {
220 const int *playerUID = CArrayGet(&pl->playerUIDs, i);
221 PlayerData *p = PlayerDataGetByUID(*playerUID);
222 if (p == NULL)
223 {
224 continue;
225 }
226 if (maxScore < GetModeScore(p))
227 {
228 maxScore = GetModeScore(p);
229 }
230
231 x = xStart;
232 // Highlight local players using different coloured text
233 const color_t textColor = p->IsLocal ? colorPurple : colorWhite;
234
235 // Draw the players offset on alternate rows
236 DisplayCharacterAndName(
237 Vec2iNew(x + (i & 1) * 16, y + 4), &p->Char, DIRECTION_DOWN,
238 p->name, textColor);
239
240 // Draw score
241 x += 100;
242 char buf[256];
243 sprintf(buf, "%d", p->Totals.Score);
244 FontStrMask(buf, Vec2iNew(x, y), textColor);
245
246 // Draw kills
247 x += 32;
248 sprintf(buf, "%d", p->Totals.Kills);
249 FontStrMask(buf, Vec2iNew(x, y), textColor);
250
251 // Draw winner/award text
252 x += 32;
253 if (pl->showWinners && GetModeScore(p) == maxScore)
254 {
255 FontStrMask("Winner!", Vec2iNew(x, y), colorGreen);
256 }
257 else if (pl->showLastMan && p->Lives > 0 &&
258 gCampaign.Entry.Mode == GAME_MODE_DEATHMATCH)
259 {
260 // Only show last man standing on deathmatch mode
261 FontStrMask("Last man standing!", Vec2iNew(x, y), colorGreen);
262 }
263
264 y += PLAYER_LIST_ROW_HEIGHT;
265 }
266
267 // Draw indicator arrows if there's enough to scroll
268 if (pl->scroll > 0)
269 {
270 FontStr("^", Vec2iNew(
271 CENTER_X(pos, size, FontStrW("^")), pos.y + FontH()));
272 }
273 if (pl->scroll < PlayerListMaxScroll(pl))
274 {
275 FontStr("v", Vec2iNew(
276 CENTER_X(pos, size, FontStrW("v")), pos.y + size.y - FontH()));
277 }
278
279 // Finally draw any custom stuff
280 if (pl->drawFunc)
281 {
282 pl->drawFunc(pl->data);
283 }
284}
285static int PlayerListInput(int cmd, void *data)
286{
287 // Input: up/down scrolls list
288 // CMD 1/2: exit
289 PlayerList *pl = data;
290
291 // Note: players can leave due to network disconnection
292 // Update our lists
293 CA_FOREACH(const int, playerUID, pl->playerUIDs)
294 const PlayerData *p = PlayerDataGetByUID(*playerUID);
295 if (p == NULL)
296 {
297 CArrayDelete(&pl->playerUIDs, _ca_index);
298 _ca_index--;
299 }
300 CA_FOREACH_END()
301
302 if (cmd == CMD_DOWN)
303 {
304 SoundPlay(&gSoundDevice, StrSound("door"));
305 pl->scroll++;
306 }
307 else if (cmd == CMD_UP)
308 {
309 SoundPlay(&gSoundDevice, StrSound("door"));
310 pl->scroll--;
311 }
312 else if (AnyButton(cmd))
313 {
314 SoundPlay(&gSoundDevice, StrSound("pickup"));
315 return 1;
316 }
317 // Scroll wrap-around
318 pl->scroll = CLAMP_OPPOSITE(pl->scroll, 0, PlayerListMaxScroll(pl));
319 return 0;
320}
321static int PlayerListMaxScroll(const PlayerList *pl)
322{
323 return MAX((int)pl->playerUIDs.size - PlayerListMaxRows(pl), 0);
324}
325static int PlayerListMaxRows(const PlayerList *pl)
326{
327 return (pl->size.y - FontH() * 3) / PLAYER_LIST_ROW_HEIGHT - 2;
328}

336static void VictoryDraw(void *data);
337GameLoopData *ScreenVictory(CampaignOptions *c)
338{
339 SoundPlay(&gSoundDevice, StrSound("victory"));
340 VictoryData *data;
341 CMALLOC(data, sizeof *data);
342 data->Campaign = c;
343 const char *finalWordsSingle[] = {
344 "Ha, next time I'll use my good hand",
345 "Over already? I was just warming up...",
346 "There's just no good opposition to be found these days!",
347 "Well, maybe I'll just do my monthly reload then",
348 "Woof woof",
349 "I'll just bury the bones in the back yard, he-he",
350 "I just wish they'd let me try bare-handed",
351 "Rambo? Who's Rambo?",
352 "<in Austrian accent:> I'll be back",
353 "Gee, my trigger finger is sore",
354 "I need more practice. I think I missed a few shots at times"
355 };
356 const char *finalWordsMulti[] = {
357 "United we stand, divided we conquer",
358 "Nothing like good teamwork, is there?",
359 "Which way is the camera?",
360 "We eat bullets for breakfast and have grenades as dessert",
361 "We're so cool we have to wear mittens",
362 };
363 if (GetNumPlayers(PLAYER_ANY, false, true) == 1)
364 {
365 const int numWords = sizeof finalWordsSingle / sizeof(char *);
366 data->FinalWords = finalWordsSingle[rand() % numWords];
367 }
368 else
369 {
370 const int numWords = sizeof finalWordsMulti / sizeof(char *);
371 data->FinalWords = finalWordsMulti[rand() % numWords];
372 }
373 PlayerList *pl = PlayerListNew(
374 PlayerListUpdate, VictoryDraw, data, true, false);
375 pl->pos.y = 75;
376 pl->size.y -= pl->pos.y;
377 return PlayerListLoop(pl);
378}
379static void VictoryDraw(void *data)
380{
381 const VictoryData *vd = data;
382
383 const int w = gGraphicsDevice.cachedConfig.Res.x;
384 FontOpts opts = FontOptsNew();
385 opts.HAlign = ALIGN_CENTER;
386 opts.Area = gGraphicsDevice.cachedConfig.Res;
387 int y = 30;
388
389 // Congratulations text
390#define CONGRATULATIONS "Congratulations, you have completed "
391 FontStrOpt(CONGRATULATIONS, Vec2iNew(0, y), opts);
392 y += 15;
393 opts.Mask = colorRed;
394 FontStrOpt(vd->Campaign->Setting.Title, Vec2iNew(0, y), opts);
395 y += 15;
396
397 // Final words
398 Vec2i pos = Vec2iNew((w - FontStrW(vd->FinalWords)) / 2, y);
399 pos = FontChMask('"', pos, colorDarker);
400 pos = FontStrMask(vd->FinalWords, pos, colorPurple);
401 FontChMask('"', pos, colorDarker);
402}

404static GameLoopResult DogfightScoresUpdate(GameLoopData *data, LoopRunner *l);
405GameLoopData *ScreenDogfightScores(void)
406{
407 PlayerList *pl = PlayerListNew(
408 DogfightScoresUpdate, NULL, NULL, false, false);
409 pl->pos.y = 24;
410 pl->size.y -= pl->pos.y;
411 return PlayerListLoop(pl);
412}
413static GameLoopResult DogfightScoresUpdate(GameLoopData *data, LoopRunner *l)
414{
415 PlayerList *pl = data->Data;
416
417 const GameLoopResult result = MenuUpdate(&pl->ms);
418 if (result == UPDATE_RESULT_OK)
419 {
420 // Calculate PVP rounds won
421 int maxScore = 0;
422 CA_FOREACH(PlayerData, p, gPlayerDatas)
423 if (IsPlayerAlive(p))
424 {
425 p->Totals.Score++;
426 maxScore = MAX(maxScore, p->Totals.Score);
427 }
428 CA_FOREACH_END()
429 gCampaign.IsComplete =
430 maxScore == ModeMaxRoundsWon(gCampaign.Entry.Mode);
431 CASSERT(maxScore <= ModeMaxRoundsWon(gCampaign.Entry.Mode),
432 "score exceeds max rounds won");
433 if (gCampaign.IsComplete)
434 {
435 LoopRunnerChange(l, ScreenDogfightFinalScores());
436 }
437 else
438 {
439 LoopRunnerChange(
440 l, HighScoresScreen(&gCampaign, &gGraphicsDevice));
441 }
442 }
443 return result;
444}
