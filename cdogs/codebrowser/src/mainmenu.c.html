63GameLoopData *MainMenu(GraphicsDevice *graphics, LoopRunner *l)
64{
65 MainMenuData *data;
66 CMALLOC(data, sizeof *data);
67 data->graphics = graphics;
68 memset(&data->creditsDisplayer, 0, sizeof data->creditsDisplayer);
69 LoadCredits(&data->creditsDisplayer, colorPurple, colorDarker);
70 memset(&data->campaigns, 0, sizeof data->campaigns);
71 LoadAllCampaigns(&data->campaigns);
72 data->lastGameMode = GAME_MODE_QUICK_PLAY;
73 data->wasClient = false;
74 MenuCreateAll(
75 &data->ms, l, &data->campaigns, &gEventHandlers, data->graphics);
76 MenuSetCreditsDisplayer(&data->ms, &data->creditsDisplayer);
77 return GameLoopDataNew(
78 data, MainMenuTerminate, MainMenuOnEnter, MainMenuOnExit,
79 NULL, MainMenuUpdate, MainMenuDraw);
80}
81static void MainMenuTerminate(GameLoopData *data)
82{
83 MainMenuData *mData = data->Data;
84
85 MenuSystemTerminate(&mData->ms);
86 UnloadCredits(&mData->creditsDisplayer);
87 UnloadAllCampaigns(&mData->campaigns);
88 CFREE(mData);
89}
90static menu_t *FindSubmenuByName(menu_t *menu, const char *name);
91static void MainMenuOnEnter(GameLoopData *data)
92{
93 MainMenuData *mData = data->Data;
94
95 if (gCampaign.IsLoaded)
96 {
97 // Loaded game already; skip menus and go straight to game
98 return;
99 }
100
101 GrafxMakeRandomBackground(
102 &gGraphicsDevice, &gCampaign, &gMission, &gMap);
103 NetClientDisconnect(&gNetClient);
104 NetServerClose(&gNetServer);
105 GameEventsTerminate(&gGameEvents);
106 // Reset config - could have been set to other values by server
107 ConfigResetChanged(&gConfig);
108 CampaignSettingTerminate(&gCampaign.Setting);
109
110 // Auto-enter the submenu corresponding to the last game mode
111 menu_t *startMenu = FindSubmenuByName(mData->ms.root, "Start");
112 if (mData->wasClient)
113 {
114 mData->ms.current = startMenu;
115 }
116 else
117 {
118 switch (mData->lastGameMode)
119 {
120 case GAME_MODE_NORMAL:
121 mData->ms.current = FindSubmenuByName(startMenu, "Campaign");
122 break;
123 case GAME_MODE_DOGFIGHT:
124 mData->ms.current = FindSubmenuByName(startMenu, "Dogfight");
125 break;
126 case GAME_MODE_DEATHMATCH:
127 mData->ms.current = FindSubmenuByName(startMenu, "Deathmatch");
128 break;
129 default:
130 mData->ms.current = mData->ms.root;
131 break;
132 }
133 }
134}
135static menu_t *FindSubmenuByName(menu_t *menu, const char *name)
136{
137 CASSERT(menu->type == MENU_TYPE_NORMAL, "invalid menu type");
138 CA_FOREACH(menu_t, submenu, menu->u.normal.subMenus)
139 if (strcmp(submenu->name, name) == 0) return submenu;
140 CA_FOREACH_END()
141 return menu;
142}
143static void MainMenuOnExit(GameLoopData *data)
144{
145 MainMenuData *mData = data->Data;
146
147 // Reset player datas
148 PlayerDataTerminate(&gPlayerDatas);
149 PlayerDataInit(&gPlayerDatas);
150 // Initialise game events; we need this for init as well as the game
151 GameEventsInit(&gGameEvents);
152
153 mData->lastGameMode = gCampaign.Entry.Mode;
154 mData->wasClient = gCampaign.IsClient;
155}
156static GameLoopResult MainMenuUpdate(GameLoopData *data, LoopRunner *l)
157{
158 MainMenuData *mData = data->Data;
159
160 if (gCampaign.IsLoaded)
161 {
162 // Loaded game already; skip menus and go straight to game
163 LoopRunnerPush(l, ScreenCampaignIntro(&gCampaign.Setting));
164 return UPDATE_RESULT_OK;
165 }
166
167 const GameLoopResult result = MenuUpdate(&mData->ms);
168 if (result == UPDATE_RESULT_OK)
169 {
170 if (gCampaign.IsLoaded)
171 {
172 LoopRunnerPush(l, ScreenCampaignIntro(&gCampaign.Setting));
173 }
174 else
175 {
176 LoopRunnerPop(l);
177 }
178 }
179 return result;
180}
181static void MainMenuDraw(GameLoopData *data)
182{
183 MainMenuData *mData = data->Data;
184 MenuDraw(&mData->ms);
185}

193static void MenuCreateAll(
194 MenuSystem *ms, LoopRunner *l,
195 custom_campaigns_t *campaigns,
196 EventHandlers *handlers,
197 GraphicsDevice *graphics)
198{
199 MenuSystemInit(
200 ms,
201 handlers, graphics,
202 Vec2iZero(),
203 Vec2iNew(
204 graphics->cachedConfig.Res.x,
205 graphics->cachedConfig.Res.y));
206 ms->root = ms->current = MenuCreateNormal(
207 "",
208 "",
209 MENU_TYPE_NORMAL,
210 MENU_DISPLAY_ITEMS_CREDITS | MENU_DISPLAY_ITEMS_AUTHORS);
211 MenuAddSubmenu(ms->root, MenuCreateStart("Start", ms, l, campaigns));
212 MenuAddSubmenu(ms->root, MenuCreateOptions("Options...", ms));
213#ifndef __EMSCRIPTEN__
214 MenuAddSubmenu(ms->root, MenuCreateQuit("Quit"));
215 MenuAddExitType(ms, MENU_TYPE_QUIT);
216#endif
217 MenuAddExitType(ms, MENU_TYPE_RETURN);
218}

234static menu_t *MenuCreateStart(
235 const char *name, MenuSystem *ms, LoopRunner *l,
236 custom_campaigns_t *campaigns)
237{
238 menu_t *menu = MenuCreateNormal(name, "Start:", MENU_TYPE_NORMAL, 0);
239 MenuAddSubmenu(
240 menu,
241 MenuCreateContinue("Continue", &gAutosave.LastMission.Campaign));
242 const int menuContinueIndex = (int)menu->u.normal.subMenus.size - 1;
243 MenuAddSubmenu(
244 menu,
245 MenuCreateQuickPlay("Quick Play", &campaigns->quickPlayEntry));
246 MenuAddSubmenu(
247 menu,
248 MenuCreateCampaigns(
249 "Campaign",
250 "Select a campaign:",
251 &campaigns->campaignList,
252 GAME_MODE_NORMAL));
253 MenuAddSubmenu(
254 menu,
255 MenuCreateCampaigns(
256 "Dogfight",
257 "Select a scenario:",
258 &campaigns->dogfightList,
259 GAME_MODE_DOGFIGHT));
260 MenuAddSubmenu(
261 menu,
262 MenuCreateCampaigns(
263 "Deathmatch",
264 "Select a scenario:",
265 &campaigns->dogfightList,
266 GAME_MODE_DEATHMATCH));
267 MenuAddSubmenu(
268 menu, CreateJoinLANGame("Join LAN game", "Choose LAN server", ms, l));
269 CheckLANServerData *cdata;
270 CMALLOC(cdata, sizeof *cdata);
271 cdata->MenuJoinIndex = (int)menu->u.normal.subMenus.size - 1;
272 MenuAddSubmenu(menu, MenuCreateSeparator(""));
273 MenuAddSubmenu(menu, MenuCreateBack("Back"));
274
275 if (strlen(gAutosave.LastMission.Password) == 0 ||
276 !gAutosave.LastMission.IsValid)
277 {
278 MenuDisableSubmenu(menu, menuContinueIndex);
279 }
280
281 MenuDisableSubmenu(menu, cdata->MenuJoinIndex);
282 // Periodically check if LAN servers are available
283 MenuSetPostUpdateFunc(menu, CheckLANServers, cdata, true);
284
285 return menu;
286}

307static void StartGameMode(menu_t *menu, void *data)
308{
309 UNUSED(menu);
310 StartGameModeData *mData = data;
311 gCampaign.Entry.Mode = mData->GameMode;
312 debug(D_NORMAL, "Starting game mode %s %d\n",
313 mData->Entry->Path, (int)mData->GameMode);
314 if (!CampaignLoad(&gCampaign, mData->Entry))
315 {
316 // Failed to load
317 printf("Error: cannot load campaign %s\n", mData->Entry->Info);
318 }
319}

332static void CampaignsDisplayFilename(
333 const menu_t *menu, GraphicsDevice *g,
334 const Vec2i pos, const Vec2i size, const void *data)
335{
336 const menu_t *subMenu =
337 CArrayGet(&menu->u.normal.subMenus, menu->u.normal.index);
338 UNUSED(g);
339 UNUSED(data);
340 if (subMenu->type != MENU_TYPE_BASIC ||
341 subMenu->customPostInputData == NULL)
342 {
343 return;
344 }
345 const StartGameModeData *mData = subMenu->customPostInputData;
346 char s[CDOGS_FILENAME_MAX];
347 sprintf(s, "( %s )", mData->Entry->Filename);
348
349 FontOpts opts = FontOptsNew();
350 opts.HAlign = ALIGN_CENTER;
351 opts.VAlign = ALIGN_END;
352 opts.Area = size;
353 opts.Pad.x = size.x / 12;
354 FontStrOpt(s, pos, opts);
355}
356static menu_t *MenuCreateCampaigns(
357 const char *name, const char *title,
358 campaign_list_t *list, const GameMode mode)
359{
360 menu_t *menu = MenuCreateNormal(name, title, MENU_TYPE_NORMAL, 0);
361 menu->u.normal.maxItems = 20;
362 menu->u.normal.align = MENU_ALIGN_CENTER;
363 CA_FOREACH(campaign_list_t, subList, list->subFolders)
364 char folderName[CDOGS_FILENAME_MAX];
365 sprintf(folderName, "%s/", subList->Name);
366 MenuAddSubmenu(
367 menu, MenuCreateCampaigns(folderName, title, subList, mode));
368 CA_FOREACH_END()
369 CA_FOREACH(CampaignEntry, e, list->list)
370 MenuAddSubmenu(menu, MenuCreateCampaignItem(e, mode));
371 CA_FOREACH_END()
372 MenuSetCustomDisplay(menu, CampaignsDisplayFilename, NULL);
373 return menu;
374}

376static menu_t *MenuCreateCampaignItem(
377 CampaignEntry *entry, const GameMode mode)
378{
379 menu_t *menu = CreateStartGameMode(entry->Info, mode, entry);
380 // Special colors:
381 // - Green for new campaigns
382 // - White (normal) for in-progress campaigns
383 // - Grey for complete campaigns
384 MissionSave m;
385 AutosaveLoadMission(&gAutosave, &m, entry->Path);
386 if (m.MissionsCompleted == entry->NumMissions)
387 {
388 // Completed campaign
389 menu->color = colorGray;
390 }
391 else if (m.MissionsCompleted > 0)
392 {
393 // Campaign in progress
394 menu->color = colorYellow;
395 }
396
397 return menu;
398}

406static menu_t *CreateJoinLANGame(
407 const char *name, const char *title, MenuSystem *ms, LoopRunner *l)
408{
409 menu_t *menu = MenuCreateNormal(name, title, MENU_TYPE_NORMAL, 0);
410 // We'll create our menu items dynamically after entering
411 // Creating an item for each scanned server address
412 CreateJoinLANGameData *data;

414 data->MS = ms;
415 data->L = l;
416 MenuSetPostEnterFunc(menu, CreateLANServerMenuItems, data, true);
417 return menu;

419typedef struct
420{
421 MenuSystem *MS;
422 LoopRunner *L;
423 int AddrIndex;
424} JoinLANGameData;
425static void JoinLANGame(menu_t *menu, void *data);
426static void CreateLANServerMenuItems(menu_t *menu, void *data)
427{
428 CreateJoinLANGameData *cData = data;
429
430 // Clear and recreate all menu items
431 MenuClearSubmenus(menu);
432 CA_FOREACH(ScanInfo, si, gNetClient.ScannedAddrs)
433 char buf[512];
434 char ipbuf[256];
435 if (enet_address_get_host_ip(&si->Addr, ipbuf, sizeof ipbuf) < 0)
436 {
437 LOG(LM_MAIN, LL_WARN, "cannot find host ip");
438 ipbuf[0] = '?';
439 ipbuf[1] = '\0';
440 };
441 // e.g. "Bob's Server (123.45.67.89:12345) - Campaign: Ogre Rampage #4, p: 4/16 350ms"
442 sprintf(buf, "%s (%s:%u) - %s: %s (# %d), p: %d/%d %dms",
443 si->ServerInfo.Hostname, ipbuf, si->Addr.port,
444 GameModeStr(si->ServerInfo.GameMode),
445 si->ServerInfo.CampaignName, si->ServerInfo.MissionNumber,
446 si->ServerInfo.NumPlayers, si->ServerInfo.MaxPlayers,
447 si->LatencyMS);
448 menu_t *serverMenu = MenuCreate(buf, MENU_TYPE_RETURN);
449 serverMenu->enterSound = MENU_SOUND_START;
450 JoinLANGameData *jdata;
451 CMALLOC(jdata, sizeof *jdata);
452 jdata->MS = cData->MS;
453 jdata->L = cData->L;
454 jdata->AddrIndex = _ca_index;
455 MenuSetPostEnterFunc(serverMenu, JoinLANGame, jdata, true);
456 MenuAddSubmenu(menu, serverMenu);
457 CA_FOREACH_END()
458 MenuAddSubmenu(menu, MenuCreateSeparator(""));
459 MenuAddSubmenu(menu, MenuCreateBack("Back"));
460}
461static void JoinLANGame(menu_t *menu, void *data)
462{
463 JoinLANGameData *jdata = data;
464 if (jdata->AddrIndex >= (int)gNetClient.ScannedAddrs.size)
465 {
466 goto bail;
467 }
468 const ScanInfo *sinfo = CArrayGet(
469 &gNetClient.ScannedAddrs, jdata->AddrIndex);
470 LOG(LM_MAIN, LL_INFO, "joining LAN game...");
471 if (NetClientTryConnect(&gNetClient, sinfo->Addr))
472 {
473 LoopRunnerPush(jdata->L, ScreenWaitForCampaignDef());
474 goto bail;
475 }
476 else
477 {
478 LOG(LM_MAIN, LL_INFO, "failed to connect to LAN game");
479 }
480 return;
481
482bail:
483 // Don't activate the menu item
484 jdata->MS->current = menu->parentMenu;
485}
486static void CheckLANServers(menu_t *menu, void *data)
487{
488 CheckLANServerData *cdata = data;
489 if (gNetClient.ScannedAddrs.size > 0)
490 {
491 MenuEnableSubmenu(menu, cdata->MenuJoinIndex);
492 }
493 else
494 {
495 // We haven't found any LAN servers in the latest scan
496 MenuDisableSubmenu(menu, cdata->MenuJoinIndex);
497 }
498 if (gNetClient.ScanTicks <= 0)
499 {
500 LOG(LM_MAIN, LL_DEBUG, "finding LAN server...");
501 NetClientFindLANServers(&gNetClient);
502 }
503}
