41void LoadCredits(
42 credits_displayer_t *displayer, color_t nameColor, color_t textColor)
43{
44 char buf[CDOGS_PATH_MAX];
45 int nameOrMessageCounter = 0;
46 FILE *file;
47
48 debug(D_NORMAL, "Reading CREDITS...\n");
49
50 GetDataFilePath(buf, "doc/CREDITS");
51 file = fopen(buf, "r");
52 if (file == NULL)
53 {
54 printf("Error: cannot load %s\n", buf);
55 return;
56 }
57
58 assert(displayer != NULL);
59
60 CArrayInit(&displayer->credits, sizeof(credit_t));
61 displayer->lastUpdateTime = time(NULL);
62 displayer->creditsIndex = 0;
63 displayer->nameColor = nameColor;
64 displayer->textColor = textColor;
65 while (fgets(buf, CDOGS_PATH_MAX, file) != NULL)
66 {
67 char nameBuf[256];
68 if (strlen(buf) > 0 && buf[strlen(buf) - 1] == '\n')
69 {
70 buf[strlen(buf) - 1] = '\0';
71 }
72 if (strlen(buf) == 0)
73 {
74 nameOrMessageCounter = 0;
75 continue;
76 }
77 if (nameOrMessageCounter == 0)
78 {
79 strcpy(nameBuf, buf);
80 nameOrMessageCounter = 1;
81 }
82 else
83 {
84 credit_t credit;
85 CSTRDUP(credit.name, nameBuf);
86 CSTRDUP(credit.message, buf + 1);
87 CArrayPushBack(&displayer->credits, &credit);
88 nameOrMessageCounter = 0;
89
90 debug(D_VERBOSE, "Read credits for \"%s\"\n", credit.name);
91 }
92 }
93
94 debug(D_NORMAL, "%d credits read\n", (int)displayer->credits.size);
95
96 fclose(file);
97}

99void UnloadCredits(credits_displayer_t *displayer)
100{
101 if (displayer == NULL)
102 {
103 return;
104 }
105 CA_FOREACH(credit_t, credit, displayer->credits)
106 CFREE(credit->name);
107 CFREE(credit->message);
108 CA_FOREACH_END()
109 CArrayTerminate(&displayer->credits);
110 displayer->creditsIndex = 0;
111}

113#define CREDIT_DISPLAY_PERIOD_SECONDS 10.0
114void ShowCredits(credits_displayer_t *displayer)
115{
116 CASSERT(displayer != NULL, "null pointer");
117 if ((int)displayer->credits.size > 0)
118 {
119 time_t now = time(NULL);
120 credit_t *credits =
121 CArrayGet(&displayer->credits, displayer->creditsIndex);
122 int y = gGraphicsDevice.cachedConfig.Res.y - 50;
123
124 FontStrMask("Credits:", Vec2iNew(16, y), displayer->textColor);
125 y += FontH() + 2;
126 FontStrMask(credits->name, Vec2iNew(20, y), displayer->nameColor);
127 y += FontH();
128 FontStrMask(credits->message, Vec2iNew(20, y), displayer->textColor);
129
130 if (difftime(now, displayer->lastUpdateTime) > CREDIT_DISPLAY_PERIOD_SECONDS)
131 {
132 displayer->creditsIndex++;
133 if (displayer->creditsIndex >= (int)displayer->credits.size)
134 {
135 displayer->creditsIndex = 0;
136 }
137 displayer->lastUpdateTime = now;
138 }
139 }
