170_TINYDIR_FUNC
171int tinydir_open(tinydir_dir *dir, const char *path)
172{
173#ifndef _MSC_VER
174#ifndef _TINYDIR_USE_READDIR
175 int error;
176 int size; /* using int size */
177#endif
178#else
179 char path_buf[_TINYDIR_PATH_MAX];
180#endif
181 char *pathp;
182
183 if (dir == NULL || path == NULL || strlen(path) == 0)
184 {
185 errno = EINVAL;
186 return -1;
187 }
188 if (strlen(path) + _TINYDIR_PATH_EXTRA >= _TINYDIR_PATH_MAX)
189 {
190 errno = ENAMETOOLONG;
191 return -1;
192 }
193
194 /* initialise dir */
195 dir->_files = NULL;
196#ifdef _MSC_VER
197 dir->_h = INVALID_HANDLE_VALUE;
198#else
199 dir->_d = NULL;
200#ifndef _TINYDIR_USE_READDIR
201 dir->_ep = NULL;
202#endif
203#endif
204 tinydir_close(dir);
205
206 strcpy(dir->path, path);
207 /* Remove trailing slashes */
208 pathp = &dir->path[strlen(dir->path) - 1];
209 while (pathp != dir->path && (*pathp == '\\' || *pathp == '/'))
210 {
211 *pathp = '\0';
212 pathp++;
213 }
214#ifdef _MSC_VER
215 strcpy(path_buf, dir->path);
216 strcat(path_buf, "\\*");
217 dir->_h = FindFirstFileA(path_buf, &dir->_f);
218 if (dir->_h == INVALID_HANDLE_VALUE)
219 {
220 errno = ENOENT;
221#else
222 dir->_d = opendir(path);
223 if (dir->_d == NULL)
224 {
225#endif
226 goto bail;
227 }
228
229 /* read first file */
230 dir->has_next = 1;
231#ifndef _MSC_VER
232#ifdef _TINYDIR_USE_READDIR
233 dir->_e = readdir(dir->_d);
234#else
235 /* allocate dirent buffer for readdir_r */
236 size = _tinydir_dirent_buf_size(dir->_d); /* conversion to int */
237 if (size == -1) return -1;
238 dir->_ep = (struct dirent*)_TINYDIR_MALLOC(size);
239 if (dir->_ep == NULL) return -1;
240
241 error = readdir_r(dir->_d, dir->_ep, &dir->_e);
242 if (error != 0) return -1;
243#endif
244 if (dir->_e == NULL)
245 {
246 dir->has_next = 0;
247 }
248#endif
249
250 return 0;
251
252bail:
253 tinydir_close(dir);
254 return -1;
255}

257_TINYDIR_FUNC
258int tinydir_open_sorted(tinydir_dir *dir, const char *path)
259{
260 /* Count the number of files first, to pre-allocate the files array */
261 size_t n_files = 0;
262 if (tinydir_open(dir, path) == -1)
263 {
264 return -1;
265 }
266 while (dir->has_next)
267 {
268 n_files++;
269 if (tinydir_next(dir) == -1)
270 {
271 goto bail;
272 }
273 }
274 tinydir_close(dir);
275
276 if (tinydir_open(dir, path) == -1)
277 {
278 return -1;
279 }
280
281 dir->n_files = 0;
282 dir->_files = (tinydir_file *)_TINYDIR_MALLOC(sizeof *dir->_files * n_files);
283 if (dir->_files == NULL)
284 {
285 goto bail;
286 }
287 while (dir->has_next)
288 {
289 tinydir_file *p_file;
290 dir->n_files++;
291
292 p_file = &dir->_files[dir->n_files - 1];
293 if (tinydir_readfile(dir, p_file) == -1)
294 {
295 goto bail;
296 }
297
298 if (tinydir_next(dir) == -1)
299 {
300 goto bail;
301 }
302
303 /* Just in case the number of files has changed between the first and
304 second reads, terminate without writing into unallocated memory */
305 if (dir->n_files == n_files)
306 {
307 break;
308 }
309 }
310
311 qsort(dir->_files, dir->n_files, sizeof(tinydir_file), _tinydir_file_cmp);
312
313 return 0;
314
315bail:
316 tinydir_close(dir);
317 return -1;
318}

320_TINYDIR_FUNC
321void tinydir_close(tinydir_dir *dir)
322{
323 if (dir == NULL)
324 {
325 return;
326 }
327
328 memset(dir->path, 0, sizeof(dir->path));
329 dir->has_next = 0;
330 dir->n_files = 0;
331 _TINYDIR_FREE(dir->_files);
332 dir->_files = NULL;
333#ifdef _MSC_VER
334 if (dir->_h != INVALID_HANDLE_VALUE)
335 {
336 FindClose(dir->_h);
337 }
338 dir->_h = INVALID_HANDLE_VALUE;
339#else
340 if (dir->_d)
341 {
342 closedir(dir->_d);
343 }
344 dir->_d = NULL;
345 dir->_e = NULL;
346#ifndef _TINYDIR_USE_READDIR
347 _TINYDIR_FREE(dir->_ep);
348 dir->_ep = NULL;
349#endif
350#endif
351}

353_TINYDIR_FUNC
354int tinydir_next(tinydir_dir *dir)
355{
356 if (dir == NULL)
357 {
358 errno = EINVAL;
359 return -1;
360 }
361 if (!dir->has_next)
362 {
363 errno = ENOENT;
364 return -1;
365 }
366
367#ifdef _MSC_VER
368 if (FindNextFileA(dir->_h, &dir->_f) == 0)
369#else
370#ifdef _TINYDIR_USE_READDIR
371 dir->_e = readdir(dir->_d);
372#else
373 if (dir->_ep == NULL)
374 {
375 return -1;
376 }
377 if (readdir_r(dir->_d, dir->_ep, &dir->_e) != 0)
378 {
379 return -1;
380 }
381#endif
382 if (dir->_e == NULL)
383#endif
384 {
385 dir->has_next = 0;
386#ifdef _MSC_VER
387 if (GetLastError() != ERROR_SUCCESS &&
388 GetLastError() != ERROR_NO_MORE_FILES)
389 {
390 tinydir_close(dir);
391 errno = EIO;
392 return -1;
393 }
394#endif
395 }
396
397 return 0;
398}

400_TINYDIR_FUNC
401int tinydir_readfile(const tinydir_dir *dir, tinydir_file *file)
402{
403 if (dir == NULL || file == NULL)
404 {
405 errno = EINVAL;
406 return -1;
407 }
408#ifdef _MSC_VER
409 if (dir->_h == INVALID_HANDLE_VALUE)
410#else
411 if (dir->_e == NULL)
412#endif
413 {
414 errno = ENOENT;
415 return -1;
416 }
417 if (strlen(dir->path) +
418 strlen(
419#ifdef _MSC_VER
420 dir->_f.cFileName
421#else
422 dir->_e->d_name
423#endif
424 ) + 1 + _TINYDIR_PATH_EXTRA >=
425 _TINYDIR_PATH_MAX)
426 {
427 /* the path for the file will be too long */
428 errno = ENAMETOOLONG;
429 return -1;
430 }
431 if (strlen(
432#ifdef _MSC_VER
433 dir->_f.cFileName
434#else
435 dir->_e->d_name
436#endif
437 ) >= _TINYDIR_FILENAME_MAX)
438 {
439 errno = ENAMETOOLONG;
440 return -1;
441 }
442
443 strcpy(file->path, dir->path);
444 strcat(file->path, "/");
445 strcpy(file->name,
446#ifdef _MSC_VER
447 dir->_f.cFileName
448#else
449 dir->_e->d_name
450#endif
451 );
452 strcat(file->path, file->name);
453#ifndef _MSC_VER
454 if (stat(file->path, &file->_s) == -1)
455 {
456 return -1;
457 }
458#endif
459 _tinydir_get_ext(file);
460
461 file->is_dir =
462#ifdef _MSC_VER
463 !!(dir->_f.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY);
464#else
465 S_ISDIR(file->_s.st_mode);
466#endif
467 file->is_reg =
468#ifdef _MSC_VER
469 !!(dir->_f.dwFileAttributes & FILE_ATTRIBUTE_NORMAL) ||
470 (
471 !(dir->_f.dwFileAttributes & FILE_ATTRIBUTE_DEVICE) &&
472 !(dir->_f.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY) &&
473 !(dir->_f.dwFileAttributes & FILE_ATTRIBUTE_ENCRYPTED) &&
474#ifdef FILE_ATTRIBUTE_INTEGRITY_STREAM
475 !(dir->_f.dwFileAttributes & FILE_ATTRIBUTE_INTEGRITY_STREAM) &&
476#endif
477#ifdef FILE_ATTRIBUTE_NO_SCRUB_DATA
478 !(dir->_f.dwFileAttributes & FILE_ATTRIBUTE_NO_SCRUB_DATA) &&
479#endif
480 !(dir->_f.dwFileAttributes & FILE_ATTRIBUTE_OFFLINE) &&
481 !(dir->_f.dwFileAttributes & FILE_ATTRIBUTE_TEMPORARY));
482#else
483 S_ISREG(file->_s.st_mode);
484#endif
485
486 return 0;
487}

489_TINYDIR_FUNC
490int tinydir_readfile_n(const tinydir_dir *dir, tinydir_file *file, size_t i)
491{
492 if (dir == NULL || file == NULL)
493 {
494 errno = EINVAL;
495 return -1;
496 }
497 if (i >= dir->n_files)
498 {
499 errno = ENOENT;
500 return -1;
501 }
502
503 memcpy(file, &dir->_files[i], sizeof(tinydir_file));
504 _tinydir_get_ext(file);
505
506 return 0;
507}

509_TINYDIR_FUNC
510int tinydir_open_subdir_n(tinydir_dir *dir, size_t i)
511{
512 char path[_TINYDIR_PATH_MAX];
513 if (dir == NULL)
514 {
515 errno = EINVAL;
516 return -1;
517 }
518 if (i >= dir->n_files || !dir->_files[i].is_dir)
519 {
520 errno = ENOENT;
521 return -1;
522 }
523
524 strcpy(path, dir->_files[i].path);
525 tinydir_close(dir);
526 if (tinydir_open_sorted(dir, path) == -1)
527 {
528 return -1;
529 }
530
531 return 0;
532}

534/* Open a single file given its path */
535_TINYDIR_FUNC
536int tinydir_file_open(tinydir_file *file, const char *path)
537{
538 tinydir_dir dir;
539 int result = 0;
540 int found = 0;
541 char dir_name_buf[_TINYDIR_PATH_MAX];
542 char file_name_buf[_TINYDIR_FILENAME_MAX];
543 char *dir_name;
544 char *base_name;
545#ifdef _MSC_VER
546 char drive_buf[_TINYDIR_PATH_MAX];
547 char ext_buf[_TINYDIR_FILENAME_MAX];
548#endif
549
550 if (file == NULL || path == NULL || strlen(path) == 0)
551 {
552 errno = EINVAL;
553 return -1;
554 }
555 if (strlen(path) + _TINYDIR_PATH_EXTRA >= _TINYDIR_PATH_MAX)
556 {
557 errno = ENAMETOOLONG;
558 return -1;
559 }
560
561 /* Get the parent path */
562#ifdef _MSC_VER
563 if (_splitpath_s(
564 path,
565 drive_buf, sizeof drive_buf,
566 dir_name_buf, sizeof dir_name_buf,
567 file_name_buf, sizeof file_name_buf,
568 ext_buf, sizeof ext_buf))
569 {
570 errno = EINVAL;
571 return -1;
572 }
573 /* Concatenate the drive letter and dir name to form full dir name */
574 strcat(drive_buf, dir_name_buf);
575 dir_name = drive_buf;
576 /* Concatenate the file name and extension to form base name */
577 strcat(file_name_buf, ext_buf);
578 base_name = file_name_buf;
579#else
580 strcpy(dir_name_buf, path);
581 dir_name = dirname(dir_name_buf);
582 strcpy(file_name_buf, path);
583 base_name = basename(file_name_buf);
584#endif
585
586 /* Open the parent directory */
587 if (tinydir_open(&dir, dir_name) == -1)
588 {
589 return -1;
590 }
591
592 /* Read through the parent directory and look for the file */
593 while (dir.has_next)
594 {
595 if (tinydir_readfile(&dir, file) == -1)
596 {
597 result = -1;
598 goto bail;
599 }
600 if (strcmp(file->name, base_name) == 0)
601 {
602 /* File found */
603 found = 1;
604 break;
605 }
606 tinydir_next(&dir);
607 }
608 if (!found)
609 {
610 result = -1;
611 errno = ENOENT;
612 }
613
614bail:
615 tinydir_close(&dir);
616 return result;
617}

619_TINYDIR_FUNC
620void _tinydir_get_ext(tinydir_file *file)
621{
622 char *period = strrchr(file->name, '.');
623 if (period == NULL)
624 {
625 file->extension = &(file->name[strlen(file->name)]);
626 }
627 else
628 {
629 file->extension = period + 1;
630 }
631}

633_TINYDIR_FUNC
634int _tinydir_file_cmp(const void *a, const void *b)
635{
636 const tinydir_file *fa = (const tinydir_file *)a;
637 const tinydir_file *fb = (const tinydir_file *)b;
638 if (fa->is_dir != fb->is_dir)
639 {
640 return -(fa->is_dir - fb->is_dir);
641 }
642 return strncmp(fa->name, fb->name, _TINYDIR_FILENAME_MAX);
643}

647/*
648The following authored by Ben Hutchings <ben@decadent.org.uk>
649from https://womble.decadent.org.uk/readdir_r-advisory.html
650*/
651/* Calculate the required buffer size (in bytes) for directory *
652* entries read from the given directory handle. Return -1 if this *
653* this cannot be done. *
654* *
655* This code does not trust values of NAME_MAX that are less than *
656* 255, since some systems (including at least HP-UX) incorrectly *
657* define it to be a smaller value. */
658_TINYDIR_FUNC
659size_t _tinydir_dirent_buf_size(DIR *dirp)
660{
661 long name_max;
662 size_t name_end;
663 /* parameter may be unused */
664 (void)dirp;
665
666#if defined _TINYDIR_USE_FPATHCONF
667 name_max = fpathconf(dirfd(dirp), _PC_NAME_MAX);
668 if (name_max == -1)
669#if defined(NAME_MAX)
670 name_max = (NAME_MAX > 255) ? NAME_MAX : 255;
671#else
672 return (size_t)(-1);
673#endif
674#elif defined(NAME_MAX)
675 name_max = (NAME_MAX > 255) ? NAME_MAX : 255;
676#else
677#error "buffer size for readdir_r cannot be determined"
678#endif
679 name_end = (size_t)offsetof(struct dirent, d_name) + name_max + 1;
680 return (name_end > sizeof(struct dirent) ?
681 name_end : sizeof(struct dirent));
682}
683#endif
684#endif
