73static bool IsFacingPlayer(TActor *actor, direction_e d)
74{
75 CA_FOREACH(const PlayerData, p, gPlayerDatas)
76 if (!IsPlayerAlive(p))
77 {
78 continue;
79 }
80 const TActor *player = ActorGetByUID(p->ActorUID);
81 if (AIIsFacing(actor, player->Pos, d))
82 {
83 return true;
84 }
85 CA_FOREACH_END()
86 return false;
87}

90#define Distance(a,b) CHEBYSHEV_DISTANCE(a->x, a->y, b->x, b->y)

92static bool IsCloseToPlayer(const Vec2i fullPos, const int fullDistance)
93{
94 TActor *closestPlayer = AIGetClosestPlayer(fullPos);
95 return closestPlayer && CHEBYSHEV_DISTANCE(
96 fullPos.x, fullPos.y,
97 closestPlayer->Pos.x, closestPlayer->Pos.y) < fullDistance;
98}

100static bool CanSeeAPlayer(const TActor *a)
101{
102 const Vec2i realPos = Vec2iFull2Real(a->Pos);
103 CA_FOREACH(const PlayerData, p, gPlayerDatas)
104 if (!IsPlayerAlive(p))
105 {
106 continue;
107 }
108 const TActor *player = ActorGetByUID(p->ActorUID);
109 const Vec2i playerRealPos = Vec2iFull2Real(player->Pos);
110 // Can see player if:
111 // - Clear line of sight, and
112 // - If they are close, or if facing and they are not too far
113 if (!AIHasClearShot(realPos, playerRealPos))
114 {
115 continue;
116 }
117 const int distance = CHEBYSHEV_DISTANCE(
118 realPos.x, realPos.y, playerRealPos.x, playerRealPos.y);
119 const bool isClose = distance < 16 * 4;
120 const bool isNotTooFar = distance < 16 * 30;
121 if (isClose ||
122 (isNotTooFar && AIIsFacing(a, player->Pos, a->direction)))
123 {
124 return true;
125 }
126 CA_FOREACH_END()
127 return false;
128}

131static bool IsPosOK(TActor *actor, Vec2i pos)
132{
133 const Vec2i realPos = Vec2iFull2Real(pos);
134 if (IsCollisionDiamond(&gMap, realPos, actor->tileItem.size))
135 {
136 return false;
137 }
138 const CollisionParams params =
139 {
140 TILEITEM_IMPASSABLE, CalcCollisionTeam(true, actor),
141 IsPVP(gCampaign.Entry.Mode)
142 };
143 if (OverlapGetFirstItem(
144 &actor->tileItem, pos, actor->tileItem.size, params))
145 {
146 return false;
147 }
148 return true;
149}

151#define STEPSIZE 1024
152
153static bool IsDirectionOK(TActor *a, int dir)
154{
155 switch (dir) {
156 case DIRECTION_UP:
157 return IsPosOK(a, Vec2iAdd(a->Pos, Vec2iNew(0, -STEPSIZE)));
158 case DIRECTION_UPLEFT:
159 return
160 IsPosOK(a, Vec2iAdd(a->Pos, Vec2iNew(-STEPSIZE, -STEPSIZE))) ||
161 IsPosOK(a, Vec2iAdd(a->Pos, Vec2iNew(-STEPSIZE, 0))) ||
162 IsPosOK(a, Vec2iAdd(a->Pos, Vec2iNew(0, -STEPSIZE)));
163 case DIRECTION_LEFT:
164 return IsPosOK(a, Vec2iAdd(a->Pos, Vec2iNew(-STEPSIZE, 0)));
165 case DIRECTION_DOWNLEFT:
166 return
167 IsPosOK(a, Vec2iAdd(a->Pos, Vec2iNew(-STEPSIZE, STEPSIZE))) ||
168 IsPosOK(a, Vec2iAdd(a->Pos, Vec2iNew(-STEPSIZE, 0))) ||
169 IsPosOK(a, Vec2iAdd(a->Pos, Vec2iNew(0, STEPSIZE)));
170 case DIRECTION_DOWN:
171 return IsPosOK(a, Vec2iAdd(a->Pos, Vec2iNew(0, STEPSIZE)));
172 case DIRECTION_DOWNRIGHT:
173 return
174 IsPosOK(a, Vec2iAdd(a->Pos, Vec2iNew(STEPSIZE, STEPSIZE))) ||
175 IsPosOK(a, Vec2iAdd(a->Pos, Vec2iNew(STEPSIZE, 0))) ||
176 IsPosOK(a, Vec2iAdd(a->Pos, Vec2iNew(0, STEPSIZE)));
177 case DIRECTION_RIGHT:
178 return IsPosOK(a, Vec2iAdd(a->Pos, Vec2iNew(STEPSIZE, 0)));
179 case DIRECTION_UPRIGHT:
180 return
181 IsPosOK(a, Vec2iAdd(a->Pos, Vec2iNew(STEPSIZE, -STEPSIZE))) ||
182 IsPosOK(a, Vec2iAdd(a->Pos, Vec2iNew(STEPSIZE, 0))) ||
183 IsPosOK(a, Vec2iAdd(a->Pos, Vec2iNew(0, -STEPSIZE)));
184 }
185 return 0;
186}

189static int BrightWalk(TActor * actor, int roll)
190{
191 const CharBot *bot = ActorGetCharacter(actor)->bot;
192 if (!!(actor->flags & FLAGS_VISIBLE) && roll < bot->probabilityToTrack)
193 {
194 actor->flags &= ~FLAGS_DETOURING;
195 return AIHuntClosest(actor);
196 }
197
198 if (actor->flags & FLAGS_TRYRIGHT)
199 {
200 if (IsDirectionOK(actor, (actor->direction + 7) % 8))
201 {
202 actor->direction = (actor->direction + 7) % 8;
203 actor->turns--;
204 if (actor->turns == 0)
205 {
206 actor->flags &= ~FLAGS_DETOURING;
207 }
208 }
209 else if (!IsDirectionOK(actor, actor->direction))
210 {
211 actor->direction = (actor->direction + 1) % 8;
212 actor->turns++;
213 if (actor->turns == 4) {
214 actor->flags &=
215 ~(FLAGS_DETOURING | FLAGS_TRYRIGHT);
216 actor->turns = 0;
217 }
218 }
219 }
220 else
221 {
222 if (IsDirectionOK(actor, (actor->direction + 1) % 8))
223 {
224 actor->direction = (actor->direction + 1) % 8;
225 actor->turns--;
226 if (actor->turns == 0)
227 actor->flags &= ~FLAGS_DETOURING;
228 }
229 else if (!IsDirectionOK(actor, actor->direction))
230 {
231 actor->direction = (actor->direction + 7) % 8;
232 actor->turns++;
233 if (actor->turns == 4) {
234 actor->flags &=
235 ~(FLAGS_DETOURING | FLAGS_TRYRIGHT);
236 actor->turns = 0;
237 }
238 }
239 }
240 return DirectionToCmd(actor->direction);
241}

243static int WillFire(TActor * actor, int roll)
244{
245 const CharBot *bot = ActorGetCharacter(actor)->bot;
246 if ((actor->flags & FLAGS_VISIBLE) != 0 &&
247 ActorCanFire(actor) &&
248 roll < bot->probabilityToShoot)
249 {
250 if ((actor->flags & FLAGS_GOOD_GUY) != 0)
251 return 1; //!FacingPlayer( actor);
252 else if (sAreGoodGuysPresent)
253 {
254 return 1;
255 }
256 else
257 {
258 return IsFacingPlayer(actor, actor->direction);
259 }
260 }
261 return 0;
262}

264void Detour(TActor * actor)
265{
266 actor->flags |= FLAGS_DETOURING;
267 actor->turns = 1;
268 if (actor->flags & FLAGS_TRYRIGHT)
269 actor->direction =
270 (CmdToDirection(actor->lastCmd) + 1) % 8;
271 else
272 actor->direction =
273 (CmdToDirection(actor->lastCmd) + 7) % 8;
274}

294static int GetCmd(TActor *actor, const int delayModifier, const int rollLimit);
295int AICommand(const int ticks)
296{
297 int count = 0;
298 int delayModifier;
299 int rollLimit;
300
301 switch (ConfigGetEnum(&gConfig, "Game.Difficulty"))
302 {
303 case DIFFICULTY_VERYEASY:
304 delayModifier = 4;
305 rollLimit = 300;
306 break;
307 case DIFFICULTY_EASY:
308 delayModifier = 2;
309 rollLimit = 200;
310 break;
311 case DIFFICULTY_HARD:
312 delayModifier = 1;
313 rollLimit = 75;
314 break;
315 case DIFFICULTY_VERYHARD:
316 delayModifier = 1;
317 rollLimit = 50;
318 break;
319 default:
320 delayModifier = 1;
321 rollLimit = 100;
322 break;
323 }
324
325 CA_FOREACH(TActor, actor, gActors)
326 if (!actor->isInUse || actor->PlayerUID >= 0 || actor->dead)
327 {
328 continue;
329 }
330 int cmd = 0;
331 if (!(actor->flags & FLAGS_PRISONER))
332 {
333 if (actor->flags & (FLAGS_VICTIM | FLAGS_GOOD_GUY))
334 {
335 sAreGoodGuysPresent = true;
336 }
337 cmd = GetCmd(actor, delayModifier, rollLimit);
338 actor->aiContext->Delay = MAX(0, actor->aiContext->Delay - ticks);
339 }
340 CommandActor(actor, cmd, ticks);
341 actor->aiContext->LastCmd = cmd;
342 count++;
343 CA_FOREACH_END()
344 return count;
345}
346static int GetCmd(TActor *actor, const int delayModifier, const int rollLimit)
347{
348 const CharBot *bot = ActorGetCharacter(actor)->bot;
349
350 int cmd = 0;
351
352 // Wake up if it can see a player
353 if ((actor->flags & FLAGS_SLEEPING) && actor->aiContext->Delay == 0)
354 {
355 if (CanSeeAPlayer(actor))
356 {
357 actor->flags &= ~FLAGS_SLEEPING;
358 ActorSetAIState(actor, AI_STATE_NONE);
359 }
360 actor->aiContext->Delay = bot->actionDelay * delayModifier;
361 // Randomly change direction
362 int newDir = (int)actor->direction + ((rand() % 2) * 2 - 1);
363 if (newDir < (int)DIRECTION_UP)
364 {
365 newDir = (int)DIRECTION_UPLEFT;
366 }
367 if (newDir == (int)DIRECTION_COUNT)
368 {
369 newDir = (int)DIRECTION_UP;
370 }
371 cmd = DirectionToCmd((int)newDir);
372 }
373 // Go to sleep if the player's too far away
374 if (!(actor->flags & FLAGS_SLEEPING) &&
375 actor->aiContext->Delay == 0 &&
376 !(actor->flags & FLAGS_AWAKEALWAYS))
377 {
378 if (!IsCloseToPlayer(actor->Pos, (40 * 16) << 8))
379 {
380 actor->flags |= FLAGS_SLEEPING;
381 ActorSetAIState(actor, AI_STATE_IDLE);
382 }
383 }
384
385 if (actor->flags & FLAGS_SLEEPING)
386 {
387 return cmd;
388 }
389
390 bool bypass = false;
391 const int roll = rand() % rollLimit;
392 if (actor->flags & FLAGS_FOLLOWER)
393 {
394 cmd = Follow(actor);
395 }
396 else if (!!(actor->flags & FLAGS_SNEAKY) &&
397 !!(actor->flags & FLAGS_VISIBLE) &&
398 DidPlayerShoot())
399 {
400 cmd = AIHuntClosest(actor) | CMD_BUTTON1;
401 if (actor->flags & FLAGS_RUNS_AWAY)
402 {
403 // Turn back and shoot for running away characters
404 cmd = AIReverseDirection(cmd);
405 }
406 bypass = true;
407 ActorSetAIState(actor, AI_STATE_HUNT);
408 }
409 else if (actor->flags & FLAGS_DETOURING)
410 {
411 cmd = BrightWalk(actor, roll);
412 ActorSetAIState(actor, AI_STATE_TRACK);
413 }
414 else if (actor->flags & FLAGS_RESCUED)
415 {
416 // If we haven't completed all objectives, act as follower
417 if (!CanCompleteMission(&gMission))
418 {
419 cmd = Follow(actor);
420 }
421 else
422 {
423 // Run towards exit
424 const Vec2i exitPos = MapGetExitPos(&gMap);
425 cmd = AIGoto(actor, exitPos, false);
426 }
427 }
428 else if (actor->aiContext->Delay > 0)
429 {
430 cmd = actor->lastCmd & ~CMD_BUTTON1;
431 }
432 else
433 {
434 if (roll < bot->probabilityToTrack)
435 {
436 cmd = AIHuntClosest(actor);
437 ActorSetAIState(actor, AI_STATE_HUNT);
438 }
439 else if (roll < bot->probabilityToMove)
440 {
441 cmd = DirectionToCmd(rand() & 7);
442 ActorSetAIState(actor, AI_STATE_TRACK);
443 }
444 actor->aiContext->Delay = bot->actionDelay * delayModifier;
445 }
446 if (!bypass)
447 {
448 if (WillFire(actor, roll))
449 {
450 cmd |= CMD_BUTTON1;
451 if (!!(actor->flags & FLAGS_FOLLOWER) &&
452 (actor->flags & FLAGS_GOOD_GUY))
453 {
454 // Shoot in a random direction away
455 for (int j = 0; j < 10; j++)
456 {
457 direction_e d =
458 (direction_e)(rand() % DIRECTION_COUNT);
459 if (!IsFacingPlayer(actor, d))
460 {
461 cmd = DirectionToCmd(d) | CMD_BUTTON1;
462 break;
463 }
464 }
465 }
466 if (actor->flags & FLAGS_RUNS_AWAY)
467 {
468 // Turn back and shoot for running away characters
469 cmd |= AIReverseDirection(AIHuntClosest(actor));
470 }
471 ActorSetAIState(actor, AI_STATE_HUNT);
472 }
473 else
474 {
475 if ((actor->flags & FLAGS_VISIBLE) == 0)
476 {
477 // I think this is some hack to make sure invisible enemies don't fire so much
478 ActorGetGun(actor)->lock = 40;
479 }
480 if (cmd && !IsDirectionOK(actor, CmdToDirection(cmd)) &&
481 (actor->flags & FLAGS_DETOURING) == 0)
482 {
483 Detour(actor);
484 cmd = 0;
485 ActorSetAIState(actor, AI_STATE_TRACK);
486 }
487 }
488 }
489 return cmd;
490}
491static int Follow(TActor *a)
492{
493 // If we are a rescue objective and we are in the exit
494 // area, stop following and stay in the rescue area
495 const Character *ch = ActorGetCharacter(a);
496 const CharacterStore *store = &gCampaign.Setting.characters;
497 if (CharacterIsPrisoner(store, ch) && CanCompleteMission(&gMission) &&
498 MapIsTileInExit(&gMap, &a->tileItem))
499 {
500 a->flags &= ~FLAGS_FOLLOWER;
501 a->flags |= FLAGS_RESCUED;
502 return 0;
503 }
504 else if (IsCloseToPlayer(a->Pos, 32 << 8))
505 {
506 ActorSetAIState(a, AI_STATE_IDLE);
507 return 0;
508 }
509 else
510 {
511 ActorSetAIState(a, AI_STATE_FOLLOW);
512 return AIGoto(a, AIGetClosestPlayerPos(a->Pos), true);
513 }

516void AICommandLast(const int ticks)
517{
518 CA_FOREACH(TActor, actor, gActors)
519 if (!actor->isInUse || actor->PlayerUID >= 0 || actor->dead ||
520 (actor->flags & FLAGS_PRISONER))
521 {
522 continue;
523 }
524 const int cmd = actor->aiContext->LastCmd;
525 actor->aiContext->Delay = MAX(0, actor->aiContext->Delay - ticks);
526 CommandActor(actor, cmd, ticks);
527 CA_FOREACH_END()
528}

530void AIAddRandomEnemies(const int enemies, const Mission *m)
531{
532 if (m->Enemies.size > 0 && m->EnemyDensity > 0 &&
533 enemies < MAX(1, (m->EnemyDensity * ConfigGetInt(&gConfig, "Game.EnemyDensity")) / 100))
534 {
535 NActorAdd aa = NActorAdd_init_default;
536 aa.UID = ActorsGetNextUID();
537 aa.CharId = CharacterStoreGetRandomBaddieId(
538 &gCampaign.Setting.characters);
539 aa.Direction = rand() % DIRECTION_COUNT;
540 const Character *c =
541 CArrayGet(&gCampaign.Setting.characters.OtherChars, aa.CharId);
542 aa.Health = CharacterGetStartingHealth(c, true);
543 aa.FullPos = PlaceAwayFromPlayers(&gMap, true);
544 GameEvent e = GameEventNew(GAME_EVENT_ACTOR_ADD);
545 e.u.ActorAdd = aa;
546 GameEventsEnqueue(&gGameEvents, e);
547 gBaddieCount++;
548 }
549}

551void InitializeBadGuys(void)
552{
553 CA_FOREACH(Objective, o, gMission.missionData->Objectives)
554 if (o->Type == OBJECTIVE_KILL &&
555 gMission.missionData->SpecialChars.size > 0)
556 {
557 for (; o->placed < o->Count; o->placed++)
558 {
559 NActorAdd aa = NActorAdd_init_default;
560 aa.UID = ActorsGetNextUID();
561 aa.CharId = CharacterStoreGetRandomSpecialId(
562 &gCampaign.Setting.characters);
563 aa.TileItemFlags = ObjectiveToTileItem(_ca_index);
564 aa.Direction = rand() % DIRECTION_COUNT;
565 const Character *c =
566 CArrayGet(&gCampaign.Setting.characters.OtherChars, aa.CharId);
567 aa.Health = CharacterGetStartingHealth(c, true);
568 aa.FullPos = PlaceAwayFromPlayers(&gMap, false);
569 GameEvent e = GameEventNew(GAME_EVENT_ACTOR_ADD);
570 e.u.ActorAdd = aa;
571 GameEventsEnqueue(&gGameEvents, e);
572
573 // Process the events that actually place the actors
574 HandleGameEvents(&gGameEvents, NULL, NULL, NULL);
575 }
576 }
577 else if (o->Type == OBJECTIVE_RESCUE)
578 {
579 for (; o->placed < o->Count; o->placed++)
580 {
581 NActorAdd aa = NActorAdd_init_default;
582 aa.UID = ActorsGetNextUID();
583 aa.CharId = CharacterStoreGetPrisonerId(
584 &gCampaign.Setting.characters, 0);
585 aa.TileItemFlags = ObjectiveToTileItem(_ca_index);
586 aa.Direction = rand() % DIRECTION_COUNT;
587 const Character *c =
588 CArrayGet(&gCampaign.Setting.characters.OtherChars, aa.CharId);
589 aa.Health = CharacterGetStartingHealth(c, true);
590 if (MapHasLockedRooms(&gMap))
591 {
592 aa.FullPos = PlacePrisoner(&gMap);
593 }
594 else
595 {
596 aa.FullPos = PlaceAwayFromPlayers(&gMap, false);
597 }
598 GameEvent e = GameEventNew(GAME_EVENT_ACTOR_ADD);
599 e.u.ActorAdd = aa;
600 GameEventsEnqueue(&gGameEvents, e);
601
602 // Process the events that actually place the actors
603 HandleGameEvents(&gGameEvents, NULL, NULL, NULL);
604 }
605 }
606 CA_FOREACH_END()
607
608 gBaddieCount = gMission.index * 4;
609 sAreGoodGuysPresent = false;
610}

612void CreateEnemies(void)
613{
614 if (gCampaign.Setting.characters.baddieIds.size == 0)
615 {
616 return;
617 }
618
619 const int density =
620 gMission.missionData->EnemyDensity *
621 ConfigGetInt(&gConfig, "Game.EnemyDensity");
622 for (int i = 0; i < density / 100; i++)
623 {
624 NActorAdd aa = NActorAdd_init_default;
625 aa.UID = ActorsGetNextUID();
626 aa.CharId = CharacterStoreGetRandomBaddieId(
627 &gCampaign.Setting.characters);
628 aa.FullPos = PlaceAwayFromPlayers(&gMap, true);
629 aa.Direction = rand() % DIRECTION_COUNT;
630 const Character *c =
631 CArrayGet(&gCampaign.Setting.characters.OtherChars, aa.CharId);
632 aa.Health = CharacterGetStartingHealth(c, true);
633 GameEvent e = GameEventNew(GAME_EVENT_ACTOR_ADD);
634 e.u.ActorAdd = aa;
635 GameEventsEnqueue(&gGameEvents, e);
636 gBaddieCount++;
637
638 // Process the events that actually place the actors
639 HandleGameEvents(&gGameEvents, NULL, NULL, NULL);
640 }
