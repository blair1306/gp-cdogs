53#define KEYBOARD_REPEAT_DELAY 500
54#define KEYBOARD_REPEAT_TICKS 100

56void KeyInit(keyboard_t *keyboard)
57{
58 memset(keyboard, 0, sizeof *keyboard);
59 keyboard->modState = KMOD_NONE;
60 keyboard->ticks = 0;
61 keyboard->repeatedTicks = 0;
62 keyboard->isFirstRepeat = 1;
63 for (int i = 0; i < MAX_KEYBOARD_CONFIGS; i++)
64 {
65 char buf[256];
66 sprintf(buf, "Input.PlayerCodes%d", i);
67 keyboard->PlayerKeys[i] = KeyLoadPlayerKeys(ConfigGet(&gConfig, buf));
68 }
69}
70InputKeys KeyLoadPlayerKeys(Config *c)
71{
72 InputKeys k;
73 k.left = (SDL_Scancode)ConfigGetInt(c, "left");
74 k.right = (SDL_Scancode)ConfigGetInt(c, "right");
75 k.up = (SDL_Scancode)ConfigGetInt(c, "up");
76 k.down = (SDL_Scancode)ConfigGetInt(c, "down");
77 k.button1 = (SDL_Scancode)ConfigGetInt(c, "button1");
78 k.button2 = (SDL_Scancode)ConfigGetInt(c, "button2");
79 k.map = (SDL_Scancode)ConfigGetInt(c, "map");
80 return k;
81}

83void KeyPrePoll(keyboard_t *keyboard)
84{
85 memcpy(
86 keyboard->previousKeys,
87 keyboard->currentKeys,
88 sizeof keyboard->previousKeys);
89 keyboard->modState = SDL_GetModState();
90 keyboard->Typed[0] = '\0';
91}

93void KeyOnKeyDown(keyboard_t *keyboard, const SDL_Keysym s)
94{
95 keyboard->currentKeys[s.scancode].isPressed = true;
96 if (s.sym >= (SDL_Keycode)' ' && s.sym <= (SDL_Keycode)'z')
97 {
98 keyboard->currentKeys[s.scancode].keycode = s.sym;
99 }
100 else
101 {
102 keyboard->currentKeys[s.scancode].keycode = 0;
103 }
104}
105void KeyOnKeyUp(keyboard_t *keyboard, const SDL_Keysym s)
106{
107 keyboard->currentKeys[s.scancode].isPressed = 0;
108}

110void KeyPostPoll(keyboard_t *keyboard, Uint32 ticks)
111{
112 int isRepeating = 0;
113 int areSameKeysPressed = 1;
114 for (int i = 0; i < SDL_NUM_SCANCODES; i++)
115 {
116 if (keyboard->previousKeys[i].isPressed ^
117 keyboard->currentKeys[i].isPressed)
118 {
119 areSameKeysPressed = 0;
120 break;
121 }
122 }
123 // If same keys have been pressed, remember how long they have been pressed
124 if (areSameKeysPressed)
125 {
126 Uint32 ticksElapsed = ticks - keyboard->ticks;
127 keyboard->repeatedTicks += ticksElapsed;
128 }
129 else
130 {
131 keyboard->repeatedTicks = 0;
132 keyboard->isFirstRepeat = 1;
133 }
134 // If more time has elapsed, forget about previous keys for repeating
135 if (keyboard->repeatedTicks > KEYBOARD_REPEAT_DELAY &&
136 keyboard->isFirstRepeat)
137 {
138 keyboard->repeatedTicks -= KEYBOARD_REPEAT_DELAY;
139 isRepeating = 1;
140 }
141 else if (keyboard->repeatedTicks > KEYBOARD_REPEAT_TICKS &&
142 !keyboard->isFirstRepeat)
143 {
144 keyboard->repeatedTicks -= KEYBOARD_REPEAT_TICKS;
145 isRepeating = 1;
146 }
147 if (isRepeating)
148 {
149 memcpy(
150 keyboard->pressedKeys,
151 keyboard->currentKeys,
152 sizeof keyboard->pressedKeys);
153 keyboard->isFirstRepeat = false;
154 // Ignore the keys that tend to stay pressed/unpressed
155 // i.e. lock keys
156 keyboard->pressedKeys[SDL_SCANCODE_NUMLOCKCLEAR].isPressed = false;
157 keyboard->pressedKeys[SDL_SCANCODE_CAPSLOCK].isPressed = false;
158 keyboard->pressedKeys[SDL_SCANCODE_SCROLLLOCK].isPressed = false;
159 }
160 else
161 {
162 for (int i = 0; i < SDL_NUM_SCANCODES; i++)
163 {
164 keyboard->pressedKeys[i].isPressed =
165 keyboard->currentKeys[i].isPressed &&
166 !keyboard->previousKeys[i].isPressed;
167 }
168 }
169 keyboard->ticks = ticks;
170}

172bool KeyIsDown(const keyboard_t *k, const int key)
173{
174 return k->currentKeys[key].isPressed;
175}
176
177bool KeyIsPressed(const keyboard_t *k, const int key)
178{
179 return k->pressedKeys[key].isPressed;
180}
181
182bool KeyIsReleased(const keyboard_t *k, const int key)
183{
184 return !KeyIsDown(k, key) && k->previousKeys[key].isPressed;
185}
186
187SDL_Scancode KeyGetPressed(const keyboard_t *k)
188{
189 for (SDL_Scancode i = 0; i < SDL_NUM_SCANCODES; i++)
190 {
191 if (KeyIsPressed(k, i))
192 {
193 return i;
194 }
195 }
196 return 0;
197}
198
199SDL_Scancode KeyGet(const InputKeys *keys, const key_code_e keyCode)
200{
201 switch (keyCode)
202 {
203 case KEY_CODE_LEFT:
204 return keys->left;
205 case KEY_CODE_RIGHT:
206 return keys->right;
207 case KEY_CODE_UP:
208 return keys->up;
209 case KEY_CODE_DOWN:
210 return keys->down;
211 case KEY_CODE_BUTTON1:
212 return keys->button1;
213 case KEY_CODE_BUTTON2:
214 return keys->button2;
215 case KEY_CODE_MAP:
216 return keys->map;
217 default:
218 LOG(LM_MAIN, LL_ERROR, "Unhandled key code %d\n", (int)keyCode);
219 return SDL_SCANCODE_UNKNOWN;
220 }
221}
