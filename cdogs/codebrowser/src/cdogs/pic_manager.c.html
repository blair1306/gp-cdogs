57static void PicManagerAdd(
58 map_t pics, map_t sprites, const char *name, SDL_Surface *imageIn)
59{
60 char buf[CDOGS_FILENAME_MAX];
61 const char *dot = strrchr(name, '.');
62 if (dot)
63 {
64 strncpy(buf, name, dot - name);
65 buf[dot - name] = '\0';
66 }
67 else
68 {
69 strcpy(buf, name);
70 }
71 // TODO: check if name already exists
72 // TODO: use efficient data structure like trie
73 // Special case: if the file name is in the form foobar_WxH.ext,
74 // this is a spritesheet where each sprite is W wide by H high
75 // Load multiple images from this single sheet
76 Vec2i size = Vec2iNew(imageIn->w, imageIn->h);
77 bool isSpritesheet = false;
78 char *underscore = strrchr(buf, '_');
79 const char *x = strrchr(buf, 'x');
80 if (underscore != NULL && x != NULL &&
81 underscore + 1 < x && x + 1 < buf + strlen(buf))
82 {
83 if (sscanf(underscore, "_%dx%d", &size.x, &size.y) != 2)
84 {
85 size = Vec2iNew(imageIn->w, imageIn->h);
86 }
87 else
88 {
89 *underscore = '\0';
90 isSpritesheet = true;
91 }
92 }
93 NamedSprites *nsp = NULL;
94 NamedPic *np = NULL;
95 if (isSpritesheet)
96 {
97 nsp = AddNamedSprites(sprites, buf);
98 }
99 else
100 {
101 np = AddNamedPic(pics, buf, NULL);
102 }
103 // Use 32-bit image
104 SDL_Surface *image = SDL_ConvertSurfaceFormat(
105 imageIn, SDL_PIXELFORMAT_RGBA8888, 0);
106 SDL_FreeSurface(imageIn);
107 SDL_LockSurface(image);
108 Vec2i offset;
109 for (offset.y = 0; offset.y < image->h; offset.y += size.y)
110 {
111 for (offset.x = 0; offset.x < image->w; offset.x += size.x)
112 {
113 Pic *pic;
114 if (isSpritesheet)
115 {
116 Pic p;
117 CArrayPushBack(&nsp->pics, &p);
118 pic = CArrayGet(&nsp->pics, nsp->pics.size - 1);
119 }
120 else
121 {
122 pic = &np->pic;
123 }
124 PicLoad(pic, size, offset, image);
125
126 if (strncmp("chars/", buf, strlen("chars/")) == 0)
127 {
128 // Convert char pics to multichannel version
129 for (int i = 0; i < pic->size.x * pic->size.y; i++)
130 {
131 color_t c = PIXEL2COLOR(pic->Data[i]);
132 // Don't bother if the alpha has already been modified; it
133 // means we have already processed this pixel
134 if (c.a != 255)
135 {
136 continue;
137 }
138 const uint8_t value = MAX(MAX(c.r, c.g), c.b);
139 if (abs((int)c.r - c.g) < 5 && abs((int)c.g - c.b) < 5)
140 {
141 // don't convert greyscale colours
142 }
143 else if ((c.g < 5 && c.b < 5) ||
144 (abs((int)c.g - c.b) < 5 && c.r > 250))
145 {
146 // Skin
147 c.r = c.g = c.b = value;
148 c.a = 254;
149 }
150 else if ((c.r < 5 && c.b < 5) ||
151 (abs((int)c.r - c.b) < 5 && c.g > 250))
152 {
153 // Hair
154 c.r = c.g = c.b = value;
155 c.a = 250;
156 }
157 else if ((c.r < 5 && c.g < 5) ||
158 (abs((int)c.r - c.g) < 5 && c.b > 250))
159 {
160 // Arms
161 c.r = c.g = c.b = value;
162 c.a = 253;
163 }
164 else if (c.b < 5 || (c.r > 250 && c.g > 250))
165 {
166 // Body
167 c.r = c.g = c.b = value;
168 c.a = 252;
169 }
170 else if (c.r < 5 || (c.g > 250 && c.b > 250))
171 {
172 // Legs
173 c.r = c.g = c.b = value;
174 c.a = 251;
175 }
176 pic->Data[i] = COLOR2PIXEL(c);
177 }
178 }
179 }
180 }
181 SDL_UnlockSurface(image);
182 SDL_FreeSurface(image);
183
184 AfterAdd(&gPicManager);
185}

187void PicManagerLoadDir(
188 PicManager *pm, const char *path, const char *prefix,
189 map_t pics, map_t sprites)
190{
191 tinydir_dir dir;
192 if (tinydir_open(&dir, path) == -1)
193 {
194 if (errno != ENOENT)
195 {
196 LOG(LM_MAIN, LL_ERROR, "Error opening image dir '%s': %s",
197 path, strerror(errno));
198 }
199 goto bail;
200 }
201
202 for (; dir.has_next; tinydir_next(&dir))
203 {
204 tinydir_file file;
205 if (tinydir_readfile(&dir, &file) == -1)
206 {
207 LOG(LM_MAIN, LL_ERROR, "Cannot read file '%s': %s",
208 file.path, strerror(errno));
209 goto bail;
210 }
211 if (file.is_reg)
212 {
213 SDL_RWops *rwops = SDL_RWFromFile(file.path, "rb");
214 const bool isPng = IMG_isPNG(rwops);
215 if (isPng)
216 {
217 SDL_Surface *data = IMG_Load_RW(rwops, 0);
218 if (!data)
219 {
220 LOG(LM_MAIN, LL_ERROR, "Cannot load image IMG_Load: %s",
221 IMG_GetError());
222 }
223 else
224 {
225 char buf[CDOGS_PATH_MAX];
226 if (prefix)
227 {
228 char buf1[CDOGS_PATH_MAX];
229 sprintf(buf1, "%s/%s", prefix, file.name);
230 PathGetWithoutExtension(buf, buf1);
231 }
232 else
233 {
234 PathGetBasenameWithoutExtension(buf, file.name);
235 }
236 PicManagerAdd(pics, sprites, buf, data);
237 }
238 }
239 rwops->close(rwops);
240 }
241 else if (file.is_dir && file.name[0] != '.')
242 {
243 if (prefix)
244 {
245 char buf[CDOGS_PATH_MAX];
246 sprintf(buf, "%s/%s", prefix, file.name);
247 PicManagerLoadDir(pm, file.path, buf, pics, sprites);
248 }
249 else
250 {
251 PicManagerLoadDir(
252 pm, file.path, file.name, pics, sprites);
253 }
254 }
255 }
256
257bail:
258 tinydir_close(&dir);
259}
260void PicManagerLoad(PicManager *pm, const char *path)
261{
262 if (!IMG_Init(IMG_INIT_PNG))
263 {
264 perror("Cannot initialise SDL_Image");
265 return;
266 }
267 char buf[CDOGS_PATH_MAX];
268 GetDataFilePath(buf, path);
269 PicManagerLoadDir(pm, buf, NULL, pm->pics, pm->sprites);
270}

288static int CompareStyleNames(const void *v1, const void *v2);
289static void FindStylePics(
290 PicManager *pm, CArray *styleNames, PFany hashmapFunc)
291{
292 // Scan all pics for style pics
293 CA_FOREACH(char *, styleName, *styleNames)
294 CFREE(*styleName);
295 CA_FOREACH_END()
296 CArrayClear(styleNames);
297 hashmap_iterate(pm->customPics, hashmapFunc, pm);
298 hashmap_iterate(pm->pics, hashmapFunc, pm);
299 // Sort the style names alphabetically
300 // This prevents the list from reordering unpredictably, when the editor
301 // is used and masked pics get added
302 qsort(
303 styleNames->data, styleNames->size, styleNames->elemSize,
304 CompareStyleNames);
305}
306static int CompareStyleNames(const void *v1, const void *v2)
307{
308 const char * const *s1 = v1;
309 const char * const *s2 = v2;
310 return strcmp(*s1, *s2);
311}
312static void MaybeAddStyleName(
313 const NamedPic *p, const char *prefix, CArray *styleNames)
314{
315 // Look for style names within a full name of the form:
316 // prefix/style/suffix
317 // NOTE: prefix should include slash
318 const char *picName = p->name;
319 if (strncmp(picName, prefix, strlen(prefix)) != 0)
320 {
321 return;
322 }
323 const char *nextSlash = strchr(picName + strlen(prefix), '/');
324 char buf[CDOGS_PATH_MAX];
325 const size_t len = nextSlash - picName - strlen(prefix);
326 strncpy(buf, picName + strlen(prefix), len);
327 buf[len] = '\0';
328 // Check if we already have the style name
329 // This can happen if a custom pic uses the same name as a built in one
330 CA_FOREACH(char *, styleName, *styleNames)
331 if (strcmp(*styleName, buf) == 0)
332 {
333 return;
334 }
335 CA_FOREACH_END()
336
337 char *s;
338 CSTRDUP(s, buf);
339 CArrayPushBack(styleNames, &s);
340}

404{

498void PicManagerGenerateMaskedPic(
499 PicManager *pm, const char *name,
500 const color_t mask, const color_t maskAlt)
501{
502 char maskedName[256];
503 GetMaskedName(maskedName, name, mask, maskAlt);
504 // Check if the masked pic already exists
505 if (PicManagerGetPic(pm, maskedName) != NULL) return;
506
507 // Check if the original pic is available; if not then it's impossible to
508 // create the masked version
509 Pic *original = PicManagerGetPic(pm, name);
510 CASSERT(original != NULL, "Cannot find original pic for masking");
511
512 // Create the new pic by masking the original pic
513 Pic p = PicCopy(original);
514 debug(D_VERBOSE, "Creating new masked pic %s (%d x %d)\n",
515 maskedName, p.size.x, p.size.y);
516 for (int i = 0; i < p.size.x * p.size.y; i++)
517 {
518 color_t c = PIXEL2COLOR(original->Data[i]);
519 // Apply mask based on which channel each pixel is
520 if (c.g <= 2 && c.b <= 2)
521 {
522 // Restore to white before masking
523 c.g = c.r;
524 c.b = c.r;
525 c = ColorMult(c, maskAlt);
526 }
527 else
528 {
529 c = ColorMult(c, mask);
530 }
531 p.Data[i] = COLOR2PIXEL(c);
532 // TODO: more channels
533 }
534 AddNamedPic(pm->customPics, maskedName, &p);
535
536 AfterAdd(pm);
537}
538void PicManagerGenerateMaskedStylePic(
539 PicManager *pm, const char *name, const char *style, const char *type,

542 char buf[256];
543 sprintf(buf, "%s/%s/%s", name, style, type);

547static void GetMaskedName(
548 char *buf, const char *name, const color_t mask, const color_t maskAlt)
549{
550 char maskName[8];
551 ColorStr(maskName, mask);
552 char maskAltName[8];
553 ColorStr(maskAltName, maskAlt);
554 sprintf(buf, "%s/%s/%s", name, maskName, maskAltName);

557static NamedPic *AddNamedPic(map_t pics, const char *name, const Pic *p)
558{
559 NamedPic *n;
560 CMALLOC(n, sizeof *n);
561 if (p != NULL) n->pic = *p;
562 CSTRDUP(n->name, name);
563 const int error = hashmap_put(pics, name, n);
564 if (error != MAP_OK)
565 {
566 LOG(LM_MAIN, LL_ERROR, "failed to add named pic %s: %d", name, error);
567 CFREE(n);
568 return NULL;
569 }
570 return n;
571}
572static NamedSprites *AddNamedSprites(map_t sprites, const char *name)
573{
574 NamedSprites *ns;
575 CMALLOC(ns, sizeof *ns);
576 NamedSpritesInit(ns, name);
577 const int error = hashmap_put(sprites, name, ns);
578 if (error != MAP_OK)
579 {
580 LOG(LM_MAIN, LL_ERROR, "failed to add named sprites %s: %d",
581 name, error);
582 CFREE(ns);
583 return NULL;
584 }
585 return ns;
586}
