30#include <json/json.h>
31
32#include "pic.h"
33#include "tile.h"
34
35typedef enum
36{
37 PARTICLE_PIC,
38 PARTICLE_TEXT
39} ParticleType;
40ParticleType StrParticleType(const char *s);
41
42typedef struct
43{
44 char *Name;
45 ParticleType Type;
46 union
47 {
48 CPic Pic;
49 color_t TextColor;
50 } u;
51 const NamedSprites *Sprites;
52 // -1 is infinite range
53 int RangeLow;
54 int RangeHigh;
55 int GravityFactor;
56 bool HitsWalls;
57 bool Bounces;
58 bool WallBounces;
59} ParticleClass;
60typedef struct
61{
62 CArray Classes; // of ParticleClass
63 CArray CustomClasses; // of ParticleClass
64} ParticleClasses;
65extern ParticleClasses gParticleClasses;
66
67typedef struct
68{
69 const ParticleClass *Class;
70 union
71 {
72 CPic Pic;
73 char *Text;
74 } u;
75 // Coordinates are in full
76 Vec2i Pos;
77 int Z;
78 double Angle;
79 int DZ;
80 double Spin;
81 int Count;
82 int Range;
83 TTileItem tileItem;
84 bool isInUse;
85} Particle;
86extern CArray gParticles; // of Particle
87
88typedef struct
89{
90 const ParticleClass *Class;
91 Vec2i FullPos;
92 int Z;
93 Vec2i Vel;
94 double Angle;
95 int DZ;
96 double Spin;
97 char Text[128];
98} AddParticle;
99
100void ParticleClassesInit(ParticleClasses *classes, const char *filename);
101void ParticleClassesLoadJSON(CArray *classes, json_t *root);
102void ParticleClassesTerminate(ParticleClasses *classes);
103void ParticleClassesClear(CArray *classes);
104const ParticleClass *StrParticleClass(
105 const ParticleClasses *classes, const char *name);
106
107void ParticlesInit(CArray *particles);
108void ParticlesTerminate(CArray *particles);
109void ParticlesUpdate(CArray *particles, const int ticks);
110
111int ParticleAdd(CArray *particles, const AddParticle add);
112void ParticleDestroy(CArray *particles, const int id);
