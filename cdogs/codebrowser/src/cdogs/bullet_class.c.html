69BulletClass *StrBulletClass(const char *s)
70{
71 if (s == NULL || strlen(s) == 0)
72 {
73 return NULL;
74 }
75 CA_FOREACH(BulletClass, b, gBulletClasses.CustomClasses)
76 if (strcmp(s, b->Name) == 0)
77 {
78 return b;
79 }
80 CA_FOREACH_END()
81 CA_FOREACH(BulletClass, b, gBulletClasses.Classes)
82 if (strcmp(s, b->Name) == 0)
83 {
84 return b;
85 }
86 CA_FOREACH_END()
87 CASSERT(false, "cannot parse bullet name");
88 return NULL;
89}

93static CPicDrawContext GetBulletDrawContext(const int id)
94{
95 const TMobileObject *obj = CArrayGet(&gMobObjs, id);
96 CASSERT(obj->isInUse, "Cannot draw non-existent mobobj");
97 // Calculate direction based on velocity
98 const direction_e dir =
99 RadiansToDirection(Vec2iToRadians(obj->tileItem.VelFull));
100 const Pic *pic = CPicGetPic(&obj->tileItem.CPic, dir);
101 CPicDrawContext c;
102 c.Dir = dir;
103 if (pic != NULL)
104 {
105 c.Offset = Vec2iNew(
106 pic->size.x / -2, pic->size.y / -2 - obj->z / Z_FACTOR);
107 }
108 return c;
109}

112static Vec2i SeekTowards(
113 const Vec2i pos, const Vec2i vel, const double speedMin,
114 const Vec2i targetPos, const int seekFactor)
115{
116 // Compensate for bullet's velocity
117 const Vec2i targetVel = Vec2iMinus(Vec2iMinus(targetPos, pos), vel);
118 // Don't seek if the coordinates are too big
119 if (abs(targetVel.x) > 10000 || abs(targetVel.y) > 10000 ||
120 Vec2iIsZero(targetVel))
121 {
122 return vel;
123 }
124 const double targetMag = sqrt(
125 targetVel.x*targetVel.x + targetVel.y*targetVel.y);
126 const double magnitude = MAX(speedMin,
127 Vec2iEqual(vel, Vec2iZero()) ? speedMin : sqrt(vel.x*vel.x + vel.y*vel.y));
128 const double combinedX =
129 vel.x / magnitude * seekFactor + targetVel.x / targetMag;
130 const double combinedY =
131 vel.y / magnitude * seekFactor + targetVel.y / targetMag;
132 return Vec2iNew(
133 (int)round(combinedX * magnitude / (seekFactor + 1)),
134 (int)round(combinedY * magnitude / (seekFactor + 1)));
135}

145static HitResult HitItem(
146 TMobileObject *obj, const Vec2i pos, const bool multipleHits);
147bool UpdateBullet(struct MobileObject *obj, const int ticks)
148{
149 TileItemUpdate(&obj->tileItem, ticks);
150 obj->count += ticks;
151 obj->specialLock = MAX(0, obj->specialLock - ticks);
152 if (obj->count < obj->bulletClass->Delay)
153 {
154 return true;
155 }
156
157 if (obj->range >= 0 && obj->count > obj->range)
158 {
159 if (!gCampaign.IsClient)
160 {
161 FireGuns(obj, &obj->bulletClass->OutOfRangeGuns);
162 }
163 if (obj->bulletClass->OutOfRangeSpark != NULL)
164 {
165 GameEvent s = GameEventNew(GAME_EVENT_ADD_PARTICLE);
166 s.u.AddParticle.Class = obj->bulletClass->OutOfRangeSpark;
167 s.u.AddParticle.FullPos = Vec2iNew(obj->x, obj->y);
168 s.u.AddParticle.Z = obj->z;
169 GameEventsEnqueue(&gGameEvents, s);
170 }
171 return false;
172 }
173
174 const Vec2i posStart = Vec2iNew(obj->x, obj->y);
175
176 if (obj->bulletClass->SeekFactor > 0)
177 {
178 // Find the closest target to this bullet and steer towards it
179 const TActor *owner = ActorGetByUID(obj->ActorUID);
180 if (owner == NULL)
181 {
182 return false;
183 }
184 const TActor *target = AIGetClosestEnemy(posStart, owner, obj->flags);
185 if (target && !target->dead)
186 {
187 for (int i = 0; i < ticks; i++)
188 {
189 obj->tileItem.VelFull = SeekTowards(
190 posStart, obj->tileItem.VelFull,
191 obj->bulletClass->SpeedLow, target->Pos,
192 obj->bulletClass->SeekFactor);
193 }
194 }
195 }
196
197 HitResult hit = { HIT_NONE, Vec2iZero(), Vec2iZero() };
198 if (!gCampaign.IsClient)
199 {
200 hit = HitItem(obj, posStart, obj->bulletClass->Persists);
201 }
202 Vec2i pos =
203 Vec2iScale(Vec2iAdd(posStart, obj->tileItem.VelFull), ticks);
204
205 if (hit.Type != HIT_NONE)
206 {
207 GameEvent b = GameEventNew(GAME_EVENT_BULLET_BOUNCE);
208 b.u.BulletBounce.UID = obj->UID;
209 b.u.BulletBounce.HitType = (int)hit.Type;
210 bool alive = true;
211 if ((hit.Type == HIT_WALL && !obj->bulletClass->WallBounces) ||
212 ((hit.Type == HIT_OBJECT || hit.Type == HIT_FLESH) &&
213 obj->bulletClass->HitsObjects))
214 {
215 b.u.BulletBounce.Spark = true;
216 CASSERT(!gCampaign.IsClient, "Cannot process bounces as client");
217 FireGuns(obj, &obj->bulletClass->HitGuns);
218 if (hit.Type == HIT_WALL || !obj->bulletClass->Persists)
219 {
220 alive = false;
221 }
222 }
223 const Vec2i hitPos = hit.Type != HIT_NONE ? hit.Pos : pos;
224 b.u.BulletBounce.BouncePos = Vec2i2Net(hitPos);
225 b.u.BulletBounce.Pos = Vec2i2Net(pos);
226 b.u.BulletBounce.Vel = Vec2i2Net(obj->tileItem.VelFull);
227 if (hit.Type == HIT_WALL && !Vec2iIsZero(obj->tileItem.VelFull))
228 {
229 // Bouncing
230 GetWallBouncePosVelFull(
231 posStart, obj->tileItem.VelFull, hit.Pos, hit.Normal,
232 &pos, &obj->tileItem.VelFull);
233 b.u.BulletBounce.Pos = Vec2i2Net(pos);
234 b.u.BulletBounce.Vel = Vec2i2Net(obj->tileItem.VelFull);
235 }
236 b.u.BulletBounce.HitSound = obj->tileItem.SoundLock == 0;
237 if (obj->tileItem.SoundLock == 0)
238 {
239 obj->tileItem.SoundLock += SOUND_LOCK_TILE_OBJECT;
240 }
241 GameEventsEnqueue(&gGameEvents, b);
242 if (!alive)
243 {
244 return false;
245 }
246 }
247
248 const Vec2i realPos = Vec2iFull2Real(pos);
249
250 // Falling (grenades)
251 if (obj->bulletClass->Falling.GravityFactor != 0)
252 {
253 bool hasDropped = obj->z <= 0;
254 for (int i = 0; i < ticks; i++)
255 {
256 obj->z += obj->dz;
257 if (obj->z <= 0)
258 {
259 obj->z = 0;
260 if (obj->bulletClass->Falling.Bounces)
261 {
262 obj->dz = -obj->dz / 2;
263 }
264 else
265 {
266 obj->dz = 0;
267 }
268 if (!hasDropped)
269 {
270 if (!gCampaign.IsClient)
271 {
272 FireGuns(obj, &obj->bulletClass->Falling.DropGuns);
273 }
274 }
275 hasDropped = true;
276 if (obj->bulletClass->Falling.DestroyOnDrop)
277 {
278 return false;
279 }
280 SoundPlayAt(
281 &gSoundDevice,
282 StrSound(obj->bulletClass->HitSound.Wall), realPos);
283 }
284 else
285 {
286 obj->dz -= obj->bulletClass->Falling.GravityFactor;
287 }
288 if (!obj->bulletClass->Falling.FallsDown)
289 {
290 obj->dz = MAX(0, obj->dz);
291 }
292 }
293 }
294
295 // Friction
296 const bool isDiagonal =
297 obj->tileItem.VelFull.x != 0 && obj->tileItem.VelFull.y != 0;
298 const int frictionComponent = isDiagonal ?
299 (int)round(obj->bulletClass->Friction / sqrt(2)) :
300 obj->bulletClass->Friction;
301 for (int i = 0; i < ticks; i++)
302 {
303 if (obj->tileItem.VelFull.x > 0)
304 {
305 obj->tileItem.VelFull.x -= frictionComponent;
306 }
307 else if (obj->tileItem.VelFull.x < 0)
308 {
309 obj->tileItem.VelFull.x += frictionComponent;
310 }
311
312 if (obj->tileItem.VelFull.y > 0)
313 {
314 obj->tileItem.VelFull.y -= frictionComponent;
315 }
316 else if (obj->tileItem.VelFull.y < 0)
317 {
318 obj->tileItem.VelFull.y += frictionComponent;
319 }
320 }
321 if (!MapTryMoveTileItem(&gMap, &obj->tileItem, realPos))
322 {
323 obj->count = obj->range;
324 return false;
325 }
326 obj->x = pos.x;
327 obj->y = pos.y;
328
329 if (obj->bulletClass->Erratic)
330 {
331 for (int i = 0; i < ticks; i++)
332 {
333 obj->tileItem.VelFull.x += ((rand() % 3) - 1) * 128;
334 obj->tileItem.VelFull.y += ((rand() % 3) - 1) * 128;
335 }
336 }
337
338 // Proximity function, destroy
339 // Only check proximity every now and then
340 if (obj->bulletClass->ProximityGuns.size > 0 && !(obj->count & 3))
341 {
342 if (!gCampaign.IsClient)
343 {
344 // Detonate the mine if there are characters in the tiles around it
345 const Vec2i tv =
346 Vec2iToTile(Vec2iFull2Real(pos));
347 Vec2i dv;
348 for (dv.y = -1; dv.y <= 1; dv.y++)
349 {
350 for (dv.x = -1; dv.x <= 1; dv.x++)
351 {
352 const Vec2i dtv = Vec2iAdd(tv, dv);
353 if (!MapIsTileIn(&gMap, dtv))
354 {
355 continue;
356 }
357 if (TileHasCharacter(MapGetTile(&gMap, dtv)))
358 {
359 FireGuns(obj, &obj->bulletClass->ProximityGuns);
360 return false;
361 }
362 }
363 }
364 }
365 }
366
367 return true;
368}
369static void FireGuns(const TMobileObject *obj, const CArray *guns)
370{
371 const Vec2i fullPos = Vec2iNew(obj->x, obj->y);
372 const double angle = Vec2iToRadians(obj->tileItem.VelFull);
373 for (int i = 0; i < (int)guns->size; i++)
374 {
375 const GunDescription **g = CArrayGet(guns, i);
376 GunFire(
377 *g, fullPos, obj->z, angle, obj->flags, obj->PlayerUID, obj->ActorUID,
378 true, false);
379 }

381typedef struct
382{
383 HitType HitType;
384 bool MultipleHits;
385 TMobileObject *Obj;
386 union
387 {
388 TTileItem *Target;
389 Vec2i TilePos;
390 } u;
391 Vec2i ColPos;
392 Vec2i ColNormal;
393 int ColPosDistRealSquared;
394} HitItemData;
395static bool HitItemFunc(
396 TTileItem *ti, void *data, const Vec2i colA, const Vec2i colB,
397 const Vec2i normal);
398static bool CheckWall(const Vec2i tilePos);
399static bool HitWallFunc(
400 const Vec2i tilePos, void *data, const Vec2i col, const Vec2i normal);
401static void OnHit(HitItemData *data, TTileItem *target);
402static HitResult HitItem(
403 TMobileObject *obj, const Vec2i pos, const bool multipleHits)
404{
405 // Get all items that collide
406 HitItemData data;
407 data.HitType = HIT_NONE;
408 data.MultipleHits = multipleHits;
409 data.Obj = obj;
410 data.ColPos = pos;
411 data.ColNormal = Vec2iZero();
412 data.ColPosDistRealSquared = -1;
413 const CollisionParams params =
414 {
415 TILEITEM_CAN_BE_SHOT, COLLISIONTEAM_NONE, IsPVP(gCampaign.Entry.Mode)
416 };
417 OverlapTileItems(
418 &obj->tileItem, pos,
419 obj->tileItem.size, params, HitItemFunc, &data,
420 CheckWall, HitWallFunc, &data);
421 if (!multipleHits && data.ColPosDistRealSquared >= 0)
422 {
423 if (data.HitType == HIT_OBJECT || data.HitType == HIT_FLESH)
424 {
425 OnHit(&data, data.u.Target);
426 }
427 }
428 HitResult hit = { data.HitType, data.ColPos, data.ColNormal };
429 return hit;
430}
431static HitType GetHitType(
432 const TTileItem *ti, const TMobileObject *bullet, int *targetUID);
433static void SetClosestCollision(
434 HitItemData *data, const Vec2i col, const Vec2i normal, HitType ht,
435 TTileItem *target, const Vec2i tilePos);
436static bool HitItemFunc(
437 TTileItem *ti, void *data, const Vec2i colA, const Vec2i colB,
438 const Vec2i normal)
439{
440 UNUSED(colB);
441 HitItemData *hData = data;
442
443 // Check bullet-to-other collisions
444 if (!CanHit(hData->Obj->flags, hData->Obj->ActorUID, ti))
445 {
446 goto bail;
447 }
448
449 // If we can hit multiple targets, just process those hits immediately
450 // Otherwise, find the closest target and only process the hit for that one
451 // at the end.
452 if (hData->MultipleHits)
453 {
454 OnHit(hData, ti);
455 }
456 else
457 {
458 SetClosestCollision(
459 hData, colA, normal, GetHitType(ti, hData->Obj, NULL),
460 ti, Vec2iZero());
461 }
462
463bail:
464 return true;
465}
466static HitType GetHitType(
467 const TTileItem *ti, const TMobileObject *bullet, int *targetUID)
468{
469 int tUID = -1;
470 HitType ht = HIT_NONE;
471 switch (ti->kind)
472 {
473 case KIND_CHARACTER:
474 ht = HIT_FLESH;
475 tUID = ((const TActor *)CArrayGet(&gActors, ti->id))->uid;
476 break;
477 case KIND_OBJECT:
478 ht = HIT_OBJECT;
479 tUID = ((const TObject *)CArrayGet(&gObjs, ti->id))->uid;
480 break;
481 default:
482 CASSERT(false, "cannot damage target kind");
483 break;
484 }
485 if (bullet->tileItem.SoundLock > 0 ||
486 !HasHitSound(
487 bullet->flags, bullet->PlayerUID, ti->kind, tUID,
488 bullet->bulletClass->Special, true))
489 {
490 ht = HIT_NONE;
491 }
492 if (targetUID != NULL)
493 {
494 *targetUID = tUID;
495 }
496 return ht;
497}
498static bool CheckWall(const Vec2i tilePos)
499{
500 const Tile *t = MapGetTile(&gMap, tilePos);
501 return t == NULL || t->flags & MAPTILE_NO_SHOOT;
502}
503static bool HitWallFunc(
504 const Vec2i tilePos, void *data, const Vec2i col, const Vec2i normal)
505{
506 HitItemData *hData = data;
507
508 SetClosestCollision(hData, col, normal, HIT_WALL, NULL, tilePos);
509
510 return true;
511}
512static void SetClosestCollision(
513 HitItemData *data, const Vec2i col, const Vec2i normal, HitType ht,
514 TTileItem *target, const Vec2i tilePos)
515{
516 // Choose the best collision point (i.e. closest to origin)
517 const int d2 = DistanceSquared(
518 Vec2iFull2Real(col),
519 Vec2iFull2Real(Vec2iNew(data->Obj->x, data->Obj->y)));
520 if (data->ColPosDistRealSquared < 0 || d2 < data->ColPosDistRealSquared)
521 {
522 data->ColPos = col;
523 data->ColPosDistRealSquared = d2;
524 data->ColNormal = normal;
525 data->HitType = ht;
526 if (ht == HIT_WALL)
527 {
528 data->u.TilePos = tilePos;
529 }
530 else
531 {
532 data->u.Target = target;
533 }
534 }
535}
536static void OnHit(HitItemData *data, TTileItem *target)
537{
538 int targetUID = -1;
539 data->HitType = GetHitType(target, data->Obj, &targetUID);
540 Damage(
541 data->Obj->tileItem.VelFull,
542 data->Obj->bulletClass->Power, data->Obj->bulletClass->Mass,
543 data->Obj->flags, data->Obj->PlayerUID, data->Obj->ActorUID,
544 target->kind, targetUID,
545 data->Obj->bulletClass->Special);
546 if (data->Obj->tileItem.SoundLock <= 0)
547 {
548 data->Obj->tileItem.SoundLock += SOUND_LOCK_TILE_OBJECT;
549 }
550 if (target->SoundLock <= 0)
551 {
552 target->SoundLock += SOUND_LOCK_TILE_OBJECT;
553 }
554 if (data->Obj->specialLock <= 0)
555 {
556 data->Obj->specialLock += SPECIAL_LOCK;
557 }
558}

571static void BulletClassFree(BulletClass *b);
572void BulletLoadJSON(
573 BulletClasses *bullets, CArray *classes, json_t *bulletNode)
574{
575 LOG(LM_MAP, LL_DEBUG, "loading bullets");
576 int version;
577 LoadInt(&version, bulletNode, "Version");
578 if (version > VERSION || version <= 0)
579 {
580 CASSERT(false, "cannot read bullets file version");
581 return;
582 }
583
584 // Defaults
585 json_t *defaultNode = json_find_first_label(bulletNode, "DefaultBullet");
586 if (defaultNode != NULL)
587 {
588 BulletClassFree(&bullets->Default);
589 LoadBullet(&bullets->Default, defaultNode->child, NULL, version);
590 }
591
592 json_t *bulletsNode = json_find_first_label(bulletNode, "Bullets")->child;
593 for (json_t *child = bulletsNode->child; child; child = child->next)
594 {
595 BulletClass b;
596 LoadBullet(&b, child, &bullets->Default, version);
597 CArrayPushBack(classes, &b);
598 }
599
600 bullets->root = bulletNode;
601}

603 char **hitsound, json_t *node, const char *name, const int version);
604static void LoadBullet(
605 BulletClass *b, json_t *node, const BulletClass *defaultBullet,
606 const int version)
607{
608 memset(b, 0, sizeof *b);
609 if (defaultBullet != NULL)
610 {
611 memcpy(b, defaultBullet, sizeof *b);
612 if (defaultBullet->Name != NULL)
613 {
614 CSTRDUP(b->Name, defaultBullet->Name);
615 }
616 if (defaultBullet->HitSound.Object != NULL)
617 {
618 CSTRDUP(b->HitSound.Object, defaultBullet->HitSound.Object);
619 }
620 if (defaultBullet->HitSound.Flesh != NULL)
621 {
622 CSTRDUP(b->HitSound.Flesh, defaultBullet->HitSound.Flesh);
623 }
624 if (defaultBullet->HitSound.Wall != NULL)
625 {
626 CSTRDUP(b->HitSound.Wall, defaultBullet->HitSound.Wall);
627 }
628 // TODO: enable default bullet guns?
629 memset(&b->Falling.DropGuns, 0, sizeof b->Falling.DropGuns);
630 memset(&b->OutOfRangeGuns, 0, sizeof b->OutOfRangeGuns);
631 memset(&b->HitGuns, 0, sizeof b->HitGuns);
632 memset(&b->ProximityGuns, 0, sizeof b->ProximityGuns);
633 }
634
635 char *tmp;
636
637 tmp = NULL;
638 LoadStr(&tmp, node, "Name");
639 if (tmp != NULL)
640 {
641 CFREE(b->Name);
642 b->Name = tmp;
643 }
644 if (json_find_first_label(node, "Pic"))
645 {
646 CPicLoadJSON(&b->CPic, json_find_first_label(node, "Pic")->child);
647 }
648 LoadVec2i(&b->ShadowSize, node, "ShadowSize");
649 LoadInt(&b->Delay, node, "Delay");
650 if (json_find_first_label(node, "Speed"))
651 {
652 LoadInt(&b->SpeedLow, node, "Speed");
653 b->SpeedHigh = b->SpeedLow;
654 }
655 LoadInt(&b->SpeedLow, node, "SpeedLow");
656 LoadInt(&b->SpeedHigh, node, "SpeedHigh");
657 b->SpeedLow = MIN(b->SpeedLow, b->SpeedHigh);
658 b->SpeedHigh = MAX(b->SpeedLow, b->SpeedHigh);
659 LoadBool(&b->SpeedScale, node, "SpeedScale");
660 LoadInt(&b->Friction, node, "Friction");
661 if (json_find_first_label(node, "Range"))
662 {
663 LoadInt(&b->RangeLow, node, "Range");
664 b->RangeHigh = b->RangeLow;
665 }
666 LoadInt(&b->RangeLow, node, "RangeLow");
667 LoadInt(&b->RangeHigh, node, "RangeHigh");
668 b->RangeLow = MIN(b->RangeLow, b->RangeHigh);
669 b->RangeHigh = MAX(b->RangeLow, b->RangeHigh);
670 LoadInt(&b->Power, node, "Power");
671
672 if (version < 2)
673 {
674 // Old version default mass = power
675 b->Mass = b->Power;
676 }
677 else
678 {
679 LoadDouble(&b->Mass, node, "Mass");
680 }
681
682 LoadVec2i(&b->Size, node, "Size");
683 tmp = NULL;
684 LoadStr(&tmp, node, "Special");
685 if (tmp != NULL)
686 {
687 b->Special = StrSpecialDamage(tmp);
688 CFREE(tmp);
689 }
690 LoadBool(&b->HurtAlways, node, "HurtAlways");
691 LoadBool(&b->Persists, node, "Persists");
692 tmp = NULL;
693 LoadStr(&tmp, node, "Spark");
694 if (tmp != NULL)
695 {
696 b->Spark = StrParticleClass(&gParticleClasses, tmp);
697 CFREE(tmp);
698 }
699 tmp = NULL;
700 LoadStr(&tmp, node, "OutOfRangeSpark");
701 if (tmp != NULL)
702 {
703 b->OutOfRangeSpark = StrParticleClass(&gParticleClasses, tmp);
704 CFREE(tmp);
705 }
706 if (json_find_first_label(node, "HitSounds"))
707 {
708 json_t *hitSounds = json_find_first_label(node, "HitSounds")->child;
709 LoadHitsound(&b->HitSound.Object, hitSounds, "Object", version);
710 LoadHitsound(&b->HitSound.Flesh, hitSounds, "Flesh", version);
711 LoadHitsound(&b->HitSound.Wall, hitSounds, "Wall", version);
712 }
713 LoadBool(&b->WallBounces, node, "WallBounces");
714 LoadBool(&b->HitsObjects, node, "HitsObjects");
715 if (json_find_first_label(node, "Falling"))
716 {
717 json_t *falling = json_find_first_label(node, "Falling")->child;
718 LoadInt(&b->Falling.GravityFactor, falling, "GravityFactor");
719 LoadBool(&b->Falling.FallsDown, falling, "FallsDown");
720 LoadBool(&b->Falling.DestroyOnDrop, falling, "DestroyOnDrop");
721 LoadBool(&b->Falling.Bounces, falling, "Bounces");
722 }
723 LoadInt(&b->SeekFactor, node, "SeekFactor");

851void BulletAdd(const NAddBullet add)
852{
853 const Vec2i pos = Net2Vec2i(add.MuzzlePos);
854
855 // Find an empty slot in mobobj list
856 TMobileObject *obj = NULL;
857 int i;
858 for (i = 0; i < (int)gMobObjs.size; i++)
859 {
860 TMobileObject *m = CArrayGet(&gMobObjs, i);
861 if (!m->isInUse)
862 {
863 obj = m;
864 break;
865 }
866 }
867 if (obj == NULL)
868 {
869 TMobileObject m;
870 memset(&m, 0, sizeof m);
871 CArrayPushBack(&gMobObjs, &m);
872 i = (int)gMobObjs.size - 1;
873 obj = CArrayGet(&gMobObjs, i);
874 }
875 memset(obj, 0, sizeof *obj);
876 obj->UID = add.UID;
877 obj->bulletClass = StrBulletClass(add.BulletClass);
878 obj->x = pos.x;
879 obj->y = pos.y;
880 obj->z = add.MuzzleHeight;
881 obj->dz = add.Elevation;
882
883 obj->tileItem.VelFull = Vec2iFull2Real(Vec2iScale(
884 GetFullVectorsForRadians(add.Angle),
885 RAND_INT(obj->bulletClass->SpeedLow, obj->bulletClass->SpeedHigh)));
886 if (obj->bulletClass->SpeedScale)
887 {
888 obj->tileItem.VelFull.y =
889 obj->tileItem.VelFull.y * TILE_WIDTH / TILE_HEIGHT;
890 }
891
892 obj->PlayerUID = add.PlayerUID;
893 obj->ActorUID = add.ActorUID;
894 obj->range = RAND_INT(
895 obj->bulletClass->RangeLow, obj->bulletClass->RangeHigh);
896
897 obj->flags = add.Flags;
898 if (obj->bulletClass->HurtAlways)
899 {
900 obj->flags |= FLAGS_HURTALWAYS;
901 }
902
903 obj->tileItem.kind = KIND_MOBILEOBJECT;
904 obj->tileItem.id = i;
905 obj->isInUse = true;
906 obj->tileItem.x = obj->tileItem.y = -1;
907 obj->tileItem.getPicFunc = NULL;
908 obj->tileItem.drawFunc = NULL;
909 obj->tileItem.drawData.MobObjId = i;
910 obj->tileItem.CPic = obj->bulletClass->CPic;
911 obj->tileItem.CPicFunc = GetBulletDrawContext;
912 obj->tileItem.size = obj->bulletClass->Size;
913 obj->tileItem.ShadowSize = obj->bulletClass->ShadowSize;
914 obj->updateFunc = UpdateBullet;
915 MapTryMoveTileItem(&gMap, &obj->tileItem, Vec2iFull2Real(pos));
916}

936 break;
937 }
