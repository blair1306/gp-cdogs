29#pragma once
30
31#include <SDL_stdinc.h>
32
33#include "pic.h"
34#include "vector.h"
35
36typedef struct
37{
38 bool previousButtons[8];
39 bool currentButtons[8];
40 bool pressedButtons[8];
41 Vec2i previousPos;
42 Vec2i currentPos;
43 Vec2i wheel;
44 Pic *cursor;
45 Pic *trail;
46 Uint32 ticks;
47 Uint32 repeatedTicks;
48
49 bool hideMouse;
50 Vec2i mouseMovePos;
51} Mouse;
52
53void MouseInit(Mouse *mouse, Pic *cursor, Pic *trail, const bool hideMouse);
54void MousePrePoll(Mouse *mouse);
55void MouseOnButtonDown(Mouse *mouse, Uint8 button);
56void MouseOnButtonUp(Mouse *mouse, Uint8 button);
57void MouseOnWheel(Mouse *m, const Sint32 x, const Sint32 y);
58void MousePostPoll(Mouse *mouse, Uint32 ticks);
59bool MouseHasMoved(const Mouse *m);
60int MouseGetPressed(const Mouse *m);
61bool MouseIsDown(const Mouse *m, const int button);
62bool MouseIsPressed(const Mouse *m, const int button);
63// Get wheel movement since last poll
64Vec2i MouseWheel(const Mouse *m);
65// Get mouse movement from a screen position
66// Note: also sets whether the mouse trail is drawn, and from where
67int MouseGetMove(Mouse *mouse, const Vec2i pos);
68void MouseDraw(const Mouse *mouse);
