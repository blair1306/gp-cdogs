72static bool checkreturn buf_read(pb_istream_t *stream, uint8_t *buf, size_t count)
73{
74 uint8_t *source = (uint8_t*)stream->state;
75 stream->state = source + count;
76
77 if (buf != NULL)
78 {
79 while (count--)
80 *buf++ = *source++;
81 }
82
83 return true;
84}
85
86bool checkreturn pb_read(pb_istream_t *stream, uint8_t *buf, size_t count)
87{
88#ifndef PB_BUFFER_ONLY
89 if (buf == NULL && stream->callback != buf_read)
90 {
91 /* Skip input bytes */
92 uint8_t tmp[16];
93 while (count > 16)
94 {
95 if (!pb_read(stream, tmp, 16))
96 return false;
97
98 count -= 16;
99 }
100
101 return pb_read(stream, tmp, count);
102 }
103#endif
104
105 if (stream->bytes_left < count)
106 PB_RETURN_ERROR(stream, "end-of-stream");
107
108#ifndef PB_BUFFER_ONLY
109 if (!stream->callback(stream, buf, count))
110 PB_RETURN_ERROR(stream, "io error");
111#else
112 if (!buf_read(stream, buf, count))
113 return false;
114#endif
115
116 stream->bytes_left -= count;
117 return true;
118}

120/* Read a single byte from input stream. buf may not be NULL.
121 * This is an optimization for the varint decoding. */
122static bool checkreturn pb_readbyte(pb_istream_t *stream, uint8_t *buf)
123{
124 if (stream->bytes_left == 0)
125 PB_RETURN_ERROR(stream, "end-of-stream");
126
127#ifndef PB_BUFFER_ONLY
128 if (!stream->callback(stream, buf, 1))
129 PB_RETURN_ERROR(stream, "io error");
130#else
131 *buf = *(uint8_t*)stream->state;
132 stream->state = (uint8_t*)stream->state + 1;
133#endif
134
135 stream->bytes_left--;
136
137 return true;
138}

140pb_istream_t pb_istream_from_buffer(uint8_t *buf, size_t bufsize)
141{
142 pb_istream_t stream;
143#ifdef PB_BUFFER_ONLY
144 stream.callback = NULL;
145#else
146 stream.callback = &buf_read;
147#endif
148 stream.state = buf;
149 stream.bytes_left = bufsize;
150#ifndef PB_NO_ERRMSG
151 stream.errmsg = NULL;
152#endif
153 return stream;
154}

160static bool checkreturn pb_decode_varint32(pb_istream_t *stream, uint32_t *dest)
161{
162 uint8_t byte;
163 uint32_t result;
164
165 if (!pb_readbyte(stream, &byte))
166 return false;
167
168 if ((byte & 0x80) == 0)
169 {
170 /* Quick case, 1 byte value */
171 result = byte;
172 }
173 else
174 {
175 /* Multibyte case */
176 uint8_t bitpos = 7;
177 result = byte & 0x7F;
178
179 do
180 {
181 if (bitpos >= 32)
182 PB_RETURN_ERROR(stream, "varint overflow");
183
184 if (!pb_readbyte(stream, &byte))
185 return false;
186
187 result |= (uint32_t)(byte & 0x7F) << bitpos;
188 bitpos = (uint8_t)(bitpos + 7);
189 } while (byte & 0x80);
190 }
191
192 *dest = result;
193 return true;
194}
195
196bool checkreturn pb_decode_varint(pb_istream_t *stream, uint64_t *dest)
197{
198 uint8_t byte;
199 uint8_t bitpos = 0;
200 uint64_t result = 0;
201
202 do
203 {
204 if (bitpos >= 64)
205 PB_RETURN_ERROR(stream, "varint overflow");
206
207 if (!pb_readbyte(stream, &byte))
208 return false;
209
210 result |= (uint64_t)(byte & 0x7F) << bitpos;
211 bitpos = (uint8_t)(bitpos + 7);
212 } while (byte & 0x80);
213
214 *dest = result;
215 return true;
216}

218bool checkreturn pb_skip_varint(pb_istream_t *stream)
219{
220 uint8_t byte;
221 do
222 {
223 if (!pb_read(stream, &byte, 1))
224 return false;
225 } while (byte & 0x80);
226 return true;
227}
228
229bool checkreturn pb_skip_string(pb_istream_t *stream)
230{
231 uint32_t length;
232 if (!pb_decode_varint32(stream, &length))
233 return false;
234
235 return pb_read(stream, NULL, length);
236}

238bool checkreturn pb_decode_tag(pb_istream_t *stream, pb_wire_type_t *wire_type, uint32_t *tag, bool *eof)
239{
240 uint32_t temp;
241 *eof = false;
242 *wire_type = (pb_wire_type_t) 0;
243 *tag = 0;
244
245 if (!pb_decode_varint32(stream, &temp))
246 {
247 if (stream->bytes_left == 0)
248 *eof = true;
249
250 return false;
251 }
252
253 if (temp == 0)
254 {
255 *eof = true; /* Special feature: allow 0-terminated messages. */
256 return false;
257 }
258
259 *tag = temp >> 3;
260 *wire_type = (pb_wire_type_t)(temp & 7);
261 return true;
262}

264bool checkreturn pb_skip_field(pb_istream_t *stream, pb_wire_type_t wire_type)
265{
266 switch (wire_type)
267 {
268 case PB_WT_VARINT: return pb_skip_varint(stream);
269 case PB_WT_64BIT: return pb_read(stream, NULL, 8);
270 case PB_WT_STRING: return pb_skip_string(stream);
271 case PB_WT_32BIT: return pb_read(stream, NULL, 4);
272 default: PB_RETURN_ERROR(stream, "invalid wire_type");
273 }
274}

276/* Read a raw value to buffer, for the purpose of passing it to callback as
277 * a substream. Size is maximum size on call, and actual size on return.
278 */
279static bool checkreturn read_raw_value(pb_istream_t *stream, pb_wire_type_t wire_type, uint8_t *buf, size_t *size)
280{
281 size_t max_size = *size;
282 switch (wire_type)
283 {
284 case PB_WT_VARINT:
285 *size = 0;
286 do
287 {
288 (*size)++;
289 if (*size > max_size) return false;
290 if (!pb_read(stream, buf, 1)) return false;
291 } while (*buf++ & 0x80);
292 return true;
293
294 case PB_WT_64BIT:
295 *size = 8;
296 return pb_read(stream, buf, 8);
297
298 case PB_WT_32BIT:
299 *size = 4;
300 return pb_read(stream, buf, 4);
301
302 default: PB_RETURN_ERROR(stream, "invalid wire_type");
303 }
304}

306/* Decode string length from stream and return a substream with limited length.
307 * Remember to close the substream using pb_close_string_substream().
308 */
309bool checkreturn pb_make_string_substream(pb_istream_t *stream, pb_istream_t *substream)
310{
311 uint32_t size;
312 if (!pb_decode_varint32(stream, &size))
313 return false;
314
315 *substream = *stream;
316 if (substream->bytes_left < size)
317 PB_RETURN_ERROR(stream, "parent stream too short");
318
319 substream->bytes_left = size;
320 stream->bytes_left -= size;
321 return true;
322}
323
324void pb_close_string_substream(pb_istream_t *stream, pb_istream_t *substream)
325{
326 stream->state = substream->state;
327
328#ifndef PB_NO_ERRMSG
329 stream->errmsg = substream->errmsg;
330#endif
331}

337static bool checkreturn decode_static_field(pb_istream_t *stream, pb_wire_type_t wire_type, pb_field_iter_t *iter)
338{
339 pb_type_t type;
340 pb_decoder_t func;
341
342 type = iter->pos->type;
343 func = PB_DECODERS[PB_LTYPE(type)];
344
345 switch (PB_HTYPE(type))
346 {
347 case PB_HTYPE_REQUIRED:
348 return func(stream, iter->pos, iter->pData);
349
350 case PB_HTYPE_OPTIONAL:
351 *(bool*)iter->pSize = true;
352 return func(stream, iter->pos, iter->pData);
353
354 case PB_HTYPE_REPEATED:
355 if (wire_type == PB_WT_STRING
356 && PB_LTYPE(type) <= PB_LTYPE_LAST_PACKABLE)
357 {
358 /* Packed array */
359 bool status = true;
360 pb_size_t *size = (pb_size_t*)iter->pSize;
361 pb_istream_t substream;
362 if (!pb_make_string_substream(stream, &substream))
363 return false;
364
365 while (substream.bytes_left > 0 && *size < iter->pos->array_size)
366 {
367 void *pItem = (uint8_t*)iter->pData + iter->pos->data_size * (*size);
368 if (!func(&substream, iter->pos, pItem))
369 {
370 status = false;
371 break;
372 }
373 (*size)++;
374 }
375 pb_close_string_substream(stream, &substream);
376
377 if (substream.bytes_left != 0)
378 PB_RETURN_ERROR(stream, "array overflow");
379
380 return status;
381 }
382 else
383 {
384 /* Repeated field */
385 pb_size_t *size = (pb_size_t*)iter->pSize;
386 void *pItem = (uint8_t*)iter->pData + iter->pos->data_size * (*size);
387 if (*size >= iter->pos->array_size)
388 PB_RETURN_ERROR(stream, "array overflow");
389
390 (*size)++;
391 return func(stream, iter->pos, pItem);
392 }
393
394 default:
395 PB_RETURN_ERROR(stream, "invalid field type");
396 }
397}

399#ifdef PB_ENABLE_MALLOC
400/* Allocate storage for the field and store the pointer at iter->pData.
401 * array_size is the number of entries to reserve in an array.
402 * Zero size is not allowed, use pb_free() for releasing.
403 */
404static bool checkreturn allocate_field(pb_istream_t *stream, void *pData, size_t data_size, size_t array_size)
405{
406 void *ptr = *(void**)pData;
407
408 /* Check for multiplication overflows.
409 * This code avoids the costly division if the sizes are small enough.
410 * Multiplication is safe as long as only half of bits are set
411 * in either multiplicand.
412 */
413 const size_t check_limit = (size_t)1 << (sizeof(size_t) * 4);
414 if (data_size >= check_limit || array_size >= check_limit)
415 {
416 const size_t size_max = (size_t)-1;
417 if (size_max / array_size < data_size)
418 {
419 PB_RETURN_ERROR(stream, "size too large");
420 }
421 }
422
423 /* Allocate new or expand previous allocation */
424 /* Note: on failure the old pointer will remain in the structure,
425 * the message must be freed by caller also on error return. */
426 ptr = pb_realloc(ptr, array_size * data_size);
427 if (ptr == NULL)
428 PB_RETURN_ERROR(stream, "realloc failed");
429
430 *(void**)pData = ptr;
431 return true;
432}

434/* Clear a newly allocated item in case it contains a pointer, or is a submessage. */
435static void initialize_pointer_field(void *pItem, pb_field_iter_t *iter)
436{
437 if (PB_LTYPE(iter->pos->type) == PB_LTYPE_STRING ||
438 PB_LTYPE(iter->pos->type) == PB_LTYPE_BYTES)
439 {
440 *(void**)pItem = NULL;
441 }
442 else if (PB_LTYPE(iter->pos->type) == PB_LTYPE_SUBMESSAGE)
443 {
444 pb_message_set_to_defaults((const pb_field_t *) iter->pos->ptr, pItem);
445 }
446}
447#endif

449static bool checkreturn decode_pointer_field(pb_istream_t *stream, pb_wire_type_t wire_type, pb_field_iter_t *iter)
450{
451#ifndef PB_ENABLE_MALLOC
452 PB_UNUSED(wire_type);
453 PB_UNUSED(iter);
454 PB_RETURN_ERROR(stream, "no malloc support");
455#else
456 pb_type_t type;
457 pb_decoder_t func;
458
459 type = iter->pos->type;
460 func = PB_DECODERS[PB_LTYPE(type)];
461
462 switch (PB_HTYPE(type))
463 {
464 case PB_HTYPE_REQUIRED:
465 case PB_HTYPE_OPTIONAL:
466 if (PB_LTYPE(type) == PB_LTYPE_STRING ||
467 PB_LTYPE(type) == PB_LTYPE_BYTES)
468 {
469 return func(stream, iter->pos, iter->pData);
470 }
471 else
472 {
473 if (!allocate_field(stream, iter->pData, iter->pos->data_size, 1))
474 return false;
475
476 initialize_pointer_field(*(void**)iter->pData, iter);
477 return func(stream, iter->pos, *(void**)iter->pData);
478 }
479
480 case PB_HTYPE_REPEATED:
481 if (wire_type == PB_WT_STRING
482 && PB_LTYPE(type) <= PB_LTYPE_LAST_PACKABLE)
483 {
484 /* Packed array, multiple items come in at once. */
485 bool status = true;
486 pb_size_t *size = (pb_size_t*)iter->pSize;
487 size_t allocated_size = *size;
488 void *pItem;
489 pb_istream_t substream;
490
491 if (!pb_make_string_substream(stream, &substream))
492 return false;
493
494 while (substream.bytes_left)
495 {
496 if ((size_t)*size + 1 > allocated_size)
497 {
498 /* Allocate more storage. This tries to guess the
499 * number of remaining entries. Round the division
500 * upwards. */
501 allocated_size += (substream.bytes_left - 1) / iter->pos->data_size + 1;
502
503 if (!allocate_field(&substream, iter->pData, iter->pos->data_size, allocated_size))
504 {
505 status = false;
506 break;
507 }
508 }
509
510 /* Decode the array entry */
511 pItem = *(uint8_t**)iter->pData + iter->pos->data_size * (*size);
512 initialize_pointer_field(pItem, iter);
513 if (!func(&substream, iter->pos, pItem))
514 {
515 status = false;
516 break;
517 }
518
519 if (*size == PB_SIZE_MAX)
520 {
521#ifndef PB_NO_ERRMSG
522 stream->errmsg = "too many array entries";
523#endif
524 status = false;
525 break;
526 }
527
528 (*size)++;
529 }
530 pb_close_string_substream(stream, &substream);
531
532 return status;
533 }
534 else
535 {
536 /* Normal repeated field, i.e. only one item at a time. */
537 pb_size_t *size = (pb_size_t*)iter->pSize;
538 void *pItem;
539
540 if (*size == PB_SIZE_MAX)
541 PB_RETURN_ERROR(stream, "too many array entries");
542
543 (*size)++;
544 if (!allocate_field(stream, iter->pData, iter->pos->data_size, *size))
545 return false;
546
547 pItem = *(uint8_t**)iter->pData + iter->pos->data_size * (*size - 1);
548 initialize_pointer_field(pItem, iter);
549 return func(stream, iter->pos, pItem);
550 }
551
552 default:
553 PB_RETURN_ERROR(stream, "invalid field type");
554 }
555#endif
556}

558static bool checkreturn decode_callback_field(pb_istream_t *stream, pb_wire_type_t wire_type, pb_field_iter_t *iter)
559{
560 pb_callback_t *pCallback = (pb_callback_t*)iter->pData;
561
562#ifdef PB_OLD_CALLBACK_STYLE
563 void *arg = pCallback->arg;
564#else
565 void **arg = &(pCallback->arg);
566#endif
567
568 if (pCallback->funcs.decode == NULL)
569 return pb_skip_field(stream, wire_type);
570
571 if (wire_type == PB_WT_STRING)
572 {
573 pb_istream_t substream;
574
575 if (!pb_make_string_substream(stream, &substream))
576 return false;
577
578 do
579 {
580 if (!pCallback->funcs.decode(&substream, iter->pos, arg))
581 PB_RETURN_ERROR(stream, "callback failed");
582 } while (substream.bytes_left);
583
584 pb_close_string_substream(stream, &substream);
585 return true;
586 }
587 else
588 {
589 /* Copy the single scalar value to stack.
590 * This is required so that we can limit the stream length,
591 * which in turn allows to use same callback for packed and
592 * not-packed fields. */
593 pb_istream_t substream;
594 uint8_t buffer[10];
595 size_t size = sizeof(buffer);
596
597 if (!read_raw_value(stream, wire_type, buffer, &size))
598 return false;
599 substream = pb_istream_from_buffer(buffer, size);
600
601 return pCallback->funcs.decode(&substream, iter->pos, arg);
602 }
603}

605static bool checkreturn decode_field(pb_istream_t *stream, pb_wire_type_t wire_type, pb_field_iter_t *iter)
606{
607 switch (PB_ATYPE(iter->pos->type))
608 {
609 case PB_ATYPE_STATIC:
610 return decode_static_field(stream, wire_type, iter);
611
612 case PB_ATYPE_POINTER:
613 return decode_pointer_field(stream, wire_type, iter);
614
615 case PB_ATYPE_CALLBACK:
616 return decode_callback_field(stream, wire_type, iter);
617
618 default:
619 PB_RETURN_ERROR(stream, "invalid field type");
620 }
621}

623/* Default handler for extension fields. Expects a pb_field_t structure
624 * in extension->type->arg. */
625static bool checkreturn default_extension_decoder(pb_istream_t *stream,
626 pb_extension_t *extension, uint32_t tag, pb_wire_type_t wire_type)
627{
628 const pb_field_t *field = (const pb_field_t*)extension->type->arg;
629 pb_field_iter_t iter;
630
631 if (field->tag != tag)
632 return true;
633
634 /* Fake a field iterator for the extension field.
635 * It is not actually safe to advance this iterator, but decode_field
636 * will not even try to. */
637 (void)pb_field_iter_begin(&iter, field, extension->dest);
638 iter.pData = extension->dest;
639 iter.pSize = &extension->found;
640
641 return decode_field(stream, wire_type, &iter);
642}

644/* Try to decode an unknown field as an extension field. Tries each extension
645 * decoder in turn, until one of them handles the field or loop ends. */
646static bool checkreturn decode_extension(pb_istream_t *stream,
647 uint32_t tag, pb_wire_type_t wire_type, pb_field_iter_t *iter)
648{
649 pb_extension_t *extension = *(pb_extension_t* const *)iter->pData;
650 size_t pos = stream->bytes_left;
651
652 while (extension != NULL && pos == stream->bytes_left)
653 {
654 bool status;
655 if (extension->type->decode)
656 status = extension->type->decode(stream, extension, tag, wire_type);
657 else
658 status = default_extension_decoder(stream, extension, tag, wire_type);
659
660 if (!status)
661 return false;
662
663 extension = extension->next;
664 }
665
666 return true;
667}

669/* Step through the iterator until an extension field is found or until all
670 * entries have been checked. There can be only one extension field per
671 * message. Returns false if no extension field is found. */
672static bool checkreturn find_extension_field(pb_field_iter_t *iter)
673{
674 const pb_field_t *start = iter->pos;
675
676 do {
677 if (PB_LTYPE(iter->pos->type) == PB_LTYPE_EXTENSION)
678 return true;
679 (void)pb_field_iter_next(iter);
680 } while (iter->pos != start);
681
682 return false;
683}

685/* Initialize message fields to default values, recursively */
686static void pb_message_set_to_defaults(const pb_field_t fields[], void *dest_struct)
687{
688 pb_field_iter_t iter;
689
690 if (!pb_field_iter_begin(&iter, fields, dest_struct))
691 return; /* Empty message type */
692
693 do
694 {
695 pb_type_t type;
696 type = iter.pos->type;
697
698 if (PB_ATYPE(type) == PB_ATYPE_STATIC)
699 {
700 if (PB_HTYPE(type) == PB_HTYPE_OPTIONAL)
701 {
702 /* Set has_field to false. Still initialize the optional field
703 * itself also. */
704 *(bool*)iter.pSize = false;
705 }
706 else if (PB_HTYPE(type) == PB_HTYPE_REPEATED)
707 {
708 /* Set array count to 0, no need to initialize contents. */
709 *(pb_size_t*)iter.pSize = 0;
710 continue;
711 }
712
713 if (PB_LTYPE(iter.pos->type) == PB_LTYPE_SUBMESSAGE)
714 {
715 /* Initialize submessage to defaults */
716 pb_message_set_to_defaults((const pb_field_t *) iter.pos->ptr, iter.pData);
717 }
718 else if (iter.pos->ptr != NULL)
719 {
720 /* Initialize to default value */
721 memcpy(iter.pData, iter.pos->ptr, iter.pos->data_size);
722 }
723 else
724 {
725 /* Initialize to zeros */
726 memset(iter.pData, 0, iter.pos->data_size);
727 }
728 }
729 else if (PB_ATYPE(type) == PB_ATYPE_POINTER)
730 {
731 /* Initialize the pointer to NULL. */
732 *(void**)iter.pData = NULL;
733
734 /* Initialize array count to 0. */
735 if (PB_HTYPE(type) == PB_HTYPE_REPEATED)
736 {
737 *(pb_size_t*)iter.pSize = 0;
738 }
739 }
740 else if (PB_ATYPE(type) == PB_ATYPE_CALLBACK)
741 {
742 /* Don't overwrite callback */
743 }
744 } while (pb_field_iter_next(&iter));
745}

747/*********************
748 * Decode all fields *
749 *********************/
750
751bool checkreturn pb_decode_noinit(pb_istream_t *stream, const pb_field_t fields[], void *dest_struct)
752{
753 uint8_t fields_seen[(PB_MAX_REQUIRED_FIELDS + 7) / 8] = {0, 0, 0, 0, 0, 0, 0, 0};
754 uint32_t extension_range_start = 0;
755 pb_field_iter_t iter;
756
757 /* Return value ignored, as empty message types will be correctly handled by
758 * pb_field_iter_find() anyway. */
759 (void)pb_field_iter_begin(&iter, fields, dest_struct);
760
761 while (stream->bytes_left)
762 {
763 uint32_t tag;
764 pb_wire_type_t wire_type;
765 bool eof;
766
767 if (!pb_decode_tag(stream, &wire_type, &tag, &eof))
768 {
769 if (eof)
770 break;
771 else
772 return false;
773 }
774
775 if (!pb_field_iter_find(&iter, tag))
776 {
777 /* No match found, check if it matches an extension. */
778 if (tag >= extension_range_start)
779 {
780 if (!find_extension_field(&iter))
781 extension_range_start = (uint32_t)-1;
782 else
783 extension_range_start = iter.pos->tag;
784
785 if (tag >= extension_range_start)
786 {
787 size_t pos = stream->bytes_left;
788
789 if (!decode_extension(stream, tag, wire_type, &iter))
790 return false;
791
792 if (pos != stream->bytes_left)
793 {
794 /* The field was handled */
795 continue;
796 }
797 }
798 }
799
800 /* No match found, skip data */
801 if (!pb_skip_field(stream, wire_type))
802 return false;
803 continue;
804 }
805
806 if (PB_HTYPE(iter.pos->type) == PB_HTYPE_REQUIRED
807 && iter.required_field_index < PB_MAX_REQUIRED_FIELDS)
808 {
809 fields_seen[iter.required_field_index >> 3] |= (uint8_t)(1 << (iter.required_field_index & 7));
810 }
811
812 if (!decode_field(stream, wire_type, &iter))
813 return false;
814 }
815
816 /* Check that all required fields were present. */
817 {
818 /* First figure out the number of required fields by
819 * seeking to the end of the field array. Usually we
820 * are already close to end after decoding.
821 */
822 unsigned req_field_count;
823 pb_type_t last_type;
824 unsigned i;
825 do {
826 req_field_count = iter.required_field_index;
827 last_type = iter.pos->type;
828 } while (pb_field_iter_next(&iter));
829
830 /* Fixup if last field was also required. */
831 if (PB_HTYPE(last_type) == PB_HTYPE_REQUIRED && iter.pos->tag != 0)
832 req_field_count++;
833
834 /* Check the whole bytes */
835 for (i = 0; i < (req_field_count >> 3); i++)
836 {
837 if (fields_seen[i] != 0xFF)
838 PB_RETURN_ERROR(stream, "missing required field");
839 }
840
841 /* Check the remaining bits */
842 if (fields_seen[req_field_count >> 3] != (0xFF >> (8 - (req_field_count & 7))))
843 PB_RETURN_ERROR(stream, "missing required field");
844 }
845
846 return true;
847}

876#ifdef PB_ENABLE_MALLOC
877void pb_release(const pb_field_t fields[], void *dest_struct)
878{
879 pb_field_iter_t iter;
880
881 if (!pb_field_iter_begin(&iter, fields, dest_struct))
882 return; /* Empty message type */
883
884 do
885 {
886 pb_type_t type;
887 type = iter.pos->type;
888
889 if (PB_ATYPE(type) == PB_ATYPE_POINTER)
890 {
891 if (PB_HTYPE(type) == PB_HTYPE_REPEATED &&
892 (PB_LTYPE(type) == PB_LTYPE_STRING ||
893 PB_LTYPE(type) == PB_LTYPE_BYTES))
894 {
895 /* Release entries in repeated string or bytes array */
896 void **pItem = *(void***)iter.pData;
897 pb_size_t count = *(pb_size_t*)iter.pSize;
898 while (count--)
899 {
900 pb_free(*pItem);
901 *pItem++ = NULL;
902 }
903 }
904 else if (PB_LTYPE(type) == PB_LTYPE_SUBMESSAGE)
905 {
906 /* Release fields in submessages */
907 void *pItem = *(void**)iter.pData;
908 pb_size_t count = (pItem ? 1 : 0);
909
910 if (PB_HTYPE(type) == PB_HTYPE_REPEATED)
911 {
912 count = *(pb_size_t*)iter.pSize;
913 }
914
915 while (count--)
916 {
917 pb_release((const pb_field_t*)iter.pos->ptr, pItem);
918 pItem = (uint8_t*)pItem + iter.pos->data_size;
919 }
920 }
921
922 /* Release main item */
923 pb_free(*(void**)iter.pData);
924 *(void**)iter.pData = NULL;
925 }
926 } while (pb_field_iter_next(&iter));
927}
928#endif

1050static bool checkreturn pb_dec_bytes(pb_istream_t *stream, const pb_field_t *field, void *dest)
1051{
1052 uint32_t size;
1053 pb_bytes_array_t *bdest;
1054
1055 if (!pb_decode_varint32(stream, &size))
1056 return false;
1057
1058 if (PB_ATYPE(field->type) == PB_ATYPE_POINTER)
1059 {
1060#ifndef PB_ENABLE_MALLOC
1061 PB_RETURN_ERROR(stream, "no malloc support");
1062#else
1063 if (!allocate_field(stream, dest, PB_BYTES_ARRAY_T_ALLOCSIZE(size), 1))
1064 return false;
1065 bdest = *(pb_bytes_array_t**)dest;
1066#endif
1067 }
1068 else
1069 {
1070 if (PB_BYTES_ARRAY_T_ALLOCSIZE(size) > field->data_size)
1071 PB_RETURN_ERROR(stream, "bytes overflow");
1072 bdest = (pb_bytes_array_t*)dest;
1073 }
1074
1075 if (size > PB_SIZE_MAX)
1076 {
1077 PB_RETURN_ERROR(stream, "bytes overflow");
1078 }
1079
1080 bdest->size = (pb_size_t)size;
1081 return pb_read(stream, bdest->bytes, size);
1082}

1084static bool checkreturn pb_dec_string(pb_istream_t *stream, const pb_field_t *field, void *dest)
1085{
1086 uint32_t size;
1087 size_t alloc_size;
1088 bool status;
1089 if (!pb_decode_varint32(stream, &size))
1090 return false;
1091
1092 /* Space for null terminator */
1093 alloc_size = size + 1;
1094
1095 if (PB_ATYPE(field->type) == PB_ATYPE_POINTER)
1096 {
1097#ifndef PB_ENABLE_MALLOC
1098 PB_RETURN_ERROR(stream, "no malloc support");
1099#else
1100 if (!allocate_field(stream, dest, alloc_size, 1))
1101 return false;
1102 dest = *(void**)dest;
1103#endif
1104 }
1105 else
1106 {
1107 if (alloc_size > field->data_size)
1108 PB_RETURN_ERROR(stream, "string overflow");
1109 }
1110
1111 status = pb_read(stream, (uint8_t*)dest, size);
1112 *((uint8_t*)dest + size) = 0;
1113 return status;
1114}

1116static bool checkreturn pb_dec_submessage(pb_istream_t *stream, const pb_field_t *field, void *dest)
1117{
1118 bool status;
1119 pb_istream_t substream;
1120 const pb_field_t* submsg_fields = (const pb_field_t*)field->ptr;
1121
1122 if (!pb_make_string_substream(stream, &substream))
1123 return false;
1124
1125 if (field->ptr == NULL)
1126 PB_RETURN_ERROR(stream, "invalid field descriptor");
1127
1128 /* New array entries need to be initialized, while required and optional
1129 * submessages have already been initialized in the top-level pb_decode. */
1130 if (PB_HTYPE(field->type) == PB_HTYPE_REPEATED)
1131 status = pb_decode(&substream, submsg_fields, dest);
1132 else
1133 status = pb_decode_noinit(&substream, submsg_fields, dest);
1134
1135 pb_close_string_substream(stream, &substream);
1136 return status;
1137}
