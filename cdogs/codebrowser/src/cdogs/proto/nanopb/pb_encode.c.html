64static bool checkreturn buf_write(pb_ostream_t *stream, const uint8_t *buf, size_t count)
65{
66 uint8_t *dest = (uint8_t*)stream->state;
67 stream->state = dest + count;
68
69 while (count--)
70 *dest++ = *buf++;
71
72 return true;
73}

75pb_ostream_t pb_ostream_from_buffer(uint8_t *buf, size_t bufsize)
76{
77 pb_ostream_t stream;
78#ifdef PB_BUFFER_ONLY
79 stream.callback = (void*)1; /* Just a marker value */
80#else
81 stream.callback = &buf_write;
82#endif
83 stream.state = buf;
84 stream.max_size = bufsize;
85 stream.bytes_written = 0;
86#ifndef PB_NO_ERRMSG
87 stream.errmsg = NULL;
88#endif
89 return stream;
90}

92bool checkreturn pb_write(pb_ostream_t *stream, const uint8_t *buf, size_t count)
93{
94 if (stream->callback != NULL)
95 {
96 if (stream->bytes_written + count > stream->max_size)
97 PB_RETURN_ERROR(stream, "stream full");
98
99#ifdef PB_BUFFER_ONLY
100 if (!buf_write(stream, buf, count))
101 PB_RETURN_ERROR(stream, "io error");
102#else
103 if (!stream->callback(stream, buf, count))
104 PB_RETURN_ERROR(stream, "io error");
105#endif
106 }
107
108 stream->bytes_written += count;
109 return true;
110}

116/* Encode a static array. Handles the size calculations and possible packing. */
117static bool checkreturn encode_array(pb_ostream_t *stream, const pb_field_t *field,
118 const void *pData, size_t count, pb_encoder_t func)
119{
120 size_t i;
121 const void *p;
122 size_t size;
123
124 if (count == 0)
125 return true;
126
127 if (PB_ATYPE(field->type) != PB_ATYPE_POINTER && count > field->array_size)
128 PB_RETURN_ERROR(stream, "array max size exceeded");
129
130 /* We always pack arrays if the datatype allows it. */
131 if (PB_LTYPE(field->type) <= PB_LTYPE_LAST_PACKABLE)
132 {
133 if (!pb_encode_tag(stream, PB_WT_STRING, field->tag))
134 return false;
135
136 /* Determine the total size of packed array. */
137 if (PB_LTYPE(field->type) == PB_LTYPE_FIXED32)
138 {
139 size = 4 * count;
140 }
141 else if (PB_LTYPE(field->type) == PB_LTYPE_FIXED64)
142 {
143 size = 8 * count;
144 }
145 else
146 {
147 pb_ostream_t sizestream = PB_OSTREAM_SIZING;
148 p = pData;
149 for (i = 0; i < count; i++)
150 {
151 if (!func(&sizestream, field, p))
152 return false;
153 p = (const char*)p + field->data_size;
154 }
155 size = sizestream.bytes_written;
156 }
157
158 if (!pb_encode_varint(stream, (uint64_t)size))
159 return false;
160
161 if (stream->callback == NULL)
162 return pb_write(stream, NULL, size); /* Just sizing.. */
163
164 /* Write the data */
165 p = pData;
166 for (i = 0; i < count; i++)
167 {
168 if (!func(stream, field, p))
169 return false;
170 p = (const char*)p + field->data_size;
171 }
172 }
173 else
174 {
175 p = pData;
176 for (i = 0; i < count; i++)
177 {
178 if (!pb_encode_tag_for_field(stream, field))
179 return false;
180
181 /* Normally the data is stored directly in the array entries, but
182 * for pointer-type string and bytes fields, the array entries are
183 * actually pointers themselves also. So we have to dereference once
184 * more to get to the actual data. */
185 if (PB_ATYPE(field->type) == PB_ATYPE_POINTER &&
186 (PB_LTYPE(field->type) == PB_LTYPE_STRING ||
187 PB_LTYPE(field->type) == PB_LTYPE_BYTES))
188 {
189 if (!func(stream, field, *(const void* const*)p))
190 return false;
191 }
192 else
193 {
194 if (!func(stream, field, p))
195 return false;
196 }
197 p = (const char*)p + field->data_size;
198 }
199 }
200
201 return true;
202}

204/* Encode a field with static or pointer allocation, i.e. one whose data
205 * is available to the encoder directly. */
206static bool checkreturn encode_basic_field(pb_ostream_t *stream,
207 const pb_field_t *field, const void *pData)
208{
209 pb_encoder_t func;
210 const void *pSize;
211 bool implicit_has = true;
212
213 func = PB_ENCODERS[PB_LTYPE(field->type)];
214
215 if (field->size_offset)
216 pSize = (const char*)pData + field->size_offset;
217 else
218 pSize = &implicit_has;
219
220 if (PB_ATYPE(field->type) == PB_ATYPE_POINTER)
221 {
222 /* pData is a pointer to the field, which contains pointer to
223 * the data. If the 2nd pointer is NULL, it is interpreted as if
224 * the has_field was false.
225 */
226
227 pData = *(const void* const*)pData;
228 implicit_has = (pData != NULL);
229 }
230
231 switch (PB_HTYPE(field->type))
232 {
233 case PB_HTYPE_REQUIRED:
234 if (!pData)
235 PB_RETURN_ERROR(stream, "missing required field");
236 if (!pb_encode_tag_for_field(stream, field))
237 return false;
238 if (!func(stream, field, pData))
239 return false;
240 break;
241
242 case PB_HTYPE_OPTIONAL:
243 if (*(const bool*)pSize)
244 {
245 if (!pb_encode_tag_for_field(stream, field))
246 return false;
247
248 if (!func(stream, field, pData))
249 return false;
250 }
251 break;
252
253 case PB_HTYPE_REPEATED:
254 if (!encode_array(stream, field, pData, *(const pb_size_t*)pSize, func))
255 return false;
256 break;
257
258 default:
259 PB_RETURN_ERROR(stream, "invalid field type");
260 }
261
262 return true;
263}

265/* Encode a field with callback semantics. This means that a user function is
266 * called to provide and encode the actual data. */
267static bool checkreturn encode_callback_field(pb_ostream_t *stream,
268 const pb_field_t *field, const void *pData)
269{
270 const pb_callback_t *callback = (const pb_callback_t*)pData;
271
272#ifdef PB_OLD_CALLBACK_STYLE
273 const void *arg = callback->arg;
274#else
275 void * const *arg = &(callback->arg);
276#endif
277
278 if (callback->funcs.encode != NULL)
279 {
280 if (!callback->funcs.encode(stream, field, arg))
281 PB_RETURN_ERROR(stream, "callback error");
282 }
283 return true;
284}

286/* Encode a single field of any callback or static type. */
287static bool checkreturn encode_field(pb_ostream_t *stream,
288 const pb_field_t *field, const void *pData)
289{
290 switch (PB_ATYPE(field->type))
291 {
292 case PB_ATYPE_STATIC:
293 case PB_ATYPE_POINTER:
294 return encode_basic_field(stream, field, pData);
295
296 case PB_ATYPE_CALLBACK:
297 return encode_callback_field(stream, field, pData);
298
299 default:
300 PB_RETURN_ERROR(stream, "invalid field type");
301 }
302}

304/* Default handler for extension fields. Expects to have a pb_field_t
305 * pointer in the extension->type->arg field. */
306static bool checkreturn default_extension_encoder(pb_ostream_t *stream,
307 const pb_extension_t *extension)
308{
309 const pb_field_t *field = (const pb_field_t*)extension->type->arg;
310 return encode_field(stream, field, extension->dest);
311}
312
313/* Walk through all the registered extensions and give them a chance
314 * to encode themselves. */
315static bool checkreturn encode_extension_field(pb_ostream_t *stream,
316 const pb_field_t *field, const void *pData)
317{
318 const pb_extension_t *extension = *(const pb_extension_t* const *)pData;
319 PB_UNUSED(field);
320
321 while (extension)
322 {
323 bool status;
324 if (extension->type->encode)
325 status = extension->type->encode(stream, extension);
326 else
327 status = default_extension_encoder(stream, extension);
328
329 if (!status)
330 return false;
331
332 extension = extension->next;
333 }
334
335 return true;
336}

342static void *remove_const(const void *p)
343{
344 /* Note: this casts away const, in order to use the common field iterator
345 * logic for both encoding and decoding. */
346 union {
347 void *p1;
348 const void *p2;
349 } t;
350 t.p2 = p;
351 return t.p1;
352}
353
354bool checkreturn pb_encode(pb_ostream_t *stream, const pb_field_t fields[], const void *src_struct)
355{
356 pb_field_iter_t iter;
357 if (!pb_field_iter_begin(&iter, fields, remove_const(src_struct)))
358 return true; /* Empty message type */
359
360 do {
361 if (PB_LTYPE(iter.pos->type) == PB_LTYPE_EXTENSION)
362 {
363 /* Special case for the extension field placeholder */
364 if (!encode_extension_field(stream, iter.pos, iter.pData))
365 return false;
366 }
367 else
368 {
369 /* Regular field */
370 if (!encode_field(stream, iter.pos, iter.pData))
371 return false;
372 }
373 } while (pb_field_iter_next(&iter));
374
375 return true;
376}

397bool checkreturn pb_encode_varint(pb_ostream_t *stream, uint64_t value)
398{
399 uint8_t buffer[10];
400 size_t i = 0;
401
402 if (value == 0)
403 return pb_write(stream, (uint8_t*)&value, 1);
404
405 while (value)
406 {
407 buffer[i] = (uint8_t)((value & 0x7F) | 0x80);
408 value >>= 7;
409 i++;
410 }
411 buffer[i-1] &= 0x7F; /* Unset top bit on last byte */
412
413 return pb_write(stream, buffer, i);
414}

416bool checkreturn pb_encode_svarint(pb_ostream_t *stream, int64_t value)
417{
418 uint64_t zigzagged;
419 if (value < 0)
420 zigzagged = ~((uint64_t)value << 1);
421 else
422 zigzagged = (uint64_t)value << 1;
423
424 return pb_encode_varint(stream, zigzagged);
425}

427bool checkreturn pb_encode_fixed32(pb_ostream_t *stream, const void *value)
428{
429 #ifdef __BIG_ENDIAN__
430 const uint8_t *bytes = value;
431 uint8_t lebytes[4];
432 lebytes[0] = bytes[3];
433 lebytes[1] = bytes[2];
434 lebytes[2] = bytes[1];
435 lebytes[3] = bytes[0];
436 return pb_write(stream, lebytes, 4);
437 #else
438 return pb_write(stream, (const uint8_t*)value, 4);
439 #endif
440}

442bool checkreturn pb_encode_fixed64(pb_ostream_t *stream, const void *value)
443{
444 #ifdef __BIG_ENDIAN__
445 const uint8_t *bytes = value;
446 uint8_t lebytes[8];
447 lebytes[0] = bytes[7];
448 lebytes[1] = bytes[6];
449 lebytes[2] = bytes[5];
450 lebytes[3] = bytes[4];
451 lebytes[4] = bytes[3];
452 lebytes[5] = bytes[2];
453 lebytes[6] = bytes[1];
454 lebytes[7] = bytes[0];
455 return pb_write(stream, lebytes, 8);
456 #else
457 return pb_write(stream, (const uint8_t*)value, 8);
458 #endif

461bool checkreturn pb_encode_tag(pb_ostream_t *stream, pb_wire_type_t wiretype, uint32_t field_number)
462{
463 uint64_t tag = ((uint64_t)field_number << 3) | wiretype;

467bool checkreturn pb_encode_tag_for_field(pb_ostream_t *stream, const pb_field_t *field)
468{
469 pb_wire_type_t wiretype;
470 switch (PB_LTYPE(field->type))
471 {
472 case PB_LTYPE_VARINT:
473 case PB_LTYPE_UVARINT:
474 case PB_LTYPE_SVARINT:
475 wiretype = PB_WT_VARINT;
476 break;
477
478 case PB_LTYPE_FIXED32:
479 wiretype = PB_WT_32BIT;
480 break;
481
482 case PB_LTYPE_FIXED64:
483 wiretype = PB_WT_64BIT;
484 break;
485
486 case PB_LTYPE_BYTES:
487 case PB_LTYPE_STRING:
488 case PB_LTYPE_SUBMESSAGE:
489 wiretype = PB_WT_STRING;
490 break;
491
492 default:
493 PB_RETURN_ERROR(stream, "invalid field type");
494 }
495
496 return pb_encode_tag(stream, wiretype, field->tag);
497}

507bool checkreturn pb_encode_submessage(pb_ostream_t *stream, const pb_field_t fields[], const void *src_struct)
508{
509 /* First calculate the message size using a non-writing substream. */
510 pb_ostream_t substream = PB_OSTREAM_SIZING;
511 size_t size;
512 bool status;
513
514 if (!pb_encode(&substream, fields, src_struct))
515 {
516#ifndef PB_NO_ERRMSG
517 stream->errmsg = substream.errmsg;
518#endif
519 return false;
520 }
521
522 size = substream.bytes_written;
523
524 if (!pb_encode_varint(stream, (uint64_t)size))
525 return false;
526
527 if (stream->callback == NULL)
528 return pb_write(stream, NULL, size); /* Just sizing */
529
530 if (stream->bytes_written + size > stream->max_size)
531 PB_RETURN_ERROR(stream, "stream full");
532
533 /* Use a substream to verify that a callback doesn't write more than
534 * what it did the first time. */
535 substream.callback = stream->callback;
536 substream.state = stream->state;
537 substream.max_size = size;
538 substream.bytes_written = 0;
539#ifndef PB_NO_ERRMSG
540 substream.errmsg = NULL;
541#endif
542
543 status = pb_encode(&substream, fields, src_struct);
544
545 stream->bytes_written += substream.bytes_written;
546 stream->state = substream.state;
547#ifndef PB_NO_ERRMSG
548 stream->errmsg = substream.errmsg;
549#endif
550
551 if (substream.bytes_written != size)
552 PB_RETURN_ERROR(stream, "submsg size changed");
553
554 return status;
555}

559static bool checkreturn pb_enc_varint(pb_ostream_t *stream, const pb_field_t *field, const void *src)
560{
561 int64_t value = 0;
562
563 /* Cases 1 and 2 are for compilers that have smaller types for bool
564 * or enums. */
565 switch (field->data_size)
566 {
567 case 1: value = *(const int8_t*)src; break;
568 case 2: value = *(const int16_t*)src; break;
569 case 4: value = *(const int32_t*)src; break;
570 case 8: value = *(const int64_t*)src; break;
571 default: PB_RETURN_ERROR(stream, "invalid data_size");
572 }
573
574 return pb_encode_varint(stream, (uint64_t)value);
575}

617static bool checkreturn pb_enc_bytes(pb_ostream_t *stream, const pb_field_t *field, const void *src)
618{
619 const pb_bytes_array_t *bytes = (const pb_bytes_array_t*)src;
620
621 if (src == NULL)
622 {
623 /* Threat null pointer as an empty bytes field */
624 return pb_encode_string(stream, NULL, 0);
625 }
626
627 if (PB_ATYPE(field->type) == PB_ATYPE_STATIC &&
628 PB_BYTES_ARRAY_T_ALLOCSIZE(bytes->size) > field->data_size)
629 {
630 PB_RETURN_ERROR(stream, "bytes size exceeded");
631 }
632
633 return pb_encode_string(stream, bytes->bytes, bytes->size);
634}

636static bool checkreturn pb_enc_string(pb_ostream_t *stream, const pb_field_t *field, const void *src)
637{
638 size_t size = 0;
639 size_t max_size = field->data_size;
640 const char *p = (const char*)src;
641
642 if (PB_ATYPE(field->type) == PB_ATYPE_POINTER)
643 max_size = (size_t)-1;
644
645 if (src == NULL)
646 {
647 size = 0; /* Threat null pointer as an empty string */
648 }
649 else
650 {
651 /* strnlen() is not always available, so just use a loop */
652 while (size < max_size && *p != '\0')
653 {
654 size++;
655 p++;
656 }
657 }
658
659 return pb_encode_string(stream, (const uint8_t*)src, size);
660}
