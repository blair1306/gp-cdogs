39void MapCaveLoad(
40 Map *map, const struct MissionOptions *mo, const CampaignOptions* co)
41{
42 // Re-seed RNG so results are consistent
43 CampaignSeedRandom(co);
44
45 const Mission *m = mo->missionData;
46
47 // Randomly set a percentage of the tiles as walls
48 for (int i = 0;
49 i < m->u.Cave.FillPercent*map->Size.x*map->Size.y / 100;
50 i++)
51 {
52 const Vec2i pos = Vec2iNew(i % map->Size.x, i / map->Size.x);
53 IMapSet(map, pos, MAP_WALL);
54 }
55 // Shuffle
56 CArrayShuffle(&map->iMap);
57 // Repetitions
58 for (int i = 0; i < m->u.Cave.Repeat; i++)
59 {
60 CaveRep(map, m->u.Cave.R1, m->u.Cave.R2);
61 }
62
63 LinkDisconnectedAreas(map);
64
65 FixCorridors(map, m->u.Cave.CorridorWidth);
66
67 PlaceSquares(map, m->u.Cave.Squares);
68
69 PlaceRooms(map, m->u.Cave.Rooms);
70}

72// Perform one generation of cellular automata
73// If the number of walls within 1 distance is at least R1, OR
74// if the number of walls within 2 distance is at most R2, then the tile
75// becomes a wall; otherwise it is a floor
76static int CountTilesAround(
77 const Map *map, const Vec2i pos, const int radius,
78 const unsigned short tile);
79static void CaveRep(Map *map, const int r1, const int r2)
80{
81 CArray buf;
82 CArrayInit(&buf, map->iMap.elemSize);
83 const unsigned short floor = MAP_FLOOR;
84 CArrayResize(&buf, map->iMap.size, &floor);
85 Vec2i v;
86 for (v.y = 0; v.y < map->Size.y; v.y++)
87 {
88 for (v.x = 0; v.x < map->Size.x; v.x++)
89 {
90 const int idx = v.x + v.y * map->Size.x;
91 unsigned short *tile = CArrayGet(&buf, idx);
92 if (CountTilesAround(map, v, 1, MAP_WALL) >= r1 ||
93 CountTilesAround(map, v, 2, MAP_WALL) <= r2)
94 {
95 *tile = MAP_WALL;
96 }
97 else
98 {
99 *tile = MAP_FLOOR;
100 }
101 }
102 }
103 CArrayCopy(&map->iMap, &buf);
104 CArrayTerminate(&buf);
105}
106static int CountTilesAround(
107 const Map *map, const Vec2i pos, const int radius,
108 const unsigned short tile)
109{
110 int c = 0;
111 for (int x = pos.x - radius; x <= pos.x + radius; x++)
112 {
113 for (int y = pos.y - radius; y <= pos.y + radius; y++)
114 {
115 // Also count edge of maps
116 if (x < 0 || x >= map->Size.x || y < 0 || y >= map->Size.y ||
117 IMapGet(map, Vec2iNew(x, y)) == tile)
118 {
119 c++;
120 }
121 }
122 }
123 return c;
124}

131static void LinkDisconnectedAreas(Map *map)
132{
133 // Use flood fill to identify disconnected areas
134 CArray fl;
135 CArrayInit(&fl, sizeof(int));
136 const int zero = 0;
137 CArrayResize(&fl, map->iMap.size, &zero);
138 // First copy across the wall tiles (as -1)
139 CA_FOREACH(const unsigned short, tile, map->iMap)
140 if (*tile == MAP_WALL)
141 {
142 *(int *)CArrayGet(&fl, _ca_index) = -1;
143 }
144 CA_FOREACH_END()
145 // Then perform flood fill conditionally on the flood layer
146 int idx = 1;
147 CA_FOREACH(int, i, fl)
148 if (*i == 0)
149 {
150 MapFloodFill(&fl, map->Size, _ca_index, idx);
151 idx++;
152 }
153 CA_FOREACH_END()
154
155 const int numAreas = idx - 1;
156 // Connect the disconnected areas, first to second, second to third etc.
157 // Select random tile from each area, using index shuffle
158 CArray areaTiles;
159 CArrayInit(&areaTiles, sizeof(int));
160 CA_FOREACH(int, i, fl)
161 UNUSED(i);
162 CArrayPushBack(&areaTiles, &_ca_index);
163 CA_FOREACH_END()
164 CArrayShuffle(&areaTiles);
165 CArray areaStarts;
166 CArrayInit(&areaStarts, sizeof(int));
167 CArrayResize(&areaStarts, numAreas, &zero);
168 CA_FOREACH(int, areaIdx, areaTiles)
169 const int tile = *(int *)CArrayGet(&fl, *areaIdx) - 1;
170 if (tile >= 0 && tile < numAreas)
171 {
172 int *areaStart = CArrayGet(&areaStarts, tile);
173 if (*areaStart == 0)
174 {
175 *areaStart = *areaIdx;
176 }
177 }
178 CA_FOREACH_END()
179 CArrayTerminate(&fl);
180 CArrayTerminate(&areaTiles);
181 // Connect consecutive pairs of tiles
182 for (int i = 0; i < (int)areaStarts.size - 1; i++)
183 {
184 const int *a1 = CArrayGet(&areaStarts, i);
185 const int *a2 = CArrayGet(&areaStarts, i + 1);
186 const Vec2i v1 = Vec2iNew(*a1 % map->Size.x, *a1 / map->Size.x);
187 const Vec2i v2 = Vec2iNew(*a2 % map->Size.x, *a2 / map->Size.x);
188 const Vec2i delta = Vec2iNew(abs(v1.x - v2.x), abs(v1.y - v2.y));
189 const int dx = delta.x > delta.y ? 1 : 0;
190 const int dy = 1 - dx;
191 AddCorridor(map, v1, v2, Vec2iNew(dx, dy), MAP_FLOOR);
192 }
193 CArrayTerminate(&areaStarts);
194}

196static void MapFloodFill(
197 CArray *fl, const Vec2i size, const int idx, const int elem)
198{
199 CArray indices;
200 CArrayInit(&indices, sizeof(int));
201 CArrayPushBack(&indices, &idx);
202 const int floodTile = *(int *)CArrayGet(fl, idx);
203 *(int *)CArrayGet(fl, idx) = elem;
204 CA_FOREACH(int, i, indices)
205 const int x = *i % size.x;
206 const int y = *i / size.x;
207 int next;
208 // top
209 next = (y - 1)*size.x + x;
210 if (y > 0 && *(int *)CArrayGet(fl, next) == floodTile)
211 {
212 CArrayPushBack(&indices, &next);
213 *(int *)CArrayGet(fl, next) = elem;
214 }
215 // bottom
216 next = (y + 1)*size.x + x;
217 if (y < size.y - 1 && *(int *)CArrayGet(fl, next) == floodTile)
218 {
219 CArrayPushBack(&indices, &next);
220 *(int *)CArrayGet(fl, next) = elem;
221 }
222 // left
223 next = y*size.x + x - 1;
224 if (x > 0 && *(int *)CArrayGet(fl, next) == floodTile)
225 {
226 CArrayPushBack(&indices, &next);
227 *(int *)CArrayGet(fl, next) = elem;
228 }
229 // right
230 next = y*size.x + x + 1;
231 if (x < size.x - 1 && *(int *)CArrayGet(fl, next) == floodTile)
232 {
233 CArrayPushBack(&indices, &next);
234 *(int *)CArrayGet(fl, next) = elem;
235 }
236 CA_FOREACH_END()
237 CArrayTerminate(&indices);
238}

240// Add an S-shaped corridor from one point to another, filling it with a
241// certain tile value. The corridor starts in a specific direction d, then
242// makes a turn in the middle, then turns back to the original direction.
243static void AddCorridor(
244 Map *map, const Vec2i v1, const Vec2i v2, const Vec2i dInit,
245 const unsigned short tile)
246{
247 Vec2i dAlt;
248 // Location of the turn
249 Vec2i half;
250 Vec2i start = v1;
251 Vec2i end = v2;
252 Vec2i d = dInit;
253 if (d.x > 0)
254 {
255 // horizontal
256 d = Vec2iNew(1, 0);
257 if (start.x > end.x)
258 {
259 // Swap
260 const Vec2i tmp = start;
261 start = end;
262 end = tmp;
263 }
264 dAlt = Vec2iNew(0, 1);
265 half = Vec2iNew((end.x - start.x) / 2 + start.x, end.y + 1);
266 if (end.y < start.y)
267 {
268 dAlt.y = -1;
269 half.y = end.y - 1;
270 }
271 }
272 else
273 {
274 // vertical
275 d = Vec2iNew(0, 1);
276 if (start.y > end.y)
277 {
278 // Swap
279 const Vec2i tmp = start;
280 start = end;
281 end = tmp;
282 }
283 dAlt = Vec2iNew(1, 0);
284 half = Vec2iNew(end.x + 1, (end.y - start.y) / 2 + start.y);
285 if (end.x < start.x)
286 {
287 dAlt.x = -1;
288 half.x = end.x - 1;
289 }
290 }
291 // Initial direction
292 Vec2i v = start;
293 for (; v.x != half.x && v.y != half.y; v = Vec2iAdd(v, d))
294 {
295 IMapSet(map, v, tile);
296 }
297 // Turn
298 for (; v.x != end.x && v.y != end.y; v = Vec2iAdd(v, dAlt))
299 {
300 IMapSet(map, v, tile);
301 }
302 // Finish
303 for (; v.x != end.x || v.y != end.y; v = Vec2iAdd(v, d))
304 {
305 IMapSet(map, v, tile);
306 }
307 IMapSet(map, end, tile);
308}

310static bool CheckCorridorsAroundTile(
311 const Map *map, const int corridorWidth, const Vec2i v);
312// Make sure corridors are wide enough
313static void FixCorridors(Map *map, const int corridorWidth)
314{
315 // Find all instances where a corridor is too narrow
316 // Do this by drawing a line from each wall tile to its surrounding square
317 // area, and that this line of tiles that each wall tile is either
318 // - blocked by a neighbouring wall tile, or
319 // - only consists of floor tiles
320 // If not, then replace that wall with a floor tile
321 Vec2i v;
322 for (v.y = corridorWidth; v.y < map->Size.y - corridorWidth; v.y++)
323 {
324 for (v.x = corridorWidth; v.x < map->Size.x - corridorWidth; v.x++)
325 {
326 const int idx = v.x + v.y * map->Size.x;
327 unsigned short *tile = CArrayGet(&map->iMap, idx);
328 // Make sure that the tile is a wall
329 if (*tile != MAP_WALL)
330 {
331 continue;
332 }
333 if (!CheckCorridorsAroundTile(map, corridorWidth, v))
334 {
335 // Corridor checks failed; replace with a floor tile
336 *tile = MAP_FLOOR;
337 }
338 }
339 }
340}
341typedef struct
342{
343 const Map *M;
344 int Counter;
345 bool IsFirstWall;
346 bool AreAllFloors;
347} FixCorridorOnTileData;
348static void FixCorridorOnTile(void *data, Vec2i v);
349static bool CheckCorridorsAroundTile(
350 const Map *map, const int corridorWidth, const Vec2i v)
351{
352 AlgoLineDrawData data;
353 data.Draw = FixCorridorOnTile;
354 FixCorridorOnTileData onTileData;
355 onTileData.M = map;
356 data.data = &onTileData;
357 Vec2i v1;
358 for (v1.y = v.y - corridorWidth; v1.y <= v.y + corridorWidth; v1.y++)
359 {
360 for (v1.x = v.x - corridorWidth; v1.x <= v.x + corridorWidth; v1.x++)
361 {
362 // only draw out to edges
363 if (v1.y != v.y - corridorWidth && v1.y != v.y + corridorWidth &&
364 v1.x != v.x - corridorWidth && v1.x != v.x + corridorWidth)
365 {
366 continue;
367 }
368 onTileData.Counter = 0;
369 onTileData.IsFirstWall = false;
370 onTileData.AreAllFloors = true;
371 BresenhamLineDraw(v, v1, &data);
372 if (!onTileData.IsFirstWall && !onTileData.AreAllFloors)
373 {
374 return false;
375 }
376 }
377 }
378 return true;
379}
380static void FixCorridorOnTile(void *data, Vec2i v)
381{
382 FixCorridorOnTileData *onTileData = data;
383 if (onTileData->Counter == 1)
384 {
385 // This is the first tile after the starting tile
386 // If this tile is a wall, result is good, and we don't care about
387 // the rest of the tiles anymore
388 onTileData->IsFirstWall = IMapGet(onTileData->M, v) == MAP_WALL;
389 }
390 if (onTileData->Counter > 0)
391 {
392 if (IMapGet(onTileData->M, v) != MAP_FLOOR)
393 {
394 onTileData->AreAllFloors = false;
395 }
396 }
397 onTileData->Counter++;
398}

402static void PlaceSquares(Map *map, const int squares)
403{
404 // Place empty square areas on the map
405 // This can only be done if at least one tile in the square is a floor type
406 int count = 0;
407 for (int i = 0; i < 1000 && count < squares; i++)
408 {
409 const Vec2i v = MapGetRandomTile(map);
410 const Vec2i size = Vec2iNew(rand() % 9 + 8, rand() % 9 + 8);
411 if (!MapIsAreaClearForCaveSquare(map, v, size))
412 {
413 continue;
414 }
415 MapMakeSquare(map, v, size);
416 count++;
417 }
418}
419static bool MapIsAreaClearForCaveSquare(
420 const Map *map, const Vec2i pos, const Vec2i size)
421{
422 if (!MapIsAreaInside(map, pos, size))
423 {
424 return false;
425 }
426
427 // For area to be clear, it must have:
428 // - At least one floor tile
429 // - No square tiles
430 Vec2i v;
431 bool hasFloor = false;
432 for (v.y = pos.y; v.y < pos.y + size.y; v.y++)
433 {
434 for (v.x = pos.x; v.x < pos.x + size.x; v.x++)
435 {
436 switch (IMapGet(map, v))
437 {
438 case MAP_FLOOR:
439 hasFloor = true;
440 break;
441 case MAP_WALL:
442 break;
443 default:
444 // Any other tile type is disallowed
445 return false;
446 }
447 }
448 }
449 return hasFloor;
450}

481static bool CaveRoomOutsideOk(const Map *map, const Vec2i v);
482static bool MapIsAreaClearForCaveRoom(
483 const Map *map, const Vec2i pos, const Vec2i size)
484{
485 if (!MapIsAreaInside(map, pos, size))
486 {
487 return false;
488 }
489
490 // For area to be clear, it must have:
491 // - Area has at least one floor tile
492 // - Can only contain floor, wall or room tiles
493 Vec2i v;
494 bool hasFloor = false;
495 for (v.y = pos.y; v.y < pos.y + size.y; v.y++)
496 {
497 for (v.x = pos.x; v.x < pos.x + size.x; v.x++)
498 {
499 switch (IMapGet(map, v))
500 {
501 case MAP_FLOOR:
502 hasFloor = true;
503 break;
504 case MAP_WALL: // passthrough
505 case MAP_ROOM:
506 break;
507 default:
508 // Any other tile type is disallowed
509 return false;
510 }
511 }
512 }
513 if (!hasFloor)
514 {
515 return false;
516 }
517
518 // For edges:
519 // - Either the edge is a wall, or
520 // - Edge is floor/room, and outside is floor/room/square
521 // This is to prevent rooms from cutting off areas of the map
522 for (v.y = pos.y; v.y < pos.y + size.y; v.y++)
523 {
524 for (v.x = pos.x; v.x < pos.x + size.x; v.x++)
525 {
526 if (v.y > pos.y && v.y < pos.y + size.y - 1 &&
527 v.x > pos.x && v.x < pos.x + size.x - 1)
528 {
529 continue;
530 }
531 const bool isTop = v.y == pos.y;
532 const bool isBottom = v.y == pos.y + size.y - 1;
533 const bool isLeft = v.x == pos.x;
534 const bool isRight = v.x == pos.x + size.x - 1;
535 const Vec2i outside = Vec2iNew(
536 isLeft ? pos.x - 1 : (isRight ? pos.x + size.x : v.x),
537 isTop ? pos.y - 1 : (isBottom ? pos.y + size.y : v.y));
538 const Vec2i outsideX = Vec2iNew(
539 (isLeft || isRight) ? outside.x : v.x, v.y);
540 const Vec2i outsideY = Vec2iNew(
541 v.x, (isTop || isBottom) ? outside.y : v.y);
542 switch (IMapGet(map, v))
543 {
544 case MAP_WALL:
545 break;
546 case MAP_FLOOR: // passthrough
547 case MAP_ROOM:
548 // Check outside tiles
549 if (!CaveRoomOutsideOk(map, outside) ||
550 !CaveRoomOutsideOk(map, outsideX) ||
551 !CaveRoomOutsideOk(map, outsideY))
552 {
553 return false;
554 }
555 break;
556 default:
557 CASSERT(false, "unexpected tile type");
558 return false;
559 }
560 }
561 }
562
563 return true;
564}
565static bool CaveRoomOutsideOk(const Map *map, const Vec2i v)
566{
567 const unsigned short t = IMapGet(map, v);
568 return t == MAP_FLOOR || t == MAP_ROOM || t == MAP_SQUARE;
569}

571static void MapBuildRoom(
572 Map *map,
573 const Vec2i pos,
574 const Vec2i size,
575 const RoomParams r,
576 const bool hasKeys,
577 const bool isOverlapRoom,
578 const unsigned short overlapAccess)
579{
580 // For edges, any tile that was a floor must be turned into
581 // a door, unless it was a corner - then it must be a wall
582 Vec2i v;
583 for (v.y = pos.y; v.y < pos.y + size.y; v.y++)
584 {
585 for (v.x = pos.x; v.x < pos.x + size.x; v.x++)
586 {
587 if (v.y > pos.y && v.y < pos.y + size.y - 1 &&
588 v.x > pos.x && v.x < pos.x + size.x - 1)
589 {
590 continue;
591 }
592 const bool isTop = v.y == pos.y;
593 const bool isBottom = v.y == pos.y + size.y - 1;
594 const bool isLeft = v.x == pos.x;
595 const bool isRight = v.x == pos.x + size.x - 1;
596 const Vec2i outside = Vec2iNew(
597 isLeft ? pos.x - 1 : (isRight ? pos.x + size.x : v.x),
598 isTop ? pos.y - 1 : (isBottom ? pos.y + size.y : v.y));
599 const Vec2i outsideX = Vec2iNew(
600 (isLeft || isRight) ? outside.x : v.x, v.y);
601 const Vec2i outsideY = Vec2iNew(
602 v.x, (isTop || isBottom) ? outside.y : v.y);
603 const bool atEdgeOfMap =
604 v.y == 0 || v.y == map->Size.y - 1 ||
605 v.x == 0 || v.x == map->Size.x - 1;
606 switch (IMapGet(map, v))
607 {
608 case MAP_FLOOR:
609 // Check outside tiles
610 if (!atEdgeOfMap &&
611 CaveRoomOutsideOk(map, outside) &&
612 CaveRoomOutsideOk(map, outsideX) &&
613 CaveRoomOutsideOk(map, outsideY))
614 {
615 IMapSet(map, v, MAP_DOOR);
616 }
617 else
618 {
619 IMapSet(map, v, MAP_WALL);
620 }
621 break;
622 default:
623 // do nothing
624 break;
625 }
626
627 const bool leftOrRightEdge = isLeft || isRight;
628 const bool topOrBottomEdge = isTop || isBottom;
629 if (leftOrRightEdge && topOrBottomEdge)
630 {
631 // corner
632 IMapSet(map, v, MAP_WALL);
633 }
634 }
635 }
636
637 MapMakeRoom(map, pos, size, false);
638
639 // TODO: keys/access level
640 UNUSED(hasKeys);
641 UNUSED(isOverlapRoom);
642 UNUSED(overlapAccess);
643
644 MapMakeRoomWalls(map, r);
645}
