25#ifndef _SDL_joystickbuttonnames_h
26#define _SDL_joystickbuttonnames_h
27
28#include <SDL_gamecontroller.h>
29
30#ifdef __cplusplus
31extern "C" {
32#endif
33
34/**
35 * \return -1 on error; 0 on success
36 */
37int SDLJBN_Init(void);
38
39void SDLJBN_Quit(void);
40
41/**
42 * Load a set of mappings from a file
43 *
44 * \return number of mappings added, -1 on error
45 */
46int SDLJBN_AddMappingsFromFile(const char *file);
47
48/**
49 * Get the button name and color for an SDL_GameControllerButton
50 * Arguments can be set to NULL if they are not required
51 * Use SDLJBN_GetError to get the error reason
52 *
53 * \return -1 on error; 0 on success
54 */
55int SDLJBN_GetButtonNameAndColor(SDL_Joystick *joystick,
56 SDL_GameControllerButton button,
57 char *s, Uint8 *r, Uint8 *g, Uint8 *b);
58/**
59 * Get the axis name and color for an SDL_GameControllerAxis
60 * Arguments can be set to NULL if they are not required
61 * Use SDLJBN_GetError to get the error reason
62 *
63 * \return -1 on error; 0 on success
64 */
65int SDLJBN_GetAxisNameAndColor(SDL_Joystick *joystick,
66 SDL_GameControllerAxis axis,
67 char *s, Uint8 *r, Uint8 *g, Uint8 *b);
68
69/* Convenience macros */
70#define SDLJBN_GetButtonName(joystick, button, s)\
71 SDLJBN_GetButtonNameAndColor(joystick, button, s, NULL, NULL, NULL);
72#define SDLJBN_GetButtonColor(joystick, button, r, g, b)\
73 SDLJBN_GetButtonNameAndColor(joystick, button, NULL, r, g, b);
74#define SDLJBN_GetAxisName(joystick, axis, s)\
75 SDLJBN_GetAxisNameAndColor(joystick, axis, s, NULL, NULL, NULL);
76#define SDLJBN_GetAxisColor(joystick, axis, r, g, b)\
77 SDLJBN_GetAxisNameAndColor(joystick, axis, NULL, r, g, b);
78
79const char *SDLJBN_GetError(void);
80

82}
