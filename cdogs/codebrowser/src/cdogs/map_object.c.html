60const char *PlacementFlagStr(const int i)
61{
62 switch (i)
63 {
64 T2S(PLACEMENT_OUTSIDE, "Outside");
65 T2S(PLACEMENT_INSIDE, "Inside");
66 T2S(PLACEMENT_NO_WALLS, "NoWalls");
67 T2S(PLACEMENT_ONE_WALL, "OneWall");
68 T2S(PLACEMENT_ONE_OR_MORE_WALLS, "OneOrMoreWalls");
69 T2S(PLACEMENT_FREE_IN_FRONT, "FreeInFront");
70 T2S(PLACEMENT_ON_WALL, "OnWall");
71 default:
72 return "";
73 }

95MapObject *StrMapObject(const char *s)
96{
97 if (s == NULL || strlen(s) == 0)
98 {
99 return NULL;
100 }
101 CA_FOREACH(MapObject, c, gMapObjects.CustomClasses)
102 if (strcmp(s, c->Name) == 0)
103 {
104 return c;
105 }
106 CA_FOREACH_END()
107 CA_FOREACH(MapObject, c, gMapObjects.Classes)
108 if (strcmp(s, c->Name) == 0)
109 {
110 return c;
111 }
112 CA_FOREACH_END()
113 return NULL;
114}

169{

202int MapObjectGetFlags(const MapObject *mo)
203{
204 int flags = 0;
205 if (mo->DrawLast)
206 {
207 flags |= TILEITEM_DRAW_LAST;
208 }
209 if (mo->Health > 0 && !(mo->Flags & (1 << PLACEMENT_ON_WALL)))
210 {
211 flags |= TILEITEM_IMPASSABLE;
212 flags |= TILEITEM_CAN_BE_SHOT;
213 }
214 return flags;
215}

219void MapObjectsInit(
220 MapObjects *classes, const char *filename,
221 const AmmoClasses *ammo, const GunClasses *guns)
222{
223 CArrayInit(&classes->Classes, sizeof(MapObject));
224 CArrayInit(&classes->CustomClasses, sizeof(MapObject));
225 CArrayInit(&classes->Destructibles, sizeof(char *));
226 CArrayInit(&classes->Bloods, sizeof(char *));
227
228 char buf[CDOGS_PATH_MAX];
229 GetDataFilePath(buf, filename);
230 FILE *f = fopen(buf, "r");
231 json_t *root = NULL;
232 if (f == NULL)
233 {
234 LOG(LM_MAIN, LL_ERROR, "Error: cannot load map objects file %s", buf);
235 goto bail;
236 }
237 enum json_error e = json_stream_parse(f, &root);
238 if (e != JSON_OK)
239 {
240 LOG(LM_MAIN, LL_ERROR, "Error parsing map objects file %s", buf);
241 goto bail;
242 }
243 MapObjectsLoadJSON(&classes->Classes, root);
244
245 // Load initial ammo/weapon spawners
246 MapObjectsLoadAmmoAndGunSpawners(classes, ammo, guns, false);
247
248bail:
249 if (f != NULL)
250 {
251 fclose(f);
252 }
253 json_free_value(&root);
254}
255static bool TryLoadMapObject(MapObject *m, json_t *node, const int version);
256static void ReloadDestructibles(MapObjects *mo);
257void MapObjectsLoadJSON(CArray *classes, json_t *root)
258{
259 int version;
260 LoadInt(&version, root, "Version");
261 if (version > VERSION || version <= 0)
262 {
263 CASSERT(false, "cannot read map objects file version");
264 return;
265 }
266
267 json_t *pickupsNode = json_find_first_label(root, "MapObjects")->child;
268 for (json_t *child = pickupsNode->child; child; child = child->next)
269 {
270 MapObject m;
271 if (TryLoadMapObject(&m, child, version))
272 {
273 CArrayPushBack(classes, &m);
274 }
275 }
276
277 ReloadDestructibles(&gMapObjects);
278 // Load blood objects
279 CArrayClear(&gMapObjects.Bloods);
280 for (int i = 0;; i++)
281 {
282 char buf[CDOGS_FILENAME_MAX];
283 sprintf(buf, "blood%d", i);
284 if (StrMapObject(buf) == NULL)
285 {
286 break;
287 }
288 char *tmp;
289 CSTRDUP(tmp, buf);
290 CArrayPushBack(&gMapObjects.Bloods, &tmp);
291 }
292}
293static bool TryLoadMapObject(MapObject *m, json_t *node, const int version)
294{
295 memset(m, 0, sizeof *m);
296
297 m->Name = GetString(node, "Name");
298
299 // Pic
300 json_t *normalNode = json_find_first_label(node, "Pic");
301 if (normalNode != NULL && normalNode->child != NULL)
302 {
303 if (version < 2)
304 {
305 CPicLoadNormal(&m->Pic, normalNode->child);
306 }
307 else
308 {
309 CPicLoadJSON(&m->Pic, normalNode->child);
310 }
311 // Pic required for map object
312 if (!CPicIsLoaded(&m->Pic))
313 {
314 LOG(LM_MAIN, LL_ERROR, "pic not found for map object(%s)", m->Name);
315 goto bail;
316 }
317 }
318 if (CPicIsLoaded(&m->Pic))
319 {
320 // Default offset: centered X, align bottom of tile and sprite
321 const Vec2i size = CPicGetSize(&m->Pic);
322 m->Offset = Vec2iNew(-size.x / 2, TILE_HEIGHT / 2 - size.y);
323 LoadVec2i(&m->Offset, node, "Offset");
324 }
325
326 // Wreck
327 if (version < 3)
328 {
329 // Assume old wreck pic is wreck
330 json_t *wreckNode = json_find_first_label(node, "WreckPic");
331 if (wreckNode != NULL && wreckNode->child != NULL)
332 {
333 if (version < 2)
334 {
335 LoadStr(&m->Wreck, node, "WreckPic");
336 }
337 else
338 {
339 LoadStr(&m->Wreck, wreckNode, "Pic");
340 }
341 }
342 }
343 else
344 {
345 LoadStr(&m->Wreck, node, "Wreck");
346 }
347
348 // Default tile size
349 m->Size = TILE_SIZE;
350 LoadVec2i(&m->Size, node, "Size");
351 LoadVec2i(&m->PosOffset, node, "PosOffset");
352 LoadInt(&m->Health, node, "Health");
353 LoadBulletGuns(&m->DestroyGuns, node, "DestroyGuns");
354
355 // Flags
356 json_t *flagsNode = json_find_first_label(node, "Flags");
357 if (flagsNode != NULL && flagsNode->child != NULL)
358 {
359 for (json_t *flagNode = flagsNode->child->child;
360 flagNode;
361 flagNode = flagNode->next)
362 {
363 m->Flags |= 1 << StrPlacementFlag(flagNode->text);
364 }
365 }
366
367 LoadBool(&m->DrawLast, node, "DrawLast");
368
369 // Special types
370 JSON_UTILS_LOAD_ENUM(m->Type, node, "Type", StrMapObjectType);
371 switch (m->Type)
372 {
373 case MAP_OBJECT_TYPE_NORMAL:
374 // Do nothing
375 break;
376 case MAP_OBJECT_TYPE_PICKUP_SPAWNER:
377 {
378 char *tmp = GetString(node, "Pickup");
379 m->u.PickupClass = StrPickupClass(tmp);
380 CFREE(tmp);
381 }
382 break;
383 default:
384 CASSERT(false, "unknown error");
385 break;
386 }
387
388 // DestroySpawn - pickups to spawn on destruction
389 json_t *destroySpawnNode = json_find_first_label(node, "DestroySpawn");
390 if (destroySpawnNode != NULL && destroySpawnNode->child != NULL)
391 {
392 CArrayInit(&m->DestroySpawn, sizeof(MapObjectDestroySpawn));
393 for (json_t *dsNode = destroySpawnNode->child->child;
394 dsNode;
395 dsNode = dsNode->next)
396 {
397 MapObjectDestroySpawn mods;
398 memset(&mods, 0, sizeof mods);
399 JSON_UTILS_LOAD_ENUM(mods.Type, dsNode, "Type", StrPickupType);
400 LoadDouble(&mods.SpawnChance, dsNode, "SpawnChance");
401 CArrayPushBack(&m->DestroySpawn, &mods);
402 }
403 }
404
405 return true;
406
407bail:
408 return false;
409}
410static void AddDestructibles(MapObjects *mo, const CArray *classes);
411static void ReloadDestructibles(MapObjects *mo)
412{
413 CA_FOREACH(char *, s, mo->Destructibles)
414 CFREE(*s);
415 CA_FOREACH_END()
416 CArrayClear(&mo->Destructibles);
417 AddDestructibles(mo, &mo->Classes);
418 AddDestructibles(mo, &mo->CustomClasses);
419}
420static void AddDestructibles(MapObjects *m, const CArray *classes)
421{
422 for (int i = 0; i < (int)classes->size; i++)
423 {
424 const MapObject *mo = CArrayGet(classes, i);
425 if (mo->Health > 0)
426 {
427 char *s;
428 CSTRDUP(s, mo->Name);
429 CArrayPushBack(&m->Destructibles, &s);
430 }
431 }
432}

435static void LoadGunSpawners(CArray *classes, const CArray *guns);
436void MapObjectsLoadAmmoAndGunSpawners(
437 MapObjects *classes, const AmmoClasses *ammo, const GunClasses *guns,
438 const bool isCustom)
439{
440 if (isCustom)
441 {
442 LoadAmmoSpawners(&classes->CustomClasses, &ammo->CustomAmmo);
443 LoadGunSpawners(&classes->CustomClasses, &guns->CustomGuns);
444 }
445 else
446 {
447 // Load built-in classes
448 LoadAmmoSpawners(&classes->Classes, &ammo->Ammo);
449 LoadGunSpawners(&classes->Classes, &guns->Guns);
450 }
451}

454 MapObject *m, const char *spawnerName, const char *pickupClassName);

456static void LoadAmmoSpawners(CArray *classes, const CArray *ammo)
457{
458 for (int i = 0; i < (int)ammo->size; i++)
459 {
460 const Ammo *a = CArrayGet(ammo, i);
461 MapObject m;
462 char spawnerName[256];
463 char pickupClassName[256];
464 sprintf(spawnerName, "%s ammo spawner", a->Name);
465 sprintf(pickupClassName, "ammo_%s", a->Name);
466 SetupSpawner(&m, spawnerName, pickupClassName);
467 CArrayPushBack(classes, &m);
468 }
469}
470static void LoadGunSpawners(CArray *classes, const CArray *guns)
471{
472 for (int i = 0; i < (int)guns->size; i++)
473 {
474 const GunDescription *g = CArrayGet(guns, i);
475 if (!g->IsRealGun)
476 {
477 continue;
478 }
479 MapObject m;
480 char spawnerName[256];
481 char pickupClassName[256];
482 sprintf(spawnerName, "%s spawner", g->name);
483 sprintf(pickupClassName, "gun_%s", g->name);
484 SetupSpawner(&m, spawnerName, pickupClassName);
485 CArrayPushBack(classes, &m);
486 }
487}

489static void SetupSpawner(
490 MapObject *m, const char *spawnerName, const char *pickupClassName)
491{
492 memset(m, 0, sizeof *m);
493 CSTRDUP(m->Name, spawnerName);
494 m->Pic.Type = PICTYPE_NORMAL;
495 m->Pic.u.Pic = PicManagerGetPic(&gPicManager, "spawn_pad");
496 m->Pic.UseMask = true;
497 m->Pic.u1.Mask = colorWhite;
498 const Vec2i size = CPicGetSize(&m->Pic);
499 m->Offset = Vec2iNew(-size.x / 2, TILE_HEIGHT / 2 - size.y);
500 m->Size = TILE_SIZE;
501 m->Health = 0;
502 m->DrawLast = true;
503 m->Type = MAP_OBJECT_TYPE_PICKUP_SPAWNER;
504 m->u.PickupClass = StrPickupClass(pickupClassName);
505}

507void MapObjectsClear(CArray *classes)
508{
509 for (int i = 0; i < (int)classes->size; i++)
510 {
511 MapObject *c = CArrayGet(classes, i);
512 CFREE(c->Name);
513 CArrayTerminate(&c->DestroyGuns);

516 CArrayClear(classes);
517}
518void MapObjectsTerminate(MapObjects *classes)
519{
520 MapObjectsClear(&classes->Classes);
521 CArrayTerminate(&classes->Classes);
522 MapObjectsClear(&classes->CustomClasses);
523 CArrayTerminate(&classes->CustomClasses);
524 CA_FOREACH(char *, s, classes->Destructibles)
525 CFREE(*s);
526 CA_FOREACH_END()
527 CArrayTerminate(&classes->Destructibles);
528 CA_FOREACH(char *, s, classes->Bloods)
529 CFREE(*s);
530 CA_FOREACH_END()
531 CArrayTerminate(&classes->Bloods);
532}

547bool MapObjectIsTileOK(
548 const MapObject *obj, unsigned short tile, const bool isEmpty,
549 unsigned short tileAbove)
550{
551 tile &= MAP_MASKACCESS;
552 if (tile != MAP_FLOOR && tile != MAP_SQUARE && tile != MAP_ROOM)
553 {
554 return 0;
555 }
556 if (!isEmpty)
557 {
558 return 0;
559 }
560 tileAbove &= MAP_MASKACCESS;
561 if ((obj->Flags & (1 << PLACEMENT_ON_WALL)) && tileAbove != MAP_WALL)
562 {
563 return 0;
564 }
565 return 1;
566}
567bool MapObjectIsTileOKStrict(
568 const MapObject *obj, const unsigned short tile, const bool isEmpty,
569 const unsigned short tileAbove, const unsigned short tileBelow,
570 const int numWallsAdjacent, const int numWallsAround)
571{
572 if (!MapObjectIsTileOK(obj, tile, isEmpty, tileAbove))
573 {
574 return 0;
575 }
576 unsigned short tileAccess = tile & MAP_MASKACCESS;
577 if (tile & MAP_LEAVEFREE)
578 {
579 return 0;
580 }
581
582 if (obj->Flags & (1 << PLACEMENT_OUTSIDE) && tileAccess == MAP_ROOM)
583 {
584 return false;
585 }
586 if ((obj->Flags & (1 << PLACEMENT_INSIDE)) && tileAccess != MAP_ROOM)
587 {
588 return false;
589 }
590 if ((obj->Flags & (1 << PLACEMENT_NO_WALLS)) && numWallsAround != 0)
591 {
592 return false;
593 }
594 if ((obj->Flags & (1 << PLACEMENT_ONE_WALL)) && numWallsAdjacent != 1)
595 {
596 return false;
597 }
598 if ((obj->Flags & (1 << PLACEMENT_ONE_OR_MORE_WALLS)) && numWallsAdjacent < 1)
599 {
600 return false;
601 }
602 if ((obj->Flags & (1 << PLACEMENT_FREE_IN_FRONT)) &&
603 (tileBelow & MAP_MASKACCESS) != MAP_FLOOR &&
604 (tileBelow & MAP_MASKACCESS) != MAP_SQUARE &&
605 (tileBelow & MAP_MASKACCESS) != MAP_ROOM)
606 {
607 return false;
608 }
609
610 return true;
611}
612Vec2i MapObjectGetPlacementPos(const MapObject *mo, const Vec2i tilePos)
613{
614 Vec2i pos = Vec2iCenterOfTile(tilePos);
615 pos = Vec2iAdd(pos, mo->PosOffset);
616 // For on-wall objects, set their position to the top of the tile
617 // This guarantees that they are drawn last
618 if (mo->Flags & (1 << PLACEMENT_ON_WALL))
619 {
620 pos.y -= TILE_HEIGHT / 2 + 1;
621 }
622 return pos;
623}
