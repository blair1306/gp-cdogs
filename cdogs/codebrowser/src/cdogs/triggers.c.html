59CArray gWatches; // of TWatch
60static int watchIndex = 1;
61
62// Number of frames to wait before repeating the "cannot activate" event
63#define CANNOT_ACTIVATE_LOCK 50
64
65
66Trigger *TriggerNew(void)
67{
68 Trigger *t;
69 CCALLOC(t, sizeof *t);
70 t->isActive = 1;
71 CArrayInit(&t->actions, sizeof(Action));
72 return t;
73}
74void TriggerTerminate(Trigger *t)
75{
76 CArrayTerminate(&t->actions);
77 CFREE(t);
78}
79Action *TriggerAddAction(Trigger *t)
80{
81 Action a;
82 memset(&a, 0, sizeof a);
83 CArrayPushBack(&t->actions, &a);
84 return CArrayGet(&t->actions, t->actions.size - 1);
85}

87TWatch *WatchNew(void)
88{
89 TWatch t;
90 memset(&t, 0, sizeof(TWatch));
91 t.index = watchIndex++;
92 CArrayInit(&t.actions, sizeof(Action));
93 CArrayInit(&t.conditions, sizeof(Condition));
94 t.active = false;
95 CArrayPushBack(&gWatches, &t);
96 return CArrayGet(&gWatches, gWatches.size - 1);
97}
98Condition *WatchAddCondition(
99 TWatch *w, const ConditionType type, const int counterMax,
100 const Vec2i pos)
101{
102 Condition c;
103 memset(&c, 0, sizeof c);
104 c.Type = type;
105 c.CounterMax = counterMax;
106 c.Pos = pos;
107 CArrayPushBack(&w->conditions, &c);
108 return CArrayGet(&w->conditions, w->conditions.size - 1);
109}
110Action *WatchAddAction(TWatch *w)
111{
112 Action a;
113 memset(&a, 0, sizeof a);
114 CArrayPushBack(&w->actions, &a);
115 return CArrayGet(&w->actions, w->actions.size - 1);
116}

118static void ActivateWatch(int idx)
119{
120 CA_FOREACH(TWatch, w, gWatches)
121 if (w->index == idx)
122 {
123 w->active = true;
124
125 // Reset all conditions related to watch
126 for (int j = 0; j < (int)w->conditions.size; j++)
127 {
128 Condition *c = CArrayGet(&w->conditions, j);
129 c->Counter = 0;
130 }
131 return;
132 }
133 CA_FOREACH_END()
134 CASSERT(false, "Cannot find watch");
135}
136
137static void DeactivateWatch(int idx)
138{
139 CA_FOREACH(TWatch, w, gWatches)
140 if (w->index == idx)
141 {
142 w->active = false;
143 return;
144 }
145 CA_FOREACH_END()
146 CASSERT(false, "Cannot find watch");
147}

149void WatchesInit(void)
150{
151 CArrayInit(&gWatches, sizeof(TWatch));

153void WatchesTerminate(void)
154{
155 CA_FOREACH(TWatch, w, gWatches)
156 CArrayTerminate(&w->conditions);
157 CArrayTerminate(&w->actions);
158 CA_FOREACH_END()
159 CArrayTerminate(&gWatches);
160}

162static void ActionRun(Action *a, CArray *mapTriggers)
163{
164 switch (a->Type)
165 {
166 case ACTION_NULL:
167 return;
168
169 case ACTION_SETTRIGGER:
170 for (int i = 0; i < (int)mapTriggers->size; i++)
171 {
172 Trigger *tr = *(Trigger **)CArrayGet(mapTriggers, i);
173 if (tr->id == a->u.index)
174 {
175 tr->isActive = 1;
176 break;
177 }
178 }
179 break;
180
181 case ACTION_CLEARTRIGGER:
182 for (int i = 0; i < (int)mapTriggers->size; i++)
183 {
184 Trigger *tr = *(Trigger **)CArrayGet(mapTriggers, i);
185 if (tr->id == a->u.index)
186 {
187 tr->isActive = 0;
188 break;
189 }
190 }
191 break;
192
193 case ACTION_EVENT:
194 GameEventsEnqueue(&gGameEvents, a->a.Event);
195 break;
196
197 case ACTION_ACTIVATEWATCH:
198 ActivateWatch(a->u.index);
199 break;
200
201 case ACTION_DEACTIVATEWATCH:
202 DeactivateWatch(a->u.index);
203 break;
204
205 case ACTION_SOUND:
206 SoundPlayAt(&gSoundDevice, a->a.Sound, a->u.pos);
207 break;
208 }
209}

211static bool ConditionsMet(CArray *conditions, const int ticks)
212{
213 bool allConditionsMet = true;
214 for (int i = 0; i < (int)conditions->size; i++)
215 {
216 Condition *c = CArrayGet(conditions, i);
217 bool conditionMet = false;
218 switch (c->Type)
219 {
220 case CONDITION_TILECLEAR:
221 conditionMet = TileIsClear(MapGetTile(&gMap, c->Pos));
222 break;
223 }
224 if (conditionMet)
225 {
226 c->Counter += ticks;
227 allConditionsMet =
228 allConditionsMet && c->Counter >= c->CounterMax;
229 }
230 else
231 {
232 c->Counter = 0;
233 allConditionsMet = false;
234 }
235 }
236 return allConditionsMet;
237}

239bool TriggerTryActivate(Trigger *t, const int flags, const Vec2i tilePos)
240{
241 const bool canActivate =
242 t->isActive && (t->flags == 0 || (t->flags & flags));
243 if (canActivate)
244 {
245 GameEvent e = GameEventNew(GAME_EVENT_TRIGGER);
246 e.u.TriggerEvent.ID = t->id;
247 e.u.TriggerEvent.Tile = Vec2i2Net(tilePos);
248 GameEventsEnqueue(&gGameEvents, e);
249 }
250 else
251 {
252 if (t->cannotActivateLock > 0)
253 {
254 t->cannotActivateLock--;
255 }
256 }
257 return canActivate;
258}
259
260bool TriggerCannotActivate(const Trigger *t)
261{
262 return t->cannotActivateLock == 0;
263}
264
265void TriggerSetCannotActivate(Trigger *t)
266{
267 t->cannotActivateLock = CANNOT_ACTIVATE_LOCK;
268}
269
270void TriggerActivate(Trigger *t, CArray *mapTriggers)
271{
272 CA_FOREACH(Action, a, t->actions)
273 ActionRun(a, mapTriggers);
274 CA_FOREACH_END()
275}
276
277void UpdateWatches(CArray *mapTriggers, const int ticks)
278{
279 CA_FOREACH(TWatch, w, gWatches)
280 if (!w->active) continue;
281 if (ConditionsMet(&w->conditions, ticks))
282 {
283 for (int j = 0; j < (int)w->actions.size; j++)
284 {
285 ActionRun(CArrayGet(&w->actions, j), mapTriggers);
286 }
287 }
288 CA_FOREACH_END()
289}
