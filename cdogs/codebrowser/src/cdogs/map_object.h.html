77// Pickups to spawn when map objects are destroyed
78typedef struct
79{
80 PickupType Type;

82} MapObjectDestroySpawn;

84// A static map object, taking up an entire tile
85typedef struct
86{
87 char *Name;
88 CPic Pic;
89 Vec2i Offset;
90 char *Wreck;
91 Vec2i Size;
92 Vec2i PosOffset;
93 int Health;
94 // Guns that are fired when this map object is destroyed
95 // i.e. explosion on destruction
96 CArray DestroyGuns; // of const GunDescription *
97 // Bit field composed of bits shifted by PlacementFlags
98 int Flags;
99 bool DrawLast;
100 MapObjectType Type;
101 union
102 {
103 const PickupClass *PickupClass;
104 } u;
105 CArray DestroySpawn; // of MapObjectDestroySpawn
106} MapObject;
107typedef struct
108{
109 CArray Classes; // of MapObject
110 CArray CustomClasses; // of MapObject
111 // Names of special types of map objects; for editor support
112 // Reset on load
113 CArray Destructibles; // of char *
114 // Map objects that match "blood%d" - left over when actors die
115 CArray Bloods; // of char *
116} MapObjects;
