70// Which "team" the actor's on, for collision
71// Actors on the same team don't have to collide
72typedef enum
73{
74 COLLISIONTEAM_NONE,
75 COLLISIONTEAM_GOOD,
76 COLLISIONTEAM_BAD
77} CollisionTeam;

79CollisionTeam CalcCollisionTeam(const bool isActor, const TActor *actor);

94// Get all TTileItem that overlap with a target TTileItem, with callback.
95// The callback returns bool continue, as multiple callbacks can result.
96typedef bool (*CollideItemFunc)(
97 TTileItem *, void *, const Vec2i, const Vec2i, const Vec2i);
98typedef bool (*CheckWallFunc)(const Vec2i);
99typedef bool (*CollideWallFunc)(const Vec2i, void *, const Vec2i, const Vec2i);
100void OverlapTileItems(
101 const TTileItem *item, const Vec2i pos, const Vec2i size,
102 const CollisionParams params, CollideItemFunc func, void *data,
103 CheckWallFunc checkWallFunc, CollideWallFunc wallFunc, void *wallData);
104// Get the first TTileItem that overlaps
105TTileItem *OverlapGetFirstItem(
106 const TTileItem *item, const Vec2i pos, const Vec2i size,
107 const CollisionParams params);

109bool AABBOverlap(
