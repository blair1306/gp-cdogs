70static void TileCacheAddImpl(
71 CArray *tc, const Vec2i v, const bool addAdjacents);
72static void TileCacheAdd(CArray *tc, const Vec2i v)
73{
74 TileCacheAddImpl(tc, v, true);
75}
76static void TileCacheAddImpl(
77 CArray *tc, const Vec2i v, const bool addAdjacents)
78{
79 if (!MapIsTileIn(&gMap, v))
80 {
81 return;
82 }
83 // Add tile in y/x order
84 CA_FOREACH(const Vec2i, t, *tc)
85 if (t->y > v.y || (t->y == v.y && t->x > v.x))
86 {
87 CArrayInsert(tc, _ca_index, &v);
88 break;
89 }
90 else if (Vec2iEqual(*t, v))
91 {
92 // Don't add the same tile twice
93 return;
94 }
95 CA_FOREACH_END()
96 CArrayPushBack(tc, &v);
97
98 // Also add the adjacencies for the tile
99 if (addAdjacents)
100 {
101 Vec2i dv;
102 for (dv.y = -1; dv.y <= 1; dv.y++)
103 {
104 for (dv.x = -1; dv.x <= 1; dv.x++)
105 {
106 if (Vec2iIsZero(dv))
107 {
108 continue;
109 }
110 const Vec2i dtv = Vec2iAdd(v, dv);
111 if (!MapIsTileIn(&gMap, dtv))
112 {
113 continue;
114 }
115 TileCacheAddImpl(tc, dtv, false);
116 }
117 }
118 }
119}

138CollisionTeam CalcCollisionTeam(const bool isActor, const TActor *actor)
139{
140 // Need to have prisoners collide with everything otherwise they will not
141 // be "rescued"
142 // Also need victims to collide with everyone
143 if (!isActor || (actor->flags & (FLAGS_PRISONER | FLAGS_VICTIM)) ||
144 IsPVP(gCampaign.Entry.Mode))
145 {
146 return COLLISIONTEAM_NONE;
147 }
148 if (actor->PlayerUID >= 0 || (actor->flags & FLAGS_GOOD_GUY))
149 {
150 return COLLISIONTEAM_GOOD;
151 }
152 return COLLISIONTEAM_BAD;
153}

155bool IsCollisionWithWall(const Vec2i pos, const Vec2i fullSize)
156{
157 Vec2i size = Vec2iScaleDiv(fullSize, 2);
158 if (pos.x - size.x < 0 ||
159 pos.y - size.y < 0 ||
160 pos.x + size.x >= gMap.Size.x * TILE_WIDTH ||
161 pos.y + size.y >= gMap.Size.y * TILE_HEIGHT)
162 {
163 return true;
164 }
165 if (HitWall(pos.x - size.x, pos.y - size.y) ||
166 HitWall(pos.x - size.x, pos.y) ||
167 HitWall(pos.x - size.x, pos.y + size.y) ||
168 HitWall(pos.x, pos.y + size.y) ||
169 HitWall(pos.x + size.x, pos.y + size.y) ||
170 HitWall(pos.x + size.x, pos.y) ||
171 HitWall(pos.x + size.x, pos.y - size.y) ||
172 HitWall(pos.x, pos.y - size.y))
173 {
174 return true;
175 }
176 return false;
177}

179// Check collision with a diamond shape
180// This means that the bounding box could be in collision, but the bounding
181// "radius" is not. The diamond is expressed with a single "radius" - that is,
182// the diamond is the same height and width.
183// This arrangement is used so that axis movement can slide off corners by
184// moving in a diagonal direction.
185// E.g. this is not a collision:
186// x
187// x x
188// x x
189// x x
190// x x
191// x x wwwww
192// x w
193// w
194// Where 'x' denotes the bounding diamond, and 'w' represents a wall corner.
195bool IsCollisionDiamond(const Map *map, const Vec2i pos, const Vec2i fullSize)
196{
197 const Vec2i mapSize =
198 Vec2iNew(map->Size.x * TILE_WIDTH, map->Size.y * TILE_HEIGHT);
199 const Vec2i size = Vec2iScaleDiv(fullSize, 2);
200 if (pos.x - size.x < 0 || pos.x + size.x >= mapSize.x ||
201 pos.y - size.y < 0 || pos.y + size.y >= mapSize.y)
202 {
203 return true;
204 }
205
206 // Only support wider-than-taller collision diamonds for now
207 CASSERT(size.x >= size.y, "not implemented, taller than wider diamond");
208 const double gradient = (double)size.y / size.x;
209
210 // Now we need to check in a diamond pattern that the boundary does not
211 // collide
212 // Top to right
213 for (int i = 0; i < size.x; i++)
214 {
215 const int y = (int)Round((-size.x + i)* gradient);
216 const Vec2i p = Vec2iAdd(pos, Vec2iNew(i, y));
217 if (HitWall(p.x, p.y))
218 {
219 return true;
220 }
221 }
222 // Right to bottom
223 for (int i = 0; i < size.x; i++)
224 {
225 const int y = (int)Round(i * gradient);
226 const Vec2i p = Vec2iAdd(pos, Vec2iNew(size.x - i, y));
227 if (HitWall(p.x, p.y))
228 {
229 return true;
230 }
231 }
232 // Bottom to left
233 for (int i = 0; i < size.x; i++)
234 {
235 const int y = (int)Round((size.x - i) * gradient);
236 const Vec2i p = Vec2iAdd(pos, Vec2iNew(-i, y));
237 if (HitWall(p.x, p.y))
238 {
239 return true;
240 }
241 }
242 // Left to top
243 for (int i = 0; i < size.x; i++)
244 {
245 const int y = (int)Round(-i * gradient);
246 const Vec2i p = Vec2iAdd(pos, Vec2iNew(-size.x + i, y));
247 if (HitWall(p.x, p.y))
248 {
249 return true;
250 }
251 }
252 return false;
253}

255bool AABBOverlap(
256 const Vec2i pos1, const Vec2i pos2, const Vec2i size1, const Vec2i size2)
257{
258 // Use Minkowski addition to check overlap of two rects
259 const Vec2i d = Vec2iNew(abs(pos1.x - pos2.x), abs(pos1.y - pos2.y));
260 const Vec2i r = Vec2iScaleDiv(Vec2iAdd(size1, size2), 2);
261 return d.x < r.x && d.y < r.y;
262}

264static bool CollisionIsOnSameTeam(
265 const TTileItem *i, const CollisionTeam team, const bool isPVP)
266{
267 if (gCollisionSystem.allyCollision == ALLYCOLLISION_NORMAL)
268 {
269 return false;
270 }
271 CollisionTeam itemTeam = COLLISIONTEAM_NONE;
272 if (i->kind == KIND_CHARACTER)
273 {
274 const TActor *a = CArrayGet(&gActors, i->id);
275 itemTeam = CalcCollisionTeam(1, a);
276 }
277 return
278 team != COLLISIONTEAM_NONE &&
279 itemTeam != COLLISIONTEAM_NONE &&
280 team == itemTeam &&
281 !isPVP;
282}

287static void AddPosToTileCache(void *data, Vec2i pos);
288static bool CheckOverlaps(
289 const TTileItem *item, const Vec2i pos, const Vec2i vel, const Vec2i size,
290 const CollisionParams params, CollideItemFunc func, void *data,
291 CheckWallFunc checkWallFunc, CollideWallFunc wallFunc, void *wallData,
292 const Vec2i tilePos);
293void OverlapTileItems(
294 const TTileItem *item, const Vec2i pos, const Vec2i size,
295 const CollisionParams params, CollideItemFunc func, void *data,
296 CheckWallFunc checkWallFunc, CollideWallFunc wallFunc, void *wallData)
297{
298 TileCacheReset(&gCollisionSystem.tileCache);
299 // Add all the tiles along the motion path
300 AlgoLineDrawData drawData;
301 drawData.Draw = AddPosToTileCache;
302 drawData.data = &gCollisionSystem.tileCache;
303 const Vec2i posReal = Vec2iFull2Real(pos);
304 const Vec2i vel = Vec2iFull2Real(item->VelFull);
305 BresenhamLineDraw(posReal, Vec2iAdd(posReal, vel), &drawData);
306
307 // Check collisions with all tiles in the cache
308 CA_FOREACH(const Vec2i, dtv, gCollisionSystem.tileCache)
309 if (!CheckOverlaps(
310 item, pos, item->VelFull, size, params, func, data,
311 checkWallFunc, wallFunc, wallData,
312 *dtv))
313 {
314 return;
315 }
316 CA_FOREACH_END()
317}
318static void AddPosToTileCache(void *data, Vec2i pos)
319{
320 CArray *tileCache = data;
321 const Vec2i tv = Vec2iToTile(pos);
322 TileCacheAdd(tileCache, tv);
323}
324static bool CheckOverlaps(
325 const TTileItem *item, const Vec2i pos, const Vec2i vel, const Vec2i size,
326 const CollisionParams params, CollideItemFunc func, void *data,
327 CheckWallFunc checkWallFunc, CollideWallFunc wallFunc, void *wallData,
328 const Vec2i tilePos)
329{
330 Vec2i colA, colB, normal;
331 // Check item collisions
332 if (func != NULL)
333 {
334 const CArray *tileThings = &MapGetTile(&gMap, tilePos)->things;
335 CA_FOREACH(const ThingId, tid, *tileThings)
336 TTileItem *ti = ThingIdGetTileItem(tid);
337 if (!CheckParams(params, item, ti))
338 {
339 continue;
340 }
341 if (!MinkowskiHexCollide(
342 pos, vel, Vec2iReal2Full(size),
343 Vec2iReal2Full(Vec2iNew(ti->x, ti->y)), ti->VelFull,
344 Vec2iReal2Full(ti->size), &colA, &colB, &normal))
345 {
346 continue;
347 }
348 // Collision callback and check continue
349 if (!func(ti, data, colA, colB, normal))
350 {
351 return false;
352 }
353 CA_FOREACH_END()
354 }
355 // Check wall collisions
356 if (checkWallFunc != NULL && wallFunc != NULL && checkWallFunc(tilePos))
357 {
358 // Hack: bullets always considered 0x0 when colliding with walls
359 // TODO: bullet size for walls
360 const Vec2i sizeForWall =
361 item->kind == KIND_MOBILEOBJECT ? Vec2iZero() : size;
362 if (MinkowskiHexCollide(
363 pos, vel, Vec2iReal2Full(sizeForWall),
364 Vec2iReal2Full(Vec2iCenterOfTile(tilePos)), Vec2iZero(),
365 Vec2iReal2Full(TILE_SIZE), &colA, &colB, &normal) &&
366 !wallFunc(tilePos, wallData, colA, normal))
367 {
368 return false;
369 }
370 }
371 return true;
372}

400static bool CheckParams(
401 const CollisionParams params, const TTileItem *a, const TTileItem *b)
402{
403 // Don't collide if items are on the same team
404 if (CollisionIsOnSameTeam(b, params.Team, params.IsPVP))
405 {
406 return false;
407 }
408 // No same-item collision
409 if (a == b) return false;
410 if (params.TileItemMask != 0 && !(b->flags & params.TileItemMask))
411 {
412 return false;
413 }
414
415 return true;
416}

418void GetWallBouncePosVelFull(
419 const Vec2i pos, const Vec2i vel, const Vec2i colPos,
420 const Vec2i colNormal, Vec2i *outPos, Vec2i *outVel)
421{
422 const Vec2i velBeforeCol = Vec2iMinus(colPos, pos);
423 const Vec2i velAfterCol = Vec2iMinus(vel, velBeforeCol);
424
425 // If normal is zero, this means we were in collision from the start
426 if (Vec2iIsZero(colNormal))
427 {
428 // Reverse the vector
429 *outPos = Vec2iMinus(pos, vel);
430 *outVel = Vec2iScale(vel, -1);
431 return;
432 }
433
434 // Reflect the out position by the collision normal about the collision pos
435 const Vec2i velReflected = Vec2iNew(
436 colNormal.x == 0 ? velAfterCol.x : colNormal.x * abs(velAfterCol.x),
437 colNormal.y == 0 ? velAfterCol.y : colNormal.y * abs(velAfterCol.y));
438 *outPos = Vec2iAdd(colPos, velReflected);
439
440 // Out velocity follows the collision normal
441 *outVel = Vec2iNew(
442 colNormal.x == 0 ? vel.x : colNormal.x * abs(vel.x),
443 colNormal.y == 0 ? vel.y : colNormal.y * abs(vel.y));
444}
