31static bool RectangleLineIntersect(
32 const Vec2i rectPos, const Vec2i rectSize,

34bool MinkowskiHexCollide(
35 const Vec2i posA, const Vec2i velA, const Vec2i sizeA,
36 const Vec2i posB, const Vec2i velB, const Vec2i sizeB,
37 Vec2i *colA, Vec2i *colB, Vec2i *normal)
38{
39 // Find rectangle C by combining A and B
40 const Vec2i sizeC = Vec2iAdd(sizeA, sizeB);
41
42 // Subtract velA from vB
43 const Vec2i velBA = Vec2iMinus(velB, velA);
44
45 // Find the intersection between velBA and the rectangle C centered on posA
46 float s;
47 if (!RectangleLineIntersect(
48 posA, sizeC, posB, Vec2iAdd(posB, velBA), &s, normal))
49 {
50 return false;
51 }
52
53 // If intersection is at the start, it means we were overlapping already
54 if (s == 0)
55 {
56 *colA = posA;
57 *colB = posB;
58 }
59 else
60 {
61 // Find the actual intersection points based on the result
62 *colA = Vec2iAdd(posA, Vec2iScaleD(velA, s));
63 *colB = Vec2iAdd(posB, Vec2iScaleD(velB, s));
64 }
65
66 return true;
67}

72static bool RectangleLineIntersect(
73 const Vec2i rectPos, const Vec2i rectSize,
74 const Vec2i lineStart, const Vec2i lineEnd, float *s, Vec2i *normal)
75{
76 // Find the closest point at which a line intersects a rectangle
77 // Do this by finding intersections between the line and all four sides of
78 // the rectangle, and returning the closest one
79 const int left = rectPos.x - rectSize.x / 2;
80 const int right = left + rectSize.x;
81 const int top = rectPos.y - rectSize.y / 2;
82 const int bottom = top + rectSize.y;
83 const Vec2i topLeft = Vec2iNew(left, top);
84 const Vec2i topRight = Vec2iNew(right, top);
85 const Vec2i bottomRight = Vec2iNew(right, bottom);
86 const Vec2i bottomLeft = Vec2iNew(left, bottom);
87
88 // Border case: check if line start is entirely within rectangle
89 if (lineStart.x > left && lineStart.x < right &&
90 lineStart.y > top && lineStart.y < bottom)
91 {
92 *s = 0;
93 *normal = Vec2iZero();
94 return true;
95 }
96
97 // Find closest of four intersections
98 *s = -1;
99 float sPart;
100#define _CHECK_MIN_DISTANCE(_p1, _p2, _n) \
101 if (LinesIntersect(_p1, _p2, lineStart, lineEnd, &sPart) &&\
102 (*s < 0 || sPart < *s))\
103 {\
104 *s = sPart;\
105 *normal = _n;\
106 }
107 _CHECK_MIN_DISTANCE(topLeft, topRight, Vec2iNew(0, 1));
108 _CHECK_MIN_DISTANCE(topRight, bottomRight, Vec2iNew(-1, 0));
109 _CHECK_MIN_DISTANCE(bottomLeft, bottomRight, Vec2iNew(0, -1));
110 _CHECK_MIN_DISTANCE(topLeft, bottomLeft, Vec2iNew(1, 0));
111
112 return *s >= 0 && *s <= 1;
113}
114static bool LinesIntersect(
115 const Vec2i p1Start, const Vec2i p1End,
116 const Vec2i p2Start, const Vec2i p2End,
117 float *s)
118{
119 // Return whether two line segments intersect, and at which fraction
120 // http://stackoverflow.com/a/1968345/2038264
121 const Vec2i v1 = Vec2iMinus(p1End, p1Start);
122 const Vec2i v2 = Vec2iMinus(p2End, p2Start);
123
124 const int divisor = -v2.x * v1.y + v1.x * v2.y;
125 if (divisor == 0)
126 {
127 // Lines are parallel/collinear, just assume no collision
128 return false;
129 }
130 const Vec2i p12Start = Vec2iMinus(p1Start, p2Start);
131 *s = (float)(-v1.y * p12Start.x + v1.x * p12Start.y) / divisor;
132 const float t = (float)(v2.x * p12Start.y - v2.y * p12Start.x) / divisor;
133
134 return *s >= 0 && *s <= 1 && t >= 0 && t <= 1;
135}
