68#define VERSION 2
69void WeaponInitialize(GunClasses *g)
70{
71 memset(g, 0, sizeof *g);
72 CArrayInit(&g->Guns, sizeof(GunDescription));
73 CArrayInit(&g->CustomGuns, sizeof(GunDescription));

75static void LoadGunDescription(
76 GunDescription *g, json_t *node, const GunDescription *defaultGun,
77 const int version);
78static void GunDescriptionTerminate(GunDescription *g);
79void WeaponLoadJSON(GunClasses *g, CArray *classes, json_t *root)
80{
81 LOG(LM_MAP, LL_DEBUG, "loading weapons");
82 int version;
83 LoadInt(&version, root, "Version");
84 if (version > VERSION || version <= 0)
85 {
86 CASSERT(false, "cannot read guns file version");
87 return;
88 }
89
90 GunDescription *defaultDesc = &g->Default;
91 // Only load default gun from main game data
92 if (classes == &g->Guns)
93 {
94 json_t *defaultNode = json_find_first_label(root, "DefaultGun");
95 if (defaultNode != NULL)
96 {
97 LoadGunDescription(defaultDesc, defaultNode->child, NULL, version);
98 }
99 else
100 {
101 memset(defaultDesc, 0, sizeof *defaultDesc);
102 }
103 CASSERT(g->Guns.size == 0, "guns not empty");
104 for (int i = 0; i < GUN_COUNT; i++)
105 {
106 GunDescription gd;
107 if (defaultNode != NULL)
108 {
109 LoadGunDescription(&gd, defaultNode->child, NULL, version);
110 }
111 else
112 {
113 memset(&gd, 0, sizeof gd);
114 }
115 CArrayPushBack(&g->Guns, &gd);
116 }
117 }
118 json_t *gunsNode = json_find_first_label(root, "Guns")->child;
119 for (json_t *child = gunsNode->child; child; child = child->next)
120 {
121 GunDescription gd;
122 LoadGunDescription(&gd, child, defaultDesc, version);
123 int idx = -1;
124 // Only allow index for non-custom guns
125 if (classes == &g->Guns)
126 {
127 LoadInt(&idx, child, "Index");
128 }
129 if (idx >= 0 && idx < GUN_COUNT)
130 {
131 GunDescription *gExisting = CArrayGet(&g->Guns, idx);
132 GunDescriptionTerminate(gExisting);
133 memcpy(gExisting, &gd, sizeof gd);
134 }
135 else
136 {
137 CArrayPushBack(classes, &gd);
138 }
139 }
140 json_t *pseudoGunsNode = json_find_first_label(root, "PseudoGuns");
141 if (pseudoGunsNode != NULL)
142 {
143 for (json_t *child = pseudoGunsNode->child->child;
144 child;
145 child = child->next)
146 {
147 GunDescription gd;
148 LoadGunDescription(&gd, child, defaultDesc, version);
149 gd.IsRealGun = false;
150 CArrayPushBack(classes, &gd);
151 }
152 }
153}
154static void LoadGunDescription(
155 GunDescription *g, json_t *node, const GunDescription *defaultGun,
156 const int version)
157{
158 memset(g, 0, sizeof *g);
159 g->AmmoId = -1;
160 if (defaultGun)
161 {
162 memcpy(g, defaultGun, sizeof *g);
163 if (defaultGun->name)
164 {
165 CSTRDUP(g->name, defaultGun->name);
166 }
167 if (defaultGun->Description)
168 {
169 CSTRDUP(g->Description, defaultGun->Description);
170 }
171 g->MuzzleHeight /= Z_FACTOR;
172 }
173 char *tmp;
174
175 tmp = NULL;
176 LoadStr(&tmp, node, "Pic");
177 if (tmp != NULL)
178 {
179 char buf[CDOGS_PATH_MAX];
180 sprintf(buf, "chars/guns/%s", tmp);
181 g->Pic = PicManagerGetSprites(&gPicManager, buf);
182 CFREE(tmp);
183 }
184
185 const Pic *icon = NULL;
186 LoadPic(&icon, node, "Icon");
187 if (icon != NULL)
188 {
189 g->Icon = icon;
190 }
191
192 tmp = NULL;
193 LoadStr(&tmp, node, "Name");
194 if (tmp != NULL)
195 {
196 CFREE(g->name);
197 g->name = tmp;
198 }
199
200 tmp = NULL;
201 LoadStr(&tmp, node, "Description");
202 if (tmp != NULL)
203 {
204 CFREE(g->Description);
205 g->Description = tmp;
206 }
207
208 tmp = NULL;
209 LoadStr(&tmp, node, "Bullet");
210 if (tmp != NULL)
211 {
212 g->Bullet = StrBulletClass(tmp);
213 CFREE(tmp);
214 }
215
216 tmp = NULL;
217 LoadStr(&tmp, node, "Ammo");
218 if (tmp != NULL)
219 {
220 g->AmmoId = StrAmmoId(tmp);
221 CFREE(tmp);
222 }
223
224 LoadInt(&g->Cost, node, "Cost");
225
226 LoadInt(&g->Lock, node, "Lock");
227
228 LoadInt(&g->ReloadLead, node, "ReloadLead");
229
230 LoadSoundFromNode(&g->Sound, node, "Sound");
231 LoadSoundFromNode(&g->ReloadSound, node, "ReloadSound");
232 LoadSoundFromNode(&g->SwitchSound, node, "SwitchSound");
233
234 LoadInt(&g->SoundLockLength, node, "SoundLockLength");
235
236 LoadDouble(&g->Recoil, node, "Recoil");
237
238 LoadInt(&g->Spread.Count, node, "SpreadCount");
239 LoadDouble(&g->Spread.Width, node, "SpreadWidth");
240 LoadDouble(&g->AngleOffset, node, "AngleOffset");
241
242 LoadInt(&g->MuzzleHeight, node, "MuzzleHeight");
243 g->MuzzleHeight *= Z_FACTOR;
244 if (json_find_first_label(node, "Elevation"))
245 {
246 LoadInt(&g->ElevationLow, node, "Elevation");
247 g->ElevationHigh = g->ElevationLow;
248 }
249 LoadInt(&g->ElevationLow, node, "ElevationLow");
250 LoadInt(&g->ElevationHigh, node, "ElevationHigh");
251 g->ElevationLow = MIN(g->ElevationLow, g->ElevationHigh);
252 g->ElevationHigh = MAX(g->ElevationLow, g->ElevationHigh);
253 tmp = NULL;
254 LoadStr(&tmp, node, "MuzzleFlashParticle");
255 if (tmp != NULL)
256 {
257 g->MuzzleFlash = StrParticleClass(&gParticleClasses, tmp);
258 CFREE(tmp);
259 }
260
261 tmp = NULL;
262 LoadStr(&tmp, node, "Brass");
263 if (tmp != NULL)
264 {
265 g->Brass = StrParticleClass(&gParticleClasses, tmp);
266 CFREE(tmp);
267 }
268
269 LoadBool(&g->CanShoot, node, "CanShoot");
270
271 LoadBool(&g->CanDrop, node, "CanDrop");
272
273 LoadInt(&g->ShakeAmount, node, "ShakeAmount");
274
275 g->IsRealGun = true;
276
277 if (version < 2)
278 {
279 if (!g->CanShoot)
280 {
281 g->Lock = 0;
282 }
283 }
284
285 LOG(LM_MAP, LL_DEBUG,
286 "loaded gun name(%s) bullet(%s) ammo(%d) cost(%d) lock(%d)...",
287 g->name, g->Bullet != NULL ? g->Bullet->Name : "", g->AmmoId, g->Cost,
288 g->Lock);
289 LOG(LM_MAP, LL_DEBUG,
290 "...reloadLead(%d) soundLockLength(%d) recoil(%f)...",
291 g->ReloadLead, g->SoundLockLength, g->Recoil);
292 LOG(LM_MAP, LL_DEBUG,
293 "...spread(%frad x%d) angleOffset(%f) muzzleHeight(%d)...",
294 g->Spread.Width, g->Spread.Count, g->AngleOffset, g->MuzzleHeight);
295 LOG(LM_MAP, LL_DEBUG,
296 "...elevation(%d-%d) muzzleFlash(%s) brass(%s) canShoot(%s)...",
297 g->ElevationLow, g->ElevationHigh,
298 g->MuzzleFlash != NULL ? g->MuzzleFlash->Name : "",
299 g->Brass != NULL ? g->Brass->Name : "",
300 g->CanShoot ? "true" : "false");
301 LOG(LM_MAP, LL_DEBUG,
302 "...canDrop(%s) shakeAmount(%d)",
303 g->CanDrop ? "true" : "false", g->ShakeAmount);
304}
305void WeaponTerminate(GunClasses *g)
306{
307 WeaponClassesClear(&g->Guns);
308 CArrayTerminate(&g->Guns);
309 WeaponClassesClear(&g->CustomGuns);
310 CArrayTerminate(&g->CustomGuns);
311 GunDescriptionTerminate(&g->Default);
312}
313void WeaponClassesClear(CArray *classes)
314{
315 CA_FOREACH(GunDescription, g, *classes)
316 GunDescriptionTerminate(g);
317 CA_FOREACH_END()
318 CArrayClear(classes);
319}
320static void GunDescriptionTerminate(GunDescription *g)
321{
322 CFREE(g->name);
323 CFREE(g->Description);
324 memset(g, 0, sizeof *g);
325}

332Weapon WeaponCreate(const GunDescription *gun)
333{
334 Weapon w;

345const GunDescription *StrGunDescription(const char *s)
346{
347 CA_FOREACH(const GunDescription, gd, gGunDescriptions.CustomGuns)
348 if (strcmp(s, gd->name) == 0)
349 {
350 return gd;
351 }
352 CA_FOREACH_END()
353 CA_FOREACH(const GunDescription, gd, gGunDescriptions.Guns)
354 if (strcmp(s, gd->name) == 0)
355 {
356 return gd;
357 }
358 CA_FOREACH_END()
359 fprintf(stderr, "Cannot parse gun name: %s\n", s);
360 return NULL;
361}
362GunDescription *IdGunDescription(const int i)
363{
364 CASSERT(
365 i >= 0 &&
366 i < (int)gGunDescriptions.Guns.size +
367 (int)gGunDescriptions.CustomGuns.size,
368 "Gun index out of bounds");
369 if (i < (int)gGunDescriptions.Guns.size)
370 {
371 return CArrayGet(&gGunDescriptions.Guns, i);
372 }
373 return CArrayGet(
374 &gGunDescriptions.CustomGuns, i - gGunDescriptions.Guns.size);
375}
376int GunDescriptionId(const GunDescription *g)
377{
378 int idx = 0;
379 for (int i = 0; i < (int)gGunDescriptions.Guns.size; i++, idx++)
380 {
381 const GunDescription *c = CArrayGet(&gGunDescriptions.Guns, i);
382 if (c == g)
383 {
384 return idx;
385 }
386 }
387 for (int i = 0; i < (int)gGunDescriptions.CustomGuns.size; i++, idx++)
388 {
389 const GunDescription *c = CArrayGet(&gGunDescriptions.CustomGuns, i);
390 if (c == g)
391 {
392 return idx;
393 }
394 }
395 CASSERT(false, "cannot find gun");
396 return -1;
397}
398GunDescription *IndexGunDescriptionReal(const int i)
399{
400 int j = 0;
401 CA_FOREACH(GunDescription, g, gGunDescriptions.Guns)
402 if (!g->IsRealGun)
403 {
404 continue;
405 }
406 if (j == i)
407 {
408 return g;
409 }
410 j++;
411 CA_FOREACH_END()
412 CA_FOREACH(GunDescription, g, gGunDescriptions.CustomGuns)
413 if (!g->IsRealGun)
414 {
415 continue;
416 }
417 if (j == i)
418 {
419 return g;
420 }
421 j++;
422 CA_FOREACH_END()
423 CASSERT(false, "cannot find gun");
424 return NULL;
425}

427void WeaponUpdate(Weapon *w, const int ticks)
428{
429 w->lock -= ticks;
430 if (w->lock < 0)
431 {
432 w->lock = 0;
433 }
434 w->soundLock -= ticks;
435 if (w->soundLock < 0)
436 {
437 w->soundLock = 0;
438 }
439 w->clickLock -= ticks;
440 if (w->clickLock < 0)
441 {
442 w->clickLock = 0;
443 }
444 if (w->stateCounter >= 0)
445 {
446 w->stateCounter = MAX(0, w->stateCounter - ticks);
447 if (w->stateCounter == 0)
448 {
449 switch (w->state)
450 {
451 case GUNSTATE_FIRING:
452 WeaponSetState(w, GUNSTATE_RECOIL);
453 break;
454 case GUNSTATE_RECOIL:
455 WeaponSetState(w, GUNSTATE_FIRING);
456 break;
457 default:
458 // do nothing
459 break;
460 }
461 }
462 }
463}

470void GunFire(
471 const GunDescription *g, const Vec2i fullPos, const int z,
472 const double radians,
473 const int flags, const int playerUID, const int uid,
474 const bool playSound, const bool isGun)
475{
476 GameEvent e = GameEventNew(GAME_EVENT_GUN_FIRE);
477 e.u.GunFire.UID = uid;
478 e.u.GunFire.PlayerUID = playerUID;
479 strcpy(e.u.GunFire.Gun, g->name);
480 e.u.GunFire.MuzzleFullPos = Vec2i2Net(fullPos);
481 e.u.GunFire.Z = z;
482 e.u.GunFire.Angle = (float)radians;
483 e.u.GunFire.Sound = playSound;
484 e.u.GunFire.Flags = flags;
485 e.u.GunFire.IsGun = isGun;
486 GameEventsEnqueue(&gGameEvents, e);
487}

489void GunAddBrass(
490 const GunDescription *g, const direction_e d, const Vec2i pos)
491{
492 // Check configuration
493 if (!ConfigGetBool(&gConfig, "Graphics.Brass"))
494 {
495 return;
496 }
497 CASSERT(g->Brass, "Cannot create brass for no-brass weapon");
498 GameEvent e = GameEventNew(GAME_EVENT_ADD_PARTICLE);
499 e.u.AddParticle.Class = g->Brass;
500 double x, y;
501 const double radians = dir2radians[d];
502 GetVectorsForRadians(radians, &x, &y);
503 const Vec2i ejectionPortOffset = Vec2iReal2Full(Vec2iScale(Vec2iNew(
504 (int)round(x), (int)round(y)), 7));
505 e.u.AddParticle.FullPos = Vec2iMinus(pos, ejectionPortOffset);
506 e.u.AddParticle.Z = g->MuzzleHeight;
507 e.u.AddParticle.Vel = Vec2iScaleDiv(
508 GetFullVectorsForRadians(radians + PI / 2), 3);
509 e.u.AddParticle.Vel.x += (rand() % 128) - 64;
510 e.u.AddParticle.Vel.y += (rand() % 128) - 64;
511 e.u.AddParticle.Angle = RAND_DOUBLE(0, PI * 2);
512 e.u.AddParticle.DZ = (rand() % 6) + 6;
513 e.u.AddParticle.Spin = RAND_DOUBLE(-0.1, 0.1);
514 GameEventsEnqueue(&gGameEvents, e);
515}

517static Vec2i GetMuzzleOffset(const direction_e d);
518Vec2i GunGetMuzzleOffset(
519 const GunDescription *desc, const CharSprites *cs, const direction_e dir)
520{
521 if (!GunHasMuzzle(desc))
522 {
523 return Vec2iZero();
524 }
525 CASSERT(desc->Pic != NULL, "Gun has no pic");
526 const Vec2i gunOffset = cs->Offsets.Dir[BODY_PART_GUN][dir];
527 const Vec2i position = Vec2iAdd(gunOffset, GetMuzzleOffset(dir));
528 return Vec2iReal2Full(position);
529}
530static Vec2i GetMuzzleOffset(const direction_e d)
531{
532 // TODO: gun-specific muzzle offsets
533 #define BARREL_LENGTH 10
534 Vec2i v = Vec2iFromPolar(BARREL_LENGTH, dir2radians[d]);
535 v.y = v.y * 3 / 4;
536 return v;
537}

539void WeaponSetState(Weapon *w, const gunstate_e state)
540{
541 w->state = state;
542 switch (state)
543 {
544 case GUNSTATE_FIRING:
545 w->stateCounter = 4;
546 break;
547 case GUNSTATE_RECOIL:
548 // This is to make sure the gun stays recoiled as long as the gun is
549 // "locked", i.e. cannot fire
550 w->stateCounter = MAX(1, w->lock - 3);
551 break;
552 default:
553 w->stateCounter = -1;
554 break;
555 }
556}

562bool IsHighDPS(const GunDescription *g)
563{
564 if (g->Bullet == NULL)
565 {
566 return false;
567 }
568 // TODO: generalised way of determining explosive bullets
569 return
570 g->Bullet->Falling.DropGuns.size > 0 ||
571 g->Bullet->OutOfRangeGuns.size > 0 ||
572 g->Bullet->HitGuns.size > 0;
573}
574int GunGetRange(const GunDescription *g)
575{
576 const BulletClass *b = g->Bullet;
577 if (b == NULL)
578 {
579 return 0;
580 }
581 const int speed = (b->SpeedLow + b->SpeedHigh) / 2;
582 const int range = (b->RangeLow + b->RangeHigh) / 2;
583 int effectiveRange = speed * range;
584 if (b->Falling.GravityFactor != 0 && b->Falling.DestroyOnDrop)
585 {
586 // Halve effective range
587 // TODO: this assumes a certain bouncing range
588 effectiveRange /= 2;
589 }
590 return effectiveRange / 256;
591}
592bool IsLongRange(const GunDescription *g)
593{
594 return GunGetRange(g) > 130;
595}
596bool IsShortRange(const GunDescription *g)
597{
598 return GunGetRange(g) < 100;
599}

601void BulletAndWeaponInitialize(
602 BulletClasses *b, GunClasses *g, const char *bpath, const char *gpath)
603{
604 BulletInitialize(b);
605
606 FILE *bf = NULL;
607 FILE *gf = NULL;
608 json_t *broot = NULL;
609 json_t *groot = NULL;
610 enum json_error e;
611
612 // 2-pass bullet loading will free root for us
613 bool freeBRoot = true;
614 char buf[CDOGS_PATH_MAX];
615 GetDataFilePath(buf, bpath);
616 bf = fopen(buf, "r");
617 if (bf == NULL)
618 {
619 LOG(LM_MAP, LL_ERROR, "Error: cannot load bullets file %s", buf);
620 goto bail;
621 }
622 e = json_stream_parse(bf, &broot);
623 if (e != JSON_OK)
624 {
625 LOG(LM_MAP, LL_ERROR, "Error parsing bullets file %s [error %d]",
626 buf, (int)e);
627 goto bail;
628 }
629 BulletLoadJSON(b, &b->Classes, broot);
630
631 WeaponInitialize(g);
632 GetDataFilePath(buf, gpath);
633 gf = fopen(buf, "r");
634 if (gf == NULL)
635 {
636 LOG(LM_MAP, LL_ERROR, "Error: cannot load guns file %s", buf);
637 goto bail;
638 }
639 e = json_stream_parse(gf, &groot);
640 if (e != JSON_OK)
641 {
642 LOG(LM_MAP, LL_ERROR, "Error parsing guns file %s [error %d]",
643 buf, (int)e);
644 goto bail;
645 }
646 WeaponLoadJSON(g, &g->Guns, groot);
647
648 BulletLoadWeapons(b);
649 freeBRoot = false;
650
651bail:
652 if (bf)
653 {
654 fclose(bf);
655 }
656 if (gf)
657 {
658 fclose(gf);
659 }
660 if (freeBRoot)
661 {
662 json_free_value(&broot);
663 }
664 json_free_value(&groot);
665}
