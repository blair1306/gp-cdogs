38PicType StrPicType(const char *s)
39{
40 S2T(PICTYPE_NORMAL, "Normal");
41 S2T(PICTYPE_DIRECTIONAL, "Directional");
42 S2T(PICTYPE_ANIMATED, "Animated");
43 S2T(PICTYPE_ANIMATED_RANDOM, "AnimatedRandom");
44 CASSERT(false, "unknown pic type");
45 return PICTYPE_NORMAL;
46}

56void NamedSpritesInit(NamedSprites *ns, const char *name)
57{
58 CSTRDUP(ns->name, name);
59 CArrayInit(&ns->pics, sizeof(Pic));
60}
61void NamedSpritesFree(NamedSprites *ns)
62{
63 if (ns == NULL)
64 {
65 return;
66 }
67 CFREE(ns->name);
68 for (int i = 0; i < (int)ns->pics.size; i++)
69 {
70 PicFree(CArrayGet(&ns->pics, i));
71 }
72 CArrayTerminate(&ns->pics);
73}

78void CPicLoadJSON(CPic *p, json_t *node)
79{
80 char *tmp = GetString(node, "Type");
81 p->Type = StrPicType(tmp);
82 CFREE(tmp);
83 tmp = NULL;
84 switch (p->Type)
85 {
86 case PICTYPE_NORMAL:
87 LoadNormal(p, json_find_first_label(node, "Pic")->child);
88 break;
89 case PICTYPE_DIRECTIONAL:
90 LoadStr(&tmp, node, "Sprites");
91 if (tmp == NULL)
92 {
93 LOG(LM_GFX, LL_ERROR, "cannot load sprites");
94 goto bail;
95 }
96 p->u.Sprites = &PicManagerGetSprites(&gPicManager, tmp)->pics;
97 CFREE(tmp);
98 break;
99 case PICTYPE_ANIMATED: // fallthrough
100 case PICTYPE_ANIMATED_RANDOM:
101 LoadStr(&tmp, node, "Sprites");
102 if (tmp == NULL)
103 {
104 LOG(LM_GFX, LL_ERROR, "cannot load sprites");
105 goto bail;
106 }
107 p->u.Animated.Sprites =
108 &PicManagerGetSprites(&gPicManager, tmp)->pics;
109 CFREE(tmp);
110 LoadInt(&p->u.Animated.Count, node, "Count");
111 LoadInt(&p->u.Animated.TicksPerFrame, node, "TicksPerFrame");
112 p->u.Animated.TicksPerFrame = MAX(p->u.Animated.TicksPerFrame, 0);
113 break;
114 default:
115 CASSERT(false, "unknown pic type");
116 break;
117 }
118 LoadMaskTint(p, node);
119bail:
120 // TODO: return error
121 return;
122}

124void CPicLoadNormal(CPic *p, json_t *node)
125{
126 p->Type = PICTYPE_NORMAL;
127 LoadNormal(p, node);
128 p->UseMask = true;
129 p->u1.Mask = colorWhite;
130}

132static void LoadNormal(CPic *p, json_t *node)
133{
134 char *tmp = json_unescape(node->text);
135 p->u.Pic = PicManagerGetPic(&gPicManager, tmp);
136 CFREE(tmp);
137}

139static void LoadMaskTint(CPic *p, json_t *node)
140{
141 p->UseMask = true;
142 p->u1.Mask = colorWhite;
143 if (json_find_first_label(node, "Mask"))
144 {
145 char *tmp = GetString(node, "Mask");
146 p->u1.Mask = StrColor(tmp);
147 CFREE(tmp);
148 }
149 else if (json_find_first_label(node, "Tint"))
150 {
151 p->UseMask = false;
152 json_t *tint = json_find_first_label(node, "Tint")->child->child;
153 p->u1.Tint.h = atof(tint->text);
154 tint = tint->next;
155 p->u1.Tint.s = atof(tint->text);
156 tint = tint->next;
157 p->u1.Tint.v = atof(tint->text);
158 }
159}

161bool CPicIsLoaded(const CPic *p)
162{
163 switch (p->Type)
164 {
165 case PICTYPE_NORMAL:
166 return p->u.Pic != NULL;
167 case PICTYPE_DIRECTIONAL:
168 return p->u.Sprites != NULL;
169 case PICTYPE_ANIMATED:
170 case PICTYPE_ANIMATED_RANDOM:
171 return p->u.Animated.Sprites != NULL;
172 default:
173 CASSERT(false, "unknown pic type");
174 return false;
175 }
176}

188void CPicCopyPic(CPic *dest, const CPic *src)
189{
190 dest->Type = src->Type;
191 switch (src->Type)
192 {
193 case PICTYPE_NORMAL:
194 dest->u.Pic = src->u.Pic;
195 break;
196 case PICTYPE_DIRECTIONAL:
197 dest->u.Sprites = src->u.Sprites;
198 break;
199 case PICTYPE_ANIMATED:
200 case PICTYPE_ANIMATED_RANDOM:
201 dest->u.Animated.Sprites = src->u.Sprites;
202 dest->u.Animated.TicksPerFrame = src->u.Animated.TicksPerFrame;
203 break;
204 default:
205 CASSERT(false, "unknown pic type");
206 break;
207 }
208 dest->UseMask = src->UseMask;
209 dest->u1 = src->u1;
210}

212void CPicUpdate(CPic *p, const int ticks)
213{
214 switch (p->Type)
215 {
216 case PICTYPE_ANIMATED:
217 {
218 p->u.Animated.Count += ticks;
219 if (p->u.Animated.TicksPerFrame > 0)
220 {
221 while (p->u.Animated.Count >= p->u.Animated.TicksPerFrame)
222 {
223 p->u.Animated.Frame++;
224 p->u.Animated.Count -= p->u.Animated.TicksPerFrame;
225 }
226 while (p->u.Animated.Frame >= (int)p->u.Animated.Sprites->size)
227 {
228 p->u.Animated.Frame -= (int)p->u.Animated.Sprites->size;
229 }
230 }
231 }
232 break;
233 case PICTYPE_ANIMATED_RANDOM:
234 if (p->u.Animated.Count == 0)
235 {
236 // Initial frame
237 p->u.Animated.Frame = rand() % (int)p->u.Animated.Sprites->size;
238 }
239 p->u.Animated.Count += ticks;
240 if (p->u.Animated.TicksPerFrame > 0 &&
241 p->u.Animated.Count >= p->u.Animated.TicksPerFrame)
242 {
243 p->u.Animated.Frame = rand() % (int)p->u.Animated.Sprites->size;
244 p->u.Animated.Count = 0;
245 }
246 break;
247 default:
248 // Do nothing
249 break;
250 }

252const Pic *CPicGetPic(const CPic *p, const int idx)
253{
254 switch (p->Type)
255 {
256 case PICTYPE_NORMAL:
257 return p->u.Pic;
258 case PICTYPE_DIRECTIONAL:
259 return CArrayGet(p->u.Sprites, idx);
260 case PICTYPE_ANIMATED:
261 case PICTYPE_ANIMATED_RANDOM:
262 if (p->u.Animated.Frame < 0 ||
263 p->u.Animated.Frame >= (int)p->u.Animated.Sprites->size)
264 {
265 return NULL;
266 }
267 return CArrayGet(p->u.Animated.Sprites, p->u.Animated.Frame);
268 default:
269 CASSERT(false, "unknown pic type");
270 return NULL;
271 }

273void CPicDraw(
274 GraphicsDevice *g, const CPic *p,
275 const Vec2i pos, const CPicDrawContext *context)
276{
277 const Pic *pic = CPicGetPic(p, context->Dir);
278 if (pic == NULL)
279 {
280 return;
281 }
282 const Vec2i picPos = Vec2iAdd(pos, context->Offset);
283 if (p->UseMask)
284 {
285 BlitMasked(g, pic, picPos, p->u1.Mask, true);
286 }
287 else
288 {
289 BlitBackground(g, pic, picPos, &p->u1.Tint, true);
290 }
