45static void LoadPlayerTemplate(
46 CArray *templates, json_t *node, const int version)
47{
48 PlayerTemplate t;
49 strcpy(t.name, json_find_first_label(node, "Name")->child->text);
50 t.CharClassName = GetString(node, "Face");
51 if (version == 1)
52 {
53 // Version 1 used integer palettes
54 int skin, arms, body, legs, hair;
55 LoadInt(&skin, node, "Skin");
56 LoadInt(&arms, node, "Arms");
57 LoadInt(&body, node, "Body");
58 LoadInt(&legs, node, "Legs");
59 LoadInt(&hair, node, "Hair");
60 ConvertCharacterColors(skin, arms, body, legs, hair, &t.Colors);
61 }
62 else
63 {
64 LoadColor(&t.Colors.Skin, node, "Skin");
65 LoadColor(&t.Colors.Arms, node, "Arms");
66 LoadColor(&t.Colors.Body, node, "Body");
67 LoadColor(&t.Colors.Legs, node, "Legs");
68 LoadColor(&t.Colors.Hair, node, "Hair");
69 }
70 CArrayPushBack(templates, &t);
71 LOG(LM_MAIN, LL_DEBUG, "loaded player template %s (%s)",
72 t.name, t.CharClassName);
73}
74void PlayerTemplatesLoad(
75 CArray *templates, const CharacterClasses *classes, const char *filename)
76{
77 // Note: not used, but included in function to express dependency
78 CASSERT(classes->Classes.size > 0,
79 "cannot load player templates without character classes");
80 json_t *root = NULL;
81 int version = 1;
82
83 // initialise templates
84 CArrayInit(templates, sizeof(PlayerTemplate));
85 FILE *f = fopen(GetConfigFilePath(filename), "r");
86 if (!f)
87 {
88 printf("Error loading player templates '%s'\n", filename);
89 goto bail;
90 }
91
92 if (json_stream_parse(f, &root) != JSON_OK)
93 {
94 printf("Error parsing player templates '%s'\n", filename);
95 goto bail;
96 }
97
98 LoadInt(&version, root, "Version");
99
100 if (json_find_first_label(root, "PlayerTemplates") == NULL)
101 {
102 printf("Error: unknown player templates format\n");
103 goto bail;
104 }
105 json_t *child = json_find_first_label(root, "PlayerTemplates")->child->child;
106 while (child != NULL)
107 {
108 LoadPlayerTemplate(templates, child, version);
109 child = child->next;
110 }
111
112bail:
113 json_free_value(&root);
114 if (f != NULL)
115 {
116 fclose(f);
117 }
118}

120void PlayerTemplatesTerminate(CArray *t)
121{
122 CA_FOREACH(PlayerTemplate, pt, *t)
123 CFREE(pt->CharClassName);
124 CA_FOREACH_END()
125 CArrayTerminate(t);
126}

128static void SavePlayerTemplate(PlayerTemplate *t, json_t *templates)
129{
130 json_t *template = json_new_object();
131 AddStringPair(template, "Name", t->name);
132 AddStringPair(template, "Face", t->CharClassName);
133 AddColorPair(template, "Body", t->Colors.Body);
134 AddColorPair(template, "Arms", t->Colors.Arms);
135 AddColorPair(template, "Legs", t->Colors.Legs);
136 AddColorPair(template, "Skin", t->Colors.Skin);
137 AddColorPair(template, "Hair", t->Colors.Hair);
138 json_insert_child(templates, template);
139}
140void SavePlayerTemplates(const CArray *templates, const char *filename)
141{
142 FILE *f = fopen(GetConfigFilePath(filename), "w");
143 char *text = NULL;
144 char *formatText = NULL;
145 json_t *root = json_new_object();
146
147 if (f == NULL)
148 {
149 printf("Error saving player templates '%s'\n", filename);
150 goto bail;
151 }
152
153 debug(D_NORMAL, "begin\n");
154
155 json_insert_pair_into_object(
156 root, "Version", json_new_number(TOSTRING(VERSION)));
157 json_t *templatesNode = json_new_array();
158 for (int i = 0; i < (int)templates->size; i++)
159 {
160 PlayerTemplate *t = CArrayGet(templates, i);
161 SavePlayerTemplate(t, templatesNode);
162 }
163 json_insert_pair_into_object(root, "PlayerTemplates", templatesNode);
164
165 json_tree_to_string(root, &text);
166 formatText = json_format_string(text);
167 fputs(formatText, f);
168
169bail:
170 // clean up
171 free(formatText);
172 free(text);
173 json_free_value(&root);
174 if (f != NULL)
175 {
176 fclose(f);
177#ifdef __EMSCRIPTEN__
178 EM_ASM(
179 //persist changes
180 FS.syncfs(false,function (err) {
181 assert(!err);
182 });
183 );
184#endif
185 }
