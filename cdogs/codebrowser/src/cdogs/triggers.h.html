63typedef enum
64{
65 ACTION_NULL,
66 ACTION_SETTRIGGER,
67 ACTION_CLEARTRIGGER,
68 ACTION_EVENT,
69 ACTION_ACTIVATEWATCH,
70 ACTION_DEACTIVATEWATCH,
71 ACTION_SOUND
72} ActionType;

80 {
81 Vec2i pos;

85 {

93{
94 int id;
95 int flags;
96 int isActive;
97 int cannotActivateLock;

102typedef enum
103{
104 CONDITION_TILECLEAR = 1

106typedef struct
107{
108 ConditionType Type;
109 // How many ticks has this condition been fulfilled
110 // Reset to 0 when condition failed
111 int Counter;
112 int CounterMax;
113 Vec2i Pos;
114} Condition;

117typedef struct

119 int index;
120 CArray conditions; // of Condition
121 CArray actions; // of Action
122 bool active;
123} TWatch;
