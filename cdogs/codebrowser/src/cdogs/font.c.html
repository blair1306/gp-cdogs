53void FontLoad(Font *f, const char *imgPath, const bool isProportional)
54{
55 char buf[CDOGS_PATH_MAX];
56 GetDataFilePath(buf, imgPath);
57 SDL_RWops *rwops = SDL_RWFromFile(buf, "rb");
58 CASSERT(IMG_isPNG(rwops), "Error: font file is not PNG");
59 SDL_Surface *image = IMG_Load_RW(rwops, 0);
60 if (!image)
61 {
62 fprintf(stderr, "Cannot load font image: %s\n", IMG_GetError());
63 goto bail;
64 }
65 if (image->format->BytesPerPixel != 4)
66 {
67 perror("Cannot load non-32-bit image");
68 fprintf(stderr, "Only 32-bit depth images supported\n");
69 goto bail;
70 }
71
72 CArrayInit(&f->Chars, sizeof(Pic));
73
74 // Check that the image is big enough for the dimensions
75 const Vec2i step = Vec2iNew(
76 f->Size.x + f->Padding.Left + f->Padding.Right,
77 f->Size.y + f->Padding.Top + f->Padding.Bottom);
78 if (step.x * f->Stride > image->w || step.y > image->h)
79 {
80 fprintf(stderr,
81 "Error: font image not big enough for font data "
82 "Image %dx%d Size %dx%d Stride %d Padding %d,%d,%d,%d\n",
83 image->w, image->h,
84 f->Size.x, f->Size.y, f->Stride,
85 f->Padding.Left, f->Padding.Top,
86 f->Padding.Right, f->Padding.Bottom);
87 goto bail;
88 }
89
90 // Load letters from image file
91 SDL_LockSurface(image);
92 int chars = 0;
93 for (Vec2i pos = Vec2iZero();
94 pos.y + step.y <= image->h && chars < LAST_CHAR - FIRST_CHAR + 1;
95 pos.y += step.y)
96 {
97 int x = 0;
98 for (pos.x = 0;
99 x < f->Stride && chars < LAST_CHAR - FIRST_CHAR + 1;
100 pos.x += step.x, x++, chars++)
101 {
102 Pic p;
103 p.size = f->Size;
104 p.offset = Vec2iZero();
105 PicLoad(
106 &p, f->Size,
107 Vec2iAdd(pos, Vec2iNew(f->Padding.Left, f->Padding.Top)),
108 image);
109 if (isProportional)
110 {
111 PicTrim(&p, true, false);
112 }
113 CArrayPushBack(&f->Chars, &p);
114 }
115 }
116 SDL_UnlockSurface(image);
117
118bail:
119 SDL_FreeSurface(image);
120 rwops->close(rwops);
121}
122void FontTerminate(Font *f)
123{
124 CA_FOREACH(Pic, p, f->Chars)
125 PicFree(p);
126 CA_FOREACH_END()
127 CArrayTerminate(&f->Chars);
128}

130int FontW(const char c)
131{
132 const Pic *p = CArrayGet(&gFont.Chars, (int)c - FIRST_CHAR);
133 return p->size.x + gFont.Gap.x;
134}
135int FontH(void)
136{
137 return gFont.Size.y + gFont.Gap.y;
138}
139int FontStrW(const char *s)
140{
141 return FontSubstrW(s, strlen(s));
142}
143int FontSubstrW(const char *s, int len)
144{
145 if (len > (int)strlen(s))
146 {
147 len = (int)strlen(s);
148 }
149 // Find the width of the longest line, if this is a multi-line
150 int maxWidth = 0;
151 int w = 0;
152 for (int i = 0; i < len; i++, s++)
153 {
154 if (*s == '\n')
155 {
156 maxWidth = MAX(maxWidth, w);
157 w = 0;
158 }
159 else
160 {
161 w += FontW(*s);
162 }
163 }
164 maxWidth = MAX(maxWidth, w);
165 return maxWidth;
166}
167int FontStrH(const char *s)
168{
169 return FontStrNumLines(s) * FontH();
170}
171Vec2i FontStrSize(const char *s)
172{
173 Vec2i size = Vec2iZero();
174 while (*s)
175 {
176 char *lineEnd = strchr(s, '\n');
177 size.y += FontH();
178 if (lineEnd)
179 {
180 size.x = MAX(size.x, FontSubstrW(s, lineEnd - s));
181 s = lineEnd + 1;
182 }
183 else
184 {
185 size.x = MAX(size.x, FontStrW(s));
186 s += strlen(s);
187 }
188 }
189 return size;
190}
191int FontStrNumLines(const char *s)
192{
193 int lines;
194 for (lines = 0; s != NULL; lines++)
195 {
196 s = strchr(s, '\n');
197 if (s)
198 {
199 s++;
200 }
201 }
202 return lines;
203}

205Vec2i FontCh(const char c, const Vec2i pos)
206{
207 return FontChMask(c, pos, colorWhite);
208}
209static Vec2i FontChColor(
210 const char c, const Vec2i pos, const color_t color, const bool blend);
211Vec2i FontChMask(const char c, const Vec2i pos, const color_t mask)
212{
213 return FontChColor(c, pos, mask, false);
214}
215static Vec2i FontChColor(
216 const char c, const Vec2i pos, const color_t color, const bool blend)
217{
218 int idx = (int)c - FIRST_CHAR;
219 if (idx < 0)
220 {
221 idx += 256;
222 }
223 if (idx < FIRST_CHAR || idx > LAST_CHAR)
224 {
225 fprintf(stderr, "invalid char %d\n", idx);
226 idx = FIRST_CHAR;
227 }
228 const Pic *pic = CArrayGet(&gFont.Chars, idx);
229 if (blend)
230 {
231 BlitBlend(&gGraphicsDevice, pic, pos, color);
232 }
233 else
234 {
235 BlitMasked(&gGraphicsDevice, pic, pos, color, true);
236 }
237 // Add gap between characters
238 return Vec2iNew(pos.x + pic->size.x + gFont.Gap.x, pos.y);
239}
240Vec2i FontStr(const char *s, Vec2i pos)
241{
242 return FontStrMask(s, pos, colorWhite);
243}
244static Vec2i FontStrColor(
245 const char *s, Vec2i pos, const color_t c, const bool blend);
246Vec2i FontStrMask(const char *s, Vec2i pos, const color_t mask)
247{
248 return FontStrColor(s, pos, mask, false);
249}
250static Vec2i FontStrColor(
251 const char *s, Vec2i pos, const color_t c, const bool blend)
252{
253 if (s == NULL)
254 {
255 return pos;
256 }
257 int left = pos.x;
258 while (*s)
259 {
260 if (*s == '\n')
261 {
262 pos.x = left;
263 pos.y += FontH();
264 }
265 else
266 {
267 pos = FontChColor(*s, pos, c, blend);
268 }
269 s++;
270 }
271 return pos;
272}
273Vec2i FontStrMaskWrap(const char *s, Vec2i pos, color_t mask, const int width)
274{
275 char buf[1024];
276 CASSERT(strlen(s) < 1024, "string too long to wrap");
277 FontSplitLines(s, buf, width);
278 return FontStrMask(buf, pos, mask);
279}
280static Vec2i GetStrPos(const char *s, Vec2i pos, const FontOpts opts);
281void FontStrOpt(const char *s, Vec2i pos, const FontOpts opts)
282{
283 if (s == NULL)
284 {
285 return;
286 }
287 pos = GetStrPos(s, pos, opts);
288 FontStrMask(s, pos, opts.Mask);
289}
290static int GetAlign(
291 const FontAlign align,
292 const int pos, const int pad, const int area, const int size);
293static Vec2i GetStrPos(const char *s, Vec2i pos, const FontOpts opts)
294{
295 const Vec2i textSize = FontStrSize(s);
296 return Vec2iNew(
297 GetAlign(opts.HAlign, pos.x, opts.Pad.x, opts.Area.x, textSize.x),
298 GetAlign(opts.VAlign, pos.y, opts.Pad.y, opts.Area.y, textSize.y));
299}
300static int GetAlign(
301 const FontAlign align,
302 const int pos, const int pad, const int area, const int size)
303{
304 switch (align)
305 {
306 case ALIGN_START:
307 return pos + pad;
308 break;
309 case ALIGN_CENTER:
310 return pos + (area - size) / 2;
311 break;
312 case ALIGN_END:
313 return pos + area - size - pad;
314 break;
315 default:
316 CASSERT(false, "unknown align");
317 return pos;
318 }
319}

330void FontSplitLines(const char *text, char *buf, const int width)
331{
332 if (width == 0)
333 {
334 strcpy(buf, text);
335 return;
336 }
337 int ix, x;
338 const char *ws, *word, *s;
339
340 ix = x = CenterX(width);
341 s = ws = word = text;
342
343 while (*s)
344 {
345 // Skip spaces
346 ws = s;
347 while (*s == ' ' || *s == '\n')
348 {
349 *buf++ = *s;
350 if (*s == '\n')
351 {
352 // We've already skipped a word, so reset the word count
353 ws = s + 1;
354 x = ix;
355 }
356 s++;
357 }
358
359 // Find word
360 word = s;
361 while (*s != 0 && *s != ' ' && *s != '\n')
362 {
363 s++;
364 }
365 // Calculate width of word
366 int w;
367 const char *p;
368 for (w = 0, p = ws; p < s; p++)
369 {
370 w += FontW(*p);
371 }
372
373 // Create new line if text too wide
374 if (x + w > width + ix && w < width)
375 {
376 x = ix;
377 ws = word;
378 *buf++ = '\n';
379 }
380
381 for (p = ws; p < word; p++)
382 {
383 x += FontW(*p);
384 }
385
386 for (p = word; p < s; p++)
387 {
388 *buf++ = *p;
389 x += FontW(*p);
390 }
391 }
392 *buf = '\0';
393}

395Vec2i Vec2iAligned(
396 const Vec2i v, const Vec2i size,
397 const FontAlign hAlign, const FontAlign vAlign, const Vec2i area)
398{
399 Vec2i vAligned = v;
400 switch (hAlign)
401 {
402 case ALIGN_CENTER:
403 vAligned.x += (area.x - size.x) / 2;
404 break;
405 case ALIGN_END:
406 vAligned.x = -v.x + area.x - size.x;
407 break;
408 default:
409 break;
410 }
411 switch (vAlign)
412 {
413 case ALIGN_CENTER:
414 vAligned.y += (area.y - size.y) / 2;
415 break;
416 case ALIGN_END:
417 vAligned.y = -v.y + area.y - size.y;
418 break;
419 default:
420 break;
421 }
422 return vAligned;
423}
