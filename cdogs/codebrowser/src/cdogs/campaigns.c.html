42void CampaignInit(CampaignOptions *campaign)
43{
44 memset(campaign, 0, sizeof *campaign);
45 CampaignSettingInit(&campaign->Setting);
46}
47void CampaignTerminate(CampaignOptions *campaign)
48{
49 CampaignUnload(&gCampaign);
50 CampaignSettingTerminate(&campaign->Setting);
51}
52void CampaignSettingInit(CampaignSetting *setting)
53{
54 memset(setting, 0, sizeof *setting);
55 CArrayInit(&setting->Missions, sizeof(Mission));
56 CharacterStoreInit(&setting->characters);
57
58 PickupClassesLoadKeys(&gPickupClasses.KeyClasses);
59}
60void CampaignSettingTerminate(CampaignSetting *setting)
61{
62 CFREE(setting->Title);
63 CFREE(setting->Author);
64 CFREE(setting->Description);
65 CA_FOREACH(Mission, m, setting->Missions)
66 MissionTerminate(m);
67 CA_FOREACH_END()
68 CArrayTerminate(&setting->Missions);
69 CharacterStoreTerminate(&setting->characters);
70 memset(setting, 0, sizeof *setting);
71
72 // Unload previous custom data
73 SoundClear(gSoundDevice.customSounds);
74 PicManagerClearCustom(&gPicManager);
75 ParticleClassesClear(&gParticleClasses.CustomClasses);
76 AmmoClassesClear(&gAmmo.CustomAmmo);
77 CharacterClassesClear(&gCharacterClasses.CustomClasses);
78 BulletClassesClear(&gBulletClasses.CustomClasses);
79 WeaponClassesClear(&gGunDescriptions.CustomGuns);
80 PickupClassesClear(&gPickupClasses.CustomClasses);
81 PickupClassesClear(&gPickupClasses.KeyClasses);
82 MapObjectsClear(&gMapObjects.CustomClasses);
83}

92void LoadAllCampaigns(custom_campaigns_t *campaigns)
93{
94 char buf[CDOGS_PATH_MAX];
95
96 CampaignListInit(&campaigns->campaignList);
97 CampaignListInit(&campaigns->dogfightList);
98
99 GetDataFilePath(buf, CDOGS_CAMPAIGN_DIR);
100 LOG(LM_MAIN, LL_INFO, "Load campaigns from dir %s...", buf);
101 LoadCampaignsFromFolder(
102 &campaigns->campaignList,
103 "",
104 buf,
105 GAME_MODE_NORMAL);
106
107 GetDataFilePath(buf, CDOGS_DOGFIGHT_DIR);
108 LOG(LM_MAIN, LL_INFO, "Load dogfights from dir %s...", buf);
109 LoadCampaignsFromFolder(
110 &campaigns->dogfightList,
111 "",
112 buf,
113 GAME_MODE_DOGFIGHT);
114
115 LOG(LM_MAIN, LL_INFO, "Load quick play...");
116 LoadQuickPlayEntry(&campaigns->quickPlayEntry);
117}

128static void CampaignListInit(campaign_list_t *list)
129{
130 list->Name = NULL;
131 CArrayInit(&list->subFolders, sizeof(campaign_list_t));
132 CArrayInit(&list->list, sizeof(CampaignEntry));
133}
134static void CampaignListTerminate(campaign_list_t *list)
135{
136 CFREE(list->Name);
137 CA_FOREACH(campaign_list_t, sublist, list->subFolders)
138 CampaignListTerminate(sublist);
139 CA_FOREACH_END()
140 CArrayTerminate(&list->subFolders);
141 CA_FOREACH(CampaignEntry, e, list->list)
142 CampaignEntryTerminate(e);
143 CA_FOREACH_END()
144 CArrayTerminate(&list->list);
145}

155static void LoadCampaignsFromFolder(
156 campaign_list_t *list, const char *name, const char *path,
157 const GameMode mode)
158{
159 tinydir_dir dir;
160 int i;
161
162 CSTRDUP(list->Name, name);
163 if (tinydir_open_sorted(&dir, path) == -1)
164 {
165 printf("Cannot load campaigns from path %s\n", path);
166 return;
167 }
168
169 for (i = 0; i < (int)dir.n_files; i++)
170 {
171 tinydir_file file;
172 tinydir_readfile_n(&dir, &file, i);
173
174 // Ignore campaigns that start with a ~
175 // These are autosaved
176
177 const bool isArchive =
178 strcmp(file.extension, "cdogscpn") == 0 ||
179 strcmp(file.extension, "CDOGSCPN") == 0;
180 if (file.is_dir && !isArchive &&
181 strcmp(file.name, ".") != 0 && strcmp(file.name, "..") != 0)
182 {
183 campaign_list_t subFolder;
184 CampaignListInit(&subFolder);
185 LoadCampaignsFromFolder(&subFolder, file.name, file.path, mode);
186 CArrayPushBack(&list->subFolders, &subFolder);
187 }
188 else if ((file.is_reg || isArchive) && file.name[0] != '~')
189 {
190 CampaignEntry entry;
191 if (CampaignEntryTryLoad(&entry, file.path, mode))
192 {
193 CArrayPushBack(&list->list, &entry);
194 }
195 }
196 }
197
198 tinydir_close(&dir);
199}

201Mission *CampaignGetCurrentMission(CampaignOptions *campaign)
202{
203 if (campaign->MissionIndex >= (int)campaign->Setting.Missions.size)
204 {
205 return NULL;
206 }
207 return CArrayGet(&campaign->Setting.Missions, campaign->MissionIndex);
208}

210void CampaignSeedRandom(const CampaignOptions *campaign)
211{
212 const int seed =
213 10 * campaign->MissionIndex + ConfigGetInt(&gConfig, "Game.RandomSeed");
214 LOG(LM_MAIN, LL_INFO, "Seeding with %d", seed);
215 srand((unsigned int)seed);
216}

218void CampaignAndMissionSetup(
219 CampaignOptions *campaign, struct MissionOptions *mo)
220{
221 Mission *m = CampaignGetCurrentMission(campaign);
222 if (m == NULL)
223 {
224 return;
225 }
226 CampaignSeedRandom(campaign);
227 SetupMission(m, mo, campaign->MissionIndex);
228}
