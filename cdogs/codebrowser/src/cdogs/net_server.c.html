65static ENetHost *HostOpen(void);
66static bool ListenSocketTryOpen(ENetSocket *listen);
67void NetServerOpen(NetServer *n)
68{
69 if (n->server)
70 {
71 return;
72 }
73
74 n->server = HostOpen();
75 if (n->server == NULL)
76 {
77 return;
78 }
79
80 // Start listen socket, to respond to UDP scans
81 if (!ListenSocketTryOpen(&n->listen))
82 {
83 return;
84 }
85
86 // Save limited length hostname for LAN scanners
87 // This is because getting hostname locally is much faster
88 if (enet_address_get_host(
89 &n->server->address, n->hostname, sizeof n->hostname) != 0)
90 {
91 LOG(LM_NET, LL_WARN, "Failed to get hostname");
92 n->hostname[0] = '\0';
93 }
94}
95static ENetHost *HostOpen(void)
96{
97 ENetAddress address;
98 address.host = ENET_HOST_ANY;
99 address.port = ENET_PORT_ANY;
100 ENetHost *host = enet_host_create(&address, NET_SERVER_MAX_CLIENTS, 2, 0, 0);
101 if (host == NULL)
102 {
103 LOG(LM_NET, LL_ERROR, "cannot create server host");
104 return NULL;
105 }
106 else
107 {
108 char buf[256];
109 enet_address_get_host_ip(&host->address, buf, sizeof buf);
110 LOG(LM_NET, LL_INFO, "starting server on %s:%u",
111 buf, host->address.port);
112 }
113 return host;
114}
115static bool ListenSocketTryOpen(ENetSocket *listen)
116{
117 *listen = enet_socket_create(ENET_SOCKET_TYPE_DATAGRAM);
118 if (*listen == ENET_SOCKET_NULL)
119 {
120 LOG(LM_NET, LL_ERROR, "Failed to create socket");
121 return false;
122 }
123 if (enet_socket_set_option(*listen, ENET_SOCKOPT_REUSEADDR, 1) != 0)
124 {
125 LOG(LM_NET, LL_ERROR, "Failed to enable reuse address");
126 return false;
127 }
128 ENetAddress addr;
129 addr.host = ENET_HOST_ANY;
130 addr.port = NET_LISTEN_PORT;
131 if (enet_socket_bind(*listen, &addr) != 0)
132 {
133 LOG(LM_NET, LL_ERROR, "failed to bind listen socket");
134 return false;
135 }
136 if (enet_socket_get_address(*listen, &addr) != 0)
137 {
138 LOG(LM_NET, LL_ERROR, "cannot get listen socket address");
139 return false;
140 }
141 LOG(LM_NET, LL_INFO, "listening for scans on port %u", addr.port);
142 return true;
143}

145void NetServerClose(NetServer *n)
146{
147 if (n->server)
148 {
149 for (int i = 0; i < (int)n->server->peerCount; i++)
150 {
151 ENetPeer *peer = n->server->peers + i;
152 enet_peer_disconnect_now(peer, 0);
153 }
154 enet_host_destroy(n->server);
155 }
156 n->server = NULL;
157}

159static void PollListener(NetServer *n);
160static void OnReceive(NetServer *n, ENetEvent event);
161static void OnDisconnect(const ENetEvent event);
162void NetServerPoll(NetServer *n)
163{
164 if (!n->server)
165 {
166 return;
167 }
168
169 // Check our listening socket for scanning clients
170 PollListener(n);
171
172 n->PrevCmd = n->Cmd;
173 n->Cmd = 0;
174 int check;
175 do
176 {
177 ENetEvent event;
178 check = enet_host_service(n->server, &event, 0);
179 if (check < 0)
180 {
181 fprintf(stderr, "Host check event failure\n");
182 enet_host_destroy(n->server);
183 n->server = NULL;
184 return;
185 }
186 else if (check > 0)
187 {
188 switch (event.type)
189 {
190 case ENET_EVENT_TYPE_CONNECT:
191 {
192 char buf[256];
193 enet_address_get_host_ip(
194 &event.peer->address, buf, sizeof buf);
195 LOG(LM_NET, LL_INFO, "client connection from %s:%u",
196 buf, event.peer->address.port);
197 }
198 break;
199 case ENET_EVENT_TYPE_RECEIVE:
200 OnReceive(n, event);
201 break;
202 case ENET_EVENT_TYPE_DISCONNECT:
203 OnDisconnect(event);
204 break;
205 default:
206 CASSERT(false, "Unknown event");
207 break;
208 }
209 }
210 } while (check > 0);
211
212 NetServerFlush(n);
213}
214static void PollListener(NetServer *n)
215{
216 // Check for data to recv
217 ENetSocketSet set;
218 ENET_SOCKETSET_EMPTY(set);
219 ENET_SOCKETSET_ADD(set, n->listen);
220 if (enet_socketset_select(n->listen, &set, NULL, 0) <= 0)
221 {
222 return;
223 }
224
225 ENetAddress addr;
226 char buf;
227 ENetBuffer recvbuf;
228 recvbuf.data = &buf;
229 recvbuf.dataLength = 1;
230 const int recvlen = enet_socket_receive(n->listen, &addr, &recvbuf, 1);
231 if (recvlen <= 0)
232 {
233 return;
234 }
235 char addrbuf[256];
236 enet_address_get_host_ip(&addr, addrbuf, sizeof addrbuf);
237 LOG(LM_NET, LL_DEBUG, "listener received from %s:%u", addrbuf, addr.port);
238 // Reply to scanner client with our server host/address
239 NServerInfo sinfo;
240 sinfo.ProtocolVersion = NET_PROTOCOL_VERSION;
241 sinfo.ENetPort = n->server->address.port;
242 sinfo.Hostname[0] = '\0';
243 strncat(sinfo.Hostname, n->hostname, sizeof sinfo.Hostname - 1);
244 sinfo.GameMode = gCampaign.Entry.Mode;
245 sinfo.CampaignName[0] = '\0';
246 strncat(sinfo.CampaignName, gCampaign.Setting.Title,
247 sizeof sinfo.CampaignName - 1);
248 sinfo.MissionNumber = gCampaign.MissionIndex + 1;
249 sinfo.NumPlayers = GetNumPlayers(PLAYER_ANY, false, false);
250 sinfo.MaxPlayers =
251 NET_SERVER_MAX_CLIENTS * MAX_LOCAL_PLAYERS +
252 GetNumPlayers(PLAYER_ANY, false, true);
253 // Encode our packet
254 uint8_t encbuf[1024];
255 pb_ostream_t stream = pb_ostream_from_buffer(encbuf, sizeof encbuf);
256 const bool status = pb_encode(&stream, NServerInfo_fields, &sinfo);
257 CASSERT(status, "Failed to encode pb");
258 recvbuf.data = encbuf;
259 recvbuf.dataLength = stream.bytes_written;
260 if (enet_socket_send(n->listen, &addr, &recvbuf, 1) != (int)recvbuf.dataLength)
261 {
262 LOG(LM_NET, LL_ERROR, "Failed to reply to scanner");
263 }
264}
265static void OnConnect(NetServer *n, ENetEvent event);
266static void OnReceive(NetServer *n, ENetEvent event)
267{
268 const GameEventType msg = (GameEventType)*(uint32_t *)event.packet->data;
269 int peerId = -1;
270 if (event.peer->data != NULL)
271 {
272 // We may not have assigned peer ID
273 peerId = ((NetPeerData *)event.peer->data)->Id;
274 LOG(LM_NET, LL_TRACE, "recv message from peerId(%d) msg(%d)",
275 peerId, (int)msg);
276 }
277 const GameEventEntry gee = GameEventGetEntry(msg);
278 if (gee.Enqueue)
279 {
280 // Game event message; decode and add to event queue
281 LOG(LM_NET, LL_TRACE, "recv gameEvent(%d)", (int)gee.Type);
282 GameEvent e = GameEventNew(gee.Type);
283 NetDecode(event.packet, &e.u, gee.Fields);
284 GameEventsEnqueue(&gGameEvents, e);
285 }
286 else
287 {
288 switch (gee.Type)
289 {
290 case GAME_EVENT_CLIENT_CONNECT:
291 OnConnect(n, event);
292 break;
293 case GAME_EVENT_CLIENT_READY:
294 CASSERT(peerId >= 0, "peer id unset");
295 // Flush game events to make sure we add the players
296 HandleGameEvents(&gGameEvents, NULL, NULL, NULL);
297 // Reset player data
298 for (int i = 0; i < MAX_LOCAL_PLAYERS; i++)
299 {
300 const int cid = (peerId + 1) * MAX_LOCAL_PLAYERS + i;
301 const PlayerData *pData = PlayerDataGetByUID(cid);
302 if (pData == NULL) continue;
303 GameEvent e = GameEventNew(GAME_EVENT_PLAYER_DATA);
304 e.u.PlayerData = PlayerDataMissionReset(pData);
305 GameEventsEnqueue(&gGameEvents, e);
306 }
307 // Flush game events to make sure we reset player data
308 HandleGameEvents(&gGameEvents, NULL, NULL, NULL);
309
310 // Send game start messages if we've started already
311 if (gMission.HasStarted)
312 {
313 NetServerSendGameStartMessages(n, peerId);
314
315 // Find a current player to spawn next to
316 Vec2i defaultSpawnPosition = Vec2iZero();
317 const TActor *closestActor = AIGetClosestPlayer(Vec2iZero());
318 if (closestActor != NULL) defaultSpawnPosition = closestActor->Pos;
319
320 // Add the client's actors
321 for (int i = 0; i < MAX_LOCAL_PLAYERS; i++)
322 {
323 const int cid = (peerId + 1) * MAX_LOCAL_PLAYERS + i;
324 const PlayerData *pData = PlayerDataGetByUID(cid);
325 if (pData != NULL)
326 {
327 defaultSpawnPosition = PlacePlayer(
328 &gMap, pData, defaultSpawnPosition, true);
329 }
330 }
331 }
332 if (gMission.HasBegun)
333 {
334 NGameBegin gb = NGameBegin_init_default;
335 gb.MissionTime = gMission.time;
336 NetServerSendMsg(n, peerId, GAME_EVENT_GAME_BEGIN, &gb);
337 }
338
339 NetServerFlush(n);
340 break;
341 default:
342 CASSERT(false, "unexpected message type");
343 break;
344 }
345 }
346 enet_packet_destroy(event.packet);
347}
348static void OnConnect(NetServer *n, ENetEvent event)
349{
350 char buf[256];
351 enet_address_get_host_ip(&event.peer->address, buf, sizeof buf);
352 LOG(LM_NET, LL_INFO, "new client connected from %s:%u",
353 buf, event.peer->address.port);
354 /* Store any relevant client information here. */
355 CMALLOC(event.peer->data, sizeof(NetPeerData));
356 const int peerId = n->peerId;
357 ((NetPeerData *)event.peer->data)->Id = peerId;
358 n->peerId++;
359
360 // Send the client ID
361 LOG(LM_NET, LL_DEBUG, "NetServer: sending client ID %d", peerId);
362 NClientId cid;
363 cid.Id = peerId;
364 cid.FirstPlayerUID = (peerId + 1) * MAX_LOCAL_PLAYERS;
365 NetServerSendMsg(n, peerId, GAME_EVENT_CLIENT_ID, &cid);
366
367 // Send the current campaign details over
368 LOG(LM_NET, LL_DEBUG, "NetServer: sending campaign entry");
369 NCampaignDef def = NMakeCampaignDef(&gCampaign);
370 NetServerSendMsg(n, peerId, GAME_EVENT_CAMPAIGN_DEF, &def);
371
372 SoundPlay(&gSoundDevice, StrSound("hahaha"));
373 LOG(LM_NET, LL_DEBUG, "NetServer: client connection complete");
374
375 NetServerFlush(n);
376}
377static void OnDisconnect(const ENetEvent event)
378{
379 int peerId = -1;
380 if (event.peer->data != NULL)
381 {
382 peerId = ((NetPeerData *)event.peer->data)->Id;
383 CFREE(event.peer->data);
384 event.peer->data = NULL;
385 }
386 CASSERT(peerId >= 0, "Cannot find disconnected peer id");
387 char buf[256];
388 enet_address_get_host_ip(&event.peer->address, buf, sizeof buf);
389 LOG(LM_NET, LL_INFO, "peerId(%d) disconnected %s:%u",
390 peerId, buf, event.peer->address.port);
391 // Remove client's players
392 for (int i = 0; i < MAX_LOCAL_PLAYERS; i++)
393 {
394 GameEvent e = GameEventNew(GAME_EVENT_PLAYER_REMOVE);
395 e.u.PlayerRemove.UID = (peerId + 1) * MAX_LOCAL_PLAYERS + i;
396 GameEventsEnqueue(&gGameEvents, e);
397 }
398}

408void NetServerSendGameStartMessages(NetServer *n, const int peerId)
409{
410 // Send details of all current players
411 CA_FOREACH(const PlayerData, pOther, gPlayerDatas)
412 NPlayerData pd = NMakePlayerData(pOther);
413 NetServerSendMsg(n, peerId, GAME_EVENT_PLAYER_DATA, &pd);
414 LOG(LM_NET, LL_DEBUG, "send player data uid(%d) maxHealth(%d)",
415 (int)pd.UID, (int)pd.MaxHealth);
416 CA_FOREACH_END()
417
418 // Send all game-specific config values
419 SendConfig(&gConfig, "Game.FriendlyFire", n, peerId);
420 SendConfig(&gConfig, "Game.FPS", n, peerId);
421 SendConfig(&gConfig, "Game.Ammo", n, peerId);
422 SendConfig(&gConfig, "Game.Fog", n, peerId);
423 SendConfig(&gConfig, "Game.SightRange", n, peerId);
424 SendConfig(&gConfig, "Game.AllyCollision", n, peerId);
425
426 NetServerSendMsg(n, peerId, GAME_EVENT_NET_GAME_START, NULL);
427
428 // Send all actors
429 CA_FOREACH(const TActor, a, gActors)
430 if (!a->isInUse)
431 {
432 continue;
433 }
434 NActorAdd aa = NActorAdd_init_default;
435 aa.UID = a->uid;
436 aa.CharId = a->charId;
437 aa.Health = a->health;
438 aa.Direction = (int32_t)a->direction;
439 aa.PlayerUID = a->PlayerUID;
440 aa.TileItemFlags = a->tileItem.flags;
441 aa.FullPos.x = a->Pos.x;
442 aa.FullPos.y = a->Pos.y;
443 LOG(LM_NET, LL_DEBUG, "send add actor UID(%d) playerUID(%d)",
444 (int)aa.UID, (int)aa.PlayerUID);
445 NetServerSendMsg(n, peerId, GAME_EVENT_ACTOR_ADD, &aa);
446 CA_FOREACH_END()
447
448 // Send key state
449 NAddKeys ak = NAddKeys_init_default;
450 ak.KeyFlags = gMission.KeyFlags;
451 NetServerSendMsg(n, peerId, GAME_EVENT_ADD_KEYS, &ak);
452
453 // Send objective counts
454 CA_FOREACH(const Objective, o, gMission.missionData->Objectives)
455 NObjectiveUpdate ou = NObjectiveUpdate_init_default;
456 ou.ObjectiveId = _ca_index;
457 ou.Count = o->done;
458 NetServerSendMsg(n, peerId, GAME_EVENT_OBJECTIVE_UPDATE, &ou);
459 CA_FOREACH_END()
460
461 // Send all tiles, RLE
462 const Tile *tLast = NULL;
463 NTileSet ts = NTileSet_init_default;
464 Vec2i pos;
465 for (pos.y = 0; pos.y < gMap.Size.y; pos.y++)
466 {
467 for (pos.x = 0; pos.x < gMap.Size.x; pos.x++)
468 {
469 const Tile *t = MapGetTile(&gMap, pos);
470 // Use RLE, so check if the current tile is the same as the last
471 if (tLast != NULL &&
472 t->pic == tLast->pic && t->picAlt == tLast->picAlt &&
473 t->flags == tLast->flags)
474 {
475 ts.RunLength++;
476 }
477 else
478 {
479 // Send the last run
480 if (tLast != NULL)
481 {
482 NetServerSendMsg(n, peerId, GAME_EVENT_TILE_SET, &ts);
483 }
484 // Begin the next run
485 memset(&ts, 0, sizeof ts);
486 ts.Pos = Vec2i2Net(pos);
487 if (t->pic != NULL) strcpy(ts.PicName, t->pic->name);
488 if (t->picAlt != NULL) strcpy(ts.PicAltName, t->picAlt->name);
489 ts.Flags = t->flags;
490 ts.RunLength = 0;
491 }
492 tLast = t;
493 }
494 }
495 NetServerSendMsg(n, peerId, GAME_EVENT_TILE_SET, &ts);
496
497 // Send all the tiles visited so far
498 NExploreTiles et = NExploreTiles_init_default;
499 et.Runs_count = 0;
500 et.Runs[0].Run = 0;
501 bool run = false;
502 for (pos.y = 0; pos.y < gMap.Size.y; pos.y++)
503 {
504 for (pos.x = 0; pos.x < gMap.Size.x; pos.x++)
505 {
506 const Tile *t = MapGetTile(&gMap, pos);
507 if (LOSAddRun(&et, &run, pos, t->isVisited))
508 {
509 NetServerSendMsg(n, peerId, GAME_EVENT_EXPLORE_TILES, &et);
510 et.Runs_count = 0;
511 et.Runs[0].Run = 0;
512 run = false;
513 }
514 }
515 }
516 if (et.Runs_count > 0)
517 {
518 NetServerSendMsg(n, peerId, GAME_EVENT_EXPLORE_TILES, &et);
519 }
520
521 // Send all pickups
522 CA_FOREACH(const Pickup, p, gPickups)
523 if (!p->isInUse) continue;
524 NAddPickup api = NAddPickup_init_default;
525 api.UID = p->UID;
526 strcpy(api.PickupClass, p->class->Name);
527 api.IsRandomSpawned = p->IsRandomSpawned;
528 api.SpawnerUID = p->SpawnerUID;
529 api.TileItemFlags = p->tileItem.flags;
530 api.Pos = Vec2i2Net(Vec2iNew(p->tileItem.x, p->tileItem.y));
531 NetServerSendMsg(n, peerId, GAME_EVENT_ADD_PICKUP, &api);
532 CA_FOREACH_END()
533
534 // Send all map objects
535 CA_FOREACH(const TObject, o, gObjs)
536 if (!o->isInUse) continue;
537 NMapObjectAdd amo = NMapObjectAdd_init_default;
538 amo.UID = o->uid;
539 strcpy(amo.MapObjectClass, o->Class->Name);
540 amo.Pos = Vec2i2Net(Vec2iNew(o->tileItem.x, o->tileItem.y));
541 amo.TileItemFlags = o->tileItem.flags;
542 amo.Health = o->Health;
543 LOG(LM_NET, LL_DEBUG,
544 "send add map object UID(%d) pos(%d, %d) flags(%x) health(%d)",
545 (int)amo.UID, amo.Pos.x, amo.Pos.y, amo.TileItemFlags, amo.Health);
546 NetServerSendMsg(n, peerId, GAME_EVENT_MAP_OBJECT_ADD, &amo);
547 CA_FOREACH_END()
548
549 // If mission complete already, send message
550 if (CanCompleteMission(&gMission))
551 {
552 NMissionComplete mc = NMakeMissionComplete(&gMission, &gMap);
553 NetServerSendMsg(n, peerId, GAME_EVENT_MISSION_COMPLETE, &mc);
554 }

556static void SendConfig(
557 Config *config, const char *name, NetServer *n, const int peerId)
558{
559 NConfig msg = NConfig_init_default;
560 const Config *c = ConfigGet(config, name);
561 strcpy(msg.Name, name);
562 switch (c->Type)
563 {
564 case CONFIG_TYPE_STRING:
565 CASSERT(false, "unimplemented");
566 break;
567 case CONFIG_TYPE_INT:
568 sprintf(msg.Value, "%d", c->u.Int.Value);
569 break;
570 case CONFIG_TYPE_FLOAT:
571 sprintf(msg.Value, "%f", c->u.Float.Value);
572 break;
573 case CONFIG_TYPE_BOOL:
574 strcpy(msg.Value, c->u.Bool.Value ? "true" : "false");
575 break;
576 case CONFIG_TYPE_ENUM:
577 sprintf(msg.Value, "%d", (int)c->u.Enum.Value);
578 break;
579 case CONFIG_TYPE_GROUP:
580 CASSERT(false, "Cannot send groups over net");
581 break;
582 default:
583 CASSERT(false, "Unknown config type");
584 break;
585 }
586 NetServerSendMsg(n, peerId, GAME_EVENT_CONFIG, &msg);
587}

589void NetServerSendMsg(
590 NetServer *n, const int peerId, const GameEventType e, const void *data)
591{
592 if (!n->server) return;
593
594 if (peerId >= 0)
595 {
596 LOG(LM_NET, LL_TRACE, "send msg(%d) to peers(%d)",
597 (int)e, (int)n->server->connectedPeers);
598 // Find the peer and send
599 for (int i = 0; i < (int)n->server->peerCount; i++)
600 {
601 ENetPeer *peer = n->server->peers + i;
602 if (peer->data != NULL &&
603 ((NetPeerData *)peer->data)->Id == peerId)
604 {
605 enet_peer_send(peer, 0, NetEncode(e, data));
606 return;
607 }
608 }
609 CASSERT(false, "Cannot find peer by id");
610 }
611 else
612 {
613 LOG(LM_NET, LL_TRACE, "bcast msg(%d) to peers(%d)",
614 (int)e, (int)n->server->connectedPeers);
615 enet_host_broadcast(n->server, 0, NetEncode(e, data));
616 }
