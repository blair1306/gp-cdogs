85bool CampaignLoad(CampaignOptions *co, CampaignEntry *entry)
86{
87 CASSERT(!co->IsLoaded, "loading campaign without unloading last one");
88 // Note: use the mode already set by the menus
89 const GameMode mode = co->Entry.Mode;
90 CampaignEntryCopy(&co->Entry, entry);
91 co->Entry.Mode = mode;
92 CampaignSettingInit(&co->Setting);
93 if (entry->Mode == GAME_MODE_QUICK_PLAY)
94 {
95 SetupQuickPlayCampaign(&co->Setting);
96 co->IsLoaded = true;
97 }
98 else
99 {
100 // Normalise the path
101 char buf[CDOGS_PATH_MAX];
102 GetDataFilePath(buf, entry->Path);
103 if (MapNewLoad(buf, &co->Setting))
104 {
105 LOG(LM_MAIN, LL_ERROR, "failed to load campaign %s!", buf);
106 CASSERT(false, "Failed to load campaign");
107 }
108 else
109 {
110 co->IsLoaded = true;
111 }
112 }
113
114 if (co->IsLoaded)
115 {
116 LOG(LM_MAIN, LL_INFO, "loaded campaign/dogfight");
117 }
118 return co->IsLoaded;
119}
120void CampaignUnload(CampaignOptions *co)
121{
122 co->IsLoaded = false;
123 co->IsClient = false; // TODO: select is client from menu
124 co->OptionsSet = false;
125 co->IsComplete = false;
126 CampaignEntryTerminate(&co->Entry);
127}

129void MissionOptionsInit(struct MissionOptions *mo)
130{
131 memset(mo, 0, sizeof *mo);
132 CArrayInit(&mo->Weapons, sizeof(GunDescription *));
133}
134void MissionOptionsTerminate(struct MissionOptions *mo)
135{
136 ActorsTerminate();
137 ObjsTerminate();
138 MobObjsTerminate();
139 PickupsTerminate();
140 ParticlesTerminate(&gParticles);
141 WatchesTerminate();
142 CA_FOREACH(PlayerData, p, gPlayerDatas)
143 p->ActorUID = -1;
144 CA_FOREACH_END()
145 gMission.HasStarted = false;
146 gMission.HasBegun = false;
147 CArrayTerminate(&mo->Weapons);
148
149 memset(mo, 0, sizeof *mo);
150}

156void AddSong(struct SongDef **songList, const char *path)
157{
158 struct SongDef *s;
159 CMALLOC(s, sizeof(struct SongDef));
160 strcpy(s->path, path);
161 s->next = *songList;
162 *songList = s;
163}
164
165void ShiftSongs(struct SongDef **songList)
166{
167 struct SongDef **h;
168
169 if (!*songList || !(*songList)->next)
170 return;
171
172 h = songList;
173 while (*h)
174 h = &(*h)->next;
175
176 *h = *songList;
177 *songList = (*songList)->next;
178 (*h)->next = NULL;
179}
180
181void FreeSongs(struct SongDef **songList)
182{
183 struct SongDef *s;
184
185 while (*songList) {
186 s = *songList;
187 *songList = s->next;
188 CFREE(s);
189 }
190}

200void LoadSongList(struct SongDef **songList, const char *dirPath)
201{
202 tinydir_dir dir;
203 int errsv;
204 if (tinydir_open(&dir, dirPath) == -1)
205 {
206 errsv = errno;
207 printf("Cannot open music dir: %s\n", strerror(errsv));
208 goto bail;
209 }
210
211 for (; dir.has_next; tinydir_next(&dir))
212 {
213 Mix_Music *m;
214 tinydir_file file;
215 if (tinydir_readfile(&dir, &file) == -1)
216 {
217 errsv = errno;
218 debug(D_VERBOSE, "cannot read file: %s\n", strerror(errsv));
219 goto bail;
220 }
221 if (!file.is_reg)
222 {
223 debug(D_VERBOSE, "not a regular file %s\n", file.name);
224 continue;
225 }
226 if (strcmp(file.extension, "txt") == 0 ||
227 strcmp(file.extension, "TXT") == 0)
228 {
229 debug(D_VERBOSE, "Skipping text file %s\n", file.name);
230 continue;
231 }
232
233 m = Mix_LoadMUS(file.path);
234 if (m == NULL)
235 {
236 debug(D_VERBOSE, "not a music file %s\n", file.name);
237 continue;
238 }
239 Mix_FreeMusic(m);
240 AddSong(songList, file.path);
241 }
242
243bail:
244 tinydir_close(&dir);
245}

248bool GameIsMouseUsed(void)
249{
250 CA_FOREACH(const PlayerData, p, gPlayerDatas)
251 if (p->IsLocal && p->inputDevice == INPUT_DEVICE_MOUSE)
252 {
253 const TActor *a = ActorGetByUID(p->ActorUID);
254 if (a == NULL) continue;
255 if (a->dead) continue;
256 return true;
257 }
258 CA_FOREACH_END()
259 return false;
260}
