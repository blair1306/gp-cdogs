45int MapNewScanArchive(
46 const char *filename, char **title, int *numMissions)
47{
48 int err = 0;
49 json_t *root = ReadArchiveJSON(filename, "campaign.json");
50 if (root == NULL)
51 {
52 err = -1;
53 goto bail;
54 }
55 err = MapNewScanJSON(root, title, numMissions);
56 if (err < 0)
57 {
58 goto bail;
59 }
60
61bail:
62 json_free_value(&root);
63 return err;
64}

69int MapNewLoadArchive(const char *filename, CampaignSetting *c)
70{
71 LOG(LM_MAP, LL_DEBUG, "Loading archive map %s", filename);
72 int err = 0;
73 json_t *root = ReadArchiveJSON(filename, "campaign.json");
74 if (root == NULL)
75 {
76 err = -1;
77 goto bail;
78 }
79 int version;
80 LoadInt(&version, root, "Version");
81 if (version > MAP_VERSION || version <= 2)
82 {
83 err = -1;
84 goto bail;
85 }
86 MapNewLoadCampaignJSON(root, c);
87 json_free_value(&root);
88
89 // Load any custom data
90 LoadArchiveSounds(&gSoundDevice, filename, "sounds");
91
92 LoadArchivePics(&gPicManager, gCharSpriteClasses.customClasses, filename);
93
94 root = ReadArchiveJSON(filename, "particles.json");
95 if (root != NULL)
96 {
97 ParticleClassesLoadJSON(&gParticleClasses.CustomClasses, root);
98 }
99
100 root = ReadArchiveJSON(filename, "character_classes.json");
101 if (root != NULL)
102 {
103 CharacterClassesLoadJSON(
104 &gCharacterClasses.CustomClasses, root);
105 }
106
107 root = ReadArchiveJSON(filename, "bullets.json");
108 if (root != NULL)
109 {
110 BulletLoadJSON(
111 &gBulletClasses, &gBulletClasses.CustomClasses, root);
112 }
113
114 root = ReadArchiveJSON(filename, "ammo.json");
115 if (root != NULL)
116 {
117 AmmoLoadJSON(&gAmmo.CustomAmmo, root);
118 json_free_value(&root);
119 }
120
121 root = ReadArchiveJSON(filename, "guns.json");
122 if (root != NULL)
123 {
124 WeaponLoadJSON(
125 &gGunDescriptions, &gGunDescriptions.CustomGuns, root);
126 json_free_value(&root);
127 }
128
129 BulletLoadWeapons(&gBulletClasses);
130
131 root = ReadArchiveJSON(filename, "pickups.json");
132 if (root != NULL)
133 {
134 PickupClassesLoadJSON(&gPickupClasses.CustomClasses, root);
135 }
136 PickupClassesLoadAmmo(&gPickupClasses.CustomClasses, &gAmmo.CustomAmmo);
137 PickupClassesLoadGuns(
138 &gPickupClasses.CustomClasses, &gGunDescriptions.CustomGuns);
139 PickupClassesLoadKeys(&gPickupClasses.KeyClasses);
140
141 // Reset custom map objects
142 MapObjectsClear(&gMapObjects.CustomClasses);
143 root = ReadArchiveJSON(filename, "map_objects.json");
144 if (root != NULL)
145 {
146 MapObjectsLoadJSON(&gMapObjects.CustomClasses, root);
147 }
148 MapObjectsLoadAmmoAndGunSpawners(
149 &gMapObjects, &gAmmo, &gGunDescriptions, true);
150
151
152 root = ReadArchiveJSON(filename, "missions.json");
153 if (root == NULL)
154 {
155 err = -1;
156 goto bail;
157 }
158 LoadMissions(
159 &c->Missions, json_find_first_label(root, "Missions")->child, version);
160 json_free_value(&root);
161
162 // Note: some campaigns don't have characters (e.g. dogfights)
163 root = ReadArchiveJSON(filename, "characters.json");
164 if (root != NULL)
165 {
166 CharacterLoadJSON(&c->characters, root, version);
167 }
168
169bail:
170 json_free_value(&root);
171 return err;
172}

174static json_t *ReadArchiveJSON(const char *archive, const char *filename)
175{
176 json_t *root = NULL;
177 debug(D_VERBOSE, "Loading archive json %s %s\n", archive, filename);
178 char path[CDOGS_PATH_MAX];
179 sprintf(path, "%s/%s", archive, filename);
180 long len;
181 char *buf = ReadFileIntoBuf(path, "rb", &len);
182 if (buf == NULL) goto bail;
183 const enum json_error e = json_parse_document(&root, buf);
184 if (e != JSON_OK)
185 {
186 LOG(LM_MAP, LL_ERROR, "Invalid syntax in JSON file (%s) error(%d)",
187 filename, (int)e);
188 root = NULL;
189 goto bail;
190 }
191
192bail:
193 CFREE(buf);
194 return root;
195}

212static char *ReadFileIntoBuf(const char *path, const char *mode, long *len)
213{
214 char *buf = NULL;
215 FILE *f = fopen(path, mode);
216 if (f == NULL)
217 {
218 debug(D_NORMAL, "Did not open file %s: %s.\n", path, strerror(errno));
219 goto bail;
220 }
221
222 // Read into buffer
223 if (fseek(f, 0L, SEEK_END) != 0)
224 {
225 debug(D_NORMAL, "Cannot seek file %s: %s.\n", path, strerror(errno));
226 goto bail;
227 }
228 *len = ftell(f);
229 if (*len == -1)
230 {
231 debug(D_NORMAL, "Cannot tell file %s: %s.\n", path, strerror(errno));
232 goto bail;
233 }
234 CCALLOC(buf, *len + 1);
235 if (fseek(f, 0L, SEEK_SET) != 0)
236 {
237 debug(D_NORMAL, "Cannot seek file %s: %s.\n", path, strerror(errno));
238 goto bail;
239 }
240 if (fread(buf, 1, *len, f) == 0)
241 {
242 debug(D_NORMAL, "Cannot read file %s: %s.\n", path, strerror(errno));
243 goto bail;
244 }
245
246 goto end;
247
248bail:
249 CFREE(buf);
250 buf = NULL;
251
252end:
253 if (f != NULL && fclose(f) != 0)
254 {
255 debug(D_NORMAL, "Cannot close file %s: %s.\n", path, strerror(errno));
256 }
257 return buf;
258}

261static json_t *SaveMissions(CArray *a);
262int MapArchiveSave(const char *filename, CampaignSetting *c)
263{
264 int res = 1;
265 json_t *root = NULL;
266
267 char relbuf[CDOGS_PATH_MAX];
268 if (strcmp(StrGetFileExt(filename), "cdogscpn") == 0 ||
269 strcmp(StrGetFileExt(filename), "CDOGSCPN") == 0)
270 {
271 strcpy(relbuf, filename);
272 }
273 else
274 {
275 sprintf(relbuf, "%s.cdogscpn", filename);
276 }
277 char buf[CDOGS_PATH_MAX];
278 RealPath(relbuf, buf);
279 // Make dir but ignore error, as we may be saving over an existing dir
280 mkdir_deep(buf);
281
282 // Campaign
283 root = json_new_object();
284 AddIntPair(root, "Version", MAP_VERSION);
285 AddStringPair(root, "Title", c->Title);
286 AddStringPair(root, "Author", c->Author);
287 AddStringPair(root, "Description", c->Description);
288 AddIntPair(root, "Missions", c->Missions.size);
289 char buf2[CDOGS_PATH_MAX];
290 sprintf(buf2, "%s/campaign.json", buf);
291 if (!TrySaveJSONFile(root, buf2))
292 {
293 res = 0;
294 goto bail;
295 }
296
297 json_free_value(&root);
298 root = json_new_object();
299 json_insert_pair_into_object(root, "Missions", SaveMissions(&c->Missions));
300 sprintf(buf2, "%s/missions.json", buf);
301 if (!TrySaveJSONFile(root, buf2))
302 {
303 res = 0;
304 goto bail;
305 }
306
307 if (!CharacterSave(&c->characters, buf))
308 {
309 res = 0;
310 goto bail;
311 }
312
313bail:
314 json_free_value(&root);
315 return res;
316}

330static json_t *SaveMissions(CArray *a)
331{
332 json_t *missionsNode = json_new_array();
333 for (int i = 0; i < (int)a->size; i++)
334 {
335 json_t *node = json_new_object();
336 Mission *mission = CArrayGet(a, i);
337 AddStringPair(node, "Title", mission->Title);
338 AddStringPair(node, "Description", mission->Description);
339 AddStringPair(node, "Type", MapTypeStr(mission->Type));
340 AddIntPair(node, "Width", mission->Size.x);
341 AddIntPair(node, "Height", mission->Size.y);
342
343 AddStringPair(node, "WallStyle", mission->WallStyle);
344 AddStringPair(node, "FloorStyle", mission->FloorStyle);
345 AddStringPair(node, "RoomStyle", mission->RoomStyle);
346 AddStringPair(node, "ExitStyle", mission->ExitStyle);
347 AddStringPair(node, "KeyStyle", mission->KeyStyle);
348 AddStringPair(node, "DoorStyle", mission->DoorStyle);
349
350 json_insert_pair_into_object(
351 node, "Objectives", SaveObjectives(&mission->Objectives));
352 json_insert_pair_into_object(
353 node, "Enemies", SaveIntArray(&mission->Enemies));
354 json_insert_pair_into_object(
355 node, "SpecialChars", SaveIntArray(&mission->SpecialChars));
356 json_t *modsNode = json_new_array();
357 for (int j = 0; j < (int)mission->MapObjectDensities.size; j++)
358 {
359 const MapObjectDensity *mod =
360 CArrayGet(&mission->MapObjectDensities, j);
361 json_t *modNode = json_new_object();
362 AddStringPair(modNode, "MapObject", mod->M->Name);
363 AddIntPair(modNode, "Density", mod->Density);
364 json_insert_child(modsNode, modNode);
365 }
366 json_insert_pair_into_object(node, "MapObjectDensities", modsNode);
367
368 AddIntPair(node, "EnemyDensity", mission->EnemyDensity);
369 json_insert_pair_into_object(
370 node, "Weapons", SaveWeapons(&mission->Weapons));
371
372 json_insert_pair_into_object(
373 node, "Song", json_new_string(mission->Song));
374
375 AddColorPair(node, "WallMask", mission->WallMask);
376 AddColorPair(node, "FloorMask", mission->FloorMask);
377 AddColorPair(node, "RoomMask", mission->RoomMask);
378 AddColorPair(node, "AltMask", mission->AltMask);
379
380 switch (mission->Type)
381 {
382 case MAPTYPE_CLASSIC:
383 AddIntPair(node, "Walls", mission->u.Classic.Walls);
384 AddIntPair(node, "WallLength", mission->u.Classic.WallLength);
385 AddIntPair(
386 node, "CorridorWidth", mission->u.Classic.CorridorWidth);
387 json_insert_pair_into_object(
388 node, "Rooms", SaveRooms(mission->u.Classic.Rooms));
389 AddIntPair(node, "Squares", mission->u.Classic.Squares);
390 json_insert_pair_into_object(
391 node, "Doors", SaveClassicDoors(mission));
392 json_insert_pair_into_object(
393 node, "Pillars", SaveClassicPillars(mission));
394 break;
395 case MAPTYPE_STATIC:
396 {
397 json_insert_pair_into_object(
398 node, "Tiles", SaveStaticTiles(mission));
399 json_insert_pair_into_object(
400 node, "StaticItems", SaveStaticItems(mission));
401 json_insert_pair_into_object(
402 node, "StaticCharacters", SaveStaticCharacters(mission));
403 json_insert_pair_into_object(
404 node, "StaticObjectives", SaveStaticObjectives(mission));
405 json_insert_pair_into_object(
406 node, "StaticKeys", SaveStaticKeys(mission));
407
408 json_insert_pair_into_object(
409 node, "Start", SaveVec2i(mission->u.Static.Start));
410 json_t *exitNode = json_new_object();
411 json_insert_pair_into_object(
412 exitNode, "Start",
413 SaveVec2i(mission->u.Static.Exit.Start));
414 json_insert_pair_into_object(
415 exitNode, "End",
416 SaveVec2i(mission->u.Static.Exit.End));
417 json_insert_pair_into_object(node, "Exit", exitNode);
418 }
419 break;
420 case MAPTYPE_CAVE:
421 AddIntPair(node, "FillPercent", mission->u.Cave.FillPercent);
422 AddIntPair(node, "Repeat", mission->u.Cave.Repeat);
423 AddIntPair(node, "R1", mission->u.Cave.R1);
424 AddIntPair(node, "R2", mission->u.Cave.R2);
425 json_insert_pair_into_object(
426 node, "Rooms", SaveRooms(mission->u.Cave.Rooms));
427 AddIntPair(node, "Squares", mission->u.Cave.Squares);
428 break;
429 default:
430 assert(0 && "unknown map type");
431 break;
432 }
433
434 json_insert_child(missionsNode, node);
435 }
436 return missionsNode;
437}

468static json_t *SaveStaticTiles(Mission *m)
469{
470 // Create a text buffer for CSV
471 // The buffer will contain n*5 chars (tiles, allow 5 chars each),
472 // and n - 1 commas, so 6n total
473 const int size = (int)m->u.Static.Tiles.size;
474 if (size == 0)
475 {
476 return json_new_string("");
477 }
478 char *bigbuf;
479 CCALLOC(bigbuf, size * 6);
480 char *pBuf = bigbuf;
481 CASSERT(pBuf != NULL, "memory error");
482 for (int i = 0; i < size; i++)
483 {
484 char buf[32];
485 sprintf(buf, "%d", *(unsigned short *)CArrayGet(
486 &m->u.Static.Tiles, i));
487 strcpy(pBuf, buf);
488 pBuf += strlen(buf);
489 if (i < size - 1)
490 {
491 *pBuf = ',';
492 pBuf++;
493 }
494 }
495 json_t *node = json_new_string(bigbuf);
496 CFREE(bigbuf);
497 return node;
498}
499static json_t *SaveStaticItems(Mission *m)
500{
501 json_t *items = json_new_array();
502 CA_FOREACH(MapObjectPositions, mop, m->u.Static.Items)
503 json_t *itemNode = json_new_object();
504 AddStringPair(itemNode, "MapObject", mop->M->Name);
505 json_t *positions = json_new_array();
506 for (int j = 0; j < (int)mop->Positions.size; j++)
507 {
508 Vec2i *pos = CArrayGet(&mop->Positions, j);
509 json_insert_child(positions, SaveVec2i(*pos));
510 }
511 json_insert_pair_into_object(
512 itemNode, "Positions", positions);
513 json_insert_child(items, itemNode);
514 CA_FOREACH_END()
515 return items;
516}

574static json_t *SaveObjectives(CArray *a)
575{
576 json_t *objectivesNode = json_new_array();
577 CA_FOREACH(const Objective, o, *a)
578 json_t *objNode = json_new_object();
579 AddStringPair(objNode, "Description", o->Description);
580 AddStringPair(objNode, "Type", ObjectiveTypeStr(o->Type));
581 switch (o->Type)
582 {
583 case OBJECTIVE_COLLECT:
584 AddStringPair(objNode, "Pickup", o->u.Pickup->Name);
585 break;
586 case OBJECTIVE_DESTROY:
587 AddStringPair(objNode, "MapObject", o->u.MapObject->Name);
588 break;
589 default:
590 AddIntPair(objNode, "Index", o->u.Index);
591 break;
592 }
593 AddIntPair(objNode, "Count", o->Count);
594 AddIntPair(objNode, "Required", o->Required);
595 AddIntPair(objNode, "Flags", o->Flags);
596 json_insert_child(objectivesNode, objNode);
597 CA_FOREACH_END()
598 return objectivesNode;
599}
