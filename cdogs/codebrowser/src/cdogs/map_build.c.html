58void MapSetTile(Map *map, Vec2i pos, unsigned short tileType, Mission *m)
59{
60 IMapSet(map, pos, tileType);
61 // Update the tile as well, plus neighbours as they may be affected
62 // by shadows etc. especially walls
63 MapSetupTile(map, pos, m);
64 MapSetupTile(map, Vec2iNew(pos.x - 1, pos.y), m);
65 MapSetupTile(map, Vec2iNew(pos.x + 1, pos.y), m);
66 MapSetupTile(map, Vec2iNew(pos.x, pos.y - 1), m);
67 MapSetupTile(map, Vec2iNew(pos.x, pos.y + 1), m);
68}

70void MapSetupTilesAndWalls(Map *map, const Mission *m)
71{
72 // Pre-load the tile pics that this map will use
73 // TODO: multiple styles and colours
74 // Walls
75 for (int i = 0; i < WALL_TYPE_COUNT; i++)
76 {
77 PicManagerGenerateMaskedStylePic(
78 &gPicManager, "wall", m->WallStyle, IntWallType(i),
79 m->WallMask, m->AltMask);
80 }
81 // Floors
82 for (int i = 0; i < FLOOR_TYPES; i++)
83 {
84 PicManagerGenerateMaskedStylePic(
85 &gPicManager, "tile", m->FloorStyle, IntTileType(i),
86 m->FloorMask, m->AltMask);
87 }
88 // Rooms
89 for (int i = 0; i < ROOMFLOOR_TYPES; i++)
90 {
91 PicManagerGenerateMaskedStylePic(
92 &gPicManager, "tile", m->RoomStyle, IntTileType(i),
93 m->RoomMask, m->AltMask);
94 }
95
96 Vec2i v;
97 for (v.x = 0; v.x < map->Size.x; v.x++)
98 {
99 for (v.y = 0; v.y < map->Size.y; v.y++)
100 {
101 MapSetupTile(map, v, m);
102 }
103 }
104
105 // Randomly change normal floor tiles to alternative floor tiles
106 for (int i = 0; i < map->Size.x*map->Size.y / 22; i++)
107 {
108 Tile *t = MapGetTile(
109 map, Vec2iNew(rand() % map->Size.x, rand() % map->Size.y));
110 if (TileIsNormalFloor(t))
111 {
112 TileSetAlternateFloor(t, PicManagerGetMaskedStylePic(
113 &gPicManager, "tile", m->FloorStyle, "alt1",
114 m->FloorMask, m->AltMask));
115 }
116 }
117 for (int i = 0; i < map->Size.x*map->Size.y / 16; i++)
118 {
119 Tile *t = MapGetTile(
120 map, Vec2iNew(rand() % map->Size.x, rand() % map->Size.y));
121 if (TileIsNormalFloor(t))
122 {
123 TileSetAlternateFloor(t, PicManagerGetMaskedStylePic(
124 &gPicManager, "tile", m->FloorStyle, "alt2",
125 m->FloorMask, m->AltMask));
126 }
127 }
128}
129static const char *MapGetWallPic(const Map *m, const Vec2i pos);
130// Set tile properties for a map tile, such as picture to use
131static void MapSetupTile(Map *map, const Vec2i pos, const Mission *m)
132{
133 Tile *tAbove = MapGetTile(map, Vec2iNew(pos.x, pos.y - 1));
134 bool canSeeTileAbove = !(tAbove != NULL && !TileCanSee(tAbove));
135 Tile *t = MapGetTile(map, pos);
136 if (!t)
137 {
138 return;
139 }
140 switch (IMapGet(map, pos) & MAP_MASKACCESS)
141 {
142 case MAP_FLOOR:
143 case MAP_SQUARE:
144 t->pic = PicManagerGetMaskedStylePic(
145 &gPicManager, "tile", m->FloorStyle,
146 canSeeTileAbove ? "normal" : "shadow",
147 m->FloorMask, m->AltMask);
148 if (canSeeTileAbove)
149 {
150 // Normal floor tiles can be replaced randomly with
151 // special floor tiles such as drainage
152 t->flags |= MAPTILE_IS_NORMAL_FLOOR;
153 }
154 break;
155
156 case MAP_ROOM:
157 case MAP_DOOR:
158 t->pic = PicManagerGetMaskedStylePic(
159 &gPicManager, "tile", m->RoomStyle,
160 canSeeTileAbove ? "normal" : "shadow",
161 m->RoomMask, m->AltMask);
162 break;
163
164 case MAP_WALL:
165 t->pic = PicManagerGetMaskedStylePic(
166 &gPicManager, "wall", m->WallStyle, MapGetWallPic(map, pos),
167 m->WallMask, m->AltMask);
168 t->flags =
169 MAPTILE_NO_WALK | MAPTILE_NO_SHOOT |
170 MAPTILE_NO_SEE | MAPTILE_IS_WALL;
171 break;
172
173 case MAP_NOTHING:
174 t->pic = NULL;
175 t->flags =
176 MAPTILE_NO_WALK | MAPTILE_IS_NOTHING;
177 break;
178 }
179}

252int MapIsValidStartForWall(
253 Map *map, int x, int y, unsigned short tileType, int pad)
254{
255 Vec2i d;
256 if (x == 0 || y == 0 || x == map->Size.x - 1 || y == map->Size.y - 1)
257 {
258 return 0;
259 }
260 for (d.x = x - pad; d.x <= x + pad; d.x++)
261 {
262 for (d.y = y - pad; d.y <= y + pad; d.y++)
263 {
264 if (IMapGet(map, d) != tileType)
265 {
266 return 0;
267 }
268 }
269 }
270 return 1;
271}

278Vec2i MapGetRoomSize(const RoomParams r, const int doorMin)
279{
280 // Work out dimensions of room
281 // make sure room is large enough to accommodate doors
282 const int roomMin = MAX(r.Min, doorMin + 4);
283 const int roomMax = MAX(r.Max, doorMin + 4);
284 return Vec2iNew(
285 rand() % (roomMax - roomMin + 1) + roomMin,
286 rand() % (roomMax - roomMin + 1) + roomMin);

289void MapMakeRoom(Map *map, const Vec2i pos, const Vec2i size, const bool walls)
290{
291 Vec2i v;
292 // Set the perimeter walls and interior
293 // If the tile is a room interior already, do not turn it into a wall
294 // This is due to overlapping rooms
295 for (v.y = pos.y; v.y < pos.y + size.y; v.y++)
296 {
297 for (v.x = pos.x; v.x < pos.x + size.x; v.x++)
298 {
299 if (v.y == pos.y || v.y == pos.y + size.y - 1 ||
300 v.x == pos.x || v.x == pos.x + size.x - 1)
301 {
302 if (walls && IMapGet(map, v) != MAP_ROOM)
303 {
304 IMapSet(map, v, MAP_WALL);
305 }
306 }
307 else
308 {
309 IMapSet(map, v, MAP_ROOM);
310 }
311 }
312 }
313 // Check perimeter again; if there are walls where both sides contain
314 // rooms, remove the wall as the rooms have merged
315 for (v.y = pos.y; v.y < pos.y + size.y; v.y++)
316 {
317 for (v.x = pos.x; v.x < pos.x + size.x; v.x++)
318 {
319 if (v.y == pos.y || v.y == pos.y + size.y - 1 ||
320 v.x == pos.x || v.x == pos.x + size.x - 1)
321 {
322 if (((IMapGet(map, Vec2iNew(v.x + 1, v.y)) & MAP_MASKACCESS) == MAP_ROOM &&
323 (IMapGet(map, Vec2iNew(v.x - 1, v.y)) & MAP_MASKACCESS) == MAP_ROOM) ||
324 ((IMapGet(map, Vec2iNew(v.x, v.y + 1)) & MAP_MASKACCESS) == MAP_ROOM &&
325 (IMapGet(map, Vec2iNew(v.x, v.y - 1)) & MAP_MASKACCESS) == MAP_ROOM))
326 {
327 IMapSet(map, v, MAP_ROOM);
328 }
329 }
330 }
331 }

334void MapMakeRoomWalls(Map *map, const RoomParams r)
335{
336 int count = 0;
337 for (int i = 0; i < 100 && count < r.Walls; i++)
338 {
339 if (!MapTryBuildWall(map, MAP_ROOM, MAX(r.WallPad, 1), r.WallLength))
340 {
341 continue;
342 }
343 count++;
344 }

347static void MapGrowWall(
348 Map *map, int x, int y,
349 unsigned short tileType, int pad, int d, int length);
350bool MapTryBuildWall(
351 Map *map, const unsigned short tileType, const int pad,
352 const int wallLength)
353{
354 const Vec2i v = MapGetRandomTile(map);
355 if (MapIsValidStartForWall(map, v.x, v.y, tileType, pad))
356 {
357 IMapSet(map, v, MAP_WALL);
358 MapGrowWall(map, v.x, v.y, tileType, pad, rand() & 3, wallLength);
359 return true;
360 }
361 return false;
362}
363static void MapGrowWall(
364 Map *map, int x, int y,
365 unsigned short tileType, int pad, int d, int length)
366{
367 int l;
368 Vec2i v;
369
370 if (length <= 0)
371 return;
372
373 switch (d) {
374 case 0:
375 if (y < 2 + pad)
376 {
377 return;
378 }
379 // Check tiles above
380 // xxxxx
381 // xxx
382 // o
383 for (v.y = y - 2; v.y > y - 2 - pad; v.y--)
384 {
385 int level = v.y - (y - 2);
386 for (v.x = x - 1 - level; v.x <= x + 1 + level; v.x++)
387 {
388 if (IMapGet(map, v) != tileType)
389 {
390 return;
391 }
392 }
393 }
394 y--;
395 break;
396 case 1:
397 // Check tiles to the right
398 // x
399 // xx
400 // oxx
401 // xx
402 // x
403 for (v.x = x + 2; v.x < x + 2 + pad; v.x++)
404 {
405 int level = v.x - (x + 2);
406 for (v.y = y - 1 - level; v.y <= y + 1 + level; v.y++)
407 {
408 if (IMapGet(map, v) != tileType)
409 {
410 return;
411 }
412 }
413 }
414 x++;
415 break;
416 case 2:
417 // Check tiles below
418 // o
419 // xxx
420 // xxxxx
421 for (v.y = y + 2; v.y < y + 2 + pad; v.y++)
422 {
423 int level = v.y - (y + 2);
424 for (v.x = x - 1 - level; v.x <= x + 1 + level; v.x++)
425 {
426 if (IMapGet(map, v) != tileType)
427 {
428 return;
429 }
430 }
431 }
432 y++;
433 break;
434 case 4:
435 if (x < 2 + pad)
436 {
437 return;
438 }
439 // Check tiles to the left
440 // x
441 // xx
442 // xxo
443 // xx
444 // x
445 for (v.x = x - 2; v.x > x - 2 - pad; v.x--)
446 {
447 int level = v.x - (x - 2);
448 for (v.y = y - 1 - level; v.y <= y + 1 + level; v.y++)
449 {
450 if (IMapGet(map, v) != tileType)
451 {
452 return;
453 }
454 }
455 }
456 x--;
457 break;
458 }
459 IMapSet(map, Vec2iNew(x, y), MAP_WALL);
460 length--;
461 if (length > 0 && (rand() & 3) == 0)
462 {
463 // Randomly try to grow the wall in a different direction
464 l = rand() % length;
465 MapGrowWall(map, x, y, tileType, pad, rand() & 3, l);
466 length -= l;
467 }
468 // Keep growing wall in same direction
469 MapGrowWall(map, x, y, tileType, pad, d, length);
470}

474void MapPlaceDoors(
475 Map *map, Vec2i pos, Vec2i size,
476 int hasDoors, int doors[4], int doorMin, int doorMax,
477 unsigned short accessMask)
478{
479 int x, y;
480 int i;
481 unsigned short doorTile = hasDoors ? MAP_DOOR : MAP_ROOM;
482 Vec2i v;
483
484 // Set access mask
485 for (y = pos.y + 1; y < pos.y + size.y - 1; y++)
486 {
487 for (x = pos.x + 1; x < pos.x + size.x - 1; x++)
488 {
489 if ((IMapGet(map, Vec2iNew(x, y)) & MAP_MASKACCESS) == MAP_ROOM)
490 {
491 IMapSet(map, Vec2iNew(x, y), MAP_ROOM | accessMask);
492 }
493 }
494 }
495
496 // Set the doors
497 if (doors[0])
498 {
499 int doorSize = MIN(
500 (doorMax > doorMin ? (rand() % (doorMax - doorMin + 1)) : 0) + doorMin,
501 size.y - 4);
502 for (i = -doorSize / 2; i < (doorSize + 1) / 2; i++)
503 {
504 v = Vec2iNew(pos.x, pos.y + size.y / 2 + i);
505 if (!TryPlaceDoorTile(map, v, Vec2iNew(1, 0), doorTile))
506 {
507 break;
508 }
509 }
510 }
511 if (doors[1])
512 {
513 int doorSize = MIN(
514 (doorMax > doorMin ? (rand() % (doorMax - doorMin + 1)) : 0) + doorMin,
515 size.y - 4);
516 for (i = -doorSize / 2; i < (doorSize + 1) / 2; i++)
517 {
518 v = Vec2iNew(pos.x + size.x - 1, pos.y + size.y / 2 + i);
519 if (!TryPlaceDoorTile(map, v, Vec2iNew(1, 0), doorTile))
520 {
521 break;
522 }
523 }
524 }
525 if (doors[2])
526 {
527 int doorSize = MIN(
528 (doorMax > doorMin ? (rand() % (doorMax - doorMin + 1)) : 0) + doorMin,
529 size.x - 4);
530 for (i = -doorSize / 2; i < (doorSize + 1) / 2; i++)
531 {
532 v = Vec2iNew(pos.x + size.x / 2 + i, pos.y);
533 if (!TryPlaceDoorTile(map, v, Vec2iNew(0, 1), doorTile))
534 {
535 break;
536 }
537 }
538 }
539 if (doors[3])
540 {
541 int doorSize = MIN(
542 (doorMax > doorMin ? (rand() % (doorMax - doorMin + 1)) : 0) + doorMin,
543 size.x - 4);
544 for (i = -doorSize / 2; i < (doorSize + 1) / 2; i++)
545 {
546 v = Vec2iNew(pos.x + size.x / 2 + i, pos.y + size.y - 1);
547 if (!TryPlaceDoorTile(map, v, Vec2iNew(0, 1), doorTile))
548 {
549 break;
550 }
551 }
552 }
553}
554static bool TryPlaceDoorTile(
555 Map *map, const Vec2i v, const Vec2i d, const unsigned short t)
556{
557 if (IMapGet(map, v) == MAP_WALL &&
558 IMapGet(map, Vec2iAdd(v, d)) != MAP_WALL &&
559 IMapGet(map, Vec2iMinus(v, d)) != MAP_WALL)
560 {
561 IMapSet(map, v, t);
562 return true;
563 }
564 return false;
565}

593static bool MapTileIsPartOfRoom(const Map *map, const Vec2i pos)
594{
595 Vec2i v2;
596 int isRoom = 0;
597 int isFloor = 0;
598 // Find whether a wall tile is part of a room perimeter
599 // The surrounding tiles must have normal floor and room tiles
600 // to be a perimeter
601 for (v2.y = pos.y - 1; v2.y <= pos.y + 1; v2.y++)
602 {
603 for (v2.x = pos.x - 1; v2.x <= pos.x + 1; v2.x++)
604 {
605 if ((IMapGet(map, v2) & MAP_MASKACCESS) == MAP_ROOM)
606 {
607 isRoom = 1;
608 }
609 else if ((IMapGet(map, v2) & MAP_MASKACCESS) == MAP_FLOOR)
610 {
611 isFloor = 1;
612 }
613 }
614 }
615 return isRoom && isFloor;
616}
617bool MapIsAreaClearOrRoom(const Map *map, const Vec2i pos, const Vec2i size)
618{
619 Vec2i v;
620
621 if (pos.x < 0 || pos.y < 0 ||
622 pos.x + size.x >= map->Size.x || pos.y + size.y >= map->Size.y)
623 {
624 return 0;
625 }
626
627 for (v.y = pos.y; v.y < pos.y + size.y; v.y++)
628 {
629 for (v.x = pos.x; v.x < pos.x + size.x; v.x++)
630 {
631 unsigned short tile = IMapGet(map, v) & MAP_MASKACCESS;
632 switch (tile)
633 {
634 case MAP_FLOOR: // fallthrough
635 case MAP_ROOM:
636 break;
637 case MAP_WALL:
638 // Check if this wall is part of a room
639 if (!MapTileIsPartOfRoom(map, v))
640 {
641 return 0;
642 }
643 break;
644 default:
645 return 0;
646 }
647 }
648 }
649
650 return 1;
651}
652int MapIsAreaClearOrWall(Map *map, Vec2i pos, Vec2i size)
653{
654 Vec2i v;
655
656 if (pos.x < 0 || pos.y < 0 ||
657 pos.x + size.x >= map->Size.x || pos.y + size.y >= map->Size.y)
658 {
659 return 0;
660 }
661
662 for (v.y = pos.y; v.y < pos.y + size.y; v.y++)
663 {
664 for (v.x = pos.x; v.x < pos.x + size.x; v.x++)
665 {
666 switch (IMapGet(map, v) & MAP_MASKACCESS)
667 {
668 case MAP_FLOOR:
669 break;
670 case MAP_WALL:
671 // need to check if this is not a room wall
672 if (MapTileIsPartOfRoom(map, v))
673 {
674 return 0;
675 }
676 break;
677 default:
678 return 0;
679 }
680 }
681 }
682
683 return 1;

685// Find the size of the passage created by the overlap of two rooms
686// To find whether an overlap is valid,
687// collect the perimeter walls that overlap
688// Two possible cases where there is a valid overlap:
689// - if there are exactly two overlapping perimeter walls
690// i.e.
691// X
692// room 2 XXXXXX
693// X X <-- all tiles between either belong to room 1 or room 2
694// XXXXXX
695// X room 1
696//
697// - if the collection of overlapping tiles are contiguous
698// i.e.
699// X room 1 X
700// XXXXXXXXXXXXXXX
701// X room 2 X
702//
703// In both cases, the overlap is valid if all tiles in between are room or
704// perimeter tiles. The size of the passage is given by the largest difference
705// in the x or y coordinates between the first and last intersection tiles,
706// minus 1
707bool MapGetRoomOverlapSize(
708 const Map *map,
709 const Vec2i pos,
710 const Vec2i size,
711 unsigned short *overlapAccess)
712{
713 Vec2i v;
714 int numOverlaps = 0;
715 Vec2i overlapMin = Vec2iZero();
716 Vec2i overlapMax = Vec2iZero();
717
718 if (pos.x < 0 || pos.y < 0 ||
719 pos.x + size.x >= map->Size.x || pos.y + size.y >= map->Size.y)
720 {
721 return 0;
722 }
723
724 // Find perimeter tiles that overlap
725 for (v.y = pos.y; v.y < pos.y + size.y; v.y++)
726 {
727 for (v.x = pos.x; v.x < pos.x + size.x; v.x++)
728 {
729 // only check perimeter
730 if (v.x != pos.x && v.x != pos.x + size.x - 1 &&
731 v.y != pos.y && v.y != pos.y + size.y - 1)
732 {
733 continue;
734 }
735 switch (IMapGet(map, v))
736 {
737 case MAP_WALL:
738 // Check if this wall is part of a room
739 if (MapTileIsPartOfRoom(map, v))
740 {
741 // Get the access level of the room
742 Vec2i v2;
743 for (v2.y = v.y - 1; v2.y <= v.y + 1; v2.y++)
744 {
745 for (v2.x = v.x - 1; v2.x <= v.x + 1; v2.x++)
746 {
747 if ((IMapGet(map, v2) & MAP_MASKACCESS) == MAP_ROOM)
748 {
749 *overlapAccess |=
750 IMapGet(map, v2) & MAP_ACCESSBITS;
751 }
752 }
753 }
754 if (numOverlaps == 0)
755 {
756 overlapMin = overlapMax = v;
757 }
758 else
759 {
760 overlapMin = Vec2iMin(overlapMin, v);
761 overlapMax = Vec2iMax(overlapMax, v);
762 }
763 numOverlaps++;
764 }
765 break;
766 default:
767 break;
768 }
769 }
770 }
771 if (numOverlaps < 2)
772 {
773 return 0;
774 }
775
776 // Now check that all tiles between the first and last tiles are room or
777 // perimeter tiles
778 for (v.y = overlapMin.y; v.y <= overlapMax.y; v.y++)
779 {
780 for (v.x = overlapMin.x; v.x <= overlapMax.x; v.x++)
781 {
782 switch (IMapGet(map, v) & MAP_MASKACCESS)
783 {
784 case MAP_ROOM:
785 break;
786 case MAP_WALL:
787 // Check if this wall is part of a room
788 if (!MapTileIsPartOfRoom(map, v))
789 {
790 return 0;
791 }
792 break;
793 default:
794 // invalid tile type
795 return 0;
796 }
797 }
798 }
799
800 return MAX(overlapMax.x - overlapMin.x, overlapMax.y - overlapMin.y) - 1;
801}
802// Check that this area does not overlap two or more "walls"
803int MapIsLessThanTwoWallOverlaps(Map *map, Vec2i pos, Vec2i size)
804{
805 Vec2i v;
806 int numOverlaps = 0;
807 Vec2i overlapMin = Vec2iZero();
808 Vec2i overlapMax = Vec2iZero();
809
810 if (pos.x < 0 || pos.y < 0 ||
811 pos.x + size.x >= map->Size.x || pos.y + size.y >= map->Size.y)
812 {
813 return 0;
814 }
815
816 for (v.y = pos.y; v.y < pos.y + size.y; v.y++)
817 {
818 for (v.x = pos.x; v.x < pos.x + size.x; v.x++)
819 {
820 // only check perimeter
821 if (v.x != pos.x && v.x != pos.x + size.x - 1 &&
822 v.y != pos.y && v.y != pos.y + size.y - 1)
823 {
824 continue;
825 }
826 switch (IMapGet(map, v))
827 {
828 case MAP_WALL:
829 // Check if this wall is part of a room
830 if (!MapTileIsPartOfRoom(map, v))
831 {
832 if (numOverlaps == 0)
833 {
834 overlapMin = overlapMax = v;
835 }
836 else
837 {
838 overlapMin = Vec2iMin(overlapMin, v);
839 overlapMax = Vec2iMax(overlapMax, v);
840 }
841 numOverlaps++;
842 }
843 break;
844 default:
845 break;
846 }
847 }
848 }
849 if (numOverlaps < 2)
850 {
851 return 1;
852 }
853
854 // Now check that all tiles between the first and last tiles are
855 // pillar tiles
856 for (v.y = overlapMin.y; v.y <= overlapMax.y; v.y++)
857 {
858 for (v.x = overlapMin.x; v.x <= overlapMax.x; v.x++)
859 {
860 switch (IMapGet(map, v) & MAP_MASKACCESS)
861 {
862 case MAP_WALL:
863 // Check if this wall is not part of a room
864 if (MapTileIsPartOfRoom(map, v))
865 {
866 return 0;
867 }
868 break;
869 default:
870 // invalid tile type
871 return 0;
872 }
873 }
874 }
875
876 return 1;
877}
