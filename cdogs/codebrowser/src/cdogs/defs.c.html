56int CmdGetReverse(int cmd)
57{
58 int newCmd = cmd & ~(CMD_LEFT | CMD_RIGHT | CMD_UP | CMD_DOWN);
59 if (cmd & CMD_LEFT) newCmd |= CMD_RIGHT;
60 if (cmd & CMD_RIGHT) newCmd |= CMD_LEFT;
61 if (cmd & CMD_UP) newCmd |= CMD_DOWN;
62 if (cmd & CMD_DOWN) newCmd |= CMD_UP;
63 return newCmd;
64}

119void GetVectorsForRadians(const double radians, double *x, double *y)
120{
121 *x = sin(radians);
122 *y = -cos(radians) * TILE_HEIGHT / TILE_WIDTH;
123}
124Vec2i GetFullVectorsForRadians(double radians)
125{
126 int scalar = 256;
127 // Scale Y so that they match the tile ratios
128 return Vec2iNew(
129 (int)(sin(radians) * scalar),
130 (int)(-cos(radians) * scalar * TILE_HEIGHT / TILE_WIDTH));
131}
132double Vec2iToRadians(const Vec2i v)
133{
134 return atan2(v.y, v.x) + PI / 2.0;
135}
136direction_e RadiansToDirection(const double r)
137{
138 // constrain to range [0, 2PI)
139 double radians = r;
140 while (radians < 0)
141 {
142 radians += 2 * PI;
143 }
144 while (radians >= 2 * PI)
145 {
146 radians -= 2 * PI;
147 }
148 int d = (int)floor((radians + PI / 8.0) / (PI / 4.0));
149 if (d < DIRECTION_UP)
150 {
151 d += DIRECTION_COUNT;
152 }
153 if (d >= DIRECTION_COUNT)
154 {
155 d -= DIRECTION_COUNT;
156 }
157 return (direction_e)d;
158}
159direction_e DirectionOpposite(const direction_e d)
160{
161 return (direction_e)(((int)d + 4) % DIRECTION_COUNT);
162}
