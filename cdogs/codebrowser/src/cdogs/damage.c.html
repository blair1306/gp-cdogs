55bool CanHitCharacter(const int flags, const int uid, const TActor *actor)
56{
57 // Don't let players hurt themselves
58 if (!(flags & FLAGS_HURTALWAYS) && uid == actor->uid)
59 {
60 return false;
61 }
62 return true;
63}
64
65bool CanDamageCharacter(
66 const int flags, const int playerUID, const int uid,
67 const TActor *actor, const special_damage_e special)
68{
69 if (!CanHitCharacter(flags, uid, actor))
70 {
71 return false;
72 }
73 if (ActorIsImmune(actor, special))
74 {
75 return false;
76 }
77 return !ActorIsInvulnerable(actor, flags, playerUID, gCampaign.Entry.Mode);
78}

80static void TrackKills(PlayerData *pd, const TActor *victim);
81void DamageActor(TActor *victim, const int power, const int hitterPlayerUID)
82{
83 const int startingHealth = victim->health;
84 InjureActor(victim, power);
85 if (startingHealth > 0 && victim->health <= 0 && hitterPlayerUID >= 0)
86 {
87 TrackKills(PlayerDataGetByUID(hitterPlayerUID), victim);
88 }
89}
90static void TrackKills(PlayerData *pd, const TActor *victim)
91{
92 if (!IsPVP(gCampaign.Entry.Mode) &&
93 (victim->PlayerUID >= 0 ||
94 (victim->flags & (FLAGS_GOOD_GUY | FLAGS_PENALTY))))
95 {
96 pd->Stats.Friendlies++;
97 pd->Totals.Friendlies++;
98 }
99 else if (pd->UID == victim->PlayerUID)
100 {
101 pd->Stats.Suicides++;
102 pd->Totals.Suicides++;
103 }
104 else
105 {
106 pd->Stats.Kills++;
107 pd->Totals.Kills++;
108 }
109}
