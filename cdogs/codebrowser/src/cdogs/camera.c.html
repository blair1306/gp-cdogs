40void CameraInit(Camera *camera)
41{
42 memset(camera, 0, sizeof *camera);
43 DrawBufferInit(
44 &camera->Buffer, Vec2iNew(X_TILES, Y_TILES), &gGraphicsDevice);
45 camera->lastPosition = Vec2iZero();
46 HUDInit(&camera->HUD, &gGraphicsDevice, &gMission);
47 camera->shake = ScreenShakeZero();
48}
49
50void CameraTerminate(Camera *camera)
51{
52 DrawBufferTerminate(&camera->Buffer);
53 HUDTerminate(&camera->HUD);
54}

56void CameraInput(Camera *camera, const int cmd, const int lastCmd)
57{
58 // Control the camera
59 if (camera->spectateMode == SPECTATE_NONE)
60 {
61 return;
62 }
63 // Arrows: pan camera
64 // CMD1/2: choose next player to follow
65 if (CMD_HAS_DIRECTION(cmd))
66 {
67 camera->spectateMode = SPECTATE_FREE;
68 const int pan = PAN_SPEED;
69 if (cmd & CMD_LEFT) camera->lastPosition.x -= pan;
70 else if (cmd & CMD_RIGHT) camera->lastPosition.x += pan;
71 if (cmd & CMD_UP) camera->lastPosition.y -= pan;
72 else if (cmd & CMD_DOWN) camera->lastPosition.y += pan;
73 }
74 else if ((AnyButton(cmd) && !AnyButton(lastCmd)) ||
75 camera->FollowNextPlayer)
76 {
77 // Can't follow if there are no players
78 if (GetNumPlayers(PLAYER_ALIVE_OR_DYING, false, false) == 0)
79 {
80 return;
81 }
82 camera->spectateMode = SPECTATE_FOLLOW;
83 // Find index of player
84 int playerIndex = -1;
85 CA_FOREACH(const PlayerData, p, gPlayerDatas)
86 if (p->UID == camera->FollowPlayerUID)
87 {
88 playerIndex = _ca_index;
89 break;
90 }
91 CA_FOREACH_END()
92 // Get the next player by index that has an actor in the game
93 const int d = (cmd & CMD_BUTTON1) ? 1 : -1;
94 for (int i = playerIndex + d;; i += d)
95 {
96 i = CLAMP_OPPOSITE(i, 0, (int)gPlayerDatas.size - 1);
97 // Check if clamping made us hit the termination condition
98 if (i == playerIndex) break;
99 const PlayerData *p = CArrayGet(&gPlayerDatas, i);
100 if (IsPlayerAliveOrDying(p))
101 {
102 // Follow this player
103 camera->FollowPlayerUID = p->UID;
104 camera->FollowNextPlayer = false;
105 break;
106 }
107 }
108 }
109}

117static void FollowPlayer(Vec2i *pos, const int playerUID);
118static void DoBuffer(
119 DrawBuffer *b, Vec2i center, int w, Vec2i noise, Vec2i offset);
120void CameraDraw(Camera *camera)
121{
122 Vec2i centerOffset = Vec2iZero();
123 const PlayerData *firstPlayer = NULL;
124 const int numPlayersScreen = GetNumPlayersScreen(&firstPlayer);
125 const int w = gGraphicsDevice.cachedConfig.Res.x;
126 const int h = gGraphicsDevice.cachedConfig.Res.y;
127
128 const Vec2i noise = ScreenShakeGetDelta(camera->shake);
129
130 GraphicsResetBlitClip(&gGraphicsDevice);
131 if (numPlayersScreen == 0)
132 {
133 // Try to spectate if there are remote players
134 if (camera->spectateMode == SPECTATE_NONE)
135 {
136 // Enter spectator mode
137 // Free-look mode
138 camera->spectateMode = SPECTATE_FREE;
139 // If there are remote players, follow them
140 CA_FOREACH(const PlayerData, p, gPlayerDatas)
141 if (p->Lives > 0 && !p->IsLocal)
142 {
143 camera->spectateMode = SPECTATE_FOLLOW;
144 camera->FollowPlayerUID = p->UID;
145 break;
146 }
147 CA_FOREACH_END()
148 }
149 if (camera->spectateMode == SPECTATE_FOLLOW)
150 {
151 FollowPlayer(&camera->lastPosition, camera->FollowPlayerUID);
152 }
153 DoBuffer(
154 &camera->Buffer,
155 camera->lastPosition,
156 X_TILES, noise, centerOffset);
157 SoundSetEars(camera->lastPosition);
158 }
159 else
160 {
161 // Don't spectate
162 camera->spectateMode = SPECTATE_NONE;
163 // Redo LOS if PVP, so that each split screen has its own LOS
164 if (IsPVP(gCampaign.Entry.Mode) && numPlayersScreen > 0)
165 {
166 LOSReset(&gMap.LOS);
167 }
168 const bool onePlayer = numPlayersScreen == 1;
169 const bool singleScreen = CameraIsSingleScreen();
170 if (onePlayer || singleScreen)
171 {
172 // Single camera screen
173 if (onePlayer)
174 {
175 const TActor *p = ActorGetByUID(firstPlayer->ActorUID);
176 camera->lastPosition = Vec2iNew(p->tileItem.x, p->tileItem.y);
177 }
178 else if (singleScreen)
179 {
180 // One screen
181 camera->lastPosition = PlayersGetMidpoint();
182 }
183
184 // Special case: if map is smaller than screen, center the camera
185 // However, it is important to keep the ear positions unmodified
186 // so that sounds don't get muffled just because there's a wall
187 // between player and camera center
188 const Vec2i earPos = camera->lastPosition;
189 if (gMap.Size.x * TILE_WIDTH < gGraphicsDevice.cachedConfig.Res.x)
190 {
191 camera->lastPosition.x = gMap.Size.x * TILE_WIDTH / 2;
192 }
193 if (gMap.Size.y * TILE_HEIGHT < gGraphicsDevice.cachedConfig.Res.y)
194 {
195 camera->lastPosition.y = gMap.Size.y * TILE_HEIGHT / 2;
196 }
197
198 // Redo LOS for every local human player
199 if (IsPVP(gCampaign.Entry.Mode))
200 {
201 CA_FOREACH(const PlayerData, p, gPlayerDatas)
202 if (!p->IsLocal || !IsPlayerAliveOrDying(p) ||
203 !IsPlayerHuman(p))
204 {
205 continue;
206 }
207 const TActor *a = ActorGetByUID(p->ActorUID);
208 LOSCalcFrom(
209 &gMap,
210 Vec2iToTile(Vec2iNew(a->tileItem.x, a->tileItem.y)),
211 false);
212 CA_FOREACH_END()
213 }
214
215 DoBuffer(
216 &camera->Buffer,
217 camera->lastPosition,
218 X_TILES, noise, centerOffset);
219 SoundSetEars(earPos);
220 }
221 else if (numPlayersScreen == 2)
222 {
223 // side-by-side split
224 int idx = 0;
225 for (int i = 0; i < (int)gPlayerDatas.size; i++, idx++)
226 {
227 const PlayerData *p = CArrayGet(&gPlayerDatas, i);
228 if (!IsPlayerScreen(p))
229 {
230 idx--;
231 continue;
232 }
233 const TActor *a = ActorGetByUID(p->ActorUID);
234 camera->lastPosition = Vec2iNew(a->tileItem.x, a->tileItem.y);
235 Vec2i centerOffsetPlayer = centerOffset;
236 int clipLeft = (idx & 1) ? w / 2 : 0;
237 int clipRight = (idx & 1) ? w - 1 : (w / 2) - 1;
238 GraphicsSetBlitClip(
239 &gGraphicsDevice, clipLeft, 0, clipRight, h - 1);
240 if (idx == 1)
241 {
242 centerOffsetPlayer.x += w / 2;
243 }
244
245 LOSCalcFrom(&gMap, Vec2iToTile(camera->lastPosition), false);
246 DoBuffer(
247 &camera->Buffer,
248 camera->lastPosition,
249 X_TILES_HALF, noise, centerOffsetPlayer);
250 SoundSetEarsSide(idx == 0, camera->lastPosition);
251 }
252 Draw_Line(w / 2 - 1, 0, w / 2 - 1, h - 1, colorBlack);
253 Draw_Line(w / 2, 0, w / 2, h - 1, colorBlack);
254 }
255 else if (numPlayersScreen >= 3 && numPlayersScreen <= 4)
256 {
257 // 4 player split screen
258 int idx = 0;
259 bool isLocalPlayerAlive[4];
260 memset(isLocalPlayerAlive, 0, sizeof isLocalPlayerAlive);
261 for (int i = 0; i < (int)gPlayerDatas.size; i++, idx++)
262 {
263 const PlayerData *p = CArrayGet(&gPlayerDatas, i);
264 if (!IsPlayerScreen(p))
265 {
266 idx--;
267 continue;
268 }
269 Vec2i centerOffsetPlayer = centerOffset;
270 const int clipLeft = (idx & 1) ? w / 2 : 0;
271 const int clipTop = (idx < 2) ? 0 : h / 2 - 1;
272 const int clipRight = (idx & 1) ? w - 1 : (w / 2) - 1;
273 const int clipBottom = (idx < 2) ? h / 2 : h - 1;
274 isLocalPlayerAlive[idx] = IsPlayerAliveOrDying(p);
275 if (!isLocalPlayerAlive[idx])
276 {
277 continue;
278 }
279 const TActor *a = ActorGetByUID(p->ActorUID);
280 camera->lastPosition = Vec2iNew(a->tileItem.x, a->tileItem.y);
281 GraphicsSetBlitClip(
282 &gGraphicsDevice,
283 clipLeft, clipTop, clipRight, clipBottom);
284 if (idx & 1)
285 {
286 centerOffsetPlayer.x += w / 2;
287 }
288 if (idx < 2)
289 {
290 centerOffsetPlayer.y -= h / 4;
291 }
292 else
293 {
294 centerOffsetPlayer.y += h / 4;
295 }
296 LOSCalcFrom(&gMap, Vec2iToTile(camera->lastPosition), false);
297 DoBuffer(
298 &camera->Buffer,
299 camera->lastPosition,
300 X_TILES_HALF, noise, centerOffsetPlayer);
301
302 // Set the sound "ears"
303 const bool isLeft = idx == 0 || idx == 2;
304 const bool isUpper = idx <= 2;
305 SoundSetEar(isLeft, isUpper ? 0 : 1, camera->lastPosition);
306 // If any player is dead, that ear reverts to the other ear
307 // of the same side of the remaining player
308 const int otherIdxOnSameSide = idx ^ 2;
309 if (!isLocalPlayerAlive[otherIdxOnSameSide])
310 {
311 SoundSetEar(
312 isLeft, !isUpper ? 0 : 1, camera->lastPosition);
313 }
314 else if (!isLocalPlayerAlive[3 - idx] &&
315 !isLocalPlayerAlive[3 - otherIdxOnSameSide])
316 {
317 // If both players of one side are dead,
318 // those ears revert to any of the other remaining
319 // players
320 SoundSetEarsSide(!isLeft, camera->lastPosition);
321 }
322 }
323 Draw_Line(w / 2 - 1, 0, w / 2 - 1, h - 1, colorBlack);
324 Draw_Line(w / 2, 0, w / 2, h - 1, colorBlack);
325 Draw_Line(0, h / 2 - 1, w - 1, h / 2 - 1, colorBlack);
326 Draw_Line(0, h / 2, w - 1, h / 2, colorBlack);
327 }
328 else
329 {
330 assert(0 && "not implemented yet");
331 }
332 }
333 GraphicsResetBlitClip(&gGraphicsDevice);
334}
335// Try to follow a player
336static void FollowPlayer(Vec2i *pos, const int playerUID)
337{
338 const PlayerData *p = PlayerDataGetByUID(playerUID);
339 if (p == NULL) return;
340 const TActor *a = ActorGetByUID(p->ActorUID);
341 if (a == NULL) return;
342 *pos = Vec2iFull2Real(a->Pos);
343}
344static void DoBuffer(
345 DrawBuffer *b, Vec2i center, int w, Vec2i noise, Vec2i offset)
346{
347 DrawBufferSetFromMap(b, &gMap, Vec2iAdd(center, noise), w);
348 if (gPlayerDatas.size > 0)
349 {
350 DrawBufferFix(b);
351 }
352 DrawBufferDraw(b, offset, NULL);
353}

355void CameraDrawMode(const Camera *camera)
356{
357 // Draw camera mode
358 char cameraNameBuf[256];
359 bool drawCameraMode = false;
360 switch (camera->spectateMode)
361 {
362 case SPECTATE_NONE:
363 // do nothing
364 break;
365 case SPECTATE_FOLLOW:
366 {
367 const PlayerData *p = PlayerDataGetByUID(camera->FollowPlayerUID);
368 if (p == NULL) break;
369 sprintf(cameraNameBuf, "Following %s", p->name);
370 drawCameraMode = true;
371 }
372 break;
373 case SPECTATE_FREE:
374 strcpy(cameraNameBuf, "Free-look Mode");
375 drawCameraMode = true;
376 break;
377 default:
378 CASSERT(false, "Unknown spectate mode");
379 break;
380 }
381 if (!drawCameraMode)
382 {
383 return;
384 }
385
386 const int w = gGraphicsDevice.cachedConfig.Res.x;
387 const int h = gGraphicsDevice.cachedConfig.Res.y;
388
389 // Draw the message centered at the bottom
390 FontStrMask(
391 cameraNameBuf,
392 Vec2iNew((w - FontStrW(cameraNameBuf)) / 2, h - FontH() * 2),
393 colorYellow);
394
395 // Show camera controls
396 const PlayerData *p = GetFirstPlayer(false, true, true);
397 // Use default keyboard controls
398 input_device_e inputDevice = INPUT_DEVICE_KEYBOARD;
399 int deviceIndex = 0;
400 if (p != NULL)
401 {
402 inputDevice = p->inputDevice;
403 deviceIndex = p->deviceIndex;
404 }
405 Vec2i pos = Vec2iNew(
406 (w - FontStrW("foo/bar to follow player, baz to free-look")) / 2,
407 h - FontH());
408 char buf[256];
409 color_t c = colorYellow;
410 InputGetButtonNameColor(
411 inputDevice, deviceIndex, CMD_BUTTON1, buf, &c);
412 pos = FontStrMask(buf, pos, c);
413 pos = FontStrMask("/", pos, colorYellow);
414 c = colorYellow;
415 InputGetButtonNameColor(
416 inputDevice, deviceIndex, CMD_BUTTON2, buf, &c);
417 pos = FontStrMask(buf, pos, c);
418 pos = FontStrMask(" to follow player, ", pos, colorYellow);
419 InputGetDirectionNames(buf, inputDevice, deviceIndex);
420 pos = FontStrMask(buf, pos, colorYellow);
421 FontStrMask(" to free-look", pos, colorYellow);
422}

424bool CameraIsSingleScreen(void)
425{
426 if (ConfigGetEnum(&gConfig, "Interface.Splitscreen") == SPLITSCREEN_ALWAYS)
427 {
428 return false;
429 }
430 // Do split screen for PVP, unless whole map fits on camera, or there
431 // are no human players alive or dying, then just do single screen
432 const bool mapFitsInScreen =
433 gMap.Size.x * TILE_WIDTH < gGraphicsDevice.cachedConfig.Res.x &&
434 gMap.Size.y * TILE_HEIGHT < gGraphicsDevice.cachedConfig.Res.y;
435 if (IsPVP(gCampaign.Entry.Mode) &&
436 !mapFitsInScreen &&
437 GetNumPlayers(PLAYER_ALIVE_OR_DYING, true, true) > 0)
438 {
439 return false;
440 }
441 // Otherwise, if we are forcing never splitscreen, use single screen
442 // regardless of whether the players are within camera range
443 if (ConfigGetEnum(&gConfig, "Interface.Splitscreen") == SPLITSCREEN_NEVER)
444 {
445 return true;
446 }
447 // Finally, use split screen if players don't fit on camera
448 Vec2i min;
449 Vec2i max;
450 PlayersGetBoundingRectangle(&min, &max);
451 return
452 max.x - min.x < gGraphicsDevice.cachedConfig.Res.x - CAMERA_SPLIT_PADDING &&
453 max.y - min.y < gGraphicsDevice.cachedConfig.Res.y - CAMERA_SPLIT_PADDING;
454}
