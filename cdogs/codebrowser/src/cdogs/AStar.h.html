31#ifndef AStar_h
32#define AStar_h
33
34#include <stdlib.h>
35
36typedef struct __ASNeighborList *ASNeighborList;
37typedef struct __ASPath *ASPath;
38
39typedef struct {
40 size_t nodeSize; // the size of the structure being used for the nodes - important since nodes are copied into the resulting path
41 void (*nodeNeighbors)(ASNeighborList neighbors, void *node, void *context); // add nodes to the neighbor list if they are connected to this node
42 float (*pathCostHeuristic)(void *fromNode, void *toNode, void *context); // estimated cost to transition from the first node to the second node -- optional, uses 0 if not specified
43 int (*earlyExit)(size_t visitedCount, void *visitingNode, void *goalNode, void *context); // early termination, return 1 for success, -1 for failure, 0 to continue searching -- optional
44 int (*nodeComparator)(void *node1, void *node2, void *context); // must return a sort order for the nodes (-1, 0, 1) -- optional, uses memcmp if not specified
45} ASPathNodeSource;
46
47// use in the nodeNeighbors callback to return neighbors
48void ASNeighborListAdd(ASNeighborList neighbors, void *node, float edgeCost);
49
50// if goalNode is NULL, it searches the entire graph and returns the cheapest deepest path
51// context is optional and is simply passed through to the callback functions
52// startNode and nodeSource is required
53// as a path is created, the relevant nodes are copied into the path
54ASPath ASPathCreate(const ASPathNodeSource *nodeSource, void *context, void *startNode, void *goalNode);
55
56// paths created with ASPathCreate() must be destroyed or else it will leak memory
57void ASPathDestroy(ASPath path);
58
59// if you want to make a copy of a path result, this function will do the job
60// you must call ASPathDestroy() with the resulting path to clean it up or it will cause a leak
61ASPath ASPathCopy(ASPath path);
62
63// fetches the number of nodes in the path
64size_t ASPathGetCount(ASPath path);
65
66// returns a pointer to the given node in the path
67void *ASPathGetNode(ASPath path, size_t index);
68
69#endif
