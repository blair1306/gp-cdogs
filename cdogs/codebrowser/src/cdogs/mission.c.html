102void MissionInit(Mission *m)
103{
104 memset(m, 0, sizeof *m);
105 // Initialise with default styles
106 strcpy(m->WallStyle, IntWallStyle(0));
107 strcpy(m->FloorStyle, IntFloorStyle(0));
108 strcpy(m->RoomStyle, IntRoomStyle(0));
109 strcpy(m->ExitStyle, IntExitStyle(0));
110 strcpy(m->KeyStyle, IntKeyStyle(0));
111 strcpy(m->DoorStyle, IntDoorStyle(0));
112 m->WallMask = colorBattleshipGrey;
113 m->FloorMask = colorGravel;
114 m->RoomMask = colorDoveGray;
115 m->AltMask = colorOfficeGreen;
116 CArrayInit(&m->Objectives, sizeof(Objective));
117 CArrayInit(&m->Enemies, sizeof(int));
118 CArrayInit(&m->SpecialChars, sizeof(int));
119 CArrayInit(&m->MapObjectDensities, sizeof(MapObjectDensity));
120 CArrayInit(&m->Weapons, sizeof(const GunDescription *));
121}
122void MissionCopy(Mission *dst, const Mission *src)
123{
124 if (src == NULL)
125 {
126 return;
127 }
128 MissionTerminate(dst);
129 MissionInit(dst);
130 if (src->Title)
131 {
132 CSTRDUP(dst->Title, src->Title);
133 }
134 if (src->Description)
135 {
136 CSTRDUP(dst->Description, src->Description);
137 }
138 dst->Type = src->Type;
139 dst->Size = src->Size;
140
141 strcpy(dst->WallStyle, src->WallStyle);
142 strcpy(dst->FloorStyle, src->FloorStyle);
143 strcpy(dst->RoomStyle, src->RoomStyle);
144 strcpy(dst->ExitStyle, src->ExitStyle);
145 strcpy(dst->KeyStyle, src->KeyStyle);
146 strcpy(dst->DoorStyle, src->DoorStyle);
147
148 CA_FOREACH(const Objective, srco, src->Objectives)
149 Objective dsto;
150 ObjectiveCopy(&dsto, srco);
151 CArrayPushBack(&dst->Objectives, &dsto);
152 CA_FOREACH_END()
153 CArrayCopy(&dst->Enemies, &src->Enemies);
154 CArrayCopy(&dst->SpecialChars, &src->SpecialChars);
155 CArrayCopy(&dst->MapObjectDensities, &src->MapObjectDensities);
156
157 dst->EnemyDensity = src->EnemyDensity;
158 CArrayCopy(&dst->Weapons, &src->Weapons);
159
160 memcpy(dst->Song, src->Song, sizeof dst->Song);
161
162 dst->WallMask = src->WallMask;
163 dst->FloorMask = src->FloorMask;
164 dst->RoomMask = src->RoomMask;
165 dst->AltMask = src->AltMask;
166
167 switch (dst->Type)
168 {
169 case MAPTYPE_STATIC:
170 CArrayCopy(&dst->u.Static.Tiles, &src->u.Static.Tiles);
171 CArrayCopy(&dst->u.Static.Items, &src->u.Static.Items);
172 CArrayCopy(&dst->u.Static.Characters, &src->u.Static.Characters);
173 CArrayCopy(&dst->u.Static.Objectives, &src->u.Static.Objectives);
174 CArrayCopy(&dst->u.Static.Keys, &src->u.Static.Keys);
175
176 dst->u.Static.Start = src->u.Static.Start;
177 dst->u.Static.Exit = src->u.Static.Exit;
178 break;
179 default:
180 memcpy(&dst->u, &src->u, sizeof dst->u);
181 break;
182 }
183}
184void MissionTerminate(Mission *m)
185{
186 if (m == NULL) return;
187 CFREE(m->Title);
188 CFREE(m->Description);
189 CA_FOREACH(Objective, o, m->Objectives)
190 ObjectiveTerminate(o);
191 CA_FOREACH_END()
192 CArrayTerminate(&m->Objectives);
193 CArrayTerminate(&m->Enemies);
194 CArrayTerminate(&m->SpecialChars);
195 CArrayTerminate(&m->MapObjectDensities);
196 CArrayTerminate(&m->Weapons);
197 switch (m->Type)
198 {
199 case MAPTYPE_CLASSIC:
200 break;
201 case MAPTYPE_STATIC:
202 CArrayTerminate(&m->u.Static.Tiles);
203 CArrayTerminate(&m->u.Static.Items);
204 CArrayTerminate(&m->u.Static.Characters);
205 CArrayTerminate(&m->u.Static.Objectives);
206 CArrayTerminate(&m->u.Static.Keys);
207 break;
208 case MAPTYPE_CAVE:
209 break;
210 default:
211 CASSERT(false, "unknown map type");
212 break;
213 }
214 memset(m, 0, sizeof *m);
215}

222static void SetupBadguysForMission(Mission *mission)
223{
224 CharacterStore *s = &gCampaign.Setting.characters;
225
226 CharacterStoreResetOthers(s);
227
228 if (s->OtherChars.size == 0)
229 {
230 return;
231 }
232
233 CA_FOREACH(const Objective, o, mission->Objectives)
234 if (o->Type == OBJECTIVE_RESCUE)
235 {
236 CharacterStoreAddPrisoner(s, o->u.Index);
237 break; // TODO: multiple prisoners
238 }
239 CA_FOREACH_END()
240
241 CA_FOREACH(int, e, mission->Enemies)
242 CharacterStoreAddBaddie(s, *e);
243 CA_FOREACH_END()
244
245 CA_FOREACH(int, sc, mission->SpecialChars)
246 CharacterStoreAddSpecial(s, *sc);
247 CA_FOREACH_END()
248}

281void MissionSetMessageIfComplete(struct MissionOptions *options)
282{
283 if (!gCampaign.IsClient)
284 {
285 if (CanCompleteMission(options))
286 {
287 GameEvent msg = GameEventNew(GAME_EVENT_MISSION_COMPLETE);
288 msg.u.MissionComplete = NMakeMissionComplete(options, &gMap);
289 GameEventsEnqueue(&gGameEvents, msg);
290 }
291 else if (options->HasBegun && gCampaign.Entry.Mode == GAME_MODE_NORMAL)
292 {
293 // Check if the game is impossible to end
294 // i.e. not enough rescue objectives left alive
295 CA_FOREACH(const Objective, o, options->missionData->Objectives)
296 if (o->Type == OBJECTIVE_RESCUE)
297 {
298 if (ObjectiveActorsAlive(_ca_index) < o->Required)
299 {
300 GameEvent e = GameEventNew(GAME_EVENT_MISSION_END);
301 e.u.MissionEnd.Delay = GAME_OVER_DELAY;
302 strcpy(e.u.MissionEnd.Msg, "Mission failed");
303 GameEventsEnqueue(&gGameEvents, e);
304 }
305 }
306 CA_FOREACH_END()
307 }
308 }
309}
310// Get the number of actors alive for an objective
311static int ObjectiveActorsAlive(const int objective)
312{
313 int count = 0;
314 CA_FOREACH(const TActor, a, gActors)
315 if (a->isInUse && a->health > 0 &&
316 ObjectiveFromTileItem(a->tileItem.flags) == objective)
317 {
318 count++;
319 }
320 CA_FOREACH_END()
321 return count;
322}

332void UpdateMissionObjective(
333 const struct MissionOptions *options,
334 const int flags, const ObjectiveType type, const int count)
335{
336 if (!(flags & TILEITEM_OBJECTIVE))
337 {
338 return;
339 }
340 const int idx = ObjectiveFromTileItem(flags);
341 const Objective *o = CArrayGet(&options->missionData->Objectives, idx);
342 if (o->Type != type)
343 {
344 return;
345 }
346 if (!gCampaign.IsClient)
347 {
348 GameEvent e = GameEventNew(GAME_EVENT_OBJECTIVE_UPDATE);
349 e.u.ObjectiveUpdate.ObjectiveId = idx;
350 e.u.ObjectiveUpdate.Count = count;
351 GameEventsEnqueue(&gGameEvents, e);
352 }
353}

355bool MissionCanBegin(void)
356{
357 // Need at least two players to begin PVP
358 if (IsPVP(gCampaign.Entry.Mode))
359 {
360 return GetNumPlayers(PLAYER_ALIVE_OR_DYING, false, false) > 1;
361 }
362 // Otherwise, just one player will do
363 return GetNumPlayers(PLAYER_ALIVE_OR_DYING, false, false) > 0;
364}

366void MissionBegin(struct MissionOptions *m, const NGameBegin gb)
367{
368 m->HasBegun = true;
369 m->state = MISSION_STATE_PLAY;
370 MusicPlayGame(&gSoundDevice, gCampaign.Entry.Path, m->missionData->Song);
371 if (MusicGetStatus(&gSoundDevice) == MUSIC_NOLOAD)
372 {
373 // Display music error message for 2 seconds
374 GameEvent e = GameEventNew(GAME_EVENT_SET_MESSAGE);
375 strncat(
376 e.u.SetMessage.Message, MusicGetErrorMessage(&gSoundDevice),
377 sizeof e.u.SetMessage.Message - 1);
378 e.u.SetMessage.Ticks = FPS_FRAMELIMIT * 2;
379 GameEventsEnqueue(&gGameEvents, e);
380 }
381 m->time = gb.MissionTime;
382 m->pickupTime = 0;
383}

385bool CanCompleteMission(const struct MissionOptions *options)
386{
387 // Can't complete if not started yet
388 if (!options->HasBegun)
389 {
390 return false;
391 }
392
393 // Death is the only escape from PVP and quick play
394 if (IsPVP(gCampaign.Entry.Mode))
395 {
396 // If we're in deathmatch with 1 player only, never complete the game
397 // Instead we'll be showing a "waiting for players..." message
398 return
399 GetNumPlayers(PLAYER_ANY, false, false) > 1 &&
400 GetNumPlayers(PLAYER_ALIVE_OR_DYING, false, false) <= 1;
401 }
402 else if (gCampaign.Entry.Mode == GAME_MODE_QUICK_PLAY)
403 {
404 return GetNumPlayers(PLAYER_ALIVE_OR_DYING, false, false) == 0;
405 }
406
407 return MissionAllObjectivesComplete(options);
408}

421
422bool IsMissionComplete(const struct MissionOptions *mo)
423{
424 if (!CanCompleteMission(mo))
425 {
426 return false;
427 }
428
429 // Check if dogfight is complete
430 if (IsPVP(gCampaign.Entry.Mode) &&
431 GetNumPlayers(PLAYER_ALIVE_OR_DYING, false, false) <= 1)
432 {
433 // Also check that only one player has lives left
434 int numPlayersWithLives = 0;
435 CA_FOREACH(const PlayerData, p, gPlayerDatas)
436 if (p->Lives > 0) numPlayersWithLives++;
437 CA_FOREACH_END()
438 if (numPlayersWithLives <= 1)
439 {
440 return true;
441 }
442 }
443
444 if (!AllSurvivingPlayersInExit())
445 {
446 return false;
447 }
448
449 if (MoreRescuesNeeded(mo))
450 {
451 return false;
452 }
453
454 return true;
455}
456
457bool MissionNeedsMoreRescuesInExit(const struct MissionOptions *mo)
458{
459 return
460 CanCompleteMission(mo) &&
461 AllSurvivingPlayersInExit() &&
462 MoreRescuesNeeded(mo);
463}
464
465static bool AllSurvivingPlayersInExit(void)
466{
467 // Check that all surviving players are in exit zone
468 // Note: players are still in the exit area if they are dying there;
469 // this is the basis for the "resurrection penalty"
470 CA_FOREACH(const PlayerData, p, gPlayerDatas)
471 if (!IsPlayerAliveOrDying(p)) continue;
472 const TActor *player = ActorGetByUID(p->ActorUID);
473 if (!MapIsTileInExit(&gMap, &player->tileItem)) return false;
474 CA_FOREACH_END()
475 return true;
476}
477
478static bool MoreRescuesNeeded(const struct MissionOptions *mo)
479{
480 int rescuesRequired = 0;
481 // Find number of rescues required
482 // TODO: support multiple rescue objectives
483 CA_FOREACH(const Objective, o, mo->missionData->Objectives)
484 if (o->Type == OBJECTIVE_RESCUE)
485 {
486 rescuesRequired = o->Required;
487 break;
488 }
489 CA_FOREACH_END()
490 // Check that enough prisoners are in exit zone
491 if (rescuesRequired > 0)
492 {
493 int prisonersRescued = 0;
494 CA_FOREACH(const TActor, a, gActors)
495 if (!a->isInUse) continue;
496 if (CharacterIsPrisoner(&gCampaign.Setting.characters, ActorGetCharacter(a)) &&
497 MapIsTileInExit(&gMap, &a->tileItem))
498 {
499 prisonersRescued++;
500 }
501 CA_FOREACH_END()
502 if (prisonersRescued < rescuesRequired)
503 {
504 return true;
505 }
506 }
507 return false;
508}

517int KeycardCount(int flags)
518{
519 int count = 0;
520 if (flags & FLAGS_KEYCARD_RED) count++;
521 if (flags & FLAGS_KEYCARD_BLUE) count++;
522 if (flags & FLAGS_KEYCARD_GREEN) count++;
523 if (flags & FLAGS_KEYCARD_YELLOW) count++;
524 return count;
