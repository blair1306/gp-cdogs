75{
76 const char *dot = strrchr(filename, '.');
77 if (!dot || dot == filename)

79 return "";
80 }
81 return dot + 1;

84// Note: includes trailing slash
85void PathGetDirname(char *buf, const char *path)
86{
87 const char *basename = PathGetBasename(path);
88 if (basename == path)
89 {
90 strcpy(buf, "");
91 }
92 else
93 {
94 strncpy(buf, path, basename - path);
95 buf[basename - path] = '\0';
96 }
97}
98const char *PathGetBasename(const char *path)
99{
100 const char *fslash = strrchr(path, '/');
101 const char *bslash = strrchr(path, '\\');
102 const char *slash =
103 fslash ? (bslash ? MAX(fslash, bslash) : fslash) : bslash;
104 if (slash == NULL)
105 {
106 // no slashes found, simply return path
107 return path;
108 }
109 else
110 {
111 return slash + 1;
112 }
113}
114void PathGetWithoutExtension(char *buf, const char *path)
115{
116 const char *dot = strrchr(path, '.');
117 if (dot)
118 {
119 strncpy(buf, path, dot - path);
120 buf[dot - path] = '\0';

122 else
123 {
124 strcpy(buf, path);
125 }
126}
127void PathGetBasenameWithoutExtension(char *buf, const char *path)
128{
129 const char *basename = PathGetBasename(path);
130 PathGetWithoutExtension(buf, basename);
131}

137static bool IsAbsolutePath(const char *path);
138void RealPath(const char *src, char *dest)
139{
140 char *res;
141#ifndef _WIN32
142 // realpath will fail if the file does not exist; if this is the
143 // case then resolve the path ourselves
144 tinydir_file file;
145 const bool exists = tinydir_file_open(&file, src) == 0;
146 if (!exists)
147 {
148 char srcBuf[CDOGS_PATH_MAX];
149 // First, convert slashes
150 srcBuf[0] = '\0';
151 strncat(srcBuf, src, CDOGS_PATH_MAX - 1);
152 for (char *c = srcBuf; *c != '\0'; c++)
153 {
154 if (*c == '\\')
155 {
156 *c = '/';
157 }
158 }
159
160 // Then, add on the CWD if the path is not absolute
161 char resolveBuf[CDOGS_PATH_MAX];
162 resolveBuf[0] = '\0';
163 if (!IsAbsolutePath(srcBuf))
164 {
165 CDogsGetCWD(resolveBuf);
166 strcat(resolveBuf, "/");
167 }
168 // Add the path
169 strcat(resolveBuf, srcBuf);
170
171 // Finally, resolve the path one level at a time, ignoring '//'s, '.'s
172 // and resolving '..'s to the parent level
173 char *cOut = dest;
174 char cLast = '\0';
175 for (const char *c = resolveBuf; *c != '\0'; c++)
176 {
177 if (*c == '.' && (cLast == '.' || cLast == '/'))
178 {
179 if (cLast == '.')
180 {
181 // '..' parent dir
182 // Rewind the out ptr to the last path separator
183 if (cOut > dest + 1)
184 {
185 // Skip past the last slash
186 cOut -= 2;
187 while (*cOut != '/' && cOut > dest)
188 {
189 cOut--;
190 }
191 // Go back in front of the slash
192 if (*cOut == '/')
193 {
194 cOut++;
195 }
196 }
197 }
198 else if (cLast == '/')
199 {
200 // ignore for now
201 }
202 }
203 else if (*c == '/' && (cLast == '/' || cLast == '.'))
204 {
205 // Double slash; ignore
206 }
207 else
208 {
209 *cOut = *c;
210 cOut++;
211 }
212 cLast = *c;
213 }
214 // Write terminating char
215 *cOut = '\0';
216 res = dest;
217 }
218 else
219#endif
220 res = realpath(src, dest);
221 if (!res)
222 {
223 fprintf(stderr, "Cannot resolve relative path %s: %s\n",
224 src, strerror(errno));
225 // Default to relative path
226 strcpy(dest, src);
227 }
228 // Convert \'s to /'s (for consistency)
229 for (char *c = dest; *c != '\0'; c++)
230 {
231 if (*c == '\\')
232 {
233 *c = '/';
234 }
235 }
236}

238// Convert an absolute path to a relative path
239// e.g. /a/path/from/here, /a/path/to -> ../to
240static void TrimSlashes(char *s);
241void RelPath(char *buf, const char *to, const char *from)
242{
243 // Make sure both to/from paths were generated using RealPath,
244 // so that common substring means they share part of their paths,
245 // and not due to alternate renderings of the same path like
246 // different path separator chars, or relative paths.
247 char toBuf[CDOGS_PATH_MAX];
248 char fromBuf[CDOGS_PATH_MAX];
249 RealPath(to, toBuf);
250 RealPath(from, fromBuf);
251
252 // Remove trailing slashes
253 TrimSlashes(toBuf);
254 TrimSlashes(fromBuf);
255
256 // First, find common string prefix
257 const char *t = toBuf;
258 const char *f = fromBuf;
259 const char *tSlash = t;
260 const char *fSlash = f;
261 while (*t && *f)
262 {
263 if (*t != *f)
264 {
265 break;
266 }
267 if (*t == '/')
268 {
269 tSlash = t;
270 fSlash = f;
271 }
272 t++;
273 f++;
274 }
275 // e.g.: tSlash = "/to", fSlash = "/from/here"
276
277 // Check if one is a complete substring of the other
278 if (!*f)
279 {
280 tSlash = t;
281 fSlash = f;
282 }
283
284 // For every folder in "from", add "..", and finally add "to"
285 strcpy(buf, "");
286 while (*fSlash)
287 {
288 if (*fSlash == '/')
289 {
290 strcat(buf, "../");
291 }
292 fSlash++;
293 }
294 if (*tSlash == '/')
295 {
296 tSlash++;
297 }
298 strcat(buf, tSlash);
299}
300static void TrimSlashes(char *s)
301{
302 char *end = s + strlen(s) - 1;
303 while (end > s && *end == '/')
304 {
305 end--;
306 }
307 *(end + 1) = '\0';
308}

313char *CDogsGetCWD(char *buf)
314{
315#ifdef __APPLE__
316 uint32_t size = CDOGS_PATH_MAX;
317 if (_NSGetExecutablePath(buf, &size))
318 {
319 return NULL;
320 }
321 // This gives us the executable path; find the dirname
322 *strrchr(buf, '/') = '\0';
323 return buf;
324#else
325 return getcwd(buf, CDOGS_PATH_MAX);
326#endif
327}
328void RelPathFromCWD(char *buf, const char *to)
329{
330 if (to == NULL || strlen(to) == 0)
331 {
332 return;
333 }
334 char cwd[CDOGS_PATH_MAX];
335 CASSERT(CDogsGetCWD(cwd) != NULL, "error");
336 if (CDogsGetCWD(cwd) == NULL)
337 {
338 fprintf(stderr, "Error getting CWD; %s\n", strerror(errno));
339 strcpy(buf, to);
340 }
341 else
342 {
343 RelPath(buf, to, cwd);
344 }
345}

347void GetDataFilePath(char *buf, const char *path)
348{
349 char relbuf[CDOGS_PATH_MAX];
350 // Don't bother prepending CWD if data dir already an absolute path
351 if (IsAbsolutePath(CDOGS_DATA_DIR))
352 {
353 sprintf(relbuf, "%s%s", CDOGS_DATA_DIR, path);
354 }
355 else
356 {
357 char cwd[CDOGS_PATH_MAX];
358 if (CDogsGetCWD(cwd) == NULL)
359 {
360 fprintf(stderr, "Error getting CWD; %s\n", strerror(errno));
361 strcpy(cwd, "");
362 }
363 sprintf(relbuf, "%s/%s%s", cwd, CDOGS_DATA_DIR, path);
364 }
365 RealPath(relbuf, buf);
366}

368static bool IsAbsolutePath(const char *path)
369{
370#ifdef _WIN32
371 return strlen(path) > 1 && path[1] == ':';
372#else
373 return path[0] == '/';

445void CamelToTitle(char *buf, const char *src)
446{
447 const char *first = src;
448#define IS_UPPER(_x) ((_x) >= 'A' && (_x) <= 'Z')
449 while (*src)
450 {
451 // Word boundaries marked by capital letters, as long as:
452 // - It's not the first letter, and
453 // - The previous letter is lower case, or
454 // - The next letter is lower case
455 if (IS_UPPER(*src) &&
456 src != first &&
457 (!IS_UPPER(*(src - 1)) || (*(src + 1) && !IS_UPPER(*(src + 1)))))
458 {
459 *buf++ = ' ';
460 }
461 *buf++ = *src++;
462 }
463 *buf = '\0';
464}
465// From answer by plinth
466// http://stackoverflow.com/a/744822/2038264
467// License: http://creativecommons.org/licenses/by-sa/3.0/
468// Author profile: http://stackoverflow.com/users/20481/plinth
469bool StrEndsWith(const char *str, const char *suffix)
470{
471 if (str == NULL || suffix == NULL)
472 {
473 return false;
474 }
475 const size_t lenStr = strlen(str);
476 const size_t lenSuffix = strlen(suffix);
477 if (lenSuffix > lenStr)
478 {
479 return false;
480 }
481 return strncmp(str + lenStr - lenSuffix, suffix, lenSuffix) == 0;
482}
