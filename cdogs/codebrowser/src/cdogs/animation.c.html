55Animation AnimationGetActorAnimation(const ActorAnimation aa)
56{
57 switch (aa)
58 {
59 case ACTORANIMATION_IDLE: return animIdling;
60 case ACTORANIMATION_WALKING: return animWalking;
61 default: CASSERT(false, "Unknown actor animation"); return animIdling;
62 }
63}

65void AnimationUpdate(Animation *a, const int ticks)
66{
67 a->frameCounter += ticks;
68 a->newFrame = false;
69 if (a->frameCounter > a->ticksPerFrame[a->frame])
70 {
71 a->frameCounter -= a->ticksPerFrame[a->frame];
72 a->newFrame = true;
73 // Advance to next frame
74 if (a->randomFrames)
75 {
76 // If we're not on the first frame, return to first frame
77 // Otherwise, pick a random non-first frame
78 if (a->frame == 0)
79 {
80 // Note: -1 means frame not used, so pick another frame
81 do
82 {
83 a->frame = (rand() % (ANIMATION_MAX_FRAMES - 1)) + 1;
84 } while (a->ticksPerFrame[a->frame] < 0);
85 }
86 else
87 {
88 a->frame = 0;
89 }
90 }
91 else
92 {
93 a->frame++;
94 if (a->frame >= ANIMATION_MAX_FRAMES ||
95 a->ticksPerFrame[a->frame] < 0)
96 {
97 a->frame = 0;
98 }
99 }
100 }
101}
102int AnimationGetFrame(const Animation *a)
103{
104 return a->frames[a->frame];
105}
