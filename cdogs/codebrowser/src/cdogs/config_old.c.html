32bool ConfigIsOld(FILE *f)
33{
34 int x1, x2, x3, x4, x5, x6, x7;
35 // See if the first line is made of numbers
36 int fscanfres = fscanf(f, "%d %d %d %d %d %d %d\n",
37 &x1, &x2, &x3, &x4, &x5, &x6, &x7);
38 return fscanfres == 7;
39}
