65#define X_TILES_HALF ((X_TILES + 1) / 2)

71typedef enum
72{
73 MAPTILE_NO_WALK = 0x0001,
74 MAPTILE_NO_SEE = 0x0002,
75 MAPTILE_NO_SHOOT = 0x0004,
76 MAPTILE_IS_WALL = 0x0008,
77 MAPTILE_IS_NOTHING = 0x0010,
78 MAPTILE_IS_NORMAL_FLOOR = 0x0020,
79 MAPTILE_OFFSET_PIC = 0x0040,
80// These constants are used internally in draw, it is never set in the map
81 MAPTILE_DELAY_DRAW = 0x0080,
82 MAPTILE_OUT_OF_SIGHT = 0x0100
83} MapTileFlags;

103typedef struct
104{
105 int MobObjId;

107 {
108 struct
109 {
110 const CArray *Sprites;
111 direction_e Dir;
112 color_t Color;
113 } MuzzleFlash;
114 } u;
115} TileItemDrawFuncData;
116typedef void (*TileItemDrawFunc)(const Vec2i, const TileItemDrawFuncData *);
117typedef struct TileItem
118{
119 int x, y;
120 // Velocity in full coordinates
121 Vec2i VelFull;
122 Vec2i size;
123 TileItemKind kind;
124 int id; // Id of item (actor, mobobj or obj)
125 int flags;
126 TileItemGetPicFunc getPicFunc;
127 TileItemDrawFunc drawFunc;
128 TileItemDrawFuncData drawData;
129 CPic CPic;
130 GetDrawContextFunc CPicFunc;
131 Vec2i ShadowSize;
132 int SoundLock;
133} TTileItem;

151} Tile;
