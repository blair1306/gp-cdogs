26const char *IntWallType(const int i)
27{
28 static const char *wallTypes[] = {
29 "o", "w", "e", "n", "s", "nw", "ne", "sw", "se",
30 "wt", "et", "nt", "st", "v", "h", "x"
31 };
32 return wallTypes[i];
33}
34const char *IntWallStyle(const int i)
35{
36 static const char *wallStyles[] = {
37 "steel", "brick", "carbon", "steelwood", "stone", "plasteel",
38 "granite"
39 };
40 return wallStyles[abs(i) % WALL_STYLE_COUNT];
41}
42
43const char *IntTileType(const int i)
44{
45 static const char *tileTypes[] = { "shadow", "normal", "alt1", "alt2" };
46 return tileTypes[abs(i) % FLOOR_TYPES];
47}
48const char *IntFloorStyle(const int i)
49{
50 static const char *floorStyles[] = {
51 "recessed", "tile", "dirt", "flat", "striped", "smallsquare", "stone",
52 "wood", "biggrid", "grid"
53 };
54 return floorStyles[abs(i) % FLOOR_STYLE_COUNT];
55}
56// Note: room styles subtly different from floor styles
57const char *IntRoomStyle(const int i)
58{
59 static const char *roomStyles[] = {
60 "recessed", "tile", "dirt", "flat", "striped", "smallsquare", "stone",
61 "wood", "grid", "biggrid", "checker"
62 };
63 return roomStyles[abs(i) % ROOM_STYLE_COUNT];
64}
