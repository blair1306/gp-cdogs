62TActor *AIGetClosestPlayer(Vec2i fullpos)
63{
64 int minDistance = -1;
65 TActor *closestPlayer = NULL;
66 CA_FOREACH(const PlayerData, pd, gPlayerDatas)
67 if (!IsPlayerAlive(pd))
68 {
69 continue;
70 }
71 TActor *p = ActorGetByUID(pd->ActorUID);
72 const int distance = CHEBYSHEV_DISTANCE(
73 fullpos.x, fullpos.y, p->Pos.x, p->Pos.y);
74 if (!closestPlayer || distance < minDistance)
75 {
76 closestPlayer = p;
77 minDistance = distance;
78 }
79 CA_FOREACH_END()
80 return closestPlayer;
81}

83static TActor *AIGetClosestActor(
84 const Vec2i fromPos, const TActor *from,
85 bool (*compFunc)(const TActor *, const TActor *))
86{
87 // Search all the actors and find the closest one that
88 // satisfies the condition
89 TActor *closest = NULL;
90 int minDistance = -1;
91 CA_FOREACH(TActor, a, gActors)
92 if (!a->isInUse || a->dead)
93 {
94 continue;
95 }
96 // Never target invulnerables or civilians
97 if (a->flags & (FLAGS_INVULNERABLE | FLAGS_PENALTY))
98 {
99 continue;
100 }
101 if (compFunc(a, from))
102 {
103 int distance = CHEBYSHEV_DISTANCE(
104 fromPos.x, fromPos.y, a->Pos.x, a->Pos.y);
105 if (!closest || distance < minDistance)
106 {
107 minDistance = distance;
108 closest = a;
109 }
110 }
111 CA_FOREACH_END()
112 return closest;
113}

115static bool IsGood(const TActor *a, const TActor *b)
116{
117 UNUSED(b);
118 return a->PlayerUID >= 0 || (a->flags & FLAGS_GOOD_GUY);
119}
120static bool IsBad(const TActor *a, const TActor *b)
121{
122 return !IsGood(a, b);
123}
124static bool IsDifferent(const TActor *a, const TActor *b)
125{
126 return a != b;
127}
128const TActor *AIGetClosestEnemy(
129 const Vec2i from, const TActor *a, const int flags)
130{
131 if (IsPVP(gCampaign.Entry.Mode))
132 {
133 // free for all; look for anybody else
134 return AIGetClosestActor(from, a, IsDifferent);
135 }
136 else if ((!a || a->PlayerUID < 0) && !(flags & FLAGS_GOOD_GUY))
137 {
138 // we are bad; look for good guys
139 return AIGetClosestActor(from, a, IsGood);
140 }
141 else
142 {
143 // we are good; look for bad guys
144 return AIGetClosestActor(from, a, IsBad);
145 }
146}

148static bool IsGoodAndVisible(const TActor *a, const TActor *b)
149{
150 return IsGood(a, b) && (a->flags & FLAGS_VISIBLE);
151}
152static bool IsBadAndVisible(const TActor *a, const TActor *b)
153{
154 return IsBad(a, b) && (a->flags & FLAGS_VISIBLE);
155}
156const TActor *AIGetClosestVisibleEnemy(
157 const TActor *from, const bool isPlayer)
158{
159 if (IsPVP(gCampaign.Entry.Mode))
160 {
161 // free for all; look for anybody
162 return AIGetClosestActor(from->Pos, from, IsDifferent);
163 }
164 else if (!isPlayer && !(from->flags & FLAGS_GOOD_GUY))
165 {
166 // we are bad; look for good guys
167 return AIGetClosestActor(from->Pos, from, IsGoodAndVisible);
168 }
169 else
170 {
171 // we are good; look for bad guys
172 return AIGetClosestActor(from->Pos, from, IsBadAndVisible);
173 }
174}

189int AIReverseDirection(int cmd)
190{
191 if (cmd & (CMD_LEFT | CMD_RIGHT))
192 {
193 cmd ^= CMD_LEFT | CMD_RIGHT;
194 }
195 if (cmd & (CMD_UP | CMD_DOWN))
196 {
197 cmd ^= CMD_UP | CMD_DOWN;
198 }
199 return cmd;
200}

207bool AIHasClearPath(
208 const Vec2i from, const Vec2i to, const bool ignoreObjects)
209{
210 IsBlockedFunc f = ignoreObjects ? IsTileNoWalk : IsTileNoWalkAroundObjects;
211 return AIHasClearLine(Vec2iToTile(from), Vec2iToTile(to), f);

213static bool AIHasClearLine(
214 Vec2i from, Vec2i to, IsBlockedFunc isBlockedFunc)
215{
216 // Find all tiles that overlap with the line (from, to)
217 HasClearLineData data;
218 data.IsBlocked = isBlockedFunc;
219 data.data = &gMap;
220
221 return HasClearLineJMRaytrace(from, to, &data);
222}
223static bool IsTileWalkableOrOpenable(Map *map, Vec2i pos);
224bool IsTileWalkable(Map *map, const Vec2i pos)
225{
226 if (!IsTileWalkableOrOpenable(map, pos))
227 {
228 return false;
229 }
230 // Check if tile has a dangerous (explosive) item on it
231 // For AI, we don't want to shoot it, so just walk around
232 Tile *t = MapGetTile(map, pos);
233 CA_FOREACH(ThingId, tid, t->things)
234 // Only look for explosive objects
235 if (tid->Kind != KIND_OBJECT)
236 {
237 continue;
238 }
239 const TObject *o = CArrayGet(&gObjs, tid->Id);
240 if (ObjIsDangerous(o))
241 {
242 return false;
243 }
244 CA_FOREACH_END()
245 return true;
246}
247static bool IsTileNoWalk(void *data, const Vec2i pos)
248{
249 return !IsTileWalkable(data, pos);
250}
251bool IsTileWalkableAroundObjects(Map *map, const Vec2i pos)
252{
253 if (!IsTileWalkableOrOpenable(map, pos))
254 {
255 return false;
256 }
257 // Check if tile has any item on it
258 Tile *t = MapGetTile(map, pos);
259 CA_FOREACH(ThingId, tid, t->things)
260 if (tid->Kind == KIND_OBJECT)
261 {
262 // Check that the object has hitbox - i.e. health > 0
263 const TObject *o = CArrayGet(&gObjs, tid->Id);
264 if (o->Health > 0)
265 {
266 return false;
267 }
268 }
269 else if (tid->Kind == KIND_CHARACTER)
270 {
271 switch (gCollisionSystem.allyCollision)
272 {
273 case ALLYCOLLISION_NORMAL:
274 return false;
275 case ALLYCOLLISION_REPEL:
276 // TODO: implement
277 // Need to know collision team of player
278 // to know if collision will result in repelling
279 break;
280 case ALLYCOLLISION_NONE:
281 continue;
282 default:
283 CASSERT(false, "unknown collision type");
284 break;
285 }
286 }
287 CA_FOREACH_END()
288 return true;
289}
290static bool IsTileNoWalkAroundObjects(void *data, const Vec2i pos)
291{
292 return !IsTileWalkableAroundObjects(data, pos);
293}
294static bool IsTileWalkableOrOpenable(Map *map, Vec2i pos)
295{
296 const Tile *tile = MapGetTile(map, pos);
297 if (tile == NULL)
298 {
299 return false;
300 }
301 const int tileFlags = tile->flags;
302 if (!(tileFlags & MAPTILE_NO_WALK))
303 {
304 return true;
305 }
306 if (tileFlags & MAPTILE_OFFSET_PIC)
307 {
308 // A door; check if we can open it
309 int keycard = MapGetDoorKeycardFlag(map, pos);
310 if (!keycard)
311 {
312 // Unlocked door
313 return true;
314 }
315 return !!(keycard & gMission.KeyFlags);
316 }
317 // Otherwise, we cannot walk over this tile
318 return false;
319}
320static bool IsPosNoSee(void *data, Vec2i pos);
321bool AIHasClearShot(const Vec2i from, const Vec2i to)
322{
323 // Perform 4 line tests - above, below, left and right
324 // This is to account for possible positions for the muzzle
325 Vec2i fromOffset = from;
326
327 const int pad = 2;
328 fromOffset.x = from.x - (ACTOR_W + pad) / 2;
329 if (Vec2iToTile(fromOffset).x >= 0 &&
330 !AIHasClearLine(fromOffset, to, IsPosNoSee))
331 {
332 return false;
333 }
334 fromOffset.x = from.x + (ACTOR_W + pad) / 2;
335 if (Vec2iToTile(fromOffset).x < gMap.Size.x &&
336 !AIHasClearLine(fromOffset, to, IsPosNoSee))
337 {
338 return false;
339 }
340 fromOffset.x = from.x;
341 fromOffset.y = from.y - (ACTOR_H + pad) / 2;
342 if (Vec2iToTile(fromOffset).y >= 0 &&
343 !AIHasClearLine(fromOffset, to, IsPosNoSee))
344 {
345 return false;
346 }
347 fromOffset.y = from.y + (ACTOR_H + pad) / 2;
348 if (Vec2iToTile(fromOffset).y < gMap.Size.y &&
349 !AIHasClearLine(fromOffset, to, IsPosNoSee))
350 {
351 return false;
352 }
353 return true;
354}
355static bool IsPosNoSee(void *data, Vec2i pos)
356{
357 return MapGetTile(data, Vec2iToTile(pos))->flags & MAPTILE_NO_SEE;
358}

360TObject *AIGetObjectRunningInto(TActor *a, int cmd)
361{
362 // Check the position just in front of the character;
363 // check if there's a (non-dangerous) object in front of it
364 Vec2i frontPos = Vec2iFull2Real(a->Pos);
365 TTileItem *item;
366 if (cmd & CMD_LEFT)
367 {
368 frontPos.x--;
369 }
370 else if (cmd & CMD_RIGHT)
371 {
372 frontPos.x++;
373 }
374 if (cmd & CMD_UP)
375 {
376 frontPos.y--;
377 }
378 else if (cmd & CMD_DOWN)
379 {
380 frontPos.y++;
381 }
382 const CollisionParams params =
383 {
384 TILEITEM_IMPASSABLE, CalcCollisionTeam(true, a),
385 IsPVP(gCampaign.Entry.Mode)
386 };
387 item = OverlapGetFirstItem(
388 &a->tileItem, Vec2iReal2Full(frontPos), a->tileItem.size, params);
389 if (!item || item->kind != KIND_OBJECT)
390 {
391 return NULL;
392 }
393 return CArrayGet(&gObjs, item->id);
394}

396bool AIIsFacing(const TActor *a, const Vec2i targetFull, const direction_e d)
397{
398 const bool isUpperOrLowerOctants =
399 abs(a->Pos.x - targetFull.x) < abs(a->Pos.y - targetFull.y);
400 const bool isRight = a->Pos.x < targetFull.x;
401 const bool isAbove = a->Pos.y > targetFull.y;
402 switch (d)
403 {
404 case DIRECTION_UP:
405 return isAbove && isUpperOrLowerOctants;
406 case DIRECTION_UPLEFT:
407 return !isRight && isAbove;
408 case DIRECTION_LEFT:
409 return !isRight && !isUpperOrLowerOctants;
410 case DIRECTION_DOWNLEFT:
411 return !isRight && !isAbove;
412 case DIRECTION_DOWN:
413 return !isAbove && isUpperOrLowerOctants;
414 case DIRECTION_DOWNRIGHT:
415 return isRight && !isAbove;
416 case DIRECTION_RIGHT:
417 return isRight && !isUpperOrLowerOctants;
418 case DIRECTION_UPRIGHT:
419 return isRight && isAbove;
420 default:
421 CASSERT(false, "invalid direction");
422 break;
423 }
424 return false;
425}

428// Use pathfinding to check that there is a path between
429// source and destination tiles
430bool AIHasPath(const Vec2i from, const Vec2i to, const bool ignoreObjects)
431{
432 // Quick first test: check there is a clear path
433 if (AIHasClearPath(from, to, ignoreObjects))
434 {
435 return true;
436 }
437 // Pathfind
438 const Vec2i fromTile = Vec2iToTile(from);
439 const Vec2i toTile = MapSearchTileAround(
440 &gMap, Vec2iToTile(to),
441 ignoreObjects ? IsTileWalkable : IsTileWalkableAroundObjects);
442 CachedPath path = PathCacheCreate(
443 &gPathCache, fromTile, toTile, ignoreObjects, true);
444 const size_t pathCount = ASPathGetCount(path.Path);
445 CachedPathDestroy(&path);
446 return pathCount >= 1;
447}

449int AIGotoDirect(const Vec2i a, const Vec2i p)
450{
451 int cmd = 0;
452
453 if (a.x < p.x) cmd |= CMD_RIGHT;
454 else if (a.x > p.x) cmd |= CMD_LEFT;
455
456 if (a.y < p.y) cmd |= CMD_DOWN;
457 else if (a.y > p.y) cmd |= CMD_UP;
458
459 return cmd;
460}

462// Follow the current A* path
463static int AStarFollow(
464 AIGotoContext *c, Vec2i currentTile, TTileItem *i, Vec2i a)
465{
466 Vec2i *pathTile = ASPathGetNode(c->Path.Path, c->PathIndex);
467 c->IsFollowing = 1;
468 // Check if we need to follow the next step in the path
469 // Note: need to make sure the actor is fully within the current tile
470 // otherwise it may get stuck at corners
471 if (Vec2iEqual(currentTile, *pathTile) &&
472 IsTileItemInsideTile(i, currentTile))
473 {
474 c->PathIndex++;
475 pathTile = ASPathGetNode(c->Path.Path, c->PathIndex);
476 }
477 // Go directly to the center of the next tile
478 return AIGotoDirect(a, Vec2iCenterOfTile(*pathTile));
479}
480// Check that we are still close to the start of the A* path,
481// and the end of the path is close to our goal
482static int AStarCloseToPath(
483 AIGotoContext *c, Vec2i currentTile, Vec2i goalTile)
484{
485 Vec2i *pathTile;
486 Vec2i *pathEnd;
487 if (!c ||
488 c->PathIndex >= (int)ASPathGetCount(c->Path.Path) - 1) // at end of path
489 {
490 return 0;
491 }
492 // Check if we're too far from the current start of the path
493 pathTile = ASPathGetNode(c->Path.Path, c->PathIndex);
494 if (CHEBYSHEV_DISTANCE(
495 currentTile.x, currentTile.y, pathTile->x, pathTile->y) > 2)
496 {
497 return 0;
498 }
499 // Check if we're too far from the end of the path
500 pathEnd = ASPathGetNode(c->Path.Path, ASPathGetCount(c->Path.Path) - 1);
501 if (CHEBYSHEV_DISTANCE(
502 goalTile.x, goalTile.y, pathEnd->x, pathEnd->y) > 0)
503 {
504 return 0;
505 }
506 return 1;
507}
508int AIGoto(TActor *actor, Vec2i p, bool ignoreObjects)
509{
510 Vec2i a = Vec2iFull2Real(actor->Pos);
511 Vec2i currentTile = Vec2iToTile(a);
512 Vec2i goalTile = Vec2iToTile(p);
513 AIGotoContext *c = &actor->aiContext->Goto;
514
515 CASSERT(c != NULL, "no AI context");
516
517 // If we are already there, go directly to the goal
518 // This can happen if AI is trying to track the player,
519 // but the player has died, for example.
520 if (Vec2iEqual(currentTile, goalTile))
521 {
522 return AIGotoDirect(a, p);
523 }
524
525 // If we are currently following an A* path,
526 // and it is still valid, keep following it until
527 // we have reached a new tile
528 if (c && c->IsFollowing && AStarCloseToPath(c, currentTile, goalTile))
529 {
530 return AStarFollow(c, currentTile, &actor->tileItem, a);
531 }
532 else if (AIHasClearPath(a, p, ignoreObjects))
533 {
534 // Simple case: if there's a clear line between AI and target,
535 // walk straight towards it
536 return AIGotoDirect(a, p);
537 }
538 else
539 {
540 // We need to recalculate A*
541
542 // First, if the goal tile is blocked itself,
543 // find a nearby tile that can be walked to
544 c->Goal = MapSearchTileAround(
545 &gMap, goalTile,
546 ignoreObjects ? IsTileWalkable : IsTileWalkableAroundObjects);
547
548 c->PathIndex = 1; // start navigating to the next path node
549 CachedPathDestroy(&c->Path);
550 c->Path = PathCacheCreate(
551 &gPathCache, currentTile, c->Goal, ignoreObjects, true);
552
553 // In case we can't calculate A* for some reason,
554 // try simple navigation again
555 if (ASPathGetCount(c->Path.Path) <= 1)
556 {
557 debug(
558 D_MAX,
559 "Error: can't calculate path from {%d, %d} to {%d, %d}",
560 currentTile.x, currentTile.y,
561 goalTile.x, goalTile.y);
562 return AIGotoDirect(a, p);
563 }
564
565 return AStarFollow(c, currentTile, &actor->tileItem, a);
566 }
567}

569// Hunt moves an Actor towards a target, using the most efficient direction.
570// That is, given the following octant:
571// x A xxxx
572// x xxxx
573// x xxxx
574// x xxxx B
575// x xxx
576// xxxxxxxxxxxxxxxxxxxxxxx
577// Those in slice A will move down-left and those in slice B will move left.
578int AIHunt(TActor *actor, Vec2i targetPos)
579{
580 const Vec2i fullPos = Vec2iAdd(actor->Pos, ActorGetGunMuzzleOffset(actor));
581 const int dx = abs(targetPos.x - fullPos.x);
582 const int dy = abs(targetPos.y - fullPos.y);
583
584 int cmd = 0;
585 if (2 * dx > dy)
586 {
587 if (fullPos.x < targetPos.x) cmd |= CMD_RIGHT;
588 else if (fullPos.x > targetPos.x) cmd |= CMD_LEFT;
589 }
590 if (2 * dy > dx)
591 {
592 if (fullPos.y < targetPos.y) cmd |= CMD_DOWN;
593 else if (fullPos.y > targetPos.y) cmd |= CMD_UP;
594 }
595 // If it's a coward, reverse directions...
596 if (actor->flags & FLAGS_RUNS_AWAY)
597 {
598 cmd = AIReverseDirection(cmd);
599 }
600
601 return cmd;
602}
603int AIHuntClosest(TActor *actor)
604{
605 Vec2i targetPos = actor->Pos;
606 if (!(actor->PlayerUID >= 0 || (actor->flags & FLAGS_GOOD_GUY)))
607 {
608 targetPos = AIGetClosestPlayerPos(actor->Pos);
609 }
610
611 if (actor->flags & FLAGS_VISIBLE)
612 {
613 const TActor *a =
614 AIGetClosestVisibleEnemy(actor, actor->PlayerUID >= 0);
615 if (a)
616 {
617 targetPos = a->Pos;
618 }
619 }
620 return AIHunt(actor, targetPos);
621}

630// Track moves an Actor towards a target, but in such a fashion that the Actor
631// will come into 8-axis alignment with the target soonest.
632// That is, given the following octant:
633// x A xxxx
634// x xxxx
635// x xxxx
636// x xxxx B
637// x xxx
638// xxxxxxxxxxxxxxxxxxxxxxx
639// Those in slice A will move left and those in slice B will move down-left.
640int AITrack(TActor *actor, const Vec2i targetPos)
641{
642 const Vec2i fullPos = Vec2iAdd(actor->Pos, ActorGetGunMuzzleOffset(actor));
643 const int dx = abs(targetPos.x - fullPos.x);
644 const int dy = abs(targetPos.y - fullPos.y);
645
646 int cmd = 0;
647 // Terminology: imagine the compass directions sliced into 16 equal parts,
648 // and labelled in bearing/clock order. That is, the slices on the right
649 // half are labelled 1-8.
650 // In order to construct the movement, note that:
651 // - If the target is to our left, we need to move left...
652 // - Except if they are in slice 2 or 7
653 // Slice 2 and 7 (and 10 and 15) can be found by (dy < 2dx && dy > dx)
654 // - call this X-exception
655 // Repeat this for all 4 cardinal directions.
656 // Furthermore, give a bit of leeway for the 8 axes so we don't
657 // fluctuate between perpendicular movement vectors.
658 const bool xException = dy < 2 * dx && dy > (int)(dx * 1.1);
659 if (!xException)
660 {
661 if (fullPos.x - targetPos.x < (int)(dy * 0.1)) cmd |= CMD_RIGHT;
662 else if (fullPos.x - targetPos.x > -(int)(dy * 0.1)) cmd |= CMD_LEFT;
663 }
664 const bool yException = dx < 2 * dy && dx > (int)(dy * 1.1);
665 if (!yException)
666 {
667 if (fullPos.y - targetPos.y < (int)(dx * 0.1)) cmd |= CMD_DOWN;
668 else if (fullPos.y - targetPos.y > -(int)(dx * 0.1)) cmd |= CMD_UP;
669 }
670
671 return cmd;
672}

674int AIMoveAwayFromLine(
675 const Vec2i fullPos, const Vec2i lineStartFull, const direction_e lineD)
676{
677 const Vec2i dv = Vec2iMinus(fullPos, lineStartFull);
678 switch (lineD)
679 {
680 case DIRECTION_UP:
681 case DIRECTION_DOWN: // fallthrough
682 return dv.x < 0 ? CMD_LEFT : CMD_RIGHT;
683 case DIRECTION_UPLEFT:
684 case DIRECTION_DOWNRIGHT: // fallthrough
685 return dv.x > dv.y ? (CMD_RIGHT | CMD_UP) : (CMD_LEFT | CMD_DOWN);
686 case DIRECTION_LEFT:
687 case DIRECTION_RIGHT: // fallthrough
688 return dv.y < 0 ? CMD_UP : CMD_DOWN;
689 case DIRECTION_DOWNLEFT:
690 case DIRECTION_UPRIGHT: // fallthrough
691 return dv.x < -dv.y ? (CMD_LEFT | CMD_UP) : (CMD_RIGHT | CMD_DOWN);
692 default:
693 CASSERT(false, "unknown direction");
694 return 0;
695 }
