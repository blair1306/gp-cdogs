135// Generic config entry
136typedef struct
137{
138 char *Name;
139 ConfigType Type;
140 union
141 {
142#define VALUES(_name, _type) \
143 struct \
144 { \
145 _type Value; \
146 _type Last; \
147 _type Default; \
148 _type Min; \
149 _type Max; \
150 _type Increment; \
151 } _name
152 VALUES(String, char *);
153 VALUES(Float, double);
154 VALUES(Bool, bool);
155#undef VALUES
156#define FORMATTED_VALUES(_name, _type, _toStrType) \
157 struct \
158 { \
159 _type Value; \
160 _type Last; \
161 _type Default; \
162 _type Min; \
163 _type Max; \
164 _type Increment; \
165 _type (*StrTo##_name)(const char *); \
166 _toStrType *(*_name##ToStr)(_type); \
167 } _name
168 FORMATTED_VALUES(Int, int, char);
169 FORMATTED_VALUES(Enum, int, const char);
170#undef FORMATTED_VALUES
171 CArray Group; // of Config
172 } u;
173} Config;
