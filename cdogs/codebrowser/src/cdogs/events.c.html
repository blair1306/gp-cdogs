44void EventInit(
45 EventHandlers *handlers, Pic *mouseCursor, Pic *mouseTrail,
46 const bool hideMouse)
47{
48 memset(handlers, 0, sizeof *handlers);
49 KeyInit(&handlers->keyboard);
50 JoyInit(&handlers->joysticks);
51 MouseInit(&handlers->mouse, mouseCursor, mouseTrail, hideMouse);
52}
53void EventTerminate(EventHandlers *handlers)
54{
55 JoyTerminate(&handlers->joysticks);
56}
57void EventReset(EventHandlers *handlers, Pic *mouseCursor, Pic *mouseTrail)
58{
59 KeyInit(&handlers->keyboard);
60 JoyReset(&handlers->joysticks);
61 MouseInit(
62 &handlers->mouse, mouseCursor, mouseTrail, handlers->mouse.hideMouse);
63}

65void EventPoll(EventHandlers *handlers, Uint32 ticks)
66{
67 SDL_Event e;
68 handlers->HasResolutionChanged = false;
69 handlers->HasLostFocus = false;
70 KeyPrePoll(&handlers->keyboard);
71 MousePrePoll(&handlers->mouse);
72 JoyPrePoll(&handlers->joysticks);
73 SDL_free(handlers->DropFile);
74 handlers->DropFile = NULL;
75 // Don't process mouse events if focus just regained this cycle
76 // This is to prevent bogus click events outside the window, e.g. in the
77 // title bar
78 bool regainedFocus = false;
79 while (SDL_PollEvent(&e))
80 {
81 switch (e.type)
82 {
83 case SDL_KEYDOWN:
84 if (e.key.repeat)
85 {
86 break;
87 }
88 KeyOnKeyDown(&handlers->keyboard, e.key.keysym);
89 break;
90 case SDL_KEYUP:
91 KeyOnKeyUp(&handlers->keyboard, e.key.keysym);
92 break;
93 case SDL_TEXTINPUT:
94 strcpy(handlers->keyboard.Typed, e.text.text);
95 break;
96
97 case SDL_CONTROLLERDEVICEADDED:
98 {
99 const SDL_JoystickID jid = JoyAdded(e.cdevice.which);
100 if (jid == -1)
101 {
102 break;
103 }
104 // If there are players with unset devices,
105 // set this controller to them
106 CA_FOREACH(PlayerData, p, gPlayerDatas)
107 if (p->inputDevice == INPUT_DEVICE_UNSET)
108 {
109 PlayerTrySetInputDevice(p, INPUT_DEVICE_JOYSTICK, jid);
110 LOG(LM_INPUT, LL_INFO,
111 "Joystick %d assigned to player %d", jid, p->UID);
112 break;
113 }
114 CA_FOREACH_END()
115 }
116 break;
117
118 case SDL_CONTROLLERDEVICEREMOVED:
119 JoyRemoved(e.cdevice.which);
120 // If there was a player using this joystick,
121 // set their input device to nothing
122 CA_FOREACH(PlayerData, p, gPlayerDatas)
123 if (p->inputDevice == INPUT_DEVICE_JOYSTICK &&
124 p->deviceIndex == e.cdevice.which)
125 {
126 PlayerTrySetInputDevice(p, INPUT_DEVICE_UNSET, 0);
127 LOG(LM_INPUT, LL_WARN, "Joystick for player %d removed",
128 p->UID);
129 break;
130 }
131 CA_FOREACH_END()
132 break;
133
134 case SDL_CONTROLLERBUTTONDOWN:
135 JoyOnButtonDown(e.cbutton);
136 break;
137
138 case SDL_CONTROLLERBUTTONUP:
139 JoyOnButtonUp(e.cbutton);
140 break;
141
142 case SDL_CONTROLLERAXISMOTION:
143 JoyOnAxis(e.caxis);
144 break;
145
146 case SDL_MOUSEBUTTONDOWN:
147 if (regainedFocus) break;
148 MouseOnButtonDown(&handlers->mouse, e.button.button);
149 break;
150 case SDL_MOUSEBUTTONUP:
151 if (regainedFocus) break;
152 MouseOnButtonUp(&handlers->mouse, e.button.button);
153 break;
154 case SDL_MOUSEWHEEL:
155 if (regainedFocus) break;
156 MouseOnWheel(&handlers->mouse, e.wheel.x, e.wheel.y);
157 break;
158 case SDL_WINDOWEVENT:
159 switch (e.window.event)
160 {
161 case SDL_WINDOWEVENT_FOCUS_GAINED:
162 regainedFocus = true;
163 MusicSetPlaying(&gSoundDevice, true);
164 break;
165 case SDL_WINDOWEVENT_FOCUS_LOST:
166 if (!gCampaign.IsClient && !ConfigGetBool(&gConfig, "StartServer"))
167 {
168 MusicSetPlaying(&gSoundDevice, false);
169 handlers->HasLostFocus = true;
170 }
171 // Reset input handlers
172 EventReset(
173 handlers, handlers->mouse.cursor, handlers->mouse.trail);
174 break;
175 case SDL_WINDOWEVENT_SIZE_CHANGED:
176 handlers->HasResolutionChanged = true;
177 if (gGraphicsDevice.cachedConfig.IsEditor)
178 {
179 const int scale = ConfigGetInt(&gConfig, "Graphics.ScaleFactor");
180 GraphicsConfigSet(
181 &gGraphicsDevice.cachedConfig,
182 Vec2iScaleDiv(
183 Vec2iNew(e.window.data1, e.window.data2), scale),
184 false,
185 scale,
186 gGraphicsDevice.cachedConfig.ScaleMode,
187 gGraphicsDevice.cachedConfig.Brightness);
188 GraphicsInitialize(&gGraphicsDevice);
189 }
190 break;
191 default:
192 // do nothing
193 break;
194 }
195 break;
196 case SDL_QUIT:
197 handlers->HasQuit = true;
198 break;
199 case SDL_DROPFILE:
200 handlers->DropFile = e.drop.file;
201 break;
202 default:
203 break;
204 }
205 }
206 KeyPostPoll(&handlers->keyboard, ticks);
207 MousePostPoll(&handlers->mouse, ticks);
208}

210int GetKeyboardCmd(
211 keyboard_t *keyboard, const int kbIndex, const bool isPressed)
212{
213 int cmd = 0;
214 bool (*keyFunc)(const keyboard_t *, const int) =
215 isPressed ? KeyIsPressed : KeyIsDown;
216 const InputKeys *keys = &keyboard->PlayerKeys[kbIndex];
217
218 if (keyFunc(keyboard, keys->left)) cmd |= CMD_LEFT;
219 else if (keyFunc(keyboard, keys->right)) cmd |= CMD_RIGHT;
220
221 if (keyFunc(keyboard, keys->up)) cmd |= CMD_UP;
222 else if (keyFunc(keyboard, keys->down)) cmd |= CMD_DOWN;
223
224 if (keyFunc(keyboard, keys->button1)) cmd |= CMD_BUTTON1;
225
226 if (keyFunc(keyboard, keys->button2)) cmd |= CMD_BUTTON2;
227
228 return cmd;
229}
230static int GetMouseCmd(
231 Mouse *mouse, bool isPressed, int useMouseMove, Vec2i pos)
232{
233 int cmd = 0;
234 bool (*mouseFunc)(const Mouse *, const int) =
235 isPressed ? MouseIsPressed : MouseIsDown;
236
237 if (useMouseMove)
238 {
239 cmd |= MouseGetMove(mouse, pos);
240 }
241 else
242 {
243 if (MouseWheel(mouse).y > 0) cmd |= CMD_UP;
244 else if (MouseWheel(mouse).y < 0) cmd |= CMD_DOWN;
245 }
246
247 if (mouseFunc(mouse, SDL_BUTTON_LEFT)) cmd |= CMD_BUTTON1;
248 if (mouseFunc(mouse, SDL_BUTTON_RIGHT)) cmd |= CMD_BUTTON2;
249 if (mouseFunc(mouse, SDL_BUTTON_MIDDLE)) cmd |= CMD_MAP;
250
251 return cmd;
252}

254static int GetJoystickCmd(const SDL_JoystickID id, bool isPressed)
255{
256 int cmd = 0;
257 bool (*joyFunc)(const SDL_JoystickID, const int) =
258 isPressed ? JoyIsPressed : JoyIsDown;
259
260 if (joyFunc(id, CMD_LEFT)) cmd |= CMD_LEFT;
261 else if (joyFunc(id, CMD_RIGHT)) cmd |= CMD_RIGHT;
262
263 if (joyFunc(id, CMD_UP)) cmd |= CMD_UP;
264 else if (joyFunc(id, CMD_DOWN)) cmd |= CMD_DOWN;
265
266 if (joyFunc(id, CMD_BUTTON1)) cmd |= CMD_BUTTON1;
267
268 if (joyFunc(id, CMD_BUTTON2)) cmd |= CMD_BUTTON2;
269
270 if (joyFunc(id, CMD_MAP)) cmd |= CMD_MAP;
271
272 if (joyFunc(id, CMD_ESC)) cmd |= CMD_ESC;
273
274 return cmd;
275}

277int GetGameCmd(
278 EventHandlers *handlers,
279 const PlayerData *playerData, const Vec2i playerPos)
280{
281 int cmd = 0;
282
283 switch (playerData->inputDevice)
284 {
285 case INPUT_DEVICE_KEYBOARD:
286 cmd = GetKeyboardCmd(
287 &handlers->keyboard, playerData->deviceIndex, false);
288 break;
289 case INPUT_DEVICE_MOUSE:
290 cmd = GetMouseCmd(&handlers->mouse, false, 1, playerPos);
291 break;
292 case INPUT_DEVICE_JOYSTICK:
293 cmd = GetJoystickCmd(playerData->deviceIndex, false);
294 break;
295 default:
296 // do nothing
297 break;
298 }
299
300 return cmd;
301}

303int GetOnePlayerCmd(
304 EventHandlers *handlers, const bool isPressed,
305 const input_device_e device, const int deviceIndex)
306{
307 int cmd = 0;
308 switch (device)
309 {
310 case INPUT_DEVICE_KEYBOARD:
311 cmd = GetKeyboardCmd(&handlers->keyboard, deviceIndex, isPressed);
312 break;
313 case INPUT_DEVICE_MOUSE:
314 cmd = GetMouseCmd(&handlers->mouse, isPressed, 0, Vec2iZero());
315 break;
316 case INPUT_DEVICE_JOYSTICK:
317 cmd = GetJoystickCmd(deviceIndex, isPressed);
318 break;
319 default:
320 // Do nothing
321 break;
322 }
323 return cmd;
324}

326void GetPlayerCmds(EventHandlers *handlers, int (*cmds)[MAX_LOCAL_PLAYERS])
327{
328 int idx = 0;
329 for (int i = 0; i < (int)gPlayerDatas.size; i++, idx++)
330 {
331 const PlayerData *p = CArrayGet(&gPlayerDatas, i);
332 if (!p->IsLocal)
333 {
334 idx--;
335 continue;
336 }
337 if (p->inputDevice != INPUT_DEVICE_UNSET)
338 {
339 (*cmds)[idx] = GetOnePlayerCmd(
340 handlers, true, p->inputDevice, p->deviceIndex);
341 }
342 }
343}

345int GetMenuCmd(EventHandlers *handlers)
346{
347 keyboard_t *kb = &handlers->keyboard;
348 bool firstJoyPressedEsc = false;
349 SDL_JoystickID firstJoyId = 0;
350 if (handlers->joysticks.size > 0)
351 {
352 const Joystick *firstJoy = CArrayGet(&handlers->joysticks, 0);
353 firstJoyId = firstJoy->id;
354 firstJoyPressedEsc = JoyIsPressed(firstJoyId, CMD_ESC);
355 }
356 if (KeyIsPressed(kb, SDL_SCANCODE_ESCAPE) || firstJoyPressedEsc)
357 {
358 return CMD_ESC;
359 }
360
361 // Check first player keyboard
362 int cmd = GetOnePlayerCmd(handlers, true, INPUT_DEVICE_KEYBOARD, 0);
363 if (!cmd)
364 {
365 // Check keyboard
366 if (KeyIsPressed(kb, SDL_SCANCODE_LEFT)) cmd |= CMD_LEFT;
367 else if (KeyIsPressed(kb, SDL_SCANCODE_RIGHT)) cmd |= CMD_RIGHT;
368
369 if (KeyIsPressed(kb, SDL_SCANCODE_UP)) cmd |= CMD_UP;
370 else if (KeyIsPressed(kb, SDL_SCANCODE_DOWN)) cmd |= CMD_DOWN;
371
372 if (KeyIsPressed(kb, SDL_SCANCODE_RETURN)) cmd |= CMD_BUTTON1;
373
374 if (KeyIsPressed(kb, SDL_SCANCODE_BACKSPACE)) cmd |= CMD_BUTTON2;
375 }
376 if (!cmd && handlers->joysticks.size > 0)
377 {
378 // Check joystick 1
379 cmd = GetOnePlayerCmd(
380 handlers, true, INPUT_DEVICE_JOYSTICK, firstJoyId);
381 }
382 if (!cmd)
383 {
384 // Check mouse
385 cmd = GetOnePlayerCmd(handlers, true, INPUT_DEVICE_MOUSE, 0);
386 }
387
388 return cmd;
389}

392void InputGetButtonNameColor(
393 const input_device_e d, const int dIndex, const int cmd,
394 char *buf, color_t *color)
395{
396 switch (d)
397 {
398 case INPUT_DEVICE_KEYBOARD:
399 #ifdef __GCWZERO__
400 if (color != NULL)
401 {
402 *color = colorBlue;
403 }
404 switch (cmd)
405 {
406 case CMD_LEFT: strcpy(buf, "left"); return;
407 case CMD_RIGHT: strcpy(buf, "right"); return;
408 case CMD_UP: strcpy(buf, "up"); return;
409 case CMD_DOWN:strcpy(buf, "down"); return;
410 case CMD_BUTTON1: strcpy(buf, "A"); return;
411 case CMD_BUTTON2: strcpy(buf, "B"); return;
412 case CMD_MAP: *color = colorGray; strcpy(buf, "L"); return;
413 case CMD_ESC: strcpy(buf, "SELECT"); return;
414 default: CASSERT(false, "unknown button"); return;
415 }
416 #else
417 {
418 const InputKeys *keys =
419 &gEventHandlers.keyboard.PlayerKeys[dIndex];
420 switch (cmd)
421 {
422 case CMD_LEFT: strcpy(buf, SDL_GetScancodeName(keys->left)); return;
423 case CMD_RIGHT: strcpy(buf, SDL_GetScancodeName(keys->right)); return;
424 case CMD_UP: strcpy(buf, SDL_GetScancodeName(keys->up)); return;
425 case CMD_DOWN: strcpy(buf, SDL_GetScancodeName(keys->down)); return;
426 case CMD_BUTTON1: strcpy(buf, SDL_GetScancodeName(keys->button1)); return;
427 case CMD_BUTTON2: strcpy(buf, SDL_GetScancodeName(keys->button2)); return;
428 case CMD_MAP: strcpy(buf, SDL_GetScancodeName(keys->map)); return;
429 case CMD_ESC: strcpy(buf, SDL_GetScancodeName(SDL_SCANCODE_ESCAPE)); return;
430 default: CASSERT(false, "unknown button"); return;
431 }
432 }
433 #endif
434 break;
435 case INPUT_DEVICE_MOUSE:
436 switch (cmd)
437 {
438 case CMD_LEFT: strcpy(buf, "left"); return;
439 case CMD_RIGHT: strcpy(buf, "right"); return;
440 case CMD_UP: strcpy(buf, "up"); return;
441 case CMD_DOWN: strcpy(buf, "down"); return;
442 case CMD_BUTTON1: strcpy(buf, "left click"); return;
443 case CMD_BUTTON2: strcpy(buf, "right click"); return;
444 case CMD_MAP: strcpy(buf, "middle click"); return;
445 case CMD_ESC: strcpy(buf, ""); return;
446 default: CASSERT(false, "unknown button"); return;
447 }
448 break;
449 case INPUT_DEVICE_JOYSTICK:
450 JoyButtonNameColor(dIndex, cmd, buf, color);
451 return;
452 case INPUT_DEVICE_AI:
453 return;
454 default:
455 CASSERT(false, "unknown input device");
456 return;
457 }
458}
459void InputGetDirectionNames(
460 char *buf, const input_device_e d, const int dIndex)
461{
462 strcpy(buf, "");
463 switch (d)
464 {
465 case INPUT_DEVICE_KEYBOARD:
466 {
467 char left[256], right[256], up[256], down[256];
468 InputGetButtonName(d, dIndex, CMD_LEFT, left);
469 InputGetButtonName(d, dIndex, CMD_RIGHT, right),
470 InputGetButtonName(d, dIndex, CMD_UP, up),
471 InputGetButtonName(d, dIndex, CMD_DOWN, down);
472 sprintf(buf, "%s, %s, %s, %s", left, right, up, down);
473 }
474 break;
475 case INPUT_DEVICE_MOUSE:
476 strcpy(buf, "mouse wheel");
477 break;
478 case INPUT_DEVICE_JOYSTICK:
479 strcpy(buf, "directions");
480 break;
481 case INPUT_DEVICE_AI:
482 break;
483 case INPUT_DEVICE_UNSET:
484 break;
485 default:
486 CASSERT(false, "unknown device");
487 break;
488 }
489}

491SDL_Scancode GetKey(EventHandlers *handlers)
492{
493 SDL_Scancode k = SDL_SCANCODE_UNKNOWN;
494 do
495 {
496 EventPoll(handlers, SDL_GetTicks());
497 k = KeyGetPressed(&handlers->keyboard);
498 SDL_Delay(10);
499 } while (k == SDL_SCANCODE_UNKNOWN);
500 return k;
501}

503SDL_Scancode EventWaitKeyOrText(EventHandlers *handlers)
504{
505 SDL_Scancode k = SDL_SCANCODE_UNKNOWN;
506 do
507 {
508 EventPoll(handlers, SDL_GetTicks());
509 k = KeyGetPressed(&handlers->keyboard);
510 SDL_Delay(10);
511 } while (k == SDL_SCANCODE_UNKNOWN && handlers->keyboard.Typed[0] == '\0');
512 return k;
513}

515static EventWaitResult WaitResult(const bool result);
516EventWaitResult EventWaitForAnyKeyOrButton(void)
517{
518 int cmds[MAX_LOCAL_PLAYERS];
519 memset(cmds, 0, sizeof cmds);
520 GetPlayerCmds(&gEventHandlers, &cmds);
521 for (int i = 0; i < MAX_LOCAL_PLAYERS; i++)
522 {
523 if (cmds[i] & (CMD_BUTTON1 | CMD_BUTTON2))
524 {
525 // Interpret anything other than CMD_BUTTON1 as cancel
526 return WaitResult(cmds[i] & CMD_BUTTON1);
527 }
528 }
529
530 // Check menu commands
531 const int menuCmd = GetMenuCmd(&gEventHandlers);
532 if (menuCmd & (CMD_BUTTON1 | CMD_BUTTON2))
533 {
534 // Interpret anything other than CMD_BUTTON1 as cancel
535 return WaitResult(menuCmd & CMD_BUTTON1);
536 }
537
538 // Check if anyone pressed escape
539 if (EventIsEscape(&gEventHandlers, cmds, menuCmd))
540 {
541 return WaitResult(false);
542 }
543
544 return EVENT_WAIT_CONTINUE;
545}
546static EventWaitResult WaitResult(const bool result)
547{
548 return result ? EVENT_WAIT_OK : EVENT_WAIT_CANCEL;
549}
550
551bool EventIsEscape(
552 EventHandlers *handlers,
553 const int cmds[MAX_LOCAL_PLAYERS], const int menuCmd)
554{
555 for (int i = 0; i < MAX_LOCAL_PLAYERS; i++)
556 {
557 if (cmds[i] & CMD_ESC)
558 {
559 return true;
560 }
561 }
562
563 // Check keyboard escape
564 if (KeyIsPressed(&handlers->keyboard, SDL_SCANCODE_ESCAPE) ||
565 handlers->HasQuit)
566 {
567 return true;
568 }
569
570 // Check menu commands
571 if (menuCmd & CMD_ESC)
572 {
573 return true;
574 }
575
576 return false;
577}
