58// Commands
59#define CMD_LEFT 1
60#define CMD_RIGHT 2
61#define CMD_UP 4
62#define CMD_DOWN 8
63#define CMD_BUTTON1 16
64#define CMD_BUTTON2 32
65#define CMD_MAP 64
66#define CMD_ESC 128
67
68// Command macros
69#define Left(x) (((x) & CMD_LEFT) != 0)
70#define Right(x) (((x) & CMD_RIGHT) != 0)
71#define Up(x) (((x) & CMD_UP) != 0)
72#define Down(x) (((x) & CMD_DOWN) != 0)
73#define Button1(x) (((x) & CMD_BUTTON1) != 0)
74#define Button2(x) (((x) & CMD_BUTTON2) != 0)
75#define AnyButton(x) (((x) & (CMD_BUTTON1 | CMD_BUTTON2)) != 0)
76#define CMD_HAS_DIRECTION(x)\
77 ((x) & (CMD_LEFT | CMD_RIGHT | CMD_UP | CMD_DOWN))
78
