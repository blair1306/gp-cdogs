119Tile *MapGetTile(Map *map, Vec2i pos)
120{
121 if (pos.x < 0 || pos.x >= map->Size.x || pos.y < 0 || pos.y >= map->Size.y)
122 {
123 return NULL;
124 }
125 return CArrayGet(&map->Tiles, pos.y * map->Size.x + pos.x);
126}

128bool MapIsTileIn(const Map *map, const Vec2i pos)
129{
130 // Check that the tile pos is within the interior of the map
131 return !(pos.x < 0 || pos.y < 0 ||
132 pos.x > map->Size.x - 1 || pos.y > map->Size.y - 1);
133}
134bool MapIsRealPosIn(const Map *map, const Vec2i realPos)
135{
136 // Check that the real pos is within the interior of the map
137 // Note: can't use Vec2iToTile as division will cause small
138 // negative values to appear to be 0 i.e. valid
139 return !(realPos.x < 0 || realPos.y < 0 ||
140 realPos.x / TILE_WIDTH > map->Size.x - 1 ||
141 realPos.y / TILE_HEIGHT > map->Size.y - 1);
142}

144bool MapIsTileInExit(const Map *map, const TTileItem *ti)
145{
146 return
147 ti->x / TILE_WIDTH >= map->ExitStart.x &&
148 ti->x / TILE_WIDTH <= map->ExitEnd.x &&
149 ti->y / TILE_HEIGHT >= map->ExitStart.y &&
150 ti->y / TILE_HEIGHT <= map->ExitEnd.y;
151}

159static void AddItemToTile(TTileItem *t, Tile *tile);
160bool MapTryMoveTileItem(Map *map, TTileItem *t, Vec2i pos)
161{
162 // Check if we can move to new position
163 if (!MapIsRealPosIn(map, pos))
164 {
165 return false;
166 }
167 // When first initialised, position is -1
168 bool doRemove = t->x >= 0 && t->y >= 0;
169 Vec2i t1 = Vec2iToTile(Vec2iNew(t->x, t->y));
170 Vec2i t2 = Vec2iToTile(pos);
171 // If we'll be in the same tile, do nothing
172 if (Vec2iEqual(t1, t2) && doRemove)
173 {
174 t->x = pos.x;
175 t->y = pos.y;
176 return true;
177 }
178 // Moving; remove from old tile...
179 if (doRemove)
180 {
181 MapRemoveTileItem(map, t);
182 }
183 // ...move and add to new tile
184 t->x = pos.x;
185 t->y = pos.y;
186 AddItemToTile(t, MapGetTile(map, t2));
187 return true;
188}
189static void AddItemToTile(TTileItem *t, Tile *tile)
190{
191 ThingId tid;
192 tid.Id = t->id;
193 tid.Kind = t->kind;
194 CASSERT(tid.Id >= 0, "invalid ThingId");
195 CASSERT(tid.Kind >= 0 && tid.Kind <= KIND_PICKUP, "unknown thing kind");
196 CArrayPushBack(&tile->things, &tid);
197}
198
199void MapRemoveTileItem(Map *map, TTileItem *t)
200{
201 if (!MapIsRealPosIn(map, Vec2iNew(t->x, t->y)))
202 {
203 return;
204 }
205 Tile *tile = MapGetTileOfItem(map, t);
206 CA_FOREACH(ThingId, tid, tile->things)
207 if (tid->Id == t->id && tid->Kind == t->kind)
208 {
209 CArrayDelete(&tile->things, _ca_index);
210 return;
211 }
212 CA_FOREACH_END()
213 CASSERT(false, "Did not find element to delete");
214}

237void MapChangeFloor(
238 Map *map, const Vec2i pos, NamedPic *normal, NamedPic *shadow)
239{
240 Tile *tAbove = MapGetTile(map, Vec2iNew(pos.x, pos.y - 1));
241 int canSeeTileAbove = !(pos.y > 0 && !TileCanSee(tAbove));
242 Tile *t = MapGetTile(map, pos);
243 switch (IMapGet(map, pos) & MAP_MASKACCESS)
244 {
245 case MAP_FLOOR:
246 case MAP_SQUARE:
247 case MAP_ROOM:
248 if (!canSeeTileAbove)
249 {
250 t->pic = shadow;
251 }
252 else
253 {
254 t->pic = normal;
255 }
256 break;
257 default:
258 // do nothing
259 break;
260 }
261}

263void MapShowExitArea(Map *map, const Vec2i exitStart, const Vec2i exitEnd)
264{
265 const int left = exitStart.x;
266 const int right = exitEnd.x;
267 const int top = exitStart.y;
268 const int bottom = exitEnd.y;
269
270 NamedPic *exitPic = PicManagerGetExitPic(
271 &gPicManager, gMission.missionData->ExitStyle, false);
272 NamedPic *exitShadowPic = PicManagerGetExitPic(
273 &gPicManager, gMission.missionData->ExitStyle, true);
274
275 Vec2i v;
276 v.y = top;
277 for (v.x = left; v.x <= right; v.x++)
278 {
279 MapChangeFloor(map, v, exitPic, exitShadowPic);
280 }
281 v.y = bottom;
282 for (v.x = left; v.x <= right; v.x++)
283 {
284 MapChangeFloor(map, v, exitPic, exitShadowPic);
285 }
286 v.x = left;
287 for (v.y = top + 1; v.y < bottom; v.y++)
288 {
289 MapChangeFloor(map, v, exitPic, exitShadowPic);
290 }
291 v.x = right;
292 for (v.y = top + 1; v.y < bottom; v.y++)
293 {
294 MapChangeFloor(map, v, exitPic, exitShadowPic);
295 }
296}

304// Adjacent means to the left, right, above or below
305static int MapGetNumWallsAdjacentTile(Map *map, Vec2i v)
306{
307 int count = 0;
308 if (v.x > 0 && v.y > 0 && v.x < map->Size.x - 1 && v.y < map->Size.y - 1)
309 {
310 if (!TileCanWalk(MapGetTile(map, Vec2iNew(v.x - 1, v.y))))
311 {
312 count++;
313 }
314 if (!TileCanWalk(MapGetTile(map, Vec2iNew(v.x + 1, v.y))))
315 {
316 count++;
317 }
318 if (!TileCanWalk(MapGetTile(map, Vec2iNew(v.x, v.y - 1))))
319 {
320 count++;
321 }
322 if (!TileCanWalk(MapGetTile(map, Vec2iNew(v.x, v.y + 1))))
323 {
324 count++;
325 }
326 }
327 return count;
328}
329
330// Around means the 8 tiles surrounding the tile
331static int MapGetNumWallsAroundTile(Map *map, Vec2i v)
332{
333 int count = MapGetNumWallsAdjacentTile(map, v);
334 if (v.x > 0 && v.y > 0 && v.x < map->Size.x - 1 && v.y < map->Size.y - 1)
335 {
336 // Having checked the adjacencies, check the diagonals
337 if (!TileCanWalk(MapGetTile(map, Vec2iNew(v.x - 1, v.y - 1))))
338 {
339 count++;
340 }
341 if (!TileCanWalk(MapGetTile(map, Vec2iNew(v.x + 1, v.y + 1))))
342 {
343 count++;
344 }
345 if (!TileCanWalk(MapGetTile(map, Vec2iNew(v.x + 1, v.y - 1))))
346 {
347 count++;
348 }
349 if (!TileCanWalk(MapGetTile(map, Vec2iNew(v.x - 1, v.y + 1))))
350 {
351 count++;
352 }
353 }
354 return count;
355}

357bool MapTryPlaceOneObject(
358 Map *map, const Vec2i v, const MapObject *mo, const int extraFlags,
359 const bool isStrictMode)
360{
361 // Don't place ammo spawners if ammo is disabled
362 if (!ConfigGetBool(&gConfig, "Game.Ammo") &&
363 mo->Type == MAP_OBJECT_TYPE_PICKUP_SPAWNER &&
364 mo->u.PickupClass->Type == PICKUP_AMMO)
365 {
366 return false;
367 }
368 const Tile *t = MapGetTile(map, v);
369 unsigned short iMap = IMapGet(map, v);
370
371 const bool isEmpty = TileIsClear(t);
372 if (isStrictMode && !MapObjectIsTileOKStrict(
373 mo, iMap, isEmpty,
374 IMapGet(map, Vec2iNew(v.x, v.y - 1)),
375 IMapGet(map, Vec2iNew(v.x, v.y + 1)),
376 MapGetNumWallsAdjacentTile(map, v),
377 MapGetNumWallsAroundTile(map, v)))
378 {
379 return 0;
380 }
381 else if (!MapObjectIsTileOK(
382 mo, iMap, isEmpty, IMapGet(map, Vec2iNew(v.x, v.y - 1))))
383 {
384 return 0;
385 }
386
387 if (mo->Flags & (1 << PLACEMENT_FREE_IN_FRONT))
388 {
389 IMapSet(map, Vec2iNew(v.x, v.y + 1), IMapGet(map, Vec2iNew(v.x, v.y + 1)) | MAP_LEAVEFREE);
390 }
391
392 NMapObjectAdd amo = NMapObjectAdd_init_default;
393 amo.UID = ObjsGetNextUID();
394 strcpy(amo.MapObjectClass, mo->Name);
395 amo.Pos = Vec2i2Net(MapObjectGetPlacementPos(mo, v));
396 amo.TileItemFlags = MapObjectGetFlags(mo) | extraFlags;
397 amo.Health = mo->Health;
398 ObjAdd(amo);
399 return true;
400}

413void MapPlaceCollectible(
414 const struct MissionOptions *mo, const int objective, const Vec2i realPos)
415{
416 const Objective *o = CArrayGet(&mo->missionData->Objectives, objective);
417 GameEvent e = GameEventNew(GAME_EVENT_ADD_PICKUP);
418 e.u.AddPickup.UID = PickupsGetNextUID();
419 strcpy(e.u.AddPickup.PickupClass, o->u.Pickup->Name);
420 e.u.AddPickup.IsRandomSpawned = false;
421 e.u.AddPickup.SpawnerUID = -1;
422 e.u.AddPickup.TileItemFlags = ObjectiveToTileItem(objective);
423 e.u.AddPickup.Pos = Vec2i2Net(realPos);
424 GameEventsEnqueue(&gGameEvents, e);
425}
426static int MapTryPlaceCollectible(
427 Map *map, const Mission *mission, const struct MissionOptions *mo,
428 const int objective)
429{
430 const Objective *o = CArrayGet(&mission->Objectives, objective);
431 const bool hasLockedRooms =
432 (o->Flags & OBJECTIVE_HIACCESS) && MapHasLockedRooms(map);
433 const bool noaccess = o->Flags & OBJECTIVE_NOACCESS;
434 int i = (noaccess || hasLockedRooms) ? 1000 : 100;
435
436 while (i)
437 {
438 Vec2i v = GuessPixelCoords(map);
439 Vec2i size = Vec2iNew(COLLECTABLE_W, COLLECTABLE_H);
440 if (!IsCollisionWithWall(v, size))
441 {
442 if ((!hasLockedRooms || MapPosIsInLockedRoom(map, v)) &&
443 (!noaccess || !MapPosIsInLockedRoom(map, v)))
444 {
445 MapPlaceCollectible(mo, objective, v);
446 return 1;
447 }
448 }
449 i--;
450 }
451 return 0;
452}

454Vec2i MapGenerateFreePosition(Map *map, Vec2i size)
455{
456 for (int i = 0; i < 100; i++)
457 {
458 Vec2i v = GuessPixelCoords(map);
459 if (!IsCollisionWithWall(v, size))
460 {
461 return v;
462 }
463 }
464 return Vec2iZero();
465}

467static bool MapTryPlaceBlowup(
468 Map *map, const Mission *mission, const int objective)
469{
470 const Objective *o = CArrayGet(&mission->Objectives, objective);
471 const bool hasLockedRooms =
472 (o->Flags & OBJECTIVE_HIACCESS) && MapHasLockedRooms(map);
473 const bool noaccess = o->Flags & OBJECTIVE_NOACCESS;
474 int i = (noaccess || hasLockedRooms) ? 1000 : 100;
475
476 while (i > 0)
477 {
478 const Vec2i v = MapGetRandomTile(map);
479 if ((!hasLockedRooms || (IMapGet(map, v) >> 8)) &&
480 (!noaccess || (IMapGet(map, v) >> 8) == 0))
481 {
482 if (MapTryPlaceOneObject(
483 map,
484 v,
485 o->u.MapObject,
486 ObjectiveToTileItem(objective), true))
487 {
488 return 1;
489 }
490 }
491 i--;
492 }
493 return 0;
494}

513static void MapPlaceCard(Map *map, int keyIndex, int map_access)
514{
515 for (;;)
516 {
517 const Vec2i v = MapGetRandomTile(map);
518 Tile *t;
519 Tile *tBelow;
520 unsigned short iMap;
521 t = MapGetTile(map, v);
522 iMap = IMapGet(map, v);
523 tBelow = MapGetTile(map, Vec2iNew(v.x, v.y + 1));
524 if (TileIsClear(t) &&
525 (iMap & 0xF00) == map_access &&
526 (iMap & MAP_MASKACCESS) == MAP_ROOM &&
527 TileIsClear(tBelow))
528 {
529 MapPlaceKey(map, &gMission, v, keyIndex);
530 return;
531 }
532 }
533}

535static int AccessCodeToFlags(int code)
536{
537 if (code & MAP_ACCESS_RED)
538 return FLAGS_KEYCARD_RED;
539 if (code & MAP_ACCESS_BLUE)
540 return FLAGS_KEYCARD_BLUE;
541 if (code & MAP_ACCESS_GREEN)
542 return FLAGS_KEYCARD_GREEN;
543 if (code & MAP_ACCESS_YELLOW)
544 return FLAGS_KEYCARD_YELLOW;
545 return 0;
546}

569static int MapGetAccessFlags(Map *map, int x, int y)
570{
571 int flags = 0;
572 flags = MAX(flags, AccessCodeToFlags(IMapGet(map, Vec2iNew(x, y))));
573 flags = MAX(flags, AccessCodeToFlags(IMapGet(map, Vec2iNew(x - 1, y))));
574 flags = MAX(flags, AccessCodeToFlags(IMapGet(map, Vec2iNew(x + 1, y))));
575 flags = MAX(flags, AccessCodeToFlags(IMapGet(map, Vec2iNew(x, y - 1))));
576 flags = MAX(flags, AccessCodeToFlags(IMapGet(map, Vec2iNew(x, y + 1))));
577 return flags;
578}

580static void MapSetupDoors(Map *map, const Mission *m)
581{
582 Vec2i v;
583 for (v.x = 0; v.x < map->Size.x; v.x++)
584 {
585 for (v.y = 0; v.y < map->Size.y; v.y++)
586 {
587 // Check if this is the start of a door group
588 // Top or left-most door
589 if ((IMapGet(map, v) & MAP_MASKACCESS) == MAP_DOOR &&
590 (IMapGet(map, Vec2iNew(v.x - 1, v.y)) & MAP_MASKACCESS) != MAP_DOOR &&
591 (IMapGet(map, Vec2iNew(v.x, v.y - 1)) & MAP_MASKACCESS) != MAP_DOOR)
592 {
593 MapAddDoorGroup(map, m, v, MapGetAccessFlags(map, v.x, v.y));
594 }
595 }
596 }

599void MapTerminate(Map *map)
600{
601 CA_FOREACH(Trigger *, t, map->triggers)
602 TriggerTerminate(*t);
603 CA_FOREACH_END()
604 CArrayTerminate(&map->triggers);
605 Vec2i v;
606 for (v.y = 0; v.y < map->Size.y; v.y++)
607 {
608 for (v.x = 0; v.x < map->Size.x; v.x++)
609 {
610 Tile *t = MapGetTile(map, v);
611 TileDestroy(t);
612 }
613 }
614 CArrayTerminate(&map->Tiles);
615 CArrayTerminate(&map->iMap);
616 LOSTerminate(&map->LOS);
617 PathCacheTerminate(&gPathCache);
618}
619void MapLoad(
620 Map *map, const struct MissionOptions *mo, const CampaignOptions *co)
621{
622 MapTerminate(map);
623
624 // Init map
625 memset(map, 0, sizeof *map);
626 CArrayInit(&map->Tiles, sizeof(Tile));
627 CArrayInit(&map->iMap, sizeof(unsigned short));
628 const Mission *mission = mo->missionData;
629 map->Size = mission->Size;
630 LOSInit(map, map->Size);
631 CArrayInit(&map->triggers, sizeof(Trigger *));
632 PathCacheInit(&gPathCache, map);
633
634 Vec2i v;
635 for (v.y = 0; v.y < map->Size.y; v.y++)
636 {
637 for (v.x = 0; v.x < map->Size.x; v.x++)
638 {
639 Tile t;
640 unsigned short tI = MAP_FLOOR;
641 TileInit(&t);
642 CArrayPushBack(&map->Tiles, &t);
643 CArrayPushBack(&map->iMap, &tI);
644 }
645 }
646
647 switch (mission->Type)
648 {
649 case MAPTYPE_CLASSIC:
650 MapClassicLoad(map, mission, co);
651 break;
652 case MAPTYPE_STATIC:
653 MapStaticLoad(map, mo);
654 break;
655 case MAPTYPE_CAVE:
656 MapCaveLoad(map, mo, co);
657 break;
658 default:
659 CASSERT(false, "unknown map type");
660 break;
661 }
662
663 MapSetupTilesAndWalls(map, mission);
664 MapSetupDoors(map, mission);
665
666 if (mission->Type == MAPTYPE_CLASSIC)
667 {
668 // Randomly add drainage tiles for classic map type;
669 // For other map types drains are regular map objects
670 const MapObject *drain = StrMapObject("drain0");
671 for (int i = 0; i < map->Size.x*map->Size.y / 45; i++)
672 {
673 // Make sure drain tiles aren't next to each other
674 v = Vec2iNew(
675 (rand() % map->Size.x) & 0xFFFFFE,
676 (rand() % map->Size.y) & 0xFFFFFE);
677 const Tile *t = MapGetTile(map, v);
678 if (TileIsNormalFloor(t))
679 {
680 MapTryPlaceOneObject(map, v, drain, 0, false);
681 }
682 }
683 }
684
685 // Set exit now since we have set up all the tiles
686 if (Vec2iIsZero(map->ExitStart) && Vec2iIsZero(map->ExitEnd))
687 {
688 MapGenerateRandomExitArea(map);
689 }
690
691 // Count total number of reachable tiles, for explored %
692 map->NumExplorableTiles = 0;
693 for (v.y = 0; v.y < map->Size.y; v.y++)
694 {
695 for (v.x = 0; v.x < map->Size.x; v.x++)
696 {
697 if (!(MapGetTile(map, v)->flags & MAPTILE_NO_WALK))
698 {
699 map->NumExplorableTiles++;
700 }
701 }
702 }
703}

708 Map *map, const struct MissionOptions *mo, const CharacterStore *store)
709{
710 const Mission *mission = mo->missionData;
711
712 if (mission->Type == MAPTYPE_STATIC)
713 {
714 MapStaticLoadDynamic(map, mo, store);
715 }
716
717 // Add map objects
718 CA_FOREACH(const MapObjectDensity, mod, mission->MapObjectDensities)
719 for (int j = 0;
720 j < (mod->Density * map->Size.x * map->Size.y) / 1000;
721 j++)
722 {
723 MapTryPlaceOneObject(
724 map,
725 Vec2iNew(rand() % map->Size.x, rand() % map->Size.y),
726 mod->M,
727 0,
728 true);
729 }
730 CA_FOREACH_END()
731
732 if (HasObjectives(gCampaign.Entry.Mode))
733 {
734 AddObjectives(map, mo);
735 }
736
737 if (AreKeysAllowed(gCampaign.Entry.Mode))
738 {
739 AddKeys(map);
740 }
741}
742static void AddObjectives(Map *map, const struct MissionOptions *mo)
743{
744 // Try to add the objectives
745 // If we are unable to place them all, make sure to reduce the totals
746 // in case we create missions that are impossible to complete
747 CA_FOREACH(Objective, o, mo->missionData->Objectives)
748 if (o->Type != OBJECTIVE_COLLECT && o->Type != OBJECTIVE_DESTROY)
749 {
750 continue;
751 }
752 if (o->Type == OBJECTIVE_COLLECT)
753 {
754 for (int i = o->placed; i < o->Count; i++)
755 {
756 if (MapTryPlaceCollectible(
757 map, mo->missionData, mo, _ca_index))
758 {
759 o->placed++;
760 }
761 }
762 }
763 else if (o->Type == OBJECTIVE_DESTROY)
764 {
765 for (int i = o->placed; i < o->Count; i++)
766 {
767 if (MapTryPlaceBlowup(map, mo->missionData, _ca_index))
768 {
769 o->placed++;
770 }
771 }
772 }
773 o->Count = o->placed;
774 if (o->Count < o->Required)
775 {
776 o->Required = o->Count;
777 }
778 CA_FOREACH_END()
779}

816// Check if the target position is completely clear
817// This includes collisions that make the target illegal, such as walls
818// But it also includes item collisions, whether or not the collisions
819// are legal, e.g. item pickups, friendly collisions
820bool MapIsTileAreaClear(Map *map, const Vec2i fullPos, const Vec2i size)
821{
822 const Vec2i realPos = Vec2iFull2Real(fullPos);
823
824 // Wall collision
825 if (IsCollisionWithWall(realPos, size))
826 {
827 return false;
828 }
829
830 // Item collision
831 const Vec2i tv = Vec2iToTile(realPos);
832 Vec2i dv;
833 // Check collisions with all other items on this tile, in all 8 directions
834 for (dv.y = -1; dv.y <= 1; dv.y++)
835 {
836 for (dv.x = -1; dv.x <= 1; dv.x++)
837 {
838 const Vec2i dtv = Vec2iAdd(tv, dv);
839 if (!MapIsTileIn(map, dtv))
840 {
841 continue;
842 }
843 const CArray *tileThings = &MapGetTile(map, dtv)->things;
844 if (tileThings == NULL)
845 {
846 continue;
847 }
848 for (int i = 0; i < (int)tileThings->size; i++)
849 {
850 const TTileItem *ti =
851 ThingIdGetTileItem(CArrayGet(tileThings, i));
852 if (AABBOverlap(
853 realPos, Vec2iNew(ti->x, ti->y), size, ti->size))
854 {
855 return false;
856 }
857 }
858 }
859 }
860
861 return true;
862}

891Vec2i MapSearchTileAround(Map *map, Vec2i start, TileSelectFunc func)
892{
893 if (func(map, start))
894 {
895 return start;
896 }
897 // Search using an expanding box pattern around the goal
898 for (int radius = 1; radius < MAX(map->Size.x, map->Size.y); radius++)
899 {
900 Vec2i tile;
901 for (tile.x = start.x - radius;
902 tile.x <= start.x + radius;
903 tile.x++)
904 {
905 if (tile.x < 0) continue;
906 if (tile.x >= map->Size.x) break;
907 for (tile.y = start.y - radius;
908 tile.y <= start.y + radius;
909 tile.y++)
910 {
911 if (tile.y < 0) continue;
912 if (tile.y >= map->Size.y) break;
913 // Check box; don't check inside
914 if (tile.x != start.x - radius &&
915 tile.x != start.x + radius &&
916 tile.y != start.y - radius &&
917 tile.y != start.y + radius)
918 {
919 continue;
920 }
921 if (func(map, tile))
922 {
923 return tile;
924 }
925 }
926 }
927 }
928 // Should never reach this point; something is very wrong
929 CASSERT(false, "failed to find tile around tile");
930 return Vec2iZero();
931}
