293color_t PaletteToColor(unsigned char idx)
294{
295 color_t color = palette[idx];
296 color.r = (uint8_t)CLAMP(color.r * GAMMA, 0, 255);
297 color.g = (uint8_t)CLAMP(color.g * GAMMA, 0, 255);
298 color.b = (uint8_t)CLAMP(color.b * GAMMA, 0, 255);
299 color.a = 255;
300 return color;
301}
302Uint32 LookupPalette(unsigned char idx)
303{
304 return COLOR2PIXEL(PaletteToColor(idx));
305}
