33Vec2i Vec2iNew(int x, int y)
34{
35 Vec2i v;
36 v.x = x;
37 v.y = y;
38 return v;
39}
40
41Vec2i Vec2iZero(void)
42{
43 return Vec2iNew(0, 0);
44}
45Vec2i Vec2iUnit(void)
46{
47 return Vec2iNew(1, 1);
48}
49Vec2i Vec2iFromPolar(const double r, const double th)
50{
51 return Vec2iNew((int)Round(sin(th) * r), -(int)Round(cos(th) * r));
52}

54Vec2i Vec2iAdd(Vec2i a, Vec2i b)
55{
56 a.x += b.x;
57 a.y += b.y;
58 return a;
59}
60Vec2i Vec2iMinus(Vec2i a, Vec2i b)
61{
62 a.x -= b.x;
63 a.y -= b.y;
64 return a;
65}

67Vec2i Vec2iMult(const Vec2i a, const Vec2i b)
68{

72Vec2i Vec2iScale(Vec2i v, int scalar)
73{
74 v.x *= scalar;

79Vec2i Vec2iScaleD(const Vec2i v, const double scalar)

91Vec2i Vec2iNorm(Vec2i v)
92{
93 double magnitude;
94 if (Vec2iIsZero(v))
95 {
96 return v;
97 }
98 magnitude = sqrt(v.x*v.x + v.y*v.y);
99 v.x = (int)floor(v.x / magnitude + 0.5);
100 v.y = (int)floor(v.y / magnitude + 0.5);
101 return v;
102}

104bool Vec2iEqual(const Vec2i a, const Vec2i b)
105{
106 return a.x == b.x && a.y == b.y;
107}
108bool Vec2iIsZero(const Vec2i v)
109{
110 return Vec2iEqual(v, Vec2iZero());
111}

113Vec2i Vec2iMin(Vec2i a, Vec2i b)
114{
115 return Vec2iNew(MIN(a.x, b.x), MIN(a.y, b.y));

117Vec2i Vec2iMax(Vec2i a, Vec2i b)
118{
119 return Vec2iNew(MAX(a.x, b.x), MAX(a.y, b.y));

121Vec2i Vec2iClamp(Vec2i v, Vec2i lo, Vec2i hi)
122{
123 v.x = CLAMP(v.x, lo.x, hi.x);
124 v.y = CLAMP(v.y, lo.y, hi.y);
125 return v;
126}

128Vec2i Vec2iFull2Real(Vec2i v)
129{
130 return Vec2iScaleDiv(v, 256);

132Vec2i Vec2iReal2Full(Vec2i v)

136Vec2i Vec2iReal2FullCentered(const Vec2i v)
137{
138 return Vec2iAdd(
139 Vec2iScale(v, 256), Vec2iNew(128 * SIGN(v.x), 128 * SIGN(v.y)));

153int Vec2iSqrMagnitude(const Vec2i v)
154{
155 return v.x * v.x + v.y * v.y;
156}

158int DistanceSquared(const Vec2i a, const Vec2i b)
159{
160 const Vec2i d = Vec2iMinus(a, b);
161 return Vec2iSqrMagnitude(d);
162}

164Vec2i CalcClosestPointOnLineSegmentToPoint(
165 Vec2i l1, Vec2i l2, Vec2i p)
166{
167 // Using parametric representation, line l1->l2 is
168 // P(t) = l1 + t(l2 - l1)
169 // Projection of point p on line is
170 // t = ((p.x - l1.x)(l2.x - l1.x) + (p.y - l1.y)(l2.y - l1.y)) / ||l2 - l1||^2
171 int lineDistanceSquared = DistanceSquared(l1, l2);
172 int numerator;
173 double t;
174 Vec2i closestPoint;
175 // Early exit since same point means 0 distance, and div by 0
176 if (lineDistanceSquared == 0)
177 {
178 return l1;
179 }
180 numerator = (p.x - l1.x)*(l2.x - l1.x) + (p.y - l1.y)*(l2.y - l1.y);
181 t = CLAMP(numerator * 1.0 / lineDistanceSquared, 0, 1);
182 closestPoint.x = (int)Round(l1.x + t*(l2.x - l1.x));
183 closestPoint.y = (int)Round(l1.y + t*(l2.y - l1.y));
184 return closestPoint;
185}
