126#define CenterX(w) ((gGraphicsDevice.cachedConfig.Res.x - w) / 2)
127#define CenterY(h) ((gGraphicsDevice.cachedConfig.Res.y - h) / 2)

129#define CenterOf(a, b, w) ((a + (((b - a) - w) / 2)))
130
131#define CenterOfRight(w) CenterOf((gGraphicsDevice.cachedConfig.Res.x / 2), (gGraphicsDevice.cachedConfig.Res.x), w)
