35void MissionConvertToType(Mission *m, Map *map, MapType type)
36{
37 memset(&m->u, 0, sizeof m->u);
38 switch (type)
39 {
40 case MAPTYPE_CLASSIC:
41 // Setup default parameters
42 m->u.Classic.Walls = 10;
43 m->u.Classic.WallLength = 5;
44 m->u.Classic.CorridorWidth = 2;
45 m->u.Classic.Rooms.Count = 10;
46 m->u.Classic.Rooms.Min = 5;
47 m->u.Classic.Rooms.Max = 8;;
48 m->u.Classic.Rooms.Edge = true;
49 m->u.Classic.Rooms.Overlap = true;
50 m->u.Classic.Rooms.Walls = 1;
51 m->u.Classic.Rooms.WallLength = 1;
52 m->u.Classic.Rooms.WallPad = 1;
53 m->u.Classic.Squares = 1;
54 m->u.Classic.Doors.Enabled = true;
55 m->u.Classic.Doors.Min = 1;
56 m->u.Classic.Doors.Max = 2;
57 m->u.Classic.Pillars.Count = 1;
58 m->u.Classic.Pillars.Min = 2;
59 m->u.Classic.Pillars.Max = 3;
60 break;
61 case MAPTYPE_STATIC:
62 {
63 Vec2i v;
64 // Take all the tiles from the current map
65 // and save them in the static map
66 CArrayInit(&m->u.Static.Tiles, sizeof(unsigned short));
67 for (v.y = 0; v.y < m->Size.y; v.y++)
68 {
69 for (v.x = 0; v.x < m->Size.x; v.x++)
70 {
71 unsigned short tile = IMapGet(map, v);
72 CArrayPushBack(&m->u.Static.Tiles, &tile);
73 }
74 }
75 CArrayInit(&m->u.Static.Items, sizeof(MapObjectPositions));
76 CArrayInit(&m->u.Static.Characters, sizeof(CharacterPositions));
77 CArrayInit(&m->u.Static.Objectives, sizeof(ObjectivePositions));
78 CArrayInit(&m->u.Static.Keys, sizeof(KeyPositions));
79 }
80 break;
81 case MAPTYPE_CAVE:
82 // Setup default parameters
83 m->u.Cave.FillPercent = 40;
84 m->u.Cave.Repeat = 4;
85 m->u.Cave.R1 = 5;
86 m->u.Cave.R2 = 2;
87 m->u.Cave.CorridorWidth = 2;
88 m->u.Cave.Rooms.Count = 10;
89 m->u.Cave.Rooms.Min = 5;
90 m->u.Cave.Rooms.Max = 8;;
91 m->u.Cave.Rooms.Edge = true;
92 m->u.Cave.Rooms.Overlap = true;
93 m->u.Cave.Rooms.Walls = 1;
94 m->u.Cave.Rooms.WallLength = 1;
95 m->u.Cave.Rooms.WallPad = 1;
96 m->u.Cave.Squares = 1;
97 break;
98 default:
99 CASSERT(false, "unknown map type");
100 break;
101 }
102 m->Type = type;
103}

105static int IsClear(unsigned short tile)
106{
107 tile &= MAP_MASKACCESS;
108 return tile == MAP_FLOOR || tile == MAP_ROOM || tile == MAP_SQUARE;
109}
110static unsigned short GetTileAt(Mission *m, Vec2i pos)
111{
112 int idx = pos.y * m->Size.x + pos.x;
113 // check for out-of-bounds
114 if (pos.x < 0 || pos.x >= m->Size.x || pos.y < 0 || pos.y >= m->Size.y)
115 {
116 return MAP_NOTHING;
117 }
118 return *(unsigned short *)CArrayGet(&m->u.Static.Tiles, idx);
119}
120// See if the tile located at a position is a door and also needs
121// to be oriented in a certain way
122// If there are walls or doors in the neighbourhood, they can force a certain
123// orientation of the door
124static int HasDoorOrientedAt(Mission *m, Vec2i pos,int isHorizontal)
125{
126 unsigned short tile = GetTileAt(m, pos);
127 if (tile != MAP_DOOR)
128 {
129 return 0;
130 }
131 // Check for walls and doors that force the orientation of the door
132 if (GetTileAt(m, Vec2iNew(pos.x - 1, pos.y)) == MAP_WALL ||
133 GetTileAt(m, Vec2iNew(pos.x - 1, pos.y)) == MAP_DOOR ||
134 GetTileAt(m, Vec2iNew(pos.x + 1, pos.y)) == MAP_WALL ||
135 GetTileAt(m, Vec2iNew(pos.x + 1, pos.y)) == MAP_DOOR)
136 {
137 // There is a horizontal door
138 return isHorizontal;
139 }
140 else if (GetTileAt(m, Vec2iNew(pos.x, pos.y - 1)) == MAP_WALL ||
141 GetTileAt(m, Vec2iNew(pos.x, pos.y - 1)) == MAP_DOOR ||
142 GetTileAt(m, Vec2iNew(pos.x, pos.y + 1)) == MAP_WALL ||
143 GetTileAt(m, Vec2iNew(pos.x, pos.y + 1)) == MAP_DOOR)
144 {
145 // There is a vertical door
146 return !isHorizontal;
147 }
148 // There is a door but it is free to be oriented in any way
149 return 0;
150}
151bool MissionTrySetTile(Mission *m, Vec2i pos, unsigned short tile)
152{
153 if (pos.x < 0 || pos.x >= m->Size.x || pos.y < 0 || pos.y >= m->Size.y)
154 {
155 return false;
156 }
157 CASSERT(m->Type == MAPTYPE_STATIC, "cannot set tile for map type");
158 switch (tile)
159 {
160 case MAP_WALL:
161 // Check that there are no incompatible doors
162 if (HasDoorOrientedAt(m, Vec2iNew(pos.x - 1, pos.y), 0) ||
163 HasDoorOrientedAt(m, Vec2iNew(pos.x + 1, pos.y), 0) ||
164 HasDoorOrientedAt(m, Vec2iNew(pos.x, pos.y - 1), 1) ||
165 HasDoorOrientedAt(m, Vec2iNew(pos.x, pos.y + 1), 1))
166 {
167 // Can't place this wall
168 return false;
169 }
170 break;
171 case MAP_DOOR:
172 {
173 // Check that there is a clear passage through this door
174 int isHClear =
175 IsClear(GetTileAt(m, Vec2iNew(pos.x - 1, pos.y))) &&
176 IsClear(GetTileAt(m, Vec2iNew(pos.x + 1, pos.y)));
177 int isVClear =
178 IsClear(GetTileAt(m, Vec2iNew(pos.x, pos.y - 1))) &&
179 IsClear(GetTileAt(m, Vec2iNew(pos.x, pos.y + 1)));
180 if (!isHClear && !isVClear)
181 {
182 return false;
183 }
184 // Check that there are no incompatible doors
185 if (HasDoorOrientedAt(m, Vec2iNew(pos.x - 1, pos.y), 0) ||
186 HasDoorOrientedAt(m, Vec2iNew(pos.x + 1, pos.y), 0) ||
187 HasDoorOrientedAt(m, Vec2iNew(pos.x, pos.y - 1), 1) ||
188 HasDoorOrientedAt(m, Vec2iNew(pos.x, pos.y + 1), 1))
189 {
190 // Can't place this door
191 return false;
192 }
193 }
194 break;
195 }
196 const int idx = pos.y * m->Size.x + pos.x;
197 *(unsigned short *)CArrayGet(&m->u.Static.Tiles, idx) = tile;
198 return true;
199}

201unsigned short MissionGetTile(Mission *m, Vec2i pos)
202{
203 if (pos.x < 0 || pos.x >= m->Size.x || pos.y < 0 || pos.y >= m->Size.y)
204 {
205 return MAP_UNSET;
206 }
207 int idx = pos.y * m->Size.x + pos.x;
208 return *(unsigned short *)CArrayGet(&m->u.Static.Tiles, idx);
209}

211void MissionStaticLayout(Mission *m, Vec2i oldSize)
212{
213 assert(m->Type == MAPTYPE_STATIC && "invalid map type");
214 // re-layout the static map after a resize
215 // The mission contains the new size; the old dimensions are oldSize
216 // Simply try to "paint" the old tiles to the new mission
217 Vec2i v;
218 CArray oldTiles;
219 CArrayInit(&oldTiles, m->u.Static.Tiles.elemSize);
220 CArrayCopy(&oldTiles, &m->u.Static.Tiles);
221
222 // Clear the tiles first
223 CArrayTerminate(&m->u.Static.Tiles);
224 CArrayInit(&m->u.Static.Tiles, oldTiles.elemSize);
225 for (v.y = 0; v.y < m->Size.y; v.y++)
226 {
227 for (v.x = 0; v.x < m->Size.x; v.x++)
228 {
229 unsigned short tile = MAP_FLOOR;
230 CArrayPushBack(&m->u.Static.Tiles, &tile);
231 }
232 }
233
234 // Paint the old tiles back
235 for (v.y = 0; v.y < m->Size.y; v.y++)
236 {
237 for (v.x = 0; v.x < m->Size.x; v.x++)
238 {
239 if (v.x >= oldSize.x || v.y >= oldSize.y)
240 {
241 MissionTrySetTile(m, v, MAP_NOTHING);
242 }
243 else
244 {
245 int idx = v.y * oldSize.x + v.x;
246 unsigned short *tile = CArrayGet(&oldTiles, idx);
247 MissionTrySetTile(m, v, *tile);
248 }
249 }
250 }
251
252 CArrayTerminate(&oldTiles);
253
254 if (m->u.Static.Start.x >= m->Size.x || m->u.Static.Start.y >= m->Size.y)
255 {
256 m->u.Static.Start = Vec2iZero();
257 }
258}

271static bool TryAddMapObject(
272 Mission *m, const MapObject *mo, const Vec2i pos, CArray *objs)
273{
274 CASSERT(m->Type == MAPTYPE_STATIC, "invalid map type");
275 const unsigned short tile = MissionGetTile(m, pos);
276
277 // Remove any items already there
278 TryRemoveMapObjectAt(pos, objs);
279
280 if (MapObjectIsTileOK(
281 mo, tile, 1, MissionGetTile(m, Vec2iNew(pos.x, pos.y - 1))))
282 {
283 // Check if the item already has an entry, and add to its list
284 // of positions
285 bool hasAdded = false;
286 for (int i = 0; i < (int)objs->size; i++)
287 {
288 MapObjectPositions *mop = CArrayGet(objs, i);
289 if (mop->M == mo)
290 {
291 CArrayPushBack(&mop->Positions, &pos);
292 hasAdded = true;
293 break;
294 }
295 }
296 // If not, create a new entry
297 if (!hasAdded)
298 {
299 MapObjectPositions mop;
300 mop.M = mo;
301 CArrayInit(&mop.Positions, sizeof(Vec2i));
302 CArrayPushBack(&mop.Positions, &pos);
303 CArrayPushBack(objs, &mop);
304 }
305 return true;
306 }
307 return false;
308}
309static bool TryRemoveMapObjectAt(const Vec2i pos, CArray *objs)
310{
311 for (int i = 0; i < (int)objs->size; i++)
312 {
313 MapObjectPositions *mop = CArrayGet(objs, i);
314 for (int j = 0; j < (int)mop->Positions.size; j++)
315 {
316 const Vec2i *mopPos = CArrayGet(&mop->Positions, j);
317 if (Vec2iEqual(*mopPos, pos))
318 {
319 CArrayDelete(&mop->Positions, j);
320 if (mop->Positions.size == 0)
321 {
322 CArrayTerminate(&mop->Positions);
323 CArrayDelete(objs, i);
324 }
325 return true;
326 }
327 }
328 }
329 return false;
330}

332bool MissionStaticTryAddCharacter(Mission *m, int ch, Vec2i pos)
333{
334 assert(m->Type == MAPTYPE_STATIC && "invalid map type");
335 unsigned short tile = MissionGetTile(m, pos);
336
337 // Remove any characters already there
338 MissionStaticTryRemoveCharacterAt(m, pos);
339
340 if (IsClear(tile))
341 {
342 // Check if the character already has an entry, and add to its list
343 // of positions
344 bool hasAdded = false;
345 CA_FOREACH(CharacterPositions, cp, m->u.Static.Characters)
346 if (cp->Index == ch)
347 {
348 CArrayPushBack(&cp->Positions, &pos);
349 hasAdded = true;
350 break;
351 }
352 CA_FOREACH_END()
353 // If not, create a new entry
354 if (!hasAdded)
355 {
356 CharacterPositions cp;
357 cp.Index = ch;
358 CArrayInit(&cp.Positions, sizeof(Vec2i));
359 CArrayPushBack(&cp.Positions, &pos);
360 CArrayPushBack(&m->u.Static.Characters, &cp);
361 }
362 return true;
363 }
364 return false;
365}
366bool MissionStaticTryRemoveCharacterAt(Mission *m, Vec2i pos)
367{
368 CA_FOREACH(CharacterPositions, cp, m->u.Static.Characters)
369 for (int j = 0; j < (int)cp->Positions.size; j++)
370 {
371 Vec2i *cpPos = CArrayGet(&cp->Positions, j);
372 if (Vec2iEqual(*cpPos, pos))
373 {
374 CArrayDelete(&cp->Positions, j);
375 if (cp->Positions.size == 0)
376 {
377 CArrayTerminate(&cp->Positions);
378 CArrayDelete(&m->u.Static.Characters, _ca_index);
379 }
380 return true;
381 }
382 }
383 CA_FOREACH_END()
384 return false;
385}

387bool MissionStaticTryAddObjective(Mission *m, int idx, int idx2, Vec2i pos)
388{
389 assert(m->Type == MAPTYPE_STATIC && "invalid map type");
390 unsigned short tile = MissionGetTile(m, pos);
391
392 // Remove any objectives already there
393 MissionStaticTryRemoveObjectiveAt(m, pos);
394
395 if (IsClear(tile))
396 {
397 // Check if the objective already has an entry, and add to its list
398 // of positions
399 int hasAdded = 0;
400 int objectiveIndex = -1;
401 ObjectivePositions *op = NULL;
402 for (int i = 0; i < (int)m->u.Static.Objectives.size; i++)
403 {
404 op = CArrayGet(&m->u.Static.Objectives, i);
405 if (op->Index == idx)
406 {
407 CArrayPushBack(&op->Positions, &pos);
408 CArrayPushBack(&op->Indices, &idx2);
409 objectiveIndex = op->Index;
410 hasAdded = 1;
411 break;
412 }
413 }
414 // If not, create a new entry
415 if (!hasAdded)
416 {
417 ObjectivePositions newOp;
418 newOp.Index = idx;
419 CArrayInit(&newOp.Positions, sizeof(Vec2i));
420 CArrayInit(&newOp.Indices, sizeof(int));
421 CArrayPushBack(&newOp.Positions, &pos);
422 CArrayPushBack(&newOp.Indices, &idx2);
423 objectiveIndex = (int)newOp.Positions.size - 1;
424 CArrayPushBack(&m->u.Static.Objectives, &newOp);
425 }
426 // Increase number of objectives
427 Objective *o = CArrayGet(&m->Objectives, objectiveIndex);
428 o->Count++;
429 return true;
430 }
431 return false;
432}
433bool MissionStaticTryRemoveObjectiveAt(Mission *m, Vec2i pos)
434{
435 CA_FOREACH(ObjectivePositions, op, m->u.Static.Objectives)
436 for (int j = 0; j < (int)op->Positions.size; j++)
437 {
438 Vec2i *opPos = CArrayGet(&op->Positions, j);
439 if (Vec2iEqual(*opPos, pos))
440 {
441 CArrayDelete(&op->Positions, j);
442 CArrayDelete(&op->Indices, j);
443 // Decrease number of objectives
444 Objective *o = CArrayGet(&m->Objectives, op->Index);
445 o->Count--;
446 CASSERT(o->Count >= 0, "removing unknown objective");
447 if (op->Positions.size == 0)
448 {
449 CArrayTerminate(&op->Positions);
450 CArrayTerminate(&op->Indices);
451 CArrayDelete(&m->u.Static.Objectives, _ca_index);
452 }
453 return true;
454 }
455 }
456 CA_FOREACH_END()
457 return false;
458}

515typedef struct
516{
517 Mission *m;
518 unsigned short mask;
519} MissionFloodFillData;
520static void MissionFillTile(void *data, Vec2i v);
521static bool MissionIsTileSame(void *data, Vec2i v);
522bool MissionStaticTrySetKey(Mission *m, int k, Vec2i pos)
523{
524 assert(m->Type == MAPTYPE_STATIC && "invalid map type");
525 unsigned short mask = GetAccessMask(k);
526 FloodFillData data;
527 data.Fill = MissionFillTile;
528 data.IsSame = MissionIsTileSame;
529 MissionFloodFillData mData;
530 mData.m = m;
531 mData.mask = mask;
532 data.data = &mData;
533 return CFloodFill(pos, &data);
534}
535static void MissionFillTile(void *data, Vec2i v)
536{
537 MissionFloodFillData *mData = data;
538 MissionTrySetTile(mData->m, v, MAP_DOOR | mData->mask);
539}
540static bool MissionIsTileSame(void *data, Vec2i v)
541{
542 MissionFloodFillData *mData = data;
543 unsigned short tile = MissionGetTile(mData->m, v);
544 return (tile & MAP_MASKACCESS) == MAP_DOOR &&
545 (tile & MAP_ACCESSBITS) != mData->mask;
546}
