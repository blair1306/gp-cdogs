41void MapStaticLoad(Map *map, const struct MissionOptions *mo)
42{
43 const Mission *m = mo->missionData;
44
45 // Tiles
46 Vec2i v;
47 for (v.y = 0; v.y < m->Size.y; v.y++)
48 {
49 for (v.x = 0; v.x < m->Size.x; v.x++)
50 {
51 int idx = v.y * m->Size.x + v.x;
52 unsigned short tile =
53 *(unsigned short *)CArrayGet(&m->u.Static.Tiles, idx);
54 if (!AreKeysAllowed(gCampaign.Entry.Mode))
55 {
56 tile &= MAP_MASKACCESS;
57 }
58 IMapSet(map, v, tile);
59 }
60 }
61
62 // Exit area
63 if (!Vec2iIsZero(m->u.Static.Exit.Start) &&
64 !Vec2iIsZero(m->u.Static.Exit.End))
65 {
66 map->ExitStart = m->u.Static.Exit.Start;
67 map->ExitEnd = m->u.Static.Exit.End;
68 }
69}

77void MapStaticLoadDynamic(
78 Map *map, const struct MissionOptions *mo, const CharacterStore *store)
79{
80 const Mission *m = mo->missionData;
81
82 // Map objects
83 CA_FOREACH(const MapObjectPositions, mop, m->u.Static.Items)
84 for (int j = 0; j < (int)mop->Positions.size; j++)
85 {
86 const Vec2i *pos = CArrayGet(&mop->Positions, j);
87 MapTryPlaceOneObject(map, *pos, mop->M, 0, false);
88 }
89 CA_FOREACH_END()
90
91 if (ModeHasNPCs(gCampaign.Entry.Mode))
92 {
93 AddCharacters(&m->u.Static.Characters);
94 }
95
96 if (HasObjectives(gCampaign.Entry.Mode))
97 {
98 AddObjectives(map, mo, store, &m->u.Static.Objectives);
99 }
100
101 if (AreKeysAllowed(gCampaign.Entry.Mode))
102 {
103 AddKeys(map, mo, &m->u.Static.Keys);
104 }
105
106 // Process the events to place dynamic objects
107 HandleGameEvents(&gGameEvents, NULL, NULL, NULL);
108}

116static void AddCharacter(const CharacterPositions *cp)
117{
118 NActorAdd aa = NActorAdd_init_default;
119 aa.CharId = cp->Index;
120 const Character *c =
121 CArrayGet(&gCampaign.Setting.characters.OtherChars, aa.CharId);
122 aa.Health = CharacterGetStartingHealth(c, true);
123 CA_FOREACH(const Vec2i, pos, cp->Positions)
124 aa.UID = ActorsGetNextUID();
125 aa.Direction = rand() % DIRECTION_COUNT;
126 const Vec2i fullPos = Vec2iReal2Full(Vec2iCenterOfTile(*pos));
127 aa.FullPos = Vec2i2Net(fullPos);
128
129 GameEvent e = GameEventNew(GAME_EVENT_ACTOR_ADD);
130 e.u.ActorAdd = aa;
131 GameEventsEnqueue(&gGameEvents, e);
132 CA_FOREACH_END()
133}

145static void AddObjective(
146 Map *map, const struct MissionOptions *mo, const CharacterStore *store,
147 const ObjectivePositions *op)
148{
149 if (op->Index >= (int)mo->missionData->Objectives.size)
150 {
151 LOG(LM_MAP, LL_ERROR, "cannot add objective; objective #%d missing",
152 op->Index);
153 return;
154 }
155 Objective *o = CArrayGet(&mo->missionData->Objectives, op->Index);
156 CA_FOREACH(const Vec2i, pos, op->Positions)
157 const int *idx = CArrayGet(&op->Indices, _ca_index);
158 const Vec2i realPos = Vec2iCenterOfTile(*pos);
159 const Vec2i fullPos = Vec2iReal2Full(realPos);
160 switch (o->Type)
161 {
162 case OBJECTIVE_KILL:
163 {
164 NActorAdd aa = NActorAdd_init_default;
165 aa.UID = ActorsGetNextUID();
166 aa.CharId = CharacterStoreGetSpecialId(store, *idx);
167 aa.TileItemFlags = ObjectiveToTileItem(op->Index);
168 aa.Direction = rand() % DIRECTION_COUNT;
169 const Character *c =
170 CArrayGet(&gCampaign.Setting.characters.OtherChars, aa.CharId);
171 aa.Health = CharacterGetStartingHealth(c, true);
172 aa.FullPos = Vec2i2Net(fullPos);
173 GameEvent e = GameEventNew(GAME_EVENT_ACTOR_ADD);
174 e.u.ActorAdd = aa;
175 GameEventsEnqueue(&gGameEvents, e);
176 }
177 break;
178 case OBJECTIVE_COLLECT:
179 MapPlaceCollectible(mo, op->Index, realPos);
180 break;
181 case OBJECTIVE_DESTROY:
182 MapTryPlaceOneObject(
183 map,
184 *pos,
185 o->u.MapObject,
186 ObjectiveToTileItem(op->Index), false);
187 break;
188 case OBJECTIVE_RESCUE:
189 {
190 NActorAdd aa = NActorAdd_init_default;
191 aa.UID = ActorsGetNextUID();
192 aa.CharId = CharacterStoreGetPrisonerId(store, *idx);
193 aa.TileItemFlags = ObjectiveToTileItem(op->Index);
194 aa.Direction = rand() % DIRECTION_COUNT;
195 const Character *c =
196 CArrayGet(&gCampaign.Setting.characters.OtherChars, aa.CharId);
197 aa.Health = CharacterGetStartingHealth(c, true);
198 aa.FullPos = Vec2i2Net(fullPos);
199 GameEvent e = GameEventNew(GAME_EVENT_ACTOR_ADD);
200 e.u.ActorAdd = aa;
201 GameEventsEnqueue(&gGameEvents, e);
202 }
203 break;
204 default:
205 // do nothing
206 break;
207 }
208 o->placed++;
209 CA_FOREACH_END()
210}
