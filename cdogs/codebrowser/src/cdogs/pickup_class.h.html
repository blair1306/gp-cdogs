36// Effects for "pick up" objects
37typedef enum
38{
39 PICKUP_NONE,
40 PICKUP_JEWEL,
41 PICKUP_HEALTH,
42 PICKUP_AMMO,
43 PICKUP_KEYCARD,
44 PICKUP_GUN
45} PickupType;
46PickupType StrPickupType(const char *s);
47
48typedef struct
49{
50 char *Name;
51 PickupType Type;
52 union
53 {
54 int Score;
55 int Health;
56 AddAmmo Ammo;
57 int Keys; // Refer to flags in mission.h
58 int GunId;
59 } u;
60 const Pic *Pic;
61} PickupClass;
62typedef struct
63{
64 CArray Classes; // of PickupClass
65 CArray CustomClasses; // of PickupClass
66 CArray KeyClasses; // of PickupClass
67} PickupClasses;
68extern PickupClasses gPickupClasses;
