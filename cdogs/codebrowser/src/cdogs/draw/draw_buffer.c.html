57void DrawBufferInit(DrawBuffer *b, Vec2i size, GraphicsDevice *g)
58{
59 debug(D_MAX, "Initialising draw buffer %dx%d\n", size.x, size.y);
60 b->OrigSize = size;
61 CMALLOC(b->tiles, size.x * sizeof *b->tiles);
62 CMALLOC(b->tiles[0], size.x * size.y * sizeof *b->tiles[0]);
63 for (int i = 1; i < size.x; i++)
64 {
65 b->tiles[i] = b->tiles[0] + i * size.y;
66 }
67 b->g = g;
68 CArrayInit(&b->displaylist, sizeof(const TTileItem *));
69 CArrayReserve(&b->displaylist, 32);
70 debug(D_MAX, "Initialised draw buffer %dx%d\n", size.x, size.y);
71}
72void DrawBufferTerminate(DrawBuffer *b)
73{
74 CFREE(b->tiles[0]);
75 CFREE(b->tiles);
76 CArrayTerminate(&b->displaylist);
77}

79void DrawBufferSetFromMap(
80 DrawBuffer *buffer, Map *map, Vec2i origin, int width)
81{
82 int x, y;
83 Tile *bufTile;
84
85 buffer->Size = Vec2iNew(width, buffer->OrigSize.y);
86
87 buffer->xTop = origin.x - TILE_WIDTH * width / 2;
88 //buffer->yTop = y_origin - 100;
89 buffer->yTop = origin.y - TILE_HEIGHT * buffer->OrigSize.y / 2;
90
91 buffer->xStart = buffer->xTop / TILE_WIDTH;
92 buffer->yStart = buffer->yTop / TILE_HEIGHT;
93 if (buffer->xTop < 0)
94 {
95 buffer->xStart--;
96 }
97 if (buffer->yTop < 0)
98 {
99 buffer->yStart--;
100 }
101
102 buffer->dx = buffer->xStart * TILE_WIDTH - buffer->xTop;
103 buffer->dy = buffer->yStart * TILE_HEIGHT - buffer->yTop;
104
105 bufTile = &buffer->tiles[0][0];
106 for (y = buffer->yStart; y < buffer->yStart + buffer->Size.y; y++)
107 {
108 for (x = buffer->xStart;
109 x < buffer->xStart + buffer->Size.x;
110 x++, bufTile++)
111 {
112 if (x >= 0 && x < map->Size.x && y >= 0 && y < map->Size.y)
113 {
114 *bufTile = *MapGetTile(map, Vec2iNew(x, y));
115 }
116 else
117 {
118 *bufTile = TileNone();
119 }
120 }
121 bufTile += buffer->OrigSize.x - buffer->Size.x;
122 }
123}

125// Set visibility and draw order for wall/door columns
126void DrawBufferFix(DrawBuffer *buffer)
127{
128 Tile *tile = &buffer->tiles[0][0];
129 Tile *tileBelow = &buffer->tiles[0][0] + X_TILES;
130 for (int y = 0; y < Y_TILES - 1; y++)
131 {
132 for (int x = 0; x < buffer->Size.x; x++, tile++, tileBelow++)
133 {
134 if (!(tile->flags & (MAPTILE_IS_WALL | MAPTILE_OFFSET_PIC)) &&
135 (tileBelow->flags & MAPTILE_IS_WALL))
136 {
137 tile->pic = NULL;
138 }
139 else if ((tile->flags & MAPTILE_IS_WALL) &&
140 (tileBelow->flags & MAPTILE_IS_WALL))
141 {
142 tile->flags |= MAPTILE_DELAY_DRAW;
143 }
144 }
145 tile += X_TILES - buffer->Size.x;
146 tileBelow += X_TILES - buffer->Size.x;
147 }
148
149 tile = &buffer->tiles[0][0];
150 for (int y = 0; y < Y_TILES; y++)
151 {
152 for (int x = 0; x < buffer->Size.x; x++, tile++)
153 {
154 const Vec2i mapTile =
155 Vec2iNew(x + buffer->xStart, y + buffer->yStart);
156 if (!LOSTileIsVisible(&gMap, mapTile))
157 {
158 tile->flags |= MAPTILE_OUT_OF_SIGHT;
159 }
160 else
161 {
162 tile->flags &= ~MAPTILE_OUT_OF_SIGHT;
163 }
164 }
165 tile += X_TILES - buffer->Size.x;
166 }
167}

169static int CompareY(const void *v1, const void *v2);
170void DrawBufferSortDisplayList(DrawBuffer *buffer)
171{
172 qsort(
173 buffer->displaylist.data,
174 buffer->displaylist.size,
175 buffer->displaylist.elemSize,
176 CompareY);
177}
178static int CompareY(const void *v1, const void *v2)
179{
180 const TTileItem * const *t1 = v1;
181 const TTileItem * const *t2 = v2;
182 if ((*t1)->y < (*t2)->y)
183 {
184 return -1;
185 }
186 else if ((*t1)->y >(*t2)->y)
187 {
188 return 1;
189 }
190 return 0;
191}
