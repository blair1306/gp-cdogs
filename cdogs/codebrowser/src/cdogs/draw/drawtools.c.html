59void Draw_Point(const int x, const int y, color_t c)
60{
61 Uint32 *screen = gGraphicsDevice.buf;
62 int idx = PixelIndex(
63 x,
64 y,
65 gGraphicsDevice.cachedConfig.Res.x,
66 gGraphicsDevice.cachedConfig.Res.y);
67 if (x < gGraphicsDevice.clipping.left ||
68 x > gGraphicsDevice.clipping.right ||
69 y < gGraphicsDevice.clipping.top ||
70 y > gGraphicsDevice.clipping.bottom)
71 {
72 return;
73 }
74 if (c.a == 255)
75 {
76 screen[idx] = COLOR2PIXEL(c);
77 }
78 else
79 {
80 const color_t existing = PIXEL2COLOR(screen[idx]);
81 screen[idx] = COLOR2PIXEL(ColorAlphaBlend(existing, c));
82 }
83}

85static
86void
87Draw_StraightLine(
88 const int x1, const int y1, const int x2, const int y2, color_t c)
89{
90 register int i;
91 register int start, end;
92
93 if (x1 == x2) { /* vertical line */
94 if (y2 > y1) {
95 start = y1;
96 end = y2;
97 } else {
98 start = y2;
99 end = y1;
100 }
101
102 for (i = start; i <= end; i++) {
103 Draw_Point(x1, i, c);
104 }
105 } else if (y1 == y2) { /* horizontal line */
106 if (x2 > x1) {
107 start = x1;
108 end = x2;
109 } else {
110 start = x2;
111 end = x1;
112 }
113
114 for (i = start; i <= end; i++) {
115 Draw_Point(i, y1, c);
116 }
117 }
118 return;
119}

121#define ABS(x) ( x > 0 ? x : (-x) )
122
123static void Draw_DiagonalLine(const int x1, const int x2, color_t c)
124{
125 register int i;
126 register int start, end;
127
128 if (x1 < x2) {
129 start = x1;
130 end = x2;
131 } else {
132 start = x2;
133 end = x1;
134 }
135
136 for (i = start; i < end; i++) {
137 Draw_Point(i, i, c);
138 }
139
140 return;
141}

143static void DrawPointFunc(void *data, const Vec2i pos);
144void DrawLine(const Vec2i from, const Vec2i to, color_t c)
145{
146 AlgoLineDrawData data;
147 data.Draw = DrawPointFunc;
148 data.data = &c;
149 BresenhamLineDraw(from, to, &data);
150}
151static void DrawPointFunc(void *data, const Vec2i pos)
152{
153 const color_t *c = data;
154 Draw_Point(pos.x, pos.y, *c);
155}
156
157void Draw_Line(
158 const int x1, const int y1, const int x2, const int y2, color_t c)
159{
160 //debug("(%d, %d) -> (%d, %d)\n", x1, y1, x2, y2);
161
162 if (x1 == x2 || y1 == y2)
163 Draw_StraightLine(x1, y1, x2, y2, c);
164 else if (ABS((x2 - x1)) == ABS((y1 - y2)))
165 Draw_DiagonalLine(x1, x2, c);
166 /*else
167 Draw_OtherLine(x1, y1, x2, y2, c);*/
168
169 return;
170}

172void DrawPointMask(GraphicsDevice *g, Vec2i pos, color_t mask)
173{
174 if (pos.x < g->clipping.left || pos.x > g->clipping.right ||
175 pos.y < g->clipping.top || pos.y > g->clipping.bottom)
176 {
177 return;
178 }
179 const int idx = PixelIndex(
180 pos.x, pos.y, g->cachedConfig.Res.x, g->cachedConfig.Res.y);
181 Uint32 *screen = g->buf;
182 color_t c = PIXEL2COLOR(screen[idx]);
183 c = ColorMult(c, mask);
184 screen[idx] = COLOR2PIXEL(c);
185}

187void DrawPointTint(GraphicsDevice *device, Vec2i pos, HSV tint)
188{
189 Uint32 *screen = device->buf;
190 int idx = PixelIndex(
191 pos.x, pos.y,
192 device->cachedConfig.Res.x,
193 device->cachedConfig.Res.y);
194 color_t c;
195 if (pos.x < device->clipping.left || pos.x > device->clipping.right ||
196 pos.y < device->clipping.top || pos.y > device->clipping.bottom)
197 {
198 return;
199 }
200 c = PIXEL2COLOR(screen[idx]);
201 c = ColorTint(c, tint);
202 screen[idx] = COLOR2PIXEL(c);
203}

205void DrawRectangle(
206 GraphicsDevice *device, Vec2i pos, Vec2i size, color_t color, int flags)
207{
208 int y;
209 if (size.x < 3 || size.y < 3)
210 {
211 flags &= ~DRAW_FLAG_ROUNDED;
212 }
213 for (y = MAX(pos.y, device->clipping.top);
214 y < MIN(pos.y + size.y, device->clipping.bottom + 1);
215 y++)
216 {
217 int isFirstOrLastLine = y == pos.y || y == pos.y + size.y - 1;
218 if (isFirstOrLastLine && (flags & DRAW_FLAG_ROUNDED))
219 {
220 int x;
221 for (x = MAX(pos.x + 1, device->clipping.left);
222 x < MIN(pos.x + size.x - 1, device->clipping.right + 1);
223 x++)
224 {
225 Draw_Point(x, y, color);
226 }
227 }
228 else if (!isFirstOrLastLine && (flags & DRAW_FLAG_LINE))
229 {
230 Draw_Point(pos.x, y, color);
231 Draw_Point(pos.x + size.x - 1, y, color);
232 }
233 else
234 {
235 int x;
236 for (x = MAX(pos.x, device->clipping.left);
237 x < MIN(pos.x + size.x, device->clipping.right + 1);
238 x++)
239 {
240 Draw_Point(x, y, color);
241 }
242 }
243 }

246void DrawCross(GraphicsDevice *device, int x, int y, color_t color)
247{
248 Uint32 *screen = device->buf;
249 const Uint32 pixel = COLOR2PIXEL(color);
250 screen += x;
251 screen += y * gGraphicsDevice.cachedConfig.Res.x;
252 *screen = pixel;
253 *(screen - 1) = pixel;
254 *(screen + 1) = pixel;
255 *(screen - gGraphicsDevice.cachedConfig.Res.x) = pixel;
256 *(screen + gGraphicsDevice.cachedConfig.Res.x) = pixel;
257}

259void DrawShadow(GraphicsDevice *device, Vec2i pos, Vec2i size)
260{
261 if (!ConfigGetBool(&gConfig, "Graphics.Shadows"))
262 {
263 return;
264 }
265 Vec2i drawPos;
266 for (drawPos.y = pos.y - size.y; drawPos.y < pos.y + size.y; drawPos.y++)
267 {
268 if (drawPos.y >= device->clipping.bottom)
269 {
270 break;
271 }
272 if (drawPos.y < device->clipping.top)
273 {
274 continue;
275 }
276 for (drawPos.x = pos.x - size.x; drawPos.x < pos.x + size.x; drawPos.x++)
277 {
278 // Calculate value tint based on distance from center
279 Vec2i scaledPos;
280 int distance2;
281 if (drawPos.x >= device->clipping.right)
282 {
283 break;
284 }
285 if (drawPos.x < device->clipping.left)
286 {
287 continue;
288 }
289 scaledPos.x = drawPos.x;
290 scaledPos.y = (drawPos.y - pos.y) * size.x / size.y + pos.y;
291 distance2 = DistanceSquared(scaledPos, pos);
292 // Maximum distance is x, so scale distance squared by x squared
293 const HSV tint =
294 {
295 -1.0, 1.0, CLAMP(distance2 * 1.0 / (size.x*size.x), 0.0, 1.0)
296 };
297 DrawPointTint(device, drawPos, tint);
298 }
