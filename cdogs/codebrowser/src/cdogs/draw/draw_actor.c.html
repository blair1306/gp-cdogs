70static Vec2i GetActorDrawOffset(
71 const Pic *pic, const BodyPart part, const CharSprites *cs,
72 const ActorAnimation anim, const int frame, const direction_e d)
73{
74 Vec2i offset = Vec2iScaleDiv(pic->size, -2);
75 offset = Vec2iMinus(offset, CharSpritesGetOffset(
76 cs->Offsets.Frame[part],
77 anim == ACTORANIMATION_WALKING ? "run" : "idle",
78 frame));
79 offset = Vec2iAdd(offset, cs->Offsets.Dir[part][d]);
80 return offset;
81}

83static Character *ActorGetCharacterMutable(TActor *a);
84ActorPics GetCharacterPicsFromActor(TActor *a)
85{
86 const Weapon *gun = ActorGetGun(a);
87 HSV *tint = NULL;
88 color_t *mask = NULL;
89 if (a->flamed)
90 {
91 tint = &tintRed;
92 mask = &colorRed;
93 }
94 else if (a->poisoned)
95 {
96 tint = &tintPoison;
97 mask = &colorPoison;
98 }
99 else if (a->petrified)
100 {
101 tint = &tintGray;
102 mask = &colorGray;
103 }
104 else if (a->confused)
105 {
106 tint = &tintPurple;
107 mask = &colorPurple;
108 }
109 return GetCharacterPics(
110 ActorGetCharacterMutable(a),
111 RadiansToDirection(a->DrawRadians), a->anim.Type,
112 AnimationGetFrame(&a->anim),
113 gun->Gun->Pic, gun->state,
114 !!(a->flags & FLAGS_SEETHROUGH),
115 tint, mask,
116 a->dead);
117}
118static const Pic *GetBodyPic(
119 PicManager *pm, const CharSprites *cs, const direction_e dir,
120 const ActorAnimation anim, const int frame, const bool isArmed);
121static const Pic *GetLegsPic(
122 PicManager *pm, const CharSprites *cs, const direction_e dir,

124static const Pic *GetGunPic(
125 const NamedSprites *gunPics, const direction_e dir, const int gunState);
126static const Pic *GetDeathPic(PicManager *pm, const int frame);
127ActorPics GetCharacterPics(
128 Character *c, const direction_e dir,
129 const ActorAnimation anim, const int frame,
130 const NamedSprites *gunPics, const gunstate_e gunState,
131 const bool isTransparent, HSV *tint, color_t *mask,
132 const int deadPic)
133{
134 ActorPics pics;
135 memset(&pics, 0, sizeof pics);
136 pics.Colors = &c->Colors;
137
138 // Dummy return to handle invalid character class
139 if (c->Class == NULL)
140 {
141 pics.IsDead = true;
142 pics.IsDying = true;
143 pics.Body = GetDeathPic(&gPicManager, 0);
144 pics.OrderedPics[0] = pics.Body;
145 return pics;
146 }
147
148 // If the actor is dead, simply draw a dying animation
149 pics.IsDead = deadPic > 0;
150 if (pics.IsDead)
151 {
152 if (deadPic < DEATH_MAX)
153 {
154 pics.IsDying = true;
155 pics.Body = GetDeathPic(&gPicManager, deadPic - 1);
156 pics.OrderedPics[0] = pics.Body;
157 }
158 return pics;
159 }
160
161 pics.IsTransparent = isTransparent;
162 if (pics.IsTransparent)
163 {
164 pics.Tint = &tintDarker;
165 }
166 else if (tint != NULL)
167 {
168 pics.Tint = tint;
169 pics.Mask = mask;
170 }
171
172 // Head
173 direction_e headDir = dir;
174 // If idle, turn head left/right on occasion
175 if (anim == ACTORANIMATION_IDLE)
176 {
177 if (frame == IDLEHEAD_LEFT) headDir = (dir + 7) % 8;
178 else if (frame == IDLEHEAD_RIGHT) headDir = (dir + 1) % 8;
179 }
180 pics.Head = GetHeadPic(c->Class, headDir, gunState);
181 pics.HeadOffset = GetActorDrawOffset(
182 pics.Head, BODY_PART_HEAD, c->Class->Sprites, anim, frame, dir);
183
184 // Body
185 const bool isArmed = gunPics != NULL;
186 pics.Body = GetBodyPic(
187 &gPicManager, c->Class->Sprites, dir, anim, frame, isArmed);
188 pics.BodyOffset = GetActorDrawOffset(
189 pics.Body, BODY_PART_BODY, c->Class->Sprites, anim, frame, dir);
190
191 // Legs
192 pics.Legs = GetLegsPic(
193 &gPicManager, c->Class->Sprites, dir, anim, frame);
194 pics.LegsOffset = GetActorDrawOffset(
195 pics.Legs, BODY_PART_LEGS, c->Class->Sprites, anim, frame, dir);
196
197 // Gun
198 pics.Gun = NULL;
199 if (isArmed)
200 {
201 pics.Gun = GetGunPic(gunPics, dir, gunState);
202 pics.GunOffset = GetActorDrawOffset(
203 pics.Gun, BODY_PART_GUN, c->Class->Sprites, anim, frame, dir);
204 }
205
206 // Determine draw order based on the direction the player is facing
207 for (BodyPart bp = BODY_PART_HEAD; bp < BODY_PART_COUNT; bp++)
208 {
209 const BodyPart drawOrder = c->Class->Sprites->Order[dir][bp];
210 switch (drawOrder)
211 {
212 case BODY_PART_HEAD:
213 pics.OrderedPics[bp] = pics.Head;
214 pics.OrderedOffsets[bp] = pics.HeadOffset;
215 break;
216 case BODY_PART_BODY:
217 pics.OrderedPics[bp] = pics.Body;
218 pics.OrderedOffsets[bp] = pics.BodyOffset;
219 break;
220 case BODY_PART_LEGS:
221 pics.OrderedPics[bp] = pics.Legs;
222 pics.OrderedOffsets[bp] = pics.LegsOffset;
223 break;
224 case BODY_PART_GUN:
225 pics.OrderedPics[bp] = pics.Gun;
226 pics.OrderedOffsets[bp] = pics.GunOffset;
227 break;
228 default:
229 break;
230 }
231 }
232
233 pics.Sprites = c->Class->Sprites;
234
235 return pics;
236}

246static void DrawDyingBody(
247 GraphicsDevice *g, const ActorPics *pics, const Vec2i pos);
248void DrawActorPics(const ActorPics *pics, const Vec2i pos)
249{
250 if (pics->IsDead)
251 {
252 if (pics->IsDying)
253 {
254 DrawDyingBody(&gGraphicsDevice, pics, pos);
255 }
256 }
257 else
258 {
259 // Draw shadow
260 if (!pics->IsTransparent)
261 {
262 DrawShadow(&gGraphicsDevice, pos, Vec2iNew(8, 6));
263 }
264 for (int i = 0; i < BODY_PART_COUNT; i++)
265 {
266 const Pic *pic = pics->OrderedPics[i];
267 if (pic == NULL)
268 {
269 continue;
270 }
271 const Vec2i drawPos = Vec2iAdd(pos, pics->OrderedOffsets[i]);
272 if (pics->IsTransparent)
273 {
274 BlitBackground(
275 &gGraphicsDevice, pic, drawPos, pics->Tint, true);
276 }
277 else if (pics->Mask != NULL)
278 {
279 BlitMasked(&gGraphicsDevice, pic, drawPos, *pics->Mask, true);
280 }
281 else
282 {
283 BlitCharMultichannel(
284 &gGraphicsDevice, pic, drawPos, pics->Colors);
285 }
286 }
287 }

289static void DrawLaserSightSingle(
290 const Vec2i from, const double radians, const int range,
291 const color_t color);
292void DrawLaserSight(
293 const ActorPics *pics, const TActor *a, const Vec2i picPos)
294{
295 // Don't draw if dead or transparent
296 if (pics->IsDead || pics->IsTransparent) return;
297 // Check config
298 const LaserSight ls = ConfigGetEnum(&gConfig, "Game.LaserSight");
299 if (ls != LASER_SIGHT_ALL &&
300 !(ls == LASER_SIGHT_PLAYERS && a->PlayerUID >= 0))
301 {
302 return;
303 }
304 // Draw weapon indicators
305 const GunDescription *g = ActorGetGun(a)->Gun;
306 Vec2i muzzlePos = Vec2iAdd(
307 picPos, Vec2iFull2Real(ActorGetGunMuzzleOffset(a)));
308 muzzlePos.y -= g->MuzzleHeight / Z_FACTOR;
309 const double radians = dir2radians[a->direction] + g->AngleOffset;
310 const int range = GunGetRange(g);
311 color_t color = colorCyan;
312 color.a = 64;
313 const double spreadHalf =
314 (g->Spread.Count - 1) * g->Spread.Width / 2 + g->Recoil / 2;
315 if (spreadHalf > 0)
316 {
317 DrawLaserSightSingle(muzzlePos, radians - spreadHalf, range, color);
318 DrawLaserSightSingle(muzzlePos, radians + spreadHalf, range, color);
319 }
320 else
321 {
322 DrawLaserSightSingle(muzzlePos, radians, range, color);
323 }
324}
325static void DrawLaserSightSingle(
326 const Vec2i from, const double radians, const int range,
327 const color_t color)
328{
329 double x, y;
330 GetVectorsForRadians(radians, &x, &y);
331 const Vec2i to = Vec2iAdd(
332 from, Vec2iNew((int)round(x * range), (int)round(y * range)));
333 DrawLine(from, to, color);
334}

336void DrawActorHighlight(
337 const ActorPics *pics, const Vec2i pos, const color_t color)
338{
339 // Do not highlight dead, dying or transparent characters
340 if (pics->IsDead || pics->IsTransparent)
341 {
342 return;
343 }
344 BlitPicHighlight(
345 &gGraphicsDevice, pics->Head, Vec2iAdd(pos, pics->HeadOffset), color);
346 if (pics->Body != NULL)
347 {
348 BlitPicHighlight(
349 &gGraphicsDevice, pics->Body, Vec2iAdd(pos, pics->BodyOffset),
350 color);
351 }
352 if (pics->Legs != NULL)
353 {
354 BlitPicHighlight(
355 &gGraphicsDevice, pics->Legs, Vec2iAdd(pos, pics->LegsOffset),
356 color);
357 }
358 if (pics->Gun != NULL)
359 {
360 BlitPicHighlight(
361 &gGraphicsDevice, pics->Gun, Vec2iAdd(pos, pics->GunOffset), color);
362 }
363}

365static void DrawChatter(
366 const TTileItem *ti, DrawBuffer *b, const Vec2i offset);
367void DrawChatters(DrawBuffer *b, const Vec2i offset)
368{
369 const Tile *tile = &b->tiles[0][0];
370 for (int y = 0; y < Y_TILES; y++)
371 {
372 for (int x = 0; x < b->Size.x; x++, tile++)
373 {
374 CA_FOREACH(ThingId, tid, tile->things)
375 const TTileItem *ti = ThingIdGetTileItem(tid);
376 if (ti->kind != KIND_CHARACTER)
377 {
378 continue;
379 }
380 DrawChatter(ti, b, offset);
381 CA_FOREACH_END()
382 }
383 tile += X_TILES - b->Size.x;
384 }
385}
386#define ACTOR_HEIGHT 25
387static void DrawChatter(
388 const TTileItem *ti, DrawBuffer *b, const Vec2i offset)
389{
390 if (!ConfigGetBool(&gConfig, "Graphics.ShowHUD"))
391 {
392 return;
393 }
394
395 const TActor *a = CArrayGet(&gActors, ti->id);
396 // Draw character text
397 if (strlen(a->Chatter) > 0)
398 {
399 const Vec2i textPos = Vec2iNew(
400 a->tileItem.x - b->xTop + offset.x -
401 FontStrW(a->Chatter) / 2,
402 a->tileItem.y - b->yTop + offset.y - ACTOR_HEIGHT);
403 FontStr(a->Chatter, textPos);
404 }
405}

407const Pic *GetHeadPic(
408 const CharacterClass *c, const direction_e dir, const gunstate_e gunState)
409{
410 // If firing, draw the firing head pic
411 const int row =
412 (gunState == GUNSTATE_FIRING || gunState == GUNSTATE_RECOIL) ? 1 : 0;
413 const int idx = (int)dir + row * 8;
414 return CPicGetPic(&c->HeadPics, idx);
415}
416static const Pic *GetBodyPic(
417 PicManager *pm, const CharSprites *cs, const direction_e dir,
418 const ActorAnimation anim, const int frame, const bool isArmed)
419{
420 const int stride = anim == ACTORANIMATION_IDLE ? 1 : 8;
421 const int col = frame % stride;
422 const int row = (int)dir;
423 const int idx = col + row * stride;
424 char buf[CDOGS_PATH_MAX];
425 sprintf(
426 buf, "chars/bodies/%s/upper_%s%s",
427 cs->Name,
428 anim == ACTORANIMATION_IDLE ? "idle" : "run",
429 isArmed ? "_handgun" : ""); // TODO: other gun holding poses
430 return CArrayGet(&PicManagerGetSprites(pm, buf)->pics, idx);
431}
432static const Pic *GetLegsPic(
433 PicManager *pm, const CharSprites *cs, const direction_e dir,
434 const ActorAnimation anim, const int frame)
435{
436 const int stride = anim == ACTORANIMATION_IDLE ? 1 : 8;
437 const int col = frame % stride;
438 const int row = (int)dir;
439 const int idx = col + row * stride;
440 char buf[CDOGS_PATH_MAX];
441 sprintf(
442 buf, "chars/bodies/%s/legs_%s",
443 cs->Name, anim == ACTORANIMATION_IDLE ? "idle" : "run");
444 return CArrayGet(&PicManagerGetSprites(pm, buf)->pics, idx);
445}
446static const Pic *GetGunPic(
447 const NamedSprites *gunPics, const direction_e dir, const int gunState)
448{
449 const int idx = (gunState == GUNSTATE_READY ? 8 : 0) + dir;
450 return CArrayGet(&gunPics->pics, idx);
451}
452static const Pic *GetDeathPic(PicManager *pm, const int frame)
453{
454 return CArrayGet(&PicManagerGetSprites(pm, "chars/death")->pics, frame);
455}

457void DrawCharacterSimple(
458 Character *c, const Vec2i pos, const direction_e d,
459 const bool hilite, const bool showGun)
460{
461 ActorPics pics = GetCharacterPics(
462 c, d, ACTORANIMATION_IDLE, 0, NULL, GUNSTATE_READY,
463 false, NULL, NULL, 0);
464 DrawActorPics(&pics, pos);
465 if (hilite)
466 {
467 FontCh('>', Vec2iAdd(pos, Vec2iNew(-8, -16)));
468 if (showGun)
469 {
470 FontStr(c->Gun->name, Vec2iAdd(pos, Vec2iNew(-8, 8)));
471 }
472 }
473}

475void DrawHead(
476 const Character *c, const direction_e dir, const Vec2i pos)
477{
478 const Pic *head = GetHeadPic(c->Class, dir, GUNSTATE_READY);
479 const Vec2i drawPos = Vec2iMinus(pos, Vec2iNew(
480 head->size.x / 2, head->size.y / 2));
481 BlitCharMultichannel(&gGraphicsDevice, head, drawPos, &c->Colors);
482}
483#define DYING_BODY_OFFSET 3
484static void DrawDyingBody(
485 GraphicsDevice *g, const ActorPics *pics, const Vec2i pos)
486{
487 const Pic *body = pics->Body;
488 const Vec2i drawPos = Vec2iMinus(pos, Vec2iNew(
489 body->size.x / 2, body->size.y / 2 + DYING_BODY_OFFSET));
490 const color_t mask = pics->Mask != NULL ? *pics->Mask : colorWhite;
491 BlitMasked(g, pics->Body, drawPos, mask, true);
492}
