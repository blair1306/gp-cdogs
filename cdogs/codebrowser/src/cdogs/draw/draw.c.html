73// Three types of tile drawing, based on line of sight:
74// Unvisited: black
75// Out of sight: dark, or if fog disabled, black
76// In sight: full color
77typedef enum
78{
79 TILE_LOS_NORMAL,
80 TILE_LOS_FOG,
81 TILE_LOS_NONE
82} TileLOS;
83static TileLOS GetTileLOS(const Tile *tile, const bool useFog)
84{
85 if (!tile->isVisited)
86 {
87 return TILE_LOS_NONE;
88 }
89 if (tile->flags & MAPTILE_OUT_OF_SIGHT)
90 {
91 return useFog ? TILE_LOS_FOG : TILE_LOS_NONE;
92 }
93 return TILE_LOS_NORMAL;
94}
95void DrawWallColumn(int y, Vec2i pos, Tile *tile)
96{
97 const bool useFog = ConfigGetBool(&gConfig, "Game.Fog");
98 while (y >= 0 && (tile->flags & MAPTILE_IS_WALL))
99 {
100 switch (GetTileLOS(tile, useFog))
101 {
102 case TILE_LOS_NORMAL:
103 Blit(&gGraphicsDevice, &tile->pic->pic, pos);
104 break;
105 case TILE_LOS_FOG:
106 BlitMasked(&gGraphicsDevice, &tile->pic->pic, pos, colorFog, false);
107 break;
108 case TILE_LOS_NONE:
109 default:
110 // don't draw anything
111 break;
112 }
113 pos.y -= TILE_HEIGHT;
114 tile -= X_TILES;
115 y--;
116 }
117}

120static void DrawFloor(DrawBuffer *b, Vec2i offset);
121static void DrawDebris(DrawBuffer *b, Vec2i offset);

123static void DrawExtra(DrawBuffer *b, Vec2i offset, GrafxDrawExtra *extra);
124
125void DrawBufferDraw(DrawBuffer *b, Vec2i offset, GrafxDrawExtra *extra)
126{
127 // First draw the floor tiles (which do not obstruct anything)
128 DrawFloor(b, offset);
129 // Then draw debris (wrecks)
130 DrawDebris(b, offset);
131 // Now draw walls and (non-wreck) things in proper order
132 DrawWallsAndThings(b, offset);
133 // Draw objective highlights, for visible and always-visible objectives
134 DrawObjectiveHighlights(b, offset);
135 // Draw actor chatter
136 DrawChatters(b, offset);
137 // Draw editor-only things
138 if (extra)
139 {
140 DrawExtra(b, offset, extra);
141 }
142}

144static void DrawFloor(DrawBuffer *b, Vec2i offset)
145{
146 int x, y;
147 Vec2i pos;
148 const Tile *tile = &b->tiles[0][0];
149 const bool useFog = ConfigGetBool(&gConfig, "Game.Fog");
150 for (y = 0, pos.y = b->dy + offset.y;
151 y < Y_TILES;
152 y++, pos.y += TILE_HEIGHT)
153 {
154 for (x = 0, pos.x = b->dx + offset.x;
155 x < b->Size.x;
156 x++, tile++, pos.x += TILE_WIDTH)
157 {
158 if (tile->pic != NULL && tile->pic->pic.Data != NULL &&
159 !(tile->flags & MAPTILE_IS_WALL))
160 {
161 switch (GetTileLOS(tile, useFog))
162 {
163 case TILE_LOS_NORMAL:
164 Blit(&gGraphicsDevice, &tile->pic->pic, pos);
165 break;
166 case TILE_LOS_FOG:
167 BlitMasked(
168 &gGraphicsDevice,
169 &tile->pic->pic,
170 pos,
171 colorFog,
172 false);
173 break;
174 case TILE_LOS_NONE:
175 default:
176 // don't draw
177 break;
178 }
179 }
180 }
181 tile += X_TILES - b->Size.x;
182 }
183}

187static void DrawDebris(DrawBuffer *b, Vec2i offset)
188{
189 Tile *tile = &b->tiles[0][0];
190 for (int y = 0; y < Y_TILES; y++)
191 {
192 CArrayClear(&b->displaylist);
193 for (int x = 0; x < b->Size.x; x++, tile++)
194 {
195 if (tile->flags & MAPTILE_OUT_OF_SIGHT)
196 {
197 continue;
198 }
199 CA_FOREACH(ThingId, tid, tile->things)
200 const TTileItem *ti = ThingIdGetTileItem(tid);
201 if (TileItemDrawLast(ti))
202 {
203 CArrayPushBack(&b->displaylist, &ti);
204 }
205 CA_FOREACH_END()
206 }
207 DrawBufferSortDisplayList(b);
208 CA_FOREACH(const TTileItem *, tp, b->displaylist)
209 DrawThing(b, *tp, offset);
210 CA_FOREACH_END()
211 tile += X_TILES - b->Size.x;
212 }
213}

215#define WALL_OFFSET_Y (-12)
216static void DrawWallsAndThings(DrawBuffer *b, Vec2i offset)
217{
218 Vec2i pos;
219 Tile *tile = &b->tiles[0][0];
220 pos.y = b->dy + WALL_OFFSET_Y + offset.y;
221 const bool useFog = ConfigGetBool(&gConfig, "Game.Fog");
222 for (int y = 0; y < Y_TILES; y++, pos.y += TILE_HEIGHT)
223 {
224 CArrayClear(&b->displaylist);
225 pos.x = b->dx + offset.x;
226 for (int x = 0; x < b->Size.x; x++, tile++, pos.x += TILE_WIDTH)
227 {
228 if (tile->flags & MAPTILE_IS_WALL)
229 {
230 if (!(tile->flags & MAPTILE_DELAY_DRAW))
231 {
232 DrawWallColumn(y, pos, tile);
233 }
234 }
235 else if (tile->flags & MAPTILE_OFFSET_PIC)
236 {
237 // Drawing doors
238 // Doors may be offset; vertical doors are drawn centered
239 // horizontal doors are bottom aligned
240 Vec2i doorPos = pos;
241 doorPos.x += (TILE_WIDTH - tile->picAlt->pic.size.x) / 2;
242 if (tile->picAlt->pic.size.y > 16)
243 {
244 doorPos.y +=
245 TILE_HEIGHT - (tile->picAlt->pic.size.y % TILE_HEIGHT);
246 }
247 switch (GetTileLOS(tile, useFog))
248 {
249 case TILE_LOS_NORMAL:
250 Blit(&gGraphicsDevice, &tile->picAlt->pic, doorPos);
251 break;
252 case TILE_LOS_FOG:
253 BlitMasked(
254 &gGraphicsDevice,
255 &tile->picAlt->pic,
256 doorPos,
257 colorFog,
258 false);
259 break;
260 case TILE_LOS_NONE:
261 default:
262 // don't draw anything
263 break;
264 }
265 }
266
267 // Draw the items that are in LOS
268 if (tile->flags & MAPTILE_OUT_OF_SIGHT)
269 {
270 continue;
271 }
272 CA_FOREACH(ThingId, tid, tile->things)
273 const TTileItem *ti = ThingIdGetTileItem(tid);
274 // Drawn later
275 if (TileItemDrawLast(ti))
276 {
277 continue;
278 }
279 CArrayPushBack(&b->displaylist, &ti);
280 CA_FOREACH_END()
281 }
282 DrawBufferSortDisplayList(b);
283 CA_FOREACH(const TTileItem *, tp, b->displaylist)
284 DrawThing(b, *tp, offset);
285 CA_FOREACH_END()
286 tile += X_TILES - b->Size.x;
287 }
288}
289static void DrawThing(DrawBuffer *b, const TTileItem *t, const Vec2i offset)
290{
291 const Vec2i picPos = Vec2iNew(
292 t->x - b->xTop + offset.x, t->y - b->yTop + offset.y);
293
294 if (!Vec2iIsZero(t->ShadowSize))
295 {
296 DrawShadow(&gGraphicsDevice, picPos, t->ShadowSize);
297 }
298
299 if (t->CPicFunc)
300 {
301 CPicDrawContext c = t->CPicFunc(t->id);
302 CPicDraw(b->g, &t->CPic, picPos, &c);
303 }
304 else if (t->getPicFunc)
305 {
306 Vec2i picOffset;
307 const Pic *pic = t->getPicFunc(t->id, &picOffset);
308 Blit(&gGraphicsDevice, pic, Vec2iAdd(picPos, picOffset));
309 }
310 else if (t->kind == KIND_CHARACTER)
311 {
312 TActor *a = CArrayGet(&gActors, t->id);
313 ActorPics pics = GetCharacterPicsFromActor(a);
314 DrawActorPics(&pics, picPos);
315 // Draw weapon indicators
316 DrawLaserSight(&pics, a, picPos);
317 }
318 else
319 {
320 (*(t->drawFunc))(picPos, &t->drawData);
321 }
322
323#ifdef DEBUG_DRAW_HITBOXES
324 const int pulsePeriod = ConfigGetInt(&gConfig, "Game.FPS");
325 int alphaUnscaled =
326 (gMission.time % pulsePeriod) * 255 / (pulsePeriod / 2);
327 if (alphaUnscaled > 255)
328 {
329 alphaUnscaled = 255 * 2 - alphaUnscaled;
330 }
331 color_t color = colorPurple;
332 color.a = (Uint8)alphaUnscaled;
333 DrawRectangle(
334 &gGraphicsDevice, Vec2iMinus(picPos, Vec2iScaleDiv(t->size, 2)),
335 t->size, color, DRAW_FLAG_LINE);
336#endif
337}

354static void DrawEditorTiles(DrawBuffer *b, const Vec2i offset)
355{
356 Vec2i pos;
357 Tile *tile = &b->tiles[0][0];
358 pos.y = b->dy + offset.y;
359 for (int y = 0; y < Y_TILES; y++, pos.y += TILE_HEIGHT)
360 {
361 pos.x = b->dx + offset.x;
362 for (int x = 0; x < b->Size.x; x++, tile++, pos.x += TILE_WIDTH)
363 {
364 if (gMission.missionData->Type == MAPTYPE_STATIC)
365 {
366 Vec2i start = gMission.missionData->u.Static.Start;
367 if (!Vec2iIsZero(start) &&
368 Vec2iEqual(start, Vec2iNew(x + b->xStart, y + b->yStart)))
369 {
370 // mission start
371 BlitMasked(
372 &gGraphicsDevice,
373 PicManagerGetPic(&gPicManager, "editor/start"),
374 pos, colorWhite, 1);
375 }
376 }
377 }
378 tile += X_TILES - b->Size.x;
379 }
380}

382static void DrawGuideImage(
383 DrawBuffer *b, SDL_Surface *guideImage, Uint8 alpha)
384{
385 SDL_LockSurface(guideImage);
386 // Scale based on ratio between map size and guide image size,
387 // so that the guide image stretches to the map size
388 double xScale = (double)guideImage->w / (gMap.Size.x * TILE_WIDTH);
389 double yScale = (double)guideImage->h / (gMap.Size.y * TILE_HEIGHT);
390 for (int j = 0; j < b->g->cachedConfig.Res.y; j++)
391 {
392 int y = (int)round((j + b->yTop) * yScale);
393 for (int i = 0; i < b->g->cachedConfig.Res.x; i++)
394 {
395 int x = (int)round((i + b->xTop) * xScale);
396 if (x >= 0 && x < guideImage->w && y >= 0 && y < guideImage->h)
397 {
398 int imgIndex = y * guideImage->w + x;
399 Uint32 p = ((Uint32 *)guideImage->pixels)[imgIndex];
400 color_t c = PIXEL2COLOR(p);
401 c.a = alpha;
402 Draw_Point(i, j, c);
403 }
404 }
405 }
406 SDL_UnlockSurface(guideImage);
407}

409// Draw names of objects (objectives, spawners etc.)
410static void DrawObjectiveName(
411 const TTileItem *ti, DrawBuffer *b, const Vec2i offset);
412static void DrawSpawnerName(
413 const TObject *obj, DrawBuffer *b, const Vec2i offset);
414static void DrawObjectNames(DrawBuffer *b, const Vec2i offset)
415{
416 const Tile *tile = &b->tiles[0][0];
417 for (int y = 0; y < Y_TILES; y++)
418 {
419 for (int x = 0; x < b->Size.x; x++, tile++)
420 {
421 CA_FOREACH(ThingId, tid, tile->things)
422 const TTileItem *ti = ThingIdGetTileItem(tid);
423 if (ti->flags & TILEITEM_OBJECTIVE)
424 {
425 DrawObjectiveName(ti, b, offset);
426 }
427 else if (ti->kind == KIND_OBJECT)
428 {
429 const TObject *obj = CArrayGet(&gObjs, ti->id);
430 if (obj->Class->Type == MAP_OBJECT_TYPE_PICKUP_SPAWNER)
431 {
432 DrawSpawnerName(obj, b, offset);
433 }
434 }
435 CA_FOREACH_END()
436 }
437 tile += X_TILES - b->Size.x;
438 }
439}
440static void DrawObjectiveName(
441 const TTileItem *ti, DrawBuffer *b, const Vec2i offset)
442{
443 const int objective = ObjectiveFromTileItem(ti->flags);
444 const Objective *o =
445 CArrayGet(&gMission.missionData->Objectives, objective);
446 const char *typeName = ObjectiveTypeStr(o->Type);
447 const Vec2i textPos = Vec2iNew(
448 ti->x - b->xTop + offset.x - FontStrW(typeName) / 2,
449 ti->y - b->yTop + offset.y);
450 FontStr(typeName, textPos);
451}
452static void DrawSpawnerName(
453 const TObject *obj, DrawBuffer *b, const Vec2i offset)
454{
455 const char *name = obj->Class->u.PickupClass->Name;
456 const Vec2i textPos = Vec2iNew(
457 obj->tileItem.x - b->xTop + offset.x - FontStrW(name) / 2,
458 obj->tileItem.y - b->yTop + offset.y);
459 FontStr(name, textPos);
460}
