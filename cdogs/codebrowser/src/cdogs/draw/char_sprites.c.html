43const CharSprites *StrCharSpriteClass(const char *s)
44{
45 CharSprites *c;
46 int error = hashmap_get(gCharSpriteClasses.customClasses, s, (any_t *)&c);
47 if (error == MAP_OK) return c;
48 error = hashmap_get(gCharSpriteClasses.classes, s, (any_t *)&c);
49 if (error == MAP_OK) return c;
50 return NULL;
51}

53void CharSpriteClassesInit(CharSpriteClasses *c)
54{
55 memset(c, 0, sizeof *c);
56 c->classes = hashmap_new();
57 c->customClasses = hashmap_new();
58 char buf[CDOGS_PATH_MAX];
59 GetDataFilePath(buf, "");
60 CharSpriteClassesLoadDir(c->classes, buf);
61}

63static CharSprites *CharSpritesLoadJSON(const char *name, const char *path);
64void CharSpriteClassesLoadDir(map_t classes, const char *path)
65{
66 char buf[CDOGS_PATH_MAX];
67 sprintf(buf, "%s/graphics/chars/bodies", path);
68 tinydir_dir dir;
69 if (tinydir_open(&dir, buf) == -1)
70 {
71 goto bail;
72 }
73
74 for (; dir.has_next; tinydir_next(&dir))
75 {
76 tinydir_file file;
77 if (tinydir_readfile(&dir, &file) == -1)
78 {
79 LOG(LM_MAIN, LL_ERROR, "Cannot read file '%s': %s",
80 file.path, strerror(errno));
81 goto bail;
82 }
83 if (!file.is_dir || file.name[0] == '.')
84 {
85 continue;
86 }
87 CharSprites *c = CharSpritesLoadJSON(file.name, file.path);
88 if (c == NULL)
89 {
90 continue;
91 }
92 const int error = hashmap_put(classes, file.name, c);
93 if (error != MAP_OK)
94 {
95 LOG(LM_MAIN, LL_ERROR, "failed to add char sprites %s: %d",
96 file.name, error);
97 continue;
98 }
99 }
100
101bail:
102 tinydir_close(&dir);
103}
104static map_t LoadFrameOffsets(yajl_val node, const char *path);
105static void LoadDirOffsets(Vec2i *offsets, yajl_val node, const char *path);
106static CharSprites *CharSpritesLoadJSON(const char *name, const char *path)
107{
108 CharSprites *c = NULL;
109 // Try to find a data.json in this dir
110 tinydir_file dataFile;
111 yajl_val node = NULL;
112 char buf[CDOGS_PATH_MAX];
113 sprintf(buf, "%s/data.json", path);
114 if (tinydir_file_open(&dataFile, buf) != 0)
115 {
116 goto bail;
117 }
118 node = YAJLReadFile(buf);
119 if (node == NULL)
120 {
121 LOG(LM_MAIN, LL_ERROR, "Error parsing char sprite JSON '%s'", buf);
122 goto bail;
123 }
124
125 CCALLOC(c, sizeof *c);
126 CSTRDUP(c->Name, name);
127 const yajl_array order = YAJL_GET_ARRAY(YAJLFindNode(node, "Order"));
128 for (direction_e d = DIRECTION_UP; d < DIRECTION_COUNT; d++)
129 {
130 const yajl_array orderDir = YAJL_GET_ARRAY(order->values[d]);
131 for (BodyPart bp = BODY_PART_HEAD; bp < BODY_PART_COUNT; bp++)
132 {
133 c->Order[d][bp] = StrBodyPart(
134 YAJL_GET_STRING(orderDir->values[bp]));
135 }
136 }
137 c->Offsets.Frame[BODY_PART_HEAD] = LoadFrameOffsets(
138 node, "Offsets/Frame/Head");
139 c->Offsets.Frame[BODY_PART_BODY] = LoadFrameOffsets(
140 node, "Offsets/Frame/Body");
141 c->Offsets.Frame[BODY_PART_LEGS] = LoadFrameOffsets(
142 node, "Offsets/Frame/Legs");
143 c->Offsets.Frame[BODY_PART_GUN] = LoadFrameOffsets(
144 node, "Offsets/Frame/Gun");
145 LoadDirOffsets(c->Offsets.Dir[BODY_PART_HEAD], node, "Offsets/Dir/Head");
146 LoadDirOffsets(c->Offsets.Dir[BODY_PART_BODY], node, "Offsets/Dir/Body");
147 LoadDirOffsets(c->Offsets.Dir[BODY_PART_LEGS], node, "Offsets/Dir/Legs");
148 LoadDirOffsets(c->Offsets.Dir[BODY_PART_GUN], node, "Offsets/Dir/Gun");
149
150bail:
151 yajl_tree_free(node);
152 return c;
153}
154static map_t LoadFrameOffsets(yajl_val node, const char *path)
155{
156 map_t offsets = hashmap_new();
157 const yajl_object obj = YAJL_GET_OBJECT(YAJLFindNode(node, path));
158 for (int i = 0; i < (int)obj->len; i++)
159 {
160 const char *key = obj->keys[i];
161 CArray *offsetVals;
162 CMALLOC(offsetVals, sizeof *offsetVals);
163 CArrayInit(offsetVals, sizeof(Vec2i));
164 const yajl_array offsetsArray = YAJL_GET_ARRAY(obj->values[i]);
165 for (int j = 0; j < (int)offsetsArray->len; j++)
166 {
167 const Vec2i offset = YAJL_GET_VEC2I(offsetsArray->values[j]);
168 CArrayPushBack(offsetVals, &offset);
169 }
170 const int error = hashmap_put(offsets, key, offsetVals);
171 if (error != MAP_OK)
172 {
173 LOG(LM_MAIN, LL_ERROR, "Failed to add animation offsets: %d",
174 error);
175 CASSERT(false, "Failed to add animation offsets");
176 }
177 }
178 return offsets;
179}
180static void LoadDirOffsets(Vec2i *offsets, yajl_val node, const char *path)
181{
182 const yajl_array offsetsArray = YAJL_GET_ARRAY(YAJLFindNode(node, path));
183 if (offsetsArray == NULL)
184 {
185 return;
186 }
187 for (direction_e d = DIRECTION_UP; d < DIRECTION_COUNT; d++)
188 {
189 offsets[d] = YAJL_GET_VEC2I(offsetsArray->values[d]);
190 }
191}

205{
206 CharSpriteClassesClear(c->classes);
207 CharSpriteClassesClear(c->customClasses);
208}

210Vec2i CharSpritesGetOffset(
211 const map_t offsets, const char *anim, const int frame)
212{
213 CArray *animOffsets;
214 int error = hashmap_get(offsets, anim, (any_t *)&animOffsets);
215 if (error == MAP_MISSING)
216 {
217 // Use idle animation by default
218 error = hashmap_get(offsets, "idle", (any_t *)&animOffsets);
219 }
220 if (error != MAP_OK)
221 {
222 CASSERT(false, "animation not found");
223 return Vec2iZero();
224 }
225 return *(Vec2i *)CArrayGet(animOffsets, frame % animOffsets->size);
226}
