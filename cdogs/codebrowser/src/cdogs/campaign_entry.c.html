39static bool IsCampaignOK(const char *path, char **buf, int *numMissions)
40{
41 return MapNewScan(path, buf, numMissions) == 0;
42}
43
44void CampaignEntryInit(CampaignEntry *entry, const char *title, GameMode mode)
45{
46 memset(entry, 0, sizeof *entry);
47 CSTRDUP(entry->Info, title);
48 entry->Mode = mode;
49}
50void CampaignEntryCopy(CampaignEntry *dst, CampaignEntry *src)
51{
52 memcpy(dst, src, sizeof *dst);
53 if (src->Filename) CSTRDUP(dst->Filename, src->Filename);
54 if (src->Path) CSTRDUP(dst->Path, src->Path);
55 if (src->Info) CSTRDUP(dst->Info, src->Info);
56}
57bool CampaignEntryTryLoad(
58 CampaignEntry *entry, const char *path, GameMode mode)
59{
60 char *buf;
61 int numMissions;
62 if (!IsCampaignOK(path, &buf, &numMissions))
63 {
64 return false;
65 }
66 // cap length of title
67 size_t maxLen = 70;
68 if (strlen(buf) > maxLen)
69 {
70 buf[maxLen] = '\0';
71 }
72 char title[256];
73 sprintf(title, "%s (%d)", buf, numMissions);
74 CampaignEntryInit(entry, title, mode);
75 CSTRDUP(entry->Filename, PathGetBasename(path));
76 // Get relative path for the campaign entry, so when we transmit it to
77 // network clients they can load it regardless of install path
78 // Note: relative path is counted from data dir
79 char pathBuf[CDOGS_PATH_MAX];
80 char dataDirBuf[CDOGS_PATH_MAX];
81 GetDataFilePath(dataDirBuf, "");
82 RelPath(pathBuf, path, dataDirBuf);
83 CSTRDUP(entry->Path, pathBuf);
84 entry->NumMissions = numMissions;
85 CFREE(buf);
86 return true;
87}
88void CampaignEntryTerminate(CampaignEntry *entry)
89{
90 CFREE(entry->Filename);
91 CFREE(entry->Path);
92 CFREE(entry->Info);
93 memset(entry, 0, sizeof *entry);
94}
