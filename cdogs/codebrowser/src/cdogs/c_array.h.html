28#include <stdbool.h>

31// dynamic array
32typedef struct
33{
34 void *data;
35 size_t elemSize;
36 size_t size;
37 size_t capacity;
38} CArray;
39
40void CArrayInit(CArray *a, size_t elemSize);
41void CArrayReserve(CArray *a, size_t capacity);
42void CArrayCopy(CArray *dst, const CArray *src);
43void CArrayPushBack(CArray *a, const void *elem); // insert address
44void CArrayInsert(CArray *a, int index, const void *elem);
45void CArrayDelete(CArray *a, int index);
46void CArrayResize(CArray *a, const size_t size, const void *value);
47void *CArrayGet(const CArray *a, int index); // gets address
48void CArrayClear(CArray *a);
49void CArrayRemoveIf(CArray *a, bool(*removeIf)(const void *));
50void CArrayFill(CArray *a, const void *elem);
51void CArrayFillZero(CArray *a);
52void CArrayShuffle(CArray *a);
53void CArrayTerminate(CArray *a);
54
55// Convenience macro for looping through a CArray
56#define CA_FOREACH(_type, _var, _a)\
57 for (int _ca_index = 0; _ca_index < (int)(_a).size; _ca_index++)\
58 {\
59 _type *_var = CArrayGet(&(_a), _ca_index);
60#define CA_FOREACH_END() }
