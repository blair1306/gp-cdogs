62#define FLAGS_HURTALWAYS (1 << 2)
63
64// Computer characters only
65#define FLAGS_DETOURING (1 << 5)
66#define FLAGS_TRYRIGHT (1 << 6)
67#define FLAGS_SLEEPING (1 << 7)
68#define FLAGS_VISIBLE (1 << 8)
69#define FLAGS_ASBESTOS (1 << 9) // Immune to fire
70#define FLAGS_IMMUNITY (1 << 10) // Immune to poison
71#define FLAGS_SEETHROUGH (1 << 11) // Almost transparent
72
73// Special flags
74#define FLAGS_RUNS_AWAY (1 << 16) // Move directly away from player
75#define FLAGS_GOOD_GUY (1 << 17) // Shots cannot hurt player
76#define FLAGS_PRISONER (1 << 18) // Won't move until touched by player
77#define FLAGS_INVULNERABLE (1 << 19)
78#define FLAGS_FOLLOWER (1 << 20) // Follows player
79#define FLAGS_PENALTY (1 << 21) // Big score penalty if shot
80#define FLAGS_VICTIM (1 << 22) // Can be shot by everyone
81#define FLAGS_SNEAKY (1 << 23) // Always shoot back when player shoots
82#define FLAGS_SLEEPALWAYS (1 << 24)
83#define FLAGS_AWAKEALWAYS (1 << 25)
84#define FLAGS_RESCUED (1 << 26) // Run towards exit

87typedef enum
88{
89 ACTORACTION_MOVING,
90 ACTORACTION_EXITING
91} ActorAction;
92
93typedef struct Actor
94{
95 Vec2i Pos; // These are the full coordinates, including fractions
96 // Vector that the player is attempting to move in, based on input
97 Vec2i MoveVel;
98 direction_e direction;
99 // Rotation used to draw the actor, which will lag behind the actual
100 // rotation in order to show smooth rotation
101 float DrawRadians;
102 Animation anim;
103 int stateCounter;
104 int lastCmd;
105 // Whether the player ran into something whilst trying to move
106 // In this situation, we interrupt dead reckoning and resend the position
107 bool hasCollided;
108 // Whether the last special command was performed with a direction
109 // This differentiates between a special command and weapon switch
110 bool specialCmdDir;
111 // If the actor wants to pick up everything, even those that need to be
112 // manually picked up using special input
113 bool PickupAll;
114 // Flag to specify whether this actor is over a special pickup,
115 // such as weapons which will replace the current weapon.
116 // If the player presses switch when in this state, instead of switching
117 // weapons, pick up.
118 bool CanPickupSpecial;
119
120 // -1 if human character (get from player data), otherwise index into
121 // CharacterStore OtherChars
122 int charId;
123 int PlayerUID; // -1 unless a human player
124 int uid; // unique ID across all actors
125 CArray guns; // of Weapon
126 CArray ammo; // of int
127 int gunIndex;
128
129 int health;
130 // A counter for player death
131 // If 0, player is alive
132 // If >0 but less than DEATH_MAX, player is "dying" and this variable
133 // is used as the frame in the death animation
134 int dead;
135 int flamed;
136 int poisoned;
137 int petrified;
138 int confused;
139 int flags;
140
141 int turns;
142
143 int slideLock;
144
145 // What to say (text label appears above actor) and how long to say it
146 char Chatter[256];
147 int ChatterCounter;
148
149 // Gore emitters
150 Emitter blood1;
151 Emitter blood2;
152 Emitter blood3;
153 int bleedCounter;
154
155 // Signals to other AIs what this actor is doing
156 ActorAction action;
157 AIContext *aiContext;
158 TTileItem tileItem;
159 bool isInUse;
160} TActor;
161
162// Store all actors in-line in an array
163// Destroying actors does not modify the array, only switches
164// a boolean flag isInUse to denote whether this instance is
165// in use.
166// Actors should be identified by index
167// There is a small chance of invalidating pointers if
168// actors are added and the array must be resized.
169// Therefore do not hold actor pointers and reuse.
170extern CArray gActors; // of TActor
