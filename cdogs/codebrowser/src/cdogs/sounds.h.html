67typedef enum
68{
69 SOUND_NORMAL,
70 SOUND_RANDOM
71} SoundType;
72
73typedef struct
74{
75 SoundType Type;
76 union
77 {
78 Mix_Chunk *normal;
79 struct
80 {
81 CArray sounds; // of Mix_Chunk *
82 int lastPlayed;
83 } random;
84 } u;
85} SoundData;
86
87typedef enum
88{
89 MUSIC_OK,
90 MUSIC_NOLOAD,
91 MUSIC_PLAYING,
92 MUSIC_PAUSED
93} music_status_e;
94
95typedef struct
96{
97 int isInitialised;
98 Mix_Music *music;
99 music_status_e musicStatus;
100 char musicErrorMessage[128];
101 int channels;
102
103 // Two sets of ears for 4-player split screen
104 Vec2i earLeft1;
105 Vec2i earLeft2;
106 Vec2i earRight1;
107 Vec2i earRight2;
108
109 map_t sounds; // of SoundData
110 map_t customSounds; // of SoundData
111} SoundDevice;
112
113extern SoundDevice gSoundDevice;
114
115typedef struct
116{
117 int SoundVolume;
118 int MusicVolume;
119 bool Footsteps;
120 bool Hits;
121 bool Reloads;
122} SoundConfig;
123
124// DTO for playing certain sounds associated with collision
125// for a bullet or weapon type
126typedef struct
127{
128 char *Object;
129 char *Flesh;
130 char *Wall;
131} HitSounds;

133void SoundInitialize(SoundDevice *device, const char *path);
134void SoundLoadDir(map_t sounds, const char *path, const char *prefix);
135void SoundReconfigure(SoundDevice *s);
136void SoundClear(map_t sounds);
137void SoundTerminate(SoundDevice *device, const bool waitForSoundsComplete);
138void SoundPlay(SoundDevice *device, Mix_Chunk *data);
139void SoundSetEarsSide(const bool isLeft, const Vec2i pos);
140void SoundSetEar(const bool isLeft, const int idx, Vec2i pos);
141void SoundSetEars(Vec2i pos);
142void SoundPlayAt(SoundDevice *device, Mix_Chunk *data, const Vec2i pos);
143
144// Play a sound but with distance added
