73static void Gfx_ModeSet(const Vec2i *mode)
74{
75 ConfigGet(&gConfig, "Graphics.ResolutionWidth")->u.Int.Value = mode->x;
76 ConfigGet(&gConfig, "Graphics.ResolutionHeight")->u.Int.Value = mode->y;
77}

79void Gfx_ModePrev(void)
80{
81 GraphicsDevice *device = &gGraphicsDevice;
82 device->modeIndex--;
83 if (device->modeIndex < 0)
84 {
85 device->modeIndex = (int)device->validModes.size - 1;
86 }
87 Gfx_ModeSet(CArrayGet(&device->validModes, device->modeIndex));
88}
89
90void Gfx_ModeNext(void)
91{
92 GraphicsDevice *device = &gGraphicsDevice;
93 device->modeIndex++;
94 if (device->modeIndex >= (int)device->validModes.size)
95 {
96 device->modeIndex = 0;
97 }
98 Gfx_ModeSet(CArrayGet(&device->validModes, device->modeIndex));
99}

101static int FindValidMode(GraphicsDevice *device, const int w, const int h)
102{
103 CA_FOREACH(const Vec2i, mode, device->validModes)
104 if (mode->x == w && mode->y == h)
105 {
106 return _ca_index;
107 }
108 CA_FOREACH_END()
109 return -1;
110}
111
112static void AddGraphicsMode(GraphicsDevice *device, const int w, const int h)
113{
114 // Don't add if mode already exists
115 if (FindValidMode(device, w, h) != -1)
116 {
117 return;
118 }
119
120 const int size = w * h;
121 int i;
122 for (i = 0; i < (int)device->validModes.size; i++)
123 {
124 // Ordered by actual resolution ascending and scale descending
125 const Vec2i *mode = CArrayGet(&device->validModes, i);
126 const int actualResolution = mode->x * mode->y;
127 if (actualResolution >= size)
128 {
129 break;
130 }
131 }
132 Vec2i mode = Vec2iNew(w, h);
133 CArrayInsert(&device->validModes, i, &mode);
134}

136void GraphicsInit(GraphicsDevice *device, Config *c)
137{
138 memset(device, 0, sizeof *device);
139 CArrayInit(&device->validModes, sizeof(Vec2i));
140 // Add default modes
141 AddGraphicsMode(device, 320, 240);
142 AddGraphicsMode(device, 400, 300);
143 AddGraphicsMode(device, 640, 480);
144 GraphicsConfigSetFromConfig(&device->cachedConfig, c);
145}

147static void AddSupportedGraphicsModes(GraphicsDevice *device)
148{
149 // TODO: multiple window support?
150 const int numDisplayModes = SDL_GetNumDisplayModes(0);
151 if (numDisplayModes < 1)
152 {
153 LOG(LM_GFX, LL_ERROR, "no valid display modes: %s", SDL_GetError());
154 return;
155 }
156 for (int i = 0; i < numDisplayModes; i++)
157 {
158 SDL_DisplayMode mode;
159 if (SDL_GetDisplayMode(0, i, &mode) != 0)
160 {
161 LOG(LM_GFX, LL_ERROR, "cannot get display mode: %s",
162 SDL_GetError());
163 continue;
164 }
165 if (mode.w < 320 || mode.h < 240)
166 {
167 break;
168 }
169 AddGraphicsMode(device, mode.w, mode.h);
170 }
171}

173// Initialises the video subsystem.
174// To prevent needless screen flickering, config is compared with cache
175// to see if anything changed. If not, don't recreate the screen.
176static void DestroyTextures(GraphicsDevice *g);
177static SDL_Texture *CreateTexture(
178 SDL_Renderer *renderer, const SDL_TextureAccess access, const Vec2i res,
179 const SDL_BlendMode blend, const Uint8 alpha);
180void GraphicsInitialize(GraphicsDevice *g)
181{
182 if (g->IsInitialized && !g->cachedConfig.RestartFlags)
183 {
184 return;
185 }
186
187 if (!g->IsWindowInitialized)
188 {
189 char buf[CDOGS_PATH_MAX];
190 GetDataFilePath(buf, "graphics/cdogs_icon.bmp");
191 g->icon = IMG_Load(buf);
192 AddSupportedGraphicsModes(g);
193 g->IsWindowInitialized = true;
194 }
195
196 g->IsInitialized = false;
197
198 const int w = g->cachedConfig.Res.x;
199 const int h = g->cachedConfig.Res.y;
200
201 const bool initRenderer =
202 !!(g->cachedConfig.RestartFlags & RESTART_RESOLUTION);
203 const bool initTextures =
204 !!(g->cachedConfig.RestartFlags &
205 (RESTART_RESOLUTION | RESTART_SCALE_MODE));
206 const bool initBrightness =
207 !!(g->cachedConfig.RestartFlags &
208 (RESTART_RESOLUTION | RESTART_SCALE_MODE | RESTART_BRIGHTNESS));
209
210 if (initRenderer)
211 {
212 Uint32 sdlFlags = SDL_WINDOW_RESIZABLE;
213 if (g->cachedConfig.Fullscreen)
214 {
215 sdlFlags |= SDL_WINDOW_FULLSCREEN;
216 }
217
218 LOG(LM_GFX, LL_INFO, "graphics mode(%dx%d %dx)",
219 w, h, g->cachedConfig.ScaleFactor);
220 // Get the previous window's size and recreate it
221 Vec2i windowSize = Vec2iNew(
222 w * g->cachedConfig.ScaleFactor, h * g->cachedConfig.ScaleFactor);
223 if (g->window)
224 {
225 SDL_GetWindowSize(g->window, &windowSize.x, &windowSize.y);
226 }
227 LOG(LM_GFX, LL_DEBUG, "destroying previous renderer");
228 DestroyTextures(g);
229 SDL_DestroyRenderer(g->renderer);
230 SDL_FreeFormat(g->Format);
231 SDL_DestroyWindow(g->window);
232 LOG(LM_GFX, LL_DEBUG, "creating window %dx%d flags(%X)",
233 windowSize.x, windowSize.y, sdlFlags);
234 if (SDL_CreateWindowAndRenderer(
235 windowSize.x, windowSize.y, sdlFlags,
236 &g->window, &g->renderer) == -1 ||
237 g->window == NULL || g->renderer == NULL)
238 {
239 LOG(LM_GFX, LL_ERROR, "cannot create window or renderer: %s",
240 SDL_GetError());
241 return;
242 }
243 char title[32];
244 sprintf(title, "C-Dogs SDL %s%s",
245 g->cachedConfig.IsEditor ? "Editor " : "",
246 CDOGS_SDL_VERSION);
247 LOG(LM_GFX, LL_DEBUG, "setting title(%s) and icon", title);
248 SDL_SetWindowTitle(g->window, title);
249 SDL_SetWindowIcon(g->window, g->icon);
250 g->Format = SDL_AllocFormat(SDL_PIXELFORMAT_ARGB8888);
251
252 if (SDL_RenderSetLogicalSize(g->renderer, w, h) != 0)
253 {
254 LOG(LM_GFX, LL_ERROR, "cannot set renderer logical size: %s",
255 SDL_GetError());
256 return;
257 }
258
259 GraphicsSetBlitClip(
260 g, 0, 0, g->cachedConfig.Res.x - 1, g->cachedConfig.Res.y - 1);
261 }
262
263 if (initTextures)
264 {
265 if (!initRenderer)
266 {
267 DestroyTextures(g);
268 }
269
270 // Set render scale mode
271 const char *renderScaleQuality = "nearest";
272 switch ((ScaleMode)ConfigGetEnum(&gConfig, "Graphics.ScaleMode"))
273 {
274 case SCALE_MODE_NN:
275 renderScaleQuality = "nearest";
276 break;
277 case SCALE_MODE_BILINEAR:
278 renderScaleQuality = "linear";
279 break;
280 default:
281 CASSERT(false, "unknown scale mode");
282 break;
283 }
284 LOG(LM_GFX, LL_DEBUG, "setting scale quality %s", renderScaleQuality);
285 if (!SDL_SetHint(SDL_HINT_RENDER_SCALE_QUALITY, renderScaleQuality))
286 {
287 LOG(LM_GFX, LL_WARN, "cannot set render quality hint: %s",
288 SDL_GetError());
289 }
290
291 g->screen = CreateTexture(
292 g->renderer, SDL_TEXTUREACCESS_STREAMING, Vec2iNew(w, h),
293 SDL_BLENDMODE_BLEND, 255);
294 if (g->screen == NULL)
295 {
296 return;
297 }
298
299 CFREE(g->buf);
300 CCALLOC(g->buf, GraphicsGetMemSize(&g->cachedConfig));
301 g->bkg = CreateTexture(
302 g->renderer, SDL_TEXTUREACCESS_STATIC, Vec2iNew(w, h),
303 SDL_BLENDMODE_NONE, 255);
304 if (g->bkg == NULL)
305 {
306 return;
307 }
308 g->hud = CreateTexture(
309 g->renderer, SDL_TEXTUREACCESS_STREAMING, Vec2iNew(w, h),
310 SDL_BLENDMODE_BLEND, 255);
311 if (g->hud == NULL)
312 {
313 return;
314 }
315 BlitClearBuf(g);
316 BlitUpdateFromBuf(g, g->hud);
317 }
318
319 if (initBrightness)
320 {
321 if (!initRenderer && !initTextures)
322 {
323 SDL_DestroyTexture(g->brightnessOverlay);
324 }
325
326 const int brightness = ConfigGetInt(&gConfig, "Graphics.Brightness");
327 // Alpha is approximately 50% max
328 const Uint8 alpha = (Uint8)(brightness > 0 ? brightness : -brightness) * 13;
329 g->brightnessOverlay = CreateTexture(
330 g->renderer, SDL_TEXTUREACCESS_STATIC, Vec2iNew(w, h),
331 SDL_BLENDMODE_BLEND, alpha);
332 if (g->brightnessOverlay == NULL)
333 {
334 return;
335 }
336 const color_t overlayColour = brightness > 0 ? colorWhite : colorBlack;
337 DrawRectangle(g, Vec2iZero(), g->cachedConfig.Res, overlayColour, 0);
338 BlitUpdateFromBuf(g, g->brightnessOverlay);
339 g->cachedConfig.Brightness = brightness;
340 }
341
342 g->IsInitialized = true;
343 g->cachedConfig.Res.x = w;
344 g->cachedConfig.Res.y = h;
345 g->cachedConfig.RestartFlags = 0;
346}
347static void DestroyTextures(GraphicsDevice *g)
348{
349 SDL_DestroyTexture(g->screen);
350 SDL_DestroyTexture(g->bkg);
351 SDL_DestroyTexture(g->hud);
352 SDL_DestroyTexture(g->brightnessOverlay);
353}
354static SDL_Texture *CreateTexture(
355 SDL_Renderer *renderer, const SDL_TextureAccess access, const Vec2i res,
356 const SDL_BlendMode blend, const Uint8 alpha)
357{
358 SDL_Texture *t = SDL_CreateTexture(
359 renderer, SDL_PIXELFORMAT_ARGB8888, access, res.x, res.y);
360 if (t == NULL)
361 {
362 LOG(LM_GFX, LL_ERROR, "cannot create texture: %s", SDL_GetError());
363 return NULL;
364 }
365 if (SDL_SetTextureBlendMode(t, blend) != 0)
366 {
367 LOG(LM_GFX, LL_ERROR, "cannot set blend mode: %s", SDL_GetError());
368 return NULL;
369 }
370 if (SDL_SetTextureAlphaMod(t, alpha) != 0)
371 {
372 LOG(LM_GFX, LL_ERROR, "cannot set texture alpha: %s", SDL_GetError());
373 return NULL;
374 }
375 return t;
376}

378void GraphicsTerminate(GraphicsDevice *g)
379{
380 CArrayTerminate(&g->validModes);
381 SDL_FreeSurface(g->icon);
382 DestroyTextures(g);
383 SDL_DestroyTexture(g->brightnessOverlay);
384 SDL_DestroyRenderer(g->renderer);
385 SDL_FreeFormat(g->Format);
386 SDL_DestroyWindow(g->window);
387 SDL_VideoQuit();
388 CFREE(g->buf);
389}

401void GraphicsConfigSet(
402 GraphicsConfig *c,
403 const Vec2i res, const bool fullscreen,
404 const int scaleFactor, const ScaleMode scaleMode, const int brightness)
405{
406 if (!Vec2iEqual(res, c->Res))
407 {
408 c->Res = res;
409 c->RestartFlags |= RESTART_RESOLUTION;
410 }
411#define SET(_lhs, _rhs, _flag) \
412 if ((_lhs) != (_rhs)) \
413 { \
414 (_lhs) = (_rhs); \
415 c->RestartFlags |= (_flag); \
416 }
417 SET(c->Fullscreen, fullscreen, RESTART_RESOLUTION);
418 SET(c->ScaleFactor, scaleFactor, RESTART_RESOLUTION);
419 SET(c->ScaleMode, scaleMode, RESTART_SCALE_MODE);
420 SET(c->Brightness, brightness, RESTART_BRIGHTNESS);
421}

423void GraphicsConfigSetFromConfig(GraphicsConfig *gc, Config *c)
424{
425 GraphicsConfigSet(
426 gc,
427 Vec2iNew(
428 ConfigGetInt(c, "Graphics.ResolutionWidth"),
429 ConfigGetInt(c, "Graphics.ResolutionHeight")),
430 ConfigGetBool(c, "Graphics.Fullscreen"),
431 ConfigGetInt(c, "Graphics.ScaleFactor"),
432 (ScaleMode)ConfigGetEnum(c, "Graphics.ScaleMode"),
433 ConfigGetInt(c, "Graphics.Brightness"));
434}

445void GraphicsSetBlitClip(
446 GraphicsDevice *device, int left, int top, int right, int bottom)
447{
448 device->clipping.left = left;
449 device->clipping.top = top;
450 device->clipping.right = right;
451 device->clipping.bottom = bottom;
452}
