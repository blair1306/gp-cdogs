37// Generate a random partition of an integer `total` into a pair of ints x, y
38// With the restrictions that neither x, y are less than min, and
39// neither x, y are greater than max
40static Vec2i GenerateRandomPairPartitionWithRestrictions(
41 int total, int min, int max)
42{
43 Vec2i v;
44 int xLow, xHigh;
45
46 // Check for invalid input
47 // Can't proceed if exactly half of total is greater than max,
48 // or if total less than min
49 if ((total + 1) / 2 > max || total < min)
50 {
51 assert(0 && "invalid random pair partition input");
52 return Vec2iNew(total / 2, total - (total / 2));
53 }
54
55 // Find range of x first
56 // Must be at least min, or total - max
57 // Must be at most max, or total - min
58 xLow = MAX(min, total - max);
59 xHigh = MIN(max, total - min);
60 v.x = xLow + (rand() % (xHigh - xLow + 1));
61 v.y = total - v.x;
62 assert(v.x >= min);
63 assert(v.y >= min);
64 assert(v.x <= max);
65 assert(v.y <= max);
66 return v;
67}

69static Vec2i GenerateQuickPlayMapSize(QuickPlayQuantity size)
70{
71 const int minMapDim = 16;
72 const int maxMapDim = 64;
73 // Map sizes based on total dimensions (width + height)
74 // Small: 32 - 64
75 // Medium: 64 - 96
76 // Large: 96 - 128
77 // Restrictions: at least 16, at most 64 per side
78 switch (size)
79 {
80 case QUICKPLAY_QUANTITY_ANY:
81 return GenerateRandomPairPartitionWithRestrictions(
82 32 + (rand() % (128 - 32 + 1)),
83 minMapDim, maxMapDim);
84 case QUICKPLAY_QUANTITY_SMALL:
85 return GenerateRandomPairPartitionWithRestrictions(
86 32 + (rand() % (64 - 32 + 1)),
87 minMapDim, maxMapDim);
88 case QUICKPLAY_QUANTITY_MEDIUM:
89 return GenerateRandomPairPartitionWithRestrictions(
90 64 + (rand() % (96 - 64 + 1)),
91 minMapDim, maxMapDim);
92 case QUICKPLAY_QUANTITY_LARGE:
93 return GenerateRandomPairPartitionWithRestrictions(
94 96 + (rand() % (128 - 96 + 1)),
95 minMapDim, maxMapDim);
96 default:
97 assert(0 && "invalid quick play map size config");
98 return Vec2iZero();
99 }
100}

102// Generate a quick play parameter based on the quantity setting, and various
103// thresholds
104// e.g. if qty is "low", generate random number between low and medium
105static int GenerateQuickPlayParam(
106 QuickPlayQuantity qty, int low, int medium, int high, int max)
107{
108 switch (qty)
109 {
110 case QUICKPLAY_QUANTITY_ANY:
111 return low + (rand() % (max - low + 1));
112 case QUICKPLAY_QUANTITY_SMALL:
113 return low + (rand() % (medium - low + 1));
114 case QUICKPLAY_QUANTITY_MEDIUM:
115 return medium + (rand() % (high - medium + 1));
116 case QUICKPLAY_QUANTITY_LARGE:
117 return high + (rand() % (max - high + 1));
118 default:
119 assert(0);
120 return 0;
121 }
122}

124static void SetupQuickPlayEnemy(Character *enemy, const GunDescription *gun)
125{
126 CharacterShuffleAppearance(enemy);
127 enemy->Gun = gun;
128 enemy->speed =GenerateQuickPlayParam(
129 ConfigGetEnum(&gConfig, "QuickPlay.EnemySpeed"), 64, 112, 160, 256);
130 if (IsShortRange(enemy->Gun))
131 {
132 enemy->speed = enemy->speed * 4 / 3;
133 }
134 if (IsShortRange(enemy->Gun))
135 {
136 enemy->bot->probabilityToMove = 35 + (rand() % 35);
137 }
138 else
139 {
140 enemy->bot->probabilityToMove = 30 + (rand() % 30);
141 }
142 enemy->bot->probabilityToTrack = 10 + (rand() % 60);
143 if (!enemy->Gun->CanShoot)
144 {
145 enemy->bot->probabilityToShoot = 0;
146 }
147 else if (IsHighDPS(enemy->Gun))
148 {
149 enemy->bot->probabilityToShoot = 1 + (rand() % 3);
150 }
151 else
152 {
153 enemy->bot->probabilityToShoot = 1 + (rand() % 6);
154 }
155 enemy->bot->actionDelay = rand() % (50 + 1);
156 enemy->maxHealth = GenerateQuickPlayParam(
157 ConfigGetEnum(&gConfig, "QuickPlay.EnemyHealth"), 10, 20, 40, 60);
158 enemy->flags = 0;
159}

161static void SetupQuickPlayEnemies(
162 Mission *mission, const int numEnemies, CharacterStore *store)
163{
164 int i;
165 for (i = 0; i < numEnemies; i++)
166 {
167 const GunDescription *gun;
168 CArrayPushBack(&mission->Enemies, &i);
169
170 for (;;)
171 {
172 gun = CArrayGet(
173 &gGunDescriptions.Guns,
174 rand() % (int)gGunDescriptions.Guns.size);
175 if (!gun->IsRealGun)
176 {
177 continue;
178 }
179 // make at least one of each type of enemy:
180 // - Short range weapon
181 // - Long range weapon
182 // - High explosive weapon
183 if (i == 0 && !IsShortRange(gun))
184 {
185 continue;
186 }
187 if (i == 1 && !IsLongRange(gun))
188 {
189 continue;
190 }
191 if (i == 2 &&
192 ConfigGetBool(&gConfig, "QuickPlay.EnemiesWithExplosives") &&
193 !IsHighDPS(gun))
194 {
195 continue;
196 }
197
198 if (!ConfigGetBool(&gConfig, "QuickPlay.EnemiesWithExplosives") &&
199 IsHighDPS(gun))
200 {
201 continue;
202 }
203 break;
204 }
205 Character *ch = CharacterStoreAddOther(store);
206 SetupQuickPlayEnemy(ch, gun);
207 }
208}

213void SetupQuickPlayCampaign(CampaignSetting *setting)
214{
215 Mission *m;
216 CMALLOC(m, sizeof *m);
217 MissionInit(m);
218 RandomStyle(m->WallStyle, &gPicManager.wallStyleNames);
219 RandomStyle(m->FloorStyle, &gPicManager.tileStyleNames);
220 RandomStyle(m->RoomStyle, &gPicManager.tileStyleNames);
221 RandomStyle(m->ExitStyle, &gPicManager.exitStyleNames);
222 RandomStyle(m->KeyStyle, &gPicManager.keyStyleNames);
223 RandomStyle(m->DoorStyle, &gPicManager.doorStyleNames);
224 m->Size = GenerateQuickPlayMapSize(
225 ConfigGetEnum(&gConfig, "QuickPlay.MapSize"));
226 for (;;)
227 {
228 m->Type = (MapType)(rand() % MAPTYPE_COUNT);
229 // Can't randomly generate static maps
230 if (m->Type != MAPTYPE_STATIC)
231 {
232 break;
233 }
234 }
235 switch (m->Type)
236 {
237 case MAPTYPE_CLASSIC:
238 m->u.Classic.Walls = GenerateQuickPlayParam(
239 ConfigGetEnum(&gConfig, "QuickPlay.WallCount"), 0, 5, 15, 30);
240 m->u.Classic.WallLength = GenerateQuickPlayParam(
241 ConfigGetEnum(&gConfig, "QuickPlay.WallLength"), 1, 3, 6, 12);
242 m->u.Classic.CorridorWidth = rand() % 3 + 1;
243 m->u.Classic.Rooms = RandomRoomParams();
244 m->u.Classic.Squares = GenerateQuickPlayParam(
245 ConfigGetEnum(&gConfig, "QuickPlay.SquareCount"), 0, 1, 3, 6);
246 m->u.Classic.Doors.Enabled = rand() % 2;
247 m->u.Classic.Doors.Min = 1;
248 m->u.Classic.Doors.Max = 6;
249 m->u.Classic.Pillars.Count = rand() % 5;
250 m->u.Classic.Pillars.Min = rand() % 3 + 1;
251 m->u.Classic.Pillars.Max = rand() % 3 + m->u.Classic.Pillars.Min;
252 break;
253 case MAPTYPE_CAVE:
254 // TODO: quickplay configs for cave type
255 m->u.Cave.FillPercent = rand() % 40 + 10;
256 m->u.Cave.Repeat = rand() % 6;
257 m->u.Cave.R1 = rand() % 2 + 4;
258 m->u.Cave.R2 = rand() % 5 - 1;
259 m->u.Cave.CorridorWidth = rand() % 3 + 1;
260 m->u.Cave.Rooms = RandomRoomParams();
261 m->u.Cave.Squares = GenerateQuickPlayParam(
262 ConfigGetEnum(&gConfig, "QuickPlay.SquareCount"), 0, 1, 3, 6);
263 break;
264 default:
265 assert(0 && "unknown map type");
266 break;
267 }
268 CharacterStoreTerminate(&setting->characters);
269 CharacterStoreInit(&setting->characters);
270 int c = GenerateQuickPlayParam(
271 ConfigGetEnum(&gConfig, "QuickPlay.EnemyCount"), 3, 5, 8, 12);
272 SetupQuickPlayEnemies(m, c, &setting->characters);
273
274 c = GenerateQuickPlayParam(
275 ConfigGetEnum(&gConfig, "QuickPlay.ItemCount"), 0, 2, 5, 10);
276 for (int i = 0; i < c; i++)
277 {
278 MapObjectDensity mop;
279 mop.M = IndexMapObject(rand() % MapObjectsCount(&gMapObjects));
280 mop.Density = GenerateQuickPlayParam(
281 ConfigGetEnum(&gConfig, "QuickPlay.ItemCount"), 0, 5, 10, 20);
282 CArrayPushBack(&m->MapObjectDensities, &mop);
283 }
284 m->EnemyDensity = (40 + (rand() % 20)) / m->Enemies.size;
285 CA_FOREACH(const GunDescription, g, gGunDescriptions.Guns)
286 if (g->IsRealGun)
287 {
288 CArrayPushBack(&m->Weapons, &g);
289 }
290 CA_FOREACH_END()
291 m->WallMask = RandomBGColor();
292 m->FloorMask = RandomBGColor();
293 m->RoomMask = RandomBGColor();
294 m->AltMask = RandomBGColor();
295
296 CFREE(setting->Title);
297 CSTRDUP(setting->Title, "Quick play");
298 CFREE(setting->Author);
299 CSTRDUP(setting->Author, "");
300 CFREE(setting->Description);
301 CSTRDUP(setting->Description, "");
302 CArrayPushBack(&setting->Missions, m);
303 CFREE(m);
304}
305static RoomParams RandomRoomParams(void)
306{
307 RoomParams r;
308 r.Count = GenerateQuickPlayParam(
309 ConfigGetEnum(&gConfig, "QuickPlay.RoomCount"), 0, 2, 5, 12);
310 r.Min = rand() % 10 + 5;
311 r.Max = rand() % 10 + r.Min;
312 r.Edge = 1;
313 r.Overlap = 1;
314 r.Walls = rand() % 5;
315 r.WallLength = rand() % 6 + 1;
316 r.WallPad = rand() % 4 + 1;
317 return r;
318}
319static void RandomStyle(char *style, const CArray *styleNames)
320{
321 const int idx = rand() % styleNames->size;
322 strcpy(style, *(char **)CArrayGet(styleNames, idx));
323}
324static color_t RandomBGColor(void)
325{
326 color_t c;
327 c.r = rand() % 128;
328 c.g = rand() % 128;
329 c.b = rand() % 128;
330 c.a = 255;
331 return c;
332}
