40void CharacterStoreInit(CharacterStore *store)
41{
42 memset(store, 0, sizeof *store);
43 CArrayInit(&store->OtherChars, sizeof(Character));
44 CArrayInit(&store->prisonerIds, sizeof(int));
45 CArrayInit(&store->baddieIds, sizeof(int));
46 CArrayInit(&store->specialIds, sizeof(int));
47}
48
49void CharacterStoreTerminate(CharacterStore *store)
50{
51 CA_FOREACH(Character, c, store->OtherChars)
52 CFREE(c->bot);
53 CA_FOREACH_END()
54 CArrayTerminate(&store->OtherChars);
55 CArrayTerminate(&store->prisonerIds);
56 CArrayTerminate(&store->baddieIds);
57 CArrayTerminate(&store->specialIds);
58 memset(store, 0, sizeof *store);
59}

61void CharacterStoreResetOthers(CharacterStore *store)
62{
63 CArrayClear(&store->prisonerIds);
64 CArrayClear(&store->baddieIds);
65 CArrayClear(&store->specialIds);
66}

68void CharacterLoadJSON(CharacterStore *c, json_t *root, int version)
69{
70 LoadInt(&version, root, "Version");
71 json_t *child = json_find_first_label(root, "Characters")->child->child;
72 CharacterStoreTerminate(c);
73 CharacterStoreInit(c);
74 while (child)
75 {
76 Character *ch = CharacterStoreAddOther(c);
77 char *tmp;
78 if (version < 7)
79 {
80 // Old version stored character looks as palette indices
81 int face;
82 LoadInt(&face, child, "face");
83 ch->Class = IntCharacterClass(face);
84 int skin, arm, body, leg, hair;
85 LoadInt(&skin, child, "skin");
86 LoadInt(&arm, child, "arm");
87 LoadInt(&body, child, "body");
88 LoadInt(&leg, child, "leg");
89 LoadInt(&hair, child, "hair");
90 ConvertCharacterColors(skin, arm, body, leg, hair, &ch->Colors);
91 }
92 else
93 {
94 tmp = GetString(child, "Class");
95 ch->Class = StrCharacterClass(tmp);
96 CFREE(tmp);
97 LoadColor(&ch->Colors.Skin, child, "Skin");
98 LoadColor(&ch->Colors.Arms, child, "Arms");
99 LoadColor(&ch->Colors.Body, child, "Body");
100 LoadColor(&ch->Colors.Legs, child, "Legs");
101 LoadColor(&ch->Colors.Hair, child, "Hair");
102 }
103 // Hair colour correction; some characters had no hair but now with
104 // specific parts of the head colourised using the hair colour; set
105 // default "hair" colour based on the head type
106 if (version < 12)
107 {
108 const color_t darkRed = {0xC0, 0, 0, 0xFF};
109 if (strcmp(ch->Class->Name, "Cyborg") == 0)
110 {
111 // eye
112 ch->Colors.Hair = colorRed;
113 }
114 else if (strcmp(ch->Class->Name, "Ice") == 0)
115 {
116 // shades
117 ch->Colors.Hair = colorBlack;
118 }
119 else if (strcmp(ch->Class->Name, "Ogre") == 0)
120 {
121 // eyes
122 ch->Colors.Hair = darkRed;
123 }
124 else if (strcmp(ch->Class->Name, "Snake") == 0)
125 {
126 // eyepatch
127 ch->Colors.Hair = colorBlack;
128 }
129 else if (strcmp(ch->Class->Name, "WarBaby") == 0)
130 {
131 // beret
132 ch->Colors.Hair = colorRed;
133 }
134 }
135 LoadInt(&ch->speed, child, "speed");
136 tmp = GetString(child, "Gun");
137 ch->Gun = StrGunDescription(tmp);
138 CFREE(tmp);
139 LoadInt(&ch->maxHealth, child, "maxHealth");
140 int flags;
141 LoadInt(&flags, child, "flags");
142 ch->flags = flags;
143 LoadInt(&ch->bot->probabilityToMove, child, "probabilityToMove");
144 LoadInt(&ch->bot->probabilityToTrack, child, "probabilityToTrack");
145 LoadInt(&ch->bot->probabilityToShoot, child, "probabilityToShoot");
146 LoadInt(&ch->bot->actionDelay, child, "actionDelay");
147 child = child->next;
148 }
149}

151bool CharacterSave(CharacterStore *s, const char *path)
152{
153 json_t *root = json_new_object();
154 AddIntPair(root, "Version", CHARACTER_VERSION);
155 bool res = true;
156
157 json_t *charNode = json_new_array();
158 CA_FOREACH(Character, c, s->OtherChars)
159 json_t *node = json_new_object();
160 AddStringPair(node, "Class", c->Class->Name);
161 AddColorPair(node, "Skin", c->Colors.Skin);
162 AddColorPair(node, "Arms", c->Colors.Arms);
163 AddColorPair(node, "Body", c->Colors.Body);
164 AddColorPair(node, "Legs", c->Colors.Legs);
165 AddColorPair(node, "Hair", c->Colors.Hair);
166 AddIntPair(node, "speed", c->speed);
167 json_insert_pair_into_object(
168 node, "Gun", json_new_string(c->Gun->name));
169 AddIntPair(node, "maxHealth", c->maxHealth);
170 AddIntPair(node, "flags", c->flags);
171 AddIntPair(node, "probabilityToMove", c->bot->probabilityToMove);
172 AddIntPair(node, "probabilityToTrack", c->bot->probabilityToTrack);
173 AddIntPair(node, "probabilityToShoot", c->bot->probabilityToShoot);
174 AddIntPair(node, "actionDelay", c->bot->actionDelay);
175 json_insert_child(charNode, node);
176 CA_FOREACH_END()
177 json_insert_pair_into_object(root, "Characters", charNode);
178 char buf[CDOGS_PATH_MAX];
179 sprintf(buf, "%s/characters.json", path);
180 if (!TrySaveJSONFile(root, buf))
181 {
182 res = false;
183 goto bail;
184 }
185
186bail:
187 json_free_value(&root);
188 return res;
189}

191Character *CharacterStoreAddOther(CharacterStore *store)
192{
193 return CharacterStoreInsertOther(store, store->OtherChars.size);
194}
195Character *CharacterStoreInsertOther(CharacterStore *store, int idx)
196{
197 Character newChar;
198 memset(&newChar, 0, sizeof newChar);
199 CCALLOC(newChar.bot, sizeof *newChar.bot);
200 CArrayInsert(&store->OtherChars, idx, &newChar);
201 return CArrayGet(&store->OtherChars, idx);
202}

250bool CharacterIsPrisoner(const CharacterStore *store, const Character *c)
251{
252 for (int i = 0; i < (int)store->prisonerIds.size; i++)
253 {
254 const Character *prisoner = CArrayGet(
255 &store->OtherChars, CharacterStoreGetPrisonerId(store, i));
256 if (prisoner == c)
257 {
258 return true;
259 }
260 }
261 return false;
262}

264int CharacterGetStartingHealth(const Character *c, const bool isNPC)
265{
266 if (isNPC)
267 {
268 return MAX(
269 (c->maxHealth * ConfigGetInt(&gConfig, "Game.NonPlayerHP")) / 100,
270 1);
271 }
272 else
273 {
274 return c->maxHealth;
275 }
276}

278static color_t RandomColor(void);
279void CharacterShuffleAppearance(Character *c)
280{
281 // Choose a random character class
282 const int numCharClasses =
283 (int)gCharacterClasses.Classes.size +
284 (int)gCharacterClasses.CustomClasses.size;
285 const int charClass = rand() % numCharClasses;
286 if (charClass < (int)gCharacterClasses.Classes.size)
287 {
288 c->Class = CArrayGet(&gCharacterClasses.Classes, charClass);
289 }
290 else
291 {
292 c->Class = CArrayGet(
293 &gCharacterClasses.Classes,
294 charClass - gCharacterClasses.Classes.size);
295 }
296 c->Colors.Skin = RandomColor();
297 c->Colors.Arms = RandomColor();
298 c->Colors.Body = RandomColor();
299 c->Colors.Legs = RandomColor();
300 c->Colors.Hair = RandomColor();
301}
302static color_t RandomColor(void)
303{
304 color_t c;
305 c.r = rand() & 255;
306 c.g = rand() & 255;
307 c.b = rand() & 255;
308 c.a = 255;
309 return c;
310}
