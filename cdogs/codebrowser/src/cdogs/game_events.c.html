121void GameEventsEnqueue(CArray *store, GameEvent e)
122{
123 if (store->elemSize == 0)
124 {
125 return;
126 }
127 // If we're the server, broadcast any events that clients need
128 // If we're the client, pass along to server, but only if it's for a local player
129 // Otherwise we'd ping-pong the same updates from the server
130 const GameEventEntry gee = sGameEventEntries[e.Type];
131 if (gee.Broadcast)
132 {
133 NetServerSendMsg(&gNetServer, NET_SERVER_BCAST, gee.Type, &e.u);
134 }
135 if (gee.Submit)
136 {
137 int actorUID = -1;
138 bool actorIsLocal = false;
139 switch (e.Type)
140 {
141 case GAME_EVENT_ACTOR_MOVE: actorUID = e.u.ActorMove.UID; break;
142 case GAME_EVENT_ACTOR_STATE: actorUID = e.u.ActorState.UID; break;
143 case GAME_EVENT_ACTOR_DIR: actorUID = e.u.ActorDir.UID; break;
144 case GAME_EVENT_ACTOR_SLIDE: actorUID = e.u.ActorSlide.UID; break;
145 case GAME_EVENT_ACTOR_SWITCH_GUN: actorUID = e.u.ActorSwitchGun.UID; break;
146 case GAME_EVENT_ACTOR_PICKUP_ALL: actorUID = e.u.ActorPickupAll.UID; break;
147 case GAME_EVENT_ACTOR_USE_AMMO: actorUID = e.u.UseAmmo.UID; break;
148 case GAME_EVENT_ACTOR_MELEE: actorUID = e.u.Melee.UID; break;
149 case GAME_EVENT_GUN_FIRE:
150 if (e.u.GunFire.IsGun)
151 {
152 actorIsLocal = PlayerIsLocal(e.u.GunFire.PlayerUID);
153 }
154 break;
155 case GAME_EVENT_GUN_RELOAD:
156 actorIsLocal = PlayerIsLocal(e.u.GunReload.PlayerUID);
157 break;
158 case GAME_EVENT_GUN_STATE: actorUID = e.u.GunState.ActorUID; break;
159 default: break;
160 }
161 if (actorUID >= 0)
162 {
163 actorIsLocal = ActorIsLocalPlayer(actorUID);
164 }
165 if (actorIsLocal)
166 {
167 NetClientSendMsg(&gNetClient, gee.Type, &e.u);
168 }
169 }
170
171 CArrayPushBack(store, &e);
172}
173static bool EventComplete(const void *elem);
174void GameEventsClear(CArray *store)
175{
176 CArrayRemoveIf(store, EventComplete);
177}
178static bool EventComplete(const void *elem)
179{
180 return ((const GameEvent *)elem)->Delay < 0;
181}

183GameEvent GameEventNew(GameEventType type)
184{
185 GameEvent e;
186 memset(&e, 0, sizeof e);
187 e.Type = type;
188 return e;
189}
