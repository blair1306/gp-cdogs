38color_t PixelToColor(
39 const SDL_PixelFormat *f, const Uint8 aShift, const Uint32 pixel)
40{
41 color_t c;
42 SDL_GetRGBA(pixel, f, &c.r, &c.g, &c.b, &c.a);
43 // Manually apply the alpha as SDL seems to always set it to 0
44 c.a = (Uint8)((pixel & ~(f->Rmask | f->Gmask | f->Bmask)) >> aShift);
45 return c;
46}
47Uint32 ColorToPixel(
48 const SDL_PixelFormat *f, const Uint8 aShift, const color_t color)
49{
50 const Uint32 pixel = SDL_MapRGBA(f, color.r, color.g, color.b, color.a);
51 // Manually apply the alpha as SDL seems to always set it to 0
52 return (pixel & (f->Rmask | f->Gmask | f->Bmask)) | (color.a << aShift);
53}

56void PicLoad(
57 Pic *p, const Vec2i size, const Vec2i offset, const SDL_Surface *image)
58{
59 p->size = size;
60 p->offset = Vec2iZero();
61 CMALLOC(p->Data, size.x * size.y * sizeof *((Pic *)0)->Data);
62 // Manually copy the pixels and replace the alpha component,
63 // since our gfx device format has no alpha
64 int srcI = offset.y*image->w + offset.x;
65 for (int i = 0; i < size.x * size.y; i++, srcI++)
66 {
67 const Uint32 pixel = ((Uint32 *)image->pixels)[srcI];
68 color_t c;
69 SDL_GetRGBA(pixel, image->format, &c.r, &c.g, &c.b, &c.a);
70 // If completely transparent, replace rgb with black (0) too
71 // This is because transparency blitting checks entire pixel
72 if (c.a == 0)
73 {
74 p->Data[i] = 0;
75 }
76 else
77 {
78 p->Data[i] = COLOR2PIXEL(c);
79 }
80 if ((i + 1) % size.x == 0)
81 {
82 srcI += image->w - size.x;
83 }
84 }
85}

87Pic PicCopy(const Pic *src)
88{
89 Pic p = *src;
90 const size_t size = p.size.x * p.size.y * sizeof *p.Data;
91 CMALLOC(p.Data, size);
92 memcpy(p.Data, src->Data, size);
93 return p;
94}

106void PicTrim(Pic *pic, const bool xTrim, const bool yTrim)
107{
108 // Scan all pixels looking for the min/max of x and y
109 Vec2i min = pic->size;
110 Vec2i max = Vec2iZero();
111 for (Vec2i pos = Vec2iZero(); pos.y < pic->size.y; pos.y++)
112 {
113 for (pos.x = 0; pos.x < pic->size.x; pos.x++)
114 {
115 const Uint32 pixel = *(pic->Data + pos.x + pos.y * pic->size.x);
116 if (pixel > 0)
117 {
118 min.x = MIN(min.x, pos.x);
119 min.y = MIN(min.y, pos.y);
120 max.x = MAX(max.x, pos.x);
121 max.y = MAX(max.y, pos.y);
122 }
123 }
124 }
125 // If no opaque pixels found, don't trim
126 Vec2i newSize = pic->size;
127 Vec2i offset = Vec2iZero();
128 if (min.x < max.x && min.y < max.y)
129 {
130 if (xTrim)
131 {
132 newSize.x = max.x - min.x + 1;
133 offset.x = min.x;
134 }
135 if (yTrim)
136 {
137 newSize.y = max.y - min.y + 1;
138 offset.y = min.y;
139 }
140 }
141 // Trim by copying pixels
142 Uint32 *newData;
143 CMALLOC(newData, newSize.x * newSize.y * sizeof *newData);
144 for (Vec2i pos = Vec2iZero(); pos.y < newSize.y; pos.y++)
145 {
146 for (pos.x = 0; pos.x < newSize.x; pos.x++)
147 {
148 Uint32 *target = newData + pos.x + pos.y * newSize.x;
149 const int srcIdx =
150 pos.x + offset.x + (pos.y + offset.y) * pic->size.x;
151 *target = *(pic->Data + srcIdx);
152 }
153 }
154 // Replace the old data
155 CFREE(pic->Data);
156 pic->Data = newData;
157 pic->size = newSize;
158 pic->offset = Vec2iZero();
159}

161bool PicPxIsEdge(const Pic *pic, const Vec2i pos, const bool isPixel)
162{
163 const bool isTopOrBottomEdge = pos.y == -1 || pos.y == pic->size.y;
164 const bool isLeftOrRightEdge = pos.x == -1 || pos.x == pic->size.x;
165 const bool isLeft =
166 pos.x > 0 && !isTopOrBottomEdge &&
167 PIXEL2COLOR(*(pic->Data + pos.x - 1 + pos.y * pic->size.x)).a;
168 const bool isRight =
169 pos.x < pic->size.x - 1 && !isTopOrBottomEdge &&
170 PIXEL2COLOR(*(pic->Data + pos.x + 1 + pos.y * pic->size.x)).a;
171 const bool isAbove =
172 pos.y > 0 && !isLeftOrRightEdge &&
173 PIXEL2COLOR(*(pic->Data + pos.x + (pos.y - 1) * pic->size.x)).a;
174 const bool isBelow =
175 pos.y < pic->size.y - 1 && !isLeftOrRightEdge &&
176 PIXEL2COLOR(*(pic->Data + pos.x + (pos.y + 1) * pic->size.x)).a;
177 if (isPixel)
178 {
179 return !(isLeft && isRight && isAbove && isBelow);
180 }
181 else
182 {
183 return isLeft || isRight || isAbove || isBelow;
184 }
185}
