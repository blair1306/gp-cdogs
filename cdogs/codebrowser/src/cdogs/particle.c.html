54void ParticleClassesInit(ParticleClasses *classes, const char *filename)
55{
56 CArrayInit(&classes->Classes, sizeof(ParticleClass));
57 CArrayInit(&classes->CustomClasses, sizeof(ParticleClass));
58
59 char buf[CDOGS_PATH_MAX];
60 GetDataFilePath(buf, filename);
61 FILE *f = fopen(buf, "r");
62 json_t *root = NULL;
63 if (f == NULL)
64 {
65 LOG(LM_MAIN, LL_ERROR, "Error: cannot load particles file %s", buf);
66 goto bail;
67 }
68 enum json_error e = json_stream_parse(f, &root);
69 if (e != JSON_OK)
70 {
71 LOG(LM_MAIN, LL_ERROR, "Error parsing particles file %s", buf);
72 goto bail;
73 }
74 ParticleClassesLoadJSON(&classes->Classes, root);
75
76bail:
77 if (f != NULL)
78 {
79 fclose(f);
80 }
81 json_free_value(&root);
82}
83void ParticleClassesLoadJSON(CArray *classes, json_t *root)
84{
85 int version;
86 LoadInt(&version, root, "Version");
87 if (version > VERSION || version <= 0)
88 {
89 CASSERT(false, "cannot read particles file version");
90 return;
91 }
92
93 json_t *particlesNode = json_find_first_label(root, "Particles")->child;
94 for (json_t *child = particlesNode->child; child; child = child->next)
95 {
96 ParticleClass c;
97 LoadParticleClass(&c, child, version);
98 CArrayPushBack(classes, &c);
99 }
100}
101void ParticleClassesTerminate(ParticleClasses *classes)
102{
103 ParticleClassesClear(&classes->Classes);
104 CArrayTerminate(&classes->Classes);
105 ParticleClassesClear(&classes->CustomClasses);
106 CArrayTerminate(&classes->CustomClasses);
107}

109{

111 {

113 CFREE(c->Name);
114 }
115 CArrayClear(classes);
116}
117static void LoadParticleClass(
118 ParticleClass *c, json_t *node, const int version)
119{
120 memset(c, 0, sizeof *c);
121 char *tmp;
122
123 c->Name = GetString(node, "Name");
124
125 c->Type = PARTICLE_PIC;
126 if (version < 2)
127 {
128 c->u.Pic.UseMask = true;
129 c->u.Pic.u1.Mask = colorWhite;
130 if (json_find_first_label(node, "Sprites"))
131 {
132 tmp = GetString(node, "Sprites");
133 c->u.Pic.Type = PICTYPE_DIRECTIONAL;
134 c->u.Pic.u.Sprites =
135 &PicManagerGetSprites(&gPicManager, tmp)->pics;
136 CFREE(tmp);
137 }
138 else
139 {
140 LoadPic(&c->u.Pic.u.Pic, node, "Pic");
141 }
142 if (json_find_first_label(node, "Mask"))
143 {
144 tmp = GetString(node, "Mask");
145 c->u.Pic.u1.Mask = StrColor(tmp);
146 CFREE(tmp);
147 }
148 int ticksPerFrame = 0;
149 LoadInt(&ticksPerFrame, node, "TicksPerFrame");
150 if (ticksPerFrame > 0)
151 {
152 c->u.Pic.Type = PICTYPE_ANIMATED;
153 c->u.Pic.u.Animated.Sprites = c->u.Pic.u.Sprites;
154 c->u.Pic.u.Animated.TicksPerFrame = ticksPerFrame;
155 }
156 }
157 else
158 {
159 tmp = NULL;
160 LoadStr(&tmp, node, "Type");
161 if (tmp != NULL)
162 {
163 c->Type = StrParticleType(tmp);
164 CFREE(tmp);
165 }
166 switch (c->Type)
167 {
168 case PARTICLE_PIC:
169 CPicLoadJSON(
170 &c->u.Pic, json_find_first_label(node, "Pic")->child);
171 break;
172 case PARTICLE_TEXT:
173 c->u.TextColor = colorWhite;
174 tmp = NULL;
175 LoadStr(&tmp, node, "TextMask");
176 if (tmp != NULL)
177 {
178 c->u.TextColor = StrColor(tmp);
179 CFREE(tmp)
180 }
181 break;
182 default:
183 break;
184 }
185 }
186
187 if (json_find_first_label(node, "Range"))
188 {
189 LoadInt(&c->RangeLow, node, "Range");
190 c->RangeHigh = c->RangeLow;
191 }
192 if (json_find_first_label(node, "RangeLow"))
193 {
194 LoadInt(&c->RangeLow, node, "RangeLow");
195 }
196 if (json_find_first_label(node, "RangeHigh"))
197 {
198 LoadInt(&c->RangeHigh, node, "RangeHigh");
199 }
200 c->RangeLow = MIN(c->RangeLow, c->RangeHigh);
201 c->RangeHigh = MAX(c->RangeLow, c->RangeHigh);
202 LoadInt(&c->GravityFactor, node, "GravityFactor");
203 LoadBool(&c->HitsWalls, node, "HitsWalls");
204 c->Bounces = true;
205 LoadBool(&c->Bounces, node, "Bounces");
206 c->WallBounces = true;
207 LoadBool(&c->WallBounces, node, "WallBounces");
208}

210const ParticleClass *StrParticleClass(
211 const ParticleClasses *classes, const char *name)
212{
213 if (name == NULL || strlen(name) == 0)
214 {
215 return NULL;
216 }
217 CA_FOREACH(const ParticleClass, c, classes->CustomClasses)
218 if (strcmp(c->Name, name) == 0)
219 {
220 return c;
221 }
222 CA_FOREACH_END()
223 CA_FOREACH(const ParticleClass, c, classes->Classes)
224 if (strcmp(c->Name, name) == 0)
225 {
226 return c;
227 }
228 CA_FOREACH_END()
229 CASSERT(false, "Cannot find particle class");
230 return NULL;
231}

238void ParticlesTerminate(CArray *particles)
239{
240 for (int i = 0; i < (int)particles->size; i++)
241 {
242 Particle *p = CArrayGet(particles, i);
243 if (p->isInUse)
244 {
245 ParticleDestroy(particles, i);
246 }
247 }
248 CArrayTerminate(particles);
249}

251static bool ParticleUpdate(Particle *p, const int ticks);
252void ParticlesUpdate(CArray *particles, const int ticks)
253{
254 for (int i = 0; i < (int)particles->size; i++)
255 {
256 Particle *p = CArrayGet(particles, i);
257 if (!p->isInUse)
258 {
259 continue;
260 }
261 if (!ParticleUpdate(p, ticks))
262 {
263 GameEvent e = GameEventNew(GAME_EVENT_PARTICLE_REMOVE);
264 e.u.ParticleRemoveId = i;
265 GameEventsEnqueue(&gGameEvents, e);
266 }
267 }

271typedef struct
272{
273 const TTileItem *Obj;
274 Vec2i ColPos;
275 Vec2i ColNormal;
276 int ColPosDistRealSquared;
277} HitWallData;
278static bool CheckWall(const Vec2i tilePos);
279static bool HitWallFunc(
280 const Vec2i tilePos, void *data, const Vec2i col, const Vec2i normal);
281static bool ParticleUpdate(Particle *p, const int ticks)
282{
283 switch(p->Class->Type)
284 {
285 case PARTICLE_PIC:
286 CPicUpdate(&p->u.Pic, ticks);
287 break;
288 default:
289 break;
290 }
291 p->Count += ticks;
292 const Vec2i startPos = p->Pos;
293 for (int i = 0; i < ticks; i++)
294 {
295 p->Pos = Vec2iAdd(p->Pos, p->tileItem.VelFull);
296 p->Z += p->DZ;
297 if (p->Class->GravityFactor != 0)
298 {
299 if (p->Z <= 0)
300 {
301 p->Z = 0;
302 if (p->Class->Bounces)
303 {
304 p->DZ = -p->DZ / 2;
305 }
306 else
307 {
308 p->DZ = 0;
309 }
310 }
311 else
312 {
313 p->DZ -= p->Class->GravityFactor;
314 }
315 if (p->DZ == 0 && p->Z == 0)
316 {
317 p->tileItem.VelFull = Vec2iZero();
318 p->Spin = 0;
319 // Fell to ground, draw last
320 p->tileItem.flags |= TILEITEM_DRAW_LAST;
321 }
322 }
323 }
324 // Wall collision, bounce off walls
325 if (!Vec2iIsZero(p->tileItem.VelFull) && p->Class->HitsWalls)
326 {
327 const CollisionParams params =
328 {
329 0, COLLISIONTEAM_NONE, IsPVP(gCampaign.Entry.Mode)
330 };
331 HitWallData data = { &p->tileItem, Vec2iZero(), Vec2iZero(), -1 };
332 OverlapTileItems(
333 &p->tileItem, startPos,
334 p->tileItem.size, params, NULL, NULL,
335 CheckWall, HitWallFunc, &data);
336 if (data.ColPosDistRealSquared >= 0)
337 {
338 if (p->Class->WallBounces)
339 {
340 GetWallBouncePosVelFull(
341 startPos, p->tileItem.VelFull, data.ColPos, data.ColNormal,
342 &p->Pos, &p->tileItem.VelFull);
343 }
344 else
345 {
346 p->tileItem.VelFull = Vec2iZero();
347 }
348 }
349 }
350 const Vec2i realPos = Vec2iFull2Real(p->Pos);
351 if (!MapTryMoveTileItem(&gMap, &p->tileItem, realPos))
352 {
353 // Out of map; destroy
354 return false;
355 }
356
357 // Spin
358 p->Angle += p->Spin;
359 if (p->Angle > 2 * PI)
360 {
361 p->Angle -= PI * 2;
362 }
363 if (p->Angle < 0)
364 {
365 p->Angle += PI * 2;
366 }
367
368 return p->Count <= p->Range;
369}
370static void SetClosestCollision(
371 HitWallData *data, const Vec2i col, const Vec2i normal);
372static bool CheckWall(const Vec2i tilePos)
373{
374 const Tile *t = MapGetTile(&gMap, tilePos);
375 return t == NULL || t->flags & MAPTILE_NO_SHOOT;
376}
377static bool HitWallFunc(
378 const Vec2i tilePos, void *data, const Vec2i col, const Vec2i normal)
379{
380 UNUSED(tilePos);
381 HitWallData *hData = data;
382 SetClosestCollision(hData, col, normal);
383 return true;
384}
385static void SetClosestCollision(
386 HitWallData *data, const Vec2i col, const Vec2i normal)
387{
388 // Choose the best collision point (i.e. closest to origin)
389 const int d2 = DistanceSquared(
390 Vec2iFull2Real(col), Vec2iNew(data->Obj->x, data->Obj->y));
391 if (data->ColPosDistRealSquared < 0 || d2 < data->ColPosDistRealSquared)
392 {
393 data->ColPos = col;
394 data->ColPosDistRealSquared = d2;
395 data->ColNormal = normal;
396 }
397}

399static void DrawParticle(const Vec2i pos, const TileItemDrawFuncData *data);
400int ParticleAdd(CArray *particles, const AddParticle add)
401{
402 // Find an empty slot in list
403 Particle *p = NULL;
404 int i;
405 for (i = 0; i < (int)particles->size; i++)
406 {
407 Particle *pSlot = CArrayGet(particles, i);
408 if (!pSlot->isInUse)
409 {
410 p = pSlot;
411 break;
412 }
413 }
414 // If no empty slots, add a new one
415 if (p == NULL)
416 {
417 Particle pNew;
418 memset(&pNew, 0, sizeof pNew);
419 CArrayPushBack(particles, &pNew);
420 i = (int)particles->size - 1;
421 p = CArrayGet(particles, i);
422 }
423 memset(p, 0, sizeof *p);
424 p->Class = add.Class;
425 switch (p->Class->Type)
426 {
427 case PARTICLE_PIC:
428 CPicCopyPic(&p->u.Pic, &p->Class->u.Pic);
429 break;
430 case PARTICLE_TEXT:
431 CSTRDUP(p->u.Text, add.Text);
432 break;
433 default:
434 break;
435 }
436 p->Pos = add.FullPos;
437 p->Z = add.Z;
438 p->Angle = add.Angle;
439 p->DZ = add.DZ;
440 p->Spin = add.Spin;
441 p->Range = RAND_INT(add.Class->RangeLow, add.Class->RangeHigh);
442 p->isInUse = true;
443 p->tileItem.x = p->tileItem.y = -1;
444 p->tileItem.VelFull = add.Vel;
445 p->tileItem.kind = KIND_PARTICLE;
446 p->tileItem.id = i;
447 p->tileItem.drawFunc = DrawParticle;
448 p->tileItem.drawData.MobObjId = i;
449 MapTryMoveTileItem(&gMap, &p->tileItem, Vec2iFull2Real(add.FullPos));
450 return i;
451}
452void ParticleDestroy(CArray *particles, const int id)
453{
454 Particle *p = CArrayGet(particles, id);
455 CASSERT(p->isInUse, "Destroying not-in-use particle");
456 MapRemoveTileItem(&gMap, &p->tileItem);
457 if (p->Class->Type == PARTICLE_TEXT)
458 {
459 CFREE(p->u.Text);
460 }
461 p->isInUse = false;
462}

464static void DrawParticle(const Vec2i pos, const TileItemDrawFuncData *data)
465{
466 const Particle *p = CArrayGet(&gParticles, data->MobObjId);
467 CASSERT(p->isInUse, "Cannot draw non-existent particle");
468 switch (p->Class->Type)
469 {
470 case PARTICLE_PIC:
471 {
472 CPicDrawContext context;
473 context.Dir = RadiansToDirection(p->Angle);
474 const Pic *pic = CPicGetPic(&p->u.Pic, context.Dir);
475 context.Offset = Vec2iNew(
476 pic->size.x / -2, pic->size.y / -2 - p->Z / Z_FACTOR);
477 CPicDraw(
478 &gGraphicsDevice, &p->Class->u.Pic, pos, &context);
479 break;
480 }
481 case PARTICLE_TEXT:
482 {
483 FontOpts opts = FontOptsNew();
484 opts.HAlign = ALIGN_CENTER;
485 opts.Mask = p->Class->u.TextColor;
486 FontStrOpt(
487 p->u.Text, Vec2iNew(pos.x, pos.y - p->Z / Z_FACTOR), opts);
488 break;
489 }
490 default:
491 break;
492 }
