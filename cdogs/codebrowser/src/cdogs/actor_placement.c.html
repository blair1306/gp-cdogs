38static NVec2i PlaceActor(Map *map)
39{
40 Vec2i pos;
41
42 if (HasExit(gCampaign.Entry.Mode))
43 {
44 // First, try to place at least half the map away from the exit
45 const int halfMap =
46 MAX(map->Size.x * TILE_WIDTH, map->Size.y * TILE_HEIGHT) / 2;
47 const Vec2i exitPos = MapGetExitPos(map);
48 // Don't try forever trying to place
49 for (int i = 0; i < 100; i++)
50 {
51 const Vec2i realPos = Vec2iNew(
52 rand() % (map->Size.x * TILE_WIDTH),
53 rand() % (map->Size.y * TILE_HEIGHT));
54 pos = Vec2iFull2Real(realPos);
55 if (abs(realPos.x - exitPos.x) > halfMap &&
56 abs(realPos.y - exitPos.y) > halfMap &&
57 MapIsTileAreaClear(map, pos, Vec2iNew(ACTOR_W, ACTOR_H)))
58 {
59 return Vec2i2Net(pos);
60 }
61 }
62 }
63
64 // Try to place randomly
65 do
66 {
67 pos.x = ((rand() % (map->Size.x * TILE_WIDTH)) << 8);
68 pos.y = ((rand() % (map->Size.y * TILE_HEIGHT)) << 8);
69 } while (!MapIsFullPosOKforPlayer(map, pos, false) ||
70 !MapIsTileAreaClear(map, pos, Vec2iNew(ACTOR_W, ACTOR_H)));
71 return Vec2i2Net(pos);
72}

74static NVec2i PlaceActorNear(
75 Map *map, const Vec2i nearPos, const bool allowAllTiles)
76{
77 // Try a concentric rhombus pattern, clockwise from right
78 // That is, start by checking right, below, left, above,
79 // then continue with radius 2 right, below-right, below, below-left...
80 // (start from S:)
81 // 4
82 // 9 3 S 1 5
83 // 8 2 6
84 // 7
85#define TRY_LOCATION()\
86 pos = Vec2iAdd(nearPos, Vec2iNew(dx, dy));\
87 if (MapIsFullPosOKforPlayer(map, pos, allowAllTiles) && \
88 MapIsTileAreaClear(map, pos, Vec2iNew(ACTOR_W, ACTOR_H)))\
89 {\
90 return Vec2i2Net(pos);\
91 }
92 int dx = 0;
93 int dy = 0;
94 Vec2i pos;
95 TRY_LOCATION();
96 int inc = 1 << 8;
97 for (int radius = 12 << 8;; radius += 12 << 8)
98 {
99 // Going from right to below
100 for (dx = radius, dy = 0; dy < radius; dx -= inc, dy += inc)
101 {
102 TRY_LOCATION();
103 }
104 // below to left
105 for (dx = 0, dy = radius; dy > 0; dx -= inc, dy -= inc)
106 {
107 TRY_LOCATION();
108 }
109 // left to above
110 for (dx = -radius, dy = 0; dx < 0; dx += inc, dy -= inc)
111 {
112 TRY_LOCATION();
113 }
114 // above to right
115 for (dx = 0, dy = -radius; dy < 0; dx += inc, dy += inc)
116 {
117 TRY_LOCATION();
118 }
119 }
120}

122NVec2i PlaceAwayFromPlayers(Map *map, const bool giveUp)
123{
124 // Don't try forever trying to place
125 for (int i = 0; i < 100; i++)
126 {
127 // Try spawning out of players' sights
128 const Vec2i pos = Vec2iReal2Full(Vec2iNew(
129 rand() % (map->Size.x * TILE_WIDTH),
130 rand() % (map->Size.y * TILE_HEIGHT)));
131 const TActor *closestPlayer = AIGetClosestPlayer(pos);
132 if (closestPlayer && CHEBYSHEV_DISTANCE(
133 pos.x, pos.y,
134 closestPlayer->Pos.x, closestPlayer->Pos.y) >= 256 * 150 &&
135 MapIsTileAreaClear(map, pos, Vec2iNew(ACTOR_W, ACTOR_H)))
136 {
137 return Vec2i2Net(pos);
138 }
139 }
140 // Keep trying, but this time try spawning anywhere,
141 // even close to player
142 for (int i = 0; i < 10000 || !giveUp; i++)
143 {
144 const Vec2i pos = Vec2iReal2Full(Vec2iNew(
145 rand() % (map->Size.x * TILE_WIDTH),
146 rand() % (map->Size.y * TILE_HEIGHT)));
147 if (MapIsTileAreaClear(map, pos, Vec2iNew(ACTOR_W, ACTOR_H)))
148 {
149 return Vec2i2Net(pos);
150 }
151 }
152 // Uh oh
153 return Vec2i2Net(Vec2iZero());
154}

156NVec2i PlacePrisoner(Map *map)
157{
158 Vec2i fullPos;
159 do
160 {
161 do
162 {
163 fullPos.x = ((rand() % (map->Size.x * TILE_WIDTH)) << 8);
164 fullPos.y = ((rand() % (map->Size.y * TILE_HEIGHT)) << 8);
165 } while (!MapPosIsInLockedRoom(map, Vec2iFull2Real(fullPos)));
166 } while (!MapIsTileAreaClear(map, fullPos, Vec2iNew(ACTOR_W, ACTOR_H)));
167 return Vec2i2Net(fullPos);
168}

170Vec2i PlacePlayer(
171 Map *map, const PlayerData *p, const Vec2i firstPos, const bool pumpEvents)
172{
173 NActorAdd aa = NActorAdd_init_default;
174 aa.UID = ActorsGetNextUID();
175 aa.Health = p->Char.maxHealth;
176 aa.PlayerUID = p->UID;
177
178 if (IsPVP(gCampaign.Entry.Mode))
179 {
180 // In a PVP mode, always place players apart
181 aa.FullPos = PlaceAwayFromPlayers(&gMap, false);
182 }
183 else if (
184 ConfigGetEnum(&gConfig, "Interface.Splitscreen") == SPLITSCREEN_NEVER &&
185 !Vec2iIsZero(firstPos))
186 {
187 // If never split screen, try to place players near the first player
188 aa.FullPos = PlaceActorNear(map, firstPos, true);
189 }
190 else if (gMission.missionData->Type == MAPTYPE_STATIC &&
191 !Vec2iIsZero(gMission.missionData->u.Static.Start))
192 {
193 // place players near the start point
194 Vec2i startPoint = Vec2iReal2Full(Vec2iCenterOfTile(
195 gMission.missionData->u.Static.Start));
196 aa.FullPos = PlaceActorNear(map, startPoint, true);
197 }
198 else
199 {
200 aa.FullPos = PlaceActor(map);
201 }
202
203 GameEvent e = GameEventNew(GAME_EVENT_ACTOR_ADD);
204 e.u.ActorAdd = aa;
205 GameEventsEnqueue(&gGameEvents, e);
206
207 if (pumpEvents)
208 {
209 // Process the events that actually place the players
210 HandleGameEvents(&gGameEvents, NULL, NULL, NULL);
211 }
212
213 return Vec2iNew(aa.FullPos.x, aa.FullPos.y);
214}
