29#pragma once
30
31#include <SDL_events.h>
32#include <SDL_keyboard.h>
33#include <SDL_keycode.h>
34
35#include "config.h"
36
37typedef struct
38{
39 bool isPressed;
40 SDL_Keycode keycode;
41} KeyPress;
42
43typedef struct
44{
45 SDL_Scancode left;
46 SDL_Scancode right;
47 SDL_Scancode up;
48 SDL_Scancode down;
49 SDL_Scancode button1;
50 SDL_Scancode button2;
51 SDL_Scancode map;
52} InputKeys;
53#define MAX_KEYBOARD_CONFIGS 2
54
55typedef enum
56{
57 KEY_CODE_LEFT,
58 KEY_CODE_RIGHT,
59 KEY_CODE_UP,
60 KEY_CODE_DOWN,
61 KEY_CODE_BUTTON1,
62 KEY_CODE_BUTTON2,
63
64 KEY_CODE_MAP
65} key_code_e;
66const char *KeycodeStr(int k);
67
68typedef struct
69{
70 KeyPress previousKeys[SDL_NUM_SCANCODES];
71 KeyPress currentKeys[SDL_NUM_SCANCODES];
72 KeyPress pressedKeys[SDL_NUM_SCANCODES];
73 SDL_Keymod modState;
74 char Typed[32];
75 Uint32 ticks;
76 Uint32 repeatedTicks;
77 bool isFirstRepeat;
78 InputKeys PlayerKeys[MAX_KEYBOARD_CONFIGS];
79} keyboard_t;
80
81void KeyInit(keyboard_t *keyboard);
82InputKeys KeyLoadPlayerKeys(Config *c);
83void KeyPrePoll(keyboard_t *keyboard);
84void KeyOnKeyDown(keyboard_t *keyboard, const SDL_Keysym s);
85void KeyOnKeyUp(keyboard_t *keyboard, const SDL_Keysym s);
86void KeyPostPoll(keyboard_t *keyboard, Uint32 ticks);
87bool KeyIsDown(const keyboard_t *k, const int key);
88bool KeyIsPressed(const keyboard_t *k, const int key);
89bool KeyIsReleased(const keyboard_t *k, const int key);
90SDL_Scancode KeyGetPressed(const keyboard_t *k);
91
92SDL_Scancode KeyGet(const InputKeys *keys, const key_code_e keyCode);
