70void MapAddDoorGroup(
71 Map *map, const Mission *m, const Vec2i v, const int keyFlags)
72{
73 const unsigned short tileLeftType =
74 IMapGet(map, Vec2iNew(v.x - 1, v.y)) & MAP_MASKACCESS;
75 const unsigned short tileRightType =
76 IMapGet(map, Vec2iNew(v.x + 1, v.y)) & MAP_MASKACCESS;
77 const bool isHorizontal =
78 tileLeftType == MAP_WALL || tileRightType == MAP_WALL ||
79 tileLeftType == MAP_DOOR || tileRightType == MAP_DOOR ||
80 tileLeftType == MAP_NOTHING || tileRightType == MAP_NOTHING;
81 const int doorGroupCount = GetDoorCountInGroup(map, v, isHorizontal);
82 const Vec2i dv = Vec2iNew(isHorizontal ? 1 : 0, isHorizontal ? 0 : 1);
83 const Vec2i dAside = Vec2iNew(dv.y, dv.x);
84
85 const char *doorKey;
86 switch (keyFlags)
87 {
88 case FLAGS_KEYCARD_RED: doorKey = "red"; break;
89 case FLAGS_KEYCARD_BLUE: doorKey = "blue"; break;
90 case FLAGS_KEYCARD_GREEN: doorKey = "green"; break;
91 case FLAGS_KEYCARD_YELLOW: doorKey = "yellow"; break;
92 default: doorKey = "normal"; break;
93 }
94 NamedPic *doorPic =
95 GetDoorPic(&gPicManager, m->DoorStyle, doorKey, isHorizontal);
96
97 // set up the door pics
98 for (int i = 0; i < doorGroupCount; i++)
99 {
100 const Vec2i vI = Vec2iAdd(v, Vec2iScale(dv, i));
101 Tile *tile = MapGetTile(map, vI);
102 tile->picAlt = doorPic;
103 tile->pic = GetDoorBasePic(&gPicManager, m->DoorStyle, isHorizontal);
104 tile->flags = DOOR_TILE_FLAGS;
105 if (isHorizontal)
106 {
107 const Vec2i vB = Vec2iAdd(vI, dAside);
108 Tile *tileB = MapGetTile(map, vB);
109 CASSERT(TileCanWalk(MapGetTile(
110 map, Vec2iNew(vI.x - dAside.x, vI.y - dAside.y))),
111 "map gen error: entrance should be clear");
112 CASSERT(TileCanWalk(tileB),
113 "map gen error: entrance should be clear");
114 // Change the tile below to shadow, cast by this door
115 const bool isFloor = IMapGet(map, vB) == MAP_FLOOR;
116 tileB->pic = PicManagerGetMaskedStylePic(
117 &gPicManager,
118 "tile",
119 isFloor ? m->FloorStyle : m->RoomStyle,
120 "shadow",
121 isFloor ? m->FloorMask : m->RoomMask, m->AltMask);
122 }
123 }
124
125 TWatch *w = CreateCloseDoorWatch(
126 map, m, v, isHorizontal, doorGroupCount, doorPic->name);
127 Trigger *t = CreateOpenDoorTrigger(
128 map, m, v, isHorizontal, doorGroupCount, keyFlags);
129 // Connect trigger and watch up
130 Action *a = TriggerAddAction(t);
131 a->Type = ACTION_ACTIVATEWATCH;
132 a->u.index = w->index;
133 a = WatchAddAction(w);
134 a->Type = ACTION_SETTRIGGER;
135 a->u.index = t->id;
136
137 // Set tiles on and besides doors free
138 for (int i = 0; i < doorGroupCount; i++)
139 {
140 const Vec2i vI = Vec2iAdd(v, Vec2iScale(dv, i));
141 IMapSet(map, vI, IMapGet(map, vI) | MAP_LEAVEFREE);
142 const Vec2i vI1 = Vec2iAdd(vI, dAside);
143 IMapSet(map, vI1, IMapGet(map, vI1) | MAP_LEAVEFREE);
144 const Vec2i vI2 = Vec2iMinus(vI, dAside);
145 IMapSet(map, vI2, IMapGet(map, vI2) | MAP_LEAVEFREE);
146 }
147}

149// Count the number of doors that are in the same group as this door
150// Only check to the right/below
151static int GetDoorCountInGroup(
152 const Map *map, const Vec2i v, const bool isHorizontal)
153{
154 const Vec2i dv = Vec2iNew(isHorizontal ? 1 : 0, isHorizontal ? 0 : 1);
155 int count = 0;
156 for (Vec2i vi = v;
157 (IMapGet(map, vi) & MAP_MASKACCESS) == MAP_DOOR;
158 vi = Vec2iAdd(vi, dv))
159 {
160 count++;
161 }
162 return count;
163}
164// 1 second to close doors
165#define CLOSE_DOOR_TICKS FPS_FRAMELIMIT
166// Create the watch responsible for closing the door
167static TWatch *CreateCloseDoorWatch(
168 Map *map, const Mission *m, const Vec2i v,
169 const bool isHorizontal, const int doorGroupCount,
170 const char *picAltName)
171{
172 TWatch *w = WatchNew();
173 const Vec2i dv = Vec2iNew(isHorizontal ? 1 : 0, isHorizontal ? 0 : 1);
174 const Vec2i dAside = Vec2iNew(dv.y, dv.x);
175
176 // The conditions are that the tile above, at and below the doors are empty
177 for (int i = 0; i < doorGroupCount; i++)
178 {
179 const Vec2i vI = Vec2iAdd(v, Vec2iScale(dv, i));
180
181 WatchAddCondition(
182 w, CONDITION_TILECLEAR, CLOSE_DOOR_TICKS, Vec2iMinus(vI, dAside));
183 WatchAddCondition(
184 w, CONDITION_TILECLEAR, CLOSE_DOOR_TICKS, vI);
185 WatchAddCondition(
186 w, CONDITION_TILECLEAR, CLOSE_DOOR_TICKS, Vec2iAdd(vI, dAside));
187 }
188
189 // Now the actions of the watch once it's triggered
190 Action *a;
191
192 // Deactivate itself
193 a = WatchAddAction(w);
194 a->Type = ACTION_DEACTIVATEWATCH;
195 a->u.index = w->index;
196 // play close sound at the center of the door group
197 a = WatchAddAction(w);
198 a->Type = ACTION_SOUND;
199 a->u.pos = Vec2iCenterOfTile(
200 Vec2iAdd(v, Vec2iScale(dv, doorGroupCount / 2)));
201 a->a.Sound = StrSound("door_close");
202
203 // Close doors
204 for (int i = 0; i < doorGroupCount; i++)
205 {
206 const Vec2i vI = Vec2iAdd(v, Vec2iScale(dv, i));
207
208 a = WatchAddAction(w);
209 a->Type = ACTION_EVENT;
210 a->a.Event = GameEventNew(GAME_EVENT_TILE_SET);
211 a->a.Event.u.TileSet.Pos = Vec2i2Net(vI);
212 a->a.Event.u.TileSet.Flags = DOOR_TILE_FLAGS;
213 strcpy(
214 a->a.Event.u.TileSet.PicName,
215 GetDoorBasePic(&gPicManager, m->DoorStyle, isHorizontal)->name);
216 strcpy(a->a.Event.u.TileSet.PicAltName, picAltName);
217 }
218
219 // Add shadows below doors
220 if (isHorizontal)
221 {
222 for (int i = 0; i < doorGroupCount; i++)
223 {
224 const Vec2i vI = Vec2iAdd(v, Vec2iScale(dv, i));
225
226 a = WatchAddAction(w);
227 a->Type = ACTION_EVENT;
228 a->a.Event = GameEventNew(GAME_EVENT_TILE_SET);
229 const Vec2i vI2 = Vec2iNew(vI.x + dAside.x, vI.y + dAside.y);
230 a->a.Event.u.TileSet.Pos = Vec2i2Net(vI2);
231 const bool isFloor = IMapGet(map, vI2) == MAP_FLOOR;
232 strcpy(
233 a->a.Event.u.TileSet.PicName,
234 PicManagerGetMaskedStylePic(
235 &gPicManager,
236 "tile",
237 isFloor ? m->FloorStyle : m->RoomStyle,
238 "shadow",
239 isFloor ? m->FloorMask : m->RoomMask,
240 m->AltMask)->name);
241 }
242 }
243
244 return w;
245}
246static void TileAddTrigger(Tile *t, Trigger *tr);
247static Trigger *CreateOpenDoorTrigger(
248 Map *map, const Mission *m, const Vec2i v,
249 const bool isHorizontal, const int doorGroupCount, const int keyFlags)
250{
251 // All tiles on either side of the door group use the same trigger
252 const Vec2i dv = Vec2iNew(isHorizontal ? 1 : 0, isHorizontal ? 0 : 1);
253 const Vec2i dAside = Vec2iNew(dv.y, dv.x);
254 Trigger *t = MapNewTrigger(map);
255 t->flags = keyFlags;
256
257 // Deactivate itself
258 Action *a;
259 a = TriggerAddAction(t);
260 a->Type = ACTION_CLEARTRIGGER;
261 a->u.index = t->id;
262
263 // Open doors
264 for (int i = 0; i < doorGroupCount; i++)
265 {
266 const Vec2i vI = Vec2iAdd(v, Vec2iScale(dv, i));
267 a = TriggerAddAction(t);
268 a->Type = ACTION_EVENT;
269 a->a.Event = GameEventNew(GAME_EVENT_TILE_SET);
270 a->a.Event.u.TileSet.Pos = Vec2i2Net(vI);
271 a->a.Event.u.TileSet.Flags =
272 (isHorizontal || i > 0) ? 0 : MAPTILE_OFFSET_PIC;
273 strcpy(
274 a->a.Event.u.TileSet.PicName,
275 GetDoorBasePic(&gPicManager, m->DoorStyle, isHorizontal)->name);
276 if (!isHorizontal && i == 0)
277 {
278 // special door cavity picture
279 strcpy(
280 a->a.Event.u.TileSet.PicAltName,
281 GetDoorPic(&gPicManager, m->DoorStyle, "wall", false)->name);
282 }
283 }
284
285 // Change tiles below the doors
286 if (isHorizontal)
287 {
288 for (int i = 0; i < doorGroupCount; i++)
289 {
290 const Vec2i vI = Vec2iAdd(v, Vec2iScale(dv, i));
291 const Vec2i vIAside = Vec2iAdd(vI, dAside);
292 a = TriggerAddAction(t);
293 // Remove shadows below doors
294 a->Type = ACTION_EVENT;
295 a->a.Event = GameEventNew(GAME_EVENT_TILE_SET);
296 const bool isFloor = IMapGet(map, vIAside) == MAP_FLOOR;
297 a->a.Event.u.TileSet.Pos = Vec2i2Net(vIAside);
298 strcpy(
299 a->a.Event.u.TileSet.PicName,
300 PicManagerGetMaskedStylePic(
301 &gPicManager,
302 "tile",
303 isFloor? m->FloorStyle : m->RoomStyle,
304 "normal",
305 isFloor ? m->FloorMask : m->RoomMask,
306 m->AltMask)->name);
307 }
308 }
309
310 // Now place the two triggers on the tiles along either side of the doors
311 for (int i = 0; i < doorGroupCount; i++)
312 {
313 const Vec2i vI = Vec2iAdd(v, Vec2iScale(dv, i));
314 const Vec2i vIA = Vec2iMinus(vI, dAside);
315 TileAddTrigger(MapGetTile(map, vIA), t);
316 const Vec2i vIB = Vec2iAdd(vI, dAside);
317 TileAddTrigger(MapGetTile(map, vIB), t);
318 }
319
320 /// play sound at the center of the door group
321 a = TriggerAddAction(t);
322 a->Type = ACTION_SOUND;
323 a->u.pos = Vec2iCenterOfTile(
324 Vec2iAdd(v, Vec2iScale(dv, doorGroupCount / 2)));
325 a->a.Sound = StrSound("door");
326
327 return t;
328}
329static void TileAddTrigger(Tile *t, Trigger *tr)
330{
331 CArrayPushBack(&t->triggers, &tr);
332}

359NamedPic *GetDoorPic(
360 const PicManager *pm, const char *style, const char *key,
361 const bool isHorizontal)
362{
363 char buf[CDOGS_FILENAME_MAX];
364 // Construct filename
365 // If the key is "wall", it doesn't include orientation
366 sprintf(
367 buf, "door/%s/%s%s", style, key,
368 strcmp(key, "wall") == 0 ? "" : (isHorizontal ? "_h" : "_v"));
369 return PicManagerGetNamedPic(pm, buf);
370}

372#define DOORSTYLE_COUNT 4
373const char *IntDoorStyle(const int i)
374{
375 static const char *doorStyles[] = {
376 "office", "dungeon", "blast", "alien"
377 };
378 // fix bugs with old campaigns
379 return doorStyles[abs(i) % DOORSTYLE_COUNT];
380}
