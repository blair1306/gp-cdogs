69// Only log the file base name
70#ifdef _WIN32
71#define __FILENAME__ (strrchr(__FILE__, '\\') ? strrchr(__FILE__, '\\') + 1 : __FILE__)
72#else
73#define __FILENAME__ (strrchr(__FILE__, '/') ? strrchr(__FILE__, '/') + 1 : __FILE__)
74#endif
75#define LOG(_module, _level, _fmt, ...)\
76 do\
77 {\
78 if (_level >= LogModuleGetLevel(_module))\
79 {\
80 LogLine(\
81 stderr, _module, _level, __FILENAME__, __LINE__,\
82 __FUNCTION__, _fmt, ##__VA_ARGS__);\
83 LogLine(\
84 gLogFile, _module, _level, __FILENAME__, __LINE__,\
85 __FUNCTION__, _fmt, ##__VA_ARGS__);\
86 }\
87 } while ((void)0, 0)
