51PickupClass *StrPickupClass(const char *s)
52{
53 if (s == NULL || strlen(s) == 0)
54 {
55 return NULL;
56 }
57 CA_FOREACH(PickupClass, c, gPickupClasses.CustomClasses)
58 if (strcmp(s, c->Name) == 0)
59 {
60 return c;
61 }
62 CA_FOREACH_END()
63 CA_FOREACH(PickupClass, c, gPickupClasses.Classes)
64 if (strcmp(s, c->Name) == 0)
65 {
66 return c;
67 }
68 CA_FOREACH_END()
69 CA_FOREACH(PickupClass, c, gPickupClasses.KeyClasses)
70 if (strcmp(s, c->Name) == 0)
71 {
72 return c;
73 }
74 CA_FOREACH_END()
75 CASSERT(false, "cannot parse pickup class");
76 return NULL;
77}
78PickupClass *IntPickupClass(const int i)
79{
80 static const char *pickupItems[] = {
81 "folder",
82 "disk1",
83 "disk2",
84 "disk3",
85 "blueprint",
86 "cd",
87 "sack",
88 "holo",
89 "bottle",
90 "radio",
91 "pci_card",
92 "paper"
93 };
94#define PICKUPS_COUNT (sizeof(pickupItems)/sizeof(const char *))
95 if (i < 0 || i >= (int)PICKUPS_COUNT)
96 {
97 return NULL;
98 }
99 return StrPickupClass(pickupItems[i]);
100}

133{

144int StrPickupClassId(const char *s)
145{
146 if (s == NULL || strlen(s) == 0)
147 {
148 return 0;
149 }
150 CA_FOREACH(const PickupClass, c, gPickupClasses.CustomClasses)
151 if (strcmp(s, c->Name) == 0)
152 {
153 return _ca_index + (int)gPickupClasses.Classes.size;
154 }
155 CA_FOREACH_END()
156 CA_FOREACH(const PickupClass, c, gPickupClasses.Classes)
157 if (strcmp(s, c->Name) == 0)
158 {
159 return _ca_index;
160 }
161 CA_FOREACH_END()
162 CASSERT(false, "cannot parse pickup class name");
163 return 0;
164}

168void PickupClassesInit(
169 PickupClasses *classes, const char *filename,
170 const AmmoClasses *ammo, const GunClasses *guns)
171{
172 CArrayInit(&classes->Classes, sizeof(PickupClass));
173 CArrayInit(&classes->CustomClasses, sizeof(PickupClass));
174 CArrayInit(&classes->KeyClasses, sizeof(PickupClass));
175
176 char buf[CDOGS_PATH_MAX];
177 GetDataFilePath(buf, filename);
178 FILE *f = fopen(buf, "r");
179 json_t *root = NULL;
180 if (f == NULL)
181 {
182 LOG(LM_MAIN, LL_ERROR, "Error: cannot load pickups file %s", buf);
183 goto bail;
184 }
185 enum json_error e = json_stream_parse(f, &root);
186 if (e != JSON_OK)
187 {
188 LOG(LM_MAIN, LL_ERROR, "Error parsing pickups file %s", buf);
189 goto bail;
190 }
191 PickupClassesLoadJSON(&classes->Classes, root);
192 PickupClassesLoadAmmo(&classes->Classes, &ammo->Ammo);
193 PickupClassesLoadGuns(&classes->Classes, &guns->Guns);
194 PickupClassesLoadKeys(&classes->KeyClasses);
195
196bail:
197 if (f != NULL)
198 {
199 fclose(f);
200 }
201 json_free_value(&root);
202}
203static bool TryLoadPickupclass(PickupClass *c, json_t *node);
204void PickupClassesLoadJSON(CArray *classes, json_t *root)
205{
206 int version = -1;
207 LoadInt(&version, root, "Version");
208 if (version > VERSION || version <= 0)
209 {
210 CASSERT(false, "cannot read pickups file version");
211 return;
212 }
213
214 json_t *pickupsNode = json_find_first_label(root, "Pickups")->child;
215 for (json_t *child = pickupsNode->child; child; child = child->next)
216 {
217 PickupClass c;
218 if (TryLoadPickupclass(&c, child))
219 {
220 CArrayPushBack(classes, &c);
221 }
222 }
223}
224static bool TryLoadPickupclass(PickupClass *c, json_t *node)
225{
226 memset(c, 0, sizeof *c);
227 char *tmp;
228
229 JSON_UTILS_LOAD_ENUM(c->Type, node, "Type", StrPickupType);
230 switch (c->Type)
231 {
232 case PICKUP_JEWEL:
233 // Set default score
234 c->u.Score = PICKUP_SCORE;
235 LoadInt(&c->u.Score, node, "Score");
236 break;
237 case PICKUP_HEALTH:
238 // Set default heal amount
239 c->u.Health = HEALTH_PICKUP_HEAL_AMOUNT;
240 LoadInt(&c->u.Health, node, "Health");
241 break;
242 case PICKUP_AMMO:
243 tmp = GetString(node, "Ammo");
244 c->u.Ammo.Id = StrAmmoId(tmp);
245 CFREE(tmp);
246 LoadInt(&c->u.Ammo.Amount, node, "AmmoAmount");
247 break;
248 case PICKUP_KEYCARD:
249 // Do nothing; keys now loaded directly from graphics files
250 return false;
251 case PICKUP_GUN:
252 CASSERT(false, "unimplemented");
253 break;
254 default:
255 CASSERT(false, "Unknown pickup type");
256 break;
257 }
258 c->Name = GetString(node, "Name");
259 LoadPic(&c->Pic, node, "Pic");
260 return true;
261}

263void PickupClassesLoadAmmo(CArray *classes, const CArray *ammoClasses)
264{
265 for (int i = 0; i < (int)ammoClasses->size; i++)
266 {
267 const Ammo *a = CArrayGet(ammoClasses, i);
268 PickupClass c;
269 char buf[256];
270 sprintf(buf, "ammo_%s", a->Name);
271 CSTRDUP(c.Name, buf);
272 c.Pic = a->Pic;
273 c.Type = PICKUP_AMMO;
274 c.u.Ammo.Id = StrAmmoId(a->Name);
275 c.u.Ammo.Amount = a->Amount;
276 CArrayPushBack(classes, &c);
277 }
278}

280void PickupClassesLoadGuns(CArray *classes, const CArray *gunClasses)
281{
282 for (int i = 0; i < (int)gunClasses->size; i++)
283 {
284 const GunDescription *g = CArrayGet(gunClasses, i);
285 PickupClass c;
286 char buf[256];
287 sprintf(buf, "gun_%s", g->name);
288 CSTRDUP(c.Name, buf);
289 c.Pic = g->Icon;
290 c.Type = PICKUP_GUN;
291 c.u.GunId = GunDescriptionId(g);
292 CArrayPushBack(classes, &c);
293 }
294}

296void PickupClassesLoadKeys(CArray *classes)
297{
298 CA_FOREACH(const char *, keyStyleName, gPicManager.keyStyleNames)
299 for (int i = 0; i < KEY_COUNT; i++)
300 {
301 PickupClass c;
302 memset(&c, 0, sizeof c);
303 char buf[CDOGS_FILENAME_MAX];
304 sprintf(buf, "keys/%s/%s", *keyStyleName, keyColors[i]);
305 CSTRDUP(c.Name, buf);
306 c.Pic = PicManagerGetPic(&gPicManager, c.Name);
307 c.Type = PICKUP_KEYCARD;
308 c.u.Keys = StrKeycard(keyColors[i]);
309 CArrayPushBack(classes, &c);
310 }
311 CA_FOREACH_END()
312}

331int PickupClassesGetScoreIdx(const PickupClass *p)
332{
333 if (p == NULL)
334 {
335 return 0;
336 }
337 int idx = -1;
338 CA_FOREACH(const PickupClass, c, gPickupClasses.Classes)
339 if (c->Type == PICKUP_JEWEL)
340 {
341 idx++;
342 if (c == p)
343 {
344 return idx;
345 }
346 }
347 CA_FOREACH_END()
348 CA_FOREACH(PickupClass, c, gPickupClasses.CustomClasses)
349 if (c->Type == PICKUP_JEWEL)
350 {
351 idx++;
352 if (c == p)
353 {
354 return idx;
355 }
356 }
357 CA_FOREACH_END()
358 return 0;
359}
