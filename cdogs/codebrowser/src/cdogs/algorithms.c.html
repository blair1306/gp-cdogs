33typedef struct
34{
35 // Whether to use the IsBlocked func to check visibility,
36 // and also whether to early-terminate if the line is blocked
37 bool CheckBlockedAndEarlyTerminate;
38 bool (*IsBlocked)(void *, Vec2i);
39 void (*OnPoint)(void *, Vec2i);
40 void *data;
41} AlgoLineData;
42static bool BresenhamLine(Vec2i from, Vec2i to, AlgoLineData *data)
43{
44 Vec2i d = Vec2iNew(abs(to.x - from.x), abs(to.y - from.y));
45 Vec2i s = Vec2iNew(from.x < to.x ? 1 : -1, from.y < to.y ? 1 : -1);
46 int err = d.x - d.y;
47 Vec2i v = from;
48 for (;;)
49 {
50 int e2 = 2 * err;
51 if (Vec2iEqual(v, to))
52 {
53 break;
54 }
55 if (data->CheckBlockedAndEarlyTerminate)
56 {
57 if (data->IsBlocked(data->data, v))
58 {
59 return false;
60 }
61 }
62 else
63 {
64 data->OnPoint(data->data, v);
65 }
66 if (e2 > -d.y)
67 {
68 err -= d.y;
69 v.x += s.x;
70 }
71 if (Vec2iEqual(v, to))
72 {
73 break;
74 }
75 if (e2 < d.x)
76 {
77 err += d.x;
78 v.y += s.y;
79 }
80 }
81 if (data->CheckBlockedAndEarlyTerminate)
82 {
83 return !data->IsBlocked(data->data, v);
84 }
85 else
86 {
87 data->OnPoint(data->data, v);
88 }
89 return true;
90}

92// From "Raytracing on a grid" by James McNeill
93// http://playtechs.blogspot.com/2007/03/raytracing-on-grid.html
94// "The code snippets are free to use as you see fit and no attribution is
95// necessary"
96static bool JMRaytrace(int x0, int y0, int x1, int y1, AlgoLineData *data)
97{
98 int dx = abs(x1 - x0);
99 int dy = abs(y1 - y0);
100 int x = x0;
101 int y = y0;
102 int n = 1 + dx + dy;
103 int x_inc = (x1 > x0) ? 1 : -1;
104 int y_inc = (y1 > y0) ? 1 : -1;
105 int error = dx - dy;
106 dx *= 2;
107 dy *= 2;
108
109 for (; n > 0; --n)
110 {
111 const Vec2i v = Vec2iNew(x, y);
112 if (data->CheckBlockedAndEarlyTerminate)
113 {
114 if (data->IsBlocked(data->data, v))
115 {
116 return false;
117 }
118 }
119 else
120 {
121 data->OnPoint(data->data, v);
122 }
123
124 if (error > 0)
125 {
126 x += x_inc;
127 error -= dy;
128 }
129 else
130 {
131 y += y_inc;
132 error += dx;
133 }
134 }
135 return true;
136}

138bool HasClearLineBresenham(Vec2i from, Vec2i to, HasClearLineData *data)
139{
140 AlgoLineData bData;
141 bData.CheckBlockedAndEarlyTerminate = true;
142 bData.IsBlocked = data->IsBlocked;
143 bData.data = data->data;
144 return BresenhamLine(from, to, &bData);
145}
146bool HasClearLineJMRaytrace(
147 const Vec2i from, const Vec2i to, HasClearLineData *data)
148{
149 AlgoLineData bData;
150 bData.CheckBlockedAndEarlyTerminate = true;
151 bData.IsBlocked = data->IsBlocked;
152 bData.data = data->data;
153 return JMRaytrace(from.x, from.y, to.x, to.y, &bData);
154}

157void BresenhamLineDraw(Vec2i from, Vec2i to, AlgoLineDrawData *data)
158{
159 AlgoLineData bData;
160 bData.CheckBlockedAndEarlyTerminate = false;
161 bData.OnPoint = data->Draw;
162 bData.data = data->data;
163 BresenhamLine(from, to, &bData);
164}
165void JMRaytraceLineDraw(
166 const Vec2i from, const Vec2i to, AlgoLineDrawData *data)
167{
168 AlgoLineData bData;
169 bData.CheckBlockedAndEarlyTerminate = false;
170 bData.OnPoint = data->Draw;
171 bData.data = data->data;
172 JMRaytrace(from.x, from.y, to.x, to.y, &bData);
173}

175bool CFloodFill(Vec2i v, FloodFillData *data)
176{
177 if (data->IsSame(data->data, v))
178 {
179 data->Fill(data->data, v);
180 CFloodFill(Vec2iNew(v.x - 1, v.y), data);
181 CFloodFill(Vec2iNew(v.x + 1, v.y), data);
182 CFloodFill(Vec2iNew(v.x, v.y - 1), data);
183 CFloodFill(Vec2iNew(v.x, v.y + 1), data);
184 return true;
185 }
186 return false;
187}
