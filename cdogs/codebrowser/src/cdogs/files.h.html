57/*
58C-Dogs (classic) file format:
59Sizes in bytes unless otherwise stated
60
61Priviledged types (types that are used directly in the file format)
62Changing these types will break the game!
63- CampaignSettingOld
64- struct MissionOld
65- TBadGuy
66- struct MissionObjectiveOld
67
68Campaign:
69- CAMPAIGN_MAGIC (4)
70- CAMPAIGN_VERSION (4)
71- Title (40, char *)
72- Author (40, char *)
73- Description (200, char *)
74- MissionCount (4)
75- <Missions> (MissionCount * sizeof(struct MissionOld))
76- CharacterCount (4)
77- <Characters> (CharacterCount * sizeof(TBadGuy))
78*/
79#ifdef _MSC_VER
80#pragma pack(push, 1)
81#endif
82typedef struct
83{
84 int32_t armedBodyPic;
85 int32_t unarmedBodyPic;
86 int32_t facePic;
87 int32_t speed;
88 int32_t probabilityToMove;
89 int32_t probabilityToTrack;
90 int32_t probabilityToShoot;
91 int32_t actionDelay;
92 int32_t gun;
93 int32_t skinColor;
94 int32_t armColor;
95 int32_t bodyColor;
96 int32_t legColor;
97 int32_t hairColor;
98 int32_t health;
99 int32_t flags;
100}
101#ifndef _MSC_VER
102__attribute__((packed))
103#endif
104TBadGuy;
105#ifdef _MSC_VER
106#pragma pack(pop)
107#endif

109// WARNING: data type used in C format (builtin campaigns)

114 char description[200];
115 int missionCount;
