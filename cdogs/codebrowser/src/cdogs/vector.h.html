32typedef struct
33{
34 int x;
35 int y;
36} Vec2i;
37
38Vec2i Vec2iNew(int x, int y);
39Vec2i Vec2iZero(void);
40Vec2i Vec2iUnit(void); // (1, 1)
41// r radius, theta radians; result will be rounded
42// Note: 0 radians is north, continuing clockwise
43Vec2i Vec2iFromPolar(const double r, const double th);
44Vec2i Vec2iAdd(Vec2i a, Vec2i b);
45Vec2i Vec2iMinus(Vec2i a, Vec2i b);
46// Multiply the components of two Vec2is together
47Vec2i Vec2iMult(const Vec2i a, const Vec2i b);
48Vec2i Vec2iScale(Vec2i v, int scalar);
49Vec2i Vec2iScaleD(const Vec2i v, const double scalar);
50Vec2i Vec2iScaleDiv(Vec2i v, int scaleDiv);
51// TODO: due to rounding, this will always return unit component vectors
52Vec2i Vec2iNorm(Vec2i v);
53bool Vec2iEqual(const Vec2i a, const Vec2i b);
54bool Vec2iIsZero(const Vec2i v);
55Vec2i Vec2iMin(Vec2i a, Vec2i b); // Get min x and y of both vectors
56Vec2i Vec2iMax(Vec2i a, Vec2i b); // Get max x and y of both vectors
57Vec2i Vec2iClamp(Vec2i v, Vec2i lo, Vec2i hi);

59// Convert to and from real (i.e. integral) coordinates and full (fractional)

64Vec2i Vec2iToTile(Vec2i v);

68int DistanceSquared(const Vec2i a, const Vec2i b);
