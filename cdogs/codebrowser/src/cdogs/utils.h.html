83#define CASSERT(_x, _errmsg)\
84{\
85 volatile bool isOk = _x;\
86 if (!isOk)\
87 {\
88 static char _buf[1024];\
89 sprintf(\
90 _buf,\
91 "In %s %d:%s: " _errmsg " (" #_x ")",\
92 __FILE__, __LINE__, __func__);\
93 CHALT();\
94 assert(_x);\
95 }\
96}

98#define _CCHECKALLOC(_func, _var, _size)\
99{\
100 if (_var == NULL && _size > 0)\
101 {\
102 debug(D_MAX,\
103 _func "(" #_var " size %d) failed\n",\
104 (int)_size);\
105 exit(1);\
106 }\
107 debug(D_MAX,\
108 _func "(" #_var " size %d) at 0x%p\n",\
109 (int)_size, _var);\
110}
111
112#define CMALLOC(_var, _size)\
113{\
114 _var = malloc(_size);\
115 _CCHECKALLOC("CMALLOC", _var, (_size))\
116}
117#define CCALLOC(_var, _size)\
118{\
119 _var = calloc(1, _size);\
120 _CCHECKALLOC("CCALLOC", _var, (_size))\
121}
122#define CREALLOC(_var, _size)\
123{\
124 _var = realloc(_var, _size);\
125 _CCHECKALLOC("CREALLOC", _var, (_size))\
126}
127#define CSTRDUP(_var, _str)\
128{\
129 CMALLOC(_var, strlen(_str) + 1);\
130 strcpy(_var, _str);\
131}
132
133#define CFREE(_var)\
134{\
135 debug(D_MAX,\
136 "CFREE(" #_var ") at 0x%p\n",\
137 _var);\
138 free(_var);\
139}
