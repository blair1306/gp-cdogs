83static void DisplayPlayer(const TActor *player, Vec2i pos, const int scale)
84{
85 Vec2i playerPos = Vec2iToTile(
86 Vec2iNew(player->tileItem.x, player->tileItem.y));
87 pos = Vec2iAdd(pos, Vec2iScale(playerPos, scale));
88 if (scale >= 2)
89 {
90 DrawHead(ActorGetCharacter(player), DIRECTION_DOWN, pos);
91 }
92 else
93 {
94 Draw_Point(pos.x, pos.y, colorWhite);
95 }
96}

98static void DisplayObjective(
99 TTileItem *t, int objectiveIndex, Vec2i pos, int scale, int flags)
100{
101 Vec2i objectivePos = Vec2iNew(t->x / TILE_WIDTH, t->y / TILE_HEIGHT);
102 const Objective *o =
103 CArrayGet(&gMission.missionData->Objectives, objectiveIndex);
104 color_t color = o->color;
105 pos = Vec2iAdd(pos, Vec2iScale(objectivePos, scale));
106 if (flags & AUTOMAP_FLAGS_MASK)
107 {
108 color.a = MASK_ALPHA;
109 }
110 if (scale >= 2)
111 {
112 DrawCross(&gGraphicsDevice, pos.x, pos.y, color);
113 }
114 else
115 {
116 Draw_Point(pos.x, pos.y, color);
117 }
118}

120static void DisplayExit(Vec2i pos, int scale, int flags)
121{
122 Vec2i exitPos = gMap.ExitStart;
123 Vec2i exitSize = Vec2iAdd(Vec2iMinus(gMap.ExitEnd, exitPos), Vec2iUnit());
124 color_t color = colorExit;
125
126 if (!HasExit(gCampaign.Entry.Mode))
127 {
128 return;
129 }
130
131 exitPos = Vec2iScale(exitPos, scale);
132 exitSize = Vec2iScale(exitSize, scale);
133 exitPos = Vec2iAdd(exitPos, pos);
134
135 if (flags & AUTOMAP_FLAGS_MASK)
136 {
137 color.a = MASK_ALPHA;
138 }
139 DrawRectangle(&gGraphicsDevice, exitPos, exitSize, color, DRAW_FLAG_LINE);
140}

142static void DisplaySummary(void)
143{
144 char sScore[20];
145 Vec2i pos;
146 pos.y = gGraphicsDevice.cachedConfig.Res.y - 5 - FontH();
147
148 CA_FOREACH(const Objective, o, gMission.missionData->Objectives)
149 if (ObjectiveIsRequired(o)|| o->done > 0)
150 {
151 color_t textColor = colorWhite;
152 pos.x = 5;
153 // Objective color dot
154 Draw_Rect(pos.x, (pos.y + 3), 2, 2, o->color);
155
156 pos.x += 5;
157
158 sprintf(sScore, "(%d)", o->done);
159
160 if (!ObjectiveIsRequired(o))
161 {
162 textColor = colorPurple;
163 }
164 else if (ObjectiveIsComplete(o))
165 {
166 textColor = colorRed;
167 }
168 pos = FontStrMask(o->Description, pos, textColor);
169 pos.x += 5;
170 FontStrMask(sScore, pos, textColor);
171 pos.y -= (FontH() + 1);
172 }
173 CA_FOREACH_END()
174}

176color_t DoorColor(int x, int y)
177{
178 int l = MapGetDoorKeycardFlag(&gMap, Vec2iNew(x, y));
179
180 switch (l) {
181 case FLAGS_KEYCARD_YELLOW:
182 return colorYellowDoor;
183 case FLAGS_KEYCARD_GREEN:
184 return colorGreenDoor;
185 case FLAGS_KEYCARD_BLUE:
186 return colorBlueDoor;
187 case FLAGS_KEYCARD_RED:
188 return colorRedDoor;
189 default:
190 return colorDoor;
191 }
192}

201static void DrawMap(
202 Map *map,
203 Vec2i center, Vec2i centerOn, Vec2i size,
204 int scale, int flags)
205{
206 int x, y;
207 Vec2i mapPos = Vec2iAdd(center, Vec2iScale(centerOn, -scale));
208 for (y = 0; y < gMap.Size.y; y++)
209 {
210 int i;
211 for (i = 0; i < scale; i++)
212 {
213 for (x = 0; x < gMap.Size.x; x++)
214 {
215 Tile *tile = MapGetTile(map, Vec2iNew(x, y));
216 if (!(tile->flags & MAPTILE_IS_NOTHING) &&
217 (tile->isVisited || (flags & AUTOMAP_FLAGS_SHOWALL)))
218 {
219 int j;
220 for (j = 0; j < scale; j++)
221 {
222 Vec2i drawPos = Vec2iNew(
223 mapPos.x + x*scale + j,
224 mapPos.y + y*scale + i);
225 color_t color = colorRoom;
226 if (tile->flags & MAPTILE_IS_WALL)
227 {
228 color = colorWall;
229 }
230 else if (tile->flags & MAPTILE_NO_WALK)
231 {
232 color = DoorColor(x, y);
233 }
234 else if (tile->flags & MAPTILE_IS_NORMAL_FLOOR)
235 {
236 color = colorFloor;
237 }
238 if (!ColorEquals(color, colorBlack))
239 {
240 if (flags & AUTOMAP_FLAGS_MASK)
241 {
242 color.a = MASK_ALPHA;
243 }
244 Draw_Point(drawPos.x, drawPos.y, color);
245 }
246 }
247 }
248 }
249 }
250 }
251 if (flags & AUTOMAP_FLAGS_MASK)
252 {
253 color_t color = { 255, 255, 255, 128 };
254 Draw_Rect(
255 center.x - size.x / 2,
256 center.y - size.y / 2,
257 size.x, size.y,
258 color);
259 }
260}

262static void DrawTileItem(
263 TTileItem *t, Tile *tile, Vec2i pos, int scale, int flags);
264static void DrawObjectivesAndKeys(Map *map, Vec2i pos, int scale, int flags)
265{
266 for (int y = 0; y < map->Size.y; y++)
267 {
268 for (int x = 0; x < map->Size.x; x++)
269 {
270 Tile *tile = MapGetTile(map, Vec2iNew(x, y));
271 CA_FOREACH(ThingId, tid, tile->things)
272 DrawTileItem(
273 ThingIdGetTileItem(tid), tile, pos, scale, flags);
274 CA_FOREACH_END()
275 }
276 }

278static void DrawTileItem(
279 TTileItem *t, Tile *tile, Vec2i pos, int scale, int flags)
280{
281 if ((t->flags & TILEITEM_OBJECTIVE) != 0)
282 {
283 const int obj = ObjectiveFromTileItem(t->flags);
284 const Objective *o =
285 CArrayGet(&gMission.missionData->Objectives, obj);
286 if (!(o->Flags & OBJECTIVE_HIDDEN) || (flags & AUTOMAP_FLAGS_SHOWALL))
287 {
288 if ((o->Flags & OBJECTIVE_POSKNOWN) ||
289 tile->isVisited ||
290 (flags & AUTOMAP_FLAGS_SHOWALL))
291 {
292 DisplayObjective(t, obj, pos, scale, flags);
293 }
294 }
295 }
296 else if (t->kind == KIND_PICKUP && tile->isVisited)
297 {
298 const Pickup *p = CArrayGet(&gPickups, t->id);
299 if (p->class->Type == PICKUP_KEYCARD)
300 {
301 color_t dotColor = colorBlack;
302 switch (((const Pickup *)CArrayGet(&gPickups, t->id))->class->u.Keys)
303 {
304 case FLAGS_KEYCARD_RED:
305 dotColor = colorRedDoor;
306 break;
307 case FLAGS_KEYCARD_BLUE:
308 dotColor = colorBlueDoor;
309 break;
310 case FLAGS_KEYCARD_GREEN:
311 dotColor = colorGreenDoor;
312 break;
313 case FLAGS_KEYCARD_YELLOW:
314 dotColor = colorYellowDoor;
315 break;
316 default:
317 CASSERT(false, "Unknown key color");
318 break;
319 }
320 DrawDot(t, dotColor, pos, scale);
321 }
322 }

325void AutomapDraw(int flags, bool showExit)
326{
327 color_t mask = { 0, 128, 0, 255 };
328 Vec2i mapCenter = Vec2iNew(
329 gGraphicsDevice.cachedConfig.Res.x / 2,
330 gGraphicsDevice.cachedConfig.Res.y / 2);
331 Vec2i centerOn = Vec2iNew(gMap.Size.x / 2, gMap.Size.y / 2);
332 Vec2i pos = Vec2iAdd(mapCenter, Vec2iScale(centerOn, -MAP_FACTOR));
333
334 // Draw faded green overlay
335 for (int y = 0; y < gGraphicsDevice.cachedConfig.Res.y; y++)
336 {
337 for (int x = 0; x < gGraphicsDevice.cachedConfig.Res.x; x++)
338 {
339 DrawPointMask(&gGraphicsDevice, Vec2iNew(x, y), mask);
340 }
341 }
342
343 DrawMap(&gMap, mapCenter, centerOn, gMap.Size, MAP_FACTOR, flags);
344 DrawObjectivesAndKeys(&gMap, pos, MAP_FACTOR, flags);
345
346 CA_FOREACH(const PlayerData, p, gPlayerDatas)
347 if (!IsPlayerAlive(p))
348 {
349 continue;
350 }
351 DisplayPlayer(ActorGetByUID(p->ActorUID), pos, MAP_FACTOR);
352 CA_FOREACH_END()
353
354 if (showExit)
355 {
356 DisplayExit(pos, MAP_FACTOR, flags);
357 }
358 DisplaySummary();
359}

361void AutomapDrawRegion(
362 Map *map,
363 Vec2i pos, Vec2i size, Vec2i mapCenter, int flags, bool showExit)
364{
365 const int scale = 1;
366 const BlitClipping oldClip = gGraphicsDevice.clipping;
367 GraphicsSetBlitClip(
368 &gGraphicsDevice,
369 pos.x, pos.y, pos.x + size.x - 1, pos.y + size.y - 1);
370 pos = Vec2iAdd(pos, Vec2iScaleDiv(size, 2));
371 DrawMap(map, pos, mapCenter, size, scale, flags);
372 const Vec2i centerOn = Vec2iAdd(pos, Vec2iScale(mapCenter, -scale));
373 CA_FOREACH(const PlayerData, p, gPlayerDatas)
374 if (!IsPlayerAlive(p))
375 {
376 continue;
377 }
378 const TActor *player = ActorGetByUID(p->ActorUID);
379 DisplayPlayer(player, centerOn, scale);
380 CA_FOREACH_END()
381 DrawObjectivesAndKeys(&gMap, centerOn, scale, flags);
382 if (showExit)
383 {
384 DisplayExit(centerOn, scale, flags);
385 }
386 GraphicsSetBlitClip(
387 &gGraphicsDevice,
388 oldClip.left, oldClip.top, oldClip.right, oldClip.bottom);
389}
