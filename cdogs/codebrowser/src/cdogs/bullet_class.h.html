59struct MobileObject;
60typedef bool (*BulletUpdateFunc)(struct MobileObject *, int);
61typedef struct
62{
63 char *Name;
64 CPic CPic;
65 Vec2i ShadowSize;
66 int Delay; // number of frames before moving
67 int SpeedLow;
68 int SpeedHigh;
69 bool SpeedScale; // whether to scale X/Y speed based on perspective
70 int Friction; // Amount to subtract from velocity per tick
71 // -1 is infinite range
72 int RangeLow;
73 int RangeHigh;
74 int Power;
75 double Mass;
76 Vec2i Size;
77 special_damage_e Special;
78 bool HurtAlways;
79 bool Persists; // remains even after hitting walls/items
80 const ParticleClass *Spark;
81 const ParticleClass *OutOfRangeSpark;
82 HitSounds HitSound;
83 bool WallBounces;
84 bool HitsObjects;
85 struct
86 {
87 int GravityFactor; // 0 for non-falling bullets
88 bool FallsDown; // if false, DZ is never negative
89 bool DestroyOnDrop;
90 bool Bounces;
91 CArray DropGuns; // of const GunDescription *
92 } Falling;
93 int SeekFactor; // -1 to disable; higher = less seeking
94 bool Erratic;
95
96 // Special weapons to fire if certain events occur
97 CArray OutOfRangeGuns; // of const GunDescription *
98 CArray HitGuns; // of const GunDescription *
99 CArray ProximityGuns; // of const GunDescription *
100
101 // Temporary JSON object for two-step bullet loading
102 json_t *node;
103} BulletClass;
104typedef struct
105{
106 CArray Classes; // of BulletClass
107 BulletClass Default;
108 CArray CustomClasses; // of BulletClass
109 json_t *root;
110} BulletClasses;
111extern BulletClasses gBulletClasses;
112
113BulletClass *StrBulletClass(const char *s);
114
115void BulletInitialize(BulletClasses *bullets);
116void BulletLoadJSON(
117 BulletClasses *bullets, CArray *classes, json_t *bulletNode);
118// 2-step initialisation since bullet and weapon reference each other
119void BulletLoadWeapons(BulletClasses *bullets);
120void BulletClassesClear(CArray *classes);
121void BulletTerminate(BulletClasses *bullets);
122
123void BulletAdd(const NAddBullet add);
124
125bool UpdateBullet(struct MobileObject *obj, const int ticks);
