105static void CheckPickups(TActor *actor);
106void UpdateActorState(TActor * actor, int ticks)
107{
108 Weapon *gun = ActorGetGun(actor);
109 WeaponUpdate(gun, ticks);
110 ActorFireUpdate(gun, actor, ticks);
111
112 // If we're ready to pick up, always check the pickups
113 if (actor->PickupAll && !gCampaign.IsClient)
114 {
115 CheckPickups(actor);
116 }
117
118 if (actor->health > 0)
119 {
120 actor->flamed = MAX(0, actor->flamed - ticks);
121 if (actor->poisoned)
122 {
123 if ((actor->poisoned & 7) == 0)
124 {
125 InjureActor(actor, 1);
126 }
127 actor->poisoned = MAX(0, actor->poisoned - ticks);
128 }
129 actor->petrified = MAX(0, actor->petrified - ticks);
130 actor->confused = MAX(0, actor->confused - ticks);
131 }
132
133 actor->slideLock = MAX(0, actor->slideLock - ticks);
134
135 TileItemUpdate(&actor->tileItem, ticks);
136
137 actor->stateCounter = MAX(0, actor->stateCounter - ticks);
138 if (actor->stateCounter > 0)
139 {
140 return;
141 }
142
143 if (actor->health <= 0) {
144 actor->dead++;
145 actor->MoveVel = Vec2iZero();
146 actor->stateCounter = 4;
147 actor->tileItem.flags = 0;
148 return;
149 }
150
151 // Draw rotation interpolation
152 const float targetRadians = (float)dir2radians[actor->direction];
153 if (actor->DrawRadians - targetRadians > PI)
154 {
155 actor->DrawRadians -= 2 * (float)PI;
156 }
157 if (actor->DrawRadians - targetRadians < -PI)
158 {
159 actor->DrawRadians += 2 * (float)PI;
160 }
161 const float dr = actor->DrawRadians - targetRadians;
162 if (dr < 0)
163 {
164 actor->DrawRadians += (float)MIN(DRAW_RADIAN_SPEED*ticks, -dr);
165 }
166 else if (dr > 0)
167 {
168 actor->DrawRadians -= (float)MIN(DRAW_RADIAN_SPEED*ticks, dr);
169 }
170
171 // Footstep sounds
172 // Step on 2 and 6
173 // TODO: custom animation and footstep frames
174 if (ConfigGetBool(&gConfig, "Sound.Footsteps") &&
175 (AnimationGetFrame(&actor->anim) == 2 ||
176 AnimationGetFrame(&actor->anim) == 6) &&
177 actor->anim.newFrame)
178 {
179 SoundPlayAtPlusDistance(
180 &gSoundDevice, StrSound("footsteps"),
181 Vec2iNew(actor->tileItem.x, actor->tileItem.y),
182 FOOTSTEP_DISTANCE_PLUS);
183 }
184
185 // Animation
186 AnimationUpdate(&actor->anim, ticks);
187
188 // Chatting
189 actor->ChatterCounter = MAX(0, actor->ChatterCounter - ticks);
190 if (actor->ChatterCounter == 0)
191 {
192 // Stop chatting
193 strcpy(actor->Chatter, "");
194 }
195}

197static Vec2i GetConstrainedFullPos(
198 const Map *map, const Vec2i fromFull, const Vec2i toFull,
199 const Vec2i size);
200static void OnMove(TActor *a);
201bool TryMoveActor(TActor *actor, Vec2i pos)
202{
203 CASSERT(!Vec2iEqual(actor->Pos, pos), "trying to move to same position");
204
205 actor->hasCollided = true;
206 actor->CanPickupSpecial = false;
207
208 const Vec2i oldPos = actor->Pos;
209 pos = GetConstrainedFullPos(&gMap, actor->Pos, pos, actor->tileItem.size);
210 if (Vec2iEqual(oldPos, pos))
211 {
212 return false;
213 }
214
215 // Check for object collisions
216 // Only do this if we are the owner of the actor, since this may lead to
217 // melee damage
218 if ((!gCampaign.IsClient && actor->PlayerUID < 0) ||
219 ActorIsLocalPlayer(actor->uid))
220 {
221 const CollisionParams params =
222 {
223 TILEITEM_IMPASSABLE, CalcCollisionTeam(true, actor),
224 IsPVP(gCampaign.Entry.Mode)
225 };
226 TTileItem *target = OverlapGetFirstItem(
227 &actor->tileItem, pos, actor->tileItem.size, params);
228 if (target)
229 {
230 Weapon *gun = ActorGetGun(actor);
231 const TObject *object = target->kind == KIND_OBJECT ?
232 CArrayGet(&gObjs, target->id) : NULL;
233 if (ActorCanFire(actor) && !gun->Gun->CanShoot &&
234 actor->health > 0 &&
235 (!object || !ObjIsDangerous(object)))
236 {
237 if (CanHit(actor->flags, actor->uid, target))
238 {
239 // Tell the server that we want to melee something
240 GameEvent e = GameEventNew(GAME_EVENT_ACTOR_MELEE);
241 e.u.Melee.UID = actor->uid;
242 strcpy(e.u.Melee.BulletClass, gun->Gun->Bullet->Name);
243 e.u.Melee.TargetKind = target->kind;
244 switch (target->kind)
245 {
246 case KIND_CHARACTER:
247 e.u.Melee.TargetUID =
248 ((const TActor *)CArrayGet(&gActors, target->id))->uid;
249 e.u.Melee.HitType = HIT_FLESH;
250 break;
251 case KIND_OBJECT:
252 e.u.Melee.TargetUID =
253 ((const TObject *)CArrayGet(&gObjs, target->id))->uid;
254 e.u.Melee.HitType = HIT_OBJECT;
255 break;
256 default:
257 CASSERT(false, "cannot damage target kind");
258 break;
259 }
260 gun->lock = gun->Gun->Lock;
261 if (gun->soundLock <= 0)
262 {
263 gun->soundLock += gun->Gun->SoundLockLength;
264 }
265 else
266 {
267 e.u.Melee.HitType = (int)HIT_NONE;
268 }
269 GameEventsEnqueue(&gGameEvents, e);
270 }
271 return false;
272 }
273
274 const Vec2i yPos = Vec2iNew(actor->Pos.x, pos.y);
275 if (OverlapGetFirstItem(
276 &actor->tileItem, yPos, actor->tileItem.size, params))
277 {
278 pos.y = actor->Pos.y;
279 }
280 const Vec2i xPos = Vec2iNew(pos.x, actor->Pos.y);
281 if (OverlapGetFirstItem(
282 &actor->tileItem, xPos, actor->tileItem.size, params))
283 {
284 pos.x = actor->Pos.x;
285 }
286 if (pos.x != actor->Pos.x && pos.y != actor->Pos.y)
287 {
288 // Both x-only or y-only movement are viable,
289 // i.e. we are colliding corner vs corner
290 // Arbitrarily choose x-only movement
291 pos.y = actor->Pos.y;
292 }
293 if ((pos.x == actor->Pos.x && pos.y == actor->Pos.y) ||
294 IsCollisionWithWall(Vec2iFull2Real(pos), actor->tileItem.size))
295 {
296 return false;
297 }
298 }
299 }
300
301 actor->Pos = pos;
302 OnMove(actor);
303
304 actor->hasCollided = false;
305 return true;
306}
307// Get a movement position that is constrained by collisions
308// May return a position that is the same as the 'from', that is, we cannot
309// move in the direction specified.
310// Note: must use full coordinates to do collisions, despite collisions using
311// real coordinates, because fractional movement will be blocked otherwise
312// since real coordinates are the same.
313static Vec2i GetConstrainedFullPos(
314 const Map *map, const Vec2i fromFull, const Vec2i toFull,
315 const Vec2i size)
316{
317 // Check collision with wall
318 if (!IsCollisionWithWall(Vec2iFull2Real(toFull), size))
319 {
320 // Not in collision; just return where we wanted to go
321 return toFull;
322 }
323
324 CASSERT(size.x >= size.y, "tall collision not supported");
325 const Vec2i dv = Vec2iMinus(toFull, fromFull);
326
327 // If moving diagonally, use rectangular bounds and
328 // try to move in only x or y directions
329 if (dv.x != 0 && dv.y != 0)
330 {
331 // X-only movement
332 const Vec2i xVec = Vec2iNew(toFull.x, fromFull.y);
333 if (!IsCollisionWithWall(Vec2iFull2Real(xVec), size))
334 {
335 return xVec;
336 }
337 // Y-only movement
338 const Vec2i yVec = Vec2iNew(fromFull.x, toFull.y);
339 if (!IsCollisionWithWall(Vec2iFull2Real(yVec), size))
340 {
341 return yVec;
342 }
343 // If we're still stuck, we're possibly stuck on a corner which is not
344 // in collision with a diamond but is colliding with the box.
345 // If so try x- or y-only movement, but with the benefit of diamond
346 // slipping.
347 const Vec2i xPos = GetConstrainedFullPos(map, fromFull, xVec, size);
348 if (!Vec2iEqual(xPos, fromFull))
349 {
350 return xPos;
351 }
352 const Vec2i yPos = GetConstrainedFullPos(map, fromFull, yVec, size);
353 if (!Vec2iEqual(yPos, fromFull))
354 {
355 return yPos;
356 }
357 }
358
359 // Now check diagonal movement, if we were moving in an x- or y-
360 // only direction
361 // Note: we're moving at extra speed because dx/dy are only magnitude 1;
362 // if we divide then we get 0 which ruins the logic
363 if (dv.x == 0)
364 {
365 // Moving up or down; try moving to the left or right diagonally
366 // Scale X movement because our diamond is wider than tall, so we
367 // may need to scale the diamond wider.
368 const int xScale =
369 size.x > size.y ? (int)ceil((double)size.x / size.y) : 1;
370 const Vec2i diag1Vec =
371 Vec2iAdd(fromFull, Vec2iNew(-dv.y * xScale, dv.y));
372 if (!IsCollisionDiamond(map, Vec2iFull2Real(diag1Vec), size))
373 {
374 return diag1Vec;
375 }
376 const Vec2i diag2Vec =
377 Vec2iAdd(fromFull, Vec2iNew(dv.y * xScale, dv.y));
378 if (!IsCollisionDiamond(map, Vec2iFull2Real(diag2Vec), size))
379 {
380 return diag2Vec;
381 }
382 }
383 else if (dv.y == 0)
384 {
385 // Moving left or right; try moving up or down diagonally
386 const Vec2i diag1Vec =
387 Vec2iAdd(fromFull, Vec2iNew(dv.x, -dv.x));
388 if (!IsCollisionDiamond(map, Vec2iFull2Real(diag1Vec), size))
389 {
390 return diag1Vec;
391 }
392 const Vec2i diag2Vec =
393 Vec2iAdd(fromFull, Vec2iNew(dv.x, dv.x));
394 if (!IsCollisionDiamond(map, Vec2iFull2Real(diag2Vec), size))
395 {
396 return diag2Vec;
397 }
398 }
399
400 // All alternative movements are in collision; don't move
401 return fromFull;
402}

404void ActorMove(const NActorMove am)
405{
406 TActor *a = ActorGetByUID(am.UID);
407 if (a == NULL || !a->isInUse) return;
408 a->Pos = Net2Vec2i(am.Pos);
409 a->MoveVel = Net2Vec2i(am.MoveVel);
410 OnMove(a);
411}
412static void CheckTrigger(const Vec2i tilePos, const bool showLocked);
413static void CheckRescue(const TActor *a);
414static void OnMove(TActor *a)
415{
416 const Vec2i realPos = Vec2iFull2Real(a->Pos);
417 MapTryMoveTileItem(&gMap, &a->tileItem, realPos);
418 if (MapIsTileInExit(&gMap, &a->tileItem))
419 {
420 a->action = ACTORACTION_EXITING;
421 }
422 else
423 {
424 a->action = ACTORACTION_MOVING;
425 }
426
427 if (!gCampaign.IsClient)
428 {
429 CheckTrigger(Vec2iToTile(realPos), ActorIsLocalPlayer(a->uid));
430
431 CheckPickups(a);
432
433 CheckRescue(a);
434 }
435}
436static void CheckTrigger(const Vec2i tilePos, const bool showLocked)
437{
438 const Tile *t = MapGetTile(&gMap, tilePos);
439 CA_FOREACH(Trigger *, tp, t->triggers)
440 if (!TriggerTryActivate(*tp, gMission.KeyFlags, tilePos) &&
441 (*tp)->isActive &&
442 TriggerCannotActivate(*tp) &&
443 showLocked)
444 {
445 TriggerSetCannotActivate(*tp);
446 GameEvent s = GameEventNew(GAME_EVENT_ADD_PARTICLE);
447 s.u.AddParticle.Class =
448 StrParticleClass(&gParticleClasses, "locked_text");
449 s.u.AddParticle.FullPos =
450 Vec2iReal2Full(Vec2iCenterOfTile(tilePos));
451 s.u.AddParticle.Z = (BULLET_Z * 2) * Z_FACTOR;
452 sprintf(s.u.AddParticle.Text, "locked");
453 GameEventsEnqueue(&gGameEvents, s);
454 }
455 CA_FOREACH_END()
456}
457// Check if the player can pickup any item
458static bool CheckPickupFunc(
459 TTileItem *ti, void *data, const Vec2i colA, const Vec2i colB,
460 const Vec2i normal);
461static void CheckPickups(TActor *actor)
462{
463 // NPCs can't pickup
464 if (actor->PlayerUID < 0)
465 {
466 return;
467 }
468 const CollisionParams params =
469 {
470 0, CalcCollisionTeam(true, actor), IsPVP(gCampaign.Entry.Mode)
471 };
472 OverlapTileItems(
473 &actor->tileItem, actor->Pos, actor->tileItem.size,
474 params, CheckPickupFunc, actor, NULL, NULL, NULL);
475}
476static bool CheckPickupFunc(
477 TTileItem *ti, void *data, const Vec2i colA, const Vec2i colB,
478 const Vec2i normal)
479{
480 UNUSED(colA);
481 UNUSED(colB);
482 UNUSED(normal);
483 // Always return true, as we can pickup multiple items in one go
484 if (ti->kind != KIND_PICKUP) return true;
485 TActor *a = data;
486 PickupPickup(a, CArrayGet(&gPickups, ti->id), a->PickupAll);
487 return true;
488}
489static void CheckRescue(const TActor *a)
490{
491 // NPCs can't rescue
492 if (a->PlayerUID < 0) return;
493
494 // Check an area slightly bigger than the actor's size for rescue
495 // objectives
496#define RESCUE_CHECK_PAD 2
497 const CollisionParams params =
498 {
499 TILEITEM_IMPASSABLE, CalcCollisionTeam(true, a),
500 IsPVP(gCampaign.Entry.Mode)
501 };
502 const TTileItem *target = OverlapGetFirstItem(
503 &a->tileItem, a->Pos,
504 Vec2iAdd(a->tileItem.size, Vec2iNew(RESCUE_CHECK_PAD, RESCUE_CHECK_PAD)),
505 params);
506 if (target != NULL && target->kind == KIND_CHARACTER)
507 {
508 TActor *other = CArrayGet(&gActors, target->id);
509 CASSERT(other->isInUse, "Cannot find nonexistent player");
510 if (other->flags & FLAGS_PRISONER)
511 {
512 other->flags &= ~FLAGS_PRISONER;
513 GameEvent e = GameEventNew(GAME_EVENT_RESCUE_CHARACTER);
514 e.u.Rescue.UID = other->uid;
515 GameEventsEnqueue(&gGameEvents, e);
516 UpdateMissionObjective(
517 &gMission, other->tileItem.flags, OBJECTIVE_RESCUE, 1);
518 }
519 }
520}

528void InjureActor(TActor * actor, int injury)
529{
530 const int lastHealth = actor->health;
531 actor->health -= injury;
532 if (lastHealth > 0 && actor->health <= 0)
533 {
534 actor->stateCounter = 0;
535 const Vec2i pos = Vec2iNew(actor->tileItem.x, actor->tileItem.y);
536 SoundPlayAt(
537 &gSoundDevice,
538 StrSound(ActorGetCharacter(actor)->Class->Sounds.Aargh), pos);
539 if (actor->PlayerUID >= 0)
540 {
541 SoundPlayAt(
542 &gSoundDevice,
543 StrSound("hahaha"),
544 pos);
545 }
546 if (actor->tileItem.flags & TILEITEM_OBJECTIVE)
547 {
548 UpdateMissionObjective(
549 &gMission, actor->tileItem.flags, OBJECTIVE_KILL, 1);
550 // If we've killed someone we have rescued, deduct from the
551 // rescue objective
552 if (!(actor->flags & FLAGS_PRISONER))
553 {
554 UpdateMissionObjective(
555 &gMission, actor->tileItem.flags, OBJECTIVE_RESCUE, -1);
556 }
557 }
558 }

580void ActorReplaceGun(const NActorReplaceGun rg)
581{
582 TActor *a = ActorGetByUID(rg.UID);
583 if (a == NULL || !a->isInUse) return;
584 const GunDescription *gun = StrGunDescription(rg.Gun);
585 CASSERT(gun != NULL, "cannot find gun");
586 // If player already has gun, don't do anything
587 if (ActorHasGun(a, gun))
588 {
589 return;
590 }
591 LOG(LM_ACTOR, LL_DEBUG, "actor uid(%d) replacing gun(%s) idx(%d) size(%d)",
592 (int)rg.UID, rg.Gun, rg.GunIdx, (int)a->guns.size);
593 Weapon w = WeaponCreate(gun);
594 if (a->guns.size <= rg.GunIdx)
595 {
596 CASSERT(rg.GunIdx < a->guns.size + 1, "gun idx would leave gap");
597 CArrayPushBack(&a->guns, &w);
598 // Switch immediately to picked up gun
599 a->gunIndex = (int)a->guns.size - 1;
600 }
601 else
602 {
603 memcpy(CArrayGet(&a->guns, rg.GunIdx), &w, a->guns.elemSize);
604 }
605
606 SoundPlayAt(&gSoundDevice, gun->SwitchSound, Vec2iFull2Real(a->Pos));
607}

633void Shoot(TActor *actor)
634{
635 Weapon *gun = ActorGetGun(actor);
636 if (!ActorCanFire(actor))
637 {
638 if (!WeaponIsLocked(gun) && ConfigGetBool(&gConfig, "Game.Ammo"))
639 {
640 CASSERT(ActorGunGetAmmo(actor, gun) == 0, "should be out of ammo");
641 // Play a clicking sound if this gun is out of ammo
642 if (gun->clickLock <= 0)
643 {
644 SoundPlayAt(
645 &gSoundDevice,
646 StrSound("click"), Vec2iFull2Real(actor->Pos));
647 gun->clickLock = SOUND_LOCK_WEAPON_CLICK;
648 }
649 }
650 return;
651 }
652 ActorFire(gun, actor);
653 if (actor->PlayerUID >= 0)
654 {
655 if (ConfigGetBool(&gConfig, "Game.Ammo") && gun->Gun->AmmoId >= 0)
656 {
657 GameEvent e = GameEventNew(GAME_EVENT_ACTOR_USE_AMMO);
658 e.u.UseAmmo.UID = actor->uid;
659 e.u.UseAmmo.PlayerUID = actor->PlayerUID;
660 e.u.UseAmmo.AmmoId = gun->Gun->AmmoId;
661 e.u.UseAmmo.Amount = 1;
662 GameEventsEnqueue(&gGameEvents, e);
663 }
664 else if (gun->Gun->Cost != 0)
665 {
666 // Classic C-Dogs score consumption
667 GameEvent e = GameEventNew(GAME_EVENT_SCORE);
668 e.u.Score.PlayerUID = actor->PlayerUID;
669 e.u.Score.Score = -gun->Gun->Cost;
670 GameEventsEnqueue(&gGameEvents, e);
671 }
672 }
673}

675static bool ActorTryChangeDirection(
676 TActor *actor, const int cmd, const int prevCmd)
677{
678 const bool willChangeDirecton =
679 !actor->petrified &&
680 CMD_HAS_DIRECTION(cmd) &&
681 (!(cmd & CMD_BUTTON2) || ConfigGetEnum(&gConfig, "Game.SwitchMoveStyle") != SWITCHMOVE_STRAFE) &&
682 (!(prevCmd & CMD_BUTTON1) || ConfigGetEnum(&gConfig, "Game.FireMoveStyle") != FIREMOVE_STRAFE);
683 const direction_e dir = CmdToDirection(cmd);
684 if (willChangeDirecton && dir != actor->direction)
685 {
686 GameEvent e = GameEventNew(GAME_EVENT_ACTOR_DIR);
687 e.u.ActorDir.UID = actor->uid;
688 e.u.ActorDir.Dir = (int32_t)dir;
689 GameEventsEnqueue(&gGameEvents, e);
690 // Change direction immediately because this affects shooting
691 actor->direction = dir;
692 }
693 return willChangeDirecton;
694}

696int ActorTryShoot(TActor *actor, int cmd)
697{
698 int willShoot = !actor->petrified && (cmd & CMD_BUTTON1);
699 if (willShoot)
700 {
701 Shoot(actor);
702 }
703 else if (ActorGetGun(actor)->state != GUNSTATE_READY)
704 {
705 GameEvent e = GameEventNew(GAME_EVENT_GUN_STATE);
706 e.u.GunState.ActorUID = actor->uid;
707 e.u.GunState.State = GUNSTATE_READY;
708 GameEventsEnqueue(&gGameEvents, e);
709 }
710 return willShoot;
711}

713static bool ActorTryMove(TActor *actor, int cmd, int hasShot, int ticks);
714void CommandActor(TActor * actor, int cmd, int ticks)
715{
716 if (actor->confused)
717 {
718 cmd = CmdGetReverse(cmd);
719 }
720
721 if (actor->health > 0)
722 {
723 int hasChangedDirection, hasShot, hasMoved;
724 hasChangedDirection = ActorTryChangeDirection(actor, cmd, actor->lastCmd);
725 hasShot = ActorTryShoot(actor, cmd);
726 hasMoved = ActorTryMove(actor, cmd, hasShot, ticks);
727 if (!hasChangedDirection && !hasShot && !hasMoved)
728 {
729 // Idle if player hasn't done anything
730 if (actor->anim.Type != ACTORANIMATION_IDLE)
731 {
732 GameEvent e = GameEventNew(GAME_EVENT_ACTOR_STATE);
733 e.u.ActorState.UID = actor->uid;
734 e.u.ActorState.State = (int32_t)ACTORANIMATION_IDLE;
735 GameEventsEnqueue(&gGameEvents, e);
736 }
737 }
738 }
739
740 actor->lastCmd = cmd;
741 if (cmd & CMD_BUTTON2)
742 {
743 if (CMD_HAS_DIRECTION(cmd))
744 {
745 actor->specialCmdDir = true;
746 }
747 else
748 {
749 // Special: pick up things that can only be picked up on demand
750 if (!actor->PickupAll)
751 {
752 GameEvent e = GameEventNew(GAME_EVENT_ACTOR_PICKUP_ALL);
753 e.u.ActorPickupAll.UID = actor->uid;
754 e.u.ActorPickupAll.PickupAll = true;
755 GameEventsEnqueue(&gGameEvents, e);
756 }
757 actor->PickupAll = true;
758 }
759 }
760 else
761 {
762 actor->specialCmdDir = false;
763 if (actor->PickupAll)
764 {
765 GameEvent e = GameEventNew(GAME_EVENT_ACTOR_PICKUP_ALL);
766 e.u.ActorPickupAll.UID = actor->uid;
767 e.u.ActorPickupAll.PickupAll = false;
768 GameEventsEnqueue(&gGameEvents, e);
769 }
770 actor->PickupAll = false;
771 }
772}
773static bool ActorTryMove(TActor *actor, int cmd, int hasShot, int ticks)
774{
775 const bool canMoveWhenShooting =
776 ConfigGetEnum(&gConfig, "Game.FireMoveStyle") != FIREMOVE_STOP ||
777 !hasShot ||
778 (ConfigGetEnum(&gConfig, "Game.SwitchMoveStyle") == SWITCHMOVE_STRAFE &&
779 (cmd & CMD_BUTTON2));
780 const bool willMove =
781 !actor->petrified && CMD_HAS_DIRECTION(cmd) && canMoveWhenShooting;
782 actor->MoveVel = Vec2iZero();
783 if (willMove)
784 {
785 const int moveAmount = ActorGetCharacter(actor)->speed * ticks;
786 if (cmd & CMD_LEFT)
787 {
788 actor->MoveVel.x -= moveAmount;
789 }
790 else if (cmd & CMD_RIGHT)
791 {
792 actor->MoveVel.x += moveAmount;
793 }
794 if (cmd & CMD_UP)
795 {
796 actor->MoveVel.y -= moveAmount;
797 }
798 else if (cmd & CMD_DOWN)
799 {
800 actor->MoveVel.y += moveAmount;
801 }
802
803 if (actor->anim.Type != ACTORANIMATION_WALKING)
804 {
805 GameEvent e = GameEventNew(GAME_EVENT_ACTOR_STATE);
806 e.u.ActorState.UID = actor->uid;
807 e.u.ActorState.State = (int32_t)ACTORANIMATION_WALKING;
808 GameEventsEnqueue(&gGameEvents, e);
809 }
810 }
811 else
812 {
813 if (actor->anim.Type != ACTORANIMATION_IDLE)
814 {
815 GameEvent e = GameEventNew(GAME_EVENT_ACTOR_STATE);
816 e.u.ActorState.UID = actor->uid;
817 e.u.ActorState.State = (int32_t)ACTORANIMATION_IDLE;
818 GameEventsEnqueue(&gGameEvents, e);
819 }
820 }
821
822 // If we have changed our move commands, send the move event
823 if (cmd != actor->lastCmd || actor->hasCollided)
824 {
825 GameEvent e = GameEventNew(GAME_EVENT_ACTOR_MOVE);
826 e.u.ActorMove.UID = actor->uid;
827 e.u.ActorMove.Pos = Vec2i2Net(actor->Pos);
828 e.u.ActorMove.MoveVel = Vec2i2Net(actor->MoveVel);
829 GameEventsEnqueue(&gGameEvents, e);
830 }
831
832 return willMove;
833}

835void SlideActor(TActor *actor, int cmd)
836{
837 // Check that actor can slide
838 if (actor->slideLock > 0)
839 {
840 return;
841 }
842
843 if (actor->petrified)
844 return;
845
846 if (actor->confused)
847 {
848 cmd = CmdGetReverse(cmd);
849 }
850
851 GameEvent e = GameEventNew(GAME_EVENT_ACTOR_SLIDE);
852 e.u.ActorSlide.UID = actor->uid;
853 Vec2i vel = Vec2iZero();
854 if (cmd & CMD_LEFT) vel.x = -SLIDE_X * 256;
855 else if (cmd & CMD_RIGHT) vel.x = SLIDE_X * 256;
856 if (cmd & CMD_UP) vel.y = -SLIDE_Y * 256;
857 else if (cmd & CMD_DOWN) vel.y = SLIDE_Y * 256;
858 e.u.ActorSlide.Vel = Vec2i2Net(vel);
859 GameEventsEnqueue(&gGameEvents, e);
860
861 actor->slideLock = SLIDE_LOCK;
862}

866void UpdateAllActors(int ticks)
867{
868 CA_FOREACH(TActor, actor, gActors)
869 if (!actor->isInUse)
870 {
871 continue;
872 }
873 ActorUpdatePosition(actor, ticks);
874 UpdateActorState(actor, ticks);
875 if (actor->dead > DEATH_MAX)
876 {
877 if (!gCampaign.IsClient)
878 {
879 ActorDie(actor);
880 }
881 continue;
882 }
883 // Find actors that are on the same team and colliding,
884 // and repel them
885 if (!gCampaign.IsClient &&
886 gCollisionSystem.allyCollision == ALLYCOLLISION_REPEL)
887 {
888 const CollisionParams params =
889 {
890 TILEITEM_IMPASSABLE, COLLISIONTEAM_NONE,
891 IsPVP(gCampaign.Entry.Mode)
892 };
893 const TTileItem *collidingItem = OverlapGetFirstItem(
894 &actor->tileItem, actor->Pos, actor->tileItem.size, params);
895 if (collidingItem && collidingItem->kind == KIND_CHARACTER)
896 {
897 TActor *collidingActor = CArrayGet(
898 &gActors, collidingItem->id);
899 if (CalcCollisionTeam(1, collidingActor) ==
900 CalcCollisionTeam(1, actor))
901 {
902 Vec2i v = Vec2iMinus(actor->Pos, collidingActor->Pos);
903 if (Vec2iIsZero(v))
904 {
905 v = Vec2iNew(1, 0);
906 }
907 v = Vec2iScale(Vec2iNorm(v), REPEL_STRENGTH);
908 GameEvent e = GameEventNew(GAME_EVENT_ACTOR_IMPULSE);
909 e.u.ActorImpulse.UID = actor->uid;
910 e.u.ActorImpulse.Vel = Vec2i2Net(v);
911 e.u.ActorImpulse.Pos = Vec2i2Net(actor->Pos);
912 GameEventsEnqueue(&gGameEvents, e);
913 e.u.ActorImpulse.UID = collidingActor->uid;
914 e.u.ActorImpulse.Vel = Vec2i2Net(Vec2iScale(v, -1));
915 e.u.ActorImpulse.Pos = Vec2i2Net(collidingActor->Pos);
916 GameEventsEnqueue(&gGameEvents, e);
917 }
918 }
919 }
920 // If low on health, bleed
921 if (ActorIsLowHealth(actor))
922 {
923 actor->bleedCounter -= ticks;
924 if (actor->bleedCounter <= 0)
925 {
926
927 ActorAddBloodSplatters(actor, 1, 1.0, Vec2iZero());
928 actor->bleedCounter += ActorGetHealthPercent(actor);
929 }
930 }
931 CA_FOREACH_END()
932}

989static bool CheckManualPickupFunc(
990 TTileItem *ti, void *data, const Vec2i colA, const Vec2i colB,
991 const Vec2i normal)
992{
993 UNUSED(colA);
994 UNUSED(colB);
995 UNUSED(normal);
996 TActor *a = data;
997 if (ti->kind != KIND_PICKUP) return true;
998 const Pickup *p = CArrayGet(&gPickups, ti->id);
999 if (!PickupIsManual(p)) return true;
1000 // "Say" that the weapon must be picked up using a command
1001 const PlayerData *pData = PlayerDataGetByUID(a->PlayerUID);
1002 if (pData->IsLocal && IsPlayerHuman(pData))

1004 char buf[256];
1005 strcpy(buf, "");
1006 InputGetButtonName(
1007 pData->inputDevice, pData->deviceIndex, CMD_BUTTON2, buf);
1008 sprintf(a->Chatter, "%s to pick up\n%s",
1009 buf, IdGunDescription(p->class->u.GunId)->name);
1010 a->ChatterCounter = 2;
1011 }
1012 // If co-op AI, alert it so it can try to pick the gun up
1013 if (a->aiContext != NULL)
1014 {
1015 AICoopOnPickupGun(a, p->class->u.GunId);
1016 }
1017 a->CanPickupSpecial = true;
1018 return false;

1434 const GameMode mode)
1435{

1437 {

1442 {

1445 {
1446 return 1;

1451 // Friendly fire (NPCs)
1452 if (!IsPVP(mode) &&
1453 !ConfigGetBool(&gConfig, "Game.FriendlyFire") &&
1454 isGood && isTargetGood)
1455 {
1456 return 1;
1457 }
1458 // Enemies don't hurt each other
1459 if (!isGood && !isTargetGood)
1460 {
1461 return 1;

1463 }

1468void ActorAddBloodSplatters(
1469 TActor *a, const int power, const double mass, const Vec2i hitVector)
1470{
1471 const GoreAmount ga = ConfigGetEnum(&gConfig, "Graphics.Gore");
1472 if (ga == GORE_NONE) return;
1473
1474 // Emit blood based on power and gore setting
1475 int bloodPower = power * 2;
1476 // Randomly cycle through the blood types
1477 int bloodSize = 1;
1478 while (bloodPower > 0)
1479 {
1480 Emitter *em = NULL;
1481 switch (bloodSize)
1482 {
1483 case 1:
1484 em = &a->blood1;
1485 break;
1486 case 2:
1487 em = &a->blood2;
1488 break;
1489 default:
1490 em = &a->blood3;
1491 break;
1492 }
1493 bloodSize++;
1494 if (bloodSize > 3)
1495 {
1496 bloodSize = 1;
1497 }
1498 const double speed =
1499 (rand() % 8 + 8) * mass / (15 * SHOT_IMPULSE_DIVISOR);
1500 const Vec2i vel =
1501 Vec2iNew((int)(hitVector.x * speed), (int)(hitVector.y * speed));
1502 EmitterStart(em, a->Pos, 10, vel);
1503 switch (ga)
1504 {
1505 case GORE_LOW:
1506 bloodPower /= 8;
1507 break;
1508 case GORE_MEDIUM:
1509 bloodPower /= 2;
1510 break;
1511 default:
1512 bloodPower = bloodPower * 7 / 8;
1513 break;
1514 }
1515 }
1516}
