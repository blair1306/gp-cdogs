39#define TIME_DECAY_EXPONENT 1.04
40#define HEALTH_W 6
41#define HEALTH_H 6
42#define MAX_TILES_PER_PICKUP 300
43
44void PowerupSpawnerInit(PowerupSpawner *p, Map *map)
45{
46 memset(p, 0, sizeof *p);
47 p->map = map;
48 p->timer = 0;
49 p->numPickups = 0;
50 p->Enabled = true;
51}
52void PowerupSpawnerTerminate(PowerupSpawner *p)
53{
54 CFREE(p->Data);
55}
56
57static bool TryPlacePickup(PowerupSpawner *p);
58void PowerupSpawnerUpdate(PowerupSpawner *p, const int ticks)
59{
60 // Don't spawn pickups if not allowed
61 if (!p->Enabled)
62 {
63 return;
64 }
65
66 double scalar = p->RateScaleFunc(p->Data);
67 if (scalar == DBL_MAX)
68 {
69 p->timeUntilNextSpawn = INT_MAX;
70 }
71 else
72 {
73 // Scale down over time
74 scalar *= pow(TIME_DECAY_EXPONENT, p->numPickups);
75
76 p->timeUntilNextSpawn = (int)floor(scalar * p->SpawnTime);
77 debug(D_MAX, "Spawning timer %d next %d spawned %d\n",
78 p->timer, p->timeUntilNextSpawn, p->numPickups);
79 }
80
81 // Update time
82 p->timer += ticks;
83
84 // Attempt to add pickup if time reached, and we haven't placed too many
85 if (p->timer >= p->timeUntilNextSpawn &&
86 p->map->NumExplorableTiles / MAX_TILES_PER_PICKUP + 1 > p->numPickups)
87 {
88 p->timer -= p->timeUntilNextSpawn;
89
90 if (TryPlacePickup(p))
91 {
92 p->numPickups++;
93 }
94 }
95}
96static bool TryPlacePickup(PowerupSpawner *p)
97{
98 const Vec2i size = Vec2iNew(HEALTH_W, HEALTH_H);
99 // Attempt to place one in out-of-sight area
100 for (int i = 0; i < 100; i++)
101 {
102 const Vec2i v = MapGenerateFreePosition(p->map, size);
103 const Vec2i fullpos = Vec2iReal2Full(v);
104 const TActor *closestPlayer = AIGetClosestPlayer(fullpos);
105 if (!Vec2iIsZero(v) &&
106 (!closestPlayer || CHEBYSHEV_DISTANCE(
107 fullpos.x, fullpos.y,
108 closestPlayer->Pos.x, closestPlayer->Pos.y) >= 256 * 150))
109 {
110 p->PlaceFunc(v, p->Data);
111 return true;
112 }
113 }
114 // Attempt to place one anyway
115 for (int i = 0; i < 100; i++)
116 {
117 const Vec2i v = MapGenerateFreePosition(p->map, size);
118 if (!Vec2iIsZero(v))
119 {
120 p->PlaceFunc(v, p->Data);
121 return true;
122 }
123 }
124 return false;
125}

127void PowerupSpawnerRemoveOne(PowerupSpawner *p)
128{
129 p->numPickups = MAX(0, p->numPickups - 1);
130}

135static double HealthScale(void *data);
136static void HealthPlace(const Vec2i pos, void *data);
137void HealthSpawnerInit(PowerupSpawner *p, Map *map)
138{
139 PowerupSpawnerInit(p, map);
140 p->Enabled =
141 AreHealthPickupsAllowed(gCampaign.Entry.Mode) &&
142 ConfigGetBool(&gConfig, "Game.HealthPickups") &&
143 !gCampaign.IsClient;
144 p->SpawnTime = HEALTH_SPAWN_TIME;
145 p->RateScaleFunc = HealthScale;
146 p->PlaceFunc = HealthPlace;
147
148 // Update once
149 PowerupSpawnerUpdate(p, 0);
150}
151static double HealthScale(void *data)
152{
153 UNUSED(data);
154 // Update time until next spawn based on:
155 // Damage taken (find player with lowest health)
156 int minHealth = ModeMaxHealth(gCampaign.Entry.Mode);
157 int maxHealth = minHealth;
158 CA_FOREACH(const PlayerData, p, gPlayerDatas)
159 if (!IsPlayerAlive(p))
160 {
161 continue;
162 }
163 const TActor *player = ActorGetByUID(p->ActorUID);
164 minHealth = MIN(minHealth, player->health);
165 CA_FOREACH_END()
166 // Double spawn rate if near 0 health
167 return (minHealth + maxHealth) / (maxHealth * 2.0);
168}
169static void HealthPlace(const Vec2i pos, void *data)
170{
171 UNUSED(data);
172 GameEvent e = GameEventNew(GAME_EVENT_ADD_PICKUP);
173 e.u.AddPickup.UID = PickupsGetNextUID();
174 e.u.AddPickup.Pos = Vec2i2Net(pos);
175 strcpy(e.u.AddPickup.PickupClass, "health");
176 e.u.AddPickup.IsRandomSpawned = true;
177 e.u.AddPickup.SpawnerUID = -1;
178 e.u.AddPickup.TileItemFlags = 0;
179 GameEventsEnqueue(&gGameEvents, e);
180}

185static double AmmoScale(void *data);
186static void AmmoPlace(const Vec2i pos, void *data);
187void AmmoSpawnerInit(PowerupSpawner *p, Map *map, const int ammoId)
188{
189 PowerupSpawnerInit(p, map);
190 // TODO: disable ammo spawners unless classic mode
191 p->Enabled =
192 ConfigGetBool(&gConfig, "Game.Ammo") &&
193 !gCampaign.IsClient;
194 p->SpawnTime = AMMO_SPAWN_TIME;
195 p->RateScaleFunc = AmmoScale;
196 p->PlaceFunc = AmmoPlace;
197 CMALLOC(p->Data, sizeof ammoId);
198 *(int *)p->Data = ammoId;
199
200 // Update once
201 PowerupSpawnerUpdate(p, 0);
202}
203static double AmmoScale(void *data)
204{
205 const int ammoId = *(int *)data;
206 // Make sure at least one player has a gun that uses this ammo
207 const int numPlayersWithAmmo = PlayersNumUseAmmo(ammoId);
208 if (numPlayersWithAmmo == 0)
209 {
210 // No players have guns that use this ammo; never spawn
211 return DBL_MAX;
212 }
213
214 // Update time until next spawn based on:
215 // Ammo left (find player with lowest ammo)
216 int minVal = AmmoGetById(&gAmmo, ammoId)->Max;
217 int maxVal = minVal;
218 CA_FOREACH(const PlayerData, p, gPlayerDatas)
219 if (!IsPlayerAlive(p))
220 {
221 continue;
222 }
223 const TActor *player = ActorGetByUID(p->ActorUID);
224 minVal = MIN(minVal, *(int *)CArrayGet(&player->ammo, ammoId));
225 CA_FOREACH_END()
226 // 10-fold spawn rate if near 0 ammo
227 return (minVal * 9.0 + maxVal) / (maxVal * 10.0) / numPlayersWithAmmo;
228}
229static void AmmoPlace(const Vec2i pos, void *data)
230{
231 const int ammoId = *(int *)data;
232 GameEvent e = GameEventNew(GAME_EVENT_ADD_PICKUP);
233 e.u.AddPickup.UID = PickupsGetNextUID();
234 e.u.AddPickup.Pos = Vec2i2Net(pos);
235 const Ammo *a = AmmoGetById(&gAmmo, ammoId);
236 sprintf(e.u.AddPickup.PickupClass, "ammo_%s", a->Name);
237 e.u.AddPickup.IsRandomSpawned = true;
238 e.u.AddPickup.SpawnerUID = -1;
239 e.u.AddPickup.TileItemFlags = 0;
240 GameEventsEnqueue(&gGameEvents, e);
241}
