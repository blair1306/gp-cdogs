41void AddIntPair(json_t *parent, const char *name, int number)
42{
43 char buf[32];
44 sprintf(buf, "%d", number);
45 json_insert_pair_into_object(parent, name, json_new_number(buf));
46}
47void AddBoolPair(json_t *parent, const char *name, int value)
48{
49 json_insert_pair_into_object(parent, name, json_new_bool(value));
50}
51void AddStringPair(json_t *parent, const char *name, const char *s)
52{
53 if (!s)
54 {
55 json_insert_pair_into_object(parent, name, json_new_string(""));
56 }
57 else
58 {
59 json_insert_pair_into_object(
60 parent, name, json_new_string(json_escape(s)));
61 }
62}
63void AddColorPair(json_t *parent, const char *name, const color_t c)
64{
65 char buf[8];
66 ColorStr(buf, c);
67 AddStringPair(parent, name, buf);
68}

70int TryLoadValue(json_t **node, const char *name)
71{
72 if (*node == NULL || (*node)->type != JSON_OBJECT)
73 {
74 return 0;
75 }
76 *node = json_find_first_label(*node, name);
77 if (*node == NULL)
78 {
79 return 0;
80 }
81 *node = (*node)->child;
82 if (*node == NULL)
83 {
84 return 0;
85 }
86 return 1;
87}
88void LoadBool(bool *value, json_t *node, const char *name)
89{
90 if (!TryLoadValue(&node, name))
91 {
92 return;
93 }
94 *value = node->type == JSON_TRUE;
95}
96void LoadInt(int *value, json_t *node, const char *name)
97{
98 if (!TryLoadValue(&node, name))
99 {
100 return;
101 }
102 *value = atoi(node->text);
103}
104void LoadDouble(double *value, json_t *node, const char *name)
105{
106 if (!TryLoadValue(&node, name))
107 {
108 return;
109 }
110 *value = atof(node->text);
111}
112void LoadVec2i(Vec2i *value, json_t *node, const char *name)
113{
114 if (!TryLoadValue(&node, name))
115 {
116 return;
117 }
118 node = node->child;
119 value->x = atoi(node->text);
120 node = node->next;
121 value->y = atoi(node->text);
122}
123void LoadStr(char **value, json_t *node, const char *name)
124{
125 if (!TryLoadValue(&node, name))
126 {
127 return;
128 }
129 *value = json_unescape(node->text);
130}
131char *GetString(json_t *node, const char *name)
132{
133 return json_unescape(json_find_first_label(node, name)->child->text);
134}
135void LoadSoundFromNode(Mix_Chunk **value, json_t *node, const char *name)

147void LoadPic(const Pic **value, json_t *node, const char *name)
148{
149 if (json_find_first_label(node, name))
150 {
151 char *tmp = GetString(node, name);
152 *value = PicManagerGetPic(&gPicManager, tmp);
153 CFREE(tmp);
154 }
155}
156void LoadBulletGuns(CArray *guns, json_t *node, const char *name)
157{
158 node = json_find_first_label(node, name);
159 if (node == NULL || node->child == NULL)
160 {
161 return;
162 }
163 CArrayInit(guns, sizeof(const GunDescription *));
164 for (json_t *gun = node->child->child; gun; gun = gun->next)
165 {
166 const GunDescription *g = StrGunDescription(gun->text);
167 CArrayPushBack(guns, &g);
168 }
169}
170void LoadColor(color_t *c, json_t *node, const char *name)
171{
172 if (json_find_first_label(node, name) == NULL)
173 {
174 return;
175 }
176 if (!TryLoadValue(&node, name))
177 {
178 return;
179 }
180 *c = StrColor(node->text);
181}

183json_t *JSONFindNode(json_t *node, const char *path)
184{
185 char *pathCopy;
186 CSTRDUP(pathCopy, path);
187 char *pch = strtok(pathCopy, "/");
188 while (pch != NULL)
189 {
190 node = json_find_first_label(node, pch);
191 if (node == NULL)
192 {
193 goto bail;
194 }
195 node = node->child;
196 if (node == NULL)
197 {
198 goto bail;
199 }
200 pch = strtok(NULL, "/");
201 }
202bail:
203 CFREE(pathCopy);
204 return node;
205}

207bool TrySaveJSONFile(json_t *node, const char *filename)
208{
209 bool res = true;
210 char *text;
211 json_tree_to_string(node, &text);
212 char *ftext = json_format_string(text);
213 FILE *f = fopen(filename, "w");
214 if (f == NULL)
215 {
216 LOG(LM_MAIN, LL_ERROR, "failed to open file(%s) for saving: %s",
217 filename, strerror(errno));
218 res = false;
219 goto bail;
220 }
221 size_t writeLen = strlen(ftext);
222 const size_t rc = fwrite(ftext, 1, writeLen, f);
223 if (rc != writeLen)
224 {
225 LOG(LM_MAIN, LL_ERROR, "Wrote (%d) of (%d) bytes: %s",
226 (int)rc, (int)writeLen, strerror(errno));
227 res = false;
228 goto bail;
229 }
230
231bail:
232 CFREE(text);
233 CFREE(ftext);
234 if (f != NULL) fclose(f);
235 return res;
236}
