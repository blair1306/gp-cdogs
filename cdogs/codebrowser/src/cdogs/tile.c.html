57Tile TileNone(void)
58{
59 Tile t;
60 TileInit(&t);
61 t.flags = MAPTILE_NO_WALK | MAPTILE_IS_NOTHING;
62 return t;
63}
64void TileInit(Tile *t)
65{
66 memset(t, 0, sizeof *t);
67 CArrayInit(&t->triggers, sizeof(Trigger *));
68 CArrayInit(&t->things, sizeof(ThingId));
69 t->pic = NULL;
70 t->picAlt = NULL;
71}
72void TileDestroy(Tile *t)
73{
74 CArrayTerminate(&t->triggers);
75 CArrayTerminate(&t->things);
76}

78bool IsTileItemInsideTile(TTileItem *i, Vec2i tilePos)
79{
80 return
81 i->x - i->size.x / 2 >= tilePos.x * TILE_WIDTH &&
82 i->x + i->size.x / 2 < (tilePos.x + 1) * TILE_WIDTH &&
83 i->y - i->size.y / 2 >= tilePos.y * TILE_HEIGHT &&
84 i->y + i->size.y / 2 < (tilePos.y + 1) * TILE_HEIGHT;
85}

87bool TileCanSee(Tile *t)
88{
89 return !(t->flags & MAPTILE_NO_SEE);
90}
91bool TileCanWalk(const Tile *t)
92{
93 return !(t->flags & MAPTILE_NO_WALK);
94}
95bool TileIsNormalFloor(const Tile *t)
96{
97 return t->flags & MAPTILE_IS_NORMAL_FLOOR;
98}
99bool TileIsClear(const Tile *t)
100{
101 // Check if tile is normal floor
102 const int normalFloorFlags = MAPTILE_IS_NORMAL_FLOOR | MAPTILE_OFFSET_PIC;
103 if (t->flags & ~normalFloorFlags) return false;
104 // Check if tile has no things on it, excluding particles
105 CA_FOREACH(const ThingId, tid, t->things)
106 if (tid->Kind != KIND_PARTICLE) return false;
107 CA_FOREACH_END()
108 return true;
109}
110bool TileHasCharacter(Tile *t)
111{
112 CA_FOREACH(const ThingId, tid, t->things)
113 if (tid->Kind == KIND_CHARACTER)
114 {
115 return true;
116 }
117 CA_FOREACH_END()
118 return false;
119}
120
121void TileSetAlternateFloor(Tile *t, NamedPic *p)
122{
123 t->pic = p;
124 t->flags &= ~MAPTILE_IS_NORMAL_FLOOR;
125}

127void TileItemUpdate(TTileItem *t, const int ticks)
128{
129 t->SoundLock = MAX(0, t->SoundLock - ticks);
130 CPicUpdate(&t->CPic, ticks);
131}

134TTileItem *ThingIdGetTileItem(const ThingId *tid)
135{
136 TTileItem *ti = NULL;
137 switch (tid->Kind)
138 {
139 case KIND_CHARACTER:
140 ti = &((TActor *)CArrayGet(&gActors, tid->Id))->tileItem;
141 break;
142 case KIND_PARTICLE:
143 ti = &((Particle *)CArrayGet(&gParticles, tid->Id))->tileItem;
144 break;
145 case KIND_MOBILEOBJECT:
146 ti = &((TMobileObject *)CArrayGet(
147 &gMobObjs, tid->Id))->tileItem;
148 break;
149 case KIND_OBJECT:
150 ti = &((TObject *)CArrayGet(&gObjs, tid->Id))->tileItem;
151 break;
152 case KIND_PICKUP:
153 ti = &((Pickup *)CArrayGet(&gPickups, tid->Id))->tileItem;
154 break;
155 default:
156 CASSERT(false, "unknown tile item to get");
157 break;
158 }
159 return ti;
160}
