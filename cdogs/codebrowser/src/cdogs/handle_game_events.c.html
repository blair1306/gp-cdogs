49void HandleGameEvents(
50 CArray *store,
51 Camera *camera,
52 PowerupSpawner *healthSpawner,
53 CArray *ammoSpawners)
54{
55 for (int i = 0; i < (int)store->size; i++)
56 {
57 GameEvent *e = CArrayGet(store, i);
58 e->Delay--;
59 if (e->Delay >= 0)
60 {
61 continue;
62 }
63 HandleGameEvent(*e, camera, healthSpawner, ammoSpawners);
64 }
65 GameEventsClear(store);
66}
67static void HandleGameEvent(
68 const GameEvent e,
69 Camera *camera,
70 PowerupSpawner *healthSpawner,
71 CArray *ammoSpawners)
72{
73 switch (e.Type)
74 {
75 case GAME_EVENT_PLAYER_DATA:
76 PlayerDataAddOrUpdate(e.u.PlayerData);
77 break;
78 case GAME_EVENT_PLAYER_REMOVE:
79 PlayerRemove(e.u.PlayerRemove.UID);
80 if (gPlayerDatas.size == 0)
81 {
82 // Waiting for players to join, follow the first one
83 camera->FollowNextPlayer = true;
84 }
85 break;
86 case GAME_EVENT_TILE_SET:
87 {
88 Vec2i pos = Net2Vec2i(e.u.TileSet.Pos);
89 for (int i = 0; i <= e.u.TileSet.RunLength; i++)
90 {
91 Tile *t = MapGetTile(&gMap, pos);
92 t->flags = e.u.TileSet.Flags;
93 t->pic = PicManagerGetNamedPic(
94 &gPicManager, e.u.TileSet.PicName);
95 t->picAlt = PicManagerGetNamedPic(
96 &gPicManager, e.u.TileSet.PicAltName);
97 pos.x++;
98 if (pos.x == gMap.Size.x)
99 {
100 pos.x = 0;
101 pos.y++;
102 }
103 }
104 }
105 break;
106 case GAME_EVENT_MAP_OBJECT_ADD:
107 ObjAdd(e.u.MapObjectAdd);
108 break;
109 case GAME_EVENT_MAP_OBJECT_DAMAGE:
110 DamageObject(e.u.MapObjectDamage);
111 break;
112 case GAME_EVENT_MAP_OBJECT_REMOVE:
113 ObjRemove(e.u.MapObjectRemove);
114 break;
115 case GAME_EVENT_CONFIG:
116 {
117 // Temporarily set config
118 Config *c = ConfigGet(&gConfig, e.u.Config.Name);
119 switch (c->Type)
120 {
121 case CONFIG_TYPE_STRING:
122 CASSERT(false, "unimplemented");
123 break;
124 case CONFIG_TYPE_INT:
125 c->u.Int.Value = atoi(e.u.Config.Value);
126 break;
127 case CONFIG_TYPE_FLOAT:
128 c->u.Float.Value = atof(e.u.Config.Value);
129 break;
130 case CONFIG_TYPE_BOOL:
131 c->u.Bool.Value = strcmp(e.u.Config.Value, "true") == 0;
132 break;
133 case CONFIG_TYPE_ENUM:
134 c->u.Enum.Value = atoi(e.u.Config.Value);
135 break;
136 case CONFIG_TYPE_GROUP:
137 CASSERT(false, "Cannot send groups over net");
138 break;
139 default:
140 CASSERT(false, "Unknown config type");
141 break;
142 }
143 }
144 break;
145 case GAME_EVENT_SCORE:
146 // No score for dogfight
147 if (gCampaign.Entry.Mode != GAME_MODE_DOGFIGHT)
148 {
149 PlayerData *p = PlayerDataGetByUID(e.u.Score.PlayerUID);
150 PlayerScore(p, e.u.Score.Score);
151 HUDNumPopupsAdd(
152 &camera->HUD.numPopups,
153 NUMBER_POPUP_SCORE, e.u.Score.PlayerUID, e.u.Score.Score);
154 }
155 break;
156 case GAME_EVENT_SOUND_AT:
157 if (!e.u.SoundAt.IsHit || ConfigGetBool(&gConfig, "Sound.Hits"))
158 {
159 SoundPlayAt(
160 &gSoundDevice,
161 StrSound(e.u.SoundAt.Sound), Net2Vec2i(e.u.SoundAt.Pos));
162 }
163 break;
164 case GAME_EVENT_SCREEN_SHAKE:
165 camera->shake = ScreenShakeAdd(
166 camera->shake, e.u.ShakeAmount,
167 ConfigGetInt(&gConfig, "Graphics.ShakeMultiplier"));
168 // Weak rumble for all joysticks
169 CA_FOREACH(Joystick, j, gEventHandlers.joysticks)
170 JoyRumble(j->id, 0.3f, 500);
171 CA_FOREACH_END()
172 break;
173 case GAME_EVENT_SET_MESSAGE:
174 HUDDisplayMessage(
175 &camera->HUD, e.u.SetMessage.Message, e.u.SetMessage.Ticks);
176 break;
177 case GAME_EVENT_GAME_START:
178 gMission.HasStarted = true;
179 gMission.HasBegun = false;
180 break;
181 case GAME_EVENT_GAME_BEGIN:
182 MissionBegin(&gMission, e.u.GameBegin);
183 break;
184 case GAME_EVENT_ACTOR_ADD:
185 ActorAdd(e.u.ActorAdd);
186 break;
187 case GAME_EVENT_ACTOR_MOVE:
188 ActorMove(e.u.ActorMove);
189 break;
190 case GAME_EVENT_ACTOR_STATE:
191 {
192 TActor *a = ActorGetByUID(e.u.ActorState.UID);
193 if (!a->isInUse) break;
194 a->anim = AnimationGetActorAnimation(
195 (ActorAnimation)e.u.ActorState.State);
196 }
197 break;
198 case GAME_EVENT_ACTOR_DIR:
199 {
200 TActor *a = ActorGetByUID(e.u.ActorDir.UID);
201 if (!a->isInUse) break;
202 a->direction = (direction_e)e.u.ActorDir.Dir;
203 }
204 break;
205 case GAME_EVENT_ACTOR_SLIDE:
206 {
207 TActor *a = ActorGetByUID(e.u.ActorSlide.UID);
208 if (!a->isInUse) break;
209 a->tileItem.VelFull = Net2Vec2i(e.u.ActorSlide.Vel);
210 // Slide sound
211 if (ConfigGetBool(&gConfig, "Sound.Footsteps"))
212 {
213 SoundPlayAt(
214 &gSoundDevice, StrSound("slide"),
215 Vec2iNew(a->tileItem.x, a->tileItem.y));
216 }
217 }
218 break;
219 case GAME_EVENT_ACTOR_IMPULSE:
220 {
221 TActor *a = ActorGetByUID(e.u.ActorImpulse.UID);
222 if (!a->isInUse) break;
223 a->tileItem.VelFull =
224 Vec2iAdd(a->tileItem.VelFull, Net2Vec2i(e.u.ActorImpulse.Vel));
225 const Vec2i pos = Net2Vec2i(e.u.ActorImpulse.Pos);
226 if (!Vec2iIsZero(pos))
227 {
228 a->Pos = pos;
229 }
230 }
231 break;
232 case GAME_EVENT_ACTOR_SWITCH_GUN:
233 ActorSwitchGun(e.u.ActorSwitchGun);
234 break;
235 case GAME_EVENT_ACTOR_PICKUP_ALL:
236 {
237 TActor *a = ActorGetByUID(e.u.ActorPickupAll.UID);
238 if (!a->isInUse) break;
239 a->PickupAll = e.u.ActorPickupAll.PickupAll;
240 }
241 break;
242 case GAME_EVENT_ACTOR_REPLACE_GUN:
243 ActorReplaceGun(e.u.ActorReplaceGun);
244 break;
245 case GAME_EVENT_ACTOR_HEAL:
246 {
247 TActor *a = ActorGetByUID(e.u.Heal.UID);
248 if (!a->isInUse || a->dead) break;
249 ActorHeal(a, e.u.Heal.Amount);
250 // Sound of healing
251 SoundPlayAt(
252 &gSoundDevice, StrSound("health"), Vec2iFull2Real(a->Pos));
253 // Tell the spawner that we took a health so we can
254 // spawn more (but only if we're the server)
255 if (e.u.Heal.IsRandomSpawned && !gCampaign.IsClient)
256 {
257 PowerupSpawnerRemoveOne(healthSpawner);
258 }
259 if (e.u.Heal.PlayerUID >= 0)
260 {
261 GameEvent s = GameEventNew(GAME_EVENT_ADD_PARTICLE);
262 s.u.AddParticle.Class =
263 StrParticleClass(&gParticleClasses, "heal_text");
264 s.u.AddParticle.FullPos = a->Pos;
265 s.u.AddParticle.Z = BULLET_Z * Z_FACTOR;
266 s.u.AddParticle.DZ = 3;
267 sprintf(s.u.AddParticle.Text, "+%d", (int)e.u.Heal.Amount);
268 GameEventsEnqueue(&gGameEvents, s);
269 }
270 }
271 break;
272 case GAME_EVENT_ACTOR_ADD_AMMO:
273 {
274 TActor *a = ActorGetByUID(e.u.AddAmmo.UID);
275 if (!a->isInUse || a->dead) break;
276 ActorAddAmmo(a, e.u.AddAmmo.AmmoId, e.u.AddAmmo.Amount);
277 // Tell the spawner that we took ammo so we can
278 // spawn more (but only if we're the server)
279 if (e.u.AddAmmo.IsRandomSpawned && !gCampaign.IsClient)
280 {
281 PowerupSpawnerRemoveOne(
282 CArrayGet(ammoSpawners, e.u.AddAmmo.AmmoId));
283 }
284 if (e.u.AddAmmo.PlayerUID >= 0)
285 {
286 GameEvent s = GameEventNew(GAME_EVENT_ADD_PARTICLE);
287 s.u.AddParticle.Class =
288 StrParticleClass(&gParticleClasses, "ammo_text");
289 s.u.AddParticle.FullPos = a->Pos;
290 s.u.AddParticle.Z = BULLET_Z * Z_FACTOR;
291 s.u.AddParticle.DZ = 10;
292 const Ammo *ammo = AmmoGetById(&gAmmo, e.u.AddAmmo.AmmoId);
293 sprintf(
294 s.u.AddParticle.Text, "+%d %s",
295 (int)e.u.AddAmmo.Amount, ammo->Name);
296 GameEventsEnqueue(&gGameEvents, s);
297 }
298 }
299 break;
300 case GAME_EVENT_ACTOR_USE_AMMO:
301 {
302 TActor *a = ActorGetByUID(e.u.UseAmmo.UID);
303 if (!a->isInUse || a->dead) break;
304 const int ammoBefore =
305 *(int *)CArrayGet(&a->ammo, e.u.UseAmmo.AmmoId);
306 const Ammo *ammo = AmmoGetById(&gAmmo, e.u.UseAmmo.AmmoId);
307 const bool wasAmmoLow = AmmoIsLow(ammo, ammoBefore);
308 ActorAddAmmo(a, e.u.UseAmmo.AmmoId, -(int)e.u.UseAmmo.Amount);
309 const PlayerData *p = PlayerDataGetByUID(e.u.UseAmmo.PlayerUID);
310 if (p != NULL && p->IsLocal)
311 {
312 // Show low or no ammo notifications
313 const int ammoAfter =
314 *(int *)CArrayGet(&a->ammo, e.u.UseAmmo.AmmoId);
315 const bool isAmmoLow = AmmoIsLow(ammo, ammoAfter);
316 if (ammoAfter == 0)
317 {
318 // No ammo
319 SoundPlay(&gSoundDevice, StrSound("ammo_none"));
320 }
321 else if (!wasAmmoLow && isAmmoLow)
322 {
323 // Low ammo
324 SoundPlay(&gSoundDevice, StrSound("ammo_low"));
325 }
326 }
327 }
328 break;
329 case GAME_EVENT_ACTOR_DIE:
330 {
331 TActor *a = ActorGetByUID(e.u.ActorDie.UID);
332
333 // Check if the player has lives to revive
334 PlayerData *p = PlayerDataGetByUID(a->PlayerUID);
335 if (p != NULL)
336 {
337 p->Lives--;
338 CASSERT(p->Lives >= 0, "Player has died too many times");
339 if (p->Lives > 0 && !gCampaign.IsClient)
340 {
341 // Find the closest player alive; try to spawn next to that position
342 // if no other suitable position exists
343 Vec2i defaultSpawnPosition = Vec2iZero();
344 const TActor *closestActor = AIGetClosestPlayer(a->Pos);
345 if (closestActor != NULL) defaultSpawnPosition = closestActor->Pos;
346 PlacePlayer(&gMap, p, defaultSpawnPosition, false);
347 }
348 }
349
350 ActorDestroy(a);
351 }
352 break;
353 case GAME_EVENT_ACTOR_MELEE:
354 {
355 const TActor *a = ActorGetByUID(e.u.Melee.UID);
356 if (!a->isInUse) break;
357 const BulletClass *b = StrBulletClass(e.u.Melee.BulletClass);
358 if ((HitType)e.u.Melee.HitType != HIT_NONE &&
359 HasHitSound(a->flags, a->PlayerUID,
360 (TileItemKind)e.u.Melee.TargetKind, e.u.Melee.TargetUID,
361 SPECIAL_NONE, false))
362 {
363 PlayHitSound(
364 &b->HitSound, (HitType)e.u.Melee.HitType,
365 Vec2iFull2Real(a->Pos));
366 }
367 if (!gCampaign.IsClient)
368 {
369 // TODO: melee hitback (vel)?
370 Damage(
371 Vec2iZero(),
372 b->Power, b->Mass,
373 a->flags, a->PlayerUID, a->uid,
374 (TileItemKind)e.u.Melee.TargetKind, e.u.Melee.TargetUID,
375 SPECIAL_NONE);
376 }
377 }
378 break;
379 case GAME_EVENT_ADD_PICKUP:
380 PickupAdd(e.u.AddPickup);
381 // Play a spawn sound
382 SoundPlayAt(
383 &gSoundDevice,
384 StrSound("spawn_item"), Net2Vec2i(e.u.AddPickup.Pos));
385 break;
386 case GAME_EVENT_REMOVE_PICKUP:
387 PickupDestroy(e.u.RemovePickup.UID);
388 if (e.u.RemovePickup.SpawnerUID >= 0)
389 {
390 TObject *o = ObjGetByUID(e.u.RemovePickup.SpawnerUID);
391 o->counter = AMMO_SPAWNER_RESPAWN_TICKS;
392 }
393 break;
394 case GAME_EVENT_BULLET_BOUNCE:
395 {
396 TMobileObject *o = MobObjGetByUID(e.u.BulletBounce.UID);
397 if (o == NULL || !o->isInUse) break;
398 const Vec2i bouncePos = Net2Vec2i(e.u.BulletBounce.BouncePos);
399 if (e.u.BulletBounce.HitSound)
400 {
401 PlayHitSound(
402 &o->bulletClass->HitSound,
403 (HitType)e.u.BulletBounce.HitType,
404 Vec2iFull2Real(bouncePos));
405 }
406 if (e.u.BulletBounce.Spark && o->bulletClass->Spark != NULL)
407 {
408 GameEvent s = GameEventNew(GAME_EVENT_ADD_PARTICLE);
409 s.u.AddParticle.Class = o->bulletClass->Spark;
410 s.u.AddParticle.FullPos = bouncePos;
411 s.u.AddParticle.Z = o->z;
412 GameEventsEnqueue(&gGameEvents, s);
413 }
414 const Vec2i pos = Net2Vec2i(e.u.BulletBounce.Pos);
415 o->x = pos.x;
416 o->y = pos.y;
417 o->tileItem.VelFull = Net2Vec2i(e.u.BulletBounce.Vel);
418 }
419 break;
420 case GAME_EVENT_REMOVE_BULLET:
421 {
422 TMobileObject *o = MobObjGetByUID(e.u.RemoveBullet.UID);
423 if (o == NULL || !o->isInUse) break;
424 MobObjDestroy(o);
425 }
426 break;
427 case GAME_EVENT_PARTICLE_REMOVE:
428 ParticleDestroy(&gParticles, e.u.ParticleRemoveId);
429 break;
430 case GAME_EVENT_GUN_FIRE:
431 {
432 const GunDescription *g = StrGunDescription(e.u.GunFire.Gun);
433 const Vec2i fullPos = Net2Vec2i(e.u.GunFire.MuzzleFullPos);
434
435 // Add bullets
436 if (g->Bullet && !gCampaign.IsClient)
437 {
438 // Find the starting angle of the spread (clockwise)
439 // Keep in mind the fencepost problem, i.e. spread of 3 means a
440 // total spread angle of 2x width
441 const double spreadStartAngle =
442 g->AngleOffset -
443 (g->Spread.Count - 1) * g->Spread.Width / 2;
444 for (int i = 0; i < g->Spread.Count; i++)
445 {
446 const double recoil =
447 ((double)rand() / RAND_MAX * g->Recoil) -
448 g->Recoil / 2;
449 const double finalAngle =
450 e.u.GunFire.Angle + spreadStartAngle +
451 i * g->Spread.Width + recoil;
452 GameEvent ab = GameEventNew(GAME_EVENT_ADD_BULLET);
453 ab.u.AddBullet.UID = MobObjsObjsGetNextUID();
454 strcpy(ab.u.AddBullet.BulletClass, g->Bullet->Name);
455 ab.u.AddBullet.MuzzlePos = Vec2i2Net(fullPos);
456 ab.u.AddBullet.MuzzleHeight = e.u.GunFire.Z;
457 ab.u.AddBullet.Angle = (float)finalAngle;
458 ab.u.AddBullet.Elevation =
459 RAND_INT(g->ElevationLow, g->ElevationHigh);
460 ab.u.AddBullet.Flags = e.u.GunFire.Flags;
461 ab.u.AddBullet.PlayerUID = e.u.GunFire.PlayerUID;
462 ab.u.AddBullet.ActorUID = e.u.GunFire.UID;
463 GameEventsEnqueue(&gGameEvents, ab);
464 }
465 }
466
467 // Add muzzle flash
468 if (GunHasMuzzle(g) && g->MuzzleFlash != NULL)
469 {
470 GameEvent ap = GameEventNew(GAME_EVENT_ADD_PARTICLE);
471 ap.u.AddParticle.Class = g->MuzzleFlash;
472 ap.u.AddParticle.FullPos = fullPos;
473 ap.u.AddParticle.Z = e.u.GunFire.Z;
474 ap.u.AddParticle.Angle = e.u.GunFire.Angle;
475 GameEventsEnqueue(&gGameEvents, ap);
476 }
477 // Sound
478 if (e.u.GunFire.Sound && g->Sound)
479 {
480 SoundPlayAt(&gSoundDevice, g->Sound, Vec2iFull2Real(fullPos));
481 }
482 // Screen shake
483 if (g->ShakeAmount > 0)
484 {
485 GameEvent s = GameEventNew(GAME_EVENT_SCREEN_SHAKE);
486 s.u.ShakeAmount = g->ShakeAmount;
487 GameEventsEnqueue(&gGameEvents, s);
488 }
489 // Brass shells
490 // If we have a reload lead, defer the creation of shells until then
491 if (g->Brass && g->ReloadLead == 0)
492 {
493 const direction_e d = RadiansToDirection(e.u.GunFire.Angle);
494 GunAddBrass(g, d, fullPos);
495 }
496 }
497 break;
498 case GAME_EVENT_GUN_RELOAD:
499 {
500 const GunDescription *g = StrGunDescription(e.u.GunReload.Gun);
501 const Vec2i fullPos = Net2Vec2i(e.u.GunReload.FullPos);
502 SoundPlayAtPlusDistance(
503 &gSoundDevice,
504 g->ReloadSound,
505 Vec2iFull2Real(fullPos),
506 RELOAD_DISTANCE_PLUS);
507 // Brass shells
508 if (g->Brass)
509 {
510 GunAddBrass(g, (direction_e)e.u.GunReload.Direction, fullPos);
511 }
512 }
513 break;
514 case GAME_EVENT_GUN_STATE:
515 {
516 const TActor *a = ActorGetByUID(e.u.GunState.ActorUID);
517 if (!a->isInUse) break;
518 WeaponSetState(ActorGetGun(a), (gunstate_e)e.u.GunState.State);
519 }
520 break;
521 case GAME_EVENT_ADD_BULLET:
522 BulletAdd(e.u.AddBullet);
523 break;
524 case GAME_EVENT_ADD_PARTICLE:
525 ParticleAdd(&gParticles, e.u.AddParticle);
526 break;
527 case GAME_EVENT_ACTOR_HIT:
528 {
529 TActor *a = ActorGetByUID(e.u.ActorHit.UID);
530 if (!a->isInUse) break;
531 ActorTakeHit(a, e.u.ActorHit.Special);
532 if (e.u.ActorHit.Power > 0)
533 {
534 DamageActor(
535 a, e.u.ActorHit.Power, e.u.ActorHit.HitterPlayerUID);
536 GameEvent s = GameEventNew(GAME_EVENT_ADD_PARTICLE);
537 s.u.AddParticle.Class =
538 StrParticleClass(&gParticleClasses, "damage_text");
539 s.u.AddParticle.FullPos = Vec2iAdd(
540 a->Pos,
541 Vec2iReal2Full(Vec2iNew(rand() % 6 - 3, rand() % 6 - 3)));
542 s.u.AddParticle.Z = BULLET_Z * Z_FACTOR;
543 s.u.AddParticle.DZ = 3;
544 sprintf(
545 s.u.AddParticle.Text, "-%d", (int)e.u.ActorHit.Power);
546 GameEventsEnqueue(&gGameEvents, s);
547
548 ActorAddBloodSplatters(
549 a, e.u.ActorHit.Power, e.u.ActorHit.Mass, Net2Vec2i(e.u.ActorHit.Vel));
550
551 // Rumble if taking hit
552 if (a->PlayerUID >= 0)
553 {
554 const PlayerData *p = PlayerDataGetByUID(a->PlayerUID);
555 if (p->inputDevice == INPUT_DEVICE_JOYSTICK)
556 {
557 JoyImpact(p->deviceIndex);
558 }
559 }
560 }
561 }
562 break;
563 case GAME_EVENT_TRIGGER:
564 {
565 const Tile *t =
566 MapGetTile(&gMap, Net2Vec2i(e.u.TriggerEvent.Tile));
567 CA_FOREACH(Trigger *, tp, t->triggers)
568 if ((*tp)->id == (int)e.u.TriggerEvent.ID)
569 {
570 TriggerActivate(*tp, &gMap.triggers);
571 break;
572 }
573 CA_FOREACH_END()
574 }
575 break;
576 case GAME_EVENT_EXPLORE_TILES:
577 // Process runs of explored tiles
578 for (int i = 0; i < (int)e.u.ExploreTiles.Runs_count; i++)
579 {
580 Vec2i tile = Net2Vec2i(e.u.ExploreTiles.Runs[i].Tile);
581 for (int j = 0; j < e.u.ExploreTiles.Runs[i].Run; j++)
582 {
583 MapMarkAsVisited(&gMap, tile);
584 tile.x++;
585 if (tile.x == gMap.Size.x)
586 {
587 tile.x = 0;
588 tile.y++;
589 }
590 }
591 }
592 break;
593 case GAME_EVENT_RESCUE_CHARACTER:
594 {
595 TActor *a = ActorGetByUID(e.u.Rescue.UID);
596 if (!a->isInUse) break;
597 a->flags &= ~FLAGS_PRISONER;
598 // If the actor isn't a follower, make them automatically run
599 // towards the exit
600 if (!(a->flags & FLAGS_FOLLOWER))
601 {
602 a->flags |= FLAGS_RESCUED;
603 }
604 SoundPlayAt(
605 &gSoundDevice, StrSound("rescue"), Vec2iFull2Real(a->Pos));
606 }
607 break;
608 case GAME_EVENT_OBJECTIVE_UPDATE:
609 {
610 Objective *o = CArrayGet(
611 &gMission.missionData->Objectives,
612 e.u.ObjectiveUpdate.ObjectiveId);
613 o->done += e.u.ObjectiveUpdate.Count;
614 // Display a text update effect for the objective
615 if (camera != NULL)
616 {
617 HUDNumPopupsAdd(
618 &camera->HUD.numPopups, NUMBER_POPUP_OBJECTIVE,
619 e.u.ObjectiveUpdate.ObjectiveId,
620 e.u.ObjectiveUpdate.Count);
621 }
622 MissionSetMessageIfComplete(&gMission);
623 }
624 break;
625 case GAME_EVENT_ADD_KEYS:
626 {
627 gMission.KeyFlags |= e.u.AddKeys.KeyFlags;
628
629 const Vec2i pos = Net2Vec2i(e.u.AddKeys.Pos);
630
631 if (!Vec2iIsZero(pos))
632 {
633 SoundPlayAt(&gSoundDevice, StrSound("key"), pos);
634
635 GameEvent s = GameEventNew(GAME_EVENT_ADD_PARTICLE);
636 s.u.AddParticle.Class =
637 StrParticleClass(&gParticleClasses, "key_text");
638 s.u.AddParticle.FullPos = Vec2iReal2Full(pos);
639 s.u.AddParticle.Z = BULLET_Z * Z_FACTOR;
640 s.u.AddParticle.DZ = 10;
641 sprintf(s.u.AddParticle.Text, "+key");
642 GameEventsEnqueue(&gGameEvents, s);
643 }
644
645 // Clear cache since we may now have new paths
646 PathCacheClear(&gPathCache);
647 }
648 break;
649 case GAME_EVENT_MISSION_COMPLETE:
650 if (camera != NULL && e.u.MissionComplete.ShowMsg)
651 {
652 HUDDisplayMessage(&camera->HUD, "Mission complete", -1);
653 }
654 // Don't show exit area or arrow if PVP
655 if (!IsPVP(gCampaign.Entry.Mode))
656 {
657 if (camera != NULL)
658 {
659 camera->HUD.showExit = true;
660 }
661 MapShowExitArea(
662 &gMap,
663 Net2Vec2i(e.u.MissionComplete.ExitStart),
664 Net2Vec2i(e.u.MissionComplete.ExitEnd));
665 }
666 break;
667 case GAME_EVENT_MISSION_INCOMPLETE:
668 gMission.state = MISSION_STATE_PLAY;
669 break;
670 case GAME_EVENT_MISSION_PICKUP:
671 gMission.state = MISSION_STATE_PICKUP;
672 gMission.pickupTime = gMission.time;
673 SoundPlay(&gSoundDevice, StrSound("whistle"));
674 break;
675 case GAME_EVENT_MISSION_END:
676 MissionDone(&gMission, e.u.MissionEnd);
677 if (e.u.MissionEnd.Msg[0] != '\0')
678 {
679 HUDDisplayMessage(&camera->HUD, e.u.MissionEnd.Msg, -1);
680 }
681 break;
682 default:
683 assert(0 && "unknown game event");
684 break;
685 }
686}
