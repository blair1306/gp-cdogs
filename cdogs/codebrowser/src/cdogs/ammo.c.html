46int StrAmmoId(const char *s)
47{
48 if (s == NULL || strlen(s) == 0)
49 {
50 return 0;
51 }
52 CA_FOREACH(Ammo, a, gAmmo.CustomAmmo)
53 if (strcmp(s, a->Name) == 0)
54 {
55 return _ca_index + (int)gAmmo.Ammo.size;
56 }
57 CA_FOREACH_END()
58 CA_FOREACH(Ammo, a, gAmmo.Ammo)
59 if (strcmp(s, a->Name) == 0)
60 {
61 return _ca_index;
62 }
63 CA_FOREACH_END()
64 CASSERT(false, "cannot parse ammo name");
65 return 0;
66}

68#define VERSION 1
69static void LoadAmmo(Ammo *a, json_t *node);
70void AmmoInitialize(AmmoClasses *ammo, const char *path)
71{
72 memset(ammo, 0, sizeof *ammo);
73 CArrayInit(&ammo->Ammo, sizeof(Ammo));
74 CArrayInit(&ammo->CustomAmmo, sizeof(Ammo));
75
76 json_t *root = NULL;
77 enum json_error e;
78
79 char buf[CDOGS_PATH_MAX];
80 GetDataFilePath(buf, path);
81 FILE *f = fopen(buf, "r");
82 if (f == NULL)
83 {
84 LOG(LM_MAIN, LL_ERROR, "Error: cannot load ammo file %s", buf);
85 goto bail;
86 }
87 e = json_stream_parse(f, &root);
88 if (e != JSON_OK)
89 {
90 LOG(LM_MAIN, LL_ERROR, "Error parsing ammo file %s [error %d]",
91 buf, (int)e);
92 goto bail;
93 }
94 AmmoLoadJSON(&ammo->Ammo, root);
95
96bail:
97 if (f)
98 {
99 fclose(f);
100 }
101 json_free_value(&root);
102}
103void AmmoLoadJSON(CArray *ammo, json_t *node)
104{
105 int version;
106 LoadInt(&version, node, "Version");
107 if (version > VERSION || version <= 0)
108 {
109 CASSERT(false, "cannot read ammo file version");
110 return;
111 }
112
113 json_t *ammoNode = json_find_first_label(node, "Ammo")->child;
114 for (json_t *child = ammoNode->child; child; child = child->next)
115 {
116 Ammo a;
117 LoadAmmo(&a, child);
118 CArrayPushBack(ammo, &a);
119 }
120}
121static void LoadAmmo(Ammo *a, json_t *node)
122{
123 a->Name = GetString(node, "Name");
124 char *tmp;
125 tmp = GetString(node, "Pic");
126 a->Pic = PicManagerGetPic(&gPicManager, tmp);
127 CFREE(tmp);
128 a->Sound = GetString(node, "Sound");
129 LoadInt(&a->Amount, node, "Amount");
130 LoadInt(&a->Max, node, "Max");
131}
132void AmmoClassesClear(CArray *ammo)
133{
134 for (int i = 0; i < (int)ammo->size; i++)
135 {
136 Ammo *a = CArrayGet(ammo, i);
137 CFREE(a->Name);
138 CFREE(a->Sound);
139 }
140 CArrayClear(ammo);
141}
142void AmmoTerminate(AmmoClasses *ammo)
143{
144 AmmoClassesClear(&ammo->Ammo);
145 CArrayTerminate(&ammo->Ammo);
146 AmmoClassesClear(&ammo->CustomAmmo);
147 CArrayTerminate(&ammo->CustomAmmo);
148}

150Ammo *AmmoGetById(AmmoClasses *ammo, const int id)
151{
152 if (id < (int)ammo->Ammo.size)
153 {
154 return CArrayGet(&ammo->Ammo, id);
155 }
156 else
157 {
158 return CArrayGet(&ammo->CustomAmmo, id - (int)ammo->Ammo.size);
159 }
160}
161
162int AmmoGetNumClasses(const AmmoClasses *ammo)
163{
164 return (int)ammo->Ammo.size + (int)ammo->CustomAmmo.size;
165}

168bool AmmoIsLow(const Ammo *a, const int amount)
169{
170 // Low ammo if less than or equal to half the default amount
171 return amount <= (a->Amount / 2);
172}
