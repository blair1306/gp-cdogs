32static char *ReadFile(const char *filename);
33yajl_val YAJLReadFile(const char *filename)
34{
35 yajl_val node = NULL;
36 char errbuf[1024];
37 char *buf = ReadFile(filename);
38 if (buf == NULL)
39 {
40 fprintf(stderr, "Error reading JSON file '%s'\n", filename);
41 goto bail;
42 }
43 node = yajl_tree_parse(buf, errbuf, sizeof errbuf);
44 if (node == NULL)
45 {
46 fprintf(stderr, "Error parsing font JSON '%s': %s\n", filename, errbuf);
47 goto bail;
48 }
49
50bail:
51 free(buf);
52 return node;
53}
54// Read a file into a dynamic buffer
55static char *ReadFile(const char *filename)
56{
57 FILE *f = fopen(filename, "r");
58 char *buf = NULL;
59 if (f == NULL) goto bail;
60 if (fseek(f, 0L, SEEK_END) != 0) goto bail;
61 const long bufsize = ftell(f);
62 if (bufsize == -1) goto bail;
63 if (fseek(f, 0L, SEEK_SET) != 0) goto bail;
64 buf = malloc(bufsize + 1);
65 if (buf == NULL) goto bail;
66 const size_t readlen = fread(buf, 1, bufsize, f);
67 buf[readlen] = '\0';
68
69bail:
70 if (f)
71 {
72 fclose(f);
73 }
74 return buf;
75}

113bool YAJLTryLoadValue(yajl_val *node, const char *name)
114{
115 if (*node == NULL || !YAJL_IS_OBJECT(*node))
116 {
117 return false;
118 }
119 const char *path[] = { name, NULL };
120 *node = yajl_tree_get(*node, path, yajl_t_any);
121 return *node != NULL;
122}
123void YAJLBool(bool *value, yajl_val node, const char *name)
124{
125 if (!YAJLTryLoadValue(&node, name))
126 {
127 return;
128 }
129 *value = YAJL_IS_TRUE(node);
130}

155void YAJLStr(char **value, yajl_val node, const char *name)
156{
157 if (!YAJLTryLoadValue(&node, name) || !YAJL_IS_STRING(node))
158 {
159 return;
160 }
161 *value = YAJLGetStr(node, name);
162}
163char *YAJLGetStr(yajl_val node, const char *name)
164{
165 char *in = YAJL_GET_STRING(YAJLFindNode(node, name));
166 CASSERT(in != NULL, "cannot get JSON string");
167 char *out;
168 CSTRDUP(out, in);
169 return out;
170}

220yajl_val YAJLFindNode(yajl_val node, const char *path)
221{
222 // max 256 levels
223 const char *pathSplit[256];
224 for (int i = 0; i < 256; i++) pathSplit[i] = NULL;
225 char *pathCopy;
226 CSTRDUP(pathCopy, path);
227 char *pch = strtok(pathCopy, "/");
228 int i = 0;
229 yajl_val out = NULL;
230 while (pch != NULL)
231 {
232 if (i == 256)
233 {
234 fprintf(stderr, "JSON path too long: '%s'\n", path);
235 goto bail;
236 }
237 pathSplit[i] = pch;
238 i++;
239 pch = strtok(NULL, "/");
240 }
241 out = yajl_tree_get(node, pathSplit, yajl_t_any);
242
243bail:
244 CFREE(pathCopy);
245 return out;
246}
