67static CPicDrawContext GetMapObjectDrawContext(const int id)
68{
69 TObject *obj = CArrayGet(&gObjs, id);
70 CASSERT(obj->isInUse, "Cannot draw non-existent mobobj");
71 CPicDrawContext c;
72 c.Dir = DIRECTION_UP;
73 c.Offset = obj->Class->Offset;
74 CPicCopyPic(&obj->tileItem.CPic, &obj->Class->Pic);
75 return c;
76}

79void DamageObject(const NMapObjectDamage mod)
80{
81 TObject *o = ObjGetByUID(mod.UID);
82 // Don't bother if object already destroyed
83 if (o->Health <= 0)
84 {
85 return;
86 }
87
88 o->Health -= mod.Power;
89
90 // Destroying objects and all the wonderful things that happen
91 if (o->Health <= 0 && !gCampaign.IsClient)
92 {
93 GameEvent e = GameEventNew(GAME_EVENT_MAP_OBJECT_REMOVE);
94 e.u.MapObjectRemove.UID = o->uid;
95 e.u.MapObjectRemove.ActorUID = mod.UID;
96 e.u.MapObjectRemove.PlayerUID = mod.PlayerUID;
97 e.u.MapObjectRemove.Flags = mod.Flags;
98 GameEventsEnqueue(&gGameEvents, e);
99 }
100}

102static void AddPickupAtObject(const TObject *o, const PickupType type)
103{
104 GameEvent e = GameEventNew(GAME_EVENT_ADD_PICKUP);
105 switch (type)
106 {
107 case PICKUP_JEWEL: CASSERT(false, "unexpected pickup type"); break;
108 case PICKUP_HEALTH:
109 if (!ConfigGetBool(&gConfig, "Game.HealthPickups"))
110 {
111 return;
112 }
113 strcpy(e.u.AddPickup.PickupClass, "health");
114 break;
115 case PICKUP_AMMO:
116 if (!ConfigGetBool(&gConfig, "Game.Ammo"))
117 {
118 return;
119 }
120 // Pick a random ammo type and spawn it
121 {
122 const int ammoId = rand() % AmmoGetNumClasses(&gAmmo);
123 const Ammo *a = AmmoGetById(&gAmmo, ammoId);
124 sprintf(e.u.AddPickup.PickupClass, "ammo_%s", a->Name);
125 }
126 break;
127 case PICKUP_KEYCARD: CASSERT(false, "unexpected pickup type"); break;
128 case PICKUP_GUN:
129 // Pick a random mission gun type and spawn it
130 {
131 const int gunId = (int)(rand() % gMission.Weapons.size);
132 const GunDescription **gun = CArrayGet(&gMission.Weapons, gunId);
133 sprintf(e.u.AddPickup.PickupClass, "gun_%s", (*gun)->name);
134 }
135 break;
136 default: CASSERT(false, "unexpected pickup type"); break;
137 }
138 e.u.AddPickup.UID = PickupsGetNextUID();
139 e.u.AddPickup.Pos = Vec2i2Net(Vec2iNew(o->tileItem.x, o->tileItem.y));
140 e.u.AddPickup.IsRandomSpawned = true;
141 e.u.AddPickup.SpawnerUID = -1;
142 e.u.AddPickup.TileItemFlags = 0;
143 GameEventsEnqueue(&gGameEvents, e);
144}

147void ObjRemove(const NMapObjectRemove mor)
148{
149 TObject *o = ObjGetByUID(mor.UID);
150 o->Health = 0;
151
152 const Vec2i realPos = Vec2iNew(o->tileItem.x, o->tileItem.y);
153
154 if (!gCampaign.IsClient)
155 {
156 // Update objective
157 UpdateMissionObjective(
158 &gMission, o->tileItem.flags, OBJECTIVE_DESTROY, 1);
159 // Extra score if objective
160 if ((o->tileItem.flags & TILEITEM_OBJECTIVE) && mor.PlayerUID >= 0)
161 {
162 GameEvent e = GameEventNew(GAME_EVENT_SCORE);
163 e.u.Score.PlayerUID = mor.PlayerUID;
164 e.u.Score.Score = OBJECT_SCORE;
165 GameEventsEnqueue(&gGameEvents, e);
166 }
167
168 // Weapons that go off when this object is destroyed
169 const Vec2i fullPos = Vec2iReal2Full(realPos);
170 CA_FOREACH(const GunDescription *, g, o->Class->DestroyGuns)
171 GunFire(
172 *g, fullPos, 0, 0, mor.Flags, mor.PlayerUID, mor.ActorUID,
173 true, false);
174 CA_FOREACH_END()
175
176 // Random chance to add pickups in single player modes
177 if (!IsPVP(gCampaign.Entry.Mode))
178 {
179 CA_FOREACH(
180 const MapObjectDestroySpawn, mods, o->Class->DestroySpawn)
181 const double chance = (double) rand() / RAND_MAX;
182 if (chance < mods->SpawnChance)
183 {
184 AddPickupAtObject(o, mods->Type);
185 }
186 CA_FOREACH_END()
187 }
188
189 // A wreck left after the destruction of this object
190 // TODO: doesn't need to be network event
191 GameEvent e = GameEventNew(GAME_EVENT_ADD_BULLET);
192 e.u.AddBullet.UID = MobObjsObjsGetNextUID();
193 strcpy(e.u.AddBullet.BulletClass, "fireball_wreck");
194 e.u.AddBullet.MuzzlePos = Vec2i2Net(fullPos);
195 e.u.AddBullet.MuzzleHeight = 0;
196 e.u.AddBullet.Angle = 0;
197 e.u.AddBullet.Elevation = 0;
198 e.u.AddBullet.Flags = 0;
199 e.u.AddBullet.PlayerUID = -1;
200 e.u.AddBullet.ActorUID = -1;
201 GameEventsEnqueue(&gGameEvents, e);
202 }
203
204 SoundPlayAt(&gSoundDevice, StrSound("bang"), realPos);
205
206 // If wreck is available spawn it in the exact same position
207 PlaceWreck(o->Class->Wreck, &o->tileItem);
208
209 ObjDestroy(o);
210
211 // Update pathfinding cache since this object could have blocked a path
212 // before
213 PathCacheClear(&gPathCache);
214}
215static void PlaceWreck(const char *wreckClass, const TTileItem *ti)
216{
217 if (wreckClass == NULL)
218 {
219 return;
220 }
221 GameEvent e = GameEventNew(GAME_EVENT_MAP_OBJECT_ADD);
222 e.u.MapObjectAdd.UID = ObjsGetNextUID();
223 const MapObject *mo = StrMapObject(wreckClass);
224 CASSERT(mo != NULL, "cannot find wreck");
225 if (mo == NULL)
226 {
227 LOG(LM_MAIN, LL_ERROR, "wreck (%s) not found", wreckClass);
228 return;
229 }
230 strcpy(e.u.MapObjectAdd.MapObjectClass, mo->Name);
231 e.u.MapObjectAdd.Pos = Vec2i2Net(Vec2iNew(ti->x, ti->y));
232 e.u.MapObjectAdd.TileItemFlags = MapObjectGetFlags(mo);
233 e.u.MapObjectAdd.Health = mo->Health;
234 GameEventsEnqueue(&gGameEvents, e);
235}

237bool CanHit(const int flags, const int uid, const TTileItem *target)
238{
239 switch (target->kind)
240 {
241 case KIND_CHARACTER:
242 return CanHitCharacter(flags, uid, CArrayGet(&gActors, target->id));
243 case KIND_OBJECT:
244 return true;
245 default:
246 CASSERT(false, "cannot damage tile item kind");
247 break;
248 }
249 return false;
250}
251bool HasHitSound(
252 const int flags, const int playerUID,
253 const TileItemKind targetKind, const int targetUID,
254 const special_damage_e special, const bool allowFriendlyHitSound)
255{
256 switch (targetKind)
257 {
258 case KIND_CHARACTER:
259 {
260 const TActor *a = ActorGetByUID(targetUID);
261 return
262 !ActorIsImmune(a, special) &&
263 (allowFriendlyHitSound || !ActorIsInvulnerable(
264 a, flags, playerUID, gCampaign.Entry.Mode));
265 }
266 case KIND_OBJECT:
267 return true;
268 default:
269 CASSERT(false, "cannot damage tile item kind");
270 break;
271 }
272 return false;
273}

275static void DoDamageCharacter(
276 const Vec2i hitVector,
277 const int power,
278 const double mass,
279 const int flags,
280 const int playerUID,
281 const int uid,
282 TActor *actor,
283 const special_damage_e special);
284void Damage(
285 const Vec2i hitVector,
286 const int power,
287 const double mass,
288 const int flags,
289 const int playerUID,
290 const int uid,
291 const TileItemKind targetKind, const int targetUID,
292 const special_damage_e special)
293{
294 switch (targetKind)
295 {
296 case KIND_CHARACTER:
297 DoDamageCharacter(
298 hitVector,
299 power, mass, flags, playerUID, uid,
300 ActorGetByUID(targetUID), special);
301 break;
302 case KIND_OBJECT:
303 {
304 GameEvent e = GameEventNew(GAME_EVENT_MAP_OBJECT_DAMAGE);
305 e.u.MapObjectDamage.UID = targetUID;
306 e.u.MapObjectDamage.Power = power;
307 e.u.MapObjectDamage.ActorUID = uid;
308 e.u.MapObjectDamage.PlayerUID = playerUID;
309 e.u.MapObjectDamage.Flags = flags;
310 GameEventsEnqueue(&gGameEvents, e);
311 }
312 break;
313 default:
314 CASSERT(false, "cannot damage tile item kind");
315 break;
316 }
317}
318static void DoDamageCharacter(
319 const Vec2i hitVector,
320 const int power,
321 const double mass,
322 const int flags,
323 const int playerUID,
324 const int uid,
325 TActor *actor,
326 const special_damage_e special)
327{
328 // Create events: hit, damage, score
329 CASSERT(actor->isInUse, "Cannot damage nonexistent player");
330 CASSERT(CanHitCharacter(flags, uid, actor), "damaging undamageable actor");
331
332 // Shot pushback, based on mass and velocity
333 const double impulseFactor = mass / SHOT_IMPULSE_DIVISOR;
334 const Vec2i vel = Vec2iNew(
335 (int)Round(hitVector.x * impulseFactor),
336 (int)Round(hitVector.y * impulseFactor));
337 if (!Vec2iIsZero(vel))
338 {
339 GameEvent ei = GameEventNew(GAME_EVENT_ACTOR_IMPULSE);
340 ei.u.ActorImpulse.UID = actor->uid;
341 ei.u.ActorImpulse.Vel = Vec2i2Net(vel);
342 ei.u.ActorImpulse.Pos = Vec2i2Net(actor->Pos);
343 GameEventsEnqueue(&gGameEvents, ei);
344 }
345
346 const bool canDamage =
347 CanDamageCharacter(flags, playerUID, uid, actor, special);
348
349 GameEvent e = GameEventNew(GAME_EVENT_ACTOR_HIT);
350 e.u.ActorHit.UID = actor->uid;
351 e.u.ActorHit.PlayerUID = actor->PlayerUID;
352 e.u.ActorHit.HitterPlayerUID = playerUID;
353 e.u.ActorHit.Special = special;
354 e.u.ActorHit.Power = canDamage ? power : 0;
355 e.u.ActorHit.Mass = (float)mass;
356 e.u.ActorHit.Vel = Vec2i2Net(hitVector);
357 GameEventsEnqueue(&gGameEvents, e);
358
359 if (canDamage)
360 {
361 // Don't score for friendly or player hits
362 const bool isFriendly =
363 (actor->flags & FLAGS_GOOD_GUY) ||
364 (!IsPVP(gCampaign.Entry.Mode) && actor->PlayerUID >= 0);
365 if (playerUID >= 0 && power != 0 && !isFriendly)
366 {
367 // Calculate score based on
368 // if they hit a penalty character
369 e = GameEventNew(GAME_EVENT_SCORE);
370 e.u.Score.PlayerUID = playerUID;
371 if (actor->flags & FLAGS_PENALTY)
372 {
373 e.u.Score.Score = PENALTY_MULTIPLIER * power;
374 }
375 else
376 {
377 e.u.Score.Score = power;
378 }
379 GameEventsEnqueue(&gGameEvents, e);
380 }
381 }

385void UpdateMobileObjects(int ticks)
386{
387 CA_FOREACH(TMobileObject, obj, gMobObjs)
388 if (!obj->isInUse)
389 {
390 continue;
391 }
392 if (!obj->updateFunc(obj, ticks) && !gCampaign.IsClient)
393 {
394 GameEvent e = GameEventNew(GAME_EVENT_REMOVE_BULLET);
395 e.u.RemoveBullet.UID = obj->UID;
396 GameEventsEnqueue(&gGameEvents, e);
397 continue;
398 }
399 CA_FOREACH_END()
400}

424void ObjAdd(const NMapObjectAdd amo)
425{
426 // Don't add if UID exists
427 if (ObjGetByUID(amo.UID) != NULL)
428 {
429 LOG(LM_MAIN, LL_DEBUG,
430 "object uid(%d) already exists; not adding", (int)amo.UID);
431 return;
432 }
433 // Find an empty slot in object list
434 TObject *o = NULL;
435 int i;
436 for (i = 0; i < (int)gObjs.size; i++)
437 {
438 TObject *obj = CArrayGet(&gObjs, i);
439 if (!obj->isInUse)
440 {
441 o = obj;
442 break;
443 }
444 }
445 if (o == NULL)
446 {
447 TObject obj;
448 memset(&obj, 0, sizeof obj);
449 CArrayPushBack(&gObjs, &obj);
450 i = (int)gObjs.size - 1;
451 o = CArrayGet(&gObjs, i);
452 }
453 memset(o, 0, sizeof *o);
454 o->uid = amo.UID;
455 o->Class = StrMapObject(amo.MapObjectClass);
456 o->Health = amo.Health;
457 o->tileItem.x = o->tileItem.y = -1;
458 o->tileItem.flags = amo.TileItemFlags;
459 o->tileItem.kind = KIND_OBJECT;
460 o->tileItem.getPicFunc = NULL;
461 o->tileItem.CPic = o->Class->Pic;
462 o->tileItem.CPicFunc = GetMapObjectDrawContext;
463 o->tileItem.size = o->Class->Size;
464 o->tileItem.id = i;
465 MapTryMoveTileItem(&gMap, &o->tileItem, Net2Vec2i(amo.Pos));
466 o->isInUse = true;
467 LOG(LM_MAIN, LL_DEBUG,
468 "added object uid(%d) class(%s) health(%d) pos(%d, %d)",
469 (int)amo.UID, amo.MapObjectClass, amo.Health, amo.Pos.x, amo.Pos.y);
470
471 // Update pathfinding cache since this object could block a path
472 PathCacheClear(&gPathCache);
473}

488void UpdateObjects(const int ticks)
489{
490 CA_FOREACH(TObject, obj, gObjs)
491 if (!obj->isInUse)
492 {
493 continue;
494 }
495 TileItemUpdate(&obj->tileItem, ticks);
496 switch (obj->Class->Type)
497 {
498 case MAP_OBJECT_TYPE_PICKUP_SPAWNER:
499 if (gCampaign.IsClient) break;
500 // If counter -1, it is inactive i.e. already spawned pickup
501 if (obj->counter == -1)
502 {
503 break;
504 }
505 obj->counter -= ticks;
506 if (obj->counter <= 0)
507 {
508 // Deactivate spawner by setting counter to -1
509 // Spawner reactivated only when ammo taken
510 obj->counter = -1;
511 GameEvent e = GameEventNew(GAME_EVENT_ADD_PICKUP);
512 e.u.AddPickup.UID = PickupsGetNextUID();
513 strcpy(
514 e.u.AddPickup.PickupClass,
515 obj->Class->u.PickupClass->Name);
516 e.u.AddPickup.IsRandomSpawned = false;
517 e.u.AddPickup.SpawnerUID = obj->uid;
518 e.u.AddPickup.TileItemFlags = 0;
519 e.u.AddPickup.Pos =
520 Vec2i2Net(Vec2iNew(obj->tileItem.x, obj->tileItem.y));
521 GameEventsEnqueue(&gGameEvents, e);
522 }
523 break;
524 default:
525 // Do nothing
526 break;
527 }
528 CA_FOREACH_END()
529}
