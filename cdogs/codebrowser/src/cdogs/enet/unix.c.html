108int
109enet_address_set_host_ip (ENetAddress * address, const char * name)
110{
111#ifdef HAS_INET_PTON
112 if (! inet_pton (AF_INET, name, & address -> host))
113#else
114 if (! inet_aton (name, (struct in_addr *) & address -> host))
115#endif
116 return -1;
117
118 return 0;
119}

121int
122enet_address_set_host (ENetAddress * address, const char * name)
123{
124#ifdef HAS_GETADDRINFO
125 struct addrinfo hints, * resultList = NULL, * result = NULL;
126
127 memset (& hints, 0, sizeof (hints));
128 hints.ai_family = AF_INET;
129
130 if (getaddrinfo (name, NULL, NULL, & resultList) != 0)
131 return -1;
132
133 for (result = resultList; result != NULL; result = result -> ai_next)
134 {
135 if (result -> ai_family == AF_INET && result -> ai_addr != NULL && result -> ai_addrlen >= sizeof (struct sockaddr_in))
136 {
137 struct sockaddr_in * sin = (struct sockaddr_in *) result -> ai_addr;
138
139 address -> host = sin -> sin_addr.s_addr;
140
141 freeaddrinfo (resultList);
142
143 return 0;
144 }
145 }
146
147 if (resultList != NULL)
148 freeaddrinfo (resultList);
149#else
150 struct hostent * hostEntry = NULL;
151#ifdef HAS_GETHOSTBYNAME_R
152 struct hostent hostData;
153 char buffer [2048];
154 int errnum;
155
156#if defined(linux) || defined(__linux) || defined(__linux__) || defined(__FreeBSD__) || defined(__FreeBSD_kernel__) || defined(__DragonFly__)
157 gethostbyname_r (name, & hostData, buffer, sizeof (buffer), & hostEntry, & errnum);
158#else
159 hostEntry = gethostbyname_r (name, & hostData, buffer, sizeof (buffer), & errnum);
160#endif
161#else
162 hostEntry = gethostbyname (name);
163#endif
164
165 if (hostEntry != NULL && hostEntry -> h_addrtype == AF_INET)
166 {
167 address -> host = * (enet_uint32 *) hostEntry -> h_addr_list [0];
168
169 return 0;
170 }
171#endif
172
173 return enet_address_set_host_ip (address, name);
174}

176int
177enet_address_get_host_ip (const ENetAddress * address, char * name, size_t nameLength)
178{
179#ifdef HAS_INET_NTOP
180 if (inet_ntop (AF_INET, & address -> host, name, nameLength) == NULL)
181#else
182 char * addr = inet_ntoa (* (struct in_addr *) & address -> host);
183 if (addr != NULL)
184 {
185 size_t addrLen = strlen(addr);
186 if (addrLen >= nameLength)
187 return -1;
188 memcpy (name, addr, addrLen + 1);
189 }
190 else
191#endif
192 return -1;
193 return 0;
194}

196int
197enet_address_get_host (const ENetAddress * address, char * name, size_t nameLength)
198{
199#ifdef HAS_GETNAMEINFO
200 struct sockaddr_in sin;
201 int err;
202
203 memset (& sin, 0, sizeof (struct sockaddr_in));
204
205 sin.sin_family = AF_INET;
206 sin.sin_port = ENET_HOST_TO_NET_16 (address -> port);
207 sin.sin_addr.s_addr = address -> host;
208
209 err = getnameinfo ((struct sockaddr *) & sin, sizeof (sin), name, nameLength, NULL, 0, NI_NAMEREQD);
210 if (! err)
211 {
212 if (name != NULL && nameLength > 0 && ! memchr (name, '\0', nameLength))
213 return -1;
214 return 0;
215 }
216 if (err != EAI_NONAME)
217 return -1;
218#else
219 struct in_addr in;
220 struct hostent * hostEntry = NULL;
221#ifdef HAS_GETHOSTBYADDR_R
222 struct hostent hostData;
223 char buffer [2048];
224 int errnum;
225
226 in.s_addr = address -> host;
227
228#if defined(linux) || defined(__linux) || defined(__linux__) || defined(__FreeBSD__) || defined(__FreeBSD_kernel__) || defined(__DragonFly__)
229 gethostbyaddr_r ((char *) & in, sizeof (struct in_addr), AF_INET, & hostData, buffer, sizeof (buffer), & hostEntry, & errnum);
230#else
231 hostEntry = gethostbyaddr_r ((char *) & in, sizeof (struct in_addr), AF_INET, & hostData, buffer, sizeof (buffer), & errnum);
232#endif
233#else
234 in.s_addr = address -> host;
235
236 hostEntry = gethostbyaddr ((char *) & in, sizeof (struct in_addr), AF_INET);
237#endif
238
239 if (hostEntry != NULL)
240 {
241 size_t hostLen = strlen (hostEntry -> h_name);
242 if (hostLen >= nameLength)
243 return -1;
244 memcpy (name, hostEntry -> h_name, hostLen + 1);
245 return 0;
246 }
247#endif
248
249 return enet_address_get_host_ip (address, name, nameLength);
250}

252int
253enet_socket_bind (ENetSocket socket, const ENetAddress * address)
254{
255 struct sockaddr_in sin;
256
257 memset (& sin, 0, sizeof (struct sockaddr_in));
258
259 sin.sin_family = AF_INET;
260
261 if (address != NULL)
262 {
263 sin.sin_port = ENET_HOST_TO_NET_16 (address -> port);
264 sin.sin_addr.s_addr = address -> host;
265 }
266 else
267 {
268 sin.sin_port = 0;
269 sin.sin_addr.s_addr = INADDR_ANY;
270 }
271
272 return bind (socket,
273 (struct sockaddr *) & sin,
274 sizeof (struct sockaddr_in));
275}

304int
305enet_socket_set_option (ENetSocket socket, ENetSocketOption option, int value)
306{
307 int result = -1;
308 switch (option)
309 {
310 case ENET_SOCKOPT_NONBLOCK:
311#ifdef HAS_FCNTL
312 result = fcntl (socket, F_SETFL, (value ? O_NONBLOCK : 0) | (fcntl (socket, F_GETFL) & ~O_NONBLOCK));
313#else
314 result = ioctl (socket, FIONBIO, & value);
315#endif
316 break;
317
318 case ENET_SOCKOPT_BROADCAST:
319 result = setsockopt (socket, SOL_SOCKET, SO_BROADCAST, (char *) & value, sizeof (int));
320 break;
321
322 case ENET_SOCKOPT_REUSEADDR:
323 result = setsockopt (socket, SOL_SOCKET, SO_REUSEADDR, (char *) & value, sizeof (int));
324 break;
325
326 case ENET_SOCKOPT_RCVBUF:
327 result = setsockopt (socket, SOL_SOCKET, SO_RCVBUF, (char *) & value, sizeof (int));
328 break;
329
330 case ENET_SOCKOPT_SNDBUF:
331 result = setsockopt (socket, SOL_SOCKET, SO_SNDBUF, (char *) & value, sizeof (int));
332 break;
333
334 case ENET_SOCKOPT_RCVTIMEO:
335 {
336 struct timeval timeVal;
337 timeVal.tv_sec = value / 1000;
338 timeVal.tv_usec = (value % 1000) * 1000;
339 result = setsockopt (socket, SOL_SOCKET, SO_RCVTIMEO, (char *) & timeVal, sizeof (struct timeval));
340 break;
341 }
342
343 case ENET_SOCKOPT_SNDTIMEO:
344 {
345 struct timeval timeVal;
346 timeVal.tv_sec = value / 1000;
347 timeVal.tv_usec = (value % 1000) * 1000;
348 result = setsockopt (socket, SOL_SOCKET, SO_SNDTIMEO, (char *) & timeVal, sizeof (struct timeval));
349 break;
350 }
351
352 case ENET_SOCKOPT_NODELAY:
353 result = setsockopt (socket, IPPROTO_TCP, TCP_NODELAY, (char *) & value, sizeof (int));
354 break;
355
356 default:
357 break;
358 }
359 return result == -1 ? -1 : 0;
360}

362int
363enet_socket_get_option (ENetSocket socket, ENetSocketOption option, int * value)
364{
365 int result = -1;
366 socklen_t len;
367 switch (option)
368 {
369 case ENET_SOCKOPT_ERROR:
370 len = sizeof (int);
371 result = getsockopt (socket, SOL_SOCKET, SO_ERROR, value, & len);
372 break;
373
374 default:
375 break;
376 }
377 return result == -1 ? -1 : 0;
378}

380int
381enet_socket_connect (ENetSocket socket, const ENetAddress * address)
382{
383 struct sockaddr_in sin;
384 int result;
385
386 memset (& sin, 0, sizeof (struct sockaddr_in));
387
388 sin.sin_family = AF_INET;
389 sin.sin_port = ENET_HOST_TO_NET_16 (address -> port);
390 sin.sin_addr.s_addr = address -> host;
391
392 result = connect (socket, (struct sockaddr *) & sin, sizeof (struct sockaddr_in));
393 if (result == -1 && errno == EINPROGRESS)
394 return 0;
395
396 return result;
397}

400enet_socket_accept (ENetSocket socket, ENetAddress * address)
401{
402 int result;
403 struct sockaddr_in sin;
404 socklen_t sinLength = sizeof (struct sockaddr_in);
405
406 result = accept (socket,
407 address != NULL ? (struct sockaddr *) & sin : NULL,
408 address != NULL ? & sinLength : NULL);
409
410 if (result == -1)
411 return ENET_SOCKET_NULL;
412
413 if (address != NULL)
414 {
415 address -> host = (enet_uint32) sin.sin_addr.s_addr;
416 address -> port = ENET_NET_TO_HOST_16 (sin.sin_port);
417 }
418
419 return result;

435int
436enet_socket_send (ENetSocket socket,
437 const ENetAddress * address,
438 const ENetBuffer * buffers,
439 size_t bufferCount)
440{
441 struct msghdr msgHdr;
442 struct sockaddr_in sin;
443 int sentLength;
444
445 memset (& msgHdr, 0, sizeof (struct msghdr));
446
447 if (address != NULL)
448 {
449 memset (& sin, 0, sizeof (struct sockaddr_in));
450
451 sin.sin_family = AF_INET;
452 sin.sin_port = ENET_HOST_TO_NET_16 (address -> port);
453 sin.sin_addr.s_addr = address -> host;
454
455 msgHdr.msg_name = & sin;
456 msgHdr.msg_namelen = sizeof (struct sockaddr_in);
457 }
458
459 msgHdr.msg_iov = (struct iovec *) buffers;
460 msgHdr.msg_iovlen = bufferCount;
461
462 sentLength = sendmsg (socket, & msgHdr, MSG_NOSIGNAL);
463
464 if (sentLength == -1)
465 {
466 if (errno == EWOULDBLOCK)
467 return 0;
468
469 return -1;
470 }
471
472 return sentLength;
473}

475int
476enet_socket_receive (ENetSocket socket,
477 ENetAddress * address,
478 ENetBuffer * buffers,
479 size_t bufferCount)
480{
481 struct msghdr msgHdr;
482 struct sockaddr_in sin;
483 int recvLength;
484
485 memset (& msgHdr, 0, sizeof (struct msghdr));
486
487 if (address != NULL)
488 {
489 msgHdr.msg_name = & sin;
490 msgHdr.msg_namelen = sizeof (struct sockaddr_in);
491 }
492
493 msgHdr.msg_iov = (struct iovec *) buffers;
494 msgHdr.msg_iovlen = bufferCount;
495
496 recvLength = recvmsg (socket, & msgHdr, MSG_NOSIGNAL);
497
498 if (recvLength == -1)
499 {
500 if (errno == EWOULDBLOCK)
501 return 0;
502
503 return -1;
504 }
505
506#ifdef HAS_MSGHDR_FLAGS
507 if (msgHdr.msg_flags & MSG_TRUNC)
508 return -1;
509#endif
510
511 if (address != NULL)
512 {
513 address -> host = (enet_uint32) sin.sin_addr.s_addr;
514 address -> port = ENET_NET_TO_HOST_16 (sin.sin_port);
515 }
516
517 return recvLength;
518}

531int
532enet_socket_wait (ENetSocket socket, enet_uint32 * condition, enet_uint32 timeout)
533{
534#ifdef HAS_POLL
535 struct pollfd pollSocket;
536 int pollCount;
537
538 pollSocket.fd = socket;
539 pollSocket.events = 0;
540
541 if (* condition & ENET_SOCKET_WAIT_SEND)
542 pollSocket.events |= POLLOUT;
543
544 if (* condition & ENET_SOCKET_WAIT_RECEIVE)
545 pollSocket.events |= POLLIN;
546
547 pollCount = poll (& pollSocket, 1, timeout);
548
549 if (pollCount < 0)
550 {
551 if (errno == EINTR && * condition & ENET_SOCKET_WAIT_INTERRUPT)
552 {
553 * condition = ENET_SOCKET_WAIT_INTERRUPT;
554
555 return 0;
556 }
557
558 return -1;
559 }
560
561 * condition = ENET_SOCKET_WAIT_NONE;
562
563 if (pollCount == 0)
564 return 0;
565
566 if (pollSocket.revents & POLLOUT)
567 * condition |= ENET_SOCKET_WAIT_SEND;
568
569 if (pollSocket.revents & POLLIN)
570 * condition |= ENET_SOCKET_WAIT_RECEIVE;
571
572 return 0;
573#else
574 fd_set readSet, writeSet;
575 struct timeval timeVal;
576 int selectCount;
577
578 timeVal.tv_sec = timeout / 1000;
579 timeVal.tv_usec = (timeout % 1000) * 1000;
580
581 FD_ZERO (& readSet);
582 FD_ZERO (& writeSet);
583
584 if (* condition & ENET_SOCKET_WAIT_SEND)
585 FD_SET (socket, & writeSet);
586
587 if (* condition & ENET_SOCKET_WAIT_RECEIVE)
588 FD_SET (socket, & readSet);
589
590 selectCount = select (socket + 1, & readSet, & writeSet, NULL, & timeVal);
591
592 if (selectCount < 0)
593 {
594 if (errno == EINTR && * condition & ENET_SOCKET_WAIT_INTERRUPT)
595 {
596 * condition = ENET_SOCKET_WAIT_INTERRUPT;
597
598 return 0;
599 }
600
601 return -1;
602 }
603
604 * condition = ENET_SOCKET_WAIT_NONE;
605
606 if (selectCount == 0)
607 return 0;
608
609 if (FD_ISSET (socket, & writeSet))
610 * condition |= ENET_SOCKET_WAIT_SEND;
611
612 if (FD_ISSET (socket, & readSet))
613 * condition |= ENET_SOCKET_WAIT_RECEIVE;
614
615 return 0;
616#endif
617}
