28#pragma once
29
30#include "AStar.h"
31#include "c_array.h"
32#include "map.h"
33#include "vector.h"
34
35// Ref-counted path reference
36// Once refs reaches zero, can then free the path
37typedef struct
38{
39 ASPath Path;
40 int *refs;
41 Vec2i from;
42 Vec2i to;
43} CachedPath;
44
45typedef struct
46{
47 CArray paths; // of CachedPath
48 size_t head;
49 Map *map;
50} PathCache;
51
52// Cache of A* paths so similar paths don't need to be recalculated
53// Mainly to work around AI repeating the same pathfinds rapidly
54// Note: lifetime managed by Map
55extern PathCache gPathCache;
56
57void CachedPathDestroy(CachedPath *c);
58
59void PathCacheInit(PathCache *pc, Map *m);
60void PathCacheTerminate(PathCache *pc);
61
62// Clear all entries in cache
63// This is done when the underlying map changes, changing paths
64// e.g. keys
65void PathCacheClear(PathCache *pc);
66
67CachedPath PathCacheCreate(
68 PathCache *pc, Vec2i from, Vec2i to,
69 const bool ignoreObjects, const bool cache);
