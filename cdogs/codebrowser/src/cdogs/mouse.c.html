38#define MOUSE_REPEAT_TICKS 150
39#define MOUSE_MOVE_DEAD_ZONE 12
40#define TRAIL_NUM_DOTS 4

43void MouseInit(Mouse *mouse, Pic *cursor, Pic *trail, const bool hideMouse)
44{
45 memset(mouse, 0, sizeof *mouse);
46 mouse->cursor = cursor;
47 mouse->trail = trail;
48 mouse->ticks = 0;
49 mouse->repeatedTicks = 0;
50 mouse->hideMouse = cursor != NULL || hideMouse;
51 if (hideMouse)
52 {
53 SDL_ShowCursor(SDL_DISABLE);
54 }
55}

57void MousePrePoll(Mouse *mouse)
58{
59 memset(mouse->pressedButtons, 0, sizeof mouse->pressedButtons);
60 memcpy(
61 mouse->previousButtons,
62 mouse->currentButtons,
63 sizeof mouse->previousButtons);
64 mouse->wheel = Vec2iZero();
65 mouse->previousPos = mouse->currentPos;
66 SDL_GetMouseState(&mouse->currentPos.x, &mouse->currentPos.y);
67 int scale = ConfigGetInt(&gConfig, "Graphics.ScaleFactor");
68 if (scale == 0) scale = 1;
69 mouse->currentPos = Vec2iScaleDiv(mouse->currentPos, scale);
70}

73void MouseOnButtonDown(Mouse *mouse, Uint8 button)
74{
75 mouse->currentButtons[button] = 1;
76}
77void MouseOnButtonUp(Mouse *mouse, Uint8 button)
78{
79 // Certain mouse buttons can be pressed very quickly (e.g. mousewheel)
80 // Detect these mouse presses before PostPoll
81 if (mouse->currentButtons[button] && !mouse->previousButtons[button])
82 {
83 mouse->pressedButtons[button] = 1;
84 }
85 mouse->currentButtons[button] = 0;
86}
87void MouseOnWheel(Mouse *m, const Sint32 x, const Sint32 y)
88{
89 m->wheel = Vec2iNew(x, y);
90}

92void MousePostPoll(Mouse *mouse, Uint32 ticks)
93{
94 int areSameButtonsPressed = 1;
95 int i;
96 for (i = 0; i < 8; i++)
97 {
98 if (mouse->previousButtons[i] ^ mouse->currentButtons[i])
99 {
100 areSameButtonsPressed = 0;
101 break;
102 }
103 }
104 // If same buttons have been pressed, remember how long they have been pressed
105 if (areSameButtonsPressed)
106 {
107 Uint32 ticksElapsed = ticks - mouse->ticks;
108 mouse->repeatedTicks += ticksElapsed;
109 }
110 else
111 {
112 mouse->repeatedTicks = 0;
113 }
114 // If more time has elapsed, forget about previous buttons for repeating
115 if (mouse->repeatedTicks > MOUSE_REPEAT_TICKS)
116 {
117 mouse->repeatedTicks -= MOUSE_REPEAT_TICKS;
118 for (i = 0; i < 8; i++)
119 {
120 mouse->pressedButtons[i] |= mouse->currentButtons[i];
121 }
122 memcpy(
123 mouse->pressedButtons,
124 mouse->currentButtons,
125 sizeof mouse->pressedButtons);
126 }
127 else
128 {
129 for (i = 0; i < 8; i++)
130 {
131 mouse->pressedButtons[i] |=
132 mouse->currentButtons[i] && !mouse->previousButtons[i];
133 }
134 }
135 mouse->ticks = ticks;
136}

138bool MouseHasMoved(const Mouse *m)
139{
140 return !Vec2iEqual(m->previousPos, m->currentPos);
141}

143int MouseGetPressed(const Mouse *m)
144{
145 for (int i = 0; i < 8; i++)
146 {
147 if (m->pressedButtons[i])
148 {
149 return i;
150 }
151 }
152 return 0;
153}

170int MouseGetMove(Mouse *mouse, const Vec2i pos)
171{
172 int cmd = 0;
173 const int dx = abs(mouse->currentPos.x - pos.x);
174 const int dy = abs(mouse->currentPos.y - pos.y);
175 mouse->mouseMovePos = pos;
176 // HACK: player is typically drawn slightly to the right
177 mouse->mouseMovePos.x += 8;
178 const bool isInDeadZone =
179 dx <= MOUSE_MOVE_DEAD_ZONE && dy <= MOUSE_MOVE_DEAD_ZONE;
180 if (!isInDeadZone)
181 {
182 if (2 * dx > dy)
183 {
184 if (pos.x < mouse->currentPos.x) cmd |= CMD_RIGHT;
185 else if (pos.x > mouse->currentPos.x) cmd |= CMD_LEFT;
186 }
187 if (2 * dy > dx)
188 {
189 if (pos.y < mouse->currentPos.y) cmd |= CMD_DOWN;
190 else if (pos.y > mouse->currentPos.y) cmd |= CMD_UP;
191 }
192 }
193 return cmd;
194}

196void MouseDraw(const Mouse *mouse)
197{
198 if (mouse->cursor)
199 {
200 BlitMasked(
201 &gGraphicsDevice, mouse->cursor, mouse->currentPos, colorWhite, 1);
202 }
203 if (mouse->trail)
204 {
205 const int dx = abs(mouse->currentPos.x - mouse->mouseMovePos.x);
206 const int dy = abs(mouse->currentPos.y - mouse->mouseMovePos.y);
207 const bool isInDeadZone =
208 dx <= MOUSE_MOVE_DEAD_ZONE && dy <= MOUSE_MOVE_DEAD_ZONE;
209 if (!isInDeadZone)
210 {
211 // Draw a trail between the mouse move pos and mouse pos
212 // The trail is made up of a fixed number of dots
213 const Vec2i d =
214 Vec2iMinus(mouse->currentPos, mouse->mouseMovePos);
215 for (int i = 1; i <= TRAIL_NUM_DOTS; i++)
216 {
217 const Vec2i pos = Vec2iAdd(
218 mouse->mouseMovePos,
219 Vec2iScaleDiv(Vec2iScale(d, i), TRAIL_NUM_DOTS + 1));
220 BlitMasked(
221 &gGraphicsDevice, mouse->trail, pos, colorWhite, false);
222 }
223 }
224 }
