42void PickupsInit(void)
43{
44 CArrayInit(&gPickups, sizeof(Pickup));
45 CArrayReserve(&gPickups, 128);
46 sPickupUIDs = 0;
47}
48void PickupsTerminate(void)
49{
50 CA_FOREACH(const Pickup, p, gPickups)
51 if (p->isInUse)
52 {
53 PickupDestroy(p->UID);
54 }
55 CA_FOREACH_END()
56 CArrayTerminate(&gPickups);
57}
58int PickupsGetNextUID(void)
59{
60 return sPickupUIDs++;

62static const Pic *GetPickupPic(const int id, Vec2i *offset);
63void PickupAdd(const NAddPickup ap)
64{
65 // Check if existing pickup
66 Pickup *p = PickupGetByUID(ap.UID);
67 if (p != NULL && p->isInUse)
68 {
69 PickupDestroy(ap.UID);
70 }
71 // Find an empty slot in pickup list
72 p = NULL;
73 int i;
74 for (i = 0; i < (int)gPickups.size; i++)
75 {
76 Pickup *pu = CArrayGet(&gPickups, i);
77 if (!pu->isInUse)
78 {
79 p = pu;
80 break;
81 }
82 }
83 if (p == NULL)
84 {
85 Pickup pu;
86 memset(&pu, 0, sizeof pu);
87 CArrayPushBack(&gPickups, &pu);
88 i = (int)gPickups.size - 1;
89 p = CArrayGet(&gPickups, i);
90 }
91 memset(p, 0, sizeof *p);
92 p->UID = ap.UID;
93 p->class = StrPickupClass(ap.PickupClass);
94 p->tileItem.x = p->tileItem.y = -1;
95 p->tileItem.flags = ap.TileItemFlags;
96 p->tileItem.kind = KIND_PICKUP;
97 p->tileItem.getPicFunc = GetPickupPic;
98 p->tileItem.size = p->class->Pic->size;
99 p->tileItem.id = i;
100 MapTryMoveTileItem(&gMap, &p->tileItem, Net2Vec2i(ap.Pos));
101 p->IsRandomSpawned = ap.IsRandomSpawned;
102 p->PickedUp = false;
103 p->SpawnerUID = ap.SpawnerUID;
104 p->isInUse = true;
105}
106void PickupDestroy(const int uid)
107{
108 Pickup *p = PickupGetByUID(uid);
109 CASSERT(p->isInUse, "Destroying not-in-use pickup");
110 MapRemoveTileItem(&gMap, &p->tileItem);
111 p->isInUse = false;
112}

114static bool TryPickupGun(
115 TActor *a, const Pickup *p, const bool pickupAll, const char **sound);
116void PickupPickup(TActor *a, Pickup *p, const bool pickupAll)
117{
118 if (p->PickedUp) return;
119 CASSERT(a->PlayerUID >= 0, "NPCs cannot pickup");
120 bool canPickup = true;
121 const char *sound = NULL;
122 const Vec2i actorPos = Vec2iNew(a->tileItem.x, a->tileItem.y);
123 switch (p->class->Type)
124 {
125 case PICKUP_JEWEL:
126 {
127 GameEvent e = GameEventNew(GAME_EVENT_SCORE);
128 e.u.Score.PlayerUID = a->PlayerUID;
129 e.u.Score.Score = p->class->u.Score;
130 GameEventsEnqueue(&gGameEvents, e);
131 sound = "pickup";
132 UpdateMissionObjective(
133 &gMission, p->tileItem.flags, OBJECTIVE_COLLECT, 1);
134 }
135 break;
136
137 case PICKUP_HEALTH:
138 // Don't pick up unless taken damage
139 canPickup = false;
140 if (a->health < ActorGetCharacter(a)->maxHealth)
141 {
142 canPickup = true;
143 GameEvent e = GameEventNew(GAME_EVENT_ACTOR_HEAL);
144 e.u.Heal.UID = a->uid;
145 e.u.Heal.PlayerUID = a->PlayerUID;
146 e.u.Heal.Amount = p->class->u.Health;
147 e.u.Heal.IsRandomSpawned = p->IsRandomSpawned;
148 GameEventsEnqueue(&gGameEvents, e);
149 }
150 break;
151
152 case PICKUP_AMMO:
153 {
154 // Don't pickup if no guns can use ammo
155 bool hasGunUsingAmmo = false;
156 CA_FOREACH(const Weapon, w, a->guns)
157 if (w->Gun->AmmoId == p->class->u.Ammo.Id)
158 {
159 hasGunUsingAmmo = true;
160 break;
161 }
162 CA_FOREACH_END()
163 if (!hasGunUsingAmmo)
164 {
165 canPickup = false;
166 break;
167 }
168
169 // Don't pickup if ammo full
170 const Ammo *ammo = AmmoGetById(&gAmmo, p->class->u.Ammo.Id);
171 const int current = *(int *)CArrayGet(&a->ammo, p->class->u.Ammo.Id);
172 if (current >= ammo->Max)
173 {
174 canPickup = false;
175 break;
176 }
177
178 // Take ammo
179 GameEvent e = GameEventNew(GAME_EVENT_ACTOR_ADD_AMMO);
180 e.u.AddAmmo.UID = a->uid;
181 e.u.AddAmmo.PlayerUID = a->PlayerUID;
182 e.u.AddAmmo.AmmoId = p->class->u.Ammo.Id;
183 e.u.AddAmmo.Amount = p->class->u.Ammo.Amount;
184 e.u.AddAmmo.IsRandomSpawned = p->IsRandomSpawned;
185 // Note: receiving end will prevent ammo from exceeding max
186 GameEventsEnqueue(&gGameEvents, e);
187
188 sound = ammo->Sound;
189 }
190 break;
191
192 case PICKUP_KEYCARD:
193 {
194 GameEvent e = GameEventNew(GAME_EVENT_ADD_KEYS);
195 e.u.AddKeys.KeyFlags = p->class->u.Keys;
196 e.u.AddKeys.Pos = Vec2i2Net(actorPos);
197 GameEventsEnqueue(&gGameEvents, e);
198 }
199 break;
200
201 case PICKUP_GUN:
202 canPickup = TryPickupGun(a, p, pickupAll, &sound);
203 break;
204
205 default:
206 CASSERT(false, "unexpected pickup type");
207 break;
208 }
209 if (canPickup)
210 {
211 if (sound != NULL)
212 {
213 GameEvent es = GameEventNew(GAME_EVENT_SOUND_AT);
214 strcpy(es.u.SoundAt.Sound, sound);
215 es.u.SoundAt.Pos = Vec2i2Net(actorPos);
216 es.u.SoundAt.IsHit = false;
217 GameEventsEnqueue(&gGameEvents, es);
218 }
219 GameEvent e = GameEventNew(GAME_EVENT_REMOVE_PICKUP);
220 e.u.RemovePickup.UID = p->UID;
221 e.u.RemovePickup.SpawnerUID = p->SpawnerUID;
222 GameEventsEnqueue(&gGameEvents, e);
223 // Prevent multiple pickups by marking
224 p->PickedUp = true;
225 }
226}
227static bool TryPickupGun(
228 TActor *a, const Pickup *p, const bool pickupAll, const char **sound)
229{
230 // Guns can only be picked up manually
231 if (!pickupAll)
232 {
233 return false;
234 }
235 const GunDescription *gun = IdGunDescription(p->class->u.GunId);
236
237 // Pickup if:
238 // - Actor doesn't have gun
239 // - Actor has less ammo than default for the gun's ammo
240
241 int ammoDeficit = 0;
242 const Ammo *ammo = NULL;
243 const int ammoId = gun->AmmoId;
244 if (ammoId >= 0)
245 {
246 ammo = AmmoGetById(&gAmmo, ammoId);
247 ammoDeficit = ammo->Amount * 2 - *(int *)CArrayGet(&a->ammo, ammoId);
248 }
249
250 const bool pickup = !ActorHasGun(a, gun) || ammoDeficit > 0;
251 if (!pickup)
252 {
253 return false;
254 }
255
256 // Pickup gun
257 GameEvent e = GameEventNew(GAME_EVENT_ACTOR_REPLACE_GUN);
258 e.u.ActorReplaceGun.UID = a->uid;
259 e.u.ActorReplaceGun.GunIdx =
260 a->guns.size == MAX_WEAPONS ? a->gunIndex : (int)a->guns.size;
261 CASSERT(e.u.ActorReplaceGun.GunIdx <= a->guns.size,
262 "invalid replace gun index");
263 strcpy(e.u.ActorReplaceGun.Gun, gun->name);
264 GameEventsEnqueue(&gGameEvents, e);
265
266 // If the player has less ammo than the default amount,
267 // replenish up to this amount
268 if (ammoDeficit > 0)
269 {
270 e = GameEventNew(GAME_EVENT_ACTOR_ADD_AMMO);
271 e.u.AddAmmo.UID = a->uid;
272 e.u.AddAmmo.PlayerUID = a->PlayerUID;
273 e.u.AddAmmo.AmmoId = ammoId;
274 e.u.AddAmmo.Amount = ammoDeficit;
275 e.u.AddAmmo.IsRandomSpawned = false;
276 GameEventsEnqueue(&gGameEvents, e);
277
278 // Also play an ammo pickup sound
279 *sound = ammo->Sound;
280 }
281
282 return true;
283}
284
285bool PickupIsManual(const Pickup *p)
286{
287 if (p->PickedUp) return false;
288 switch (p->class->Type)
289 {
290 case PICKUP_GUN:
291 return true;
292 default:
293 return false;
294 }
295}

297static const Pic *GetPickupPic(const int id, Vec2i *offset)
298{
299 const Pickup *p = CArrayGet(&gPickups, id);
300 *offset = Vec2iScaleDiv(p->class->Pic->size, -2);
301 return p->class->Pic;
302}

304Pickup *PickupGetByUID(const int uid)
305{
306 CA_FOREACH(Pickup, p, gPickups)
307 if (p->UID == uid)
308 {
309 return p;
310 }
311 CA_FOREACH_END()
312 return NULL;
