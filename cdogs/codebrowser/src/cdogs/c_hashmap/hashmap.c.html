11#define INITIAL_SIZE (256)
12#define MAX_CHAIN_LENGTH (8)
13
14/* We need to keep keys and values */
15typedef struct _hashmap_element{
16 char* key;
17 bool in_use;
18 any_t data;
19} hashmap_element;
20
21/* A hashmap has some maximum size and current size,
22 * as well as the data to hold. */
23struct hashmap_map{
24 int table_size;
25 int size;
26 hashmap_element *data;
27};
28
29/*
30 * Return an empty hashmap, or NULL on failure.
31 */
32map_t hashmap_new(void) {
33 map_t m = malloc(sizeof(struct hashmap_map));
34 if(!m) goto err;
35
36 m->data = (hashmap_element*) calloc(INITIAL_SIZE, sizeof(hashmap_element));
37 if(!m->data) goto err;
38
39 m->table_size = INITIAL_SIZE;
40 m->size = 0;
41
42 return m;
43 err:
44 if (m)
45 hashmap_free(m);
46 return NULL;
47}

57 /* First, the polynomial itself and its table of feedback terms. The */
58 /* polynomial is */
59 /* X^32+X^26+X^23+X^22+X^16+X^12+X^11+X^10+X^8+X^7+X^5+X^4+X^2+X^1+X^0 */
60 /* */
61 /* Note that we take it "backwards" and put the highest-order term in */
62 /* the lowest-order bit. The X^32 term is "implied"; the LSB is the */
63 /* X^31 term, etc. The X^0 term (usually shown as "+1") results in */
64 /* the MSB being 1. */
65 /* */
66 /* Note that the usual hardware shift register implementation, which */
67 /* is what we're using (we're merely optimizing it by doing eight-bit */
68 /* chunks at a time) shifts bits into the lowest-order term. In our */
69 /* implementation, that means shifting towards the right. Why do we */
70 /* do it this way? Because the calculated CRC must be transmitted in */
71 /* order from highest-order term to lowest-order term. UARTs transmit */
72 /* characters in order from LSB to MSB. By storing the CRC this way, */
73 /* we hand it to the UART in the order low-byte to high-byte; the UART */
74 /* sends each low-bit to hight-bit; and the result is transmission bit */
75 /* by bit from highest- to lowest-order term without requiring any bit */
76 /* shuffling on our part. Reception works similarly. */
77 /* */
78 /* The feedback terms table consists of 256, 32-bit entries. Notes: */
79 /* */
80 /* The table can be generated at runtime if desired; code to do so */
81 /* is shown later. It might not be obvious, but the feedback */
82 /* terms simply represent the results of eight shift/xor opera- */
83 /* tions for all combinations of data and CRC register values. */
84 /* */
85 /* The values must be right-shifted by eight bits by the "updcrc" */
86 /* logic; the shift must be unsigned (bring in zeroes). On some */
87 /* hardware you could probably optimize the shift in assembler by */
88 /* using byte-swap instructions. */
89 /* polynomial $edb88320 */
90 /* */
91 /* -------------------------------------------------------------------- */

148/* Return a 32-bit CRC of the contents of the buffer. */
149
150static unsigned long crc32(const unsigned char *s, unsigned int len)
151{
152 unsigned int i;
153 unsigned long crc32val;
154
155 crc32val = 0;
156 for (i = 0; i < len; i ++)
157 {
158 crc32val =
159 crc32_tab[(crc32val ^ s[i]) & 0xff] ^
160 (crc32val >> 8);
161 }
162 return crc32val;
163}

165/*
166 * Hashing function for a string
167 */
168static unsigned int hashmap_hash_int(const map_t m, const char* keystring){
169
170 unsigned long key = crc32((const unsigned char*)keystring, strlen(keystring));
171
172 /* Robert Jenkins' 32 bit Mix Function */
173 key += (key << 12);
174 key ^= (key >> 22);
175 key += (key << 4);
176 key ^= (key >> 9);
177 key += (key << 10);
178 key ^= (key >> 2);
179 key += (key << 7);
180 key ^= (key >> 12);
181
182 /* Knuth's Multiplicative Method */
183 key = (key >> 3) * 2654435761;
184
185 return key % m->table_size;
186}

188/*
189 * Return the integer of the location in data
190 * to store the point to the item, or MAP_FULL.
191 */
192static int hashmap_hash(map_t m, const char* key){
193 int curr;
194 int i;
195
196 /* If full, return immediately */
197 if(m->size >= (m->table_size/2)) return MAP_FULL;
198
199 /* Find the best index */
200 curr = hashmap_hash_int(m, key);
201
202 /* Linear probing */
203 for(i = 0; i< MAX_CHAIN_LENGTH; i++){
204 if(m->data[curr].in_use == 0)
205 return curr;
206
207 if(m->data[curr].in_use == 1 && (strcmp(m->data[curr].key,key)==0))
208 return curr;
209
210 curr = (curr + 1) % m->table_size;
211 }
212
213 return MAP_FULL;
214}

216/*
217 * Doubles the size of the hashmap, and rehashes all the elements
218 */
219static int hashmap_rehash(map_t m){
220 int i;
221 int old_size;
222 hashmap_element* curr;
223
224 /* Setup the new elements */
225 hashmap_element* temp = (hashmap_element *)
226 calloc(2 * m->table_size, sizeof(hashmap_element));
227 if(!temp) return MAP_OMEM;
228
229 /* Update the array */
230 curr = m->data;
231 m->data = temp;
232
233 /* Update the size */
234 old_size = m->table_size;
235 m->table_size = 2 * m->table_size;
236 m->size = 0;
237
238 /* Rehash the elements */
239 for(i = 0; i < old_size; i++){
240 int status;
241
242 if (curr[i].in_use == 0)
243 continue;
244
245 status = hashmap_put(m, curr[i].key, curr[i].data);
246 if (status != MAP_OK)
247 return status;
248 }
249
250 free(curr);
251
252 return MAP_OK;
253}

255/*
256 * Add a pointer to the hashmap with some key
257 */
258int hashmap_put(map_t m, const char* key, any_t value){
259 int index;
260
261 /* Find a place to put our value */
262 index = hashmap_hash(m, key);
263 while(index == MAP_FULL){
264 if (hashmap_rehash(m) == MAP_OMEM) {
265 return MAP_OMEM;
266 }
267 index = hashmap_hash(m, key);
268 }
269
270 /* Set the data */
271 m->data[index].data = value;
272 m->data[index].key = malloc(strlen(key) + 1);
273 strcpy(m->data[index].key, key);
274 m->data[index].in_use = true;
275 m->size++;
276
277 return MAP_OK;
278}

280/*
281 * Get your pointer out of the hashmap with a key
282 */
283int hashmap_get(const map_t m, const char* key, any_t *arg){
284 int curr;
285 int i;
286
287 /* Find data location */
288 curr = hashmap_hash_int(m, key);
289
290 /* Linear probing, if necessary */
291 for(i = 0; i<MAX_CHAIN_LENGTH; i++){
292
293 int in_use = m->data[curr].in_use;
294 if (in_use == 1){
295 if (strcmp(m->data[curr].key,key)==0){
296 *arg = (m->data[curr].data);
297 return MAP_OK;
298 }
299 }
300
301 curr = (curr + 1) % m->table_size;
302 }
303
304 *arg = NULL;
305
306 /* Not found */
307 return MAP_MISSING;
308}

310/*
311 * Iterate the function parameter over each element in the hashmap. The
312 * additional any_t argument is passed to the function as its first
313 * argument and the hashmap element is the second.
314 */
315int hashmap_iterate(map_t m, PFany f, any_t item) {
316 int i;
317
318 /* On empty hashmap, return immediately */
319 if (hashmap_length(m) <= 0)
320 return MAP_MISSING;
321
322 /* Linear probing */
323 for(i = 0; i< m->table_size; i++)
324 if(m->data[i].in_use != 0) {
325 any_t data = (any_t) (m->data[i].data);
326 int status = f(item, data);
327 if (status != MAP_OK) {
328 return status;
329 }
330 }
331
332 return MAP_OK;
333}

335/*
336 * Remove an element with that key from the map
337 */
338int hashmap_remove(map_t m, char* key){
339 int i;
340 int curr;
341
342 /* Find key */
343 curr = hashmap_hash_int(m, key);
344
345 /* Linear probing, if necessary */
346 for(i = 0; i<MAX_CHAIN_LENGTH; i++){
347
348 int in_use = m->data[curr].in_use;
349 if (in_use == 1){
350 if (strcmp(m->data[curr].key,key)==0){
351 /* Blank out the fields */
352 m->data[curr].in_use = 0;
353 m->data[curr].data = NULL;
354 free(m->data[curr].key);
355 m->data[curr].key = NULL;
356
357 /* Reduce the size */
358 m->size--;
359 return MAP_OK;
360 }
361 }
362 curr = (curr + 1) % m->table_size;
363 }
364
365 /* Data not found */
366 return MAP_MISSING;
367}

369static int hashmap_destroy_item_callback(any_t a, any_t b);
370
371void hashmap_clear(map_t m, void(*callback)(any_t)){
372 hashmap_iterate(m, hashmap_destroy_item_callback, &callback);
373 // Deallocate keys
374 if (m != NULL)
375 {
376 for (int i = 0; i< m->table_size; i++)
377 if (m->data[i].in_use) {
378 m->data[i].in_use = 0;
379 free(m->data[i].key);
380 }
381 m->size = 0;
382 }
383}
384
385/* Deallocate the hashmap */
386void hashmap_free(map_t m){
387 // Deallocate keys
388 if (m != NULL)
389 {
390 for (int i = 0; i< m->table_size; i++)
391 if (m->data[i].in_use) {
392 free(m->data[i].key);
393 }
394 free(m->data);
395 }
396 free(m);
397}
398
399void hashmap_destroy(map_t in, void (*callback)(any_t)){
400 // Pass the callback into the first argument
401 // Note: pass pointer to function-pointer because standard C does not
402 // support function/data pointer conversion
403 hashmap_iterate(in, hashmap_destroy_item_callback, &callback);
404 hashmap_free(in);
405}
406
407static int hashmap_destroy_item_callback(any_t a, any_t b)
408{
409 void (**callback)(any_t) = a;
410 if (callback != NULL) {
411 (*callback)(b);
412 }
413 return MAP_OK;
414}

416/* Return the length of the hashmap */
417int hashmap_length(map_t m){
418 if(m != NULL) return m->size;
419 else return 0;
420}
