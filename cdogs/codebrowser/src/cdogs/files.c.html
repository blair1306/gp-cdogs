78void swap32 (void *d)
79{
80 if (bendian)
81 {
82 *((int32_t *)d) = SDL_Swap32(*((int32_t *)d));
83 }
84}

86size_t f_read(FILE *f, void *buf, size_t size)
87{
88 return fread(buf, size, 1, f);
89}

91size_t f_read32(FILE *f, void *buf, size_t size)
92{
93 size_t ret = 0;
94 if (buf) {
95 ret = f_read(f, buf, size);
96 swap32((int *)buf);
97 }
98 return ret;
99}

131int IsCampaignOldFile(const char *filename)
132{
133 int32_t i;
134 FILE *f = fopen(filename, "rb");
135 if (f == NULL)
136 {
137 return 0;
138 }
139 f_read32(f, &i, sizeof i);
140 fclose(f);
141 return i == CAMPAIGN_MAGIC;
142}

144int ScanCampaignOld(const char *filename, char **title, int *missions)
145{
146 FILE *f;
147 int i;
148 CampaignSettingOld setting;
149
150 debug(D_NORMAL, "filename: %s\n", filename);
151
152 f = fopen(filename, "rb");
153 if (f != NULL)
154 {
155 f_read32(f, &i, sizeof(i));
156
157 if (i != CAMPAIGN_MAGIC) {
158 fclose(f);
159 debug(D_NORMAL, "Filename: %s\n", filename);
160 debug(D_NORMAL, "Magic: %d FileM: %d\n", CAMPAIGN_MAGIC, i);
161 debug(D_NORMAL, "ScanCampaignOld - bad file!\n");
162 return -1;
163 }
164
165 f_read32(f, &i, sizeof(i));
166 if (i != CAMPAIGN_VERSION) {
167 fclose(f);
168 debug(
169 D_NORMAL,
170 "ScanCampaignOld - version mismatch (expected %d, read %d)\n",
171 CAMPAIGN_VERSION, i);
172 return -1;
173 }
174
175 f_read(f, setting.title, sizeof(setting.title));
176 f_read(f, setting.author, sizeof(setting.author));
177 f_read(f, setting.description, sizeof(setting.description));
178 f_read32(f, &setting.missionCount, sizeof(setting.missionCount));
179 CSTRDUP(*title, setting.title);
180 *missions = setting.missionCount;
181
182 fclose(f);
183
184 return 0;
185 }
186 perror("ScanCampaignOld - couldn't read file:");
187 return -1;
188}

192static void load_mission_objective(FILE *f, struct MissionObjectiveOld *o)
193{
194 f_read(f, o->description, sizeof(o->description));
195 f_read32(f, &o->type, sizeof(o->type));
196 f_read32(f, &o->index, sizeof(o->index));
197 f_read32(f, &o->count, sizeof(o->count));
198 f_read32(f, &o->required, sizeof(o->required));
199 f_read32(f, &o->flags, sizeof(o->flags));
200 debug(D_VERBOSE, " >> Objective: %s data: %d %d %d %d %d\n",
201 o->description, o->type, o->index, o->count, o->required, o->flags);
202}

204static void load_mission(FILE *f, struct MissionOld *m)
205{
206 int i;
207
208 f_read(f, m->title, sizeof(m->title));
209 f_read(f, m->description, sizeof(m->description));
210
211 debug(D_NORMAL, "== MISSION ==\n");
212 debug(D_NORMAL, "t: %s\n", m->title);
213 debug(D_NORMAL, "d: %s\n", m->description);
214
215 R32(m->wallStyle);
216 R32(m->floorStyle);
217 R32(m->roomStyle);
218 R32(m->exitStyle);
219 R32(m->keyStyle);
220 R32(m->doorStyle);
221
222 R32(m->mapWidth); R32(m->mapHeight);
223 R32(m->wallCount); R32(m->wallLength);
224 R32(m->roomCount);
225 R32(m->squareCount);
226
227 R32(m->exitLeft); R32(m->exitTop); R32(m->exitRight); R32(m->exitBottom);
228
229 R32(m->objectiveCount);
230
231 debug(D_NORMAL, "number of objectives: %d\n", m->objectiveCount);
232 for (i = 0; i < OBJECTIVE_MAX; i++) {
233 load_mission_objective(f, &m->objectives[i]);
234 }
235
236 R32(m->baddieCount);
237 for (i = 0; i < BADDIE_MAX; i++) {
238 f_read32(f, &m->baddies[i], sizeof(int));
239 }
240
241 R32(m->specialCount);
242 for (i = 0; i < SPECIAL_MAX; i++) {
243 f_read32(f, &m->specials[i], sizeof(int));
244 }
245
246 R32(m->itemCount);
247 for (i = 0; i < ITEMS_MAX; i++) {
248 f_read32(f, &m->items[i], sizeof(int));
249 }
250 for (i = 0; i < ITEMS_MAX; i++) {
251 f_read32(f, &m->itemDensity[i], sizeof(int));
252 }
253
254 R32(m->baddieDensity);
255 R32(m->weaponSelection);
256
257 f_read(f, m->song, sizeof(m->song));
258 f_read(f, m->map, sizeof(m->map));
259
260 R32(m->wallRange);
261 R32(m->floorRange);
262 R32(m->roomRange);
263 R32(m->altRange);
264
265 debug(D_VERBOSE, "number of baddies: %d\n", m->baddieCount);
266}

320void ConvertCharacter(Character *c, TBadGuy *b)
321{
322 c->Class = IntCharacterClass(b->facePic);
323 c->speed = b->speed;
324 c->bot->probabilityToMove = b->probabilityToMove;
325 c->bot->probabilityToTrack = b->probabilityToTrack;
326 c->bot->probabilityToShoot = b->probabilityToShoot;
327 c->bot->actionDelay = b->actionDelay;
328 c->Gun = CArrayGet(&gGunDescriptions.Guns, b->gun);
329 ConvertCharacterColors(
330 b->skinColor, b->armColor, b->bodyColor, b->legColor, b->hairColor,
331 &c->Colors);
332 c->maxHealth = b->health;
333 c->flags = b->flags;
334}
335static void ConvertObjective(
336 Objective *dest, struct MissionObjectiveOld *src)
337{
338 CSTRDUP(dest->Description, src->description);
339 dest->Type = src->type;
340 // Set objective colours based on type
341 dest->color = ObjectiveTypeColor(dest->Type);
342 // Index numbers used for all objective classes; convert them
343 // to their class handles
344 dest->u.Index = src->index;
345 switch (dest->Type)
346 {
347 case OBJECTIVE_COLLECT:
348 dest->u.Pickup = IntPickupClass(dest->u.Index);
349 break;
350 case OBJECTIVE_DESTROY:
351 dest->u.MapObject = IntMapObject(dest->u.Index);
352 break;
353 default:
354 // do nothing
355 break;
356 }
357 dest->Count = src->count;
358 dest->Required = src->required;
359 if (dest->Required < 0)
360 {
361 dest->Required = 0;
362 }
363 dest->Flags = src->flags;
364}
365static void ConvertMission(
366 Mission *dest, struct MissionOld *src, const int charCount)
367{
368 CFREE(dest->Title);
369 CSTRDUP(dest->Title, src->title);
370 CFREE(dest->Description);
371 CSTRDUP(dest->Description, src->description);
372 dest->Type = MAPTYPE_CLASSIC;
373 dest->Size = Vec2iNew(src->mapWidth, src->mapHeight);
374 strcpy(dest->WallStyle, IntWallStyle(src->wallStyle));
375 strcpy(dest->FloorStyle, IntFloorStyle(src->floorStyle));
376 strcpy(dest->RoomStyle, IntRoomStyle(src->roomStyle));
377 strcpy(dest->ExitStyle, IntExitStyle(src->exitStyle));
378 strcpy(dest->KeyStyle, IntKeyStyle(src->keyStyle));
379 strcpy(dest->DoorStyle, IntDoorStyle(src->doorStyle));
380 for (int i = 0; i < src->objectiveCount; i++)
381 {
382 Objective o;
383 ConvertObjective(&o, &src->objectives[i]);
384 CArrayPushBack(&dest->Objectives, &o);
385 }
386 // Note: modulo for compatibility with older, buggy missions
387 for (int i = 0; i < src->baddieCount; i++)
388 {
389 int n = src->baddies[i] % charCount;
390 CArrayPushBack(&dest->Enemies, &n);
391 }
392 for (int i = 0; i < src->specialCount; i++)
393 {
394 int n = src->specials[i] % charCount;
395 CArrayPushBack(&dest->SpecialChars, &n);
396 }
397 for (int i = 0; i < src->itemCount; i++)
398 {
399 MapObjectDensity mod;
400 mod.M = IntMapObject(src->items[i]);
401 mod.Density = src->itemDensity[i];
402 CArrayPushBack(&dest->MapObjectDensities, &mod);
403 }
404 dest->EnemyDensity = src->baddieDensity;
405 CArrayClear(&dest->Weapons);
406 for (int i = 0; i < WEAPON_MAX; i++)
407 {
408 if ((src->weaponSelection & (1 << i)) || !src->weaponSelection)
409 {
410 GunDescription *g = CArrayGet(&gGunDescriptions.Guns, i);
411 CArrayPushBack(&dest->Weapons, &g);
412 }
413 }
414 strcpy(dest->Song, src->song);
415 dest->WallMask = RangeToColor(abs(src->wallRange) % COLORRANGE_COUNT);
416 dest->FloorMask = RangeToColor(abs(src->floorRange) % COLORRANGE_COUNT);
417 dest->RoomMask = RangeToColor(abs(src->roomRange) % COLORRANGE_COUNT);
418 dest->AltMask = RangeToColor(abs(src->altRange) % COLORRANGE_COUNT);
419
420 dest->u.Classic.Walls = src->wallCount;
421 dest->u.Classic.WallLength = src->wallLength;
422 dest->u.Classic.CorridorWidth = 1;
423 dest->u.Classic.Rooms.Count = src->roomCount;
424 dest->u.Classic.Rooms.Min = 6;
425 dest->u.Classic.Rooms.Max = 10;
426 dest->u.Classic.Rooms.Edge = 0;
427 dest->u.Classic.Rooms.Walls = 0;
428 dest->u.Classic.Squares = src->squareCount;
429 dest->u.Classic.Doors.Enabled = 1;
430 dest->u.Classic.Doors.Min = dest->u.Classic.Doors.Max = 1;
431 dest->u.Classic.Pillars.Count = 0;
432}

434void ConvertCampaignSetting(CampaignSetting *dest, CampaignSettingOld *src)
435{
436 int i;
437 CFREE(dest->Title);
438 CSTRDUP(dest->Title, src->title);
439 CFREE(dest->Author);
440 CSTRDUP(dest->Author, src->author);
441 CFREE(dest->Description);
442 CSTRDUP(dest->Description, src->description);
443 for (i = 0; i < src->missionCount; i++)
444 {
445 Mission m;
446 MissionInit(&m);
447 ConvertMission(&m, &src->missions[i], src->characterCount);
448 CArrayPushBack(&dest->Missions, &m);
449 }
450 CharacterStoreTerminate(&dest->characters);
451 CharacterStoreInit(&dest->characters);
452 for (i = 0; i < src->characterCount; i++)
453 {
454 Character *ch = CharacterStoreAddOther(&dest->characters);
455 ConvertCharacter(ch, &src->characters[i]);
456 }
457}

459int LoadCampaignOld(const char *filename, CampaignSettingOld *setting)
460{
461 FILE *f = NULL;
462 int32_t i;
463 int err = 0;
464
465 debug(D_NORMAL, "f: %s\n", filename);
466 f = fopen(filename, "rb");
467 if (f == NULL)
468 {
469 err = -1;
470 goto bail;
471 }
472
473 f_read32(f, &i, sizeof(i));
474 if (i != CAMPAIGN_MAGIC)
475 {
476 debug(D_NORMAL, "LoadCampaign - bad file!\n");
477 err = -1;
478 goto bail;
479 }
480
481 f_read32(f, &i, sizeof(i));
482 if (i != CAMPAIGN_VERSION)
483 {
484 debug(D_NORMAL, "LoadCampaign - version mismatch!\n");
485 err = -1;
486 goto bail;
487 }
488
489 f_read(f, setting->title, sizeof(setting->title));
490 f_read(f, setting->author, sizeof(setting->author));
491 f_read(f, setting->description, sizeof(setting->description));
492
493 f_read32(f, &setting->missionCount, sizeof(int32_t));
494 CCALLOC(
495 setting->missions,
496 setting->missionCount * sizeof *setting->missions);
497 debug(D_NORMAL, "No. missions: %d\n", setting->missionCount);
498 for (i = 0; i < setting->missionCount; i++)
499 {
500 load_mission(f, &setting->missions[i]);
501 }
502
503 f_read32(f, &setting->characterCount, sizeof(int32_t));
504 CCALLOC(
505 setting->characters,
506 setting->characterCount * sizeof *setting->characters);
507 debug(D_NORMAL, "No. characters: %d\n", setting->characterCount);
508 for (i = 0; i < setting->characterCount; i++)
509 {
510 load_character(f, &setting->characters[i]);
511 }
512
513bail:
514 if (f != NULL)
515 {
516 fclose(f);
517 }
518 return err;
519}

521/* GetHomeDirectory ()
522 *
523 * Uses environment variables to determine the users home directory.
524 * returns some sort of path (C string)
525 *
526 * It's an ugly piece of sh*t... :/
527 */
528char *cdogs_homepath = NULL;
529const char *GetHomeDirectory(void)
530{
531#ifdef __EMSCRIPTEN__
532 return "/persistent_data/";
533#endif
534
535 const char *p;
536
537 if (cdogs_homepath != NULL)
538 {
539 return cdogs_homepath;
540 }
541
542 p = getenv("CDOGS_CONFIG_DIR");
543 if (p != NULL && strlen(p) != 0)
544 {
545 CSTRDUP(cdogs_homepath, p);
546 return cdogs_homepath;
547 }
548
549 p = getenv(HOME_DIR_ENV);
550 if (p != NULL && strlen(p) != 0)
551 {
552 CCALLOC(cdogs_homepath, strlen(p) + 2);
553 strcpy(cdogs_homepath, p);
554 cdogs_homepath[strlen(p)] = '/';
555 return cdogs_homepath;
556 }
557
558 fprintf(stderr,"%s%s%s%s",
559 "##############################################################\n",
560 "# You don't have the environment variables HOME or USER set. #\n",
561 "# It is suggested you get a better shell. :D #\n",
562 "##############################################################\n");
563 return "";
564}

586static bool doMkdir(const char *path)
587{
588 if (mkdir(path, MKDIR_MODE) != -1)
589 {
590 return true;
591 }
592 int e = errno;
593 (void)e;
594 // Mac OS X 10.4 returns EISDIR instead of EEXIST
595 // if a dir already exists...
596 return
597 errno == EEXIST ||
598 errno == EISDIR ||
599 errno == EACCES;
600}
601bool mkdir_deep(const char *path)
602{
603 for (int i = 0; i < (int)strlen(path); i++)
604 {
605 if (path[i] == '\0') break;
606 if (path[i] == '/')
607 {
608 char buf[CDOGS_PATH_MAX];
609 strncpy(buf, path, i + 1);
610 buf[i + 1] = '\0';
611 if (!doMkdir(buf))
612 {
613 return false;
614 }
615 }
616 }
617 if (path[strlen(path) - 1] != '/' && !doMkdir(path))
618 {
619 return false;
620 }
621 return true;
622}

624void SetupConfigDir(void)
625{
626 const char *cfg_p = GetConfigFilePath("");
627
628 LOG(LM_MAIN, LL_INFO, "Creating config dir %s... ", cfg_p);
629
630 if (mkdir_deep(cfg_p))
631 {
632 if (errno != EEXIST) LOG(LM_MAIN, LL_INFO, "Config dir created.");
633 else LOG(LM_MAIN, LL_INFO, "Config dir already exists.");
634 }
635 else
636 {
637 switch (errno)
638 {
639 case EACCES:
640 LOG(LM_MAIN, LL_WARN, "Permission denied");
641 break;
642 default:
643 perror("Error creating config directory:");
644 break;
645 }
646 }
647
648 return;
649}
