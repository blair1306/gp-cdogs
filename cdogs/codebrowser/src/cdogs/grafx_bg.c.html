41void GrafxMakeRandomBackground(
42 GraphicsDevice *device,
43 CampaignOptions *co, struct MissionOptions *mo, Map *map)
44{
45 CampaignSettingInit(&co->Setting);
46 SetupQuickPlayCampaign(&co->Setting);
47 const HSV tint = {
48 rand() * 360.0 / RAND_MAX, rand() * 1.0 / RAND_MAX, 0.5
49 };
50 DrawBuffer buffer;
51 DrawBufferInit(&buffer, Vec2iNew(X_TILES, Y_TILES), device);
52 co->MissionIndex = 0;
53 GrafxMakeBackground(
54 device, &buffer, co, mo, map, tint, false, Vec2iZero(), NULL);
55 DrawBufferTerminate(&buffer);
56 MissionOptionsTerminate(mo);
57 CampaignSettingTerminate(&co->Setting);
58}

60void GrafxDrawBackground(
61 GraphicsDevice *g, DrawBuffer *buffer,
62 HSV tint, Vec2i pos, GrafxDrawExtra *extra)
63{
64 DrawBufferSetFromMap(buffer, &gMap, pos, X_TILES);
65 DrawBufferDraw(buffer, Vec2iZero(), extra);
66
67 if (!HSVEquals(tint, tintNone))
68 {
69 Vec2i v;
70 for (v.y = 0; v.y < g->cachedConfig.Res.y; v.y++)
71 {
72 for (v.x = 0; v.x < g->cachedConfig.Res.x; v.x++)
73 {
74 DrawPointTint(g, v, tint);
75 }
76 }
77 }
78 BlitUpdateFromBuf(g, g->bkg);
79 BlitClearBuf(g);
80}

82void GrafxRedrawBackground(GraphicsDevice *g, const Vec2i pos)
83{
84 memset(g->buf, 0, GraphicsGetMemSize(&g->cachedConfig));
85 DrawBuffer buffer;
86 DrawBufferInit(&buffer, Vec2iNew(X_TILES, Y_TILES), g);
87 const HSV tint = {
88 rand() * 360.0 / RAND_MAX, rand() * 1.0 / RAND_MAX, 0.5
89 };
90 GrafxDrawBackground(g, &buffer, tint, pos, NULL);
91 DrawBufferTerminate(&buffer);
92}
93
94void GrafxMakeBackground(
95 GraphicsDevice *device, DrawBuffer *buffer,
96 CampaignOptions *co, struct MissionOptions *mo, Map *map, HSV tint,
97 const bool isEditor, Vec2i pos, GrafxDrawExtra *extra)
98{
99 CampaignAndMissionSetup(co, mo);
100 GameEventsInit(&gGameEvents);
101 MapLoad(map, mo, co);
102 MapLoadDynamic(map, mo, &co->Setting.characters);
103 InitializeBadGuys();
104 CreateEnemies();
105 MapMarkAllAsVisited(map);
106 if (isEditor)
107 {
108 MapShowExitArea(map, map->ExitStart, map->ExitEnd);
109 }
110 else
111 {
112 pos = Vec2iCenterOfTile(Vec2iScaleDiv(map->Size, 2));
113 }
114 // Process the events that place dynamic objects
115 HandleGameEvents(&gGameEvents, NULL, NULL, NULL);
116 GrafxDrawBackground(device, buffer, tint, pos, extra);
117 GameEventsTerminate(&gGameEvents);
118}
