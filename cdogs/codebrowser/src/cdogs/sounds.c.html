72int OpenAudio(int frequency, Uint16 format, int channels, int chunkSize)
73{
74 int qFrequency;
75 Uint16 qFormat;
76 int qChannels;
77
78 if (Mix_OpenAudio(frequency, format, channels, chunkSize) != 0)
79 {
80 printf("Couldn't open audio!: %s\n", SDL_GetError());
81 return 1;
82 }
83
84 // Check that we got the specs we wanted
85#ifndef __EMSCRIPTEN__
86 Mix_QuerySpec(&qFrequency, &qFormat, &qChannels);
87 debug(D_NORMAL, "spec: f=%d fmt=%d c=%d\n", qFrequency, qFormat, qChannels);
88 if (qFrequency != frequency || qFormat != format || qChannels != channels)
89 {
90 printf("Audio not what we want.\n");
91 return 1;
92 }
93#endif
94
95 return 0;
96}

98static Mix_Chunk *LoadSound(const char *path);
99static void AddSound(map_t sounds, const char *name, SoundData *sound);
100static void SoundLoad(map_t sounds, const char *name, const char *path)
101{
102 // If the sound basename is a number, it is part of a group of random sounds
103 char basename[CDOGS_FILENAME_MAX];
104 PathGetBasenameWithoutExtension(basename, name);
105 char nameNoExt[CDOGS_PATH_MAX];
106 PathGetWithoutExtension(nameNoExt, name);
107 bool isNumber = true;
108 for (const char *c = basename; *c != '\0'; c++)
109 {
110 if (!isdigit(*c))
111 {
112 isNumber = false;
113 break;
114 }
115 }
116 if (isNumber)
117 {
118 // Sound is random
119 // Only add the 0 number
120 const int n = atoi(basename);
121 if (n != 0)
122 {
123 return;
124 }
125 SoundData *sound;
126 CCALLOC(sound, sizeof *sound);
127 sound->Type = SOUND_RANDOM;
128 CArrayInit(&sound->u.random.sounds, sizeof(Mix_Chunk *));
129 // Remove "0.<ext>" from path
130 const char *ext = StrGetFileExt(path);
131 const int len = ext - path - 2;
132 char fmt[CDOGS_FILENAME_MAX];
133 strncpy(fmt, path, len);
134 // Create format string path/to/sound/%d.<ext>
135 sprintf(fmt + len, "%%d.%s", ext);
136 for (int i = 0;; i++)
137 {
138 char buf[CDOGS_PATH_MAX];
139 sprintf(buf, fmt, i);
140 Mix_Chunk *data = LoadSound(buf);
141 if (data == NULL) break;
142 CArrayPushBack(&sound->u.random.sounds, &data);
143 }
144 // Remove "/0" from name and add
145 *strrchr(nameNoExt, '/') = '\0';
146 AddSound(sounds, nameNoExt, sound);
147 }
148 else
149 {
150 Mix_Chunk *data = LoadSound(path);
151 if (data != NULL)
152 {
153 SoundData *sound;
154 CMALLOC(sound, sizeof *sound);
155 sound->Type = SOUND_NORMAL;
156 sound->u.normal = data;
157 AddSound(sounds, nameNoExt, sound);
158 }
159 }
160}
161static Mix_Chunk *LoadSound(const char *path)
162{
163 return Mix_LoadWAV(path);
164}

166static void AddSound(map_t sounds, const char *name, SoundData *sound)

168 const int error = hashmap_put(sounds, name, sound);
169 if (error != MAP_OK)

171 LOG(LM_MAIN, LL_ERROR, "failed to add sound %s: %d", name, error);
172 SoundDataTerminate((any_t)sound);
173 }

176void SoundInitialize(SoundDevice *device, const char *path)
177{
178 memset(device, 0, sizeof *device);
179 if (OpenAudio(44100, AUDIO_S16, 2, 1024) != 0)
180 {
181 return;
182 }
183
184 device->channels = 64;
185 SoundReconfigure(device);
186
187 device->sounds = hashmap_new();
188 device->customSounds = hashmap_new();
189 char buf[CDOGS_PATH_MAX];
190 GetDataFilePath(buf, path);
191 SoundLoadDir(device->sounds, buf, NULL);
192}
193void SoundLoadDir(map_t sounds, const char *path, const char *prefix)
194{
195 tinydir_dir dir;
196 if (tinydir_open(&dir, path) == -1)
197 {
198 if (errno != ENOENT)
199 {
200 LOG(LM_MAIN, LL_ERROR, "Cannot open sound dir '%s': %s",
201 path, strerror(errno));
202 }
203 goto bail;
204 }
205 for (; dir.has_next; tinydir_next(&dir))
206 {
207 tinydir_file file;
208 if (tinydir_readfile(&dir, &file) == -1)
209 {
210 LOG(LM_MAIN, LL_ERROR, "Cannot read sound file '%s'", file.path);
211 continue;
212 }
213 if (file.name[0] == '.')
214 {
215 continue;
216 }
217 char buf[CDOGS_PATH_MAX];
218 if (prefix != NULL)
219 {
220 sprintf(buf, "%s/%s", prefix, file.name);
221 }
222 else
223 {
224 strcpy(buf, file.name);
225 }
226 if (file.is_reg)
227 {
228 SoundLoad(sounds, buf, file.path);
229 }
230 else if (file.is_dir)
231 {
232 SoundLoadDir(sounds, file.path, buf);
233 }
234 }
235
236bail:
237 tinydir_close(&dir);
238}

240void SoundReconfigure(SoundDevice *s)
241{
242 s->isInitialised = false;
243
244 if (Mix_AllocateChannels(s->channels) != s->channels)
245 {
246 printf("Couldn't allocate channels!\n");
247 return;
248 }
249
250 Mix_Volume(-1, ConfigGetInt(&gConfig, "Sound.SoundVolume"));
251 Mix_VolumeMusic(ConfigGetInt(&gConfig, "Sound.MusicVolume"));
252 if (ConfigGetInt(&gConfig, "Sound.MusicVolume") > 0)
253 {
254 MusicResume(s);
255 }
256 else
257 {
258 MusicPause(s);
259 }
260
261 s->isInitialised = true;
262}

268void SoundTerminate(SoundDevice *device, const bool waitForSoundsComplete)
269{
270 if (!device->isInitialised)
271 {
272 return;
273 }
274
275 debug(D_NORMAL, "shutting down sound\n");
276 if (waitForSoundsComplete)
277 {
278 Uint32 waitStart = SDL_GetTicks();
279 while (Mix_Playing(-1) > 0 &&
280 SDL_GetTicks() - waitStart < 1000);
281 }
282 MusicStop(device);
283 while (Mix_Init(0))
284 {
285 Mix_Quit();
286 }
287 Mix_CloseAudio();
288
289 hashmap_destroy(device->sounds, SoundDataTerminate);
290 hashmap_destroy(device->customSounds, SoundDataTerminate);
291}
292static void SoundDataTerminate(any_t data)
293{
294 SoundData *s = data;
295 switch (s->Type)
296 {
297 case SOUND_NORMAL:
298 Mix_FreeChunk(s->u.normal);
299 break;
300 case SOUND_RANDOM:
301 CA_FOREACH(Mix_Chunk *, chunk, s->u.random.sounds)
302 Mix_FreeChunk(*chunk);
303 CA_FOREACH_END()
304 CArrayTerminate(&s->u.random.sounds);
305 break;
306 default:
307 CASSERT(false, "Unknown sound data type");
308 break;
309 }
310 CFREE(s);
311}

313#define OUT_OF_SIGHT_DISTANCE_PLUS 100
314static int GetChannel(SoundDevice *s, Mix_Chunk *data);
315static void MuffleEffect(int chan, void *stream, int len, void *udata)
316{
317 UNUSED(chan);
318 UNUSED(udata);
319 const int channels = 2;
320 const int chunk = channels * 2;
321 for (int i = 0; i < len / chunk - 2; i++)
322 {
323 int16_t *samples = (int16_t *)((char *)stream + i * chunk);
324 samples[0] = (samples[0] + samples[2] + samples[4]) / 3;
325 samples[1] = (samples[1] + samples[3] + samples[5]) / 3;
326 }
327}
328static void SetSoundEffect(
329 const int channel, const Sint16 bearingDegrees, const Uint8 distance,
330 const bool isMuffled);
331static void SoundPlayAtPosition(
332 SoundDevice *device, Mix_Chunk *data, const Vec2i dp, const bool isMuffled)
333{
334 if (!device->isInitialised || data == NULL)
335 {
336 return;
337 }
338
339 int distance = 0;
340 Sint16 bearingDegrees = 0;
341 const int screen = gGraphicsDevice.cachedConfig.Res.x;
342 const int halfScreen = screen / 2;
343 if (!Vec2iIsZero(dp))
344 {
345 // Calculate distance and bearing
346 // Sound position is calculated from an imaginary camera that's half as
347 // distant from the centre of the screen as the screen width, i.e.
348 //
349 // centre-+
350 // v
351 // Screen: |------+------|
352 // |
353 // |
354 // camera---> +
355 // Calculate real distance using Pythagoras
356 const int d =
357 (int)sqrt(Vec2iSqrMagnitude(dp) + halfScreen * halfScreen);
358 // Translate so that sounds at exactly centre are distance 0
359 const int dTrans = d - halfScreen;
360 // Scale so that sounds more than a full screen from centre have
361 // maximum distance (255)
362 const int maxDistance =
363 (int)sqrt(screen * screen + halfScreen * halfScreen) - halfScreen;
364 distance = dTrans * 255 / maxDistance;
365
366 // Calculate bearing
367 const double bearing = atan((double)dp.x / halfScreen);
368 bearingDegrees = (Sint16)(bearing * 180 / PI);
369 if (bearingDegrees < 0)
370 {
371 bearingDegrees += 360;
372 }
373 }
374 if (isMuffled)
375 {
376 distance += OUT_OF_SIGHT_DISTANCE_PLUS;
377 }
378 // Don't play anything if it's too distant
379 // This means we don't waste sound channels
380 if (distance > 255)
381 {
382 return;
383 }
384
385 LOG(LM_SOUND, LL_TRACE, "distance(%d) bearing(%d)",
386 distance, bearingDegrees);
387
388 // Get sound channel to play sound
389 const int channel = GetChannel(device, data);
390 if (channel < 0)
391 {
392 return;
393 }
394
395 SetSoundEffect(channel, bearingDegrees, (Uint8)distance, isMuffled);
396}
397static int GetChannel(SoundDevice *s, Mix_Chunk *data)
398{
399 for (;;)
400 {
401 const int channel = Mix_PlayChannel(-1, data, 0);
402 if (channel >= 0 || s->channels > 128)
403 {
404 return channel;
405 }
406 // Check if we cannot play the sound; allocate more channels
407 s->channels *= 2;
408 if (Mix_AllocateChannels(s->channels) != s->channels)
409 {
410 LOG(LM_SOUND, LL_ERROR, "Cannot allocate channels");
411 return -1;
412 }
413 // When allocating new channels, need to reset their volume
414 Mix_Volume(-1, ConfigGetInt(&gConfig, "Sound.SoundVolume"));
415 }
416}
417static void SetSoundEffect(
418 const int channel, const Sint16 bearingDegrees, const Uint8 distance,
419 const bool isMuffled)
420{
421#ifndef __EMSCRIPTEN__
422 Mix_SetPosition(channel, bearingDegrees, (Uint8)distance);
423 if (isMuffled)
424 {
425 if (!Mix_RegisterEffect(channel, MuffleEffect, NULL, NULL))
426 {
427 fprintf(stderr, "Mix_RegisterEffect: %s\n", Mix_GetError());
428 }
429 }
430#else
431 // Mix_SetPosition and Mix_RegisterEffect not supported by emscripten;
432 // use plain panning instead
433
434 // Calculate left/right channel as values from 0-180
435 int left;
436 if (bearingDegrees < 90) left = 90 - bearingDegrees;
437 else if (bearingDegrees < 270) left = bearingDegrees - 90;
438 else left = 450 - bearingDegrees;
439 const int right = 180 - left;
440 Mix_SetPanning(
441 channel,
442 (Uint8)(left * distance / 180), (Uint8)(right * distance / 180));
443#endif
444}

446void SoundPlay(SoundDevice *device, Mix_Chunk *data)

448 if (!device->isInitialised)

500static bool IsPosNoSee(void *data, Vec2i pos)
501{
502 const Tile *t = MapGetTile(data, Vec2iToTile(pos));
503 return t != NULL && (t->flags & MAPTILE_NO_SEE);
504}
505void SoundPlayAtPlusDistance(
506 SoundDevice *device, Mix_Chunk *data,
507 const Vec2i pos, const int plusDistance)
508{
509 Vec2i closestLeftEar, closestRightEar;
510
511 // Find closest set of ears to the sound
512 if (CHEBYSHEV_DISTANCE(
513 pos.x, pos.y, device->earLeft1.x, device->earLeft1.y) <
514 CHEBYSHEV_DISTANCE(
515 pos.x, pos.y, device->earLeft2.x, device->earLeft2.y))
516 {
517 closestLeftEar = device->earLeft1;
518 }
519 else
520 {
521 closestLeftEar = device->earLeft2;
522 }
523 if (CHEBYSHEV_DISTANCE(
524 pos.x, pos.y, device->earRight1.x, device->earRight1.y) <
525 CHEBYSHEV_DISTANCE(
526 pos.x, pos.y, device->earRight2.x, device->earRight2.y))
527 {
528 closestRightEar = device->earRight1;
529 }
530 else
531 {
532 closestRightEar = device->earRight2;
533 }
534
535 const Vec2i origin = CalcClosestPointOnLineSegmentToPoint(
536 closestLeftEar, closestRightEar, pos);
537 HasClearLineData lineData;
538 lineData.IsBlocked = IsPosNoSee;
539 lineData.data = &gMap;
540 bool isMuffled = false;
541 if (!HasClearLineJMRaytrace(pos, origin, &lineData))
542 {
543 isMuffled = true;
544 }
545 const Vec2i dp = Vec2iMinus(pos, origin);
546 SoundPlayAtPosition(
547 &gSoundDevice, data, Vec2iAdd(dp, Vec2iNew(0, plusDistance)),
548 isMuffled);
549}

551static Mix_Chunk *SoundDataGet(SoundData *s);
552Mix_Chunk *StrSound(const char *s)
553{
554 if (s == NULL || strlen(s) == 0 || !gSoundDevice.isInitialised)
555 {
556 return NULL;
557 }
558 SoundData *sound;
559 int error = hashmap_get(gSoundDevice.customSounds, s, (any_t *)&sound);
560 if (error == MAP_OK)
561 {
562 return SoundDataGet(sound);
563 }
564 error = hashmap_get(gSoundDevice.sounds, s, (any_t *)&sound);
565 if (error == MAP_OK)
566 {
567 return SoundDataGet(sound);
568 }
569 return NULL;
570}
571static Mix_Chunk *SoundDataGet(SoundData *s)
572{
573 switch (s->Type)
574 {
575 case SOUND_NORMAL:
576 return s->u.normal;
577 case SOUND_RANDOM:
578 {
579 // Don't get the last sound used
580 int idx = s->u.random.lastPlayed;
581 while ((int)s->u.random.sounds.size > 1 &&
582 idx == s->u.random.lastPlayed)
583 {
584 idx = rand() % s->u.random.sounds.size;
585 }
586 Mix_Chunk **sound = CArrayGet(&s->u.random.sounds, idx);
587 s->u.random.lastPlayed = idx;
588 return *sound;
589 }
590 default:
591 CASSERT(false, "Unknown sound data type");
592 return NULL;
593 }
