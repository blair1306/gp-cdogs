36// Game events represent anything that is created within the game but is
37// required by outside systems, e.g. sound events
38// This is to prevent the game from depending on these external systems

111// Which game events should be passed along to server or client
112typedef struct
113{
114 GameEventType Type;
115 // Whether this message is sent by server to clients
116 bool Broadcast;
117 // Whether clients send this message to the server
118 bool Submit;
119 // Whether to simply enqueue as game event; otherwise processed by handler
120 bool Enqueue;
121 // Whether to broadcast these events only after game start
122 bool GameStart;
123 const pb_field_t *Fields;
124} GameEventEntry;
