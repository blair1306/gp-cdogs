40void JoyInit(CArray *joys)
41{
42 char buf[CDOGS_PATH_MAX];
43 GetDataFilePath(buf, "data/gamecontrollerdb.txt");
44 if (SDL_GameControllerAddMappingsFromFile(buf) == -1)
45 {
46 LOG(LM_INPUT, LL_ERROR, "cannot load controller mappings file: %s",
47 SDL_GetError());
48 }
49 GetDataFilePath(buf, "data/gamecontrollerbuttondb.txt");
50 if (SDLJBN_AddMappingsFromFile(buf) == -1)
51 {
52 LOG(LM_INPUT, LL_ERROR, "cannot load button mappings file: %s",
53 SDLJBN_GetError());
54 }
55
56 CArrayInit(joys, sizeof(Joystick));
57
58 // Detect all current controllers
59 const int n = SDL_NumJoysticks();
60 LOG(LM_INPUT, LL_INFO, "%d controllers found", n);
61 if (n == 0)
62 {
63 return;
64 }
65 for (int i = 0; i < n; i++)
66 {
67 JoyAdded(i);
68 }
69}

71void JoyReset(CArray *joys)
72{
73 CA_FOREACH(Joystick, j, *joys)
74 j->currentCmd = j->pressedCmd = j->previousCmd = 0;
75 CA_FOREACH_END()
76}

78static void JoyTerminateOne(Joystick *j)
79{
80 SDL_GameControllerClose(j->gc);
81 SDL_HapticClose(j->haptic);
82}
83
84void JoyTerminate(CArray *joys)
85{
86 CA_FOREACH(Joystick, j, *joys)
87 JoyTerminateOne(j);
88 CA_FOREACH_END()
89 CArrayTerminate(joys);
90}

92static Joystick *GetJoystick(const SDL_JoystickID id)
93{
94 CA_FOREACH(Joystick, j, gEventHandlers.joysticks)
95 if (j->id == id) return j;
96 CA_FOREACH_END()
97 return NULL;
98}
99
100bool JoyIsDown(const SDL_JoystickID id, const int cmd)
101{
102 return !!(GetJoystick(id)->currentCmd & cmd);
103}
104
105bool JoyIsPressed(const SDL_JoystickID id, const int cmd)
106{
107 return JoyIsDown(id, cmd) && !(GetJoystick(id)->previousCmd & cmd);
108}
109
110int JoyGetPressed(const SDL_JoystickID id)
111{
112 const Joystick *j = GetJoystick(id);
113 // Pressed is current and not previous, so take bitwise complement
114 return j->currentCmd & ~j->previousCmd;
115}

117void JoyPrePoll(CArray *joys)
118{

125SDL_JoystickID JoyAdded(const Sint32 which)
126{
127 if (!SDL_IsGameController(which))
128 {
129 return -1;
130 }
131
132 Joystick j;
133 memset(&j, 0, sizeof j);
134 j.hapticEffectId = -1;
135 j.gc = SDL_GameControllerOpen(which);
136 if (j.gc == NULL)
137 {
138 LOG(LM_INPUT, LL_ERROR, "Failed to open game controller: %s",
139 SDL_GetError());
140 return -1;
141 }
142 j.j = SDL_GameControllerGetJoystick(j.gc);
143 if (j.j == NULL)
144 {
145 LOG(LM_INPUT, LL_ERROR, "Failed to open joystick: %s",
146 SDL_GetError());
147 return -1;
148 }
149 j.id = SDL_JoystickInstanceID(j.j);
150 if (j.id == -1)
151 {
152 LOG(LM_INPUT, LL_ERROR, "Failed to get joystick instance ID: %s",
153 SDL_GetError());
154 return -1;
155 }
156 const int isHaptic = SDL_JoystickIsHaptic(j.j);
157 if (isHaptic > 0)
158 {
159 j.haptic = SDL_HapticOpenFromJoystick(j.j);
160 if (j.haptic == NULL)
161 {
162 LOG(LM_INPUT, LL_ERROR, "Failed to open haptic: %s",
163 SDL_GetError());
164 }
165 else
166 {
167 if (SDL_HapticRumbleInit(j.haptic) != 0)
168 {
169 LOG(LM_INPUT, LL_ERROR, "Failed to init rumble: %s",
170 SDL_GetError());
171 }
172 const int hapticQuery = SDL_HapticQuery(j.haptic);
173 LOG(LM_INPUT, LL_INFO, "Haptic support: %x", hapticQuery);
174 if (hapticQuery & SDL_HAPTIC_CONSTANT)
175 {
176 SDL_HapticEffect he;
177 memset(&he, 0, sizeof he);
178 he.type = SDL_HAPTIC_CONSTANT;
179 he.constant.length = 100;
180 he.constant.level = 20000; // 20000/32767 strength
181 j.hapticEffectId = SDL_HapticNewEffect(j.haptic, &he);
182 if (j.hapticEffectId == -1)
183 {
184 LOG(LM_INPUT, LL_ERROR,
185 "Failed to create haptic effect: %s", SDL_GetError());
186 }
187 }
188 }
189 }
190 else if (isHaptic < 0)
191 {
192 LOG(LM_INPUT, LL_ERROR, "Failed to query haptic: %s", SDL_GetError());
193 }
194 CArrayPushBack(&gEventHandlers.joysticks, &j);
195 LOG(LM_INPUT, LL_INFO, "Added joystick index %d id %d", which, j.id);
196 return j.id;
197}
198void JoyRemoved(const Sint32 which)
199{
200 LOG(LM_INPUT, LL_INFO, "Removed joystick id %d", which);
201 CA_FOREACH(Joystick, j, gEventHandlers.joysticks)
202 if (j->id == which)
203 {
204 JoyTerminateOne(j);
205 CArrayDelete(&gEventHandlers.joysticks, _ca_index);
206 return;
207 }
208 CA_FOREACH_END()
209 CASSERT(false, "Cannot find joystick");
210}

215void JoyOnButtonDown(const SDL_ControllerButtonEvent e)
216{
217 LOG(LM_INPUT, LL_DEBUG, "Joystick %d button down %d", e.which, e.button);
218 JoyOnCmd(GetJoystick(e.which), ControllerButtonToCmd(e.button), true);
219}
220void JoyOnButtonUp(const SDL_ControllerButtonEvent e)
221{
222 LOG(LM_INPUT, LL_DEBUG, "Joystick %d button up %d", e.which, e.button);
223 JoyOnCmd(GetJoystick(e.which), ControllerButtonToCmd(e.button), false);
224}
225int ControllerButtonToCmd(const Uint8 button)
226{
227 switch (button)
228 {
229 case SDL_CONTROLLER_BUTTON_A: return CMD_BUTTON1;
230 case SDL_CONTROLLER_BUTTON_B: return CMD_BUTTON2;
231 case SDL_CONTROLLER_BUTTON_BACK: return CMD_MAP;
232 case SDL_CONTROLLER_BUTTON_START: return CMD_ESC;
233 case SDL_CONTROLLER_BUTTON_DPAD_UP: return CMD_UP;
234 case SDL_CONTROLLER_BUTTON_DPAD_DOWN: return CMD_DOWN;
235 case SDL_CONTROLLER_BUTTON_DPAD_LEFT: return CMD_LEFT;
236 case SDL_CONTROLLER_BUTTON_DPAD_RIGHT: return CMD_RIGHT;
237 default: return 0;
238 }
239}
240#define DEADZONE 16384
241void JoyOnAxis(const SDL_ControllerAxisEvent e)
242{
243 LOG(LM_INPUT, LL_DEBUG, "Joystick %d received axis %d, %d",
244 e.which, e.axis, e.value);
245 Joystick *j = GetJoystick(e.which);
246 switch (e.axis)
247 {
248 case SDL_CONTROLLER_AXIS_LEFTX:
249 JoyOnCmd(j, CMD_LEFT, e.value < -DEADZONE);
250 JoyOnCmd(j, CMD_RIGHT, e.value > DEADZONE);
251 break;
252 case SDL_CONTROLLER_AXIS_LEFTY:
253 JoyOnCmd(j, CMD_UP, e.value < -DEADZONE);
254 JoyOnCmd(j, CMD_DOWN, e.value > DEADZONE);
255 break;
256 case SDL_CONTROLLER_AXIS_TRIGGERRIGHT:
257 // Right trigger fire
258 JoyOnCmd(j, CMD_BUTTON1, e.value > DEADZONE);
259 break;
260 default:
261 // Ignore axis
262 break;
263 }
264}

266static void JoyOnCmd(Joystick *j, const int cmd, const bool isDown)
267{
268 if (isDown)
269 {
270 j->currentCmd |= cmd;
271 }
272 else
273 {
274 if ((j->currentCmd & cmd) && !(j->previousCmd & cmd))
275 {
276 j->pressedCmd |= cmd;
277 }
278 j->currentCmd &= ~cmd;
279 }
280}

282void JoyRumble(
283 const SDL_JoystickID id, const float strength, const Uint32 length)
284{
285 Joystick *j = GetJoystick(id);
286 if (j->haptic == NULL) return;
287 if (SDL_HapticRumblePlay(j->haptic, strength, length) < 0)
288 {
289 LOG(LM_INPUT, LL_ERROR, "Failed to rumble: %s", SDL_GetError());
290 }
291}
292void JoyImpact(const SDL_JoystickID id)
293{
294 Joystick *j = GetJoystick(id);
295 if (j->hapticEffectId != -1 &&
296 SDL_HapticRunEffect(j->haptic, j->hapticEffectId, 1) != 0)
297 {
298 LOG(LM_INPUT, LL_ERROR, "Failed to run haptic effect: %s",
299 SDL_GetError());
300 }
301}

303const char *JoyName(const SDL_JoystickID id)
304{
305 const Joystick *j = GetJoystick(id);
306 const char *controllerName = SDL_GameControllerName(j->gc);
307 const char *joystickName = SDL_JoystickName(j->j);
308 // Try to use the more specific name
309 if (joystickName != NULL &&
310 strcmp(controllerName, "Generic DirectInput Controller") == 0)
311 {
312 return joystickName;
313 }
314 return controllerName;
315}

317static int CmdToControllerButton(const int cmd);
318void JoyButtonNameColor(
319 const SDL_JoystickID id, const int cmd, char *buf, color_t *color)
320{
321 Joystick *j = GetJoystick(id);
322 const int button = CmdToControllerButton(cmd);
323 int res;
324 if (color != NULL)
325 {
326 res = SDLJBN_GetButtonNameAndColor(
327 j->j, button, buf, &color->r, &color->g, &color->b);
328 }
329 else
330 {
331 res = SDLJBN_GetButtonName(j->j, button, buf);
332 }
333 if (res < 0)
334 {
335 LOG(LM_INPUT, LL_WARN, "Could not get button name/colour: %s",
336 SDLJBN_GetError());
337 strcpy(buf, "?");
338 }
339}
340static int CmdToControllerButton(const int cmd)
341{
342 switch (cmd)
343 {
344 case CMD_BUTTON1: return SDL_CONTROLLER_BUTTON_A;
345 case CMD_BUTTON2: return SDL_CONTROLLER_BUTTON_B;
346 case CMD_MAP: return SDL_CONTROLLER_BUTTON_BACK;
347 case CMD_ESC: return SDL_CONTROLLER_BUTTON_START;
348 case CMD_UP: return SDL_CONTROLLER_BUTTON_DPAD_UP;
349 case CMD_DOWN: return SDL_CONTROLLER_BUTTON_DPAD_DOWN;
350 case CMD_LEFT: return SDL_CONTROLLER_BUTTON_DPAD_LEFT;
351 case CMD_RIGHT: return SDL_CONTROLLER_BUTTON_DPAD_RIGHT;
352 default: return SDL_CONTROLLER_BUTTON_INVALID;
353 }
