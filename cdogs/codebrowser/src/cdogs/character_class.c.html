41const CharacterClass *StrCharacterClass(const char *s)
42{
43 CA_FOREACH(const CharacterClass, c, gCharacterClasses.CustomClasses)
44 if (strcmp(s, c->Name) == 0)
45 {
46 return c;
47 }
48 CA_FOREACH_END()
49 CA_FOREACH(const CharacterClass, c, gCharacterClasses.Classes)
50 if (strcmp(s, c->Name) == 0)
51 {
52 return c;
53 }
54 CA_FOREACH_END()
55 LOG(LM_MAIN, LL_ERROR, "Cannot find character name: %s", s);
56 return NULL;
57}

82const CharacterClass *IndexCharacterClass(const int i)
83{
84 CASSERT(
85 i >= 0 &&
86 i < (int)gCharacterClasses.Classes.size +
87 (int)gCharacterClasses.CustomClasses.size,
88 "Character class index out of bounds");
89 if (i < (int)gCharacterClasses.Classes.size)
90 {
91 return CArrayGet(&gCharacterClasses.Classes, i);
92 }
93 return CArrayGet(
94 &gCharacterClasses.CustomClasses, i - gCharacterClasses.Classes.size);
95}
96int CharacterClassIndex(const CharacterClass *c)
97{
98 if (c == NULL)
99 {
100 return 0;
101 }
102 CA_FOREACH(const CharacterClass, cc, gCharacterClasses.Classes)
103 if (cc == c)
104 {
105 return _ca_index;
106 }
107 CA_FOREACH_END()
108 CA_FOREACH(const CharacterClass, cc, gCharacterClasses.CustomClasses)
109 if (cc == c)
110 {
111 return _ca_index + gCharacterClasses.Classes.size;
112 }
113 CA_FOREACH_END()
114 CASSERT(false, "cannot find character class");
115 return -1;
116}

118void CharacterClassesInitialize(CharacterClasses *c, const char *filename)
119{
120 memset(c, 0, sizeof *c);
121 CArrayInit(&c->Classes, sizeof(CharacterClass));
122 CArrayInit(&c->CustomClasses, sizeof(CharacterClass));
123
124 char buf[CDOGS_PATH_MAX];
125 GetDataFilePath(buf, filename);
126 FILE *f = fopen(buf, "r");
127 json_t *root = NULL;
128 if (f == NULL)
129 {
130 LOG(LM_MAIN, LL_ERROR, "cannot load characters file %s", buf);
131 goto bail;
132 }
133 enum json_error e = json_stream_parse(f, &root);
134 if (e != JSON_OK)
135 {
136 LOG(LM_MAIN, LL_ERROR, "error parsing characters file %s", buf);
137 goto bail;
138 }
139 CharacterClassesLoadJSON(&c->Classes, root);
140
141bail:
142 if (f != NULL)
143 {
144 fclose(f);
145 }
146 json_free_value(&root);
147}
148static void LoadCharacterClass(CharacterClass *c, json_t *node);
149void CharacterClassesLoadJSON(CArray *classes, json_t *root)
150{
151 int version;
152 LoadInt(&version, root, "Version");
153 if (version > VERSION || version <= 0)
154 {
155 LOG(LM_MAIN, LL_ERROR,
156 "Cannot read character file version: %d", version);
157 return;
158 }
159
160 json_t *charactersNode = json_find_first_label(root, "Characters")->child;
161 for (json_t *child = charactersNode->child; child; child = child->next)
162 {
163 CharacterClass cc;
164 LoadCharacterClass(&cc, child);
165 CArrayPushBack(classes, &cc);
166 }
167}
168static void LoadCharacterClass(CharacterClass *c, json_t *node)
169{
170 memset(c, 0, sizeof *c);
171 c->Name = GetString(node, "Name");
172 CPicLoadJSON(&c->HeadPics, json_find_first_label(node, "HeadPics")->child);
173 // TODO: custom character sprites
174 c->Sprites = StrCharSpriteClass("base");
175
176 // Default man sounds
177 CSTRDUP(c->Sounds.Aargh, "aargh/man");
178 json_t *sounds = json_find_first_label(node, "Sounds");
179 if (sounds != NULL && sounds->child != NULL)
180 {
181 char *tmp = NULL;
182 LoadStr(&tmp, sounds->child, "Aargh");
183 if (tmp != NULL)
184 {
185 CFREE(c->Sounds.Aargh);
186 c->Sounds.Aargh = tmp;
187 }
188 else
189 {
190 CFREE(tmp);
191 }
192 }
193}
194static void CharacterClassFree(CharacterClass *c);
195void CharacterClassesClear(CArray *classes)
196{
197 for (int i = 0; i < (int)classes->size; i++)
198 {
199 CharacterClassFree(CArrayGet(classes, i));
200 }
201 CArrayClear(classes);
202}
203static void CharacterClassFree(CharacterClass *c)
204{
205 CFREE(c->Name);
206 CFREE(c->Sounds.Aargh);
207}
208void CharacterClassesTerminate(CharacterClasses *c)
209{
210 CharacterClassesClear(&c->Classes);
211 CArrayTerminate(&c->Classes);
212 CharacterClassesClear(&c->CustomClasses);
213 CArrayTerminate(&c->CustomClasses);
214}
