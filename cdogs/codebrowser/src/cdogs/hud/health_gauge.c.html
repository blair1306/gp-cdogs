34#define WAIT_MS 1000

37void HealthGaugeInit(HealthGauge *h)
38{
39 h->health = 0;
40 h->waitHealth = 0;
41 h->waitMs = 0;
42}
43
44void HealthGaugeUpdate(HealthGauge *h, const TActor *a, const int ms)
45{
46 // Initialise health to actor's health
47 if (h->health == 0)
48 {
49 h->health = a->health;
50 h->waitHealth = a->health;
51 }
52 h->waitMs = MAX(0, h->waitMs - ms);
53 if (h->waitHealth != a->health)
54 {
55 // Health has changed; start waiting
56 h->waitHealth = a->health;
57 h->waitMs = WAIT_MS;
58 }
59 else if (h->waitMs == 0)
60 {
61 // End of wait; start moving health towards real health
62 if (h->health != a->health)
63 {
64 const int d = SIGN(a->health - h->health);
65 h->health += d;
66 }
67 }
68}

70void HealthGaugeDraw(
71 const HealthGauge *h, GraphicsDevice *device, const TActor *actor,
72 const Vec2i pos, const FontAlign hAlign, const FontAlign vAlign)
73{
74 char s[50];
75 Vec2i gaugePos = Vec2iAdd(pos, Vec2iNew(-1, -1));
76 Vec2i size = Vec2iNew(GAUGE_WIDTH, FontH() + 2);
77 HSV hsv = { 0.0, 1.0, 1.0 };
78 const int health = actor->health;
79 const int maxHealth = ActorGetCharacter(actor)->maxHealth;
80 if (actor->poisoned)
81 {
82 hsv.h = 120.0;
83 hsv.v = 0.5;
84 }
85 else
86 {
87 double maxHealthHue = 50.0;
88 double minHealthHue = 0.0;
89 hsv.h =
90 (maxHealthHue - minHealthHue) * health / maxHealth + minHealthHue;
91 }
92 color_t backColor = { 50, 0, 0, 255 };
93 if (h->health != health)
94 {
95 // Draw different-coloured health gauge representing health change
96 const int higherHealth = MAX(h->health, health);
97 const int innerWidthUpdate = MAX(1, size.x * higherHealth / maxHealth);
98 HUDDrawGauge(
99 device, gaugePos, size, innerWidthUpdate,
100 h->health > health ? colorMaroon : colorGreen,
101 backColor, hAlign, vAlign);
102 backColor = colorTransparent;
103 }
104 const int lowerHealth = MIN(h->health, health);
105 const int innerWidth = MAX(1, size.x * lowerHealth / maxHealth);
106 const color_t barColor = ColorTint(colorWhite, hsv);
107 HUDDrawGauge(
108 device, gaugePos, size, innerWidth, barColor, backColor,
109 hAlign, vAlign);
110 sprintf(s, "%d", health);
111
112 // Draw health number label
113 FontOpts opts = FontOptsNew();
114 opts.HAlign = hAlign;
115 opts.VAlign = vAlign;
116 opts.Area = gGraphicsDevice.cachedConfig.Res;
117 opts.Pad = pos;
118 // If low health, draw text with different colours, flashing
119 if (ActorIsLowHealth(actor))
120 {
121 // Fast flashing
122 const int fps = ConfigGetInt(&gConfig, "Game.FPS");
123 const int pulsePeriod = fps / 4;
124 if ((gMission.time % pulsePeriod) < (pulsePeriod / 2))
125 {
126 opts.Mask = colorRed;
127 }
128 }
129 FontStrOpt(s, Vec2iZero(), opts);
130}
