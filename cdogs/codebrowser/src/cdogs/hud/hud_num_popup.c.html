37// Total number of milliseconds that the numeric popup lasts for
38#define TIMER_MS 500
39#define TIMER_OBJECTIVE_MS 1500

42void HUDNumPopupsInit(
43 HUDNumPopups *popups, const struct MissionOptions *mission)
44{
45 memset(popups, 0, sizeof *popups);
46 CArrayInit(&popups->objective, sizeof(HUDNumPopup));
47 CArrayResize(
48 &popups->objective, mission->missionData->Objectives.size, NULL);
49 CArrayFillZero(&popups->objective);
50}
51
52void HUDNumPopupsTerminate(HUDNumPopups *popups)
53{
54 CArrayTerminate(&popups->objective);
55}

57static int FindLocalPlayerIndex(const int playerUID);
58static void MergePopups(HUDNumPopup *dst, const HUDNumPopup src);
59void HUDNumPopupsAdd(
60 HUDNumPopups *popups, const HUDNumPopupType type,
61 const int idxOrUID, const int amount)
62{
63 HUDNumPopup s;
64 memset(&s, 0, sizeof s);
65
66 // Index
67 int localPlayerIdx = -1;
68 switch (type)
69 {
70 case NUMBER_POPUP_SCORE:
71 localPlayerIdx = FindLocalPlayerIndex(idxOrUID);
72 if (localPlayerIdx)
73 {
74 // This popup was for a non-local player; abort
75 return;
76 }
77 s.u.PlayerUID = idxOrUID;
78 break;
79 case NUMBER_POPUP_OBJECTIVE:
80 s.u.ObjectiveIndex = idxOrUID;
81 break;
82 default:
83 CASSERT(false, "unknown HUD popup type");
84 break;
85 }
86
87 s.Amount = amount;
88
89 // Timers
90 switch (type)
91 {
92 case NUMBER_POPUP_SCORE:
93 s.Timer = TIMER_MS;
94 s.TimerMax = TIMER_MS;
95 break;
96 case NUMBER_POPUP_OBJECTIVE:
97 s.Timer = TIMER_OBJECTIVE_MS;
98 s.TimerMax = TIMER_OBJECTIVE_MS;
99 break;
100 default:
101 CASSERT(false, "unknown HUD popup type");
102 break;
103 }
104
105 // Merge with existing popups
106 switch (type)
107 {
108 case NUMBER_POPUP_SCORE:
109 MergePopups(&popups->score[localPlayerIdx], s);
110 break;
111 case NUMBER_POPUP_OBJECTIVE:
112 MergePopups(CArrayGet(&popups->objective, s.u.ObjectiveIndex), s);
113 break;
114 default:
115 CASSERT(false, "unknown HUD popup type");
116 break;
117 }
118}
119static int FindLocalPlayerIndex(const int playerUID)
120{
121 const PlayerData *p = PlayerDataGetByUID(playerUID);
122 if (p == NULL || !p->IsLocal)
123 {
124 // This update was for a non-local player; abort
125 return -1;
126 }
127 // Note: player UIDs divided by MAX_LOCAL_PLAYERS per client
128 return playerUID % MAX_LOCAL_PLAYERS;
129}
130static void MergePopups(HUDNumPopup *dst, const HUDNumPopup src)
131{
132 // Combine popup amounts
133 if (dst->Timer <= 0)
134 {
135 // Old popup finished; simply replace with new
136 dst->Amount = src.Amount;
137 }
138 else
139 {
140 // Add the updates
141 dst->Amount += src.Amount;
142 }
143 dst->Timer = src.Timer;
144 dst->TimerMax = src.TimerMax;
145 dst->u.PlayerUID = src.u.PlayerUID;
146}

148static void NumPopupUpdate(HUDNumPopup *p, const int ms);
149void HUDPopupsUpdate(HUDNumPopups *popups, const int ms)
150{
151 for (int i = 0; i < MAX_LOCAL_PLAYERS; i++)
152 {
153 NumPopupUpdate(&popups->score[i], ms);
154 }
155 CA_FOREACH(HUDNumPopup, p, popups->objective)
156 NumPopupUpdate(p, ms);
157 CA_FOREACH_END()
158}
159static void NumPopupUpdate(HUDNumPopup *p, const int ms)
160{
161 if (p->Timer > 0)
162 {
163 p->Timer -= ms;
164 }
165}

185static void DrawScoreUpdate(const HUDNumPopup *u, const int flags)
186{
187 if (!IsScoreNeeded(gCampaign.Entry.Mode))
188 {
189 return;
190 }
191 if (u->Amount == 0)
192 {
193 return;
194 }
195 const PlayerData *p = PlayerDataGetByUID(u->u.PlayerUID);
196 if (!IsPlayerAlive(p)) return;
197 const int rowHeight = 1 + FontH();
198 const int y = 5 + rowHeight;
199 DrawNumUpdate(u, "Score: %d", p->Stats.Score, Vec2iNew(5, y), flags);
200}
201// Parameters that define how the numeric update is animated
202// The update animates in the following phases:
203// 1. Pop up from text position to slightly above
204// 2. Fall down from slightly above back to text position
205// 3. Persist over text position
206#define NUM_UPDATE_POP_UP_DURATION_MS 100
207#define NUM_UPDATE_FALL_DOWN_DURATION_MS 100
208#define NUM_UPDATE_POP_UP_HEIGHT 5
209static void DrawNumUpdate(
210 const HUDNumPopup *p,
211 const char *formatText, const int currentValue, Vec2i pos, const int flags)
212{
213 if (p->Timer <= 0 || p->Amount == 0)
214 {
215 return;
216 }
217 color_t color = p->Amount > 0 ? colorGreen : colorRed;
218
219 char s[50];
220 if (!(flags & HUDFLAGS_PLACE_RIGHT))
221 {
222 // Find the right position to draw the popup
223 // Make sure the popup is displayed lined up with the lowest digits
224 // Find the position of where the normal text is displayed,
225 // and move to its right
226 sprintf(s, formatText, currentValue);
227 pos.x += FontStrW(s);
228 // Then find the size of the popup, and move left
229 sprintf(s, "%s%d", p->Amount > 0 ? "+" : "", p->Amount);
230 pos.x -= FontStrW(s);
231 // The final position should ensure the score popup's lowest digit
232 // lines up with the normal score's lowest digit
233 }
234 else
235 {
236 sprintf(s, "%s%d", p->Amount > 0 ? "+" : "", p->Amount);
237 }
238
239 // Now animate the popup based on its stage
240 int timer = p->TimerMax - p->Timer;
241 if (timer < NUM_UPDATE_POP_UP_DURATION_MS)
242 {
243 // popup is still popping up
244 // calculate height
245 int popupHeight =
246 timer * NUM_UPDATE_POP_UP_HEIGHT / NUM_UPDATE_POP_UP_DURATION_MS;
247 pos.y -= popupHeight;
248 }
249 else if (timer <
250 NUM_UPDATE_POP_UP_DURATION_MS + NUM_UPDATE_FALL_DOWN_DURATION_MS)
251 {
252 // popup is falling down
253 // calculate height
254 timer -= NUM_UPDATE_POP_UP_DURATION_MS;
255 timer = NUM_UPDATE_FALL_DOWN_DURATION_MS - timer;
256 int popupHeight =
257 timer * NUM_UPDATE_POP_UP_HEIGHT / NUM_UPDATE_FALL_DOWN_DURATION_MS;
258 pos.y -= popupHeight;
259 }
260 else
261 {
262 // Change alpha so that the popup fades away
263 color.a = (Uint8)(p->Timer * 255 / p->TimerMax);
264 }
265
266 FontOpts opts = FontOptsNew();
267 if (flags & HUDFLAGS_PLACE_RIGHT)
268 {
269 opts.HAlign = ALIGN_END;
270 }
271 if (flags & HUDFLAGS_PLACE_BOTTOM)
272 {
273 opts.VAlign = ALIGN_END;
274 pos.y += BOTTOM_PADDING;
275 }
276 opts.Area = gGraphicsDevice.cachedConfig.Res;
277 opts.Pad = pos;
278 opts.Mask = color;
279 opts.Blend = true;
280 FontStrOpt(s, Vec2iZero(), opts);
281}
