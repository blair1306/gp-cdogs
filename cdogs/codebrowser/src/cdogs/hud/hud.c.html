68void HUDInit(
69 HUD *hud,
70 GraphicsDevice *device,
71 struct MissionOptions *mission)
72{
73 memset(hud, 0, sizeof *hud);
74 hud->mission = mission;
75 strcpy(hud->message, "");
76 hud->messageTicks = 0;
77 hud->device = device;
78 FPSCounterInit(&hud->fpsCounter);
79 WallClockInit(&hud->clock);
80 HUDNumPopupsInit(&hud->numPopups, mission);
81 for (int i = 0; i < MAX_LOCAL_PLAYERS; i++)
82 {
83 HealthGaugeInit(&hud->healthGauges[i]);
84 }
85 hud->showExit = false;
86}
87void HUDTerminate(HUD *hud)
88{
89 HUDNumPopupsTerminate(&hud->numPopups);
90}
91
92void HUDDisplayMessage(HUD *hud, const char *msg, int ticks)
93{
94 strcpy(hud->message, msg);
95 hud->messageTicks = ticks;
96}
97
98void HUDUpdate(HUD *hud, int ms)
99{
100 if (hud->messageTicks >= 0)
101 {
102 hud->messageTicks -= ms;
103 if (hud->messageTicks < 0)
104 {
105 hud->messageTicks = 0;
106 }
107 }
108 FPSCounterUpdate(&hud->fpsCounter, ms);
109 WallClockUpdate(&hud->clock, ms);
110 HUDPopupsUpdate(&hud->numPopups, ms);
111 int idx = 0;
112 for (int i = 0; i < (int)gPlayerDatas.size; i++, idx++)
113 {
114 const PlayerData *p = CArrayGet(&gPlayerDatas, i);
115 if (!IsPlayerScreen(p))
116 {
117 idx--;
118 continue;
119 }
120 const TActor *a = ActorGetByUID(p->ActorUID);
121 HealthGaugeUpdate(&hud->healthGauges[i], a, ms);
122 }
123}

126// Draw a gauge with an outer background and inner level
127// +--------------+
128// |XXXXXXXX| |
129// +--------------+
130void HUDDrawGauge(
131 GraphicsDevice *device,
132 Vec2i pos, const Vec2i size, const int innerWidth,
133 const color_t barColor, const color_t backColor,
134 const FontAlign hAlign, const FontAlign vAlign)
135{
136 const Vec2i offset = Vec2iUnit();
137 Vec2i barPos = Vec2iAdd(pos, offset);
138 const Vec2i barSize = Vec2iNew(MAX(0, innerWidth - 2), size.y - 2);
139 if (hAlign == ALIGN_END)
140 {
141 int w = device->cachedConfig.Res.x;
142 pos.x = w - pos.x - size.x;
143 barPos.x = w - barPos.x - barSize.x;
144 }
145 if (vAlign == ALIGN_END)
146 {
147 int h = device->cachedConfig.Res.y;
148 pos.y = h - pos.y - size.y;
149 barPos.y = h - barPos.y - barSize.y;
150 }
151 if (backColor.a > 0)
152 {
153 DrawRectangle(device, pos, size, backColor, DRAW_FLAG_ROUNDED);
154 }
155 DrawRectangle(device, barPos, barSize, barColor, 0);
156}

158static void DrawWeaponStatus(
159 HUD *hud, const TActor *actor, Vec2i pos,
160 const FontAlign hAlign, const FontAlign vAlign)
161{
162 const Weapon *weapon = ActorGetGun(actor);
163
164 // Draw gun icon, and allocate padding to draw the gun icon
165 const GunDescription *g = ActorGetGun(actor)->Gun;
166 const Vec2i iconPos = Vec2iAligned(
167 Vec2iNew(pos.x - 2, pos.y - 2),
168 g->Icon->size, hAlign, vAlign, gGraphicsDevice.cachedConfig.Res);
169 Blit(&gGraphicsDevice, g->Icon, iconPos);
170
171 // Draw gauge if ammo or reloading
172 const bool useAmmo =
173 ConfigGetBool(&gConfig, "Game.Ammo") && weapon->Gun->AmmoId >= 0;
174 const Ammo *ammo =
175 useAmmo ? AmmoGetById(&gAmmo, weapon->Gun->AmmoId) : NULL;
176 const int amount = useAmmo ? ActorGunGetAmmo(actor, weapon) : 0;
177 if (useAmmo || weapon->lock > 0)
178 {
179 const Vec2i gaugePos = Vec2iAdd(pos, Vec2iNew(-1 + GUN_ICON_PAD, -1));
180 const Vec2i size = Vec2iNew(GAUGE_WIDTH - GUN_ICON_PAD, FontH() + 5);
181 const int maxLock = weapon->Gun->Lock;
182 color_t barColor;
183 const double reloadProgressColorMod = 0.5 +
184 0.5 * (weapon->lock / (double) maxLock);
185 HSV hsv = { 0.0, 1.0, reloadProgressColorMod };
186 barColor = ColorTint(colorWhite, hsv);
187 int innerWidth;
188 color_t backColor = { 128, 128, 128, 255 };
189 if (maxLock == 0 || weapon->lock == 0)
190 {
191 innerWidth = 0;
192 }
193 else
194 {
195 innerWidth = MAX(1, size.x * (maxLock - weapon->lock) / maxLock);
196 }
197 HUDDrawGauge(
198 hud->device, gaugePos, size, innerWidth, barColor, backColor,
199 hAlign, vAlign);
200
201 if (useAmmo)
202 {
203 // Draw ammo level as inner mini-gauge, no background
204 const int yOffset = 8;
205 const Vec2i gaugeAmmoPos =
206 Vec2iNew(gaugePos.x, gaugePos.y + yOffset);
207 const Vec2i gaugeAmmoSize = Vec2iNew(size.x, size.y - yOffset);
208 const int ammoGaugeWidth =
209 MAX(1, gaugeAmmoSize.x * amount / ammo->Max);
210 HUDDrawGauge(
211 hud->device, gaugeAmmoPos, gaugeAmmoSize, ammoGaugeWidth,
212 colorBlue, colorTransparent, hAlign, vAlign);
213 }
214 }
215 FontOpts opts = FontOptsNew();
216 opts.HAlign = hAlign;
217 opts.VAlign = vAlign;
218 opts.Area = gGraphicsDevice.cachedConfig.Res;
219 opts.Pad = Vec2iNew(pos.x + GUN_ICON_PAD, pos.y);
220 char buf[128];
221 if (useAmmo && weapon->Gun->AmmoId >= 0)
222 {
223 // Include ammo counter
224 sprintf(buf, "%s %d/%d",
225 weapon->Gun->name,
226 ActorGunGetAmmo(actor, weapon),
227 AmmoGetById(&gAmmo, weapon->Gun->AmmoId)->Max);
228
229 // If low / no ammo, draw text with different colours, flashing
230 const int fps = ConfigGetInt(&gConfig, "Game.FPS");
231 if (amount == 0)
232 {
233 // No ammo; fast flashing
234 const int pulsePeriod = fps / 4;
235 if ((gMission.time % pulsePeriod) < (pulsePeriod / 2))
236 {
237 opts.Mask = colorRed;
238 }
239 }
240 else if (AmmoIsLow(ammo, amount))
241 {
242 // Low ammo; slow flashing
243 const int pulsePeriod = fps / 2;
244 if ((gMission.time % pulsePeriod) < (pulsePeriod / 2))
245 {
246 opts.Mask = colorOrange;
247 }
248 }
249 }
250 else
251 {
252 strcpy(buf, weapon->Gun->name);
253 }
254 FontStrOpt(buf, Vec2iZero(), opts);
255}

257static void DrawLives(
258 const GraphicsDevice *device, const PlayerData *player, const Vec2i pos,
259 const FontAlign hAlign, const FontAlign vAlign)
260{
261 const int xStep = hAlign == ALIGN_START ? 10 : -10;
262 const Vec2i offset = Vec2iNew(2, 5);
263 Vec2i drawPos = Vec2iAdd(pos, offset);
264 if (hAlign == ALIGN_END)
265 {
266 const int w = device->cachedConfig.Res.x;
267 drawPos.x = w - drawPos.x - offset.x;
268 }
269 if (vAlign == ALIGN_END)
270 {
271 const int h = device->cachedConfig.Res.y;
272 drawPos.y = h - drawPos.y - 1;
273 }
274 for (int i = 0; i < player->Lives; i++)
275 {
276 DrawHead(&player->Char, DIRECTION_DOWN, drawPos);
277 drawPos.x += xStep;
278 }
279}

281#define AUTOMAP_PADDING 5
282#define AUTOMAP_SIZE 45
283static void DrawRadar(
284 GraphicsDevice *device, const TActor *p,
285 const int flags, const bool showExit)
286{
287 Vec2i pos = Vec2iZero();
288 int w = device->cachedConfig.Res.x;
289 int h = device->cachedConfig.Res.y;
290
291 if (!p)
292 {
293 return;
294 }
295
296 // Possible map positions:
297 // top-right (player 1 only)
298 // top-left (player 2 only)
299 // bottom-right (player 3 only)
300 // bottom-left (player 4 only)
301 // top-left-of-middle (player 1 when two players)
302 // top-right-of-middle (player 2 when two players)
303 // bottom-left-of-middle (player 3)
304 // bottom-right-of-middle (player 4)
305 // top (shared screen)
306 if (flags & HUDFLAGS_HALF_SCREEN)
307 {
308 // two players
309 pos.y = AUTOMAP_PADDING;
310 if (flags & HUDFLAGS_PLACE_RIGHT)
311 {
312 // player 2
313 pos.x = w / 2 + AUTOMAP_PADDING;
314 }
315 else
316 {
317 // player 1
318 pos.x = w / 2 - AUTOMAP_SIZE - AUTOMAP_PADDING;
319 }
320 }
321 else if (flags & HUDFLAGS_QUARTER_SCREEN)
322 {
323 // four players
324 if (flags & HUDFLAGS_PLACE_RIGHT)
325 {
326 // player 2 or 4
327 pos.x = w / 2 + AUTOMAP_PADDING;
328 }
329 else
330 {
331 // player 1 or 3
332 pos.x = w / 2 - AUTOMAP_SIZE - AUTOMAP_PADDING;
333 }
334
335 if (flags & HUDFLAGS_PLACE_BOTTOM)
336 {
337 // player 3 or 4
338 pos.y = h - AUTOMAP_SIZE - AUTOMAP_PADDING;
339 }
340 else
341 {
342 // player 1 or 2
343 pos.y = AUTOMAP_PADDING;
344 }
345 }
346 else
347 {
348 // one player
349 if (flags & HUDFLAGS_PLACE_RIGHT)
350 {
351 // player 2 or 4
352 pos.x = AUTOMAP_PADDING;
353 }
354 else
355 {
356 // player 1 or 3
357 pos.x = w - AUTOMAP_SIZE - AUTOMAP_PADDING;
358 }
359
360 if (flags & HUDFLAGS_PLACE_BOTTOM)
361 {
362 // player 3 or 4
363 pos.y = h - AUTOMAP_SIZE - AUTOMAP_PADDING;
364 }
365 else
366 {
367 // player 1 or 2
368 pos.y = AUTOMAP_PADDING;
369 }
370 }
371
372 if (!Vec2iIsZero(pos))
373 {
374 Vec2i playerPos = Vec2iNew(
375 p->tileItem.x / TILE_WIDTH, p->tileItem.y / TILE_HEIGHT);
376 AutomapDrawRegion(
377 &gMap,
378 pos,
379 Vec2iNew(AUTOMAP_SIZE, AUTOMAP_SIZE),
380 playerPos,
381 AUTOMAP_FLAGS_MASK,
382 showExit);
383 }
384}

404// Draw player's score, health etc.
405static void DrawPlayerStatus(
406 HUD *hud, const PlayerData *data, const TActor *p,
407 const int flags, const Rect2i r, const HealthGauge *hg)
408{
409 if (p != NULL)
410 {
411 DrawObjectiveCompass(
412 hud->device, Vec2iFull2Real(p->Pos), r, hud->showExit);
413 }
414
415 Vec2i pos = Vec2iNew(5, 5);
416
417 FontOpts opts = FontOptsNew();
418 if (flags & HUDFLAGS_PLACE_RIGHT)
419 {
420 opts.HAlign = ALIGN_END;
421 }
422 if (flags & HUDFLAGS_PLACE_BOTTOM)
423 {
424 opts.VAlign = ALIGN_END;
425 pos.y += BOTTOM_PADDING;
426 }
427 opts.Area = gGraphicsDevice.cachedConfig.Res;
428 opts.Pad = pos;
429 FontStrOpt(data->name, Vec2iZero(), opts);
430
431 const int rowHeight = 1 + FontH();
432 pos.y += rowHeight;
433 char s[50];
434 if (IsScoreNeeded(gCampaign.Entry.Mode))
435 {
436 if (ConfigGetBool(&gConfig, "Game.Ammo"))
437 {
438 // Display money instead of ammo
439 sprintf(s, "Cash: $%d", data->Stats.Score);
440 }
441 else
442 {
443 sprintf(s, "Score: %d", data->Stats.Score);
444 }
445 }
446 else
447 {
448 s[0] = 0;
449 }
450 if (p)
451 {
452 // Score/money
453 opts.Pad = pos;
454 FontStrOpt(s, Vec2iZero(), opts);
455
456 // Health
457 pos.y += rowHeight;
458 HealthGaugeDraw(hg, hud->device, p, pos, opts.HAlign, opts.VAlign);
459
460 // Lives
461 pos.y += rowHeight;
462 DrawLives(hud->device, data, pos, opts.HAlign, opts.VAlign);
463
464 // Weapon
465 pos.y += rowHeight + LIVES_ROW_EXTRA_Y;
466 DrawWeaponStatus(hud, p, pos, opts.HAlign, opts.VAlign);
467 }
468 else
469 {
470 opts.Pad = pos;
471 FontStrOpt(s, Vec2iZero(), opts);
472 }
473
474 if (ConfigGetBool(&gConfig, "Interface.ShowHUDMap") &&
475 !(flags & HUDFLAGS_SHARE_SCREEN) &&
476 IsAutoMapEnabled(gCampaign.Entry.Mode))
477 {
478 DrawRadar(hud->device, p, flags, hud->showExit);
479 }
480}

483static void DrawCompassArrow(
484 GraphicsDevice *g, Rect2i r, Vec2i pos, Vec2i playerPos, color_t mask,
485 const char *label);
486static void DrawObjectiveCompass(
487 GraphicsDevice *g, Vec2i playerPos, Rect2i r, bool showExit)
488{
489 // Draw exit position
490 if (showExit)
491 {
492 DrawCompassArrow(
493 g, r, MapGetExitPos(&gMap), playerPos, colorGreen, "Exit");
494 }
495
496 // Draw objectives
497 Map *map = &gMap;
498 Vec2i tilePos;
499 for (tilePos.y = 0; tilePos.y < map->Size.y; tilePos.y++)
500 {
501 for (tilePos.x = 0; tilePos.x < map->Size.x; tilePos.x++)
502 {
503 Tile *tile = MapGetTile(map, tilePos);
504 CA_FOREACH(ThingId, tid, tile->things)
505 TTileItem *ti = ThingIdGetTileItem(tid);
506 if (!(ti->flags & TILEITEM_OBJECTIVE))
507 {
508 continue;
509 }
510 const int objective = ObjectiveFromTileItem(ti->flags);
511 const Objective *o =
512 CArrayGet(&gMission.missionData->Objectives, objective);
513 if (o->Flags & OBJECTIVE_HIDDEN)
514 {
515 continue;
516 }
517 if (!(o->Flags & OBJECTIVE_POSKNOWN) && !tile->isVisited)
518 {
519 continue;
520 }
521 DrawCompassArrow(
522 g, r, Vec2iNew(ti->x, ti->y), playerPos, o->color, NULL);
523 CA_FOREACH_END()
524 }
525 }
526}

528#define COMP_SATURATE_DIST 350
529static void DrawCompassArrow(
530 GraphicsDevice *g, Rect2i r, Vec2i pos, Vec2i playerPos, color_t mask,
531 const char *label)
532{
533 Vec2i compassV = Vec2iMinus(pos, playerPos);
534 // Don't draw if objective is on screen
535 if (abs(pos.x - playerPos.x) < r.Size.x / 2 &&
536 abs(pos.y - playerPos.y) < r.Size.y / 2)
537 {
538 return;
539 }
540 // Saturate according to dist from screen edge
541 int xDist = abs(pos.x - playerPos.x) - r.Size.x / 2;
542 int yDist = abs(pos.y - playerPos.y) - r.Size.y / 2;
543 int lDist;
544 xDist > yDist ? lDist = xDist: (lDist = yDist);
545 HSV hsv = { -1.0, 1.0,
546 2.0 - 1.5 * MIN(lDist, COMP_SATURATE_DIST) / COMP_SATURATE_DIST };
547 mask = ColorTint(mask, hsv);
548 Vec2i textPos = Vec2iZero();
549 // Find which edge of screen is the best
550 bool hasDrawn = false;
551 if (compassV.x != 0)
552 {
553 double sx = r.Size.x / 2.0 / compassV.x;
554 int yInt = (int)floor(fabs(sx) * compassV.y + 0.5);
555 if (yInt >= -r.Size.y / 2 && yInt <= r.Size.y / 2)
556 {
557 // Intercepts either left or right side
558 hasDrawn = true;
559 if (compassV.x > 0)
560 {
561 // right edge
562 textPos = Vec2iNew(
563 r.Pos.x + r.Size.x, r.Pos.y + r.Size.y / 2 + yInt);
564 const Pic *p = PicManagerGetPic(&gPicManager, "arrow_right");
565 Vec2i drawPos = Vec2iNew(
566 textPos.x - p->size.x, textPos.y - p->size.y / 2);
567 BlitMasked(g, p, drawPos, mask, true);
568 }
569 else if (compassV.x < 0)
570 {
571 // left edge
572 textPos = Vec2iNew(r.Pos.x, r.Pos.y + r.Size.y / 2 + yInt);
573 const Pic *p = PicManagerGetPic(&gPicManager, "arrow_left");
574 Vec2i drawPos = Vec2iNew(textPos.x, textPos.y - p->size.y / 2);
575 BlitMasked(g, p, drawPos, mask, true);
576 }
577 }
578 }
579 if (!hasDrawn && compassV.y != 0)
580 {
581 double sy = r.Size.y / 2.0 / compassV.y;
582 int xInt = (int)floor(fabs(sy) * compassV.x + 0.5);
583 if (xInt >= -r.Size.x / 2 && xInt <= r.Size.x / 2)
584 {
585 // Intercepts either top or bottom side
586 if (compassV.y > 0)
587 {
588 // bottom edge
589 textPos = Vec2iNew(
590 r.Pos.x + r.Size.x / 2 + xInt, r.Pos.y + r.Size.y);
591 const Pic *p = PicManagerGetPic(&gPicManager, "arrow_down");
592 Vec2i drawPos = Vec2iNew(
593 textPos.x - p->size.x / 2, textPos.y - p->size.y);
594 BlitMasked(g, p, drawPos, mask, true);
595 }
596 else if (compassV.y < 0)
597 {
598 // top edge
599 textPos = Vec2iNew(r.Pos.x + r.Size.x / 2 + xInt, r.Pos.y);
600 const Pic *p = PicManagerGetPic(&gPicManager, "arrow_up");
601 Vec2i drawPos = Vec2iNew(textPos.x - p->size.x / 2, textPos.y);
602 BlitMasked(g, p, drawPos, mask, true);
603 }
604 }
605 }
606 if (label && strlen(label) > 0)
607 {
608 Vec2i textSize = FontStrSize(label);
609 // Center the text around the target position
610 textPos.x -= textSize.x / 2;
611 textPos.y -= textSize.y / 2;
612 // Make sure the text is inside the screen
613 int padding = 8;
614 textPos.x = MAX(textPos.x, r.Pos.x + padding);
615 textPos.x = MIN(textPos.x, r.Pos.x + r.Size.x - textSize.x - padding);
616 textPos.y = MAX(textPos.y, r.Pos.y + padding);
617 textPos.y = MIN(textPos.y, r.Pos.y + r.Size.y - textSize.y - padding);
618 FontStrMask(label, textPos, mask);
619 }
620}

659static void DrawPlayerAreas(HUD *hud)
660{
661 int flags = 0;
662 const int numPlayersCreen = GetNumPlayersScreen(NULL);
663
664 Rect2i r;
665 r.Size = Vec2iNew(
666 hud->device->cachedConfig.Res.x,
667 hud->device->cachedConfig.Res.y);
668 if (numPlayersCreen <= 1)
669 {
670 flags = 0;
671 }
672 else if (
673 ConfigGetEnum(&gConfig, "Interface.Splitscreen") == SPLITSCREEN_NEVER)
674 {
675 flags |= HUDFLAGS_SHARE_SCREEN;
676 }
677 else if (numPlayersCreen == 2)
678 {
679 r.Size.x /= 2;
680 flags |= HUDFLAGS_HALF_SCREEN;
681 }
682 else if (numPlayersCreen == 3 || numPlayersCreen == 4)
683 {
684 r.Size.x /= 2;
685 r.Size.y /= 2;
686 flags |= HUDFLAGS_QUARTER_SCREEN;
687 }
688 else
689 {
690 assert(0 && "not implemented");
691 }
692
693 int idx = 0;
694 for (int i = 0; i < (int)gPlayerDatas.size; i++, idx++)
695 {
696 const PlayerData *p = CArrayGet(&gPlayerDatas, i);
697 if (!IsPlayerScreen(p))
698 {
699 idx--;
700 continue;
701 }
702 int drawFlags = flags;
703 r.Pos = Vec2iZero();
704 if (idx & 1)
705 {
706 r.Pos.x = r.Size.x;
707 drawFlags |= HUDFLAGS_PLACE_RIGHT;
708 }
709 if (idx >= 2)
710 {
711 r.Pos.y = r.Size.y;
712 drawFlags |= HUDFLAGS_PLACE_BOTTOM;
713 }
714 TActor *player = NULL;
715 if (IsPlayerAlive(p))
716 {
717 player = ActorGetByUID(p->ActorUID);
718 }

843static void DrawMissionState(HUD *hud)
844{
845 char s[50];
846 const int numPlayersAlive =
847 GetNumPlayers(PLAYER_ALIVE_OR_DYING, false, false);
848
849 switch (hud->mission->state)
850 {
851 case MISSION_STATE_WAITING:
852 FontStrCenter("Waiting for players...");
853 break;
854 case MISSION_STATE_PLAY:
855 if (numPlayersAlive == 0 && AreAllPlayersDeadAndNoLives())
856 {
857 if (gPlayerDatas.size == 0)
858 {
859 FontStrCenter("Waiting for players...");
860 }
861 else if (!IsPVP(gCampaign.Entry.Mode))
862 {
863 FontStrCenter("Game Over!");
864 }
865 else
866 {
867 FontStrCenter("All Kill!");
868 }
869 }
870 else if (MissionNeedsMoreRescuesInExit(&gMission))
871 {
872 FontStrCenter("More rescues needed");
873 }
874 break;
875 case MISSION_STATE_PICKUP:
876 {
877 int timeLeft = gMission.pickupTime + PICKUP_LIMIT - gMission.time;
878 sprintf(s, "Pickup in %d seconds\n",
879 (timeLeft + (FPS_FRAMELIMIT - 1)) / FPS_FRAMELIMIT);
880 FontStrCenter(s);
881 }
882 break;
883 default:
884 CASSERT(false, "unknown mission state");
885 break;
886 }
887}
