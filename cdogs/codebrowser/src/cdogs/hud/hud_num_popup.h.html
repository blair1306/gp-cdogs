36// Numeric popup for health and score
37// Displays as a small pop-up coloured text overlay
38typedef struct
39{
40 union
41 {
42 int PlayerUID;
43 int ObjectiveIndex;
44 } u;
45 int Amount;
46 // Number of milliseconds that this update will last
47 int TimerMax;
48 int Timer;
49} HUDNumPopup;
50
51typedef struct
52{
53 HUDNumPopup score[MAX_LOCAL_PLAYERS];
54 CArray objective; // of HUDNumPopup, one per objective
55} HUDNumPopups;

57void HUDNumPopupsInit(
58 HUDNumPopups *popups, const struct MissionOptions *mission);

60
61typedef enum

63 NUMBER_POPUP_SCORE,
64 NUMBER_POPUP_OBJECTIVE
