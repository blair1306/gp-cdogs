35// How many ticks to stay in one confusion state
36#define CONFUSION_STATE_TICKS_MIN 25
37#define CONFUSION_STATE_TICKS_RANGE 25
38
39#define DISTANCE_TO_KEEP_OUT_OF_WAY SQUARED(4*16)

42static int AICoopGetCmdNormal(TActor *actor);
43int AICoopGetCmd(TActor *actor, const int ticks)
44{
45 // Create AI context if it isn't there already
46 // This is because co-op AIs don't have the character bot property
47 // TODO: don't require this lazy initialisation
48 if (actor->aiContext == NULL)
49 {
50 actor->aiContext = AIContextNew();
51 }
52
53 int cmd = 0;
54
55 // Special decision tree for confusion
56 // - If confused, randomly:
57 // - Perform the right action (reverse directions)
58 // - Perform a random command (except for switching weapons)
59 // - And never slide
60 if (actor->confused)
61 {
62 AIConfusionState *s = &actor->aiContext->ConfusionState;
63 // Check delay and change state
64 actor->aiContext->Delay = MAX(0, actor->aiContext->Delay - ticks);
65 if (actor->aiContext->Delay == 0)
66 {
67 actor->aiContext->Delay =
68 CONFUSION_STATE_TICKS_MIN +
69 (rand() % CONFUSION_STATE_TICKS_RANGE);
70 if (s->Type == AI_CONFUSION_CONFUSED)
71 {
72 s->Type = AI_CONFUSION_CORRECT;
73 }
74 else
75 {
76 ActorSetAIState(actor, AI_STATE_CONFUSED);
77 s->Type = AI_CONFUSION_CONFUSED;
78 // Generate the confused action
79 s->Cmd = rand() &
80 (CMD_LEFT | CMD_RIGHT | CMD_UP | CMD_DOWN |
81 CMD_BUTTON1 | CMD_BUTTON2);
82 }
83 }
84 // Choose confusion action based on state
85 switch (s->Type)
86 {
87 case AI_CONFUSION_CONFUSED:
88 // Use canned random command
89 // TODO: add state for this action
90 cmd = s->Cmd;
91 break;
92 case AI_CONFUSION_CORRECT:
93 // Reverse directions so they are correct
94 cmd = CmdGetReverse(AICoopGetCmdNormal(actor));
95 break;
96 default:
97 CASSERT(false, "Unknown state");
98 break;
99 }
100 // Don't slide
101 if ((cmd & CMD_BUTTON2) && CMD_HAS_DIRECTION(cmd))
102 {
103 cmd &= ~CMD_BUTTON2;
104 }
105 }
106 else
107 {
108 // Act normally
109 cmd = AICoopGetCmdNormal(actor);
110 }
111 return cmd;
112}

114static int SmartGoto(TActor *actor, Vec2i pos, int minDistance2);
115static bool TryCompleteNearbyObjective(
116 TActor *actor, const TActor *closestPlayer,
117 const int distanceTooFarFromPlayer, int *cmdOut);
118static int AICoopGetCmdNormal(TActor *actor)
119{
120 // Use decision tree to command the AI
121 // - Move away from dangerous bullets
122 // - Check weapons and ammo
123 // - If too far away from nearest player
124 // - Go to nearest player
125 // - else if blocking a nearby player
126 // - move out of the way
127 // - else if closest enemy is close enough
128 // - Attack enemy
129 // - else
130 // - Go to nearest player if too far, or away if too close
131
132 const Vec2i actorRealPos = Vec2iFull2Real(actor->Pos);
133 const Vec2i actorTilePos = Vec2iToTile(actorRealPos);
134
135 // Look for dangerous bullets in a 1-tile radius
136 // These are bullets with the "HurtAlways" property true
137 Vec2i v;
138 Vec2i dangerBulletFullPos = Vec2iZero();
139 for (v.x = actorTilePos.x - 1;
140 v.x <= actorTilePos.x + 1 && Vec2iIsZero(dangerBulletFullPos);
141 v.x++)
142 {
143 for (v.y = actorTilePos.y - 1;
144 v.y <= actorTilePos.y + 1 && Vec2iIsZero(dangerBulletFullPos);
145 v.y++)
146 {
147 const Tile *t = MapGetTile(&gMap, v);
148 if (t == NULL) continue;
149 CA_FOREACH(const ThingId, tid, t->things)
150 // Only look for bullets
151 if (tid->Kind != KIND_MOBILEOBJECT) continue;
152 const TMobileObject *mo = CArrayGet(&gMobObjs, tid->Id);
153 if (mo->bulletClass->HurtAlways)
154 {
155 dangerBulletFullPos = Vec2iNew(mo->x, mo->y);
156 break;
157 }
158 CA_FOREACH_END()
159 }
160 }
161 // Run away if dangerous bullet found
162 if (!Vec2iIsZero(dangerBulletFullPos))
163 {
164 return AIRetreatFrom(actor, dangerBulletFullPos);
165 }
166
167 // Check the weapon for ammo
168 int lowAmmoGun = -1;
169 if (ConfigGetBool(&gConfig, "Game.Ammo"))
170 {
171 // Check all our weapons
172 // Prefer guns using ammo
173 // This is because unlimited ammo guns tend to be worse
174 int mostPreferredWeaponWithAmmo = -1;
175 bool mostPreferredWeaponHasAmmo = false;
176 CA_FOREACH(const Weapon, w, actor->guns)
177 const int ammoAmount = ActorGunGetAmmo(actor, w);
178 if ((mostPreferredWeaponWithAmmo == -1 && ammoAmount != 0) ||
179 (!mostPreferredWeaponHasAmmo && ammoAmount > 0))
180 {
181 // Note: -1 means the gun has unlimited ammo
182 mostPreferredWeaponWithAmmo = _ca_index;
183 mostPreferredWeaponHasAmmo = ammoAmount > 0;
184 }
185 if (w->Gun->AmmoId != -1)
186 {
187 const Ammo *ammo = AmmoGetById(&gAmmo, w->Gun->AmmoId);
188 if (lowAmmoGun == -1 &&
189 ammoAmount < ammo->Amount * AMMO_STARTING_MULTIPLE)
190 {
191 lowAmmoGun = _ca_index;
192 }
193 }
194 CA_FOREACH_END()
195
196 // If we're out of ammo, switch to one with ammo
197 // Prioritise our selection based on weapon order, i.e.
198 // prefer first weapon over second over third etc.
199 if (mostPreferredWeaponWithAmmo != -1 &&
200 mostPreferredWeaponWithAmmo != actor->gunIndex)
201 {
202 return actor->lastCmd == CMD_BUTTON2 ? 0 : CMD_BUTTON2;
203 }
204 }
205
206 // If we're over a weapon pickup, check if we want to pick it up
207 // Pick up if we have:
208 // - a weapon that is low on ammo and don't yet have the gun, or
209 // - if we have a free slot
210 if (actor->aiContext->OnGunId != -1)
211 {
212 const GunDescription *gun =
213 IdGunDescription(actor->aiContext->OnGunId);
214 if ((lowAmmoGun != -1 && !ActorHasGun(actor, gun)) ||
215 actor->guns.size < MAX_WEAPONS)
216 {
217 actor->aiContext->OnGunId = -1;
218 // Pick it up
219 // TODO: this will not guarantee the new gun will replace
220 // the low ammo gun; weapon management?
221 return actor->lastCmd == CMD_BUTTON2 ? 0 : CMD_BUTTON2;
222 }
223 actor->aiContext->OnGunId = -1;
224 }
225
226 // Find nearby players, and move out of the way if any of them are facing
227 // directly at us
228 // Follow the closest player with a lower UID
229 const TActor *closestPlayer = NULL;
230 int minDistance2 = -1;
231 if (!IsPVP(gCampaign.Entry.Mode))
232 {
233 for (int uid = 0; uid < actor->PlayerUID; uid++)
234 {
235 const PlayerData *pd = PlayerDataGetByUID(uid);
236 if (pd == NULL || !IsPlayerAlive(pd)) continue;
237 const TActor *p = ActorGetByUID(pd->ActorUID);
238 const int distance2 =
239 DistanceSquared(actorRealPos, Vec2iFull2Real(p->Pos));
240 if (!closestPlayer || distance2 < minDistance2)
241 {
242 minDistance2 = distance2;
243 closestPlayer = p;
244 }
245 // TODO: AIIsFacing might be too generous?
246 if (distance2 < DISTANCE_TO_KEEP_OUT_OF_WAY &&
247 AIIsFacing(p, actor->Pos, p->direction))
248 {
249 // Move out of the way
250 return AIMoveAwayFromLine(actor->Pos, p->Pos, p->direction);
251 }
252 }
253 }
254
255 // Set distance we want to stay within the lead player
256 int distanceTooFarFromPlayer = 8;
257 // If player is exiting, we want to be very close to the player
258 if (closestPlayer && closestPlayer->action == ACTORACTION_EXITING)
259 {
260 distanceTooFarFromPlayer = 2;
261 }
262 if (closestPlayer && minDistance2 > SQUARED(distanceTooFarFromPlayer*16))
263 {
264 ActorSetAIState(actor, AI_STATE_FOLLOW);
265 return SmartGoto(actor, Vec2iFull2Real(closestPlayer->Pos), minDistance2);
266 }
267
268 // Check if closest enemy is close enough, and visible
269 const TActor *closestEnemy = AIGetClosestVisibleEnemy(actor, true);
270 if (closestEnemy)
271 {
272 int minEnemyDistance = CHEBYSHEV_DISTANCE(
273 actor->Pos.x, actor->Pos.y,
274 closestEnemy->Pos.x, closestEnemy->Pos.y);
275 // Also only engage if there's a clear shot
276 if (minEnemyDistance > 0 && minEnemyDistance < ((12 * 16) << 8) &&
277 AIHasClearShot(actorRealPos, Vec2iFull2Real(closestEnemy->Pos)))
278 {
279 ActorSetAIState(actor, AI_STATE_HUNT);
280 if (closestEnemy->uid != actor->aiContext->EnemyId)
281 {
282 // Engaging new enemy now
283 actor->aiContext->EnemyId = closestEnemy->uid;
284 actor->aiContext->GunRangeScalar = 1.0;
285 }
286 else
287 {
288 // Still attacking the same enemy; inch closer
289 actor->aiContext->GunRangeScalar *= 0.99;
290 }
291 // Move to the ideal distance for the weapon
292 int cmd = 0;
293 const GunDescription *gun = ActorGetGun(actor)->Gun;
294 const int gunRange = GunGetRange(gun);
295 const int distanceSquared = DistanceSquared(
296 Vec2iFull2Real(actor->Pos), Vec2iFull2Real(closestEnemy->Pos));
297 const bool canFire = gun->CanShoot && ActorGetGun(actor)->lock <= 0;
298 if ((double)distanceSquared >
299 SQUARED(gunRange * 3 / 4) * actor->aiContext->GunRangeScalar)
300 {
301 // Move towards the enemy, fire if able
302 // But don't bother firing if too far away
303
304#define MINIMUM_GUN_DISTANCE 30
305 // Cap the minimum distance for moving away; if we are extremely
306 // close to the target, fire anyway regardless of the gun range.
307 // This is to prevent us from not firing with a melee-range gun.
308 if (canFire &&
309 (distanceSquared < SQUARED(gunRange * 2) ||
310 distanceSquared > SQUARED(MINIMUM_GUN_DISTANCE)))
311 {
312 cmd = AIHunt(actor, closestEnemy->Pos) | CMD_BUTTON1;
313 }
314 else
315 {
316 // Track so that we end up in a favorable angle
317 cmd = AITrack(actor, closestEnemy->Pos);
318 }
319 }
320 else if ((double)distanceSquared <
321 SQUARED(gunRange * 3) * actor->aiContext->GunRangeScalar &&
322 !canFire)
323 {
324 // Move away from the enemy because we're too close
325 // Only move away if we can't fire; otherwise turn to fire
326 cmd = AIRetreatFrom(actor, closestEnemy->Pos);
327 }
328 else if (canFire)
329 {
330 // We're not too close and not too far; fire if able
331 cmd = AIHunt(actor, closestEnemy->Pos) | CMD_BUTTON1;
332 }
333 return cmd;
334 }
335 }
336
337 // Look for objectives nearby to complete
338 int cmd;
339 if (TryCompleteNearbyObjective(
340 actor, closestPlayer, distanceTooFarFromPlayer, &cmd))
341 {
342 return cmd;
343 }
344
345 // Otherwise, just go towards the closest player as long as we don't
346 // run into them, but keep away if we're too close
347 if (closestPlayer)
348 {
349 if (minDistance2 > SQUARED(2*16))
350 {
351 ActorSetAIState(actor, AI_STATE_FOLLOW);
352 return SmartGoto(
353 actor, Vec2iFull2Real(closestPlayer->Pos), minDistance2);
354 }
355 else if (minDistance2 < SQUARED(4*16/3))
356 {
357 return
358 CmdGetReverse(AIGotoDirect(actor->Pos, closestPlayer->Pos));
359 }
360 }
361
362 ActorSetAIState(actor, AI_STATE_IDLE);
363 return 0;
364}
365// Number of ticks to persist in trying to destroy an obstruction
366// before giving up and going around
367#define STUCK_TICKS 70
368// Goto with extra smarts:
369// - If clear path, slide
370// - If non-dangerous object blocking, shoot at it
371// - If stuck for a long time, pathfind around obstructing object
372static int SmartGoto(TActor *actor, Vec2i realPos, int minDistance2)
373{
374 int cmd = AIGoto(actor, realPos, true);
375 // Try to slide if there is a clear path and we are far enough away
376 if (CMD_HAS_DIRECTION(cmd) &&
377 AIHasClearPath(
378 Vec2iFull2Real(actor->Pos),
379 realPos,
380 !actor->aiContext->IsStuckTooLong) &&
381 minDistance2 > 7 * 7 * 16 * 16)
382 {
383 cmd |= CMD_BUTTON2;
384 }
385 // If running into safe object, and we're being blocked, shoot at it
386 TObject *o = AIGetObjectRunningInto(actor, cmd);
387 Vec2i tilePos = Vec2iToTile(Vec2iFull2Real(actor->Pos));
388 if (o && ObjIsDangerous(o) &&
389 Vec2iEqual(tilePos, actor->aiContext->LastTile))
390 {
391 cmd = AIGoto(
392 actor, Vec2iNew(o->tileItem.x, o->tileItem.y), true);
393 if (ActorGetGun(actor)->lock <= 0)
394 {
395 cmd |= CMD_BUTTON1;
396 }
397 }
398 // Check if we are stuck
399 if (Vec2iEqual(tilePos, actor->aiContext->LastTile))
400 {
401 actor->aiContext->Delay++;
402 if (actor->aiContext->Delay >= STUCK_TICKS)
403 {
404 // We've been stuck for too long
405 // Pathfind around it
406 // Make sure to reset the A* path if we are first realising
407 // we are stuck
408 if (!actor->aiContext->IsStuckTooLong)
409 {
410 actor->aiContext->Goto.IsFollowing = false;
411 }
412 actor->aiContext->IsStuckTooLong = true;
413 cmd = AIGoto(actor, realPos, !actor->aiContext->IsStuckTooLong);
414 }
415 }
416 else
417 {
418 actor->aiContext->Delay = 0;
419 actor->aiContext->IsStuckTooLong = false;
420 }
421 actor->aiContext->LastTile = tilePos;
422 return cmd;
423}

439 const Vec2i objRealPos, const Vec2i actorRealPos, const TActor *player,

441static int GotoObjective(TActor *actor, int objDistance);
442static bool TryCompleteNearbyObjective(
443 TActor *actor, const TActor *closestPlayer,
444 const int distanceTooFarFromPlayer, int *cmdOut)
445{
446 AIContext *context = actor->aiContext;
447 AIObjectiveState *objState = &context->ObjectiveState;
448 const Vec2i actorRealPos = Vec2iFull2Real(actor->Pos);
449
450 // Check to see if we are already attempting to complete an objective,
451 // and that it hasn't been updated since we last checked.
452 // If so keep following the path
453 if (context->State == AI_STATE_NEXT_OBJECTIVE)
454 {
455 bool hasNoUpdates = true;
456 switch (objState->Type)
457 {
458 case AI_OBJECTIVE_TYPE_KEY:
459 hasNoUpdates =
460 objState->LastDone == KeycardCount(gMission.KeyFlags);
461 break;
462 case AI_OBJECTIVE_TYPE_NORMAL:
463 hasNoUpdates =
464 objState->u.Obj && objState->LastDone == objState->u.Obj->done;
465 break;
466 case AI_OBJECTIVE_TYPE_KILL:
467 {
468 const TActor *target = ActorGetByUID(objState->u.UID);
469 hasNoUpdates = target->health > 0;
470 // Update target position
471 objState->Goal = Vec2iNew(
472 target->tileItem.x,
473 target->tileItem.y);
474 }
475 break;
476 case AI_OBJECTIVE_TYPE_PICKUP:
477 {
478 const Pickup *p = PickupGetByUID(objState->u.UID);
479 hasNoUpdates = p != NULL && p->isInUse;
480 }
481 break;
482 default:
483 // Do nothing
484 break;
485 }
486 if (hasNoUpdates)
487 {
488 int cmd = GotoObjective(actor, 0);
489 // If we can't go to the objective,
490 // one possibility is that we've reached where the objective was,
491 // but have nothing to do.
492 // Remember that actor objectives can move around!
493 if (CMD_HAS_DIRECTION(cmd))
494 {
495 *cmdOut = cmd;
496 return true;
497 }
498 }
499 }
500
501 // First, check if mission complete;
502 // if so (and there's a path) go to exit
503 const Vec2i exitPos = MapGetExitPos(&gMap);
504 if (CanCompleteMission(&gMission) && CanGetObjective(
505 exitPos, actorRealPos, closestPlayer, distanceTooFarFromPlayer))
506 {
507 ActorSetAIState(actor, AI_STATE_NEXT_OBJECTIVE);
508 objState->Type = AI_OBJECTIVE_TYPE_EXIT;
509 objState->Goal = exitPos;
510 *cmdOut = GotoObjective(actor, 0);
511 return true;
512 }
513
514 // Find all the objective/key locations, sort according to distance
515 // TODO: reuse this array, don't recreate it
516 CArray objectives;
517 FindObjectivesSortedByDistance(&objectives, actor, closestPlayer);
518
519 // Starting from the closest objectives, find one we can go to
520 CA_FOREACH(ClosestObjective, c, objectives)
521 if (CanGetObjective(
522 c->Pos, actorRealPos, closestPlayer, distanceTooFarFromPlayer))
523 {
524 ActorSetAIState(actor, AI_STATE_NEXT_OBJECTIVE);
525 objState->Type = c->Type;
526 objState->IsDestructible = c->IsDestructible;
527 switch (c->Type)
528 {
529 case AI_OBJECTIVE_TYPE_KEY:
530 objState->LastDone = KeycardCount(gMission.KeyFlags);
531 break;
532 case AI_OBJECTIVE_TYPE_NORMAL:
533 objState->u.Obj = c->u.Objective;
534 objState->LastDone = c->u.Objective->done;
535 break;
536 case AI_OBJECTIVE_TYPE_KILL:
537 objState->u.UID = c->u.UID;
538 break;
539 case AI_OBJECTIVE_TYPE_PICKUP:
540 objState->u.UID = c->u.UID;
541 break;
542 default:
543 // Do nothing
544 break;
545 }
546 objState->Goal = c->Pos;
547 *cmdOut = GotoObjective(actor, c->Distance);
548 return true;
549 }
550 CA_FOREACH_END()
551 return false;
552}
553static bool OnClosestPickupGun(
554 ClosestObjective *co, const Pickup *p,
555 const TActor *actor, const TActor *closestPlayer);
556static int CompareClosestObjective(const void *v1, const void *v2);
557static void FindObjectivesSortedByDistance(
558 CArray *objectives, const TActor *actor, const TActor *closestPlayer)
559{
560 const Vec2i actorRealPos = Vec2iFull2Real(actor->Pos);
561 CArrayInit(objectives, sizeof(ClosestObjective));
562
563 // If PVP, find the closest enemy and go to them
564 if (IsPVP(gCampaign.Entry.Mode))
565 {
566 const TActor *closestEnemy =
567 AIGetClosestVisibleEnemy(actor, true);
568 if (closestEnemy != NULL)
569 {
570 ClosestObjective co;
571 memset(&co, 0, sizeof co);
572 co.Pos = Vec2iNew(
573 closestEnemy->tileItem.x,
574 closestEnemy->tileItem.y);
575 co.IsDestructible = false;
576 co.Type = AI_OBJECTIVE_TYPE_KILL;
577 co.Distance = DistanceSquared(actorRealPos, co.Pos);
578 co.u.UID = closestEnemy->uid;
579 CArrayPushBack(objectives, &co);
580 }
581 }
582
583 // Look for pickups
584 CA_FOREACH(const Pickup, p, gPickups)
585 if (!p->isInUse)
586 {
587 continue;
588 }
589 ClosestObjective co;
590 memset(&co, 0, sizeof co);
591 co.Pos = Vec2iNew(p->tileItem.x, p->tileItem.y);
592 co.IsDestructible = false;
593 co.Type = AI_OBJECTIVE_TYPE_NORMAL;
594 switch (p->class->Type)
595 {
596 case PICKUP_KEYCARD:
597 co.Type = AI_OBJECTIVE_TYPE_KEY;
598 break;
599 case PICKUP_JEWEL:
600 break;
601 case PICKUP_HEALTH:
602 // Pick up if we are on low health, and lower than lead player
603 if (actor->health > ModeMaxHealth(gCampaign.Entry.Mode) / 4)
604 {
605 continue;
606 }
607 co.Type = AI_OBJECTIVE_TYPE_PICKUP;
608 co.u.UID = p->UID;
609 break;
610 case PICKUP_AMMO:
611 {
612 // Pick up if we use this ammo, have less ammo than starting,
613 // and lower than lead player, who uses the ammo
614 const int ammoId = p->class->u.Ammo.Id;
615 const Ammo *ammo = AmmoGetById(&gAmmo, ammoId);
616 const int ammoAmount = *(int *)CArrayGet(&actor->ammo, ammoId);
617 if (!ActorUsesAmmo(actor, ammoId) ||
618 ammoAmount > ammo->Amount * AMMO_STARTING_MULTIPLE ||
619 (closestPlayer != NULL &&
620 ActorUsesAmmo(closestPlayer, ammoId) &&
621 ammoAmount > *(int *)CArrayGet(&closestPlayer->ammo, ammoId)))
622 {
623 continue;
624 }
625 co.Type = AI_OBJECTIVE_TYPE_PICKUP;
626 co.u.UID = p->UID;
627 }
628 break;
629 case PICKUP_GUN:
630 if (!OnClosestPickupGun(&co, p, actor, closestPlayer))
631 {
632 continue;
633 }
634 break;
635 default:
636 // Not something we want to pick up
637 continue;
638 }
639 // Check if the pickup is actually accessible
640 // This is because random spawning may cause some pickups to be spawned
641 // in inaccessible areas
642 if (MapGetTile(&gMap, Vec2iToTile(co.Pos))->flags & MAPTILE_NO_WALK)
643 {
644 continue;
645 }
646 co.Distance = DistanceSquared(actorRealPos, co.Pos);
647 if (co.Type == AI_OBJECTIVE_TYPE_NORMAL)
648 {
649 const int objective = ObjectiveFromTileItem(p->tileItem.flags);
650 co.u.Objective =
651 CArrayGet(&gMission.missionData->Objectives, objective);
652 }
653 CArrayPushBack(objectives, &co);
654 CA_FOREACH_END()
655
656 // Look for destructibles
657 CA_FOREACH(const TObject, o, gObjs)
658 if (!o->isInUse)
659 {
660 continue;
661 }
662 ClosestObjective co;
663 memset(&co, 0, sizeof co);
664 co.Pos = Vec2iNew(o->tileItem.x, o->tileItem.y);
665 co.IsDestructible = true;
666 co.Type = AI_OBJECTIVE_TYPE_NORMAL;
667 if (!(o->tileItem.flags & TILEITEM_OBJECTIVE))
668 {
669 continue;
670 }
671 // Destructible objective; go towards it and fire
672 co.Distance = DistanceSquared(actorRealPos, co.Pos);
673 if (co.Type == AI_OBJECTIVE_TYPE_NORMAL)
674 {
675 const int objective = ObjectiveFromTileItem(o->tileItem.flags);
676 co.u.Objective =
677 CArrayGet(&gMission.missionData->Objectives, objective);
678 }
679 CArrayPushBack(objectives, &co);
680 CA_FOREACH_END()
681
682 // Look for kill or rescue objectives
683 CA_FOREACH(const TActor, a, gActors)
684 if (!a->isInUse)
685 {
686 continue;
687 }
688 const TTileItem *ti = &a->tileItem;
689 if (!(ti->flags & TILEITEM_OBJECTIVE))
690 {
691 continue;
692 }
693 const int objective = ObjectiveFromTileItem(ti->flags);
694 const Objective *o =
695 CArrayGet(&gMission.missionData->Objectives, objective);
696 if (o->Type != OBJECTIVE_KILL && o->Type != OBJECTIVE_RESCUE)
697 {
698 continue;
699 }
700 // Only rescue those that need to be rescued
701 if (o->Type == OBJECTIVE_RESCUE && !(a->flags & FLAGS_PRISONER))
702 {
703 continue;
704 }
705 ClosestObjective co;
706 memset(&co, 0, sizeof co);
707 co.Pos = Vec2iNew(ti->x, ti->y);
708 co.IsDestructible = false;
709 co.Type = AI_OBJECTIVE_TYPE_NORMAL;
710 co.Distance = DistanceSquared(actorRealPos, co.Pos);
711 co.u.Objective = o;
712 CArrayPushBack(objectives, &co);
713 CA_FOREACH_END()
714
715 // Look for explore objectives
716 CA_FOREACH(const Objective, o, gMission.missionData->Objectives)
717 if (o->Type != OBJECTIVE_INVESTIGATE)
718 {
719 continue;
720 }
721 if (ObjectiveIsComplete(o) || MapGetExploredPercentage(&gMap) == 100)
722 {
723 continue;
724 }
725 // Find the nearest unexplored tile
726 const Vec2i actorTile = Vec2iToTile(actorRealPos);
727 const Vec2i unexploredTile = MapSearchTileAround(
728 &gMap, actorTile, MapTileIsUnexplored);
729 ClosestObjective co;
730 memset(&co, 0, sizeof co);
731 co.Pos = Vec2iCenterOfTile(unexploredTile);
732 co.IsDestructible = false;
733 co.Type = AI_OBJECTIVE_TYPE_NORMAL;
734 co.Distance = DistanceSquared(actorRealPos, co.Pos);
735 co.u.Objective = o;
736 CArrayPushBack(objectives, &co);
737 CA_FOREACH_END()
738
739 // Sort according to distance
740 qsort(
741 objectives->data,
742 objectives->size, objectives->elemSize, CompareClosestObjective);
743}
744static bool OnClosestPickupGun(
745 ClosestObjective *co, const Pickup *p,
746 const TActor *actor, const TActor *closestPlayer)
747{
748 if (!ConfigGetBool(&gConfig, "Game.Ammo"))
749 {
750 return false;
751 }
752 // Pick up if:
753 // - we have a gun with less ammo than starting,
754 // - and lower than lead player, who uses the ammo,
755 // - or if there is a free weapon slot,
756 // - and the
757 bool hasGunLowOnAmmo = false;
758 if (actor->guns.size < MAX_WEAPONS)
759 {
760 hasGunLowOnAmmo = true;
761 }
762 else
763 {
764 CA_FOREACH(const Weapon, w, actor->guns)
765 const int ammoId = w->Gun->AmmoId;
766 if (ammoId < 0)
767 {
768 continue;
769 }
770 const Ammo *ammo = AmmoGetById(&gAmmo, ammoId);
771 const int ammoAmount = *(int *)CArrayGet(&actor->ammo, ammoId);
772 if (ammoAmount >= ammo->Amount * AMMO_STARTING_MULTIPLE ||
773 (closestPlayer != NULL &&
774 ActorUsesAmmo(closestPlayer, ammoId) &&
775 ammoAmount > *(int *)CArrayGet(&closestPlayer->ammo, ammoId)))
776 {
777 continue;
778 }
779 hasGunLowOnAmmo = true;
780 CA_FOREACH_END()
781 }
782
783 if (!hasGunLowOnAmmo)
784 {
785 return false;
786 }
787
788 if (ActorHasGun(actor, IdGunDescription(p->class->u.GunId)))
789 {
790 return false;
791 }
792
793 co->Type = AI_OBJECTIVE_TYPE_PICKUP;
794 co->u.UID = p->UID;
795 return true;
796}
797static int CompareClosestObjective(const void *v1, const void *v2)
798{
799 const ClosestObjective *c1 = v1;
800 const ClosestObjective *c2 = v2;
801 if (c1->Distance < c2->Distance)
802 {
803 return -1;
804 }
805 else if (c1->Distance > c2->Distance)
806 {
807 return 1;
808 }
809 return 0;
810}
811static bool IsPosCloseEnoughToPlayer(
812 const Vec2i realPos, const TActor *player,
813 const int distanceTooFarFromPlayer);
814static bool CanGetObjective(
815 const Vec2i objRealPos, const Vec2i actorRealPos, const TActor *player,
816 const int distanceTooFarFromPlayer)
817{
818 // Check if we can complete an objective.
819 // If we don't need to follow any player, then only check that we can
820 // pathfind to it.
821 if (!player)
822 {
823 return AIHasPath(actorRealPos, objRealPos, true);
824 }
825 // If we need to stick close to a player, then check that it is both:
826 // - Close enough from the lead player
827 // - In line of sight from us
828 if (!IsPosCloseEnoughToPlayer(
829 objRealPos, player, distanceTooFarFromPlayer))
830 {
831 return false;
832 }
833 return AIHasClearPath(actorRealPos, objRealPos, true);
834}
835static bool IsPosCloseEnoughToPlayer(
836 const Vec2i realPos, const TActor *player,
837 const int distanceTooFarFromPlayer)
838{
839 if (!player)
840 {
841 return true;
842 }
843 const int distanceFromPlayer2 =
844 DistanceSquared(realPos, Vec2iFull2Real(player->Pos));
845 const int distanceMax2 = SQUARED(distanceTooFarFromPlayer * 16);
846 return distanceFromPlayer2 < distanceMax2;
847}
848// Navigate to the current objective
849// If the objective is destructible and it makes sense to shoot it,
850// shoot at it as well
851static int GotoObjective(TActor *actor, int objDistance)
852{
853 const AIObjectiveState *objState = &actor->aiContext->ObjectiveState;
854 const Vec2i goal = objState->Goal;
855 int cmd = 0;
856 // Check to see if we need to go any closer to the objective
857 const bool isDestruction =
858 objState->Type == AI_OBJECTIVE_TYPE_NORMAL && objState->IsDestructible;
859 if (!isDestruction ||
860 DistanceSquared(Vec2iFull2Real(actor->Pos), goal) > 3 * 16 * 16 ||
861 !AIHasClearShot(Vec2iFull2Real(actor->Pos), goal))
862 {
863 cmd = SmartGoto(actor, goal, objDistance);
864 }
865 else if (isDestruction && ActorGetGun(actor)->lock <= 0)
866 {
867 cmd = AIHunt(actor, Vec2iReal2FullCentered(goal));
868 if (AIHasClearShot(Vec2iFull2Real(actor->Pos), goal))
869 {
870 cmd |= CMD_BUTTON1;
871 }
872 }
873 return cmd;
874}

877void AICoopSelectWeapons(
878 PlayerData *p, const int player, const CArray *weapons)
879{
880 p->weaponCount = 0;
881
882 // Select two weapons from available ones
883 // Offset the starting index a bit so each AI uses different guns
884 const int startIdx = (player * 3) % weapons->size;
885 for (int i = 0; i < MAX_WEAPONS; i++)
886 {
887 const int idx = (startIdx + i) % weapons->size;
888 if (idx == startIdx && i > 0)
889 {
890 // Not enough weapons
891 break;
892 }
893 const GunDescription **g = CArrayGet(weapons, idx);
894 p->weapons[i] = *g;
895 p->weaponCount++;
896 }
897
898 if (ConfigGetBool(&gConfig, "Game.Ammo"))
899 {
900 // Select pistol as an infinite-ammo backup
901 const GunDescription *pistol = StrGunDescription("Pistol");
902 if (!PlayerHasWeapon(p, pistol))
903 {
904 if (p->weaponCount == MAX_WEAPONS)
905 {
906 // Player has full weapons; replace last weapon
907 p->weapons[MAX_WEAPONS - 1] = pistol;
908 }
909 else
910 {
911 // Add the pistol
912 p->weapons[p->weaponCount] = pistol;
913 p->weaponCount++;
914 }
915 }
916 }
917}
918static bool PlayerHasWeapon(const PlayerData *p, const GunDescription *w)
919{
920 for (int i = 0; i < MAX_WEAPONS; i++)
921 {
922 const GunDescription *g = p->weapons[i];
923 if (w == g)
924 {
925 return true;
926 }
927 }
928 return false;
929}
