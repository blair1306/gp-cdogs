55const char *ObjectiveTypeStr(const ObjectiveType t)
56{
57 switch (t)
58 {
59 T2S(OBJECTIVE_KILL, "Kill");
60 T2S(OBJECTIVE_COLLECT, "Collect");
61 T2S(OBJECTIVE_DESTROY, "Destroy");
62 T2S(OBJECTIVE_RESCUE, "Rescue");
63 T2S(OBJECTIVE_INVESTIGATE, "Explore");
64 default:
65 return "";
66 }
67}
68ObjectiveType StrObjectiveType(const char *s)
69{
70 S2T(OBJECTIVE_KILL, "Kill");
71 S2T(OBJECTIVE_COLLECT, "Collect");
72 S2T(OBJECTIVE_DESTROY, "Destroy");
73 S2T(OBJECTIVE_RESCUE, "Rescue");
74 S2T(OBJECTIVE_INVESTIGATE, "Explore");
75 CASSERT(false, "unknown objective name");
76 return OBJECTIVE_KILL;
77}
78color_t ObjectiveTypeColor(const ObjectiveType t)
79{
80 switch (t)
81 {
82 case OBJECTIVE_KILL: // fallthrough
83 case OBJECTIVE_DESTROY:
84 return colorRed;
85 case OBJECTIVE_COLLECT: // fallthrough
86 case OBJECTIVE_RESCUE:
87 return colorGreen;
88 case OBJECTIVE_INVESTIGATE:
89 return colorCyan;
90 default:
91 CASSERT(false, "unknown objective type");
92 // Shouldn't get here but use a different colour in case
93 return colorYellow;
94 }
95}

97void ObjectiveLoadJSON(Objective *o, json_t *node, const int version)
98{
99 memset(o, 0, sizeof *o);
100 o->Description = GetString(node, "Description");
101 JSON_UTILS_LOAD_ENUM(o->Type, node, "Type", StrObjectiveType);
102 // Set objective colours based on type
103 o->color = ObjectiveTypeColor(o->Type);
104 if (version < 8)
105 {
106 // Index numbers used for all objective classes; convert them
107 // to their class handles
108 LoadInt(&o->u.Index, node, "Index");
109 switch (o->Type)
110 {
111 case OBJECTIVE_COLLECT:
112 o->u.Pickup = IntPickupClass(o->u.Index);
113 break;
114 case OBJECTIVE_DESTROY:
115 o->u.MapObject = IntMapObject(o->u.Index);
116 break;
117 default:
118 // do nothing
119 break;
120 }
121 }
122 else
123 {
124 char *tmp;
125 switch (o->Type)
126 {
127 case OBJECTIVE_COLLECT:
128 tmp = GetString(node, "Pickup");
129 o->u.Pickup = StrPickupClass(tmp);
130 CFREE(tmp);
131 break;
132 case OBJECTIVE_DESTROY:
133 tmp = GetString(node, "MapObject");
134 o->u.MapObject = StrMapObject(tmp);
135 CFREE(tmp);
136 break;
137 default:
138 LoadInt(&o->u.Index, node, "Index");
139 break;
140 }
141 }
142 LoadInt(&o->Count, node, "Count");
143 LoadInt(&o->Required, node, "Required");
144 LoadInt(&o->Flags, node, "Flags");
145}
146void ObjectiveSetup(Objective *o)
147{
148 o->placed = 0;
149 o->done = 0;
150}
151void ObjectiveCopy(Objective *dst, const Objective *src)
152{
153 memcpy(dst, src, sizeof *dst);
154 if (src->Description)
155 {
156 CSTRDUP(dst->Description, src->Description);
157 }
158}
159void ObjectiveTerminate(Objective *o)
160{
161 CFREE(o->Description);
162}
163
164bool ObjectiveIsRequired(const Objective *o)
165{
166 return o->Required > 0;
167}
168bool ObjectiveIsComplete(const Objective *o)
169{
170 return o->done >= o->Required;
171}
172bool ObjectiveIsPerfect(const Objective *o)
173{
174 // Don't count objectives that are fully required anyway
175 return o->done == o->Count && o->done > o->Required;
176}
