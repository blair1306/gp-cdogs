69/// Function: getch
70/// Get character without waiting for Return to be pressed.
71/// Windows has this in conio.h
72RLUTIL_INLINE int getch(void) {
73 // Here be magic.
74 struct termios oldt, newt;
75 int ch;
76 tcgetattr(STDIN_FILENO, &oldt);
77 newt = oldt;
78 newt.c_lflag &= ~(ICANON | ECHO);
79 tcsetattr(STDIN_FILENO, TCSANOW, &newt);
80 ch = getchar();
81 tcsetattr(STDIN_FILENO, TCSANOW, &oldt);
82 return ch;
83}

85/// Function: kbhit
86/// Determines if keyboard has been hit.
87/// Windows has this in conio.h
88RLUTIL_INLINE int kbhit(void) {
89 // Here be dragons.
90 static struct termios oldt, newt;
91 int cnt = 0;
92 tcgetattr(STDIN_FILENO, &oldt);
93 newt = oldt;
94 newt.c_lflag &= ~(ICANON | ECHO);
95 newt.c_iflag = 0; // input mode
96 newt.c_oflag = 0; // output mode
97 newt.c_cc[VMIN] = 1; // minimum time to wait
98 newt.c_cc[VTIME] = 1; // minimum characters to wait for
99 tcsetattr(STDIN_FILENO, TCSANOW, &newt);
100 ioctl(0, FIONREAD, &cnt); // Read count
101 struct timeval tv;
102 tv.tv_sec = 0;
103 tv.tv_usec = 100;
104 select(STDIN_FILENO+1, NULL, NULL, NULL, &tv); // A small time delay
105 tcsetattr(STDIN_FILENO, TCSANOW, &oldt);
106 return cnt; // Return number of characters
107}
108#endif // _WIN32

339/// Function: getkey
340/// Reads a key press (blocking) and returns a key code.
341///
342/// See <Key codes for keyhit()>
343///
344/// Note:
345/// Only Arrows, Esc, Enter and Space are currently working properly.
346RLUTIL_INLINE int getkey(void) {
347 #ifndef _WIN32
348 int cnt = kbhit(); // for ANSI escapes processing
349 #endif
350 int k = getch();
351 switch(k) {
352 case 0: {
353 int kk;
354 switch (kk = getch()) {
355 case 71: return KEY_NUMPAD7;
356 case 72: return KEY_NUMPAD8;
357 case 73: return KEY_NUMPAD9;
358 case 75: return KEY_NUMPAD4;
359 case 77: return KEY_NUMPAD6;
360 case 79: return KEY_NUMPAD1;
361 case 80: return KEY_NUMPAD2;
362 case 81: return KEY_NUMPAD3;
363 case 82: return KEY_NUMPAD0;
364 case 83: return KEY_NUMDEL;
365 default: return kk-59+KEY_F1; // Function keys
366 }}
367 case 224: {
368 int kk;
369 switch (kk = getch()) {
370 case 71: return KEY_HOME;
371 case 72: return KEY_UP;
372 case 73: return KEY_PGUP;
373 case 75: return KEY_LEFT;
374 case 77: return KEY_RIGHT;
375 case 79: return KEY_END;
376 case 80: return KEY_DOWN;
377 case 81: return KEY_PGDOWN;
378 case 82: return KEY_INSERT;
379 case 83: return KEY_DELETE;
380 default: return kk-123+KEY_F1; // Function keys
381 }}
382 case 13: return KEY_ENTER;
383#ifdef _WIN32
384 case 27: return KEY_ESCAPE;
385#else // _WIN32
386 case 155: // single-character CSI
387 case 27: {
388 // Process ANSI escape sequences
389 if (cnt >= 3 && getch() == '[') {
390 switch (k = getch()) {
391 case 'A': return KEY_UP;
392 case 'B': return KEY_DOWN;
393 case 'C': return KEY_RIGHT;
394 case 'D': return KEY_LEFT;
395 }
396 } else return KEY_ESCAPE;
397 }
398#endif // _WIN32
399 default: return k;
400 }
401}

456/// Don't change the background color

460#if defined(_WIN32) && !defined(RLUTIL_USE_ANSI)

467#else

490/// Function: saveDefaultColor
491/// Call once to preserve colors for use in resetColor()
492/// on Windows without ANSI, no-op otherwise
493///
494/// See <Color Codes>
495/// See <resetColor>
496RLUTIL_INLINE int saveDefaultColor(void) {
497#if defined(_WIN32) && !defined(RLUTIL_USE_ANSI)
498 static char initialized = 0; // bool
499 static WORD attributes;
500
501 if (!initialized) {
502 CONSOLE_SCREEN_BUFFER_INFO csbi;
503 GetConsoleScreenBufferInfo(GetStdHandle(STD_OUTPUT_HANDLE), &csbi);
504 attributes = csbi.wAttributes;
505 initialized = 1;
506 }
507 return (int)attributes;
508#else
509 return -1;
510#endif
511}

528/// Function: cls
529/// Clears screen, resets all attributes and moves cursor home.
530RLUTIL_INLINE void cls(void) {
531#if defined(_WIN32) && !defined(RLUTIL_USE_ANSI)
532 // Based on https://msdn.microsoft.com/en-us/library/windows/desktop/ms682022%28v=vs.85%29.aspx
533 const HANDLE hConsole = GetStdHandle(STD_OUTPUT_HANDLE);
534 const COORD coordScreen = {0, 0};
535 DWORD cCharsWritten;
536 CONSOLE_SCREEN_BUFFER_INFO csbi;
537
538 GetConsoleScreenBufferInfo(hConsole, &csbi);
539 const DWORD dwConSize = csbi.dwSize.X * csbi.dwSize.Y;
540 FillConsoleOutputCharacter(hConsole, (TCHAR)' ', dwConSize, coordScreen, &cCharsWritten);
541
542 GetConsoleScreenBufferInfo(hConsole, &csbi);
543 FillConsoleOutputAttribute(hConsole, csbi.wAttributes, dwConSize, coordScreen, &cCharsWritten);
544
545 SetConsoleCursorPosition(hConsole, coordScreen);
546#else
547 RLUTIL_PRINT(ANSI_CLS);
548 RLUTIL_PRINT(ANSI_CURSOR_HOME);
549#endif
550}

552/// Function: locate
553/// Sets the cursor position to 1-based x,y.
554RLUTIL_INLINE void locate(int x, int y) {
555#if defined(_WIN32) && !defined(RLUTIL_USE_ANSI)
556 COORD coord;
557 // TODO: clamping/assert for x/y <= 0?
558 coord.X = (SHORT)(x - 1);
559 coord.Y = (SHORT)(y - 1); // Windows uses 0-based coordinates
560 SetConsoleCursorPosition(GetStdHandle(STD_OUTPUT_HANDLE), coord);
561#else // _WIN32 || USE_ANSI
562 #ifdef __cplusplus
563 RLUTIL_PRINT("\033[" << y << ";" << x << "H");
564 #else // __cplusplus
565 char buf[32];
566 sprintf(buf, "\033[%d;%df", y, x);
567 RLUTIL_PRINT(buf);
568 #endif // __cplusplus
569#endif // _WIN32 || USE_ANSI
570}

572/// Function: setString
573/// Prints the supplied string without advancing the cursor
574#ifdef __cplusplus
575RLUTIL_INLINE void setString(const RLUTIL_STRING_T & str_) {
576 const char * const str = str_.data();
577 unsigned int len = str_.size();
578#else // __cplusplus
579RLUTIL_INLINE void setString(RLUTIL_STRING_T str) {
580 unsigned int len = strlen(str);
581#endif // __cplusplus
582#if defined(_WIN32) && !defined(RLUTIL_USE_ANSI)
583 HANDLE hConsoleOutput = GetStdHandle(STD_OUTPUT_HANDLE);
584 DWORD numberOfCharsWritten;
585 CONSOLE_SCREEN_BUFFER_INFO csbi;

588 WriteConsoleOutputCharacter(hConsoleOutput, str, len, csbi.dwCursorPosition, &numberOfCharsWritten);
589#else // _WIN32 || USE_ANSI
590 RLUTIL_PRINT(str);
591 #ifdef __cplusplus
592 RLUTIL_PRINT("\033[" << len << 'D');
593 #else // __cplusplus
594 char buf[3 + 20 + 1]; // 20 = max length of 64-bit unsigned int when printed as dec
595 sprintf(buf, "\033[%uD", len);
596 RLUTIL_PRINT(buf);
597 #endif // __cplusplus
598#endif // _WIN32 || USE_ANSI
599}

646/// Function: trows
647/// Get the number of rows in the terminal window or -1 on error.
648RLUTIL_INLINE int trows(void) {
649#ifdef _WIN32
650 CONSOLE_SCREEN_BUFFER_INFO csbi;
651 if (!GetConsoleScreenBufferInfo(GetStdHandle(STD_OUTPUT_HANDLE), &csbi))
652 return -1;
653 else
654 return csbi.srWindow.Bottom - csbi.srWindow.Top + 1; // Window height
655 // return csbi.dwSize.Y; // Buffer height
656#else
657#ifdef TIOCGSIZE
658 struct ttysize ts;
659 ioctl(STDIN_FILENO, TIOCGSIZE, &ts);
660 return ts.ts_lines;
661#elif defined(TIOCGWINSZ)
662 struct winsize ts;
663 ioctl(STDIN_FILENO, TIOCGWINSZ, &ts);
664 return ts.ws_row;
665#else // TIOCGSIZE
666 return -1;
667#endif // TIOCGSIZE

671/// Function: tcols
672/// Get the number of columns in the terminal window or -1 on error.
673RLUTIL_INLINE int tcols(void) {
674#ifdef _WIN32
675 CONSOLE_SCREEN_BUFFER_INFO csbi;
676 if (!GetConsoleScreenBufferInfo(GetStdHandle(STD_OUTPUT_HANDLE), &csbi))
677 return -1;
678 else
679 return csbi.srWindow.Right - csbi.srWindow.Left + 1; // Window width
680 // return csbi.dwSize.X; // Buffer width
681#else
682#ifdef TIOCGSIZE
683 struct ttysize ts;
684 ioctl(STDIN_FILENO, TIOCGSIZE, &ts);
685 return ts.ts_cols;
686#elif defined(TIOCGWINSZ)
687 struct winsize ts;
688 ioctl(STDIN_FILENO, TIOCGWINSZ, &ts);
689 return ts.ws_col;
690#else // TIOCGSIZE
691 return -1;
692#endif // TIOCGSIZE
693#endif // _WIN32
694}
