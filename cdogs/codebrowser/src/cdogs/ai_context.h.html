35// State data for various AI routines
36
37// State for what the AI is currently doing
38// Used for AI chatter
39typedef enum
40{
41 AI_STATE_NONE,
42 AI_STATE_IDLE,
43 AI_STATE_DIE,
44 AI_STATE_FOLLOW,
45 AI_STATE_HUNT,
46 AI_STATE_TRACK,
47 AI_STATE_FLEE,
48 AI_STATE_CONFUSED,
49 AI_STATE_NEXT_OBJECTIVE
50} AIState;
51// State for what the AI is doing when confused
52typedef enum
53{
54 AI_CONFUSION_CONFUSED, // perform a random action
55 AI_CONFUSION_CORRECT // perform the right action (correct for confusion)
56} AIConfusionType;
57typedef struct
58{
59 AIConfusionType Type;
60 int Cmd;
61} AIConfusionState;
62typedef enum
63{
64 AI_OBJECTIVE_TYPE_NORMAL,
65 AI_OBJECTIVE_TYPE_KEY,
66 AI_OBJECTIVE_TYPE_EXIT,
67 AI_OBJECTIVE_TYPE_KILL,
68 AI_OBJECTIVE_TYPE_PICKUP
69} AIObjectiveType;
70// State for AI attempting to complete an objective
71// This is to prevent excessive pathfinding calls
72typedef struct
73{
74 AIObjectiveType Type;
75 bool IsDestructible;
76 union
77 {
78 const Objective *Obj;
79 int UID;
80 } u;
81 int LastDone;
82 Vec2i Goal;
83} AIObjectiveState;
84typedef struct
85{
86 Vec2i Goal;
87 CachedPath Path;
88 int PathIndex;
89 bool IsFollowing;
90} AIGotoContext;
91typedef struct
92{
93 int LastCmd;
94 // Delay in executing consecutive actions;
95 // Used to let the AI perform one action for a set amount of time
96 int Delay;
97 AIState State;
98
99 // Counters to moderate amount of chatter
100 int ChatterCounter;
101
102 AIConfusionState ConfusionState;
103 AIObjectiveState ObjectiveState;
104 Vec2i LastTile;
105 bool IsStuckTooLong;
106 AIGotoContext Goto;
107 int EnemyId;
108 double GunRangeScalar;
109 int OnGunId;
110} AIContext;
111
112AIContext *AIContextNew(void);
113void AIContextDestroy(AIContext *c);
114
115const char *AIStateGetChatterText(const AIState s);
116bool AIContextShowChatter(const AIContext *c, const AIChatterFrequency f);
117bool AIContextSetState(AIContext *c, const AIState s);
