42int MapNewScan(const char *filename, char **title, int *numMissions)
43{
44 int err = 0;
45 json_t *root = NULL;
46 FILE *f = NULL;
47
48 if (strcmp(StrGetFileExt(filename), "cdogscpn") == 0 ||
49 strcmp(StrGetFileExt(filename), "CDOGSCPN") == 0)
50 {
51 return MapNewScanArchive(filename, title, numMissions);
52 }
53 if (IsCampaignOldFile(filename))
54 {
55 return ScanCampaignOld(filename, title, numMissions);
56 }
57 f = fopen(filename, "r");
58 if (f == NULL)
59 {
60 debug(D_NORMAL, "MapNewLoad - invalid path!\n");
61 err = -1;
62 goto bail;
63 }
64 if (json_stream_parse(f, &root) != JSON_OK)
65 {
66 err = -1;
67 goto bail;
68 }
69 err = MapNewScanJSON(root, title, numMissions);
70 if (err < 0)
71 {
72 goto bail;
73 }
74
75bail:
76 if (f)
77 {
78 fclose(f);
79 }
80 json_free_value(&root);
81 return err;
82}
83int MapNewScanJSON(json_t *root, char **title, int *numMissions)
84{
85 int err = 0;
86 int version;
87 LoadInt(&version, root, "Version");
88 if (version > MAP_VERSION || version <= 0)
89 {
90 err = -1;
91 goto bail;
92 }
93 *title = GetString(root, "Title");
94 *numMissions = 0;
95 if (version < 3)
96 {
97 for (json_t *missionNode =
98 json_find_first_label(root, "Missions")->child->child;
99 missionNode;
100 missionNode = missionNode->next)
101 {
102 (*numMissions)++;
103 }
104 }
105 else
106 {
107 LoadInt(numMissions, root, "Missions");
108 }
109
110bail:
111 return err;
112}

114int MapNewLoad(const char *filename, CampaignSetting *c)
115{
116 int err = 0;
117
118 debug(D_NORMAL, "Loading map %s\n", filename);
119
120 if (IsCampaignOldFile(filename))
121 {
122 CampaignSettingOld cOld;
123 memset(&cOld, 0, sizeof cOld);
124 err = LoadCampaignOld(filename, &cOld);
125 if (!err)
126 {
127 ConvertCampaignSetting(c, &cOld);
128 }
129 CFREE(cOld.missions);
130 CFREE(cOld.characters);
131 return err;
132 }
133
134 if (strcmp(StrGetFileExt(filename), "cdogscpn") == 0 ||
135 strcmp(StrGetFileExt(filename), "CDOGSCPN") == 0)
136 {
137 return MapNewLoadArchive(filename, c);
138 }
139
140 // try to load the new map format
141 json_t *root = NULL;
142 int version;
143 FILE *f = fopen(filename, "r");
144 if (f == NULL)
145 {
146 debug(D_NORMAL, "MapNewLoad - invalid path!\n");
147 err = -1;
148 goto bail;
149 }
150 if (json_stream_parse(f, &root) != JSON_OK)
151 {
152 printf("Error parsing campaign '%s'\n", filename);
153 err = -1;
154 goto bail;
155 }
156 LoadInt(&version, root, "Version");
157 if (version > 2 || version <= 0)
158 {
159 assert(0 && "not implemented or unknown campaign");
160 err = -1;
161 goto bail;
162 }
163 MapNewLoadCampaignJSON(root, c);
164 LoadMissions(&c->Missions, json_find_first_label(root, "Missions")->child, version);
165 CharacterLoadJSON(&c->characters, root, version);
166
167bail:
168 json_free_value(&root);
169 if (f != NULL)
170 {
171 fclose(f);
172 }
173 return err;
174}

194void LoadMissions(CArray *missions, json_t *missionsNode, int version)
195{
196 json_t *child;
197 for (child = missionsNode->child; child; child = child->next)
198 {
199 Mission m;
200 MissionInit(&m);
201 m.Title = GetString(child, "Title");
202 m.Description = GetString(child, "Description");
203 JSON_UTILS_LOAD_ENUM(m.Type, child, "Type", StrMapType);
204 LoadInt(&m.Size.x, child, "Width");
205 LoadInt(&m.Size.y, child, "Height");
206 if (version <= 10)
207 {
208 int style;
209 LoadInt(&style, child, "WallStyle");
210 strcpy(m.WallStyle, IntWallStyle(style));
211 LoadInt(&style, child, "FloorStyle");
212 strcpy(m.FloorStyle, IntFloorStyle(style));
213 LoadInt(&style, child, "RoomStyle");
214 strcpy(m.RoomStyle, IntRoomStyle(style));
215 }
216 else
217 {
218 char *tmp = GetString(child, "WallStyle");
219 strcpy(m.WallStyle, tmp);
220 CFREE(tmp);
221 tmp = GetString(child, "FloorStyle");
222 strcpy(m.FloorStyle, tmp);
223 CFREE(tmp);
224 tmp = GetString(child, "RoomStyle");
225 strcpy(m.RoomStyle, tmp);
226 CFREE(tmp);
227 }
228 if (version <= 9)
229 {
230 int style;
231 LoadInt(&style, child, "ExitStyle");
232 strcpy(m.ExitStyle, IntExitStyle(style));
233 }
234 else
235 {
236 char *tmp = GetString(child, "ExitStyle");
237 strcpy(m.ExitStyle, tmp);
238 CFREE(tmp);
239 }
240 if (version <= 8)
241 {
242 int keyStyle;
243 LoadInt(&keyStyle, child, "KeyStyle");
244 strcpy(m.KeyStyle, IntKeyStyle(keyStyle));
245 }
246 else
247 {
248 char *tmp = GetString(child, "KeyStyle");
249 strcpy(m.KeyStyle, tmp);
250 CFREE(tmp);
251 }
252 if (version <= 5)
253 {
254 int doorStyle;
255 LoadInt(&doorStyle, child, "DoorStyle");
256 strcpy(m.DoorStyle, IntDoorStyle(doorStyle));
257 }
258 else
259 {
260 char *tmp = GetString(child, "DoorStyle");
261 strcpy(m.DoorStyle, tmp);
262 CFREE(tmp);
263 }
264 LoadMissionObjectives(
265 &m.Objectives,
266 json_find_first_label(child, "Objectives")->child,
267 version);
268 LoadIntArray(&m.Enemies, child, "Enemies");
269 LoadIntArray(&m.SpecialChars, child, "SpecialChars");
270 if (version <= 3)
271 {
272 CArray items;
273 CArrayInit(&items, sizeof(int));
274 LoadIntArray(&items, child, "Items");
275 CArray densities;
276 CArrayInit(&densities, sizeof(int));
277 LoadIntArray(&densities, child, "ItemDensities");
278 for (int i = 0; i < (int)items.size; i++)
279 {
280 MapObjectDensity mod;
281 mod.M = IntMapObject(*(int *)CArrayGet(&items, i));
282 mod.Density = *(int *)CArrayGet(&densities, i);
283 CArrayPushBack(&m.MapObjectDensities, &mod);
284 }
285 }
286 else
287 {
288 json_t *modsNode =
289 json_find_first_label(child, "MapObjectDensities");
290 if (modsNode && modsNode->child)
291 {
292 modsNode = modsNode->child;
293 for (json_t *modNode = modsNode->child;
294 modNode;
295 modNode = modNode->next)
296 {
297 MapObjectDensity mod;
298 mod.M = StrMapObject(
299 json_find_first_label(modNode, "MapObject")->child->text);
300 LoadInt(&mod.Density, modNode, "Density");
301 CArrayPushBack(&m.MapObjectDensities, &mod);
302 }
303 }
304 }
305 LoadInt(&m.EnemyDensity, child, "EnemyDensity");
306 LoadWeapons(
307 &m.Weapons, json_find_first_label(child, "Weapons")->child);
308 strcpy(m.Song, json_find_first_label(child, "Song")->child->text);
309 if (version <= 4)
310 {
311 // Load colour indices
312 int wc, fc, rc, ac;
313 LoadInt(&wc, child, "WallColor");
314 LoadInt(&fc, child, "FloorColor");
315 LoadInt(&rc, child, "RoomColor");
316 LoadInt(&ac, child, "AltColor");
317 m.WallMask = RangeToColor(wc);
318 m.FloorMask = RangeToColor(fc);
319 m.RoomMask = RangeToColor(rc);
320 m.AltMask = RangeToColor(ac);
321 }
322 else
323 {
324 LoadColor(&m.WallMask, child, "WallMask");
325 LoadColor(&m.FloorMask, child, "FloorMask");
326 LoadColor(&m.RoomMask, child, "RoomMask");
327 LoadColor(&m.AltMask, child, "AltMask");
328 }
329 switch (m.Type)
330 {
331 case MAPTYPE_CLASSIC:
332 LoadInt(&m.u.Classic.Walls, child, "Walls");
333 LoadInt(&m.u.Classic.WallLength, child, "WallLength");
334 LoadInt(&m.u.Classic.CorridorWidth, child, "CorridorWidth");
335 LoadRooms(
336 &m.u.Classic.Rooms,
337 json_find_first_label(child, "Rooms")->child);
338 LoadInt(&m.u.Classic.Squares, child, "Squares");
339 LoadClassicDoors(&m, child, "Doors");
340 LoadClassicPillars(&m, child, "Pillars");
341 break;
342 case MAPTYPE_STATIC:
343 if (!TryLoadStaticMap(&m, child, version))
344 {
345 continue;
346 }
347 break;
348 case MAPTYPE_CAVE:
349 {
350 LoadInt(&m.u.Cave.FillPercent, child, "FillPercent");
351 LoadInt(&m.u.Cave.Repeat, child, "Repeat");
352 LoadInt(&m.u.Cave.R1, child, "R1");
353 LoadInt(&m.u.Cave.R2, child, "R2");
354 json_t *roomsNode = json_find_first_label(child, "Rooms");
355 if (roomsNode != NULL && roomsNode->child != NULL)
356 {
357 LoadRooms(&m.u.Cave.Rooms, roomsNode->child);
358 }
359 LoadInt(&m.u.Cave.Squares, child, "Squares");
360 }
361 break;
362 default:
363 assert(0 && "unknown map type");
364 continue;
365 }
366 CArrayPushBack(missions, &m);
367 }
368}

376static bool TryLoadStaticMap(Mission *m, json_t *node, int version)
377{
378 CArrayInit(&m->u.Static.Tiles, sizeof(unsigned short));
379 if (version == 1)
380 {
381 // JSON array
382 json_t *tiles = json_find_first_label(node, "Tiles");
383 if (!tiles || !tiles->child)
384 {
385 return false;
386 }
387 tiles = tiles->child;
388 for (tiles = tiles->child; tiles; tiles = tiles->next)
389 {
390 unsigned short n = (unsigned short)atoi(tiles->text);
391 CArrayPushBack(&m->u.Static.Tiles, &n);
392 }
393 }
394 else
395 {
396 // CSV string
397 char *tileCSV = GetString(node, "Tiles");
398 char *pch = strtok(tileCSV, ",");
399 while (pch != NULL)
400 {
401 unsigned short n = (unsigned short)atoi(pch);
402 CArrayPushBack(&m->u.Static.Tiles, &n);
403 pch = strtok(NULL, ",");
404 }
405 CFREE(tileCSV);
406 }
407
408 CArrayInit(&m->u.Static.Items, sizeof(MapObjectPositions));
409 LoadStaticItems(m, node, "StaticItems", version);
410 if (version < 13)
411 {
412 LoadStaticWrecks(m, node, "StaticWrecks");
413 }
414 LoadStaticCharacters(m, node, "StaticCharacters");
415 LoadStaticObjectives(m, node, "StaticObjectives");
416 LoadStaticKeys(m, node, "StaticKeys");
417
418 LoadVec2i(&m->u.Static.Start, node, "Start");
419 LoadStaticExit(m, node, "Exit");
420
421 return true;
422}

519static void LoadStaticItems(
520 Mission *m, json_t *node, const char *name, const int version)
521{
522 json_t *items = json_find_first_label(node, name);
523 if (!items || !items->child)
524 {
525 return;
526 }
527 items = items->child;
528 for (items = items->child; items; items = items->next)
529 {
530 MapObjectPositions mop;
531 mop.M = LoadMapObjectRef(items, version);
532 if (mop.M == NULL)
533 {
534 continue;
535 }
536 CArrayInit(&mop.Positions, sizeof(Vec2i));
537 json_t *positions = json_find_first_label(items, "Positions");
538 if (!positions || !positions->child)
539 {
540 continue;
541 }
542 positions = positions->child;
543 for (positions = positions->child;
544 positions;
545 positions = positions->next)
546 {
547 Vec2i pos;
548 json_t *position = positions->child;
549 pos.x = atoi(position->text);
550 position = position->next;
551 pos.y = atoi(position->text);
552 CArrayPushBack(&mop.Positions, &pos);
553 }
554 CArrayPushBack(&m->u.Static.Items, &mop);
555 }
556}

594static const MapObject *LoadMapObjectRef(json_t *itemNode, const int version)
595{
596 if (version <= 3)
597 {
598 int idx;
599 LoadInt(&idx, itemNode, "Index");
600 return IntMapObject(idx);
601 }
602 else
603 {
604 const char *moName =
605 json_find_first_label(itemNode, "MapObject")->child->text;
606 const MapObject *mo = StrMapObject(moName);
607 if (mo == NULL && version <= 11 && StrEndsWith(moName, " spawner"))
608 {
609 char buf[256];
610 // Old version had same name for ammo and gun spawner
611 char itemName[256];
612 strncpy(itemName, moName, strlen(moName) - strlen(" spawner"));
613 itemName[strlen(moName) - strlen(" spawner")] = '\0';
614 snprintf(buf, 256, "%s ammo spawner", itemName);
615 mo = StrMapObject(buf);
616 }
617 if (mo == NULL)
618 {
619 LOG(LM_MAP, LL_ERROR, "Failed to load map object (%s)", moName);
620 }
621 return mo;
622 }
623}
624static const MapObject *LoadMapObjectWreckRef(json_t *itemNode)
625{
626 const char *moName =
627 json_find_first_label(itemNode, "MapObject")->child->text;
628 const MapObject *mo = StrMapObject(moName);
629 if (mo == NULL)
630 {
631 LOG(LM_MAP, LL_ERROR, "Failed to load map object (%s)", moName);
632 return NULL;
633 }
634 const MapObject *wreck = StrMapObject(mo->Wreck);
635 return wreck;
636}
