87color_t ColorMult(color_t c, color_t m)
88{
89 c.r = (uint8_t)((int)c.r * m.r / 255);
90 c.g = (uint8_t)((int)c.g * m.g / 255);
91 c.b = (uint8_t)((int)c.b * m.b / 255);
92 return c;
93}
94color_t ColorAlphaBlend(color_t a, color_t b)
95{
96 a.r = (uint8_t)(((int)a.r*(255 - b.a) + (int)b.r*b.a)/255);
97 a.g = (uint8_t)(((int)a.g*(255 - b.a) + (int)b.g*b.a)/255);
98 a.b = (uint8_t)(((int)a.b*(255 - b.a) + (int)b.b*b.a)/255);
99 a.a = 255;
100 return a;
101}

113color_t ColorTint(color_t c, HSV hsv)
114{
115 // Adapted from answer by David H
116 // http://stackoverflow.com/a/6930407/2038264
117 // License: http://creativecommons.org/licenses/by-sa/3.0/
118 // Author profile: http://stackoverflow.com/users/1633251/david-h
119 color_t out;
120 int vAvg = ((int)c.r + c.g + c.b) / 3;
121 uint8_t vComponent = (uint8_t)CLAMP(hsv.v * vAvg, 0, 255);
122
123 if (hsv.s <= 0.0)
124 {
125 // No saturation; just gray
126 out.r = vComponent;
127 out.g = vComponent;
128 out.b = vComponent;
129 }
130 else if (hsv.h >= 0)
131 {
132 // set hue to h; use regular HSV to RGB conversion
133 double ff;
134 uint8_t p, q, t;
135 long i;
136 double hh = hsv.h;
137 if (hh >= 360.0)
138 {
139 hh = 0.0;
140 }
141 hh /= 60.0;
142 i = (long)hh;
143 ff = hh - i;
144 p = (uint8_t)CLAMP(vComponent * (1.0 - hsv.s), 0, 255);
145 q = (uint8_t)CLAMP(vComponent * (1.0 - (hsv.s * ff)), 0, 255);
146 t = (uint8_t)CLAMP(vComponent * (1.0 - (hsv.s * (1.0 - ff))), 0, 255);
147
148 switch(i)
149 {
150 case 0:
151 out.r = vComponent;
152 out.g = t;
153 out.b = p;
154 break;
155 case 1:
156 out.r = q;
157 out.g = vComponent;
158 out.b = p;
159 break;
160 case 2:
161 out.r = p;
162 out.g = vComponent;
163 out.b = t;
164 break;
165
166 case 3:
167 out.r = p;
168 out.g = q;
169 out.b = vComponent;
170 break;
171 case 4:
172 out.r = t;
173 out.g = p;
174 out.b = vComponent;
175 break;
176 case 5:
177 default:
178 out.r = vComponent;
179 out.g = p;
180 out.b = q;
181 break;
182 }
183 }
184 else
185 {
186 // Just set saturation and value
187 // Use weighted average to shift components towards grey for saturation
188 out.r = (uint8_t)CLAMP(hsv.v * (vAvg*(1.0-hsv.s) + hsv.s*c.r), 0, 255);
189 out.g = (uint8_t)CLAMP(hsv.v * (vAvg*(1.0-hsv.s) + hsv.s*c.g), 0, 255);
190 out.b = (uint8_t)CLAMP(hsv.v * (vAvg*(1.0-hsv.s) + hsv.s*c.b), 0, 255);
191 }
192 out.a = c.a;
193 return out;
194}

196bool ColorEquals(const color_t a, const color_t b)
197{
198 return a.r == b.r && a.g == b.g && a.b == b.b;
199}
200bool HSVEquals(const HSV a, const HSV b)
201{
202 const double epsilon = 0.000001;
203 return fabs(a.h - b.h) < epsilon && fabs(a.s - b.s) < epsilon &&
204 fabs(a.v - b.v) < epsilon;
205}

207color_t StrColor(const char *s)
208{
209 if (s == NULL || strlen(s) != 6)
210 {
211 return colorBlack;
212 }
213 int hex = (int)strtol(s, NULL, 16);
214 color_t c;
215 c.a = 255;
216 c.r = (hex >> 16) & 255;
217 c.g = (hex >> 8) & 255;
218 c.b = hex & 255;
219 return c;
220}
221void ColorStr(char *s, const color_t c)
222{
223 sprintf(s, "%02x%02x%02x", c.r, c.g, c.b);
224}
