47void PlayerDataAddOrUpdate(const NPlayerData pd)
48{
49 PlayerData *p = PlayerDataGetByUID(pd.UID);
50 if (p == NULL)
51 {
52 PlayerData pNew;
53 memset(&pNew, 0, sizeof pNew);
54 CArrayPushBack(&gPlayerDatas, &pNew);
55 p = CArrayGet(&gPlayerDatas, (int)gPlayerDatas.size - 1);
56
57 // Set defaults
58 p->ActorUID = -1;
59 p->IsLocal =
60 (int)pd.UID >= gNetClient.FirstPlayerUID &&
61 (int)pd.UID < gNetClient.FirstPlayerUID + MAX_LOCAL_PLAYERS;
62 p->inputDevice = INPUT_DEVICE_UNSET;
63
64 p->Char.speed = 256;
65
66 LOG(LM_MAIN, LL_INFO, "add default player UID(%u) local(%s)",
67 pd.UID, p->IsLocal ? "true" : "false");
68 }
69
70 p->UID = pd.UID;
71
72 strcpy(p->name, pd.Name);
73 p->Char.Class = StrCharacterClass(pd.CharacterClass);
74 if (p->Char.Class == NULL)
75 {
76 p->Char.Class = StrCharacterClass("Jones");
77 }
78 p->Char.Colors = Net2CharColors(pd.Colors);
79 p->weaponCount = pd.Weapons_count;
80 for (int i = 0; i < (int)pd.Weapons_count; i++)
81 {
82 p->weapons[i] = StrGunDescription(pd.Weapons[i]);
83 }
84 p->Lives = pd.Lives;
85 p->Stats = pd.Stats;
86 p->Totals = pd.Totals;
87 p->Char.maxHealth = pd.MaxHealth;
88 p->lastMission = pd.LastMission;
89
90 // Ready players as well
91 p->Ready = true;
92
93 LOG(LM_MAIN, LL_INFO, "update player UID(%d) maxHealth(%d)",
94 p->UID, p->Char.maxHealth);
95}

98void PlayerRemove(const int uid)
99{
100 // Find the player so we can remove by index
101 PlayerData *p = NULL;
102 int i;
103 for (i = 0; i < (int)gPlayerDatas.size; i++)
104 {
105 PlayerData *pi = CArrayGet(&gPlayerDatas, i);
106 if (pi->UID == uid)
107 {
108 p = pi;
109 break;
110 }
111 }
112 if (p == NULL)
113 {
114 return;
115 }
116 if (p->ActorUID >= 0)
117 {
118 ActorDestroy(ActorGetByUID(p->ActorUID));
119 }
120 PlayerTerminate(p);
121 CArrayDelete(&gPlayerDatas, i);
122
123 LOG(LM_MAIN, LL_INFO, "remove player UID(%d)", uid);
124}

127{
128 CFREE(p->Char.bot);
129}

254{

265int GetNumPlayers(
266 const PlayerAliveOptions alive, const bool human, const bool local)
267{
268 int numPlayers = 0;
269 CA_FOREACH(const PlayerData, p, gPlayerDatas)
270 bool life = false;
271 switch (alive)
272 {
273 case PLAYER_ANY: life = true; break;
274 case PLAYER_ALIVE: life = IsPlayerAlive(p); break;
275 case PLAYER_ALIVE_OR_DYING: life = IsPlayerAliveOrDying(p); break;
276 }
277 if (life &&
278 (!human || p->inputDevice != INPUT_DEVICE_AI) &&
279 (!local || p->IsLocal))
280 {
281 numPlayers++;
282 }
283 CA_FOREACH_END()
284 return numPlayers;
285}

311const PlayerData *GetFirstPlayer(
312 const bool alive, const bool human, const bool local)
313{
314 CA_FOREACH(const PlayerData, p, gPlayerDatas)
315 if ((!alive || IsPlayerAliveOrDying(p)) &&
316 (!human || p->inputDevice != INPUT_DEVICE_AI) &&
317 (!local || p->IsLocal))
318 {
319 return p;
320 }
321 CA_FOREACH_END()
322 return NULL;
323}

325bool IsPlayerAlive(const PlayerData *player)
326{
327 if (player->ActorUID == -1)
328 {
329 return false;
330 }
331 const TActor *p = ActorGetByUID(player->ActorUID);
332 return !p->dead;
333}
334bool IsPlayerHuman(const PlayerData *player)
335{
336 return player->inputDevice != INPUT_DEVICE_AI;
337}
338bool IsPlayerHumanAndAlive(const PlayerData *player)
339{
340 return IsPlayerAlive(player) && IsPlayerHuman(player);
341}
342bool IsPlayerAliveOrDying(const PlayerData *player)
343{
344 if (player->ActorUID == -1)
345 {
346 return false;
347 }
348 const TActor *p = ActorGetByUID(player->ActorUID);
349 return p->dead <= DEATH_MAX;
350}
351bool IsPlayerScreen(const PlayerData *p)
352{
353 const bool humanOnly =
354 IsPVP(gCampaign.Entry.Mode) ||
355 !ConfigGetBool(&gConfig, "Interface.SplitscreenAI");
356 const bool humanOrScreen = !humanOnly || p->inputDevice != INPUT_DEVICE_AI;
357 return p->IsLocal && humanOrScreen && IsPlayerAliveOrDying(p);
358}

360Vec2i PlayersGetMidpoint(void)
361{
362 // for all surviving players, find bounding rectangle, and get center
363 Vec2i min;
364 Vec2i max;
365 PlayersGetBoundingRectangle(&min, &max);
366 return Vec2iScaleDiv(Vec2iAdd(min, max), 2);
367}
368
369void PlayersGetBoundingRectangle(Vec2i *min, Vec2i *max)
370{
371 bool isFirst = true;
372 *min = Vec2iZero();
373 *max = Vec2iZero();
374 const bool humansOnly =
375 GetNumPlayers(PLAYER_ALIVE_OR_DYING, true, false) > 0;
376 CA_FOREACH(const PlayerData, p, gPlayerDatas)
377 if (!p->IsLocal)
378 {
379 continue;
380 }
381 if (humansOnly ? IsPlayerHumanAndAlive(p) : IsPlayerAlive(p))
382 {
383 const TActor *player = ActorGetByUID(p->ActorUID);
384 const TTileItem *ti = &player->tileItem;
385 if (isFirst)
386 {
387 *min = *max = Vec2iNew(ti->x, ti->y);
388 }
389 else
390 {
391 if (ti->x < min->x) min->x = ti->x;
392 if (ti->y < min->y) min->y = ti->y;
393 if (ti->x > max->x) max->x = ti->x;
394 if (ti->y > max->y) max->y = ti->y;
395 }
396 isFirst = false;
397 }
398 CA_FOREACH_END()
399}

401// Get the number of players that use this ammo
402int PlayersNumUseAmmo(const int ammoId)
403{
404 int numPlayersWithAmmo = 0;
405 CA_FOREACH(const PlayerData, p, gPlayerDatas)
406 if (!IsPlayerAlive(p))
407 {
408 continue;
409 }
410 const TActor *player = ActorGetByUID(p->ActorUID);
411 for (int j = 0; j < (int)player->guns.size; j++)
412 {
413 const Weapon *w = CArrayGet(&player->guns, j);
414 if (w->Gun->AmmoId == ammoId)
415 {
416 numPlayersWithAmmo++;
417 }
418 }
419 CA_FOREACH_END()
420 return numPlayersWithAmmo;
421}

439bool PlayerTrySetInputDevice(
440 PlayerData *p, const input_device_e d, const int idx)
441{
442 if (p->inputDevice == d && p->deviceIndex == idx)
443 {
444 return false;
445 }
446 p->inputDevice = d;
447 p->deviceIndex = idx;
448 LOG(LM_MAIN, LL_DEBUG, "playerUID(%d) assigned input device(%d %d)",
449 p->UID, (int)d, idx);
450 return true;
451}
452
453bool PlayerTrySetUnusedInputDevice(
454 PlayerData *p, const input_device_e d, const int idx)
455{
456 // Check that player's input device is unassigned
457 if (p->inputDevice != INPUT_DEVICE_UNSET) return false;
458 // Check that no players use this input device
459 CA_FOREACH(const PlayerData, pOther, gPlayerDatas)
460 if (pOther->inputDevice == d && pOther->deviceIndex == idx)
461 {
462 return false;
463 }
464 CA_FOREACH_END()
465 return PlayerTrySetInputDevice(p, d, idx);
466}
