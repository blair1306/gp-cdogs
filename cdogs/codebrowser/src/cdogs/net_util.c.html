35ENetPacket *NetEncode(const GameEventType e, const void *data)
36{
37 uint8_t buffer[1024];
38 pb_ostream_t stream = pb_ostream_from_buffer(buffer, sizeof buffer);
39 const pb_field_t *fields = GameEventGetEntry(e).Fields;
40 const bool status =
41 (data && fields) ? pb_encode(&stream, fields, data) : true;
42 CASSERT(status, "Failed to encode pb");
43 int msgId = (int)e;
44 ENetPacket *packet = enet_packet_create(
45 &msgId, NET_MSG_SIZE + stream.bytes_written,
46 ENET_PACKET_FLAG_RELIABLE);
47 memcpy(packet->data + NET_MSG_SIZE, buffer, stream.bytes_written);
48 return packet;
49}
50
51bool NetDecode(
52 ENetPacket *packet, void *dest, const pb_field_t *fields)
53{
54 pb_istream_t stream = pb_istream_from_buffer(
55 packet->data + NET_MSG_SIZE, packet->dataLength - NET_MSG_SIZE);
56 bool status = pb_decode(&stream, fields, dest);
57 CASSERT(status, "Failed to decode pb");
58 return status;
59}

62NPlayerData NMakePlayerData(const PlayerData *p)
63{
64 NPlayerData d = NPlayerData_init_default;
65 const Character *c = &p->Char;
66 strcpy(d.Name, p->name);
67 strcpy(d.CharacterClass, p->Char.Class->Name);
68 d.Colors = CharColors2Net(c->Colors);
69 d.Weapons_count = (pb_size_t)p->weaponCount;
70 for (int i = 0; i < (int)d.Weapons_count; i++)
71 {
72 strcpy(d.Weapons[i], p->weapons[i]->name);
73 }
74 d.Lives = p->Lives;
75 d.Stats = p->Stats;
76 d.Totals = p->Totals;
77 d.MaxHealth = p->Char.maxHealth;
78 d.LastMission = p->lastMission;
79 d.UID = p->UID;
80 return d;
81}
82NCampaignDef NMakeCampaignDef(const CampaignOptions *co)
83{
84 NCampaignDef def;
85 memset(&def, 0, sizeof def);
86 if (co->Entry.Path)
87 {
88 strcpy((char *)def.Path, co->Entry.Path);
89 }
90 def.GameMode = co->Entry.Mode;
91 def.Mission = co->MissionIndex;
92 return def;
93}
94NMissionComplete NMakeMissionComplete(
95 const struct MissionOptions *mo, const Map *map)
96{
97 NMissionComplete mc;
98 mc.ShowMsg = MissionHasRequiredObjectives(mo);
99 mc.ExitStart = Vec2i2Net(map->ExitStart);
100 mc.ExitEnd = Vec2i2Net(map->ExitEnd);
101 return mc;
102}

115color_t Net2Color(const NColor c)
116{
117 color_t co;
118 co.r = (uint8_t)((c.RGBA & 0xff000000) >> 24);
119 co.g = (uint8_t)((c.RGBA & 0x00ff0000) >> 16);
120 co.b = (uint8_t)((c.RGBA & 0x0000ff00) >> 8);
121 co.a = (uint8_t)(c.RGBA & 0x000000ff);
122 return co;
123}
124NColor Color2Net(const color_t c)
125{
126 NColor co;
127 co.RGBA = (c.r << 24) | (c.g << 16) | (c.b << 8) | c.a;
128 return co;
129}
130NCharColors CharColors2Net(const CharColors c)
131{
132 NCharColors co;
133 co.Skin = Color2Net(c.Skin);
134 co.Arms = Color2Net(c.Arms);
135 co.Body = Color2Net(c.Body);
136 co.Legs = Color2Net(c.Legs);
137 co.Hair = Color2Net(c.Hair);
138 return co;
139}
140CharColors Net2CharColors(const NCharColors c)
141{
142 CharColors co;
143 co.Skin = Net2Color(c.Skin);
144 co.Arms = Net2Color(c.Arms);
145 co.Body = Net2Color(c.Body);
146 co.Legs = Net2Color(c.Legs);
147 co.Hair = Net2Color(c.Hair);
148 return co;
149}
