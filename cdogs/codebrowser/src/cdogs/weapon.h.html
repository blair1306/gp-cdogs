83// Gun states
84typedef enum
85{
86 GUNSTATE_READY,
87 GUNSTATE_FIRING,
88 GUNSTATE_RECOIL,
89 GUNSTATE_COUNT
90} gunstate_e;

92typedef struct
93{
94 const NamedSprites *Pic;
95 const Pic *Icon;
96 char *name;
97 char *Description;
98 const BulletClass *Bullet;
99 int AmmoId; // -1 if the gun does not consume ammo
100 int Cost; // Cost in score to fire weapon
101 int Lock;
102 int ReloadLead;
103 Mix_Chunk *Sound;
104 Mix_Chunk *ReloadSound;
105 Mix_Chunk *SwitchSound;
106 int SoundLockLength;
107 double Recoil; // Random recoil for inaccurate weapons, in radians
108 struct
109 {
110 int Count; // Number of bullets in spread
111 double Width; // Width of individual spread, in radians
112 } Spread;
113 double AngleOffset;
114 int MuzzleHeight;
115 int ElevationLow;
116 int ElevationHigh;
117 const ParticleClass *MuzzleFlash;
118 const ParticleClass *Brass;
119 bool CanShoot;
120 bool CanDrop; // whether this gun can be dropped to be picked up
121 int ShakeAmount; // Amount of screen shake to produce
122 bool IsRealGun; // whether this gun can be used as is by players
123} GunDescription;
124typedef struct
125{
126 CArray Guns; // of GunDescription
127 GunDescription Default;
128 CArray CustomGuns; // of GunDescription
129} GunClasses;
130
131typedef struct
132{
133 const GunDescription *Gun;
134 gunstate_e state;
135 int lock;
136 int soundLock;
137 int clickLock;
138 int stateCounter;
139} Weapon;
