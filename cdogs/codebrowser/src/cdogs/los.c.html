36void LOSInit(Map *map, const Vec2i size)
37{
38 CArrayInit(&map->LOS.LOS, sizeof(bool));
39 CArrayInit(&map->LOS.Explored, sizeof(bool));
40 Vec2i v;
41 for (v.y = 0; v.y < size.y; v.y++)
42 {
43 for (v.x = 0; v.x < size.x; v.x++)
44 {
45 const bool f = false;
46 CArrayPushBack(&map->LOS.LOS, &f);
47 CArrayPushBack(&map->LOS.Explored, &f);
48 }
49 }
50}
51void LOSTerminate(LineOfSight *los)
52{
53 CArrayTerminate(&los->LOS);
54 CArrayTerminate(&los->Explored);
55}

57// Reset lines of sight by setting all cells to unseen
58void LOSReset(LineOfSight *los)
59{
60 CArrayFillZero(&los->LOS);
61 CArrayFillZero(&los->Explored);
62}
63void LOSSetAllVisible(LineOfSight *los)
64{
65 CA_FOREACH(bool, l, los->LOS)
66 *l = true;
67 CA_FOREACH_END()
68}

70typedef struct
71{
72 Map *Map;
73 Vec2i Center;
74 int SightRange2;
75 bool Explore;
76} LOSData;
77// Calculate LOS cells from a certain start position
78// Sight range based on config
79static void SetLOSVisible(Map *map, const Vec2i pos, const bool explore);
80static bool IsNextTileBlockedAndSetVisibility(void *data, Vec2i pos);
81static void SetObstructionVisible(
82 Map *map, const Vec2i pos, const bool explore);
83void LOSCalcFrom(Map *map, const Vec2i pos, const bool explore)
84{
85 // Perform LOS by casting rays from the centre to the edges, terminating
86 // whenever an obstruction or out-of-range is reached.
87
88 CArrayFillZero(&map->LOS.Explored);
89
90 // First mark center tile and all adjacent tiles as visible
91 // +-+-+-+
92 // |V|V|V|
93 // +-+-+-+
94 // |V|C|V|
95 // +-+-+-+
96 // |V|V|V| (C=center, V=visible)
97 // +-+-+-+
98 Vec2i end;
99 for (end.x = pos.x - 1; end.x <= pos.x + 1; end.x++)
100 {
101 for (end.y = pos.y - 1; end.y <= pos.y + 1; end.y++)
102 {
103 SetLOSVisible(map, end, explore);
104 }
105 }
106
107 const int sightRange = ConfigGetInt(&gConfig, "Game.SightRange");
108 if (sightRange == 0) return;
109
110 // Limit the perimeter to the sight range
111 const Vec2i origin = Vec2iNew(pos.x - sightRange, pos.y - sightRange);
112 const Vec2i perimSize = Vec2iScale(Vec2iMinus(pos, origin), 2);
113
114 LOSData data;
115 data.Map = map;
116 data.Center = pos;
117 data.SightRange2 = sightRange * sightRange;
118 data.Explore = explore;
119
120 // Start from the top-left cell, and proceed clockwise around
121 end = origin;
122 HasClearLineData lineData;
123 lineData.IsBlocked = IsNextTileBlockedAndSetVisibility;
124 lineData.data = &data;
125 // Top edge
126 for (; end.x < origin.x + perimSize.x; end.x++)
127 {
128 HasClearLineJMRaytrace(pos, end, &lineData);
129 }
130 // right edge
131 for (; end.y < origin.y + perimSize.y; end.y++)
132 {
133 HasClearLineJMRaytrace(pos, end, &lineData);
134 }
135 // bottom edge
136 for (; end.x > origin.x; end.x--)
137 {
138 HasClearLineJMRaytrace(pos, end, &lineData);
139 }
140 // left edge
141 for (; end.y > origin.y; end.y--)
142 {
143 HasClearLineJMRaytrace(pos, end, &lineData);
144 }
145
146 // Second pass: make any non-visible obstructions that are adjacent to
147 // visible non-obstructions visible too
148 // This is to ensure runs of walls stay visible
149 for (end.y = origin.y; end.y < origin.y + perimSize.y; end.y++)
150 {
151 for (end.x = origin.x; end.x < origin.x + perimSize.x; end.x++)
152 {
153 const Tile *tile = MapGetTile(map, end);
154 if (!tile || !(tile->flags & MAPTILE_NO_SEE))
155 {
156 continue;
157 }
158 // Check sight range
159 if (DistanceSquared(pos, end) >= data.SightRange2)
160 {
161 continue;
162 }
163 SetObstructionVisible(map, end, explore);
164 }
165 }
166
167 // Find all the newly visible tiles and set events for them
168 GameEvent e = GameEventNew(GAME_EVENT_EXPLORE_TILES);
169 e.u.ExploreTiles.Runs_count = 0;
170 e.u.ExploreTiles.Runs[0].Run = 0;
171 bool run = false;
172 for (end.y = 0; end.y < map->Size.y; end.y++)
173 {
174 for (end.x = 0; end.x < map->Size.x; end.x++)
175 {
176 if (LOSAddRun(
177 &e.u.ExploreTiles, &run, end,
178 *((bool *)CArrayGet(&map->LOS.Explored, end.y * map->Size.x + end.x))))
179 {
180 GameEventsEnqueue(&gGameEvents, e);
181 e.u.ExploreTiles.Runs_count = 0;
182 e.u.ExploreTiles.Runs[0].Run = 0;
183 run = false;
184 }
185 }
186 }
187 if (e.u.ExploreTiles.Runs_count > 0)
188 {
189 GameEventsEnqueue(&gGameEvents, e);
190 }
191 CArrayFillZero(&map->LOS.Explored);
192}
193static void SetLOSVisible(Map *map, const Vec2i pos, const bool explore)
194{
195 const Tile *t = MapGetTile(map, pos);
196 if (t == NULL) return;
197 *((bool *)CArrayGet(&map->LOS.LOS, pos.y * map->Size.x + pos.x)) = true;
198 if (!t->isVisited && explore)
199 {
200 // Cache the newly explored tile
201 *((bool *)CArrayGet(&map->LOS.Explored, pos.y * map->Size.x + pos.x)) = true;
202 }
203 // Mark any actors on this tile as visible
204 // This affects some AI
205 CA_FOREACH(ThingId, tid, t->things)
206 const TTileItem *ti = ThingIdGetTileItem(tid);
207 if (ti->kind == KIND_CHARACTER)
208 {
209 TActor *a = CArrayGet(&gActors, ti->id);
210 a->flags |= FLAGS_VISIBLE;
211 }
212 CA_FOREACH_END()
213}
214static bool IsNextTileBlockedAndSetVisibility(void *data, Vec2i pos)
215{
216 LOSData *lData = data;
217 // Check sight range
218 if (DistanceSquared(lData->Center, pos) >= lData->SightRange2) return true;
219 // Check map range
220 const Tile *t = MapGetTile(lData->Map, pos);
221 if (t == NULL) return true;
222 SetLOSVisible(lData->Map, pos, lData->Explore);
223 // Check if this tile is an obstruction
224 return t->flags & MAPTILE_NO_SEE;
225}
226static bool IsTileVisibleNonObstruction(Map *map, const Vec2i pos);
227static void SetObstructionVisible(
228 Map *map, const Vec2i pos, const bool explore)
229{
230 Vec2i d;
231 for (d.x = -1; d.x < 2; d.x++)
232 {
233 for (d.y = -1; d.y < 2; d.y++)
234 {
235 if (IsTileVisibleNonObstruction(map, Vec2iAdd(pos, d)))
236 {
237 SetLOSVisible(map, pos, explore);
238 return;
239 }
240 }
241 }
242}
243static bool IsTileVisibleNonObstruction(Map *map, const Vec2i pos)
244{
245 const Tile *t = MapGetTile(map, pos);
246 if (t == NULL) return false;
247 return !(t->flags & MAPTILE_NO_SEE) && LOSTileIsVisible(map, pos);
248}

250bool LOSAddRun(
251 NExploreTiles *runs, bool *run, const Vec2i tile, const bool explored)
252{
253 if (explored)
254 {
255 if (!*run)
256 {
257 // Start of new run
258 runs->Runs_count++;
259 runs->Runs[runs->Runs_count - 1].Tile = Vec2i2Net(tile);
260 runs->Runs[runs->Runs_count - 1].Run = 0;
261 }
262 runs->Runs[runs->Runs_count - 1].Run++;
263 *run = true;
264 }
265 else
266 {
267 // End of run
268 // If we have too many runs, send off the event and start a new one
269 if (runs->Runs_count == sizeof runs->Runs / sizeof runs->Runs[0])
270 {
271 return true;
272 }
273 *run = false;
274 }
275 return false;
276}

278bool LOSTileIsVisible(Map *map, const Vec2i pos)
279{
280 if (MapGetTile(map, pos) == NULL) return false;
281 return *((bool *)CArrayGet(&map->LOS.LOS, pos.y * map->Size.x + pos.x));
282}
