41static CachedPath CachedPathCopy(CachedPath *c)
42{
43 CachedPath copy;
44 memcpy(&copy, c, sizeof *c);
45 (*copy.refs)++;
46 return copy;
47}
48void CachedPathDestroy(CachedPath *c)
49{
50 if (c->Path == NULL && c->refs == NULL)
51 {
52 return;
53 }
54 (*c->refs)--;
55 CASSERT(*c->refs >= 0, "out of sync ref count");
56 if (*c->refs == 0)
57 {
58 ASPathDestroy(c->Path);
59 CFREE(c->refs);
60 }
61}

63static bool CachedPathMatches(
64 const CachedPath *c, const Vec2i from, const Vec2i to)
65{
66 return Vec2iEqual(c->from, from) && Vec2iEqual(c->to, to);
67}

70void PathCacheInit(PathCache *pc, Map *m)
71{
72 CArrayInit(&pc->paths, sizeof(CachedPath));
73 pc->head = 0;
74 pc->map = m;
75}
76void PathCacheTerminate(PathCache *pc)
77{
78 PathCacheClear(pc);
79 CArrayTerminate(&pc->paths);
80}
81
82void PathCacheClear(PathCache *pc)
83{
84 CA_FOREACH(CachedPath, c, pc->paths)
85 CachedPathDestroy(c);
86 CA_FOREACH_END()
87 CArrayClear(&pc->paths);
88 pc->head = 0;
89}

91typedef struct
92{
93 Map *Map;
94 TileSelectFunc IsTileOk;
95} AStarContext;
96static void AddTileNeighbors(
97 ASNeighborList neighbors, void *node, void *context);
98static float AStarHeuristic(void *fromNode, void *toNode, void *context);
99static ASPathNodeSource cPathNodeSource =
100{
101 sizeof(Vec2i), AddTileNeighbors, AStarHeuristic, NULL, NULL
102};
103CachedPath PathCacheCreate(
104 PathCache *pc, Vec2i from, Vec2i to,
105 const bool ignoreObjects, const bool cache)
106{
107 // Search through existing cache for path
108 CA_FOREACH(CachedPath, c, pc->paths)
109 if (CachedPathMatches(c, from, to))
110 {
111 LOG(LM_PATH, LL_TRACE, "cached path (%d, %d) to (%d, %d)...",
112 from.x, from.y, to.x, to.y);
113 return CachedPathCopy(c);
114 }
115 CA_FOREACH_END()
116
117 LOG(LM_PATH, LL_TRACE, "find path (%d, %d) to (%d, %d)...",
118 from.x, from.y, to.x, to.y);
119 const clock_t start = clock();
120
121 // Cached path not found; find the path now
122 CachedPath cp;
123 AStarContext ac;
124 ac.Map = pc->map;
125 ac.IsTileOk = ignoreObjects ? IsTileWalkable : IsTileWalkableAroundObjects;
126 cp.Path = ASPathCreate(&cPathNodeSource, &ac, &from, &to);
127 CMALLOC(cp.refs, sizeof *cp.refs);
128 (*cp.refs) = 1;
129 cp.from = from;
130 cp.to = to;
131 // Cache the path, optionally
132 if (cache)
133 {
134 (*cp.refs)++;
135 // Add to the cache if we are under the max size
136 if ((int)pc->paths.size < PATH_CACHE_MAX)
137 {
138 CArrayPushBack(&pc->paths, &cp);
139 }
140 else
141 {
142 // Replace the oldest cached path with this one
143 CachedPath *oldest = CArrayGet(&pc->paths, pc->head);
144 CachedPathDestroy(oldest);
145 memcpy(oldest, &cp, sizeof cp);
146 // Move the head
147 pc->head++;
148 if (pc->head == pc->paths.size)
149 {
150 pc->head = 0;
151 }
152 }
153 LOG(LM_PATH, LL_TRACE, "Cached %d paths", (int)pc->paths.size);
154 }
155 const clock_t diff = clock() - start;
156 const int ms = diff * 1000 / CLOCKS_PER_SEC;
157 LOG(LM_PATH, LL_DEBUG, "Pathfind time %dms", ms);
158 return cp;
159}

161static void AddTileNeighbors(
162 ASNeighborList neighbors, void *node, void *context)
163{
164 Vec2i *v = node;
165 int y;
166 AStarContext *c = context;
167 for (y = v->y - 1; y <= v->y + 1; y++)
168 {
169 int x;
170 if (y < 0 || y >= c->Map->Size.y)
171 {
172 continue;
173 }
174 for (x = v->x - 1; x <= v->x + 1; x++)
175 {
176 float cost;
177 Vec2i neighbor;
178 neighbor.x = x;
179 neighbor.y = y;
180 if (x < 0 || x >= c->Map->Size.x)
181 {
182 continue;
183 }
184 if (x == v->x && y == v->y)
185 {
186 continue;
187 }
188 // if we're moving diagonally,
189 // need to check the axis-aligned neighbours are also clear
190 if (!c->IsTileOk(c->Map, Vec2iNew(x, y)) ||
191 !c->IsTileOk(c->Map, Vec2iNew(v->x, y)) ||
192 !c->IsTileOk(c->Map, Vec2iNew(x, v->y)))
193 {
194 continue;
195 }
196 // Calculate cost of direction
197 // Note that there are different horizontal and vertical costs,
198 // due to the tiles being non-square
199 // Slightly prefer axes instead of diagonals
200 if (x != v->x && y != v->y)
201 {
202 cost = TILE_WIDTH * 1.1f;
203 }
204 else if (x != v->x)
205 {
206 cost = TILE_WIDTH;
207 }
208 else
209 {
210 cost = TILE_HEIGHT;
211 }
212 ASNeighborListAdd(neighbors, &neighbor, cost);
213 }
214 }
215}
216static float AStarHeuristic(void *fromNode, void *toNode, void *context)
217{
218 // Simple Euclidean
219 Vec2i *v1 = fromNode;
220 Vec2i *v2 = toNode;
221 UNUSED(context);
222 return (float)sqrt(DistanceSquared(
223 Vec2iCenterOfTile(*v1), Vec2iCenterOfTile(*v2)));
224}
