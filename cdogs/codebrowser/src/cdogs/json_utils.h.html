63#define JSON_UTILS_ADD_ENUM_PAIR(parent, name, value, func)\
64 json_insert_pair_into_object(\
65 (parent), (name), json_new_string(func(value)));
66
67int TryLoadValue(json_t **node, const char *name);
68#define JSON_UTILS_LOAD_ENUM(value, node, name, func)\
69 {\
70 json_t *_node = (node);\
71 if (TryLoadValue(&_node, (name)))\
72 {\
73 (value) = func(_node->text);\
74 }\
75 }
