64void MapClassicLoad(Map *map, const Mission *m, const CampaignOptions* co)
65{
66 // The classic random map generator randomly attempts to place
67 // a configured number of features on the map, in order:
68 // 1. "Squares", almost-square areas of empty floor
69 // 2. Rooms, rectangles of "room" squares surrounded by walls,
70 // some of which have doors
71 // 3. "Pillars", or rectangles of solid wall
72 // 4. Walls, single-thickness walls that sometimes bend and
73 // split
74 // All features are placed by randomly picking a position in
75 // the map and checking to see if it's possible to place the
76 // feature at that point, and repeating this process until
77 // either all features are placed or too many attempts have
78 // been done.
79 // Sometimes it's impossible to place features, either because
80 // they overlap with other incompatible features, or it may
81 // create inaccessible areas on the map.
82
83 // Re-seed RNG so results are consistent
84 CampaignSeedRandom(co);
85
86 MapSetupPerimeter(map);
87
88 const int pad = MAX(m->u.Classic.CorridorWidth, 1);
89
90 // place squares
91 int count = 0;
92 int i = 0;
93 while (i < 1000 && count < m->u.Classic.Squares)
94 {
95 if (MapTryBuildSquare(map))
96 {
97 count++;
98 }
99 i++;
100 }
101
102 // place rooms
103 count = 0;
104 for (i = 0; i < 1000 && count < m->u.Classic.Rooms.Count; i++)
105 {
106 const Vec2i v = MapGetRandomTile(map);
107 const int doorMin = CLAMP(m->u.Classic.Doors.Min, 1, 6);
108 const int doorMax = CLAMP(m->u.Classic.Doors.Max, doorMin, 6);
109 const Vec2i size = MapGetRoomSize(m->u.Classic.Rooms, doorMin);
110 bool isOverlapRoom;
111 unsigned short overlapAccess;
112 if (!MapIsAreaClearForClassicRoom(
113 map, v, size, m, pad, &isOverlapRoom, &overlapAccess))
114 {
115 continue;
116 }
117 MapBuildRoom(
118 map, v, size, m, doorMin, doorMax,
119 AreKeysAllowed(gCampaign.Entry.Mode), isOverlapRoom, overlapAccess);
120 count++;
121 }
122
123 // place pillars
124 count = 0;
125 i = 0;
126 while (i < 1000 && count < m->u.Classic.Pillars.Count)
127 {
128 if (MapTryBuildPillar(map, m, pad))
129 {
130 count++;
131 }
132 i++;
133 }
134
135 // place walls
136 count = 0;
137 i = 0;
138 while (i < 1000 && count < m->u.Classic.Walls)
139 {
140 if (MapTryBuildWall(
141 map, MAP_FLOOR, pad, m->u.Classic.WallLength))
142 {
143 count++;
144 }
145 i++;
146 }
147}

149static void MapSetupPerimeter(Map *map)
150{
151 Vec2i v;
152 for (v.y = 0; v.y < map->Size.y; v.y++)
153 {
154 for (v.x = 0; v.x < map->Size.x; v.x++)
155 {
156 if (v.y == 0 || v.y == map->Size.y - 1 ||
157 v.x == 0 || v.x == map->Size.x - 1)
158 {
159 IMapSet(map, v, MAP_WALL);
160 }
161 }
162 }
163}

165static int MapTryBuildSquare(Map *map)
166{
167 const Vec2i v = MapGetRandomTile(map);
168 Vec2i size = Vec2iNew(rand() % 9 + 8, rand() % 9 + 8);
169 if (MapIsAreaClear(map, v, size))
170 {
171 MapMakeSquare(map, v, size);
172 return 1;
173 }
174 return 0;
175}

177static bool MapIsAreaClearForClassicRoom(
178 const Map *map, const Vec2i pos, const Vec2i size, const Mission *m,
179 const int pad, bool *isOverlapRoom, unsigned short *overlapAccess)
180{
181 Vec2i clearPos = Vec2iNew(pos.x - pad, pos.y - pad);
182 Vec2i clearSize = Vec2iNew(size.x + 2 * pad, size.y + 2 * pad);
183 bool isEdgeRoom = false;
184 *isOverlapRoom = false;
185 *overlapAccess = 0;
186
187 if (m->u.Classic.Rooms.Edge)
188 {
189 // Check if room is at edge; if so only check if clear inside edge
190 if (pos.x == 0 || pos.x == 1)
191 {
192 const int dx = 1 - clearPos.x;
193 clearPos.x += dx;
194 clearSize.x -= dx;
195 isEdgeRoom = true;
196 }
197 else if (pos.x + size.x == map->Size.x - 2 ||
198 pos.x + size.x == map->Size.x - 1)
199 {
200 clearSize.x = map->Size.x - 1 - pos.x;
201 isEdgeRoom = true;
202 }
203 if (pos.y == 0 || pos.y == 1)
204 {
205 const int dy = 1 - clearPos.y;
206 clearPos.y += dy;
207 clearSize.y -= dy;
208 isEdgeRoom = true;
209 }
210 else if (pos.y + size.y == map->Size.y - 2 ||
211 pos.y + size.y == map->Size.y - 1)
212 {
213 clearSize.y = map->Size.y - 1 - pos.y;
214 isEdgeRoom = true;
215 }
216 }
217 bool isClear = MapIsAreaClear(map, clearPos, clearSize);
218 // Don't let rooms be both edge rooms and overlap rooms
219 // Otherwise dead pockets will be created
220 if (!isClear && !isEdgeRoom)
221 {
222 // If room overlap is enabled, check if it overlaps with a room
223 const bool isOverlap =
224 m->u.Classic.Rooms.Overlap &&
225 MapIsAreaClearOrRoom(map, clearPos, clearSize);
226 // Now check if the overlapping rooms will create a passage
227 // large enough
228 const int roomOverlapSize = MapGetRoomOverlapSize(
229 map, pos, size, overlapAccess);
230 isClear = isOverlap && roomOverlapSize >= m->u.Classic.CorridorWidth;
231 *isOverlapRoom = true;
232 }
233 return isClear;
234}

241static void MapBuildRoom(
242 Map *map, const Vec2i pos, const Vec2i size, const Mission *m,
243 const int doorMin, const int doorMax, const bool hasKeys,
244 const bool isOverlapRoom, const unsigned short overlapAccess)
245{
246 int doormask = rand() % 15 + 1;
247 int doors[4];
248 int doorsUnplaced = 0;
249 int i;
250 unsigned short accessMask = 0;
251
252 MapMakeRoom(map, pos, size, true);
253 // Check which walls we can place doors
254 // If we cannot place doors, remember this and try to place them
255 // on other walls
256 // We cannot place doors on: the perimeter, and on overlaps
257 MapFindAvailableDoors(map, pos, size, doorMin, doors);
258 // Try to place doors according to the random mask
259 // If we cannot place a door, remember this and try to place it
260 // on other doors
261 for (i = 0; i < 4; i++)
262 {
263 if ((doormask & (1 << i)) && !doors[i])
264 {
265 doorsUnplaced++;
266 }
267 }
268 for (i = 0; i < 4; i++)
269 {
270 if (!(doormask & (1 << i)))
271 {
272 if (doorsUnplaced == 0)
273 {
274 // If we don't need to place any doors,
275 // set this door to unavailable
276 doors[i] = 0;
277 }
278 else if (doors[i])
279 {
280 // Otherwise, if it's possible to place a door here
281 // and we need to place doors, do so
282 doorsUnplaced--;
283 }
284 }
285 }
286
287 // Work out what access level (i.e. key) this room has
288 if (hasKeys && m->u.Classic.Doors.Enabled)
289 {
290 // If this room has overlapped another room, use the same
291 // access level as that room
292 if (isOverlapRoom)
293 {
294 accessMask = overlapAccess;
295 }
296 else
297 {
298 // Otherwise, generate an access level for this room
299 accessMask = GenerateAccessMask(&map->keyAccessCount);
300 if (map->keyAccessCount < 1)
301 {
302 map->keyAccessCount = 1;
303 }
304 }
305 }
306 MapPlaceDoors(map, pos, size,
307 m->u.Classic.Doors.Enabled, doors, doorMin, doorMax, accessMask);
308
309 MapMakeRoomWalls(map, m->u.Classic.Rooms);
310}

312static bool MapTryBuildPillar(Map *map, const Mission *m, const int pad)
313{
314 int pillarMin = m->u.Classic.Pillars.Min;
315 int pillarMax = m->u.Classic.Pillars.Max;
316 Vec2i size = Vec2iNew(
317 rand() % (pillarMax - pillarMin + 1) + pillarMin,
318 rand() % (pillarMax - pillarMin + 1) + pillarMin);
319 const Vec2i pos = MapGetRandomTile(map);
320 Vec2i clearPos = Vec2iNew(pos.x - pad, pos.y - pad);
321 Vec2i clearSize = Vec2iNew(size.x + 2 * pad, size.y + 2 * pad);
322 int isEdge = 0;
323 int isClear = 0;
324
325 // Check if pillar is at edge; if so only check if clear inside edge
326 if (pos.x == 0 || pos.x == 1)
327 {
328 int dx = 1 - clearPos.x;
329 clearPos.x += dx;
330 clearSize.x -= dx;
331 isEdge = 1;
332 }
333 else if (pos.x + size.x == map->Size.x - 2 ||
334 pos.x + size.x == map->Size.x - 1)
335 {
336 clearSize.x = map->Size.x - 1 - pos.x;
337 isEdge = 1;
338 }
339 if (pos.y == 0 || pos.y == 1)
340 {
341 int dy = 1 - clearPos.y;
342 clearPos.y += dy;
343 clearSize.y -= dy;
344 isEdge = 1;
345 }
346 else if (pos.y + size.y == map->Size.y - 2 ||
347 pos.y + size.y == map->Size.y - 1)
348 {
349 clearSize.y = map->Size.y - 1 - pos.y;
350 isEdge = 1;
351 }
352
353 // Only place pillars if the area is totally clear,
354 // or if the pillar only overlaps one of the edge or another
355 // non-room wall
356 // This is to prevent dead pockets
357 isClear = MapIsAreaClear(map, clearPos, clearSize);
358 if (!isClear && !isEdge && MapIsAreaClearOrWall(map, clearPos, clearSize))
359 {
360 // Also check that the pillar does not overlap two pillars
361 isClear = MapIsLessThanTwoWallOverlaps(map, clearPos, clearSize);
362 }
363 if (isClear)
364 {
365 MapMakePillar(map, pos, size);
366 return 1;
367 }
368 return 0;
369}

374static void MapFindAvailableDoors(
375 Map *map, Vec2i pos, Vec2i size, int doorMin, int doors[4])
376{
377 int i;
378 for (i = 0; i < 4; i++)
379 {
380 doors[i] = 1;
381 }
382 // left
383 if (pos.x <= 1)
384 {
385 doors[0] = 0;
386 }
387 else if (FindWallRun(
388 map,
389 Vec2iNew(pos.x, pos.y + size.y / 2),
390 Vec2iNew(0, 1),
391 size.y - 2) < doorMin)
392 {
393 doors[0] = 0;
394 }
395 // right
396 if (pos.x + size.x >= map->Size.x - 2)
397 {
398 doors[1] = 0;
399 }
400 else if (FindWallRun(
401 map,
402 Vec2iNew(pos.x + size.x - 1, pos.y + size.y / 2),
403 Vec2iNew(0, 1),
404 size.y - 2) < doorMin)
405 {
406 doors[1] = 0;
407 }
408 // top
409 if (pos.y <= 1)
410 {
411 doors[2] = 0;
412 }
413 else if (FindWallRun(
414 map,
415 Vec2iNew(pos.x + size.x / 2, pos.y),
416 Vec2iNew(1, 0),
417 size.x - 2) < doorMin)
418 {
419 doors[2] = 0;
420 }
421 // bottom
422 if (pos.y >= map->Size.y - 2)
423 {
424 doors[3] = 0;
425 }
426 else if (FindWallRun(
427 map,
428 Vec2iNew(pos.x + size.x / 2, pos.y + size.y - 1),
429 Vec2iNew(1, 0),
430 size.x - 2) < doorMin)
431 {
432 doors[3] = 0;
433 }
434}
435static int FindWallRun(
436 const Map *map, const Vec2i mid, const Vec2i d, const int len)
437{
438 int run = 0;
439 int next = 0;
440 bool plus = false;
441 // Find the wall run by starting from a midpoint and expanding outwards in
442 // both directions, in a series 0, 1, -1, 2, -2...
443 for (int i = 0; i < len; i++, run++)
444 {
445 // Check if this is a wall so we can add a door here
446 // Also check if the two tiles aside are not walls
447
448 // Note: we must look for runs
449
450 if (plus)
451 {
452 next += i;
453 }
454 else
455 {
456 next -= i;
457 }
458 const Vec2i v = Vec2iAdd(mid, Vec2iScale(d, next));
459 plus = !plus;
460
461 if (IMapGet(map, v) != MAP_WALL ||
462 IMapGet(map, Vec2iNew(v.x + d.y, v.y + d.x)) == MAP_WALL ||
463 IMapGet(map, Vec2iNew(v.x - d.y, v.y - d.x)) == MAP_WALL)
464 {
465 break;
466 }
467 }
468 return run;
469}
