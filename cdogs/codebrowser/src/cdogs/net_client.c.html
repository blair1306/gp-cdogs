52void NetClientInit(NetClient *n)
53{
54 memset(n, 0, sizeof *n);
55 n->ClientId = -1; // -1 is unset
56 n->scanner = ENET_SOCKET_NULL;
57 n->client = enet_host_create(NULL, 1, 2,
58 57600 / 8 /* 56K modem with 56 Kbps downstream bandwidth */,
59 14400 / 8 /* 56K modem with 14 Kbps upstream bandwidth */);
60 if (n->client == NULL)
61 {
62 LOG(LM_NET, LL_ERROR, "cannot create ENet client host");
63 }
64 CArrayInit(&n->ScannedAddrs, sizeof(ScanInfo));
65 CArrayInit(&n->scannedAddrBuf, sizeof(ScanInfo));
66}
67void NetClientTerminate(NetClient *n)
68{
69 NetClientDisconnect(n);
70 n->peer = NULL;
71 enet_host_destroy(n->client);
72 n->client = NULL;
73 if (n->scanner != ENET_SOCKET_NULL)
74 {
75 if (enet_socket_shutdown(n->scanner, ENET_SOCKET_SHUTDOWN_READ_WRITE) != 0)
76 {
77 LOG(LM_NET, LL_ERROR, "Failed to shutdown listen socket");
78 }
79 enet_socket_destroy(n->scanner);
80 n->scanner = ENET_SOCKET_NULL;
81 }
82 CArrayTerminate(&n->ScannedAddrs);
83 CArrayTerminate(&n->scannedAddrBuf);
84}

86static bool TryScanHost(NetClient *n, const enet_uint32 host);
87void NetClientFindLANServers(NetClient *n)
88{
89 // If we've already begun scanning, wait for that to finish
90 if (n->ScanTicks > 0)
91 {
92 return;
93 }
94
95 // Replace the scanned addresses with the buffer
96 CArray temp = n->ScannedAddrs;
97 n->ScannedAddrs = n->scannedAddrBuf;
98 n->scannedAddrBuf = temp;
99 CArrayClear(&n->scannedAddrBuf);
100
101 // Scan for servers on LAN using broadcast host
102 if (!TryScanHost(n, ENET_HOST_BROADCAST))
103 {
104 return;
105 }
106
107 n->ScanTicks = FIND_CONNECTION_WAIT_SECONDS * FPS_FRAMELIMIT;
108}
109static bool TryScanHost(NetClient *n, const enet_uint32 host)
110{
111 // Create scanner socket if it has not been created yet
112 if (n->scanner == ENET_SOCKET_NULL)
113 {
114 n->scanner = enet_socket_create(ENET_SOCKET_TYPE_DATAGRAM);
115 if (n->scanner == ENET_SOCKET_NULL)
116 {
117 LOG(LM_NET, LL_ERROR, "Failed to create socket");
118 goto bail;
119 }
120 if (enet_socket_set_option(n->scanner, ENET_SOCKOPT_BROADCAST, 1) != 0)
121 {
122 LOG(LM_NET, LL_ERROR, "Failed to enable broadcast socket");
123 goto bail;
124 }
125 }
126
127 // Send the scanning message
128 ENetAddress addr;
129 addr.host = host;
130 addr.port = NET_LISTEN_PORT;
131 // Send a dummy payload
132 char data = 42;
133 ENetBuffer sendbuf;
134 sendbuf.data = &data;
135 sendbuf.dataLength = 1;
136 if (enet_socket_send(n->scanner, &addr, &sendbuf, 1) !=
137 (int)sendbuf.dataLength)
138 {
139 LOG(LM_NET, LL_ERROR, "Failed to scan for server");
140 goto bail;
141 }
142
143 return true;
144
145bail:
146 enet_socket_destroy(n->scanner);
147 n->scanner = ENET_SOCKET_NULL;
148 return false;
149}

151bool NetClientTryConnect(NetClient *n, const ENetAddress addr)
152{
153 NetClientDisconnect(n);
154
155 char buf[256];
156 enet_address_get_host_ip(&addr, buf, sizeof buf);
157 LOG(LM_NET, LL_INFO, "Connecting client to %s:%u...", buf, addr.port);
158
159 /* Initiate the connection, allocating the two channels 0 and 1. */
160 n->peer = enet_host_connect(n->client, &addr, 2, 0);
161 if (n->peer == NULL)
162 {
163 LOG(LM_NET, LL_WARN, "No server connection found");
164 goto bail;
165 }
166
167 /* Wait milliseconds for the connection attempt to succeed. */
168 ENetEvent event;
169 if (enet_host_service(n->client, &event, CONNECTION_WAIT_MS) > 0 &&
170 event.type == ENET_EVENT_TYPE_CONNECT)
171 {
172 fprintf(stderr, "Connected.\n");
173 }
174 else
175 {
176 /* Either the 5 seconds are up or a disconnect event was */
177 /* received. Reset the peer in the event the 5 seconds */
178 /* had run out without any significant event. */
179 LOG(LM_NET, LL_WARN, "connection failed");
180 goto bail;
181 }
182
183 // Set disconnect timeout ms
184 enet_peer_timeout(n->peer, 0, 0, TIMEOUT_MS);
185
186 // Tell the server that this is a proper connection request
187 NetClientSendMsg(n, GAME_EVENT_CLIENT_CONNECT, NULL);
188
189 return NetClientIsConnected(n);
190
191bail:
192 NetClientDisconnect(n);
193 return false;
194}

196static bool TryRecvScanForServerPort(
197 NetClient *n, const int timeoutMs,
198 NServerInfo *outInfo, ENetAddress *outAddr);
199bool NetClientTryScanAndConnect(NetClient *n, const enet_uint32 host)
200{
201 if (!TryScanHost(n, host))
202 {
203 return false;
204 }
205
206 // Wait for the scan to return
207 NServerInfo sinfo;
208 ENetAddress addr;
209 if (!TryRecvScanForServerPort(n, CONNECTION_WAIT_MS, &sinfo, &addr))
210 {
211 return false;
212 }
213
214 // Connect to address
215 return NetClientTryConnect(n, addr);
216}
217static bool TryRecvScanForServerPort(
218 NetClient *n, const int timeoutMs,
219 NServerInfo *outInfo, ENetAddress *outAddr)
220{
221 CASSERT(n->scanner != ENET_SOCKET_NULL,
222 "cannot recv scans without scanner");
223 // Check for the reply, which will give us the server address
224 ENetSocketSet set;
225 ENET_SOCKETSET_EMPTY(set);
226 ENET_SOCKETSET_ADD(set, n->scanner);
227 if (enet_socketset_select(n->scanner, &set, NULL, timeoutMs) <= 0)
228 {
229 return false;
230 }
231
232 // Receive the reply
233 uint8_t buf[1024];
234 ENetBuffer recvbuf;
235 recvbuf.data = buf;
236 recvbuf.dataLength = sizeof buf;
237 const int recvlen = enet_socket_receive(n->scanner, outAddr, &recvbuf, 1);
238 if (recvlen <= 0)
239 {
240 return false;
241 }
242 pb_istream_t stream = pb_istream_from_buffer(buf, recvlen);
243 if (!pb_decode(&stream, NServerInfo_fields, outInfo))
244 {
245 LOG(LM_NET, LL_ERROR, "Failed to decode scan reply");
246 return false;
247 }
248 // Check the protocol version, if it doesn't match we can't connect
249 if (outInfo->ProtocolVersion != NET_PROTOCOL_VERSION)
250 {
251 LOG(LM_NET, LL_DEBUG, "Protocol mismatch; %d (theirs) != %d",
252 outInfo->ProtocolVersion, NET_PROTOCOL_VERSION);
253 return false;
254 }
255 outAddr->port = (enet_uint16)outInfo->ENetPort;
256 char ipbuf[256];
257 enet_address_get_host_ip(outAddr, ipbuf, sizeof ipbuf);
258 LOG(LM_NET, LL_DEBUG, "Found server at %s:%u", ipbuf, outInfo->ENetPort);
259 return true;
260}

262void NetClientDisconnect(NetClient *n)
263{
264 if (n->peer)
265 {
266 LOG(LM_NET, LL_INFO, "disconnecting peer");
267 enet_peer_disconnect_now(n->peer, 0);
268 n->peer = NULL;
269 }
270 // Reset IDs so that when we start a server, we use our own IDs
271 n->ClientId = -1;
272 n->FirstPlayerUID = 0;
273 n->Ready = false;
274 // Also reset the scanned address buffer
275 CArrayClear(&n->ScannedAddrs);
276 CArrayClear(&n->scannedAddrBuf);
277}

279static void OnReceive(NetClient *n, ENetEvent event);
280static void Scanning(NetClient *n);
281void NetClientPoll(NetClient *n)
282{
283 // Check to see if LAN servers have been scanned
284 Scanning(n);
285
286 if (!n->client || !n->peer)
287 {
288 return;
289 }
290
291 // Service the connection
292 int check;
293 do
294 {
295 ENetEvent event;
296 check = enet_host_service(n->client, &event, 0);
297 if (check < 0)
298 {
299 LOG(LM_NET, LL_ERROR, "connection error(%d)", check);
300 NetClientTerminate(n);
301 return;
302 }
303 else if (check > 0)
304 {
305 switch (event.type)
306 {
307 case ENET_EVENT_TYPE_RECEIVE:
308 OnReceive(n, event);
309 break;
310 case ENET_EVENT_TYPE_DISCONNECT:
311 LOG(LM_NET, LL_INFO, "disconnected");
312 NetClientDisconnect(n);
313 return;
314 default:
315 LOG(LM_NET, LL_ERROR, "Unexpected event type(%d)", (int)event.type);
316 break;
317 }
318 }
319 } while (check > 0);
320}
321static void Scanning(NetClient *n)
322{
323 // If it's been too long, stop scanning
324 if (n->ScanTicks <= 0)
325 {
326 return;
327 }
328 n->ScanTicks--;
329
330 // Check to see if we have received the scan reply
331 ScanInfo sinfo;
332 while (TryRecvScanForServerPort(n, 0, &sinfo.ServerInfo, &sinfo.Addr))
333 {
334 // Check if we've seen this scan before
335 // This can happen occasionally, especially if the network is flaky
336 bool found = false;
337 CA_FOREACH(const ScanInfo, si, n->scannedAddrBuf)
338 if (si->Addr.host == sinfo.Addr.host &&
339 si->Addr.port == sinfo.Addr.port)
340 {
341 found = true;
342 break;
343 }
344 CA_FOREACH_END()
345 if (found)
346 {
347 continue;
348 }
349
350 // Save them into our buffer
351 // Calculate latency based on ticks
352 sinfo.LatencyMS =
353 (FIND_CONNECTION_WAIT_SECONDS * FPS_FRAMELIMIT - n->ScanTicks) *
354 1000 / FPS_FRAMELIMIT;
355 CArrayPushBack(&n->scannedAddrBuf, &sinfo);
356 // Also add them to the main collection, so we can find servers quickly
357 found = false;
358 CA_FOREACH(const ScanInfo, si, n->ScannedAddrs)
359 if (si->Addr.host == sinfo.Addr.host &&
360 si->Addr.port == sinfo.Addr.port)
361 {
362 found = true;
363 break;
364 }
365 CA_FOREACH_END()
366 if (!found)
367 {
368 CArrayPushBack(&n->ScannedAddrs, &sinfo);
369 }
370 }
371}
372static void OnReceive(NetClient *n, ENetEvent event)
373{
374 const GameEventType msg = (GameEventType)*(uint32_t *)event.packet->data;
375 LOG(LM_NET, LL_TRACE, "recv msg(%u)", msg);
376 const GameEventEntry gee = GameEventGetEntry(msg);
377 if (gee.Enqueue)
378 {
379 if (gee.GameStart && !gMission.HasStarted)
380 {
381 LOG(LM_NET, LL_TRACE, "ignore game start gameEvent(%d)",
382 (int)gee.Type);
383 }
384 else
385 {
386 // Game event message; decode and add to event queue
387 LOG(LM_NET, LL_TRACE, "recv gameEvent(%d)", (int)gee.Type);
388 GameEvent e = GameEventNew(gee.Type);
389 if (gee.Fields != NULL)
390 {
391 NetDecode(event.packet, &e.u, gee.Fields);
392 }
393
394 // For actor events, check if UID is not for local player
395 // TODO: repeated code (see game_events.c)
396 int actorUID = -1;
397 bool actorIsLocal = false;
398 switch (gee.Type)
399 {
400 case GAME_EVENT_ACTOR_ADD:
401 // Note: ignore checking this event
402 break;
403 case GAME_EVENT_ACTOR_MOVE: actorUID = e.u.ActorMove.UID; break;
404 case GAME_EVENT_ACTOR_STATE: actorUID = e.u.ActorState.UID; break;
405 case GAME_EVENT_ACTOR_DIR: actorUID = e.u.ActorDir.UID; break;
406 case GAME_EVENT_ACTOR_SLIDE: actorUID = e.u.ActorSlide.UID; break;
407 case GAME_EVENT_ACTOR_SWITCH_GUN: actorUID = e.u.ActorSwitchGun.UID; break;
408 case GAME_EVENT_ACTOR_PICKUP_ALL: actorUID = e.u.ActorPickupAll.UID; break;
409 case GAME_EVENT_ACTOR_USE_AMMO: actorUID = e.u.UseAmmo.UID; break;
410 case GAME_EVENT_ACTOR_MELEE: actorUID = e.u.Melee.UID; break;
411 case GAME_EVENT_GUN_FIRE:
412 if (e.u.GunFire.IsGun)
413 {
414 actorIsLocal = PlayerIsLocal(e.u.GunFire.PlayerUID);
415 }
416 break;
417 case GAME_EVENT_GUN_RELOAD:
418 actorIsLocal = PlayerIsLocal(e.u.GunReload.PlayerUID);
419 break;
420 case GAME_EVENT_GUN_STATE: actorUID = e.u.GunState.ActorUID; break;
421 default: break;
422 }
423 if (actorUID >= 0)
424 {
425 actorIsLocal = ActorIsLocalPlayer(actorUID);
426 }
427 if (actorIsLocal)
428 {
429 LOG(LM_NET, LL_TRACE, "game event is for local player, ignoring");
430 }
431 else
432 {
433 GameEventsEnqueue(&gGameEvents, e);
434 }
435 }
436 }
437 else
438 {
439 switch (gee.Type)
440 {
441 case GAME_EVENT_CLIENT_ID:
442 {
443 CASSERT(
444 n->ClientId == -1,
445 "unexpected client ID message, already set");
446 NClientId cid;
447 NetDecode(event.packet, &cid, NClientId_fields);
448 LOG(LM_NET, LL_DEBUG, "recv clientId(%u) uid(%u)",
449 cid.Id, cid.FirstPlayerUID);
450 n->ClientId = (int)cid.Id;
451 n->FirstPlayerUID = (int)cid.FirstPlayerUID;
452 }
453 break;
454 case GAME_EVENT_CAMPAIGN_DEF:
455 if (gCampaign.IsLoaded)
456 {
457 LOG(LM_NET, LL_INFO, "WARNING: unexpected campaign def msg received");
458 }
459 else
460 {
461 LOG(LM_NET, LL_DEBUG, "NetClient: received campaign def, loading...");
462 NCampaignDef def;
463 NetDecode(event.packet, &def, NCampaignDef_fields);
464 gCampaign.Entry.Mode = (GameMode)def.GameMode;
465 // Normalise the path
466 char buf[CDOGS_PATH_MAX];
467 GetDataFilePath(buf, def.Path);
468 CampaignEntry entry;
469 if (CampaignEntryTryLoad(&entry, buf, GAME_MODE_NORMAL) &&
470 CampaignLoad(&gCampaign, &entry))
471 {
472 gCampaign.IsClient = true;
473 gCampaign.MissionIndex = def.Mission;
474 }
475 else
476 {
477 LOG(LM_NET, LL_ERROR, "failed to load campaign def %s",
478 def.Path);
479 gCampaign.IsError = true;
480 }
481 }
482 break;
483 case GAME_EVENT_NET_GAME_START:
484 LOG(LM_NET, LL_DEBUG, "recv game start ready(%s)",
485 n->Ready ? "yes" : "no");
486 // Don't ready-up unless we're ready
487 if (n->Ready)
488 {
489 gMission.HasStarted = true;
490 }
491 break;
492 default:
493 CASSERT(false, "unexpected message type");
494 break;
495 }
496 }
497 enet_packet_destroy(event.packet);
498}

500void NetClientFlush(NetClient *n)
501{
502 if (n->client == NULL) return;
503 enet_host_flush(n->client);
504}
505
506void NetClientSendMsg(NetClient *n, const GameEventType e, const void *data)
507{
508 if (!NetClientIsConnected(n))
509 {
510 return;
511 }
512
513 LOG(LM_NET, LL_TRACE, "NetClient: send msg type %d", (int)e);
514 enet_peer_send(n->peer, 0, NetEncode(e, data));
515}
516
517bool NetClientIsConnected(const NetClient *n)
518{
519 return n->client && n->peer;
520}
