38// Display a character and the player name above it, with the character
39// centered around the target position
40void DisplayCharacterAndName(
41 Vec2i pos, Character *c, const direction_e d,
42 const char *name, const color_t color)
43{
44 // Move the point down a bit since the default character draw point is at
45 // its feet
46 pos.y += 8;
47 Vec2i namePos = Vec2iAdd(pos, Vec2iNew(-FontStrW(name) / 2, -30));
48 DrawCharacterSimple(c, pos, d, false, false);
49 FontStrMask(name, namePos, color);
50}

52void MenuDisplayPlayer(
53 const menu_t *menu, GraphicsDevice *g,
54 const Vec2i pos, const Vec2i size, const void *data)
55{
56 UNUSED(g);
57 const MenuDisplayPlayerData *d = data;
58 Vec2i playerPos;
59 char s[22];
60 UNUSED(menu);
61 Vec2i dPos = pos;
62 dPos.x -= size.x; // move to left half of screen
63 playerPos = Vec2iNew(
64 dPos.x + size.x * 3 / 4 - 12 / 2, CENTER_Y(dPos, size, 0));
65
66 PlayerData *pData = PlayerDataGetByUID(d->PlayerUID);
67 if (d->currentMenu && strcmp((*d->currentMenu)->name, "Name") == 0)
68 {
69 sprintf(s, "%c%s%c", '>', pData->name, '<');
70 }
71 else
72 {
73 strcpy(s, pData->name);
74 }
75
76 DisplayCharacterAndName(playerPos, &pData->Char, d->Dir, s, colorWhite);
77}

79void MenuDisplayPlayerControls(
80 const menu_t *menu, GraphicsDevice *g,
81 const Vec2i pos, const Vec2i size, const void *data)
82{
83 UNUSED(g);
84 char s[256];
85 const int *playerUID = data;
86 const int y = pos.y + size.y - FontH();
87
88 UNUSED(menu);
89
90 const PlayerData *pData = PlayerDataGetByUID(*playerUID);
91 char directionNames[256];
92 InputGetDirectionNames(
93 directionNames, pData->inputDevice, pData->deviceIndex);
94 switch (pData->inputDevice)
95 {
96 case INPUT_DEVICE_KEYBOARD:
97 {
98 char button1[256], button2[256];
99 InputGetButtonName(
100 pData->inputDevice, pData->deviceIndex, CMD_BUTTON1, button1);
101 InputGetButtonName(
102 pData->inputDevice, pData->deviceIndex, CMD_BUTTON2, button2);
103 sprintf(s, "(%s,\n%s and %s)", directionNames, button1, button2);
104 FontStr(s, Vec2iNew(pos.x - FontStrW(s) / 2, y - FontH()));
105 }
106 break;
107 case INPUT_DEVICE_MOUSE:
108 sprintf(s, "(%s to scroll,\nleft and right click)", directionNames);
109 FontStr(s, Vec2iNew(pos.x - FontStrW(s) / 2, y - FontH()));
110 break;
111 case INPUT_DEVICE_JOYSTICK:
112 {
113 sprintf(s, "(%s,",
114 InputDeviceName(pData->inputDevice, pData->deviceIndex));
115 Vec2i textPos = Vec2iNew(pos.x - FontStrW(s) / 2, y - FontH());
116 FontStr(s, textPos);
117 textPos.y += FontH();
118 color_t c = colorWhite;
119 InputGetButtonNameColor(
120 pData->inputDevice, pData->deviceIndex, CMD_BUTTON1, s, &c);
121 textPos = FontStrMask(s, textPos, c);
122 textPos = FontStr(" and ", textPos);
123 c = colorWhite;
124 InputGetButtonNameColor(
125 pData->inputDevice, pData->deviceIndex, CMD_BUTTON2, s, &c);
126 textPos = FontStrMask(s, textPos, c);
127 FontStr(")", textPos);
128 }
129 break;
130 case INPUT_DEVICE_AI:
131 sprintf(s, "(%s)",
132 InputDeviceName(pData->inputDevice, pData->deviceIndex));
133 FontStr(s, Vec2iNew(pos.x - FontStrW(s) / 2, y));
134 break;
135 case INPUT_DEVICE_UNSET:
136 strcpy(s, "(no device; plug in a controller)");
137 FontStr(s, Vec2iNew(pos.x - FontStrW(s) / 2, y));
138 break;
139 default:
140 CASSERT(false, "unknown device");
141 break;
142 }
