60typedef enum
61{
62 MENU_TYPE_NORMAL, // normal menu with items, up/down/left/right moves cursor
63 MENU_TYPE_OPTIONS, // menu with items, only up/down moves
64 MENU_TYPE_KEYS, // extra wide option menu
65 MENU_TYPE_BASIC, // no items, does nothing (use custom callbacks)
66 MENU_TYPE_SET_OPTION_TOGGLE, // no items, sets option on/off
67 MENU_TYPE_SET_OPTION_RANGE, // no items, sets option range low-high
68 MENU_TYPE_SET_OPTION_SEED, // set random seed
69 MENU_TYPE_SET_OPTION_UP_DOWN_VOID_FUNC_VOID, // set option using up/down functions
70 MENU_TYPE_SET_OPTION_RANGE_GET_SET, // set option range low-high using get/set functions
71 MENU_TYPE_SET_OPTION_CHANGE_KEY, // redefine key
72 MENU_TYPE_VOID_FUNC, // call a void (*f)(void *) function
73 MENU_TYPE_BACK,
74 MENU_TYPE_QUIT,
75 MENU_TYPE_RETURN, // Return with a code
76 MENU_TYPE_CUSTOM // use custom callbacks for input and drawing
77} menu_type_e;

81 MENU_OPTION_DISPLAY_STYLE_NONE,

92typedef struct menu menu_t;
93
94// Callback for drawing custom menu
95// menu, graphics, pos, size, data
96typedef void (*MenuDisplayFunc)(
97 const menu_t *, GraphicsDevice *, const Vec2i, const Vec2i, const void *);
98// Callback for handling user input
99// cmd, data
100// returns: 0 if stay on menu, 1 if exit from menu
101typedef int (*MenuInputFunc)(int, void *);
102// Callback for processing custom function on menu
103typedef void(*MenuFunc)(menu_t *, void *);
104// Callback for processing custom menu function post input
105// menu, cmd, data
106typedef void(*MenuPostInputFunc)(menu_t *, int cmd, void *);

123struct menu
124{
125 char *name;
126 menu_type_e type;
127 struct menu *parentMenu;
128 bool isDisabled;
129 color_t color;
130 MenuFunc customPostEnterFunc;
131 void *customPostEnterData;
132 bool isCustomPostEnterDataDynamic;
133 MenuFunc customPostUpdateFunc;
134 void *customPostUpdateData;
135 bool isCustomPostUpdateDataDynamic;
136 MenuPostInputFunc customPostInputFunc;
137 void *customPostInputData;
138 MenuDisplayFunc customDisplayFunc;
139 const void *customDisplayData;
140 MenuSound enterSound;
141 union
142 {
143 // normal menu, with sub menus
144 struct
145 {
146 char title[64];
147 CArray subMenus; // of menu_t
148 // whether to use alternate control to enter submenus
149 bool isSubmenusAlt;
150 int index;
151 int scroll;
152 int maxItems; // 0 means unlimited
153 MenuAlignStyle align;
154 int quitMenuIndex;
155 int displayItems;
156 int setOptions;
157 struct menu *changeKeyMenu; // if in change key mode, and which item
158 } normal;
159 // menu item only
160 struct
161 {
162 union
163 {
164 bool *optionToggle;
165 struct
166 {
167 int *option;
168 int low;
169 int high;
170 int increment;
171 } optionRange;
172 unsigned int *seed;
173 // function to call
174 struct
175 {
176 void (*upFunc)(void);
177 void (*downFunc)(void);
178 } upDownFuncs;
179 struct
180 {
181 int (*getFunc)(void);
182 void (*setFunc)(int);
183 int low;
184 int high;
185 int increment;
186 } optionRangeGetSet;
187 struct
188 {
189 void (*func)(void *);
190 void *data;
191 } voidFunc;
192 struct
193 {
194 input_device_e *device0;
195 input_device_e *device1;
196 } changeControl;
197 } uHook;
198 menu_option_display_style_e displayStyle;
199 union
200 {
201 char *(*str)(void);
202 const char *(*intToStr)(int);
203 } uFunc;
204 } option;
205 // change key
206 struct
207 {
208 key_code_e code;
209 int playerIndex;
210 } changeKey;
211 int returnCode;
212 struct
213 {
214 MenuDisplayFunc displayFunc;
215 MenuInputFunc inputFunc;
216 void *data;
217 } customData;
218 } u;
219};
