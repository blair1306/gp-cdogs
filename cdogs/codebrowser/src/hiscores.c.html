68// Dummy screen to calculate high scores and switch to high scores screens if
69// required
70typedef struct
71{
72 CampaignOptions *co;
73 GraphicsDevice *g;
74} HighScoresScreenData;
75static void HighScoresScreenTerminate(GameLoopData *data);
76static GameLoopResult HighScoresScreenUpdate(
77 GameLoopData *data, LoopRunner *l);
78GameLoopData *HighScoresScreen(CampaignOptions *co, GraphicsDevice *g)
79{
80 HighScoresScreenData *data;
81 CMALLOC(data, sizeof *data);
82 data->co = co;
83 data->g = g;
84 return GameLoopDataNew(
85 data, HighScoresScreenTerminate, NULL, NULL,
86 NULL, HighScoresScreenUpdate, NULL);
87}
88static void HighScoresScreenTerminate(GameLoopData *data)
89{
90 HighScoresScreenData *hData = data->Data;
91 CFREE(hData);
92}
93static GameLoopResult HighScoresScreenUpdate(
94 GameLoopData *data, LoopRunner *l)
95{
96 HighScoresScreenData *hData = data->Data;
97 LoopRunnerPop(l);
98 if (!IsPVP(hData->co->Entry.Mode) &&
99 GetNumPlayers(PLAYER_ANY, false, true) > 0)
100 {
101 LoadHighScores();
102 bool allTime = false;
103 bool todays = false;
104 CA_FOREACH(PlayerData, p, gPlayerDatas)
105 const bool isPlayerComplete =
106 (!gMission.IsQuit && !p->survived) || hData->co->IsComplete;
107 if (isPlayerComplete && p->IsLocal && IsPlayerHuman(p))
108 {
109 EnterHighScore(p);
110 allTime |= p->allTime >= 0;
111 todays |= p->today >= 0;
112 }
113
114 if (!p->survived)
115 {
116 // Reset scores because we died :(
117 memset(&p->Totals, 0, sizeof p->Totals);
118 p->missions = 0;
119 }
120 else
121 {
122 p->missions++;
123 }
124 p->lastMission = hData->co->MissionIndex;
125 CA_FOREACH_END()
126 SaveHighScores();
127
128 // Show high scores screen if high enough
129 if (todays)
130 {
131 LoopRunnerPush(l, DisplayTodaysHighScores(hData->g));
132 }
133 if (allTime)
134 {
135 LoopRunnerPush(l, DisplayAllTimeHighScores(hData->g));
136 }
137 }
138 if (!HasRounds(hData->co->Entry.Mode) && !hData->co->IsComplete)
139 {
140 hData->co->MissionIndex++;
141 }
142
143 return UPDATE_RESULT_OK;
144}

147// Warning: written as-is to file
148struct Entry {
149 char name[20];
150 int32_t unused1;
151 int32_t unused2;
152 int32_t unused3;
153 int32_t unused4;
154 int32_t unused5;
155 int32_t unused6;
156 int32_t score;
157 int32_t missions;
158 int32_t lastMission;
159};
160
161#define MAX_ENTRY 20
162
163static struct Entry allTimeHigh[MAX_ENTRY];
164static struct Entry todaysHigh[MAX_ENTRY];
165
166
167static int EnterTable(struct Entry *table, PlayerData *data)
168{
169 for (int i = 0; i < MAX_ENTRY; i++)
170 {
171 if (data->Totals.Score > table[i].score)
172 {
173 for (int j = MAX_ENTRY - 1; j > i; j--)
174 {
175 table[j] = table[j - 1];
176 }
177
178 strcpy(table[i].name, data->name);
179 table[i].score = data->Totals.Score;
180 table[i].missions = data->missions;
181 table[i].lastMission = data->lastMission;
182
183 return i;
184 }
185 }
186 return -1;
187}
188
189void EnterHighScore(PlayerData *data)
190{
191 data->allTime = EnterTable(allTimeHigh, data);
192 data->today = EnterTable(todaysHigh, data);
193}

195static void DisplayAt(int x, int y, const char *s, int hilite)
196{
197 color_t mask = hilite ? colorRed : colorWhite;
198 FontStrMask(s, Vec2iNew(x, y), mask);
199}
200
201static int DisplayEntry(
202 const int x, const int y, const int idx, const struct Entry *e,
203 const bool hilite)
204{
205 char s[10];
206
207#define INDEX_OFFSET 15
208#define SCORE_OFFSET 40
209#define MISSIONS_OFFSET 60
210#define MISSION_OFFSET 80
211#define NAME_OFFSET 85
212
213 sprintf(s, "%d.", idx + 1);
214 DisplayAt(x + INDEX_OFFSET - FontStrW(s), y, s, hilite);
215 sprintf(s, "%d", e->score);
216 DisplayAt(x + SCORE_OFFSET - FontStrW(s), y, s, hilite);
217 sprintf(s, "%d", e->missions);
218 DisplayAt(x + MISSIONS_OFFSET - FontStrW(s), y, s, hilite);
219 sprintf(s, "(%d)", e->lastMission + 1);
220 DisplayAt(x + MISSION_OFFSET - FontStrW(s), y, s, hilite);
221 DisplayAt(x + NAME_OFFSET, y, e->name, hilite);
222
223 return 1 + FontH();
224}
225
226static int DisplayPage(
227 const char *title, const int idxStart, const struct Entry *e,
228 const int highlights[MAX_LOCAL_PLAYERS])
229{
230 int x = 80;
231 int y = 5 + FontH();
232
233 FontStr(title, Vec2iNew(5, 5));
234 int idx = idxStart;
235 while (idx < MAX_ENTRY && e[idx].score > 0 && x < 300)
236 {
237 bool isHighlighted = false;
238 for (int i = 0; i < MAX_LOCAL_PLAYERS; i++)
239 {
240 if (idx == highlights[i])
241 {
242 isHighlighted = true;
243 break;
244 }
245 }
246 y += DisplayEntry(x, y, idx, &e[idx], isHighlighted);
247 if (y > 198 - FontH())
248 {
249 y = 20;
250 x += 100;
251 }
252 idx++;
253 }
254 return idx;
255}
256typedef struct
257{
258 GraphicsDevice *g;
259 const char *title;
260 const struct Entry *scores;
261 int highlights[MAX_LOCAL_PLAYERS];
262 int scoreIdx;
263} HighScoresData;
264static void HighScoreTerminate(GameLoopData *data);
265static GameLoopResult HighScoreUpdate(GameLoopData *data, LoopRunner *l);
266static void HighScoreDraw(GameLoopData *data);
267GameLoopData *DisplayAllTimeHighScores(GraphicsDevice *graphics)
268{
269 HighScoresData *data;
270 CMALLOC(data, sizeof *data);
271 data->g = graphics;
272 int idx = 0;
273 for (int i = 0; i < (int)gPlayerDatas.size; i++, idx++)
274 {
275 const PlayerData *p = CArrayGet(&gPlayerDatas, i);
276 if (!p->IsLocal)
277 {
278 idx--;
279 continue;
280 }
281 data->highlights[idx] = p->allTime;
282 }
283 data->scoreIdx = 0;
284 data->title = "All time high scores:";
285 data->scores = allTimeHigh;
286 return GameLoopDataNew(
287 data, HighScoreTerminate, NULL, NULL,
288 NULL, HighScoreUpdate, HighScoreDraw);
289}
290GameLoopData *DisplayTodaysHighScores(GraphicsDevice *graphics)
291{
292 HighScoresData *data;
293 CMALLOC(data, sizeof *data);
294 data->g = graphics;
295 int idx = 0;
296 for (int i = 0; i < (int)gPlayerDatas.size; i++, idx++)
297 {
298 const PlayerData *p = CArrayGet(&gPlayerDatas, i);
299 if (!p->IsLocal)
300 {
301 idx--;
302 continue;
303 }
304 data->highlights[idx] = p->today;
305 }
306 data->scoreIdx = 0;
307 data->title = "Today's highest score:";
308 data->scores = todaysHigh;
309 return GameLoopDataNew(
310 data, HighScoreTerminate, NULL, NULL,
311 NULL, HighScoreUpdate, HighScoreDraw);
312}
313static void HighScoreTerminate(GameLoopData *data)
314{
315 HighScoresData *hData = data->Data;
316
317 CFREE(hData);
318}
319static void HighScoreDraw(GameLoopData *data)
320{
321 HighScoresData *hData = data->Data;
322
323 BlitClearBuf(hData->g);
324 hData->scoreIdx = DisplayPage(
325 hData->title, hData->scoreIdx, hData->scores, hData->highlights);
326 BlitUpdateFromBuf(hData->g, hData->g->screen);
327}
328static GameLoopResult HighScoreUpdate(GameLoopData *data, LoopRunner *l)
329{
330 UNUSED(data);
331 const EventWaitResult result = EventWaitForAnyKeyOrButton();
332
333 if (result != EVENT_WAIT_CONTINUE)
334 {
335 LoopRunnerPop(l);
336 }
337 return UPDATE_RESULT_OK;
338}

341#define MAGIC 4711

343void SaveHighScores(void)
344{
345 int magic;
346 FILE *f;
347 time_t t;
348 struct tm *tp;
349
350 f = fopen(GetConfigFilePath(SCORES_FILE), "wb");
351 if (f != NULL) {
352 magic = MAGIC;
353
354 fwrite(&magic, sizeof(magic), 1, f);
355 fwrite(allTimeHigh, sizeof(allTimeHigh), 1, f);
356
357 t = time(NULL);
358 tp = localtime(&t);
359 debug(D_NORMAL, "time now, y: %d m: %d d: %d\n", tp->tm_year, tp->tm_mon, tp->tm_mday);
360
361 magic = tp->tm_year;
362 fwrite(&magic, sizeof(magic), 1, f);
363 magic = tp->tm_mon;
364 fwrite(&magic, sizeof(magic), 1, f);
365 magic = tp->tm_mday;
366 fwrite(&magic, sizeof(magic), 1, f);
367
368 debug(D_NORMAL, "writing today's high: %d\n", todaysHigh[0].score);
369 fwrite(todaysHigh, sizeof(todaysHigh), 1, f);
370
371 fclose(f);
372
373#ifdef __EMSCRIPTEN__
374 EM_ASM(
375 //persist changes
376 FS.syncfs(false,function (err) {
377 assert(!err);
378 });
379 );
380#endif
381
382 debug(D_NORMAL, "saved high scores\n");
383 } else
384 printf("Unable to open %s\n", SCORES_FILE);
385}
386
387void LoadHighScores(void)
388{
389 int magic;
390 FILE *f;
391 int y, m, d;
392 time_t t;
393 struct tm *tp;
394
395 debug(D_NORMAL, "Reading hi-scores...\n");
396
397 memset(allTimeHigh, 0, sizeof(allTimeHigh));
398 memset(todaysHigh, 0, sizeof(todaysHigh));
399
400 f = fopen(GetConfigFilePath(SCORES_FILE), "rb");
401 if (f != NULL) {
402 size_t elementsRead;
403 #define CHECK_FREAD(count)\
404 if (elementsRead != count) {\
405 debug(D_NORMAL, "Error reading scores file\n");\
406 fclose(f);\
407 return;\
408 }
409 elementsRead = fread(&magic, sizeof(magic), 1, f);
410 CHECK_FREAD(1)
411 if (magic != MAGIC) {
412 debug(D_NORMAL, "Scores file magic doesn't match!\n");
413 fclose(f);
414 return;
415 }
416
417 //for (i = 0; i < MAX_ENTRY; i++) {
418 elementsRead = fread(allTimeHigh, sizeof(allTimeHigh), 1, f);
419 CHECK_FREAD(1)
420 //}
421
422 t = time(NULL);
423 tp = localtime(&t);
424 elementsRead = fread(&y, sizeof(y), 1, f);
425 CHECK_FREAD(1)
426 elementsRead = fread(&m, sizeof(m), 1, f);
427 CHECK_FREAD(1)
428 elementsRead = fread(&d, sizeof(d), 1, f);
429 CHECK_FREAD(1)
430 debug(D_NORMAL, "scores time, y: %d m: %d d: %d\n", y, m, d);
431
432
433 if (tp->tm_year == y && tp->tm_mon == m && tp->tm_mday == d) {
434 elementsRead = fread(todaysHigh, sizeof(todaysHigh), 1, f);
435 CHECK_FREAD(1)
436 debug(D_NORMAL, "reading today's high: %d\n", todaysHigh[0].score);
437 }
438
439 fclose(f);
440 } else {
441 printf("Unable to open %s\n", SCORES_FILE);
442 }
443}
