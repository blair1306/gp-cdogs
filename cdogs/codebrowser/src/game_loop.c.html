51void emscriptenLoadFiles() {
52 // options.cnf
53 FILE *file_options = fopen(GetConfigFilePath(CONFIG_FILE),"r");
54 if (file_options) {
55 fclose(file_options);
56 ConfigLoadJSON(&gConfig, GetConfigFilePath(CONFIG_FILE));
57 ConfigApply(&gConfig);
58 }
59 else {
60 ConfigSave(&gConfig, GetConfigFilePath(CONFIG_FILE));
61 }
62
63 // autosave.json
64 FILE *file_autosave = fopen(GetConfigFilePath(AUTOSAVE_FILE),"r");
65 if (file_autosave) {
66 fclose(file_autosave);
67 AutosaveLoad(&gAutosave, GetConfigFilePath(AUTOSAVE_FILE));
68 }
69 else {
70 AutosaveSave(&gAutosave, GetConfigFilePath(AUTOSAVE_FILE));
71 }
72
73 // scores.dat
74 FILE *file_scores = fopen(GetConfigFilePath(SCORES_FILE),"r");
75 if (file_scores) {
76 fclose(file_scores);
77 }
78 else {
79 SaveHighScores();
80 }
81
82 // players.cnf
83 FILE *file_players = fopen(GetConfigFilePath(PLAYER_TEMPLATE_FILE),"r");
84 if (file_players) {
85 fclose(file_players);
86 PlayerTemplatesLoad(&gPlayerTemplates, &gCharacterClasses, PLAYER_TEMPLATE_FILE);
87 }
88 else {
89 SavePlayerTemplates(&gPlayerTemplates, PLAYER_TEMPLATE_FILE);
90 }
91}

93bool emscriptenPersistData() {
94 if (emscripten_fs_ready)
95 return true;
96
97 if(emscripten_run_script_int("Module.syncdone") == 1) {
98 FILE *config_file = fopen(GetConfigFilePath(CONFIG_FILE),"r");
99 if (config_file == NULL) {
100 //persist Emscripten current data to Indexed Db
101 EM_ASM(
102 Module.print("Start File sync..");
103 Module.syncdone = 0;
104 FS.syncfs(false, function(err) {
105 assert(!err);
106 Module.print("End File sync..");
107 Module.syncdone = 1;
108 });
109 );
110 emscriptenLoadFiles();
111 return false;
112 }
113 else {
114 fclose(config_file);
115 emscripten_fs_ready = true;
116 emscriptenLoadFiles();
117 return true;
118 }
119 }
120 return false;
121}

124GameLoopData *GameLoopDataNew(
125 void *data,
126 void (*onTerminate)(GameLoopData *),
127 void (*onEnter)(GameLoopData *), void (*onExit)(GameLoopData *),
128 void (*inputFunc)(GameLoopData *),
129 GameLoopResult (*updateFunc)(GameLoopData *, LoopRunner *),
130 void (*drawFunc)(GameLoopData *))
131{
132 GameLoopData *g;
133 CCALLOC(g, sizeof *g);
134 g->Data = data;
135 g->OnTerminate = onTerminate;
136 g->OnEnter = onEnter;
137 g->OnExit = onExit;
138 g->InputFunc = inputFunc;
139 g->UpdateFunc = updateFunc;
140 g->DrawFunc = drawFunc;
141 g->FPS = 30;
142 return g;
143}

147LoopRunner LoopRunnerNew(GameLoopData *newData)
148{
149 LoopRunner l;
150 CArrayInit(&l.Loops, sizeof(GameLoopData *));
151 if (newData != NULL)
152 {
153 LoopRunnerPush(&l, newData);
154 }
155 return l;
156}
157static void GameLoopTerminate(GameLoopData *data);
158void LoopRunnerTerminate(LoopRunner *l)
159{
160 for (int i = (int)l->Loops.size - 1; i >= 0; i--)
161 {
162 GameLoopData *g = CArrayGet(&l->Loops, i);
163 GameLoopTerminate(g);
164 }
165 CArrayTerminate(&l->Loops);
166}

171typedef struct
172{
173 GameLoopResult Result;
174 Uint32 TicksNow;
175 Uint32 TicksElapsed;
176 int FrameDurationMs;
177 int FramesSkipped;
178 int MaxFrameskip;
179} LoopRunParams;
180typedef struct
181{
182 LoopRunner *l;
183 GameLoopData *data;
184 LoopRunParams p;
185}LoopRunInnerData;
186static LoopRunParams LoopRunParamsNew(const GameLoopData *data);
187static bool LoopRunParamsShouldSleep(LoopRunParams *p);
188static bool LoopRunParamsShouldSkip(LoopRunParams *p);
189bool LoopRunnerRunInner(LoopRunInnerData *ctx)
190{
191#ifndef __EMSCRIPTEN__
192 // Frame rate control
193 if (LoopRunParamsShouldSleep(&(ctx->p)))
194 {
195 SDL_Delay(1);
196 return true;
197 }
198#endif
199
200 // Input
201 if ((ctx->data->Frames & 1) || !ctx->data->InputEverySecondFrame)
202 {
203 EventPoll(&gEventHandlers, ctx->p.TicksNow);
204 if (ctx->data->InputFunc)
205 {
206 ctx->data->InputFunc(ctx->data);
207 }
208 }
209
210 NetClientPoll(&gNetClient);
211 NetServerPoll(&gNetServer);
212
213 // Update
214 ctx->p.Result = ctx->data->UpdateFunc(ctx->data, ctx->l);
215 GameLoopData *newData = GetCurrentLoop(ctx->l);
216 if (newData == NULL)
217 {
218 return false;
219 }
220 else if (newData != ctx->data)
221 {
222 // State change; restart loop
223 GameLoopOnExit(ctx->data);
224 ctx->data = newData;
225 GameLoopOnEnter(ctx->data);
226 ctx->p = LoopRunParamsNew(ctx->data);
227 return true;
228 }
229
230 NetServerFlush(&gNetServer);
231 NetClientFlush(&gNetClient);
232
233 bool draw = !ctx->data->HasDrawnFirst;
234 switch (ctx->p.Result)
235 {
236 case UPDATE_RESULT_OK:
237 // Do nothing
238 break;
239 case UPDATE_RESULT_DRAW:
240 draw = true;
241 break;
242 default:
243 CASSERT(false, "Unknown loop result");
244 break;
245 }
246 ctx->data->Frames++;
247#ifndef __EMSCRIPTEN__
248 // frame skip
249 if (LoopRunParamsShouldSkip(&(ctx->p)))
250 {
251 return true;
252 }
253#endif
254
255 // Draw
256 if (draw)
257 {
258 if (ctx->data->DrawFunc)
259 {
260 ctx->data->DrawFunc(ctx->data);
261 BlitFlip(&gGraphicsDevice);
262 }
263 ctx->data->HasDrawnFirst = true;
264 }
265
266 return true;
267}

270void EmscriptenMainLoop(void *arg)
271{
272 if (!emscriptenPersistData())
273 return;
274
275 LoopRunnerRunInner((LoopRunInnerData *)arg);
276}

278void LoopRunnerRun(LoopRunner *l)
279{
280 GameLoopData *data = GetCurrentLoop(l);
281 if (data == NULL)
282 {
283 return;
284 }
285 GameLoopOnEnter(data);
286
287 LoopRunInnerData ctx;
288 ctx.l = l;
289 ctx.data = data;
290 ctx.p = LoopRunParamsNew(data);
291
292#ifdef __EMSCRIPTEN__
293 // TODO use GameLoopData->FPS instead of 30?
294 emscripten_set_main_loop_arg(EmscriptenMainLoop, &ctx, FPS_FRAMELIMIT, 1);
295#else
296 for (;;)
297 {
298 if (!LoopRunnerRunInner(&ctx))
299 {
300 break;
301 }
302 }
303#endif
304 GameLoopOnExit(data);
305}
306static LoopRunParams LoopRunParamsNew(const GameLoopData *data)
307{
308 LoopRunParams p;
309 p.Result = UPDATE_RESULT_OK;
310 p.TicksNow = SDL_GetTicks();
311 p.TicksElapsed = 0;
312 p.FrameDurationMs = 1000 / data->FPS;
313 p.FramesSkipped = 0;
314 p.MaxFrameskip = data->FPS / 5;
315 return p;
316}
317static bool LoopRunParamsShouldSleep(LoopRunParams *p)
318{
319 const Uint32 ticksThen = p->TicksNow;
320 p->TicksNow = SDL_GetTicks();
321 p->TicksElapsed += p->TicksNow - ticksThen;
322 return (int)p->TicksElapsed < p->FrameDurationMs;
323}
324static bool LoopRunParamsShouldSkip(LoopRunParams *p)
325{
326 p->TicksElapsed -= p->FrameDurationMs;
327 // frame skip
328 if ((int)p->TicksElapsed > p->FrameDurationMs)
329 {
330 p->FramesSkipped++;
331 if (p->FramesSkipped == p->MaxFrameskip)
332 {
333 // We've skipped too many frames; give up
334 p->TicksElapsed = 0;
335 }
336 else
337 {
338 return true;
339 }
340 }
341 p->FramesSkipped = 0;
342 return false;
343}

345void LoopRunnerChange(LoopRunner *l, GameLoopData *newData)
346{
347 LoopRunnerPop(l);
348 LoopRunnerPush(l, newData);
349}
350void LoopRunnerPush(LoopRunner *l, GameLoopData *newData)
351{
352 CArrayPushBack(&l->Loops, &newData);
353 newData->IsUsed = true;
354}
355void LoopRunnerPop(LoopRunner *l)
356{
357 GameLoopData *data = GetCurrentLoop(l);
358 data->IsUsed = false;
359 CArrayDelete(&l->Loops, l->Loops.size - 1);
360}
361
362static void GameLoopTerminate(GameLoopData *data)
363{
364 if (data->OnTerminate)
365 {
366 data->OnTerminate(data);
367 }
368 CFREE(data);
369}
370
371static void GameLoopOnEnter(GameLoopData *data)
372{
373 if (data->OnEnter)
374 {
375 data->OnEnter(data);
376 }
377 EventReset(
378 &gEventHandlers,
379 gEventHandlers.mouse.cursor, gEventHandlers.mouse.trail);
380}
381static void GameLoopOnExit(GameLoopData *data)
382{
383 if (data->OnExit)
384 {
385 data->OnExit(data);
386 }
387 if (!data->IsUsed)
388 {
389 GameLoopTerminate(data);
390 }
391}

393static GameLoopData *GetCurrentLoop(LoopRunner *l)
394{
395 if (l->Loops.size == 0)
396 {
397 return NULL;
398 }
399 return *(GameLoopData **)CArrayGet(&l->Loops, l->Loops.size - 1);
400}
