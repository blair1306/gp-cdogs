104static void CampaignIntroDraw(GameLoopData *data)
105{
106 const ScreenCampaignIntroData *sData = data->Data;
107
108 BlitClearBuf(&gGraphicsDevice);
109 const int w = gGraphicsDevice.cachedConfig.Res.x;
110 const int h = gGraphicsDevice.cachedConfig.Res.y;
111 const int y = h / 4;
112
113 // Display title + author
114 char *buf;
115 CMALLOC(buf, strlen(sData->c->Title) + strlen(sData->c->Author) + 16);
116 sprintf(buf, "%s by %s", sData->c->Title, sData->c->Author);
117 FontOpts opts = FontOptsNew();
118 opts.HAlign = ALIGN_CENTER;
119 opts.Area = gGraphicsDevice.cachedConfig.Res;
120 opts.Pad.y = y - 25;
121 FontStrOpt(buf, Vec2iZero(), opts);
122 CFREE(buf);
123
124 // Display campaign description
125 // allow some slack for newlines
126 if (strlen(sData->c->Description) > 0)
127 {
128 CMALLOC(buf, strlen(sData->c->Description) * 2);
129 // Pad about 1/6th of the screen width total (1/12th left and right)
130 FontSplitLines(sData->c->Description, buf, w * 5 / 6);
131 FontStr(buf, Vec2iNew(w / 12, y));
132 CFREE(buf);
133 }
134
135 BlitUpdateFromBuf(&gGraphicsDevice, gGraphicsDevice.screen);
136}

140{
141 char *Title;
142 FontOpts TitleOpts;
143 char Password[32];
144 FontOpts PasswordOpts;
145 int TypewriterCount;
146 char *Description;
147 char *TypewriterBuf;

160GameLoopData *ScreenMissionBriefing(const struct MissionOptions *m)
161{
162 const int w = gGraphicsDevice.cachedConfig.Res.x;
163 const int h = gGraphicsDevice.cachedConfig.Res.y;
164 const int y = h / 4;
165 MissionBriefingData *mData;
166 CCALLOC(mData, sizeof *mData);
167 mData->waitResult = EVENT_WAIT_CONTINUE;
168
169 // Title
170 if (m->missionData->Title)
171 {
172 CMALLOC(mData->Title, strlen(m->missionData->Title) + 32);
173 sprintf(mData->Title, "Mission %d: %s",
174 m->index + 1, m->missionData->Title);
175 mData->TitleOpts = FontOptsNew();
176 mData->TitleOpts.HAlign = ALIGN_CENTER;
177 mData->TitleOpts.Area = gGraphicsDevice.cachedConfig.Res;
178 mData->TitleOpts.Pad.y = y - 25;
179 }
180
181 // Password
182 if (m->index > 0)
183 {
184 sprintf(
185 mData->Password, "Password: %s", gAutosave.LastMission.Password);
186 mData->PasswordOpts = FontOptsNew();
187 mData->PasswordOpts.HAlign = ALIGN_CENTER;
188 mData->PasswordOpts.Area = gGraphicsDevice.cachedConfig.Res;
189 mData->PasswordOpts.Pad.y = y - 15;
190 }
191
192 // Description
193 if (m->missionData->Description)
194 {
195 // Split the description, and prepare it for typewriter effect
196 // allow some slack for newlines
197 CMALLOC(
198 mData->Description, strlen(m->missionData->Description) * 2 + 1);
199 CCALLOC(
200 mData->TypewriterBuf, strlen(m->missionData->Description) * 2 + 1);
201 // Pad about 1/6th of the screen width total (1/12th left and right)
202 FontSplitLines(
203 m->missionData->Description, mData->Description, w * 5 / 6);
204 mData->DescriptionPos = Vec2iNew(w / 12, y);
205
206 // Objectives
207 mData->ObjectiveDescPos =
208 Vec2iNew(w / 6, y + FontStrH(mData->Description) + h / 10);
209 mData->ObjectiveInfoPos =
210 Vec2iNew(w - (w / 6), mData->ObjectiveDescPos.y + FontH());
211 mData->ObjectiveHeight = h / 12;
212 }
213 mData->MissionOptions = m;
214
215 return GameLoopDataNew(
216 mData, MissionBriefingTerminate, NULL, MissionBriefingOnExit,
217 MissionBriefingInput, MissionBriefingUpdate, MissionBriefingDraw);
218}

241static void MissionBriefingInput(GameLoopData *data)
242{
243 MissionBriefingData *mData = data->Data;
244
245 int cmds[MAX_LOCAL_PLAYERS];
246 memset(cmds, 0, sizeof cmds);
247 GetPlayerCmds(&gEventHandlers, &cmds);
248 if (mData->Description)
249 {
250 // Check for player input; if any then skip to the end of the briefing
251 for (int i = 0; i < MAX_LOCAL_PLAYERS; i++)
252 {
253 if (AnyButton(cmds[i]))
254 {
255 // If the typewriter is still going, skip to end
256 if (mData->TypewriterCount <= (int)strlen(mData->Description))
257 {
258 strcpy(mData->TypewriterBuf, mData->Description);
259 mData->TypewriterCount = strlen(mData->Description);
260 return;
261 }
262 // Otherwise, exit out of loop
263 mData->waitResult = EVENT_WAIT_OK;
264 }
265 }
266 }
267 // Check if anyone pressed escape
268 if (EventIsEscape(&gEventHandlers, cmds, GetMenuCmd(&gEventHandlers)))
269 {
270 mData->waitResult = EVENT_WAIT_CANCEL;
271 }
272}
273static GameLoopResult MissionBriefingUpdate(GameLoopData *data, LoopRunner *l)
274{
275 MissionBriefingData *mData = data->Data;
276
277 if (!IsMissionBriefingNeeded(gCampaign.Entry.Mode))
278 {
279 mData->waitResult = EVENT_WAIT_OK;
280 goto bail;
281 }
282
283 // Check exit conditions from input
284 if (mData->waitResult != EVENT_WAIT_CONTINUE)
285 {
286 goto bail;
287 }
288
289 // Update the typewriter effect
290 if (mData->TypewriterCount <= (int)strlen(mData->Description))
291 {
292 mData->TypewriterBuf[mData->TypewriterCount] =
293 mData->Description[mData->TypewriterCount];
294 mData->TypewriterCount++;
295 return UPDATE_RESULT_DRAW;
296 }
297
298 return UPDATE_RESULT_OK;
299
300bail:
301 if (mData->waitResult == EVENT_WAIT_OK)
302 {
303 LoopRunnerChange(l, PlayerEquip());
304 }
305 else
306 {
307 LoopRunnerPop(l);
308 }
309 return UPDATE_RESULT_OK;
310}
311static void MissionBriefingDraw(GameLoopData *data)
312{
313 const MissionBriefingData *mData = data->Data;
314
315 BlitClearBuf(&gGraphicsDevice);
316
317 // Mission title
318 FontStrOpt(mData->Title, Vec2iZero(), mData->TitleOpts);
319 // Display password
320 FontStrOpt(mData->Password, Vec2iZero(), mData->PasswordOpts);
321 // Display description with typewriter effect
322 FontStr(mData->TypewriterBuf, mData->DescriptionPos);
323 // Display objectives
324 CA_FOREACH(
325 const Objective, o, mData->MissionOptions->missionData->Objectives)
326 // Do not brief optional objectives
327 if (o->Required == 0)
328 {
329 continue;
330 }
331 Vec2i offset = Vec2iNew(0, _ca_index * mData->ObjectiveHeight);
332 FontStr(o->Description, Vec2iAdd(mData->ObjectiveDescPos, offset));
333 // Draw the icons slightly offset so that tall icons don't overlap each
334 // other
335 offset.x = -16 * (_ca_index & 1);
336 DrawObjectiveInfo(o, Vec2iAdd(mData->ObjectiveInfoPos, offset));
337 CA_FOREACH_END()
338
339 BlitUpdateFromBuf(&gGraphicsDevice, gGraphicsDevice.screen);
340}

358GameLoopData *ScreenMissionSummary(
359 const CampaignOptions *c, struct MissionOptions *m, const bool completed)
360{
361 MissionSummaryData *mData;
362 CMALLOC(mData, sizeof *mData);
363
364 const int h = FontH() * 10;
365 MenuSystemInit(
366 &mData->ms, &gEventHandlers, &gGraphicsDevice,
367 Vec2iNew(0, gGraphicsDevice.cachedConfig.Res.y - h),
368 Vec2iNew(gGraphicsDevice.cachedConfig.Res.x, h));
369 mData->ms.current = mData->ms.root =
370 MenuCreateNormal("", "", MENU_TYPE_NORMAL, 0);
371 // Use return code 0 for whether to continue the game
372 if (completed)
373 {
374 MenuAddSubmenu(mData->ms.root, MenuCreateReturn("Continue", 0));
375 }
376 else
377 {
378 MenuAddSubmenu(mData->ms.root, MenuCreateReturn("Replay mission", 0));
379 MenuAddSubmenu(mData->ms.root, MenuCreateReturn("Back to menu", 1));
380 }
381 mData->ms.allowAborts = true;
382 MenuAddExitType(&mData->ms, MENU_TYPE_RETURN);
383 MenuSystemAddCustomDisplay(&mData->ms, MissionSummaryMenuDraw, m);
384
385 mData->c = c;
386 mData->m = m;
387 mData->completed = completed;
388
389 return GameLoopDataNew(
390 mData, MissionSummaryTerminate, MissionSummaryOnEnter, NULL,
391 NULL, MissionSummaryUpdate, MissionSummaryDraw);
392}

404static void MissionSummaryOnEnter(GameLoopData *data)
405{
406 MissionSummaryData *mData = data->Data;
407
408 if (mData->completed)
409 {
410 // Save password
411 MissionSave ms;
412 MissionSaveInit(&ms);
413 ms.Campaign = mData->c->Entry;
414 // Don't make password for next level if there is none
415 int passwordIndex = mData->m->index + 1;
416 if (passwordIndex == mData->c->Entry.NumMissions)
417 {
418 passwordIndex--;
419 }
420 strcpy(ms.Password, MakePassword(passwordIndex, 0));
421 ms.MissionsCompleted = mData->m->index + 1;
422 AutosaveAddMission(&gAutosave, &ms);
423 AutosaveSave(&gAutosave, GetConfigFilePath(AUTOSAVE_FILE));
424 }
425
426 // Calculate bonus scores
427 // Bonuses only apply if at least one player has lived
428 if (AreAnySurvived())
429 {
430 int bonus = 0;
431 // Objective bonuses
432 CA_FOREACH(const Objective, o, mData->m->missionData->Objectives)
433 if (ObjectiveIsPerfect(o))
434 {
435 bonus += PERFECT_BONUS;
436 }
437 CA_FOREACH_END()
438 bonus += GetAccessBonus(mData->m);
439 bonus += GetTimeBonus(mData->m, NULL);
440
441 CA_FOREACH(PlayerData, p, gPlayerDatas)
442 ApplyBonuses(p, bonus);
443 CA_FOREACH_END()
444 }
445}
446static GameLoopResult MissionSummaryUpdate(GameLoopData *data, LoopRunner *l)
447{
448 MissionSummaryData *mData = data->Data;
449
450 const GameLoopResult result = MenuUpdate(&mData->ms);
451 if (result == UPDATE_RESULT_OK)
452 {
453 gCampaign.IsComplete = mData->completed &&
454 gCampaign.MissionIndex == (int)gCampaign.Setting.Missions.size - 1;
455 if (gCampaign.IsComplete)
456 {
457 LoopRunnerChange(l, ScreenVictory(&gCampaign));
458 }
459 else if (!mData->completed)
460 {
461 // Check if we want to return to menu or replay mission
462 mData->m->IsQuit = mData->ms.current->u.returnCode == 1;
463 LoopRunnerPop(l);
464 }
465 else
466 {
467 LoopRunnerChange(
468 l, HighScoresScreen(&gCampaign, &gGraphicsDevice));
469 }
470 }
471 return result;
472}

479static bool AreAnySurvived(void)
480{
481 CA_FOREACH(const PlayerData, p, gPlayerDatas)
482 if (p->survived)
483 {
484 return true;
485 }
486 CA_FOREACH_END()
487 return false;

489static int GetAccessBonus(const struct MissionOptions *m)
490{
491 return KeycardCount(m->KeyFlags) * 50;
492}
493static int GetTimeBonus(const struct MissionOptions *m, int *secondsOut)
494{
495 int seconds =
496 60 +
497 (int)m->missionData->Objectives.size * 30 -
498 m->time / FPS_FRAMELIMIT;
499 if (seconds < 0)
500 {
501 seconds = 0;
502 }
503 if (secondsOut)
504 {
505 *secondsOut = seconds;
506 }
507 return seconds * 25;
508}
509static int GetHealthBonus(const PlayerData *p);
510static int GetResurrectionFee(const PlayerData *p);
511static int GetButcherPenalty(const PlayerData *p);
512static int GetNinjaBonus(const PlayerData *p);
513static int GetFriendlyBonus(const PlayerData *p);
514static void ApplyBonuses(PlayerData *p, const int bonus)
515{
516 // Apply bonuses to surviving players only
517 if (!p->survived)
518 {
519 return;
520 }
521
522 p->Totals.Score += bonus;
523
524 // Other per-player bonuses
525 p->Totals.Score += GetHealthBonus(p);
526 p->Totals.Score += GetResurrectionFee(p);
527 p->Totals.Score += GetButcherPenalty(p);
528 p->Totals.Score += GetNinjaBonus(p);
529 p->Totals.Score += GetFriendlyBonus(p);
530}
531static int GetHealthBonus(const PlayerData *p)
532{
533 const int maxHealth = ModeMaxHealth(gCampaign.Entry.Mode);
534 return p->hp > maxHealth - 50 ? (p->hp + 50 - maxHealth) * 10 : 0;
535}
536static int GetResurrectionFee(const PlayerData *p)
537{
538 return p->hp <= 0 ? -500 : 0;
539}
540static int GetButcherPenalty(const PlayerData *p)
541{
542 if (p->Stats.Friendlies > 5 && p->Stats.Friendlies > p->Stats.Kills / 2)
543 {
544 return -100 * p->Stats.Friendlies;
545 }
546 return 0;
547}
548// TODO: amend ninja bonus to check if no shots fired
549static int GetNinjaBonus(const PlayerData *p)
550{
551 if (p->weaponCount == 1 && !p->weapons[0]->CanShoot &&
552 p->Stats.Friendlies == 0 && p->Stats.Kills > 5)
553 {
554 return 50 * p->Stats.Kills;
555 }
556 return 0;
557}
558static int GetFriendlyBonus(const PlayerData *p)
559{
560 return (p->Stats.Kills == 0 && p->Stats.Friendlies == 0) ? 500 : 0;
561}

564static void MissionSummaryMenuDraw(
565 const menu_t *menu, GraphicsDevice *g,
566 const Vec2i p, const Vec2i size, const void *data)
567{
568 UNUSED(menu);
569 UNUSED(p);
570 UNUSED(size);
571 const struct MissionOptions *m = data;
572
573 const int w = gGraphicsDevice.cachedConfig.Res.x;
574 const int h = gGraphicsDevice.cachedConfig.Res.y;
575
576 // Display password
577 if (strlen(gAutosave.LastMission.Password) > 0)
578 {
579 char s[64];
580 sprintf(s, "Last password: %s", gAutosave.LastMission.Password);
581 FontOpts opts = FontOptsNew();
582 opts.HAlign = ALIGN_CENTER;
583 opts.VAlign = ALIGN_END;
584 opts.Area = g->cachedConfig.Res;
585 opts.Pad.y = opts.Area.y / 12;
586 FontStrOpt(s, Vec2iZero(), opts);
587 }
588
589 // Display objectives and bonuses
590 Vec2i pos = Vec2iNew(w / 6, h / 2 + h / 10);
591 int idx = 1;
592 CA_FOREACH(const Objective, o, m->missionData->Objectives)
593 // Do not mention optional objectives with none completed
594 if (o->done == 0 && !ObjectiveIsRequired(o))
595 {
596 continue;
597 }
598
599 // Objective icon
600 DrawObjectiveInfo(o, Vec2iAdd(pos, Vec2iNew(-26, FontH())));
601
602 // Objective completion text
603 char s[100];
604 sprintf(s, "Objective %d: %d of %d, %d required",
605 idx, o->done, o->Count, o->Required);
606 FontOpts opts = FontOptsNew();
607 opts.Area = gGraphicsDevice.cachedConfig.Res;
608 opts.Pad = pos;
609 if (!ObjectiveIsRequired(o))
610 {
611 // Show optional objectives in purple
612 opts.Mask = colorPurple;
613 }
614 FontStrOpt(s, Vec2iZero(), opts);
615
616 // Objective status text
617 opts = FontOptsNew();
618 opts.HAlign = ALIGN_END;
619 opts.Area = gGraphicsDevice.cachedConfig.Res;
620 opts.Pad = pos;
621 if (!ObjectiveIsComplete(o))
622 {
623 opts.Mask = colorRed;
624 FontStrOpt("Failed", Vec2iZero(), opts);
625 }
626 else if (ObjectiveIsPerfect(o) && AreAnySurvived())
627 {
628 opts.Mask = colorGreen;
629 char buf[16];
630 sprintf(buf, "Perfect: %d", PERFECT_BONUS);
631 FontStrOpt(buf, Vec2iZero(), opts);
632 }
633 else if (ObjectiveIsRequired(o))
634 {
635 FontStrOpt("Done", Vec2iZero(), opts);
636 }
637 else
638 {
639 FontStrOpt("Bonus!", Vec2iZero(), opts);
640 }
641
642 pos.y += 15;
643 idx++;
644 CA_FOREACH_END()
645
646 // Draw other bonuses
647 if (AreAnySurvived())
648 {
649 char s[64];
650
651 sprintf(s, "Access bonus: %d", GetAccessBonus(m));
652 FontStr(s, pos);
653
654 pos.y += FontH() + 1;
655 int seconds;
656 const int timeBonus = GetTimeBonus(m, &seconds);
657 sprintf(s, "Time bonus: %d secs x 25 = %d", seconds, timeBonus);
658 FontStr(s, pos);
659 }
660
661 // Draw per-player summaries
662 PlayerData *pds[MAX_LOCAL_PLAYERS];
663 idx = 0;
664 CA_FOREACH(PlayerData, pd, gPlayerDatas)
665 if (!pd->IsLocal)
666 {
667 continue;
668 }
669 pds[idx] = pd;
670 idx++;
671 CA_FOREACH_END()
672 Vec2i playerSize;
673 switch (idx)
674 {
675 case 1:
676 playerSize = Vec2iNew(w, h / 2);
677 DrawPlayerSummary(Vec2iZero(), playerSize, pds[0]);
678 break;
679 case 2:
680 // side by side
681 playerSize = Vec2iNew(w / 2, h / 2);
682 DrawPlayerSummary(Vec2iZero(), playerSize, pds[0]);
683 DrawPlayerSummary(Vec2iNew(w / 2, 0), playerSize, pds[1]);
684 break;
685 case 3: // fallthrough
686 case 4:
687 // 2x2
688 playerSize = Vec2iNew(w / 2, h / 4);
689 DrawPlayerSummary(Vec2iZero(), playerSize, pds[0]);
690 DrawPlayerSummary(Vec2iNew(w / 2, 0), playerSize, pds[1]);
691 DrawPlayerSummary(Vec2iNew(0, h / 4), playerSize, pds[2]);
692 if (idx == 4)
693 {
694 DrawPlayerSummary(Vec2iNew(w / 2, h / 4), playerSize, pds[3]);
695 }
696 break;
697 default:
698 CASSERT(false, "not implemented");
699 break;
700 }
701}
702// Display compact player summary, with player on left half and score summaries
703// on right half
704static void DrawPlayerSummary(
705 const Vec2i pos, const Vec2i size, PlayerData *data)
706{
707 char s[50];
708 const int totalTextHeight = FontH() * 7;
709 // display text on right half
710 Vec2i textPos = Vec2iNew(
711 pos.x + size.x / 2, CENTER_Y(pos, size, totalTextHeight));
712
713 DisplayCharacterAndName(
714 Vec2iAdd(pos, Vec2iNew(size.x / 4, size.y / 2)),
715 &data->Char, DIRECTION_DOWN, data->name, colorWhite);
716
717 if (data->survived)
718 {
719 FontStr("Completed mission", textPos);
720 }
721 else
722 {
723 FontStrMask("Failed mission", textPos, colorRed);
724 }
725
726 textPos.y += 2 * FontH();
727 sprintf(s, "Score: %d", data->Stats.Score);
728 FontStr(s, textPos);
729 textPos.y += FontH();
730 sprintf(s, "Total: %d", data->Totals.Score);
731 FontStr(s, textPos);
732 textPos.y += FontH();
733 sprintf(s, "Missions: %d", data->missions + (data->survived ? 1 : 0));
734 FontStr(s, textPos);
735 textPos.y += FontH();
736
737 // Display bonuses if the player has survived
738 if (data->survived)
739 {
740 const int healthBonus = GetHealthBonus(data);
741 if (healthBonus != 0)
742 {
743 sprintf(s, "Health bonus: %d", healthBonus);
744 }
745 const int resurrectionFee = GetResurrectionFee(data);
746 if (resurrectionFee != 0)
747 {
748 sprintf(s, "Resurrection fee: %d", resurrectionFee);
749 }
750 if (healthBonus != 0 || resurrectionFee != 0)
751 {
752 FontStr(s, textPos);
753 textPos.y += FontH();
754 }
755
756 const int butcherPenalty = GetButcherPenalty(data);
757 if (butcherPenalty != 0)
758 {
759 sprintf(s, "Butcher penalty: %d", butcherPenalty);
760 }
761 const int ninjaBonus = GetNinjaBonus(data);
762 if (ninjaBonus != 0)
763 {
764 sprintf(s, "Ninja bonus: %d", ninjaBonus);
765 }
766 const int friendlyBonus = GetFriendlyBonus(data);
767 if (friendlyBonus != 0)
768 {
769 sprintf(s, "Friendly bonus: %d", friendlyBonus);
770 }
771 if (butcherPenalty != 0 || ninjaBonus != 0 || friendlyBonus != 0)
772 {
773 FontStr(s, textPos);
774 }
775 }

778static void DrawObjectiveInfo(const Objective *o, const Vec2i pos)
779{
780 const CharacterStore *store = &gCampaign.Setting.characters;
781
782 switch (o->Type)
783 {
784 case OBJECTIVE_KILL:
785 {
786 const Character *cd = CArrayGet(
787 &store->OtherChars, CharacterStoreGetSpecialId(store, 0));
788 DrawHead(cd, DIRECTION_DOWN, pos);
789 }
790 break;
791 case OBJECTIVE_RESCUE:
792 {
793 const Character *cd = CArrayGet(
794 &store->OtherChars, CharacterStoreGetPrisonerId(store, 0));
795 DrawHead(cd, DIRECTION_DOWN, pos);
796 }
797 break;
798 case OBJECTIVE_COLLECT:
799 {
800 const Pic *p = o->u.Pickup->Pic;
801 Blit(&gGraphicsDevice, p,
802 Vec2iMinus(pos, Vec2iScaleDiv(p->size, 2)));
803 }
804 break;
805 case OBJECTIVE_DESTROY:
806 {
807 Vec2i picOffset;
808 const Pic *p = MapObjectGetPic(o->u.MapObject, &picOffset);
809 Blit(&gGraphicsDevice, p, Vec2iAdd(pos, picOffset));
810 }
811 break;
812 case OBJECTIVE_INVESTIGATE:
813 // Don't draw
814 return;
815 default:
816 CASSERT(false, "Unknown objective type");
817 return;
818 }
