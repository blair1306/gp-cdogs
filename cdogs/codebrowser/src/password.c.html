76const char *MakePassword(int mission, int isTwoPlayers)
77{
78 static char s[PASSWORD_MAX + 1];
79 int sum1, sum2, count;
80 size_t i, x;
81 static char *alphabet1 = "0123456789abcdefghijklmnopqrstuvwxyz";
82 static char *alphabet2 = "9876543210kjihgfedcbazyxwvutsrqponml";
83 char *alphabet = isTwoPlayers ? alphabet2 : alphabet1;
84 size_t base = strlen(alphabet);
85
86 sum1 = sum2 = 0;
87 for (i = 0; i < strlen(gCampaign.Setting.Title); i++)
88 {
89 sum1 += gCampaign.Setting.Title[i];
90 sum2 ^= gCampaign.Setting.Title[i];
91 }
92
93 const int seed = ConfigGetInt(&gConfig, "Game.RandomSeed");
94 x = ((sum2 << 23) | (mission << 16) | sum1) ^ (size_t)seed;
95 count = 0;
96 while (x > 0 && count < PASSWORD_MAX) {
97 i = x % base;
98 s[count++] = alphabet[i];
99 x /= base;
100 }
101 s[count] = 0;
102 return s;
103}

105static int TestPassword(const char *password)
106{
107 int i;
108
109 for (i = 0; i < (int)gCampaign.Setting.Missions.size; i++)
110 {
111 // For legacy passwords, try both one and two players
112 if (strcmp(password, MakePassword(i, 0)) == 0 ||
113 strcmp(password, MakePassword(i, 1)) == 0)
114 {
115 return i;
116 }
117 }
118 return -1;
119}

121// Give user an alpha-num entry screen and returns the decoded mission number.
122// If the password is invalid, the screen won't exit.

135{
136 EnterCodeScreenData *data;

151static void EnterCodeScreenOnExit(GameLoopData *data)
152{
153 const EnterCodeScreenData *eData = data->Data;
154 if (eData->Mission > 0)
155 {
156 gCampaign.MissionIndex = eData->Mission;
157 SoundPlay(&gSoundDevice, StrSound("mg"));
158 }
159 else
160 {
161 SoundPlay(&gSoundDevice, StrSound("switch"));
162 }
163}
164#define PASSWORD_ENTRY_COLS 10
165#define PASSWORD_LETTERS "abcdefghijklmnopqrstuvwxyz0123456789"
166static bool PasswordEntry(EnterCodeScreenData *data, const int cmd);
167static GameLoopResult EnterCodeScreenUpdate(GameLoopData *data, LoopRunner *l)
168{
169 EnterCodeScreenData *eData = data->Data;
170
171 // Check if anyone pressed escape
172 int cmds[MAX_LOCAL_PLAYERS];
173 memset(cmds, 0, sizeof cmds);
174 GetPlayerCmds(&gEventHandlers, &cmds);
175 if (EventIsEscape(&gEventHandlers, cmds, GetMenuCmd(&gEventHandlers)))
176 {
177 eData->Mission = 0;
178 LoopRunnerPop(l);
179 return UPDATE_RESULT_OK;
180 }
181
182 const int cmd = GetMenuCmd(&gEventHandlers);
183 // Returning 1 means we are still entering password
184 if (PasswordEntry(eData, cmd))
185 {
186 return UPDATE_RESULT_DRAW;
187 }
188
189 if (strlen(eData->Buffer) == 0)
190 {
191 // Nothing in the password buffer; exit
192 eData->Mission = 0;
193 LoopRunnerPop(l);
194 return UPDATE_RESULT_OK;
195 }
196
197 eData->Mission = TestPassword(eData->Buffer);
198 if (eData->Mission == 0)
199 {
200 // Password doesn't work; play a bad sound
201 SoundPlay(&gSoundDevice, StrSound("aargh/man"));
202 return UPDATE_RESULT_OK;
203 }
204
205 // Password works; exit
206 LoopRunnerPop(l);
207 return UPDATE_RESULT_OK;
208}
209static bool PasswordEntry(EnterCodeScreenData *data, const int cmd)
210{
211 if (data->Selection == -1)
212 {
213 data->Selection = (int)strlen(PASSWORD_LETTERS);
214 }
215
216 if (cmd & CMD_BUTTON1)
217 {
218 if (data->Selection == (int)strlen(PASSWORD_LETTERS))
219 {
220 // Done
221 return false;
222 }
223
224 if (strlen(data->Buffer) < PASSWORD_MAX)
225 {
226 // enter a letter
227 data->Buffer[strlen(data->Buffer)] =
228 PASSWORD_LETTERS[data->Selection];
229 SoundPlay(&gSoundDevice, StrSound("mg"));
230 }
231 else
232 {
233 // Too many letters entered
234 SoundPlay(&gSoundDevice, StrSound("aargh/man"));
235 }
236 }
237 else if (cmd & CMD_BUTTON2)
238 {
239 if (strlen(data->Buffer) > 0)
240 {
241 // Delete a letter
242 data->Buffer[strlen(data->Buffer) - 1] = 0;
243 SoundPlay(&gSoundDevice, StrSound("bang"));
244 }
245 else
246 {
247 // No letters to delete
248 SoundPlay(&gSoundDevice, StrSound("aargh/man"));
249 }
250 }
251 else if (cmd & CMD_LEFT)
252 {
253 if (data->Selection > 0)
254 {
255 data->Selection--;
256 SoundPlay(&gSoundDevice, StrSound("door"));
257 }
258 }
259 else if (cmd & CMD_RIGHT)
260 {
261 if (data->Selection < (int)strlen(PASSWORD_LETTERS))
262 {
263 data->Selection++;
264 SoundPlay(&gSoundDevice, StrSound("door"));
265 }
266 }
267 else if (cmd & CMD_UP)
268 {
269 if (data->Selection >= PASSWORD_ENTRY_COLS)
270 {
271 data->Selection -= PASSWORD_ENTRY_COLS;
272 SoundPlay(&gSoundDevice, StrSound("door"));
273 }
274 }
275 else if (cmd & CMD_DOWN)
276 {
277 if (data->Selection <=
278 (int)strlen(PASSWORD_LETTERS) - PASSWORD_ENTRY_COLS)
279 {
280 data->Selection += PASSWORD_ENTRY_COLS;
281 SoundPlay(&gSoundDevice, StrSound("door"));
282 }
283 }
284
285 return true;
286}
287static void EnterCodeScreenDraw(GameLoopData *data)
288{
289 const EnterCodeScreenData *eData = data->Data;
290
291 BlitClearBuf(&gGraphicsDevice);
292
293 // Password display
294 Vec2i pos = Vec2iNew(
295 CenterX(FontStrW(eData->Buffer) + FontW('>') + FontW('<')),
296 gGraphicsDevice.cachedConfig.Res.y / 4);
297 pos = FontCh('>', pos);
298 pos = FontStr(eData->Buffer, pos);
299 FontCh('<', pos);
300
301 FontOpts opts = FontOptsNew();
302 opts.HAlign = ALIGN_CENTER;
303 opts.Area = gGraphicsDevice.cachedConfig.Res;
304 opts.Pad.y = gGraphicsDevice.cachedConfig.Res.y / 12;
305 FontStrOpt("Enter code", Vec2iZero(), opts);
306
307 // Draw password entry letters
308#define ENTRY_SPACING 12
309
310 const int x =
311 CenterX(ENTRY_SPACING * (PASSWORD_ENTRY_COLS - 1) + FontW('a'));
312 const int y = (int)CenterY(
313 FontH() * ((strlen(PASSWORD_LETTERS) - 1) / PASSWORD_ENTRY_COLS));
314 for (int i = 0; i < (int)strlen(PASSWORD_LETTERS) + 1; i++)
315 {
316 pos = Vec2iNew(
317 x + (i % PASSWORD_ENTRY_COLS) * ENTRY_SPACING,
318 y + (i / PASSWORD_ENTRY_COLS) * FontH());
319 color_t mask = (i == eData->Selection) ? colorRed : colorWhite;
320 if (i < (int)strlen(PASSWORD_LETTERS))
321 {
322 FontChMask(PASSWORD_LETTERS[i], pos, mask);
323 }
324 else
325 {
326 FontStrMask(DONE, pos, mask);
327 }
328 }
329
330 ShowControls();
331
332 BlitUpdateFromBuf(&gGraphicsDevice, gGraphicsDevice.screen);
333}

335typedef enum
336{
337 RETURN_CODE_CONTINUE = -1,
338 RETURN_CODE_START = -2,
339 RETURN_CODE_ENTER_CODE = -3
340} ReturnCode;

342typedef struct
343{
344 MenuSystem ms;
345 MissionSave save;
346 int mission;
347} PasswordData;
348static void PasswordTerminate(GameLoopData *data);
349static void PasswordOnEnter(GameLoopData *data);
350static GameLoopResult PasswordUpdate(GameLoopData *data, LoopRunner *l);
351static void PasswordDraw(GameLoopData *data);
352static void MenuCreateStart(
353 MenuSystem *ms, const int mission, const MissionSave *save);
354GameLoopData *EnterPassword(GraphicsDevice *graphics)
355{
356 PasswordData *data;
357 CMALLOC(data, sizeof *data);
358 AutosaveLoadMission(&gAutosave, &data->save, gCampaign.Entry.Path);
359 MenuSystemInit(
360 &data->ms, &gEventHandlers, graphics, Vec2iZero(),
361 graphics->cachedConfig.Res);
362 data->mission = TestPassword(data->save.Password);
363 MenuCreateStart(&data->ms, data->mission, &data->save);
364 return GameLoopDataNew(
365 data, PasswordTerminate, PasswordOnEnter, NULL,
366 NULL, PasswordUpdate, PasswordDraw);
367}
368static void MenuCreateStart(
369 MenuSystem *ms, const int mission, const MissionSave *save)
370{
371 ms->root = ms->current = MenuCreateNormal("", "", MENU_TYPE_NORMAL, 0);
372
373 menu_t *menuContinue = MenuCreateReturn("Continue", RETURN_CODE_CONTINUE);
374 // Note: mission can be -1
375 menuContinue->isDisabled = mission <= 0;
376 MenuAddSubmenu(ms->root, menuContinue);
377
378 // Create level select menus
379 menu_t *levelSelect = MenuCreateNormal(
380 "Level select...", "Select Level", MENU_TYPE_NORMAL, 0);
381 for (int i = 0;
382 i < MIN(save->MissionsCompleted + 1, (int)gCampaign.Setting.Missions.size);
383 i++)
384 {
385 const Mission *m = CArrayGet(&gCampaign.Setting.Missions, i);
386 char buf[CDOGS_FILENAME_MAX];
387 sprintf(buf, "%d: %s", i + 1, m->Title);
388 menu_t *l = MenuCreateReturn(buf, i);
389 l->isDisabled = i > save->MissionsCompleted;
390 MenuAddSubmenu(levelSelect, l);
391 }
392 levelSelect->isDisabled = save->MissionsCompleted == 0;
393 MenuAddSubmenu(ms->root, levelSelect);
394
395 MenuAddSubmenu(ms->root, MenuCreateReturn("Start campaign", RETURN_CODE_START));
396
397 MenuAddSubmenu(ms->root, MenuCreateReturn("Enter code...", RETURN_CODE_ENTER_CODE));
398
399 MenuAddExitType(ms, MENU_TYPE_RETURN);
400}
401static void PasswordTerminate(GameLoopData *data)
402{
403 PasswordData *pData = data->Data;
404
405 MenuSystemTerminate(&pData->ms);
406 CFREE(data->Data);
407}
408static void PasswordOnEnter(GameLoopData *data)
409{
410 PasswordData *pData = data->Data;
411
412 // TODO: re-detect mission saves on enter
413 MenuReset(&pData->ms);
414 gCampaign.MissionIndex = 0;
415}
416static GameLoopResult PasswordUpdate(GameLoopData *data, LoopRunner *l)
417{
418 PasswordData *pData = data->Data;
419
420 const GameLoopResult result = MenuUpdate(&pData->ms);
421 if (result == UPDATE_RESULT_OK)
422 {
423 if (pData->ms.hasAbort)
424 {
425 LoopRunnerPop(l);
426 }
427 else
428 {
429 // Check valid password
430 const int returnCode = pData->ms.current->u.returnCode;
431 switch (returnCode)
432 {
433 case RETURN_CODE_CONTINUE:
434 gCampaign.MissionIndex = pData->mission;
435 LoopRunnerChange(l, GameOptions(gCampaign.Entry.Mode));
436 break;
437 case RETURN_CODE_START:
438 LoopRunnerChange(l, GameOptions(gCampaign.Entry.Mode));
439 break;
440 case RETURN_CODE_ENTER_CODE:
441 LoopRunnerPush(l, EnterCodeScreen(pData->save.Password));
442 break;
443 default:
444 // Return code represents the mission to start on
445 CASSERT(returnCode >= 0, "Invalid return code for password menu");
446 gCampaign.MissionIndex = returnCode;
447 LoopRunnerChange(l, GameOptions(gCampaign.Entry.Mode));
448 break;
449 }
450 }
451 }
452 return result;
453}
454static void PasswordDraw(GameLoopData *data)
455{
456 const PasswordData *pData = data->Data;
457
458 MenuDraw(&pData->ms);
459}
