69void MenuSystemInit(
70 MenuSystem *ms,
71 EventHandlers *handlers, GraphicsDevice *graphics, Vec2i pos, Vec2i size)
72{
73 memset(ms, 0, sizeof *ms);
74 ms->root = ms->current = NULL;
75 CArrayInit(&ms->exitTypes, sizeof(menu_type_e));
76 CArrayInit(&ms->customDisplayFuncs, sizeof(MenuCustomDisplayFunc));
77 ms->handlers = handlers;
78 ms->graphics = graphics;
79 ms->pos = pos;
80 ms->size = size;
81 ms->align = MENU_ALIGN_CENTER;
82}

152GameLoopResult MenuUpdate(MenuSystem *ms)
153{
154 if (ms->current->type == MENU_TYPE_KEYS &&
155 ms->current->u.normal.changeKeyMenu != NULL)
156 {
157 MenuProcessChangeKey(ms->current);
158 }
159 else
160 {
161 const int cmd = GetMenuCmd(ms->handlers);
162 if (cmd)
163 {
164 MenuProcessCmd(ms, cmd);
165 }
166 }
167 // Check if anyone pressed escape, or we need a hard exit
168 int cmds[MAX_LOCAL_PLAYERS];
169 memset(cmds, 0, sizeof cmds);
170 GetPlayerCmds(&gEventHandlers, &cmds);
171 const bool aborted =
172 ms->allowAborts &&
173 EventIsEscape(&gEventHandlers, cmds, GetMenuCmd(&gEventHandlers));
174 if (aborted || ms->handlers->HasQuit)
175 {
176 ms->hasAbort = true;
177 return UPDATE_RESULT_OK;
178 }
179 if (MenuIsExit(ms))
180 {
181 return UPDATE_RESULT_OK;
182 }
183 if (ms->current->customPostUpdateFunc)
184 {
185 ms->current->customPostUpdateFunc(
186 ms->current, ms->current->customPostUpdateData);
187 }
188 return UPDATE_RESULT_DRAW;
189}

203static void MoveIndexToNextEnabledSubmenu(menu_t *menu, const bool isDown)
204{
205 if (menu->u.normal.index >= (int)menu->u.normal.subMenus.size)
206 {
207 menu->u.normal.index = (int)menu->u.normal.subMenus.size - 1;
208 }
209 const int firstIndex = menu->u.normal.index;
210 bool isFirst = true;
211 // Move the selection to the next non-disabled submenu
212 for (;;)
213 {
214 const menu_t *currentSubmenu =
215 CArrayGet(&menu->u.normal.subMenus, menu->u.normal.index);
216 if (!currentSubmenu->isDisabled)
217 {
218 break;
219 }
220 if (menu->u.normal.index == firstIndex && !isFirst)
221 {
222 break;
223 }
224 isFirst = false;
225 if (isDown)
226 {
227 menu->u.normal.index++;
228 if (menu->u.normal.index == (int)menu->u.normal.subMenus.size)
229 {
230 menu->u.normal.index = 0;
231 }
232 }
233 else
234 {
235 menu->u.normal.index--;
236 if (menu->u.normal.index == -1)
237 {
238 menu->u.normal.index = (int)menu->u.normal.subMenus.size - 1;
239 }
240 }
241 }
242}

283Vec2i DisplayMenuItem(
284 Vec2i pos, const char *s, int selected, int isDisabled, color_t color)
285{
286 if (selected)
287 {
288 return FontStrMask(s, pos, colorRed);
289 }
290 else if (isDisabled)
291 {
292 color_t dark = { 64, 64, 64, 255 };
293 return FontStrMask(s, pos, dark);
294 }
295 else if (!ColorEquals(color, colorBlack))
296 {
297 return FontStrMask(s, pos, color);
298 }
299 else
300 {
301 return FontStr(s, pos);
302 }
303}

346static void UpdateSubmenuParentPtrs(menu_t *menu)
347{
348 CA_FOREACH(menu_t, subMenu, menu->u.normal.subMenus)
349 subMenu->parentMenu = menu;
350 if (MenuTypeHasSubMenus(subMenu->type))
351 {
352 UpdateSubmenuParentPtrs(subMenu);
353 }
354 CA_FOREACH_END()
355}
356void MenuAddSubmenu(menu_t *menu, menu_t *subMenu)
357{
358 CArrayPushBack(&menu->u.normal.subMenus, subMenu);
359 if (subMenu->type == MENU_TYPE_QUIT)
360 {
361 menu->u.normal.quitMenuIndex = (int)menu->u.normal.subMenus.size - 1;
362 }
363 CFREE(subMenu);
364
365 // update all parent pointers, in child menus
366 UpdateSubmenuParentPtrs(menu);
367
368 // move cursor in case first menu item(s) are disabled
369 MoveIndexToNextEnabledSubmenu(menu, true);
370}

400menu_t *MenuCreateConfigOptions(
401 const char *name, const char *title, const Config *c, MenuSystem *ms,
402 const bool backOrReturn)
403{
404 menu_t *menu = MenuCreateNormal(name, title, MENU_TYPE_OPTIONS, 0);
405 CASSERT(c->Type == CONFIG_TYPE_GROUP,
406 "Cannot make menu from non-group config");
407 CA_FOREACH(Config, child, c->u.Group)
408 MenuAddConfigOptionsItem(menu, child);
409 CA_FOREACH_END()
410 MenuAddSubmenu(menu, MenuCreateSeparator(""));
411 if (backOrReturn)
412 {
413 MenuAddSubmenu(menu, MenuCreateBack("Done"));
414 }
415 else
416 {
417 MenuAddSubmenu(menu, MenuCreateReturn("Done", 0));
418 }
419 MenuSetPostInputFunc(menu, PostInputConfigApply, ms);
420 return menu;
421}
422void MenuAddConfigOptionsItem(menu_t *menu, Config *c)
423{
424 char nameBuf[256];
425 CASSERT(strlen(c->Name) < sizeof nameBuf, "buffer too small");
426 CamelToTitle(nameBuf, c->Name);
427 switch (c->Type)
428 {
429 case CONFIG_TYPE_STRING:
430 CASSERT(false, "Unimplemented");
431 break;
432 case CONFIG_TYPE_INT:
433 MenuAddSubmenu(
434 menu,
435 MenuCreateOptionRange(
436 nameBuf, (int *)&c->u.Int.Value,
437 c->u.Int.Min, c->u.Int.Max, c->u.Int.Increment,
438 MENU_OPTION_DISPLAY_STYLE_INT_TO_STR_FUNC,
439 (void (*)(void))c->u.Int.IntToStr));
440 break;
441 case CONFIG_TYPE_FLOAT:
442 CASSERT(false, "Unimplemented");
443 break;
444 case CONFIG_TYPE_BOOL:
445 MenuAddSubmenu(
446 menu, MenuCreateOptionToggle(nameBuf, &c->u.Bool.Value));
447 break;
448 case CONFIG_TYPE_ENUM:
449 MenuAddSubmenu(
450 menu,
451 MenuCreateOptionRange(
452 nameBuf, (int *)&c->u.Enum.Value,
453 c->u.Enum.Min, c->u.Enum.Max, 1,
454 MENU_OPTION_DISPLAY_STYLE_INT_TO_STR_FUNC,
455 (void(*)(void))c->u.Enum.EnumToStr));
456 break;
457 case CONFIG_TYPE_GROUP:
458 // Do nothing
459 break;
460 default:
461 CASSERT(false, "Unknown config type");
462 break;
463 }
464}

583void MenuDisplay(const MenuSystem *ms)
584{
585 const menu_t *menu = ms->current;
586 if (menu->type == MENU_TYPE_CUSTOM)
587 {
588 menu->u.customData.displayFunc(
589 menu, ms->graphics, ms->pos, ms->size, menu->u.customData.data);
590 }
591 else
592 {
593 MenuDisplayItems(ms);
594
595 if (strlen(menu->u.normal.title) != 0)
596 {
597 FontOpts opts = FontOptsNew();
598 opts.HAlign = ALIGN_CENTER;
599 opts.Area = ms->size;
600 opts.Pad = Vec2iNew(20, 20);
601 FontStrOpt(menu->u.normal.title, ms->pos, opts);
602 }
603
604 MenuDisplaySubmenus(ms);
605 }
606 CA_FOREACH(MenuCustomDisplayFunc, cdf, ms->customDisplayFuncs)
607 cdf->Func(NULL, ms->graphics, ms->pos, ms->size, cdf->Data);
608 CA_FOREACH_END()
609 if (menu->customDisplayFunc)
610 {
611 menu->customDisplayFunc(
612 menu, ms->graphics, ms->pos, ms->size, menu->customDisplayData);
613 }
614}
615static void MenuDisplayItems(const MenuSystem *ms)
616{
617 int d = ms->current->u.normal.displayItems;
618 if ((d & MENU_DISPLAY_ITEMS_CREDITS) && ms->creditsDisplayer != NULL)
619 {
620 ShowCredits(ms->creditsDisplayer);
621 }
622 if (d & MENU_DISPLAY_ITEMS_AUTHORS)
623 {
624 const Pic *logo = PicManagerGetPic(&gPicManager, "logo");
625 const Vec2i pos = Vec2iNew(
626 MS_CENTER_X(*ms, logo->size.x), ms->pos.y + ms->size.y / 12);
627 Blit(&gGraphicsDevice, logo, pos);
628
629 FontOpts opts = FontOptsNew();
630 opts.HAlign = ALIGN_END;
631 opts.Area = ms->size;
632 opts.Pad = Vec2iNew(20, 20);
633 FontStrOpt("Version: " CDOGS_SDL_VERSION, ms->pos, opts);
634 }

636static int MenuOptionGetIntValue(const menu_t *menu);
637static void MenuDisplaySubmenus(const MenuSystem *ms)
638{
639 int x = 0, yStart = 0;
640 const menu_t *menu = ms->current;
641
642 switch (menu->type)
643 {
644 // TODO: refactor the three menu types (normal, options, campaign) into one
645 case MENU_TYPE_NORMAL:
646 case MENU_TYPE_OPTIONS:
647 {
648 int iStart = 0;
649 int iEnd;
650 int numMenuLines = 0;
651 int maxIEnd = (int)menu->u.normal.subMenus.size;
652 if (menu->u.normal.maxItems > 0)
653 {
654 // Calculate first/last indices
655 if (menu->u.normal.scroll != 0)
656 {
657 iStart = menu->u.normal.scroll;
658 }
659 maxIEnd = iStart + menu->u.normal.maxItems;
660 }
661 // Count the number of menu items that can fit
662 // This is to account for multi-line items
663 for (iEnd = iStart;
664 iEnd < maxIEnd && iEnd < (int)menu->u.normal.subMenus.size;
665 iEnd++)
666 {
667 const menu_t *subMenu =
668 CArrayGet(&menu->u.normal.subMenus, iEnd);
669 const int numLines = FontStrNumLines(subMenu->name);
670 if (menu->u.normal.maxItems > 0 &&
671 numMenuLines + numLines > menu->u.normal.maxItems)
672 {
673 break;
674 }
675 numMenuLines += numLines;
676 }
677
678 int maxWidth = 0;
679 CA_FOREACH(const menu_t, subMenu, menu->u.normal.subMenus)
680 const int width = FontStrW(subMenu->name);
681 if (width > maxWidth)
682 {
683 maxWidth = width;
684 }
685 CA_FOREACH_END()
686 // Limit max width if it is larger than the menu system size
687 maxWidth = MIN(ms->size.x, maxWidth);
688 const bool isCentered = menu->type == MENU_TYPE_NORMAL;
689 switch (ms->align)
690 {
691 case MENU_ALIGN_CENTER:
692 x = MS_CENTER_X(*ms, maxWidth);
693 if (!isCentered)
694 {
695 x -= 20;
696 }
697 break;
698 case MENU_ALIGN_LEFT:
699 x = ms->pos.x;
700 break;
701 default:
702 assert(0 && "unknown alignment");
703 break;
704 }
705
706 yStart = MS_CENTER_Y(*ms, numMenuLines * FontH());
707 if (menu->u.normal.maxItems > 0)
708 {
709 // Display scroll arrows
710 if (menu->u.normal.scroll != 0)
711 {
712 DisplayMenuItem(
713 Vec2iNew(
714 MS_CENTER_X(*ms, FontW('^')),
715 yStart - 2 - FontH()),
716 "^",
717 0, 0,
718 colorBlack);
719 }
720 if (iEnd < (int)menu->u.normal.subMenus.size - 1)
721 {
722 DisplayMenuItem(
723 Vec2iNew(
724 MS_CENTER_X(*ms, FontW('v')),
725 yStart + numMenuLines*FontH() + 2),
726 "v",
727 0, 0,
728 colorBlack);
729 }
730 }
731 const int xOptions = x + maxWidth + 10;
732
733 // Display normal menu items
734 Vec2i pos = Vec2iNew(x, yStart);
735 for (int i = iStart; i < iEnd; i++)
736 {
737 const menu_t *subMenu = CArrayGet(&menu->u.normal.subMenus, i);
738 char *nameBuf;
739 CMALLOC(nameBuf, strlen(subMenu->name) + 3);
740 if (subMenu->type == MENU_TYPE_NORMAL &&
741 subMenu->u.normal.isSubmenusAlt)
742 {
743 sprintf(nameBuf, "%s >", subMenu->name);
744 }
745 else
746 {
747 strcpy(nameBuf, subMenu->name);
748 }
749
750 switch (menu->u.normal.align)
751 {
752 case MENU_ALIGN_CENTER:
753 pos.x = MS_CENTER_X(*ms, FontStrW(nameBuf));
754 break;
755 case MENU_ALIGN_LEFT:
756 // Do nothing
757 break;
758 default:
759 assert(0 && "unknown alignment");
760 break;
761 }
762
763 const int yNext = DisplayMenuItem(
764 pos,
765 nameBuf,
766 i == menu->u.normal.index,
767 subMenu->isDisabled,
768 subMenu->color).y + FontH();
769 CFREE(nameBuf);
770
771 // display option value
772 const int optionInt = MenuOptionGetIntValue(subMenu);
773 const Vec2i valuePos = Vec2iNew(xOptions, pos.y);
774 if (subMenu->type == MENU_TYPE_SET_OPTION_RANGE ||
775 subMenu->type == MENU_TYPE_SET_OPTION_SEED ||
776 subMenu->type == MENU_TYPE_SET_OPTION_UP_DOWN_VOID_FUNC_VOID ||
777 subMenu->type == MENU_TYPE_SET_OPTION_RANGE_GET_SET)
778 {
779 switch (subMenu->u.option.displayStyle)
780 {
781 case MENU_OPTION_DISPLAY_STYLE_NONE:
782 // Do nothing
783 break;
784 case MENU_OPTION_DISPLAY_STYLE_STR_FUNC:
785 FontStr(subMenu->u.option.uFunc.str(), valuePos);
786 break;
787 case MENU_OPTION_DISPLAY_STYLE_INT_TO_STR_FUNC:
788 FontStr(
789 subMenu->u.option.uFunc.intToStr(optionInt),
790 valuePos);
791 break;
792 default:
793 CASSERT(false, "unknown menu display type");
794 break;
795 }
796 }
797 else if (subMenu->type == MENU_TYPE_SET_OPTION_TOGGLE)
798 {
799 FontStr(optionInt ? "Yes" : "No", valuePos);
800 }
801
802 pos.y = yNext;
803 }
804 }
805 break;
806 case MENU_TYPE_KEYS:
807 {
808 x = MS_CENTER_X(*ms, (FontW('a') * 10)) / 2;
809 const int xKeys = x * 3;
810 yStart = (gGraphicsDevice.cachedConfig.Res.y / 2) - (FontH() * 10);
811
812 CA_FOREACH(const menu_t, subMenu, menu->u.normal.subMenus)
813 int y = yStart + _ca_index * FontH();
814 bool isSelected = _ca_index == menu->u.normal.index;
815
816 const char *name = subMenu->name;
817 if (isSelected &&
818 subMenu->type != MENU_TYPE_SET_OPTION_CHANGE_KEY)
819 {
820 FontStrMask(name, Vec2iNew(x, y), colorRed);
821 }
822 else
823 {
824 FontStr(name, Vec2iNew(x, y));
825 }
826
827 if (subMenu->type == MENU_TYPE_SET_OPTION_CHANGE_KEY)
828 {
829 const char *keyName;
830 if (menu->u.normal.changeKeyMenu == subMenu)
831 {
832 keyName = "Press a key";
833 }
834 else
835 {
836 const int pi = subMenu->u.changeKey.playerIndex;
837 const InputKeys *keys =
838 &gEventHandlers.keyboard.PlayerKeys[pi];
839 const SDL_Scancode sc = KeyGet(
840 keys, subMenu->u.changeKey.code);
841 keyName = SDL_GetScancodeName(sc);
842 if (sc == SDL_SCANCODE_UNKNOWN || keyName == NULL)
843 {
844 keyName = "<Unset>";
845 }
846 }
847 DisplayMenuItem(
848 Vec2iNew(xKeys, y),
849 keyName,
850 isSelected,
851 0,
852 colorBlack);
853 }
854 CA_FOREACH_END()
855 }
856 break;
857 default:
858 // No submenus, don't display anything
859 break;
860 }

951void MenuProcessCmd(MenuSystem *ms, int cmd)
952{
953 menu_t *menu = ms->current;
954 menu_t *menuToChange = NULL;
955 if (cmd == CMD_ESC || (cmd & CMD_BUTTON2) ||
956 ((cmd & CMD_LEFT) && menu->u.normal.isSubmenusAlt))
957 {
958 menuToChange = MenuProcessEscCmd(menu);
959 if (menuToChange != NULL)
960 {
961 MenuPlaySound(MENU_SOUND_BACK);
962 ms->current = menuToChange;
963 goto bail;
964 }
965 }
966 if (menu->type == MENU_TYPE_CUSTOM)
967 {
968 if (menu->u.customData.inputFunc(cmd, menu->u.customData.data))
969 {
970 ms->current = menu->parentMenu;
971 goto bail;
972 }
973 }
974 else
975 {
976 menuToChange = MenuProcessButtonCmd(ms, menu, cmd);
977 if (menuToChange != NULL)
978 {
979 debug(D_VERBOSE, "change to menu type %d\n", menuToChange->type);
980 MenuPlaySound(menuToChange->enterSound);
981 ms->current = menuToChange;
982 goto bail;
983 }
984 MenuChangeIndex(menu, cmd);
985 }
986
987bail:
988 if (menu->customPostInputFunc)
989 {
990 menu->customPostInputFunc(menu, cmd, menu->customPostInputData);
991 }
992 if (menuToChange && menuToChange->customPostEnterFunc)
993 {
994 menuToChange->customPostEnterFunc(
995 menuToChange, menuToChange->customPostEnterData);
996 }
997}

1025menu_t *MenuProcessButtonCmd(MenuSystem *ms, menu_t *menu, int cmd)
1026{
1027 if (AnyButton(cmd) || Left(cmd) || Right(cmd))
1028 {
1029 // Ignore if menu contains no submenus
1030 if (menu->u.normal.subMenus.size == 0)
1031 {
1032 return NULL;
1033 }
1034 menu_t *subMenu =
1035 CArrayGet(&menu->u.normal.subMenus, menu->u.normal.index);
1036
1037 // Only allow menu switching on button 1
1038
1039 switch (subMenu->type)
1040 {
1041 case MENU_TYPE_NORMAL:
1042 case MENU_TYPE_OPTIONS:
1043 case MENU_TYPE_KEYS:
1044 case MENU_TYPE_CUSTOM:
1045 if (subMenu->u.normal.isSubmenusAlt ?
1046 (cmd & CMD_RIGHT) : (cmd & CMD_BUTTON1))
1047 {
1048 return subMenu;
1049 }
1050 break;
1051 case MENU_TYPE_BACK:
1052 if (cmd & CMD_BUTTON1)
1053 {
1054 return menu->parentMenu;
1055 }
1056 break;
1057 case MENU_TYPE_QUIT:
1058 if (cmd & CMD_BUTTON1)
1059 {
1060 return subMenu; // caller will check if subMenu type is QUIT
1061 }
1062 break;
1063 case MENU_TYPE_RETURN:
1064 if (cmd & CMD_BUTTON1)
1065 {
1066 return subMenu;
1067 }
1068 break;
1069 default:
1070 MenuActivate(ms, subMenu, cmd);
1071 break;
1072 }
1073 }
1074 return NULL;

1077static bool KeyAvailable(
1078 const SDL_Scancode key, const key_code_e code, const int playerIndex)
1079{
1080 if (key == SDL_SCANCODE_ESCAPE ||
1081 key == SDL_SCANCODE_F9 || key == SDL_SCANCODE_F10)
1082 {
1083 return false;
1084 }
1085 if (key == (SDL_Scancode)ConfigGetInt(&gConfig, "Input.PlayerCodes0.map"))
1086 {
1087 return false;
1088 }
1089
1090 // Check if the key is being used by another control
1091 char buf[256];
1092 sprintf(buf, "Input.PlayerCodes%d", playerIndex);
1093 const InputKeys keys = KeyLoadPlayerKeys(ConfigGet(&gConfig, buf));
1094 for (key_code_e i = 0; i < KEY_CODE_MAP; i++)
1095 {
1096 if (i != code && KeyGet(&keys, i) == key)
1097 {
1098 return false;
1099 }
1100 }
1101
1102 // Check if the other player is using the key
1103 sprintf(buf, "Input.PlayerCodes%d", 1 - playerIndex);
1104 const InputKeys keysOther = KeyLoadPlayerKeys(ConfigGet(&gConfig, buf));
1105 if (keysOther.left == key ||
1106 keysOther.right == key ||
1107 keysOther.up == key ||
1108 keysOther.down == key ||
1109 keysOther.button1 == key ||
1110 keysOther.button2 == key)
1111 {
1112 return false;
1113 }
1114
1115 return true;
1116}

1118void MenuProcessChangeKey(menu_t *menu)
1119{
1120 // wait until user has pressed a new button
1121 const SDL_Scancode key = KeyGetPressed(&gEventHandlers.keyboard);
1122 if (key == 0)
1123 {
1124 return;
1125 }
1126 const key_code_e code = menu->u.normal.changeKeyMenu->u.changeKey.code;
1127 const int pi = menu->u.normal.changeKeyMenu->u.changeKey.playerIndex;
1128 if (key == SDL_SCANCODE_ESCAPE)
1129 {
1130 MenuPlaySound(MENU_SOUND_BACK);
1131 }
1132 else if (KeyAvailable(key, code, pi))
1133 {
1134 if (code != KEY_CODE_MAP)
1135 {
1136 char buf[256];
1137 sprintf(buf, "Input.PlayerCodes%d.%s", pi, KeycodeStr(code));
1138 ConfigGet(&gConfig, buf)->u.Int.Value = key;
1139 sprintf(buf, "Input.PlayerCodes%d", pi);
1140 gEventHandlers.keyboard.PlayerKeys[pi] = KeyLoadPlayerKeys(
1141 ConfigGet(&gConfig, buf));
1142 }
1143 else
1144 {
1145 ConfigGet(&gConfig, "Input.PlayerCodes0.map")->u.Int.Value = key;
1146 gEventHandlers.keyboard.PlayerKeys[0].map = key;
1147 }
1148 MenuPlaySound(MENU_SOUND_ENTER);
1149 }
1150 else
1151 {
1152 MenuPlaySound(MENU_SOUND_ERROR);
1153 }
1154 menu->u.normal.changeKeyMenu = NULL;
1155}

1158void MenuChangeIndex(menu_t *menu, int cmd)
1159{
1160 // Ignore if no submenus
1161 if (menu->u.normal.subMenus.size == 0)
1162 {
1163 return;
1164 }
1165
1166 if (Up(cmd))
1167 {
1168 menu->u.normal.index--;
1169 if (menu->u.normal.index == -1)
1170 {
1171 menu->u.normal.index = (int)menu->u.normal.subMenus.size - 1;
1172 }
1173 MoveIndexToNextEnabledSubmenu(menu, false);
1174 MenuPlaySound(MENU_SOUND_SWITCH);
1175 }
1176 else if (Down(cmd))
1177 {
1178 menu->u.normal.index++;
1179 if (menu->u.normal.index == (int)menu->u.normal.subMenus.size)
1180 {
1181 menu->u.normal.index = 0;
1182 }
1183 MoveIndexToNextEnabledSubmenu(menu, true);
1184 MenuPlaySound(MENU_SOUND_SWITCH);
1185 }
1186 menu->u.normal.scroll =
1187 CLAMP(menu->u.normal.scroll,
1188 MAX(0, menu->u.normal.index - menu->u.normal.maxItems + 1),
1189 MIN((int)menu->u.normal.subMenus.size - 1, menu->u.normal.index + menu->u.normal.maxItems - 1));
1190 if (menu->u.normal.index < menu->u.normal.scroll)
1191 {
1192 menu->u.normal.scroll = menu->u.normal.index;
1193 }
1194}

1197void MenuActivate(MenuSystem *ms, menu_t *menu, int cmd)
1198{
1199 UNUSED(ms);
1200 MenuPlaySound(MENU_SOUND_SWITCH);
1201 switch (menu->type)
1202 {
1203 case MENU_TYPE_BASIC:
1204 // do nothing
1205 return;
1206 case MENU_TYPE_SET_OPTION_TOGGLE:
1207 *menu->u.option.uHook.optionToggle = !*menu->u.option.uHook.optionToggle;
1208 break;
1209 case MENU_TYPE_SET_OPTION_RANGE:
1210 {
1211 int option = *menu->u.option.uHook.optionRange.option;
1212 int increment = menu->u.option.uHook.optionRange.increment;
1213 int low = menu->u.option.uHook.optionRange.low;
1214 int high = menu->u.option.uHook.optionRange.high;
1215 if (Left(cmd))
1216 {
1217 if (low == option)
1218 {
1219 option = high;
1220 }
1221 else if (low + increment > option)
1222 {
1223 option = low;
1224 }
1225 else
1226 {
1227 option -= increment;
1228 }
1229 }
1230 else if (Right(cmd))
1231 {
1232 if (high == option)
1233 {
1234 option = low;
1235 }
1236 else if (high - increment < option)
1237 {
1238 option = high;
1239 }
1240 else
1241 {
1242 option += increment;
1243 }
1244 }
1245 *menu->u.option.uHook.optionRange.option = option;
1246 }
1247 break;
1248 case MENU_TYPE_SET_OPTION_SEED:
1249 {
1250 unsigned int seed = *menu->u.option.uHook.seed;
1251 unsigned int increment = 1;
1252 if (Button1(cmd))
1253 {
1254 increment *= 10;
1255 }
1256 if (Button2(cmd))
1257 {
1258 increment *= 100;
1259 }
1260 if (Left(cmd))
1261 {
1262 if (increment > seed)
1263 {
1264 seed = 0;
1265 }
1266 else
1267 {
1268 seed -= increment;
1269 }
1270 }
1271 else if (Right(cmd))
1272 {
1273 if (UINT_MAX - increment < seed)
1274 {
1275 seed = UINT_MAX;
1276 }
1277 else
1278 {
1279 seed += increment;
1280 }
1281 }
1282 *menu->u.option.uHook.seed = seed;
1283 }
1284 break;
1285 case MENU_TYPE_SET_OPTION_UP_DOWN_VOID_FUNC_VOID:
1286 if (Left(cmd))
1287 {
1288 menu->u.option.uHook.upDownFuncs.upFunc();
1289 }
1290 else if (Right(cmd))
1291 {
1292 menu->u.option.uHook.upDownFuncs.downFunc();
1293 }
1294 break;
1295 case MENU_TYPE_SET_OPTION_RANGE_GET_SET:
1296 {
1297 int option = menu->u.option.uHook.optionRangeGetSet.getFunc();
1298 int increment = menu->u.option.uHook.optionRangeGetSet.increment;
1299 if (Left(cmd))
1300 {
1301 if (menu->u.option.uHook.optionRangeGetSet.low + increment > option)
1302 {
1303 option = menu->u.option.uHook.optionRangeGetSet.low;
1304 }
1305 else
1306 {
1307 option -= increment;
1308 }
1309 }
1310 else if (Right(cmd))
1311 {
1312 if (menu->u.option.uHook.optionRangeGetSet.high - increment < option)
1313 {
1314 option = menu->u.option.uHook.optionRangeGetSet.high;
1315 }
1316 else
1317 {
1318 option += increment;
1319 }
1320 }
1321 menu->u.option.uHook.optionRangeGetSet.setFunc(option);
1322 }
1323 break;
1324 case MENU_TYPE_VOID_FUNC:
1325 if (AnyButton(cmd))
1326 {
1327 menu->u.option.uHook.voidFunc.func(
1328 menu->u.option.uHook.voidFunc.data);
1329 }
1330 break;
1331 case MENU_TYPE_SET_OPTION_CHANGE_KEY:
1332 menu->parentMenu->u.normal.changeKeyMenu = menu;
1333 break;
1334 default:
1335 printf("Error unhandled menu type %d\n", menu->type);
1336 assert(0);
1337 break;
1338 }
1339}
