36static void WeaponSelect(menu_t *menu, int cmd, void *data)
37{
38 WeaponMenuData *d = data;
39 PlayerData *p = PlayerDataGetByUID(d->display.PlayerUID);
40 const CArray *weapons = &gMission.Weapons;
41
42 // Don't process if we're not selecting a weapon
43 if ((cmd & CMD_BUTTON1) && menu->u.normal.index < (int)weapons->size)
44 {
45 // Add the selected weapon
46
47 // Check that the weapon hasn't been chosen yet
48 const GunDescription **selectedWeapon =
49 CArrayGet(weapons, menu->u.normal.index);
50 for (int i = 0; i < p->weaponCount; i++)
51 {
52 if (p->weapons[i] == *selectedWeapon)
53 {
54 return;
55 }
56 }
57
58 // Check that there are empty slots to add weapons
59 if (p->weaponCount == MAX_WEAPONS)
60 {
61 return;
62 }
63
64 p->weapons[p->weaponCount] = *selectedWeapon;
65 p->weaponCount++;
66 SoundPlay(&gSoundDevice, (*selectedWeapon)->SwitchSound);
67
68 // Note: need to enable before disabling otherwise
69 // menu index is not updated properly
70
71 // Enable "Done" menu item
72 MenuEnableSubmenu(menu, (int)menu->u.normal.subMenus.size - 1);
73
74 // Disable this menu entry
75 MenuDisableSubmenu(menu, menu->u.normal.index);
76 }
77 else if (cmd & CMD_BUTTON2)
78 {
79 // Remove a weapon
80 if (p->weaponCount > 0)
81 {
82 p->weaponCount--;
83 MenuPlaySound(MENU_SOUND_BACK);
84
85 // Re-enable the menu entry for this weapon
86 const GunDescription *removedWeapon = p->weapons[p->weaponCount];
87 for (int i = 0; i < (int)weapons->size; i++)
88 {
89 const GunDescription **g = CArrayGet(weapons, i);
90 if (*g == removedWeapon)
91 {
92 MenuEnableSubmenu(menu, i);
93 break;
94 }
95 }
96 }
97
98 // Disable "Done" if no weapons selected
99 if (p->weaponCount == 0)
100 {
101 MenuDisableSubmenu(menu, (int)menu->u.normal.subMenus.size - 1);
102 }
103 }
104}

106static void DisplayEquippedWeapons(
107 const menu_t *menu, GraphicsDevice *g,
108 const Vec2i pos, const Vec2i size, const void *data)
109{
110 UNUSED(g);
111 const WeaponMenuData *d = data;
112 Vec2i weaponsPos;
113 Vec2i maxTextSize = FontStrSize("LongestWeaponName");
114 UNUSED(menu);
115 Vec2i dPos = pos;
116 dPos.x -= size.x; // move to left half of screen
117 weaponsPos = Vec2iNew(
118 dPos.x + size.x * 3 / 4 - maxTextSize.x / 2,
119 CENTER_Y(dPos, size, 0) + 14);
120 const PlayerData *p = PlayerDataGetByUID(d->display.PlayerUID);
121 if (p->weaponCount == 0)
122 {
123 FontStr("None selected...", weaponsPos);
124 }
125 else
126 {
127 for (int i = 0; i < p->weaponCount; i++)
128 {
129 FontStr(
130 p->weapons[i]->name,
131 Vec2iAdd(weaponsPos, Vec2iNew(0, i * FontH())));
132 }
133 }
134}

142void WeaponMenuCreate(
143 WeaponMenu *menu,
144 int numPlayers, int player, const int playerUID,
145 EventHandlers *handlers, GraphicsDevice *graphics)
146{
147 MenuSystem *ms = &menu->ms;
148 WeaponMenuData *data = &menu->data;
149 Vec2i pos, size;
150 int w = graphics->cachedConfig.Res.x;
151 int h = graphics->cachedConfig.Res.y;
152
153 data->display.PlayerUID = playerUID;
154 data->display.currentMenu = &ms->current;
155 data->display.Dir = DIRECTION_DOWN;
156 data->PlayerUID = playerUID;
157
158 switch (numPlayers)
159 {
160 case 1:
161 // Single menu, entire screen
162 pos = Vec2iNew(w / 2, 0);
163 size = Vec2iNew(w / 2, h);
164 break;
165 case 2:
166 // Two menus, side by side
167 pos = Vec2iNew(player * w / 2 + w / 4, 0);
168 size = Vec2iNew(w / 4, h);
169 break;
170 case 3:
171 case 4:
172 // Four corners
173 pos = Vec2iNew((player & 1) * w / 2 + w / 4, (player / 2) * h / 2);
174 size = Vec2iNew(w / 4, h / 2);
175 break;
176 default:
177 CASSERT(false, "not implemented");
178 pos = Vec2iNew(w / 2, 0);
179 size = Vec2iNew(w / 2, h);
180 break;
181 }
182 MenuSystemInit(ms, handlers, graphics, pos, size);
183 ms->align = MENU_ALIGN_LEFT;
184 ms->root = ms->current = MenuCreateNormal(
185 "",
186 "",
187 MENU_TYPE_NORMAL,
188 0);
189 ms->root->u.normal.maxItems = 11;
190 const CArray *weapons = &gMission.Weapons;
191 for (int i = 0; i < (int)weapons->size; i++)
192 {
193 const GunDescription **g = CArrayGet(weapons, i);
194 menu_t *gunMenu;
195 if ((*g)->Description != NULL)
196 {
197 // Gun description menu
198 gunMenu = MenuCreateNormal((*g)->name, "", MENU_TYPE_NORMAL, 0);
199 char *buf;
200 CMALLOC(buf, strlen((*g)->Description) * 2);
201 FontSplitLines((*g)->Description, buf, size.x * 5 / 6);
202 MenuAddSubmenu(gunMenu, MenuCreateBack(buf));
203 CFREE(buf);
204 gunMenu->u.normal.isSubmenusAlt = true;
205 MenuSetCustomDisplay(gunMenu, DisplayDescriptionGunIcon, *g);
206 }
207 else
208 {
209 gunMenu = MenuCreate((*g)->name, MENU_TYPE_BASIC);
210 }
211 MenuAddSubmenu(ms->root, gunMenu);
212 }
213 MenuSetPostInputFunc(ms->root, WeaponSelect, &data->display);
214 // Disable menu items where the player already has the weapon
215 PlayerData *pData = PlayerDataGetByUID(playerUID);
216 for (int i = 0; i < pData->weaponCount; i++)
217 {
218 for (int j = 0; j < (int)weapons->size; j++)
219 {
220 const GunDescription **g = CArrayGet(weapons, j);
221 if (pData->weapons[i] == *g)
222 {
223 MenuDisableSubmenu(ms->root, j);
224 }
225 }
226 }
227 MenuAddSubmenu(ms->root, MenuCreateSeparator(""));
228 MenuAddSubmenu(
229 ms->root, MenuCreateNormal("(End)", "", MENU_TYPE_NORMAL, 0));
230 // Select "(End)"
231 ms->root->u.normal.index = (int)ms->root->u.normal.subMenus.size - 1;
232
233 // Disable "Done" if no weapons selected
234 if (pData->weaponCount == 0)
235 {
236 MenuDisableSubmenu(ms->root, (int)ms->root->u.normal.subMenus.size - 1);
237 }
238
239 MenuSetCustomDisplay(ms->root, DisplayGunIcon, NULL);
240 MenuSystemAddCustomDisplay(ms, MenuDisplayPlayer, &data->display);
241 MenuSystemAddCustomDisplay(ms, DisplayEquippedWeapons, data);
242 MenuSystemAddCustomDisplay(
243 ms, MenuDisplayPlayerControls, &data->PlayerUID);
244}
245static void DisplayGunIcon(
246 const menu_t *menu, GraphicsDevice *g, const Vec2i pos, const Vec2i size,
247 const void *data)
248{
249 UNUSED(data);
250 if (menu->u.normal.index >= (int)gMission.Weapons.size)
251 {
252 return;
253 }
254 // Display a gun icon next to the currently selected weapon
255 const GunDescription **gun =
256 CArrayGet(&gMission.Weapons, menu->u.normal.index);
257 const int menuItems = MIN(
258 menu->u.normal.maxItems, (int)menu->u.normal.subMenus.size);
259 const int textScroll =
260 -menuItems * FontH() / 2 +
261 (menu->u.normal.index - menu->u.normal.scroll) * FontH();
262 const Vec2i iconPos = Vec2iNew(
263 pos.x - (*gun)->Icon->size.x - 4,
264 pos.y + size.y / 2 + textScroll + (FontH() - (*gun)->Icon->size.y) / 2);
265 Blit(g, (*gun)->Icon, iconPos);
266}
267static void DisplayDescriptionGunIcon(
268 const menu_t *menu, GraphicsDevice *g, const Vec2i pos, const Vec2i size,
269 const void *data)
270{
271 UNUSED(menu);
272 UNUSED(size);
273 const GunDescription *gun = data;
274 // Display the gun just to the left of the description text
275 const Vec2i iconPos = Vec2iNew(
276 pos.x - gun->Icon->size.x - 4, pos.y + size.y / 2);
277 Blit(g, gun->Icon, iconPos);
278}
