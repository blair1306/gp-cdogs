97#include <emscripten.h>
98#endif

100int main(int argc, char *argv[])
101{
102#if defined(_MSC_VER) && !defined(NDEBUG)
103 FreeConsole();
104#endif
105 int err = 0;
106 const char *loadCampaign = NULL;
107 ENetAddress connectAddr;
108 memset(&connectAddr, 0, sizeof connectAddr);
109
110 srand((unsigned int)time(NULL));
111 LogInit();
112
113 PrintTitle();
114
115 if (getenv("DEBUG") != NULL)
116 {
117 debug = true;
118 char *dbg;
119 if ((dbg = getenv("DEBUG_LEVEL")) != NULL)
120 {
121 debug_level = CLAMP(atoi(dbg), D_NORMAL, D_MAX);
122 }
123 }
124
125#ifdef __EMSCRIPTEN__
126 // initialize IDBFS for Emscripten persistent storage
127 EM_ASM(
128 FS.mkdir('/persistent_data');
129 FS.mount(IDBFS,{},'/persistent_data');
130
131 Module.print("start file sync..");
132 Module.syncdone = 0;
133
134 FS.syncfs(true, function(err) {
135 assert(!err);
136 Module.print("end file sync..");
137 Module.syncdone = 1;
138 });
139 );
140
141 SetupConfigDir();
142 gConfig = ConfigDefault();
143#else
144 SetupConfigDir();
145 gConfig = ConfigLoad(GetConfigFilePath(CONFIG_FILE));
146#endif
147 // Set config options that are only set via command line
148 ConfigGet(&gConfig, "Graphics.ShowHUD")->u.Bool.Value = true;
149 ConfigGet(&gConfig, "Graphics.ShakeMultiplier")->u.Int.Value = 1;
150
151 AutosaveInit(&gAutosave);
152#ifndef __EMSCRIPTEN__
153 AutosaveLoad(&gAutosave, GetConfigFilePath(AUTOSAVE_FILE));
154#endif
155
156#ifndef __EMSCRIPTEN__
157 if (enet_initialize() != 0)
158 {
159 LOG(LM_MAIN, LL_ERROR, "An error occurred while initializing ENet.");
160 err = EXIT_FAILURE;
161 goto bail;
162 }
163 NetClientInit(&gNetClient);
164#endif
165
166 // Print command line
167 char buf[CDOGS_PATH_MAX];
168 ProcessCommandLine(buf, argc, argv);
169 LOG(LM_MAIN, LL_INFO, "Command line (%d args):%s", argc, buf);
170 if (!ParseArgs(argc, argv, &connectAddr, &loadCampaign))
171 {
172 goto bail;
173 }
174
175 debug(D_NORMAL, "Initialising SDL...\n");
176#ifndef __EMSCRIPTEN__
177 const int sdlFlags =
178 SDL_INIT_TIMER | SDL_INIT_AUDIO | SDL_INIT_VIDEO | SDL_INIT_HAPTIC |
179 SDL_INIT_GAMECONTROLLER;
180#else
181 const int sdlFlags = SDL_INIT_TIMER | SDL_INIT_AUDIO | SDL_INIT_VIDEO;
182#endif
183 if (SDL_Init(sdlFlags) != 0)
184 {
185 LOG(LM_MAIN, LL_ERROR, "Could not initialise SDL: %s", SDL_GetError());
186 err = EXIT_FAILURE;
187 goto bail;
188 }
189 if (SDLJBN_Init() != 0)
190 {
191 LOG(LM_MAIN, LL_ERROR, "Could not initialise SDLJBN: %s",
192 SDLJBN_GetError());
193 err = EXIT_FAILURE;
194 goto bail;
195 }
196 SDL_EventState(SDL_DROPFILE, SDL_DISABLE);
197
198 GetDataFilePath(buf, "");
199 LOG(LM_MAIN, LL_INFO, "data dir(%s)", buf);
200 LOG(LM_MAIN, LL_INFO, "config dir(%s)", GetConfigFilePath(""));
201
202 SoundInitialize(&gSoundDevice, "sounds");
203 if (!gSoundDevice.isInitialised)
204 {
205 LOG(LM_MAIN, LL_ERROR, "Sound initialization failed!");
206 }
207
208 LoadSongs();
209
210 MusicPlayMenu(&gSoundDevice);
211
212 EventInit(&gEventHandlers, NULL, NULL, true);
213 NetServerInit(&gNetServer);
214 PicManagerInit(&gPicManager);
215 GraphicsInit(&gGraphicsDevice, &gConfig);
216 GraphicsInitialize(&gGraphicsDevice);
217 if (!gGraphicsDevice.IsInitialized)
218 {
219 LOG(LM_MAIN, LL_WARN, "Cannot initialise video; trying default config");
220 ConfigResetDefault(ConfigGet(&gConfig, "Graphics"));
221 GraphicsInit(&gGraphicsDevice, &gConfig);
222 GraphicsInitialize(&gGraphicsDevice);
223 }
224 if (!gGraphicsDevice.IsInitialized)
225 {
226 LOG(LM_MAIN, LL_ERROR, "Video didn't init!");
227 err = EXIT_FAILURE;
228 goto bail;
229 }
230 FontLoadFromJSON(&gFont, "graphics/font.png", "graphics/font.json");
231 PicManagerLoad(&gPicManager, "graphics");
232 CharSpriteClassesInit(&gCharSpriteClasses);
233
234 ParticleClassesInit(&gParticleClasses, "data/particles.json");
235 AmmoInitialize(&gAmmo, "data/ammo.json");
236 BulletAndWeaponInitialize(
237 &gBulletClasses, &gGunDescriptions,
238 "data/bullets.json", "data/guns.json");
239 CharacterClassesInitialize(&gCharacterClasses, "data/character_classes.json");
240#ifndef __EMSCRIPTEN__
241 PlayerTemplatesLoad(
242 &gPlayerTemplates, &gCharacterClasses, PLAYER_TEMPLATE_FILE);
243#endif
244 PickupClassesInit(
245 &gPickupClasses, "data/pickups.json", &gAmmo, &gGunDescriptions);
246 MapObjectsInit(
247 &gMapObjects, "data/map_objects.json", &gAmmo, &gGunDescriptions);
248 CollisionSystemInit(&gCollisionSystem);
249 CampaignInit(&gCampaign);
250 PlayerDataInit(&gPlayerDatas);
251
252 LoopRunner l = LoopRunnerNew(NULL);
253 // Attempt to pre-load campaign if requested
254 if (loadCampaign != NULL)
255 {
256 LOG(LM_MAIN, LL_INFO, "Loading campaign %s...", loadCampaign);
257 gCampaign.Entry.Mode =
258 strstr(loadCampaign, "/" CDOGS_DOGFIGHT_DIR "/") != NULL ?
259 GAME_MODE_DOGFIGHT : GAME_MODE_NORMAL;
260 CampaignEntry entry;
261 if (!CampaignEntryTryLoad(&entry, loadCampaign, GAME_MODE_NORMAL) ||
262 !CampaignLoad(&gCampaign, &entry))
263 {
264 LOG(LM_MAIN, LL_ERROR, "Failed to load campaign %s", loadCampaign);
265 }
266 }
267 else if (connectAddr.host != 0)
268 {
269 if (NetClientTryScanAndConnect(&gNetClient, connectAddr.host))
270 {
271 LoopRunnerPush(&l, ScreenWaitForCampaignDef());
272 }
273 else
274 {
275 printf("Failed to connect\n");
276 }
277 }
278 if (!gCampaign.IsLoaded)
279 {
280 LoopRunnerPush(&l, MainMenu(&gGraphicsDevice, &l));
281 }
282 LOG(LM_MAIN, LL_INFO, "Starting game");
283 LoopRunnerRun(&l);
284 LoopRunnerTerminate(&l);
285
286bail:
287 NetServerTerminate(&gNetServer);
288 MapTerminate(&gMap);
289 PlayerDataTerminate(&gPlayerDatas);
290 MapObjectsTerminate(&gMapObjects);
291 PickupClassesTerminate(&gPickupClasses);
292 ParticleClassesTerminate(&gParticleClasses);
293 AmmoTerminate(&gAmmo);
294 WeaponTerminate(&gGunDescriptions);
295 BulletTerminate(&gBulletClasses);
296 CharacterClassesTerminate(&gCharacterClasses);
297 MissionOptionsTerminate(&gMission);
298 NetClientTerminate(&gNetClient);
299 atexit(enet_deinitialize);
300 EventTerminate(&gEventHandlers);
301 GraphicsTerminate(&gGraphicsDevice);
302 CampaignTerminate(&gCampaign);
303 CollisionSystemTerminate(&gCollisionSystem);
304
305 CharSpriteClassesTerminate(&gCharSpriteClasses);
306 PicManagerTerminate(&gPicManager);
307 FontTerminate(&gFont);
308 AutosaveSave(&gAutosave, GetConfigFilePath(AUTOSAVE_FILE));
309 AutosaveTerminate(&gAutosave);
310 PlayerTemplatesTerminate(&gPlayerTemplates);
311 FreeSongs(&gMenuSongs);
312 FreeSongs(&gGameSongs);
313 SoundTerminate(&gSoundDevice, true);
314 ConfigDestroy(&gConfig);
315 LogTerminate();
316
317 SDLJBN_Quit();
318 SDL_Quit();
319
320 return err;
321}
