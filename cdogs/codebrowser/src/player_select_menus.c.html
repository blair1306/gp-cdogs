45static void DrawNameMenu(
46 const menu_t *menu, GraphicsDevice *g,
47 const Vec2i pos, const Vec2i size, const void *data)
48{
49 const PlayerSelectMenuData *d = data;
50
51#define ENTRY_COLS 8
52#define ENTRY_SPACING 7
53
54 int x = pos.x;
55 int y = CENTER_Y(
56 pos, size, FontH() * ((strlen(letters) - 1) / ENTRY_COLS));
57
58 UNUSED(menu);
59 UNUSED(g);
60
61 int i;
62 for (i = 0; i < (int)strlen(letters); i++)
63 {
64 Vec2i menuPos = Vec2iNew(
65 x + (i % ENTRY_COLS) * ENTRY_SPACING,
66 y + (i / ENTRY_COLS) * FontH());
67 FontChMask(
68 letters[i], menuPos,
69 i == d->nameMenuSelection ? colorRed : colorWhite);
70 }
71
72 DisplayMenuItem(
73 Vec2iNew(
74 x + (i % ENTRY_COLS) * ENTRY_SPACING,
75 y + (i / ENTRY_COLS) * FontH()),
76 "(End)", i == d->nameMenuSelection, 0, colorBlack);
77}

79static int HandleInputNameMenu(int cmd, void *data)
80{
81 PlayerSelectMenuData *d = data;
82 PlayerData *p = PlayerDataGetByUID(d->display.PlayerUID);
83
84 if (cmd & CMD_BUTTON1)
85 {
86 if (d->nameMenuSelection == (int)strlen(letters))
87 {
88 MenuPlaySound(MENU_SOUND_ENTER);
89 return 1;
90 }
91
92 if (strlen(p->name) < sizeof p->name - 1)
93 {
94 size_t l = strlen(p->name);
95 p->name[l + 1] = 0;
96 if (l > 0 && p->name[l - 1] != ' ')
97 {
98 p->name[l] = smallLetters[d->nameMenuSelection];
99 }
100 else
101 {
102 p->name[l] = letters[d->nameMenuSelection];
103 }
104 MenuPlaySound(MENU_SOUND_ENTER);
105 }
106 else
107 {
108 MenuPlaySound(MENU_SOUND_ERROR);
109 }
110 }
111 else if (cmd & CMD_BUTTON2)
112 {
113 if (p->name[0])
114 {
115 p->name[strlen(p->name) - 1] = 0;
116 MenuPlaySound(MENU_SOUND_BACK);
117 }
118 else
119 {
120 MenuPlaySound(MENU_SOUND_ERROR);
121 }
122 }
123 else if (cmd & CMD_LEFT)
124 {
125 if (d->nameMenuSelection > 0)
126 {
127 d->nameMenuSelection--;
128 MenuPlaySound(MENU_SOUND_SWITCH);
129 }
130 }
131 else if (cmd & CMD_RIGHT)
132 {
133 if (d->nameMenuSelection < (int)strlen(letters))
134 {
135 d->nameMenuSelection++;
136 MenuPlaySound(MENU_SOUND_SWITCH);
137 }
138 }
139 else if (cmd & CMD_UP)
140 {
141 if (d->nameMenuSelection >= ENTRY_COLS)
142 {
143 d->nameMenuSelection -= ENTRY_COLS;
144 MenuPlaySound(MENU_SOUND_SWITCH);
145 }
146 }
147 else if (cmd & CMD_DOWN)
148 {
149 if (d->nameMenuSelection <= (int)strlen(letters) - ENTRY_COLS)
150 {
151 d->nameMenuSelection += ENTRY_COLS;
152 MenuPlaySound(MENU_SOUND_SWITCH);
153 }
154 else if (d->nameMenuSelection < (int)strlen(letters))
155 {
156 d->nameMenuSelection = (int)strlen(letters);
157 MenuPlaySound(MENU_SOUND_SWITCH);
158 }
159 }
160
161 return 0;
162}

167static menu_t *CreateFaceMenu(MenuDisplayPlayerData *data)
168{
169 menu_t *menu = MenuCreateNormal("Face", "", MENU_TYPE_NORMAL, 0);
170 menu->u.normal.maxItems = 11;

180static void PostInputFaceMenu(menu_t *menu, int cmd, void *data)
181{
182 const MenuDisplayPlayerData *d = data;
183 // Change player face based on current menu selection
184 PlayerData *p = PlayerDataGetByUID(d->PlayerUID);
185 Character *c = &p->Char;
186 if (menu->u.normal.index < (int)gCharacterClasses.Classes.size)
187 {
188 c->Class = CArrayGet(&gCharacterClasses.Classes, menu->u.normal.index);
189 }
190 else
191 {
192 c->Class = CArrayGet(
193 &gCharacterClasses.CustomClasses,
194 menu->u.normal.index - (int)gCharacterClasses.Classes.size);
195 }
196 PostInputRotatePlayer(menu, cmd, data);
197}

199static void DrawColorMenu(
200 const menu_t *menu, GraphicsDevice *g,
201 const Vec2i pos, const Vec2i size, const void *data);
202static int HandleInputColorMenu(int cmd, void *data);
203static menu_t *CreateColorMenu(
204 const char *name, ColorMenuData *data,
205 const CharColorType type, const int playerUID)
206{
207 data->Type = type;
208 data->PlayerUID = playerUID;
209 data->palette = PicManagerGetPic(&gPicManager, "palette");
210 // Find the closest palette colour available, and select it
211 // Use least squares method
212 int errorLowest = -1;
213 PlayerData *p = PlayerDataGetByUID(playerUID);
214 const color_t currentColour = *CharColorGetByType(&p->Char.Colors, type);
215 Vec2i v;
216 for (v.y = 0; v.y < data->palette->size.y; v.y++)
217 {
218 for (v.x = 0; v.x < data->palette->size.x; v.x++)
219 {
220 const color_t colour = PIXEL2COLOR(
221 data->palette->Data[v.x + v.y * data->palette->size.x]);
222 if (colour.a == 0)
223 {
224 continue;
225 }
226 const int er = (int)colour.r - currentColour.r;
227 const int eg = (int)colour.g - currentColour.g;
228 const int eb = (int)colour.b - currentColour.b;
229 const int error = er*er + eg*eg + eb*eb;
230 if (errorLowest == -1 || error < errorLowest)
231 {
232 errorLowest = error;
233 data->selectedColor = v;
234 }
235 }
236 }
237 return MenuCreateCustom(name, DrawColorMenu, HandleInputColorMenu, data);
238}
239static void DrawColorMenu(
240 const menu_t *menu, GraphicsDevice *g,
241 const Vec2i pos, const Vec2i size, const void *data)
242{
243 UNUSED(menu);
244 const ColorMenuData *d = data;
245 // Draw colour squares from the palette
246 const Vec2i swatchSize = Vec2iNew(4, 4);
247 const Vec2i drawPos = Vec2iNew(
248 pos.x, CENTER_Y(pos, size, d->palette->size.y * swatchSize.y));
249 Vec2i v;
250 for (v.y = 0; v.y < d->palette->size.y; v.y++)
251 {
252 for (v.x = 0; v.x < d->palette->size.x; v.x++)
253 {
254 const color_t colour = PIXEL2COLOR(
255 d->palette->Data[v.x + v.y * d->palette->size.x]);
256 if (colour.a == 0)
257 {
258 continue;
259 }
260 DrawRectangle(
261 g, Vec2iAdd(drawPos, Vec2iMult(v, swatchSize)), swatchSize,
262 colour, 0);
263 }
264 }
265 // Draw a highlight around the selected colour
266 DrawRectangle(
267 g,
268 Vec2iAdd(
269 drawPos,
270 Vec2iMinus(
271 Vec2iMult(d->selectedColor, swatchSize), Vec2iNew(1, 1))),
272 Vec2iAdd(swatchSize, Vec2iNew(2, 2)),
273 colorWhite, DRAW_FLAG_LINE);
274}
275static int HandleInputColorMenu(int cmd, void *data)
276{
277 ColorMenuData *d = data;
278 Vec2i selected = d->selectedColor;
279 switch (cmd)
280 {
281 case CMD_LEFT: selected.x--; break;
282 case CMD_RIGHT: selected.x++; break;
283 case CMD_UP: selected.y--; break;
284 case CMD_DOWN: selected.y++; break;
285 case CMD_BUTTON1:
286 case CMD_BUTTON2: // fallthrough
287 MenuPlaySound(MENU_SOUND_ENTER);
288 return cmd;
289 default:
290 break;
291 }
292 if (selected.x >= 0 && selected.x < d->palette->size.x &&
293 selected.y >= 0 && selected.y < d->palette->size.y)
294 {
295 const color_t colour = PIXEL2COLOR(
296 d->palette->Data[selected.x + selected.y * d->palette->size.x]);
297 if (colour.a != 0)
298 {
299 d->selectedColor = selected;
300 PlayerData *p = PlayerDataGetByUID(d->PlayerUID);
301 *CharColorGetByType(&p->Char.Colors, d->Type) = colour;
302 MenuPlaySound(MENU_SOUND_SWITCH);
303 }
304 }
305 return 0;
306}

308static void PostInputLoadTemplate(menu_t *menu, int cmd, void *data)
309{
310 if (cmd & CMD_BUTTON1)
311 {
312 PlayerSelectMenuData *d = data;
313 PlayerData *p = PlayerDataGetByUID(d->display.PlayerUID);
314 const PlayerTemplate *t =
315 CArrayGet(&gPlayerTemplates, menu->u.normal.index);
316 memset(p->name, 0, sizeof p->name);
317 strncpy(p->name, t->name, sizeof p->name - 1);
318 p->Char.Class = StrCharacterClass(t->CharClassName);
319 if (p->Char.Class == NULL)
320 {
321 p->Char.Class = StrCharacterClass("Jones");
322 }
323 p->Char.Colors = t->Colors;
324 }
325}

327// Load all the template names to the menu entries
328static void PostEnterLoadTemplateNames(menu_t *menu, void *data)
329{
330 bool *isSave = (bool *)data;
331 for (int i = 0; i < (int)gPlayerTemplates.size; i++)
332 {
333 // Add menu if necessary
334 if (i == (int)menu->u.normal.subMenus.size)
335 {
336 MenuAddSubmenu(menu, MenuCreateBack(""));
337 }
338 menu_t *subMenu = CArrayGet(&menu->u.normal.subMenus, i);
339 const PlayerTemplate *pt = CArrayGet(&gPlayerTemplates, i);
340 CFREE(subMenu->name);
341 CSTRDUP(subMenu->name, pt->name);
342 }
343 if (*isSave && menu->u.normal.subMenus.size == gPlayerTemplates.size)
344 {
345 MenuAddSubmenu(menu, MenuCreateBack("(new)"));
346 }
347}

359static void PostInputSaveTemplate(menu_t *menu, int cmd, void *data)
360{
361 if (!(cmd & CMD_BUTTON1))
362 {
363 return;
364 }
365 PlayerSelectMenuData *d = data;
366 PlayerData *p = PlayerDataGetByUID(d->display.PlayerUID);
367 while (menu->u.normal.index >= (int)gPlayerTemplates.size)
368 {
369 PlayerTemplate empty;
370 memset(&empty, 0, sizeof empty);
371 CArrayPushBack(&gPlayerTemplates, &empty);
372 }
373 PlayerTemplate *t =
374 CArrayGet(&gPlayerTemplates, menu->u.normal.index);
375 memset(t->name, 0, sizeof t->name);
376 strncpy(t->name, p->name, sizeof t->name - 1);
377 CFREE(t->CharClassName);
378 CSTRDUP(t->CharClassName, p->Char.Class->Name);
379 t->Colors = p->Char.Colors;
380 SavePlayerTemplates(&gPlayerTemplates, PLAYER_TEMPLATE_FILE);
381}

421void PlayerSelectMenusCreate(
422 PlayerSelectMenu *menu,
423 int numPlayers, int player, const int playerUID,
424 EventHandlers *handlers, GraphicsDevice *graphics,
425 const NameGen *ng)
426{
427 MenuSystem *ms = &menu->ms;
428 PlayerSelectMenuData *data = &menu->data;
429 Vec2i pos, size;
430 int w = graphics->cachedConfig.Res.x;
431 int h = graphics->cachedConfig.Res.y;
432
433 data->nameMenuSelection = (int)strlen(letters);
434 data->display.PlayerUID = playerUID;
435 data->display.currentMenu = &ms->current;
436 data->display.Dir = DIRECTION_DOWN;
437 data->nameGenerator = ng;
438
439 switch (numPlayers)
440 {
441 case 1:
442 // Single menu, entire screen
443 pos = Vec2iNew(w / 2, 0);
444 size = Vec2iNew(w / 2, h);
445 break;
446 case 2:
447 // Two menus, side by side
448 pos = Vec2iNew(player * w / 2 + w / 4, 0);
449 size = Vec2iNew(w / 4, h);
450 break;
451 case 3:
452 case 4:
453 // Four corners
454 pos = Vec2iNew((player & 1) * w / 2 + w / 4, (player / 2) * h / 2);
455 size = Vec2iNew(w / 4, h / 2);
456 break;
457 default:
458 CASSERT(false, "not implemented");
459 pos = Vec2iNew(w / 2, 0);
460 size = Vec2iNew(w / 2, h);
461 break;
462 }
463 MenuSystemInit(ms, handlers, graphics, pos, size);
464 ms->align = MENU_ALIGN_LEFT;
465 ms->root = ms->current = MenuCreateNormal(
466 "",
467 "",
468 MENU_TYPE_NORMAL,
469 0);
470 MenuSetPostInputFunc(ms->root, PostInputRotatePlayer, &data->display);
471 MenuAddSubmenu(ms->root, MenuCreateCustom(
472 "Name", DrawNameMenu, HandleInputNameMenu, data));
473
474 MenuAddSubmenu(ms->root, CreateCustomizeMenu("Customize...", data));
475 MenuAddSubmenu(
476 ms->root,
477 MenuCreateVoidFunc("Shuffle", ShuffleAppearance, data));
478
479 MenuAddSubmenu(ms->root, CreateUseTemplateMenu("Load", data));
480 MenuAddSubmenu(ms->root, CreateSaveTemplateMenu("Save", data));
481
482 MenuAddSubmenu(ms->root, MenuCreateSeparator(""));
483 MenuAddSubmenu(
484 ms->root, MenuCreateNormal("Done", "", MENU_TYPE_NORMAL, 0));
485 // Select "Done"
486 ms->root->u.normal.index = (int)ms->root->u.normal.subMenus.size - 1;
487 MenuAddExitType(ms, MENU_TYPE_RETURN);
488 MenuSystemAddCustomDisplay(ms, MenuDisplayPlayer, &data->display);
489 MenuSystemAddCustomDisplay(
490 ms, MenuDisplayPlayerControls, &data->display.PlayerUID);
491
492 // Detect when there have been new player templates created,
493 // to re-enable the load menu
494 CheckReenableLoadMenu(ms->root, NULL);
495 MenuSetPostEnterFunc(ms->root, CheckReenableLoadMenu, NULL, false);
496}
497static menu_t *CreateCustomizeMenu(
498 const char *name, PlayerSelectMenuData *data)
499{
500 menu_t *menu = MenuCreateNormal(name, "", MENU_TYPE_NORMAL, 0);
501
502 MenuAddSubmenu(menu, CreateFaceMenu(&data->display));
503
504 MenuAddSubmenu(menu, CreateColorMenu(
505 "Skin", &data->skinData, CHAR_COLOR_SKIN, data->display.PlayerUID));
506 MenuAddSubmenu(menu, CreateColorMenu(
507 "Hair", &data->hairData, CHAR_COLOR_HAIR, data->display.PlayerUID));
508 MenuAddSubmenu(menu, CreateColorMenu(
509 "Arms", &data->armsData, CHAR_COLOR_ARMS, data->display.PlayerUID));
510 MenuAddSubmenu(menu, CreateColorMenu(
511 "Body", &data->bodyData, CHAR_COLOR_BODY, data->display.PlayerUID));
512 MenuAddSubmenu(menu, CreateColorMenu(
513 "Legs", &data->legsData, CHAR_COLOR_LEGS, data->display.PlayerUID));
514
515 MenuAddSubmenu(menu, MenuCreateSeparator(""));
516 MenuAddSubmenu(menu, MenuCreateBack("Back"));
517
518 MenuSetPostInputFunc(menu, PostInputRotatePlayer, &data->display);
519
520 return menu;
521}
522static void ShuffleAppearance(void *data)
523{
524 PlayerSelectMenuData *pData = data;
525 char buf[512];
526 NameGenMake(pData->nameGenerator, buf);
527 PlayerData *p = PlayerDataGetByUID(pData->display.PlayerUID);
528 strncpy(p->name, buf, 20);
529 Character *c = &p->Char;
530 CharacterShuffleAppearance(c);
531}

533static void PostInputRotatePlayer(menu_t *menu, int cmd, void *data)
534{
535 UNUSED(menu);
536 MenuDisplayPlayerData *d = data;
537 // Rotate player using left/right keys
538 const int dx = (cmd & CMD_LEFT) ? 1 : ((cmd & CMD_RIGHT) ? -1 : 0);
539 if (dx != 0)
540 {
541 d->Dir = (direction_e)CLAMP_OPPOSITE(
542 (int)d->Dir + dx, DIRECTION_UP, DIRECTION_UPLEFT);
543 SoundPlay(&gSoundDevice, StrSound("footsteps"));
544 }
