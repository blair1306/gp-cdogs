92static void ScreenWaitTerminate(GameLoopData *data);
93static GameLoopResult ScreenWaitUpdate(GameLoopData *data, LoopRunner *l);
94static void ScreenWaitDraw(GameLoopData *data);
95static GameLoopData *ScreenWait(
96 const char *message, GameLoopResult (*checkFunc)(void *, LoopRunner *l),
97 void *data)
98{
99 ScreenWaitData *swData;
100 CMALLOC(swData, sizeof *swData);
101 MenuSystemInit(
102 &swData->ms, &gEventHandlers, &gGraphicsDevice,
103 Vec2iZero(),
104 gGraphicsDevice.cachedConfig.Res);
105 swData->ms.allowAborts = true;
106 swData->ms.root = swData->ms.current =
107 MenuCreateNormal("", message, MENU_TYPE_NORMAL, 0);
108 MenuAddExitType(&swData->ms, MENU_TYPE_RETURN);
109 swData->checkFunc = checkFunc;
110 swData->data = data;
111
112 return GameLoopDataNew(
113 swData, ScreenWaitTerminate, NULL, NULL,
114 NULL, ScreenWaitUpdate, ScreenWaitDraw);
115}
116static void ScreenWaitTerminate(GameLoopData *data)
117{
118 ScreenWaitData *swData = data->Data;
119
120 MenuSystemTerminate(&swData->ms);
121 CFREE(swData->data);
122 CFREE(swData);
123}
124static GameLoopResult ScreenWaitUpdate(GameLoopData *data, LoopRunner *l)
125{
126 ScreenWaitData *swData = data->Data;
127
128 const GameLoopResult result = swData->checkFunc(swData->data, l);
129 if (result != UPDATE_RESULT_DRAW)
130 {
131 return result;
132 }
133 const GameLoopResult menuResult = MenuUpdate(&swData->ms);
134 if (menuResult == UPDATE_RESULT_OK)
135 {
136 LoopRunnerPop(l);
137 }
138 return menuResult;
139}
140static void ScreenWaitDraw(GameLoopData *data)
141{
142 const ScreenWaitData *swData = data->Data;
143
144 MenuDraw(&swData->ms);
145}

217static GameLoopResult NumPlayersUpdate(GameLoopData *data, LoopRunner *l)
218{
219 MenuSystem *ms = data->Data;
220 const GameLoopResult result = MenuUpdate(ms);
221 if (result == UPDATE_RESULT_OK)
222 {
223 if (ms->hasAbort)
224 {
225 CampaignUnload(&gCampaign);
226 LoopRunnerPop(l);
227 }
228 else
229 {
230 const int numPlayers = ms->current->u.returnCode;
231 CA_FOREACH(const PlayerData, p, gPlayerDatas)
232 CASSERT(!p->IsLocal, "unexpected local player");
233 CA_FOREACH_END()
234 // Add the players
235 for (int i = 0; i < numPlayers; i++)
236 {
237 GameEvent e = GameEventNew(GAME_EVENT_PLAYER_DATA);
238 e.u.PlayerData = PlayerDataDefault(i);
239 e.u.PlayerData.UID = gNetClient.FirstPlayerUID + i;
240 GameEventsEnqueue(&gGameEvents, e);
241 }
242 // Process the events to force add the players
243 HandleGameEvents(&gGameEvents, NULL, NULL, NULL);
244 // This also causes the client to send player data to the server
245
246 // Switch to player selection
247 LoopRunnerChange(l, PlayerSelection());
248 }
249 }
250 return result;
251}

253{
254 MenuDraw(data->Data);
255}

258static void AssignPlayerInputDevices(EventHandlers *handlers)
259{
260 CA_FOREACH(PlayerData, p, gPlayerDatas)
261 if (!p->IsLocal)
262 {
263 continue;
264 }
265
266 // Try to assign devices to players
267 // For each unassigned player, check if any device has button 1 pressed
268 // If so, and that input device wasn't used, assign it to that player
269 for (int j = 0; j < MAX_KEYBOARD_CONFIGS; j++)
270 {
271 char buf[256];
272 sprintf(buf, "Input.PlayerCodes%d.button1", j);
273 if (KeyIsPressed(&handlers->keyboard, ConfigGetInt(&gConfig, buf)) &&
274 PlayerTrySetUnusedInputDevice(p, INPUT_DEVICE_KEYBOARD, j))
275 {
276 SoundPlay(&gSoundDevice, StrSound("hahaha"));
277 break;
278 }
279 }
280 if (MouseIsPressed(&handlers->mouse, SDL_BUTTON_LEFT) &&
281 PlayerTrySetUnusedInputDevice(p, INPUT_DEVICE_MOUSE, 0))
282 {
283 SoundPlay(&gSoundDevice, StrSound("hahaha"));
284 continue;
285 }
286 for (int j = 0; j < (int)handlers->joysticks.size; j++)
287 {
288 const Joystick *joy = CArrayGet(&handlers->joysticks, j);
289 if (JoyIsPressed(joy->id, CMD_BUTTON1) &&
290 PlayerTrySetUnusedInputDevice(p, INPUT_DEVICE_JOYSTICK, joy->id))
291 {
292 SoundPlay(&gSoundDevice, StrSound("hahaha"));
293 break;
294 }
295 }
296 CA_FOREACH_END()
297}

312GameLoopData *PlayerSelection(void)
313{
314 PlayerSelectionData *data;
315 CCALLOC(data, sizeof *data);
316 data->waitResult = EVENT_WAIT_CONTINUE;
317 GetDataFilePath(data->prefixes, "data/prefixes.txt");
318 GetDataFilePath(data->suffixes, "data/suffixes.txt");
319 GetDataFilePath(data->suffixnames, "data/suffixnames.txt");
320 NameGenInit(&data->g, data->prefixes, data->suffixes, data->suffixnames);
321
322 // Create selection menus for each local player
323 for (int i = 0, idx = 0; i < (int)gPlayerDatas.size; i++, idx++)
324 {
325 PlayerData *p = CArrayGet(&gPlayerDatas, i);
326 if (!p->IsLocal)
327 {
328 idx--;
329 continue;
330 }
331 PlayerSelectMenusCreate(
332 &data->menus[idx], GetNumPlayers(PLAYER_ANY, false, true),
333 idx, p->UID,
334 &gEventHandlers, &gGraphicsDevice, &data->g);
335 }
336
337 return GameLoopDataNew(
338 data, PlayerSelectionTerminate, NULL, PlayerSelectionOnExit,
339 NULL, PlayerSelectionUpdate, PlayerSelectionDraw);
340}

345 for (int i = 0; i < GetNumPlayers(PLAYER_ANY, false, true); i++)

352static void PlayerSelectionOnExit(GameLoopData *data)
353{
354 PlayerSelectionData *pData = data->Data;
355 if (pData->waitResult == EVENT_WAIT_OK)
356 {
357 for (int i = 0, idx = 0; i < (int)gPlayerDatas.size; i++, idx++)
358 {
359 PlayerData *p = CArrayGet(&gPlayerDatas, i);
360 if (!p->IsLocal)
361 {
362 idx--;
363 continue;
364 }
365
366 // For any player slots not picked, turn them into AIs
367 if (p->inputDevice == INPUT_DEVICE_UNSET)
368 {
369 PlayerTrySetInputDevice(p, INPUT_DEVICE_AI, 0);
370 }
371 }
372
373 if (!gCampaign.IsClient)
374 {
375 gCampaign.MissionIndex = 0;
376 }
377 }
378 else
379 {
380 CampaignUnload(&gCampaign);
381 }
382}
383static GameLoopResult PlayerSelectionUpdate(GameLoopData *data, LoopRunner *l)
384{
385 PlayerSelectionData *pData = data->Data;
386
387 if (GetNumPlayers(PLAYER_ANY, false, true) == 0)
388 {
389 pData->waitResult = EVENT_WAIT_OK;
390 LoopRunnerPop(l);
391 return UPDATE_RESULT_OK;
392 }
393
394 // Check if anyone pressed escape
395 int cmds[MAX_LOCAL_PLAYERS];
396 memset(cmds, 0, sizeof cmds);
397 GetPlayerCmds(&gEventHandlers, &cmds);
398 if (EventIsEscape(&gEventHandlers, cmds, GetMenuCmd(&gEventHandlers)))
399 {
400 pData->waitResult = EVENT_WAIT_CANCEL;
401 LoopRunnerPop(l);
402 return UPDATE_RESULT_OK;
403 }
404
405 // Menu input
406 int idx = 0;
407 for (int i = 0; i < (int)gPlayerDatas.size; i++, idx++)
408 {
409 const PlayerData *p = CArrayGet(&gPlayerDatas, i);
410 if (!p->IsLocal)
411 {
412 idx--;
413 continue;
414 }
415 if (p->inputDevice != INPUT_DEVICE_UNSET &&
416 !MenuIsExit(&pData->menus[idx].ms) && cmds[idx])
417 {
418 MenuProcessCmd(&pData->menus[idx].ms, cmds[idx]);
419 }
420 }
421
422 // Conditions for exit: at least one player has selected "Done",
423 // and no other players, if any, are still selecting their player
424 // The "players" with no input device are turned into AIs
425 bool hasAtLeastOneInput = false;
426 bool isDone = true;
427 idx = 0;
428 for (int i = 0; i < (int)gPlayerDatas.size; i++, idx++)
429 {
430 const PlayerData *p = CArrayGet(&gPlayerDatas, i);
431 if (!p->IsLocal)
432 {
433 idx--;
434 continue;
435 }
436 if (p->inputDevice != INPUT_DEVICE_UNSET)
437 {
438 hasAtLeastOneInput = true;
439 if (strcmp(pData->menus[idx].ms.current->name, "Done") != 0)
440 {
441 isDone = false;
442 }
443 }
444 }
445 if (isDone && hasAtLeastOneInput)
446 {
447 pData->waitResult = EVENT_WAIT_OK;
448 if (!gCampaign.IsClient && IsPasswordAllowed(gCampaign.Entry.Mode))
449 {
450 LoopRunnerChange(l, EnterPassword(&gGraphicsDevice));
451 }
452 else
453 {
454 LoopRunnerChange(l, GameOptions(gCampaign.Entry.Mode));
455 }
456 return UPDATE_RESULT_OK;
457 }
458
459 AssignPlayerInputDevices(&gEventHandlers);
460
461 return UPDATE_RESULT_DRAW;
462}
463static void PlayerSelectionDraw(GameLoopData *data)
464{
465 const PlayerSelectionData *pData = data->Data;
466
467 BlitClearBuf(&gGraphicsDevice);
468 const int w = gGraphicsDevice.cachedConfig.Res.x;
469 const int h = gGraphicsDevice.cachedConfig.Res.y;
470 int idx = 0;
471 for (int i = 0; i < (int)gPlayerDatas.size; i++, idx++)
472 {
473 const PlayerData *p = CArrayGet(&gPlayerDatas, i);
474 if (!p->IsLocal)
475 {
476 idx--;
477 continue;
478 }
479 if (p->inputDevice != INPUT_DEVICE_UNSET)
480 {
481 MenuDisplay(&pData->menus[idx].ms);
482 }
483 else
484 {
485 Vec2i center = Vec2iZero();
486 const char *prompt = "Press Fire to join...";
487 const Vec2i offset = Vec2iScaleDiv(FontStrSize(prompt), -2);
488 switch (GetNumPlayers(false, false, true))
489 {
490 case 1:
491 // Center of screen
492 center = Vec2iNew(w / 2, h / 2);
493 break;
494 case 2:
495 // Side by side
496 center = Vec2iNew(idx * w / 2 + w / 4, h / 2);
497 break;
498 case 3:
499 case 4:
500 // Four corners
501 center = Vec2iNew(
502 (idx & 1) * w / 2 + w / 4, (idx / 2) * h / 2 + h / 4);
503 break;
504 default:
505 CASSERT(false, "not implemented");
506 break;
507 }
508 FontStr(prompt, Vec2iAdd(center, offset));
509 }
510 }
511
512 BlitUpdateFromBuf(&gGraphicsDevice, gGraphicsDevice.screen);
513}

625static void GameOptionsOnEnter(GameLoopData *data)
626{
627 GameOptionsData *gData = data->Data;
628
629 if (!gMission.IsQuit)
630 {
631 MissionOptionsTerminate(&gMission);
632 CampaignAndMissionSetup(&gCampaign, &gMission);
633 }
634
635 gData->ms.current = gData->ms.root;
636 // Select "Done"
637 gData->ms.root->u.normal.index =
638 (int)gData->ms.root->u.normal.subMenus.size - 1;

640static GameLoopResult GameOptionsUpdate(GameLoopData *data, LoopRunner *l)
641{
642 GameOptionsData *gData = data->Data;
643
644 // Check end conditions:
645 // - Campaign not loaded
646 // - Campaign complete
647 // - Mission quit
648 // - No options needed
649 // - Menu complete
650 const GameLoopResult result = MenuUpdate(&gData->ms);
651 const bool isQuit =
652 !gCampaign.IsLoaded || gCampaign.IsComplete || gMission.IsQuit ||
653 gData->ms.hasAbort;
654 const bool isDone =
655 !IsGameOptionsNeeded(gCampaign.Entry.Mode) ||
656 result == UPDATE_RESULT_OK;
657 if (isQuit || isDone)
658 {
659 if (isQuit)
660 {
661 MissionOptionsTerminate(&gMission);
662 CampaignUnload(&gCampaign);
663 LoopRunnerPop(l);
664 }
665 else
666 {
667 if (!ConfigApply(&gConfig))
668 {
669 LOG(LM_MAIN, LL_ERROR,
670 "Failed to apply config; reset to last used");
671 ConfigResetChanged(&gConfig);
672 }
673 else
674 {
675 // Save options for later
676 ConfigSave(&gConfig, GetConfigFilePath(CONFIG_FILE));
677 }
678
679 // Set allowed weapons
680 // First check if the player has unwittingly disabled all weapons
681 // if so, enable all weapons
682 bool allDisabled = true;
683 for (int i = 0, j = 0; i < (int)gData->allowed.size; i++, j++)
684 {
685 const bool *allowed = CArrayGet(&gData->allowed, i);
686 if (*allowed)
687 {
688 allDisabled = false;
689 break;
690 }
691 }
692 if (!allDisabled)
693 {
694 for (int i = 0, j = 0; i < (int)gData->allowed.size; i++, j++)
695 {
696 const bool *allowed = CArrayGet(&gData->allowed, i);
697 if (!*allowed)
698 {
699 CArrayDelete(&gMission.Weapons, j);
700 j--;
701 }
702 }
703 }
704
705 gCampaign.OptionsSet = true;
706
707 // If enabled, start net server
708 if (!gCampaign.IsClient && ConfigGetBool(&gConfig, "StartServer"))
709 {
710 NetServerOpen(&gNetServer);
711 }
712 LoopRunnerPush(l, ScreenMissionBriefing(&gMission));
713 }
714 return UPDATE_RESULT_OK;
715 }
716 return UPDATE_RESULT_DRAW;
717}

724static menu_t *MenuCreateAllowedWeapons(
725 const char *name, GameOptionsData *data)
726{
727 // Create a menu to choose allowable weapons for this map
728 // The weapons will be chosen from the available weapons
729 menu_t *m =
730 MenuCreateNormal(name, "Available Weapons:", MENU_TYPE_OPTIONS, 0);
731 m->customPostInputData = data;
732 for (int i = 0; i < (int)data->weapons->size; i++)
733 {
734 const GunDescription **g = CArrayGet(data->weapons, i);
735 MenuAddSubmenu(m,
736 MenuCreateOptionToggle((*g)->name, CArrayGet(&data->allowed, i)));
737 }
738 MenuAddSubmenu(m, MenuCreateSeparator(""));
739 MenuAddSubmenu(m, MenuCreateBack("Done"));
740 return m;
741}

758 for (int i = 0, idx = 0; i < (int)gPlayerDatas.size; i++, idx++)
759 {
760 PlayerData *p = CArrayGet(&gPlayerDatas, i);
761 if (!p->IsLocal)
762 {
763 idx--;
764 continue;
765 }
766
767 // Remove unavailable weapons from players inventories
768 RemoveUnavailableWeapons(p, &gMission.Weapons);
769
770 WeaponMenuCreate(
771 &data->menus[idx], GetNumPlayers(PLAYER_ANY, false, true),
772 idx, p->UID,
773 &gEventHandlers, &gGraphicsDevice);
774 // For AI players, pre-pick their weapons and go straight to menu end
775 if (p->inputDevice == INPUT_DEVICE_AI)
776 {
777 const int lastMenuIndex =
778 (int)data->menus[idx].ms.root->u.normal.subMenus.size - 1;
779 data->menus[idx].ms.current = CArrayGet(
780 &data->menus[idx].ms.root->u.normal.subMenus, lastMenuIndex);
781 AICoopSelectWeapons(p, idx, &gMission.Weapons);
782 }
783 }
784
785 return GameLoopDataNew(
786 data, PlayerEquipTerminate, NULL, PlayerEquipOnExit,
787 NULL, PlayerEquipUpdate, PlayerEquipDraw);
788}

797 {
798 data->weapons[j - 1] = data->weapons[j];
799 }
800 data->weaponCount--;
801 }
802 }

858static GameLoopResult PlayerEquipUpdate(GameLoopData *data, LoopRunner *l)
859{
860 PlayerEquipData *pData = data->Data;
861
862 // If no human players, don't show equip screen
863 if (GetNumPlayers(PLAYER_ANY, false, true) == 0)
864 {
865 pData->waitResult = EVENT_WAIT_OK;
866 goto bail;
867 }
868
869 // Check if anyone pressed escape
870 int cmds[MAX_LOCAL_PLAYERS];
871 memset(cmds, 0, sizeof cmds);
872 GetPlayerCmds(&gEventHandlers, &cmds);
873 if (EventIsEscape(&gEventHandlers, cmds, GetMenuCmd(&gEventHandlers)))
874 {
875 pData->waitResult = EVENT_WAIT_CANCEL;
876 goto bail;
877 }
878
879 // Update menus
880 int idx = 0;
881 for (int i = 0; i < (int)gPlayerDatas.size; i++, idx++)
882 {
883 const PlayerData *p = CArrayGet(&gPlayerDatas, i);
884 if (!p->IsLocal)
885 {
886 idx--;
887 continue;
888 }
889 if (!MenuIsExit(&pData->menus[idx].ms))
890 {
891 MenuProcessCmd(&pData->menus[idx].ms, cmds[idx]);
892 }
893 else if (p->weaponCount == 0)
894 {
895 // Check exit condition; must have selected at least one weapon
896 // Otherwise reset the current menu
897 pData->menus[idx].ms.current = pData->menus[idx].ms.root;
898 }
899 }
900
901 bool isDone = true;
902 for (int i = 0; i < GetNumPlayers(PLAYER_ANY, false, true); i++)
903 {
904 if (strcmp(pData->menus[i].ms.current->name, "(End)") != 0)
905 {
906 isDone = false;
907 }
908 }
909 if (isDone)
910 {
911 pData->waitResult = EVENT_WAIT_OK;
912 goto bail;
913 }
914
915 return UPDATE_RESULT_DRAW;
916
917bail:
918 if (pData->waitResult == EVENT_WAIT_OK)
919 {
920 LoopRunnerChange(l, ScreenWaitForGameStart());
921 }
922 else
923 {
924 LoopRunnerPop(l);
925 }
926 return UPDATE_RESULT_OK;
927}
