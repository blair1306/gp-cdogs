49void MissionSaveInit(MissionSave *ms)
50{
51 memset(ms, 0, sizeof *ms);
52 ms->IsValid = 1;
53}

56void AutosaveInit(Autosave *autosave)
57{
58 memset(&autosave->LastMission.Campaign, 0, sizeof autosave->LastMission.Campaign);
59 autosave->LastMission.Campaign.Mode = GAME_MODE_NORMAL;
60 strcpy(autosave->LastMission.Password, "");
61 CArrayInit(&autosave->Missions, sizeof(MissionSave));
62}
63void AutosaveTerminate(Autosave *autosave)
64{
65 CA_FOREACH(MissionSave, m, autosave->Missions)
66 CampaignEntryTerminate(&m->Campaign);
67 CA_FOREACH_END()
68 CArrayTerminate(&autosave->Missions);
69}

71static void LoadCampaignNode(CampaignEntry *c, json_t *node)
72{
73 const char *path = json_find_first_label(node, "Path")->child->text;
74 if (path != NULL)
75 {
76 CSTRDUP(c->Path, path);
77 }
78 c->Mode = GAME_MODE_NORMAL;
79}
80static void AddCampaignNode(CampaignEntry *c, json_t *root)
81{
82 json_t *subConfig = json_new_object();
83 // Save relative path so that save files are portable across installs
84 char path[CDOGS_PATH_MAX] = "";
85 RelPathFromCWD(path, c->Path);
86 json_insert_pair_into_object(
87 subConfig, "Path", json_new_string(path));
88 json_insert_pair_into_object(root, "Campaign", subConfig);
89}
90
91static void LoadMissionNode(MissionSave *m, json_t *node)
92{
93 MissionSaveInit(m);
94 LoadCampaignNode(&m->Campaign, json_find_first_label(node, "Campaign")->child);
95 strcpy(m->Password, json_find_first_label(node, "Password")->child->text);
96 LoadInt(&m->MissionsCompleted, node, "MissionsCompleted");
97 // Check that file exists
98 char buf[CDOGS_PATH_MAX];
99 GetDataFilePath(buf, m->Campaign.Path);
100 m->IsValid = access(buf, F_OK | R_OK) != -1;
101}
102static json_t *CreateMissionNode(MissionSave *m)
103{
104 json_t *subConfig = json_new_object();
105 AddCampaignNode(&m->Campaign, subConfig);
106 json_insert_pair_into_object(subConfig, "Password", json_new_string(m->Password));
107 AddIntPair(subConfig, "MissionsCompleted", m->MissionsCompleted);
108 return subConfig;
109}

111static void LoadMissionNodes(Autosave *a, json_t *root, const char *nodeName)
112{
113 json_t *child;
114 if (json_find_first_label(root, nodeName) == NULL)
115 {
116 return;
117 }
118 child = json_find_first_label(root, nodeName)->child->child;
119 while (child != NULL)
120 {
121 MissionSave m;
122 LoadMissionNode(&m, child);
123 AutosaveAddMission(a, &m);
124 child = child->next;
125 }
126}
127static void AddMissionNodes(Autosave *a, json_t *root, const char *nodeName)
128{
129 json_t *missions = json_new_array();
130 CA_FOREACH(MissionSave, m, a->Missions)
131 json_insert_child(missions, CreateMissionNode(m));
132 CA_FOREACH_END()
133 json_insert_pair_into_object(root, nodeName, missions);
134}

136void AutosaveLoad(Autosave *autosave, const char *filename)
137{
138 FILE *f = fopen(filename, "r");
139 json_t *root = NULL;
140
141 if (f == NULL)
142 {
143 printf("Error loading autosave '%s'\n", filename);
144 goto bail;
145 }
146
147 if (json_stream_parse(f, &root) != JSON_OK)
148 {
149 printf("Error parsing autosave '%s'\n", filename);
150 goto bail;
151 }
152 // Note: need to load missions before LastMission because the former
153 // will overwrite the latter, since AutosaveAddMission also
154 // writes to LastMission
155 LoadMissionNodes(autosave, root, "Missions");
156 if (json_find_first_label(root, "LastMission"))
157 {
158 LoadMissionNode(
159 &autosave->LastMission,
160 json_find_first_label(root, "LastMission")->child);
161 }
162
163bail:
164 json_free_value(&root);
165 if (f != NULL)
166 {
167 fclose(f);
168 }
169}

171void AutosaveSave(Autosave *autosave, const char *filename)
172{
173 FILE *f = fopen(filename, "w");
174 char *text = NULL;
175 json_t *root;
176
177 if (f == NULL)
178 {
179 printf("Error saving autosave '%s'\n", filename);
180 return;
181 }
182
183 root = json_new_object();
184 json_insert_pair_into_object(root, "Version", json_new_number("2"));
185 json_insert_pair_into_object(
186 root, "LastMission", CreateMissionNode(&autosave->LastMission));
187 AddMissionNodes(autosave, root, "Missions");
188
189 json_tree_to_string(root, &text);
190 char *formatText = json_format_string(text);
191 fputs(formatText, f);
192
193 // clean up
194 free(formatText);
195 free(text);
196 json_free_value(&root);
197
198 fclose(f);
199
200#ifdef __EMSCRIPTEN__
201 EM_ASM(
202 //persist changes
203 FS.syncfs(false,function (err) {
204 assert(!err);
205 });
206 );
207#endif
208}

210MissionSave *AutosaveFindMission(Autosave *autosave, const char *path)
211{
212 if (strlen(path) == 0)
213 {
214 return NULL;
215 }
216
217 // Turn the path into a relative one, since autosave paths are all stored
218 // in this form
219 char relPath[CDOGS_PATH_MAX] = "";
220 RelPathFromCWD(relPath, path);
221 CA_FOREACH(MissionSave, m, autosave->Missions)
222 const char *campaignPath = m->Campaign.Path;
223 if (campaignPath != NULL && strcmp(campaignPath, relPath) == 0)
224 {
225 return m;
226 }
227 CA_FOREACH_END()
228 return NULL;
229}

231void AutosaveAddMission(Autosave *autosave, MissionSave *mission)
232{
233 MissionSave *existingMission = AutosaveFindMission(
234 autosave, mission->Campaign.Path);
235 if (existingMission != NULL)
236 {
237 CampaignEntryTerminate(&existingMission->Campaign);
238 }
239 else
240 {
241 CArrayPushBack(&autosave->Missions, mission);
242 existingMission =
243 CArrayGet(&autosave->Missions, autosave->Missions.size - 1);
244 memset(existingMission, 0, sizeof *existingMission);
245 }
246 const int maxMissionsCompleted =
247 MAX(existingMission->MissionsCompleted, mission->MissionsCompleted);
248 memcpy(existingMission, mission, sizeof *existingMission);
249 existingMission->MissionsCompleted = maxMissionsCompleted;
250 CampaignEntryCopy(&existingMission->Campaign, &mission->Campaign);
251 CampaignEntryTerminate(&autosave->LastMission.Campaign);
252 memcpy(&autosave->LastMission, mission, sizeof autosave->LastMission);
253 CampaignEntryCopy(&autosave->LastMission.Campaign, &mission->Campaign);
254}
255
256void AutosaveLoadMission(
257 Autosave *autosave, MissionSave *mission, const char *path)
258{
259 MissionSave *existingMission = AutosaveFindMission(autosave, path);
260 MissionSaveInit(mission);
261 if (existingMission != NULL)
262 {
263 memcpy(mission, existingMission, sizeof *mission);
264 CampaignEntryCopy(&mission->Campaign, &existingMission->Campaign);
265 }
266}
