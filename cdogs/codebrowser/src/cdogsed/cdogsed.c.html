106static Vec2i GetMouseTile(EventHandlers *e)
107{
108 Mission *m = CampaignGetCurrentMission(&gCampaign);
109 if (!m)
110 {
111 return Vec2iNew(-1, -1);
112 }
113 else
114 {
115 return Vec2iNew(
116 (e->mouse.currentPos.x - sDrawBuffer.dx) / TILE_WIDTH + sDrawBuffer.xStart,
117 (e->mouse.currentPos.y - sDrawBuffer.dy) / TILE_HEIGHT + sDrawBuffer.yStart);
118 }
119}
120static Vec2i GetScreenPos(Vec2i mapTile)
121{
122 return Vec2iNew(
123 (mapTile.x - sDrawBuffer.xStart) * TILE_WIDTH + sDrawBuffer.dx,
124 (mapTile.y - sDrawBuffer.yStart) * TILE_HEIGHT + sDrawBuffer.dy);
125}

133static void MakeBackground(GraphicsDevice *g, const bool changedMission)
134{
135 if (changedMission)
136 {
137 // Automatically pan camera to middle of screen
138 Mission *m = gMission.missionData;
139 Vec2i focusTile = Vec2iScaleDiv(m->Size, 2);
140 // Better yet, if the map has a known start position, focus on that
141 if (m->Type == MAPTYPE_STATIC && !Vec2iIsZero(m->u.Static.Start))
142 {
143 focusTile = m->u.Static.Start;
144 }
145 camera = Vec2iCenterOfTile(focusTile);
146 }
147
148 // Clear background first
149 memset(g->buf, 0, GraphicsGetMemSize(&g->cachedConfig));
150 GrafxDrawExtra extra;
151 extra.guideImage = brush.GuideImageSurface;
152 extra.guideImageAlpha = brush.GuideImageAlpha;
153
154 DrawBufferTerminate(&sDrawBuffer);
155 DrawBufferInit(&sDrawBuffer, Vec2iNew(X_TILES, Y_TILES), &gGraphicsDevice);
156 GrafxMakeBackground(
157 g, &sDrawBuffer, &gCampaign, &gMission, &gMap,
158 tintNone, true, camera, &extra);
159}

170static void Display(GraphicsDevice *g, HandleInputResult result)
171{
172 char s[128];
173 int y = 5;
174 int w = g->cachedConfig.Res.x;
175 int h = g->cachedConfig.Res.y;
176 Mission *mission = CampaignGetCurrentMission(&gCampaign);
177
178 if (mission)
179 {
180 // Re-make the background if the resolution has changed
181 if (gEventHandlers.HasResolutionChanged)
182 {
183 MakeBackground(g, false);
184 }
185 if (result.RemakeBg || brush.IsGuideImageNew)
186 {
187 // Clear background first
188 memset(g->buf, 0, GraphicsGetMemSize(&g->cachedConfig));
189 brush.IsGuideImageNew = false;
190 GrafxDrawExtra extra;
191 extra.guideImage = brush.GuideImageSurface;
192 extra.guideImageAlpha = brush.GuideImageAlpha;
193 GrafxDrawBackground(g, &sDrawBuffer, tintNone, camera, &extra);
194 }
195 BlitClearBuf(g);
196
197 // Draw brush highlight tiles
198 if (brush.IsActive && IsBrushPosValid(brush.Pos, mission))
199 {
200 EditorBrushSetHighlightedTiles(&brush);
201 }
202 CA_FOREACH(Vec2i, pos, brush.HighlightedTiles)
203 Vec2i screenPos = GetScreenPos(*pos);
204 if (screenPos.x >= 0 && screenPos.x < w &&
205 screenPos.y >= 0 && screenPos.y < h)
206 {
207 DrawRectangle(
208 g, screenPos, TILE_SIZE, colorWhite, DRAW_FLAG_LINE);
209 }
210 CA_FOREACH_END()
211
212 if (brush.LastPos.x)
213 {
214 sprintf(s, "(%d, %d)", brush.Pos.x, brush.Pos.y);
215 FontStr(s, Vec2iNew(w - 40, h - 16));
216 }
217 }
218 else
219 {
220 ClearScreen(g);
221 }
222
223 if (fileChanged)
224 {
225 // Display a disk icon to show the game needs saving
226 const Pic *pic = PicManagerGetPic(&gPicManager, "disk1");
227 Blit(&gGraphicsDevice, pic, Vec2iNew(10, y));
228 }
229
230 FontStr("Press Ctrl+E to edit characters", Vec2iNew(20, h - 20 - FontH() * 2));
231 FontStr("Press F1 for help", Vec2iNew(20, h - 20 - FontH()));
232
233 UIObjectDraw(
234 sObjs, g, Vec2iZero(), gEventHandlers.mouse.currentPos, &sDrawObjs);
235
236 if (result.WillDisplayAutomap && mission)
237 {
238 AutomapDraw(AUTOMAP_FLAGS_SHOWALL, true);
239 }
240 else
241 {
242 if (sTooltipObj && sTooltipObj->Tooltip)
243 {
244 UITooltipDraw(
245 g, gEventHandlers.mouse.currentPos, sTooltipObj->Tooltip);
246 }
247 MouseDraw(&gEventHandlers.mouse);
248 }
249 BlitUpdateFromBuf(g, g->screen);
250 BlitFlip(g);
251}

256{

267 {
268 Setup(false);

314static void AdjustYC(int *yc)
315{
316 Mission *mission = CampaignGetCurrentMission(&gCampaign);
317 if (mission)
318 {
319 if (mission->Objectives.size)
320 {
321 *yc = CLAMP_OPPOSITE(
322 *yc, 0, YC_OBJECTIVES + (int)mission->Objectives.size - 1);

326 *yc = CLAMP_OPPOSITE(*yc, 0, YC_OBJECTIVES);
327 }

331static void AdjustXC(int yc, int *xc)
332{
333 Mission *mission = CampaignGetCurrentMission(&gCampaign);
334 switch (yc)
335 {
336 case YC_CAMPAIGNTITLE:
337 *xc = CLAMP_OPPOSITE(*xc, 0, XC_CAMPAIGNDESC);
338 break;
339
340 case YC_MISSIONTITLE:
341 *xc = CLAMP_OPPOSITE(*xc, 0, XC_MUSICFILE);
342 break;
343
344 case YC_MISSIONLOOKS:
345 *xc = CLAMP_OPPOSITE(*xc, 0, XC_COLOR4);
346 break;
347
348 case YC_CHARACTERS:
349 if (mission && mission->Enemies.size > 0)
350 {
351 *xc = CLAMP_OPPOSITE(*xc, 0, (int)mission->Enemies.size - 1);
352 }
353 break;
354
355 case YC_SPECIALS:
356 if (mission && mission->SpecialChars.size > 0)
357 {
358 *xc = CLAMP_OPPOSITE(*xc, 0, (int)mission->SpecialChars.size - 1);
359 }
360 break;
361
362 case YC_ITEMS:
363 if (mission && mission->MapObjectDensities.size > 0)
364 {
365 *xc = CLAMP_OPPOSITE(
366 *xc, 0, (int)mission->MapObjectDensities.size - 1);
367 }
368 break;
369
370 default:
371 break;
372 }
373}

375static void Autosave(void)
376{
377 if (fileChanged && sTicksElapsed > ticksAutosave)
378 {
379 ticksAutosave = sTicksElapsed + AUTOSAVE_INTERVAL_SECONDS * 1000;
380 char dirname[CDOGS_PATH_MAX];
381 PathGetDirname(dirname, lastFile);
382 char buf[CDOGS_PATH_MAX];
383 sprintf(
384 buf, "%s~%d%s", dirname, sAutosaveIndex, PathGetBasename(lastFile));
385 MapArchiveSave(buf, &gCampaign.Setting);
386 sAutosaveIndex++;
387 }
388}

391{
392 Mission *m = CampaignGetCurrentMission(&gCampaign);

421static bool TryOpen(const char *filename);
422static void ShowFailedToOpenMsg(const char *filename);
423static void Open(void)
424{
425 char filename[CDOGS_PATH_MAX];
426 strcpy(filename, lastFile);
427 bool done = false;
428 while (!done)
429 {
430 ClearScreen(&gGraphicsDevice);
431 const int x = 125;
432 Vec2i pos = Vec2iNew(x, 50);
433 FontStr("Open file:", pos);
434 pos.y += FontH();
435 pos = FontCh('>', pos);
436 pos = FontStr(filename, pos);
437 pos = FontCh('<', pos);
438
439 // Based on the filename, print a list of candidates
440 tinydir_dir dir;
441 char buf[CDOGS_PATH_MAX];
442 PathGetDirname(buf, filename);
443 char tabCompleteCandidate[CDOGS_PATH_MAX];
444 tabCompleteCandidate[0] = '\0';
445 if (!tinydir_open_sorted(&dir, buf))
446 {
447 int numCandidates = 0;
448 const char *basename = PathGetBasename(filename);
449 pos.x = x;
450 pos.y += FontH() * 2;
451 for (int i = 0; i < (int)dir.n_files; i++)
452 {
453 tinydir_file file;
454 tinydir_readfile_n(&dir, &file, i);
455 if (strncmp(file.name, basename, strlen(basename)) == 0)
456 {
457 // Ignore files that aren't campaigns or interesting folders
458 if (file.name[0] == '.') continue;
459 const bool canOpen =
460 strcmp(file.extension, "cdogscpn") == 0 ||
461 strcmp(file.extension, "CDOGSCPN") == 0 ||
462 strcmp(file.extension, "cpn") == 0 ||
463 strcmp(file.extension, "CPN") == 0;
464 if (!canOpen && !file.is_dir)
465 {
466 continue;
467 }
468 numCandidates++;
469 strcpy(tabCompleteCandidate, file.path);
470 const color_t c = canOpen ? colorCyan : colorGray;
471 pos = FontStrMask(file.path, pos, c);
472 if (!canOpen)
473 {
474 FontStrMask("/", pos, c);
475 strcat(tabCompleteCandidate, "/");
476 }
477 pos.x = x;
478 pos.y += FontH();
479 }
480 }
481 tinydir_close(&dir);
482
483 // See if there is only one candidate for tab-completion
484 if (numCandidates > 1)
485 {
486 tabCompleteCandidate[0] = '\0';
487 }
488 }
489
490 BlitUpdateFromBuf(&gGraphicsDevice, gGraphicsDevice.screen);
491 BlitFlip(&gGraphicsDevice);
492
493 bool doOpen = false;
494 const SDL_Scancode sc = EventWaitKeyOrText(&gEventHandlers);
495 switch (sc)
496 {
497 case SDL_SCANCODE_RETURN:
498 case SDL_SCANCODE_KP_ENTER:
499 if (!filename[0])
500 {
501 break;
502 }
503 doOpen = true;
504 break;
505
506 case SDL_SCANCODE_ESCAPE:
507 done = true;
508 break;
509
510 case SDL_SCANCODE_BACKSPACE:
511 if (filename[0])
512 filename[strlen(filename) - 1] = 0;
513 break;
514
515 case SDL_SCANCODE_TAB:
516 // tab completion - replace filename buffer with it
517 if (tabCompleteCandidate[0])
518 {
519 strcpy(filename, tabCompleteCandidate);
520 }
521 break;
522
523 default:
524 // Do nothing
525 break;
526 }
527 GetTextInput(filename);
528 if (doOpen)
529 {
530 ClearScreen(&gGraphicsDevice);
531
532 FontStrCenter("Loading...");
533
534 BlitUpdateFromBuf(&gGraphicsDevice, gGraphicsDevice.screen);
535 BlitFlip(&gGraphicsDevice);
536 // Try original filename
537 if (TryOpen(filename))
538 {
539 break;
540 }
541 // Try adding .cdogscpn
542 sprintf(buf, "%s.cdogscpn", filename);
543 if (TryOpen(buf))
544 {
545 break;
546 }
547 // Try adding .cpn
548 sprintf(buf, "%s.cpn", filename);
549 if (TryOpen(buf))
550 {
551 break;
552 }
553 // All attempts failed
554 ShowFailedToOpenMsg(filename);
555 }
556 SDL_Delay(10);
557 }
558}
559static bool TryOpen(const char *filename)
560{
561 // Try opening a campaign
562 CampaignSettingTerminate(&gCampaign.Setting);
563 CampaignSettingInit(&gCampaign.Setting);
564 char buf[CDOGS_PATH_MAX];
565 RealPath(filename, buf);
566 if (!MapNewLoad(buf, &gCampaign.Setting))
567 {
568 fileChanged = false;
569 Setup(true);
570 strcpy(lastFile, filename);
571 sAutosaveIndex = 0;
572 ReloadUI();
573 return true;
574 }
575 return false;
576}
577static void ShowFailedToOpenMsg(const char *filename)
578{
579 char msgBuf[CDOGS_PATH_MAX];
580 sprintf(msgBuf, "Failed to open file %s", filename);
581 SDL_ShowSimpleMessageBox(
582 SDL_MESSAGEBOX_ERROR, "Error", msgBuf, gGraphicsDevice.window);
583}

585static void Save(void)
586{
587 char filename[CDOGS_PATH_MAX];
588 strcpy(filename, lastFile);
589 bool doSave = false;
590 bool done = false;
591 while (!done)
592 {
593 ClearScreen(&gGraphicsDevice);
594 Vec2i pos = Vec2iNew(125, 50);
595 FontStr("Save as:", pos);
596 pos.y += FontH();
597 pos = FontCh('>', pos);
598 pos = FontStr(filename, pos);
599 FontCh('<', pos);
600 BlitUpdateFromBuf(&gGraphicsDevice, gGraphicsDevice.screen);
601 BlitFlip(&gGraphicsDevice);
602
603 const SDL_Scancode sc = EventWaitKeyOrText(&gEventHandlers);
604 switch (sc)
605 {
606 case SDL_SCANCODE_RETURN:
607 case SDL_SCANCODE_KP_ENTER:
608 if (!filename[0])
609 {
610 break;
611 }
612 doSave = true;
613 done = true;
614 break;
615
616 case SDL_SCANCODE_ESCAPE:
617 break;
618
619 case SDL_SCANCODE_BACKSPACE:
620 if (filename[0])
621 filename[strlen(filename) - 1] = 0;
622 break;
623
624 default:
625 // Do nothing
626 break;
627 }
628 GetTextInput(filename);
629 SDL_Delay(10);
630 }
631 if (doSave)
632 {
633 ClearScreen(&gGraphicsDevice);
634
635 FontStrCenter("Saving...");
636
637 BlitUpdateFromBuf(&gGraphicsDevice, gGraphicsDevice.screen);
638 BlitFlip(&gGraphicsDevice);
639 MapArchiveSave(filename, &gCampaign.Setting);
640 fileChanged = false;
641 strcpy(lastFile, filename);
642 sAutosaveIndex = 0;
643 char msgBuf[CDOGS_PATH_MAX];
644 sprintf(msgBuf, "Saved to %s", filename);
645 SDL_ShowSimpleMessageBox(
646 SDL_MESSAGEBOX_INFORMATION, "Campaign Saved",
647 msgBuf, gGraphicsDevice.window);
648 }
649}

651// Collect keyboard text input into buffer
652// Only support ASCII and limited characters
653static void GetTextInput(char *buf)
654{
655 // Get the filename typed, ASCII only
656 char *c = gEventHandlers.keyboard.Typed;
657 while (c && strlen(buf) < CDOGS_PATH_MAX - 1 && *c >= ' ' && *c <= '~')
658 {
659 // Prohibit some characters bad for filenames and/or not in font
660 if (*c != '*' &&
661 (strlen(buf) > 1 || *c != '-') &&
662 *c != ':' && *c != '<' && *c != '>' && *c != '?' &&
663 *c != '|')
664 {
665 const size_t si = strlen(buf);
666 buf[si + 1] = 0;
667 buf[si] = *c;
668 }
669 c++;
670 }
671}

707static void Delete(int xc, int yc)
708{
709 Mission *mission = CampaignGetCurrentMission(&gCampaign);
710 bool changedMission = false;
711 switch (yc)
712 {
713 case YC_CHARACTERS:
714 if (xc >= (int)mission->Enemies.size)
715 {
716 // Nothing to delete; do nothing
717 return;
718 }
719 DeleteCharacter(mission, xc);
720 break;
721
722 case YC_SPECIALS:
723 if (xc >= (int)mission->SpecialChars.size)
724 {
725 // Nothing to delete; do nothing
726 return;
727 }
728 DeleteSpecial(mission, xc);
729 break;
730
731 case YC_ITEMS:
732 DeleteItem(mission, xc);
733 break;
734
735 default:
736 if (yc >= YC_OBJECTIVES)
737 {
738 if (mission->Objectives.size == 0)
739 {
740 // Nothing to delete; do nothing
741 return;
742 }
743 DeleteObjective(mission, yc - YC_OBJECTIVES);
744 }
745 else
746 {
747 if (gCampaign.Setting.Missions.size == 0 ||
748 gCampaign.MissionIndex >= (int)gCampaign.Setting.Missions.size)
749 {
750 // Nothing to delete; do nothing
751 return;
752 }
753 DeleteMission(&gCampaign);
754 changedMission = true;
755 }
756 AdjustYC(&yc);
757 break;
758 }
759 fileChanged = true;
760 Setup(changedMission);
761}

763static void InputInsert(int *xc, const int yc, Mission *mission);
764static void InputDelete(const int xc, const int yc);
765static HandleInputResult HandleInput(
766 SDL_Scancode sc, const int m,
767 int *xc, int *yc, int *xcOld, int *ycOld, Mission *scrap)
768{
769 HandleInputResult result = { false, false, false, false };
770 Mission *mission = CampaignGetCurrentMission(&gCampaign);
771 UIObject *o = NULL;
772 const Vec2i brushLastDrawPos = brush.Pos;
773 brush.Pos = GetMouseTile(&gEventHandlers);
774 const bool shift = gEventHandlers.keyboard.modState & KMOD_SHIFT;
775
776 // Find whether the mouse has hovered over a tooltip
777 const bool hadTooltip = sTooltipObj != NULL;
778 bool isContextMenu = false;
779 sTooltipObj = NULL;
780 const Vec2i mousePos = gEventHandlers.mouse.currentPos;
781 UITryGetObject(sLastHighlightedObj, mousePos, &sTooltipObj);
782 if (sTooltipObj == NULL)
783 {
784 UITryGetObject(sObjs, mousePos, &sTooltipObj);
785 }
786 if (sTooltipObj != NULL)
787 {
788 if (sTooltipObj->Parent &&
789 sTooltipObj->Parent->Type == UITYPE_CONTEXT_MENU)
790 {
791 isContextMenu = true;
792 }
793 if (!sTooltipObj->Tooltip)
794 {
795 sTooltipObj = NULL;
796 }
797 }
798 // Need to redraw if we either had a tooltip (draw to remove) or there's a
799 // tooltip to draw
800 if (hadTooltip || sTooltipObj || isContextMenu)
801 {
802 result.Redraw = true;
803 }
804
805 // Make sure a redraw is done immediately if the resolution changes
806 // Otherwise the resolution change is ignored and we try to redraw
807 // later, when the draw buffer has not yet been recreated
808 if (gEventHandlers.HasResolutionChanged)
809 {
810 result.Redraw = true;
811 }
812
813 // Also need to redraw if the brush is active to update the highlight
814 if (brush.IsActive && !Vec2iEqual(brushLastDrawPos, brush.Pos))
815 {
816 result.Redraw = true;
817 }
818
819 if (m &&
820 (m == SDL_BUTTON_LEFT || m == SDL_BUTTON_RIGHT ||
821 MouseWheel(&gEventHandlers.mouse).y != 0))
822 {
823 result.Redraw = true;
824 if (sLastHighlightedObj && !sLastHighlightedObj->IsBackground)
825 {
826 UITryGetObject(sLastHighlightedObj, mousePos, &o);
827 }
828 if (o == NULL)
829 {
830 UITryGetObject(sObjs, mousePos, &o);
831 }
832 if (o != NULL)
833 {
834 if (!o->DoNotHighlight)
835 {
836 if (sLastHighlightedObj)
837 {
838 if (UIObjectUnhighlight(sLastHighlightedObj, true))
839 {
840 Setup(false);
841 }
842 }
843 sLastHighlightedObj = o;
844 UIObjectHighlight(o, shift);
845 sIgnoreMouse = true;
846 }
847 CArrayTerminate(&sDrawObjs);
848 *xcOld = *xc;
849 *ycOld = *yc;
850 // Only change selection on left/right click
851 if (m == SDL_BUTTON_LEFT || m == SDL_BUTTON_RIGHT)
852 {
853 if (!(o->Flags & UI_LEAVE_YC))
854 {
855 *yc = o->Id;
856 AdjustYC(yc);
857 }
858 if (!(o->Flags & UI_LEAVE_XC))
859 {
860 *xc = o->Id2;
861 AdjustXC(*yc, xc);
862 }
863 }
864 if (!(o->Flags & UI_SELECT_ONLY) &&
865 (!(o->Flags & UI_SELECT_ONLY_FIRST) || (*xc == *xcOld && *yc == *ycOld)))
866 {
867 if (m == SDL_BUTTON_LEFT ||
868 MouseWheel(&gEventHandlers.mouse).y > 0)
869 {
870 sc = SDL_SCANCODE_PAGEUP;
871 }
872 else if (m == SDL_BUTTON_RIGHT ||
873 MouseWheel(&gEventHandlers.mouse).y < 0)
874 {
875 sc = SDL_SCANCODE_PAGEDOWN;
876 }
877 }
878 }
879 else
880 {
881 if (!(brush.IsActive && mission))
882 {
883 UIObjectUnhighlight(sObjs, true);
884 CArrayTerminate(&sDrawObjs);
885 sLastHighlightedObj = NULL;
886 }
887 }
888 }
889 if (!o &&
890 (MouseIsDown(&gEventHandlers.mouse, SDL_BUTTON_LEFT) ||
891 MouseIsDown(&gEventHandlers.mouse, SDL_BUTTON_RIGHT)) &&
892 !sIgnoreMouse)
893 {
894 result.Redraw = true;
895 if (brush.IsActive && mission->Type == MAPTYPE_STATIC)
896 {
897 // Draw a tile
898 if (IsBrushPosValid(brush.Pos, mission))
899 {
900 const bool isMain =
901 MouseIsDown(&gEventHandlers.mouse, SDL_BUTTON_LEFT);
902 const EditorResult r =
903 EditorBrushStartPainting(&brush, mission, isMain);
904 if (r & EDITOR_RESULT_CHANGED)
905 {
906 fileChanged = true;
907 Autosave();
908 result.RemakeBg = true;
909 sHasUnbakedChanges = true;
910 }
911 if (r & EDITOR_RESULT_RELOAD)
912 {
913 Setup(false);
914 }
915 }
916 }
917 }
918 else
919 {
920 if (mission)
921 {
922 // Clamp brush position
923 brush.Pos = Vec2iClamp(
924 brush.Pos,
925 Vec2iZero(), Vec2iMinus(mission->Size, Vec2iUnit()));
926 const EditorResult r = EditorBrushStopPainting(&brush, mission);
927 if (r & EDITOR_RESULT_CHANGED)
928 {
929 fileChanged = true;
930 Autosave();
931 result.Redraw = true;
932 result.RemakeBg = true;
933 sHasUnbakedChanges = true;
934 }
935 if (r & EDITOR_RESULT_RELOAD)
936 {
937 Setup(false);
938 }
939 }
940 }
941 // Pan the camera based on keyboard cursor keys
942 if (mission)
943 {
944 if (KeyIsDown(&gEventHandlers.keyboard, SDL_SCANCODE_LEFT))
945 {
946 camera.x -= CAMERA_PAN_SPEED;
947 result.Redraw = result.RemakeBg = true;
948 }
949 else if (KeyIsDown(&gEventHandlers.keyboard, SDL_SCANCODE_RIGHT))
950 {
951 camera.x += CAMERA_PAN_SPEED;
952 result.Redraw = result.RemakeBg = true;
953 }
954 if (KeyIsDown(&gEventHandlers.keyboard, SDL_SCANCODE_UP))
955 {
956 camera.y -= CAMERA_PAN_SPEED;
957 result.Redraw = result.RemakeBg = true;
958 }
959 else if (KeyIsDown(&gEventHandlers.keyboard, SDL_SCANCODE_DOWN))
960 {
961 camera.y += CAMERA_PAN_SPEED;
962 result.Redraw = result.RemakeBg = true;
963 }
964 // Also pan the camera based on middle mouse drag
965 if (MouseIsDown(&gEventHandlers.mouse, SDL_BUTTON_MIDDLE))
966 {
967 camera = Vec2iAdd(camera, Vec2iMinus(
968 gEventHandlers.mouse.previousPos,
969 gEventHandlers.mouse.currentPos));
970 result.Redraw = result.RemakeBg = true;
971 }
972 camera.x = CLAMP(camera.x, 0, Vec2iCenterOfTile(mission->Size).x);
973 camera.y = CLAMP(camera.y, 0, Vec2iCenterOfTile(mission->Size).y);
974 }
975 bool hasQuit = false;
976 if (sc != SDL_SCANCODE_UNKNOWN)
977 {
978 result.Redraw = true;
979 }
980 if (gEventHandlers.keyboard.modState & (KMOD_ALT | KMOD_CTRL))
981 {
982 const SDL_Keycode kc = SDL_GetKeyFromScancode(sc);
983 switch (kc)
984 {
985 case 'z':
986 // Undo
987 // Do this by swapping the current mission with the last mission
988 // This requires a bit of copy-acrobatics; because missions
989 // are saved in Setup(), but by this stage the mission has already
990 // changed, _two_ mission caches are used, copied in sequence.
991 // That is, if the current mission is at state B, the first cache
992 // is still at state B (copied after the mission has changed
993 // already), and the second cache is at state A.
994 // If we were to perform an undo and still maintain functionality,
995 // we need to copy such that the states change from B,B,A to
996 // A,A,B.
997
998 // However! The above is true only if we have "baked" changes
999 // The editor has been optimised to perform some changes
1000 // without reloading map files; that is, the files are actually
1001 // in states C,B,A.
1002 // In this case, another set of "acrobatics" is required
1003 if (sHasUnbakedChanges)
1004 {
1005 MissionCopy(&lastMission, mission); // B,A,Z -> B,A,B
1006 }
1007 else
1008 {
1009 MissionCopy(mission, &lastMission); // B,B,A -> A,B,A
1010 MissionCopy(&lastMission, &currentMission); // A,B,A -> A,B,B
1011 }
1012 fileChanged = true;
1013 Setup(false); // A,B,B -> A,A,B
1014 break;
1015
1016 case 'x':
1017 MissionTerminate(scrap);
1018 MissionCopy(scrap, mission);
1019 Delete(*xc, *yc);
1020 // Unhighlight everything (in case this is the last mission)
1021 UIObjectUnhighlight(sObjs, true);
1022 CArrayTerminate(&sDrawObjs);
1023 sLastHighlightedObj = NULL;
1024 break;
1025
1026 case 'c':
1027 MissionTerminate(scrap);
1028 MissionCopy(scrap, mission);
1029 break;
1030
1031 case 'v':
1032 // Use map size as a proxy to whether there's a valid scrap mission
1033 if (!Vec2iIsZero(scrap->Size))
1034 {
1035 InsertMission(&gCampaign, scrap, gCampaign.MissionIndex);
1036 fileChanged = true;
1037 Setup(false);
1038 }
1039 break;
1040
1041 case 'i':
1042 InputInsert(xc, *yc, mission);
1043 break;
1044
1045 case 'd':
1046 InputDelete(*xc, *yc);
1047 break;
1048
1049 case 'q':
1050 hasQuit = true;
1051 break;
1052
1053 case 'n':
1054 InsertMission(&gCampaign, NULL, gCampaign.Setting.Missions.size);
1055 gCampaign.MissionIndex = gCampaign.Setting.Missions.size - 1;
1056 fileChanged = true;
1057 Setup(true);
1058 break;
1059
1060 case 'o':
1061 if (!fileChanged || ConfirmScreen(
1062 "File has been modified, but not saved", "Open anyway? (Y/N)"))
1063 {
1064 Open();
1065 }
1066 break;
1067
1068 case 's':
1069 Save();
1070 break;
1071
1072 case 'm':
1073 result.WillDisplayAutomap = true;
1074 break;
1075
1076 case 'e':
1077 CharEditor(
1078 &gGraphicsDevice, &gCampaign.Setting, &gEventHandlers,
1079 &fileChanged);
1080 Setup(false);
1081 UIObjectUnhighlight(sObjs, true);
1082 CArrayTerminate(&sDrawObjs);
1083 break;
1084
1085 default:
1086 // do nothing
1087 break;
1088 }
1089 }
1090 else
1091 {
1092 switch (sc)
1093 {
1094 case SDL_SCANCODE_F1:
1095 HelpScreen();
1096 break;
1097
1098 case SDL_SCANCODE_HOME:
1099 if (gCampaign.MissionIndex > 0)
1100 {
1101 gCampaign.MissionIndex--;
1102 }
1103 Setup(true);
1104 break;
1105
1106 case SDL_SCANCODE_END:
1107 if (gCampaign.MissionIndex < (int)gCampaign.Setting.Missions.size)
1108 {
1109 gCampaign.MissionIndex++;
1110 }
1111 Setup(true);
1112 break;
1113
1114 case SDL_SCANCODE_INSERT:
1115 InputInsert(xc, *yc, mission);
1116 break;
1117
1118 case SDL_SCANCODE_DELETE:
1119 InputDelete(*xc, *yc);
1120 break;
1121
1122 case SDL_SCANCODE_PAGEUP:
1123 Change(o, 1, shift);
1124 break;
1125
1126 case SDL_SCANCODE_PAGEDOWN:
1127 Change(o, -1, shift);
1128 break;
1129
1130 case SDL_SCANCODE_ESCAPE:
1131 hasQuit = true;
1132 break;
1133
1134 case SDL_SCANCODE_BACKSPACE:
1135 fileChanged = UIObjectDelChar(sObjs) || fileChanged;
1136 break;
1137
1138 default:
1139 // Do nothing
1140 break;
1141 }
1142 // Get text input, ASCII only
1143 char *c = gEventHandlers.keyboard.Typed;
1144 while (c && *c >= ' ' && *c <= '~')
1145 {
1146 fileChanged = UIObjectAddChar(sObjs, *c) || fileChanged;
1147 c++;
1148 }
1149 }
1150 if (gEventHandlers.DropFile != NULL)
1151 {
1152 // Copy to buf because ConfirmScreen will cause the DropFile to be lost
1153 char buf[CDOGS_PATH_MAX];
1154 strcpy(buf, gEventHandlers.DropFile);
1155 if (!fileChanged || ConfirmScreen(
1156 "File has been modified, but not saved", "Open anyway? (Y/N)"))
1157 {
1158 if (!TryOpen(buf))
1159 {
1160 ShowFailedToOpenMsg(buf);
1161 }
1162 }
1163 }
1164 if (gEventHandlers.HasQuit)
1165 {
1166 hasQuit = true;
1167 // Don't let the program quit yet; wait for confirmation screen
1168 gEventHandlers.HasQuit = false;
1169 }
1170 if (hasQuit)
1171 {
1172 // Make sure we redraw so if the user has cancelled the quit confirm
1173 // the editor reappears
1174 result.Redraw = true;
1175 result.Done = !fileChanged || ConfirmScreen(
1176 "File has been modified, but not saved", "Quit anyway? (Y/N)");
1177 }
1178 if (!MouseIsDown(&gEventHandlers.mouse, SDL_BUTTON_LEFT) &&
1179 !MouseIsDown(&gEventHandlers.mouse, SDL_BUTTON_RIGHT))
1180 {
1181 sIgnoreMouse = false;
1182 }
1183 return result;
1184}
1185static void InputInsert(int *xc, const int yc, Mission *mission)
1186{
1187 bool changedMission = false;
1188 switch (yc)
1189 {
1190 case YC_CHARACTERS:
1191 if (gCampaign.Setting.characters.OtherChars.size > 0)
1192 {
1193 int ch = 0;
1194 CArrayPushBack(&mission->Enemies, &ch);
1195 CharacterStoreAddBaddie(&gCampaign.Setting.characters, ch);
1196 *xc = mission->Enemies.size - 1;
1197 }
1198 break;
1199
1200 case YC_SPECIALS:
1201 if (gCampaign.Setting.characters.OtherChars.size > 0)
1202 {
1203 int ch = 0;
1204 CArrayPushBack(&mission->SpecialChars, &ch);
1205 CharacterStoreAddSpecial(&gCampaign.Setting.characters, ch);
1206 *xc = mission->SpecialChars.size - 1;
1207 }
1208 break;
1209
1210 case YC_ITEMS:
1211 {
1212 MapObjectDensity mod;
1213 mod.M = IndexMapObject(0);
1214 mod.Density = 0;
1215 CArrayPushBack(&mission->MapObjectDensities, &mod);
1216 *xc = mission->MapObjectDensities.size - 1;
1217 }
1218 break;
1219
1220 default:
1221 if (yc >= YC_OBJECTIVES)
1222 {
1223 AddObjective(mission);
1224 }
1225 else
1226 {
1227 InsertMission(&gCampaign, NULL, gCampaign.MissionIndex);
1228 changedMission = true;
1229 }
1230 break;
1231 }
1232 fileChanged = true;
1233 Setup(changedMission);
1234}
1235static void InputDelete(const int xc, const int yc)
1236{
1237 Delete(xc, yc);
1238 // Unhighlight everything (in case this is the last mission)
1239 UIObjectUnhighlight(sObjs, true);
1240 CArrayTerminate(&sDrawObjs);
1241 sLastHighlightedObj = NULL;
1242}

1244static void EditCampaign(void)
1245{
1246 int xc = 0, yc = 0;
1247 int xcOld, ycOld;
1248 Mission scrap;
1249 memset(&scrap, 0, sizeof scrap);
1250
1251 Setup(true);
1252
1253 Uint32 ticksNow = SDL_GetTicks();
1254 sTicksElapsed = 0;
1255 ticksAutosave = AUTOSAVE_INTERVAL_SECONDS * 1000;
1256 SDL_StartTextInput();
1257 for (;;)
1258 {
1259 Uint32 ticksThen = ticksNow;
1260 ticksNow = SDL_GetTicks();
1261 sTicksElapsed += ticksNow - ticksThen;
1262 if (sTicksElapsed < 1000 / FPS_FRAMELIMIT * 2)
1263 {
1264 SDL_Delay(1);
1265 continue;
1266 }
1267
1268 EventPoll(&gEventHandlers, SDL_GetTicks());
1269 const SDL_Scancode sc = KeyGetPressed(&gEventHandlers.keyboard);
1270 const int m = MouseGetPressed(&gEventHandlers.mouse);
1271
1272 HandleInputResult result = HandleInput(
1273 sc, m, &xc, &yc, &xcOld, &ycOld, &scrap);
1274 if (result.Done)
1275 {
1276 break;
1277 }
1278 if (result.Redraw || result.RemakeBg || sJustLoaded)
1279 {
1280 sJustLoaded = false;
1281 LOG(LM_EDIT, LL_TRACE, "Drawing UI");
1282 Display(&gGraphicsDevice, result);
1283 if (result.WillDisplayAutomap)
1284 {
1285 GetKey(&gEventHandlers);
1286 }
1287 }
1288 sTicksElapsed -= 1000 / (FPS_FRAMELIMIT * 2);
1289 }
1290 SDL_StopTextInput();
1291}
