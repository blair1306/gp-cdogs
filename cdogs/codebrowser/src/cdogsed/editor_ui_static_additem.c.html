81static void DrawPickupSpawner(
82 UIObject *o, GraphicsDevice *g, Vec2i pos, void *vData)
83{
84 const IndexedEditorBrush *data = vData;
85 const MapObject *mo = data->u.MapObject;
86 DisplayMapItem(
87 Vec2iAdd(Vec2iAdd(pos, o->Pos), Vec2iScaleDiv(o->Size, 2)), mo);
88 const Pic *pic = mo->u.PickupClass->Pic;
89 pos = Vec2iMinus(pos, Vec2iScaleDiv(pic->size, 2));
90 Blit(g, pic, Vec2iAdd(Vec2iAdd(pos, o->Pos), Vec2iScaleDiv(o->Size, 2)));
91}
92static void DrawCharacter(
93 UIObject *o, GraphicsDevice *g, Vec2i pos, void *vData)
94{
95 UNUSED(g);
96 EditorBrushAndCampaign *data = vData;
97 CharacterStore *store = &data->Campaign->Setting.characters;
98 Character *c = CArrayGet(&store->OtherChars, data->Brush.u.ItemIndex);
99 DrawCharacterSimple(
100 c,
101 Vec2iAdd(Vec2iAdd(pos, o->Pos), Vec2iScaleDiv(o->Size, 2)),
102 DIRECTION_DOWN, false, false);
103}
104static void DrawObjective(
105 UIObject *o, GraphicsDevice *g, Vec2i pos, void *vData)
106{
107 UNUSED(g);
108 EditorBrushAndCampaign *data = vData;
109 Mission *m = CampaignGetCurrentMission(data->Campaign);
110 const Objective *obj = CArrayGet(&m->Objectives, data->Brush.u.ItemIndex);
111 if (obj == NULL)
112 {
113 return;
114 }
115 CharacterStore *store = &data->Campaign->Setting.characters;
116 pos = Vec2iAdd(Vec2iAdd(pos, o->Pos), Vec2iScaleDiv(o->Size, 2));
117 switch (obj->Type)
118 {
119 case OBJECTIVE_KILL:
120 {
121 Character *c = CArrayGet(
122 &store->OtherChars,
123 CharacterStoreGetSpecialId(store, data->Brush.Index2));
124 DrawCharacterSimple(c, pos, DIRECTION_DOWN, false, false);
125 }
126 break;
127 case OBJECTIVE_RESCUE:
128 {
129 Character *c = CArrayGet(
130 &store->OtherChars,
131 CharacterStoreGetPrisonerId(store, data->Brush.Index2));
132 DrawCharacterSimple(c, pos, DIRECTION_DOWN, false, false);
133 }
134 break;
135 case OBJECTIVE_COLLECT:
136 {
137 const Pic *p = obj->u.Pickup->Pic;
138 pos = Vec2iMinus(pos, Vec2iScaleDiv(p->size, 2));
139 Blit(&gGraphicsDevice, p, pos);
140 }
141 break;
142 case OBJECTIVE_DESTROY:
143 DisplayMapItem(pos, obj->u.MapObject);
144 break;
145 default:
146 assert(0 && "invalid objective type");
147 break;
148 }
149}

196UIObject *CreateAddItemObjs(
197 Vec2i pos, EditorBrush *brush, CampaignOptions *co)
198{
199 const int th = FontH();
200 UIObject *o2;
201 UIObject *c = UIObjectCreate(UITYPE_CONTEXT_MENU, 0, pos, Vec2iZero());
202
203 UIObject *o = UIObjectCreate(
204 UITYPE_LABEL, 0, Vec2iZero(), Vec2iNew(65, th));
205 o->Data = brush;
206
207 pos = Vec2iZero();
208 o2 = UIObjectCopy(o);
209 o2->Label = "Player start";
210 o2->ChangeFunc = BrushSetBrushTypeSetPlayerStart;
211 o2->Pos = pos;
212 CSTRDUP(o2->Tooltip, "Location where players start");
213 o2->OnFocusFunc = ActivateBrush;
214 o2->OnUnfocusFunc = DeactivateBrush;
215 o2->Data = brush;
216 UIObjectAddChild(c, o2);
217 pos.y += th;
218 o2 = UIObjectCopy(o);
219 o2->Label = "Map item >";
220 o2->Pos = pos;
221 UIObjectAddChild(o2, CreateAddMapItemObjs(
222 o2->Size, AddMapItemBrushObjFunc, brush, sizeof(IndexedEditorBrush),
223 true));
224 UIObjectAddChild(c, o2);
225 pos.y += th;
226 o2 = UIObjectCopy(o);
227 o2->Label = "Pickup spawner >";
228 o2->Pos = pos;
229 UIObjectAddChild(o2, CreateAddPickupSpawnerObjs(
230 o2->Size, AddPickupSpawnerBrushObjFunc, brush,
231 sizeof(IndexedEditorBrush)));
232 UIObjectAddChild(c, o2);
233 pos.y += th;
234 o2 = UIObjectCopy(o);
235 o2->Label = "Character >";
236 o2->Pos = pos;
237 UIObjectAddChild(o2, CreateAddCharacterObjs(o2->Size, brush, co));
238 UIObjectAddChild(c, o2);
239 pos.y += th;
240 o2 = UIObjectCopy(o);
241 o2->Label = "Objective >";
242 o2->Pos = pos;
243 UIObjectAddChild(o2, CreateAddObjectiveObjs(o2->Size, brush, co));
244 UIObjectAddChild(c, o2);
245 pos.y += th;
246 o2 = UIObjectCopy(o);
247 o2->Label = "Key >";
248 o2->Pos = pos;
249 UIObjectAddChild(o2, CreateAddKeyObjs(o2->Size, brush));
250 UIObjectAddChild(c, o2);
251
252 UIObjectDestroy(o);
253 return c;
254}

259static bool AddMapItemBrushObjFunc(UIObject *o, MapObject *mo, void *vData)
260{
261 if (mo->Type != MAP_OBJECT_TYPE_NORMAL)
262 {
263 return false;
264 }
265 o->ChangeFunc = BrushSetBrushTypeAddMapItem;
266 o->u.CustomDrawFunc = DrawMapItem;
267 o->OnFocusFunc = ActivateIndexedEditorBrush;
268 o->OnUnfocusFunc = DeactivateIndexedEditorBrush;
269 ((IndexedEditorBrush *)o->Data)->Brush = vData;
270 ((IndexedEditorBrush *)o->Data)->u.MapObject = mo;
271 o->Tooltip = MakeMapObjectTooltip(mo);
272 return true;
273}

283static char *MakeMapObjectTooltip(const MapObject *mo)
284{
285 // Add a descriptive tooltip for the map object
286 char buf[512];
287 // Construct text representing explosion guns
288 char exBuf[256];
289 strcpy(exBuf, "");
290 if (mo->DestroyGuns.size > 0)
291 {
292 sprintf(exBuf, "\nExplodes: ");
293 for (int i = 0; i < (int)mo->DestroyGuns.size; i++)
294 {
295 if (i > 0)
296 {
297 strcat(exBuf, ", ");
298 }
299 const GunDescription **g = CArrayGet(&mo->DestroyGuns, i);
300 strcat(exBuf, (*g)->name);
301 }
302 }
303 sprintf(buf, "%s\nHealth: %d%s", mo->Name, mo->Health, exBuf);
304 char *tmp;
305 CSTRDUP(tmp, buf);
306 return tmp;
307}

334static UIObject *CreateAddCharacterObjs(
335 Vec2i pos, EditorBrush *brush, CampaignOptions *co)
336{
337 UIObject *c = UIObjectCreate(UITYPE_CONTEXT_MENU, 0, pos, Vec2iZero());

348static void CreateAddCharacterSubObjs(UIObject *c, void *vData)
349{
350 EditorBrushAndCampaign *data = vData;
351 CharacterStore *store = &data->Campaign->Setting.characters;
352 if (c->Children.size == store->OtherChars.size)
353 {
354 return;
355 }
356 // Recreate the child UI objects
357 c->Highlighted = NULL;
358 UIObject **objs = c->Children.data;
359 for (int i = 0; i < (int)c->Children.size; i++, objs++)
360 {
361 UIObjectDestroy(*objs);
362 }
363 CArrayTerminate(&c->Children);
364 CArrayInit(&c->Children, sizeof c);
365
366 UIObject *o = UIObjectCreate(
367 UITYPE_CUSTOM, 0,
368 Vec2iZero(), Vec2iNew(TILE_WIDTH + 4, TILE_HEIGHT * 2 + 4));
369 o->ChangeFunc = BrushSetBrushTypeAddCharacter;
370 o->u.CustomDrawFunc = DrawCharacter;
371 o->OnFocusFunc = ActivateEditorBrushAndCampaignBrush;
372 o->OnUnfocusFunc = DeactivateEditorBrushAndCampaignBrush;
373 Vec2i pos = Vec2iZero();
374 int width = 8;
375 for (int i = 0; i < (int)store->OtherChars.size; i++)
376 {
377 UIObject *o2 = UIObjectCopy(o);
378 Character *ch = CArrayGet(&store->OtherChars, i);
379 CSTRDUP(o2->Tooltip, ch->Gun->name);
380 o2->IsDynamicData = 1;
381 CMALLOC(o2->Data, sizeof(EditorBrushAndCampaign));
382 ((EditorBrushAndCampaign *)o2->Data)->Brush.Brush = data->Brush.Brush;
383 ((EditorBrushAndCampaign *)o2->Data)->Campaign = data->Campaign;
384 ((EditorBrushAndCampaign *)o2->Data)->Brush.u.ItemIndex = i;
385 o2->Pos = pos;
386 UIObjectAddChild(c, o2);
387 pos.x += o->Size.x;
388 if (((i + 1) % width) == 0)
389 {
390 pos.x = 0;
391 pos.y += o->Size.y;
392 }
393 }
394 UIObjectDestroy(o);
395}

413static void CreateAddObjectiveSubObjs(UIObject *c, void *vData)
414{
415 EditorBrushAndCampaign *data = vData;
416 Mission *m = CampaignGetCurrentMission(data->Campaign);
417 // Check if the data is still the same; if so don't recreate the
418 // child UI objects
419 // This is because during the course of UI operations, this element
420 // could be highlighted again; if we recreate then we invalidate
421 // UI pointers.
422 bool needToRecreate = false;
423 int childIndex = 0;
424 CA_FOREACH(const Objective, obj, m->Objectives)
425 int secondaryCount = 1;
426 const CharacterStore *store = &data->Campaign->Setting.characters;
427 switch (obj->Type)
428 {
429 case OBJECTIVE_KILL:
430 secondaryCount = (int)store->specialIds.size;
431 break;
432 case OBJECTIVE_COLLECT:
433 break;
434 case OBJECTIVE_DESTROY:
435 break;
436 case OBJECTIVE_RESCUE:
437 secondaryCount = (int)store->prisonerIds.size;
438 break;
439 default:
440 continue;
441 }
442 for (int j = 0; j < (int)secondaryCount; j++)
443 {
444 if ((int)c->Children.size <= childIndex)
445 {
446 needToRecreate = true;
447 break;
448 }
449 UIObject *o2 = *(UIObject **)CArrayGet(&c->Children, childIndex);
450 if (((EditorBrushAndCampaign *)o2->Data)->Brush.u.ItemIndex != _ca_index ||
451 ((EditorBrushAndCampaign *)o2->Data)->Brush.Index2 != j)
452 {
453 needToRecreate = true;
454 break;
455 }
456 childIndex++;
457 }
458 if (needToRecreate)
459 {
460 break;
461 }
462 CA_FOREACH_END()
463 if (!needToRecreate)
464 {
465 return;
466 }
467
468 // Recreate the child UI objects
469 c->Highlighted = NULL;
470 CA_FOREACH(UIObject *, obj, c->Children)
471 UIObjectDestroy(*obj);
472 CA_FOREACH_END()
473 CArrayTerminate(&c->Children);
474 CArrayInit(&c->Children, sizeof c);
475
476 UIObject *o = UIObjectCreate(
477 UITYPE_CUSTOM, 0,
478 Vec2iZero(), Vec2iNew(TILE_WIDTH + 4, TILE_HEIGHT * 2 + 4));
479 o->ChangeFunc = BrushSetBrushTypeAddObjective;
480 o->u.CustomDrawFunc = DrawObjective;
481 o->OnFocusFunc = ActivateEditorBrushAndCampaignBrush;
482 o->OnUnfocusFunc = DeactivateEditorBrushAndCampaignBrush;
483 Vec2i pos = Vec2iZero();
484 CA_FOREACH(const Objective, obj, m->Objectives)
485 int secondaryCount = 1;
486 const CharacterStore *store = &data->Campaign->Setting.characters;
487 switch (obj->Type)
488 {
489 case OBJECTIVE_KILL:
490 secondaryCount = (int)store->specialIds.size;
491 o->Size.y = TILE_HEIGHT * 2 + 4;
492 break;
493 case OBJECTIVE_COLLECT:
494 o->Size.y = TILE_HEIGHT + 4;
495 break;
496 case OBJECTIVE_DESTROY:
497 o->Size.y = TILE_HEIGHT * 2 + 4;
498 break;
499 case OBJECTIVE_RESCUE:
500 secondaryCount = (int)store->prisonerIds.size;
501 o->Size.y = TILE_HEIGHT * 2 + 4;
502 break;
503 default:
504 continue;
505 }
506 for (int j = 0; j < (int)secondaryCount; j++)
507 {
508 UIObject *o2 = UIObjectCopy(o);
509 CSTRDUP(o2->Tooltip, ObjectiveTypeStr(obj->Type));
510 o2->IsDynamicData = true;
511 CMALLOC(o2->Data, sizeof(EditorBrushAndCampaign));
512 ((EditorBrushAndCampaign *)o2->Data)->Brush.Brush =
513 data->Brush.Brush;
514 ((EditorBrushAndCampaign *)o2->Data)->Campaign = data->Campaign;
515 ((EditorBrushAndCampaign *)o2->Data)->Brush.u.ItemIndex = _ca_index;
516 ((EditorBrushAndCampaign *)o2->Data)->Brush.Index2 = j;
517 o2->Pos = pos;
518 UIObjectAddChild(c, o2);
519 pos.x += o->Size.x;
520 }
521 pos.x = 0;
522 pos.y += o->Size.y;
523 CA_FOREACH_END()
524 UIObjectDestroy(o);
525}
526static UIObject *CreateAddKeyObjs(Vec2i pos, EditorBrush *brush)
527{
528 UIObject *o2;
529 UIObject *c = UIObjectCreate(UITYPE_CONTEXT_MENU, 0, pos, Vec2iZero());
530
531 UIObject *o = UIObjectCreate(
532 UITYPE_CUSTOM, 0,
533 Vec2iZero(), Vec2iNew(TILE_WIDTH + 4, TILE_HEIGHT + 4));
534 o->ChangeFunc = BrushSetBrushTypeAddKey;
535 o->u.CustomDrawFunc = DrawKey;
536 o->OnFocusFunc = ActivateIndexedEditorBrush;
537 o->OnUnfocusFunc = DeactivateIndexedEditorBrush;
538 pos = Vec2iZero();
539 int width = 4;
540 for (int i = 0; i < KEY_COUNT; i++)
541 {
542 o2 = UIObjectCopy(o);
543 o2->IsDynamicData = 1;
544 CMALLOC(o2->Data, sizeof(IndexedEditorBrush));
545 ((IndexedEditorBrush *)o2->Data)->Brush = brush;
546 ((IndexedEditorBrush *)o2->Data)->u.ItemIndex = i;
547 o2->Pos = pos;
548 UIObjectAddChild(c, o2);
549 pos.x += o->Size.x;
550 if (((i + 1) % width) == 0)
551 {
552 pos.x = 0;
553 pos.y += o->Size.y;
554 }
555 }
556
557 UIObjectDestroy(o);
558 return c;
