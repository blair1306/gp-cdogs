38void EditorBrushInit(EditorBrush *b)
39{
40 memset(b, 0, sizeof *b);
41 b->Type = BRUSHTYPE_POINT;
42 b->MainType = MAP_WALL;
43 b->SecondaryType = MAP_FLOOR;
44 b->PaintType = b->MainType;
45 b->IsActive = 0;
46 b->IsPainting = 0;
47 b->BrushSize = 1;
48 b->LastPos = Vec2iNew(-1, -1);
49 b->Pos = Vec2iNew(-1, -1);
50 b->GuideImageAlpha = 64;
51 CArrayInit(&b->HighlightedTiles, sizeof(Vec2i));
52}
53void EditorBrushTerminate(EditorBrush *b)
54{
55 CArrayTerminate(&b->HighlightedTiles);
56 SDL_FreeSurface(b->GuideImageSurface);
57}

59static void EditorBrushHighlightPoint(void *data, Vec2i p)
60{
61 EditorBrush *b = data;
62 Vec2i v;
63 for (v.y = 0; v.y < b->BrushSize; v.y++)
64 {
65 for (v.x = 0; v.x < b->BrushSize; v.x++)
66 {
67 Vec2i pos = Vec2iAdd(p, v);
68 CArrayPushBack(&b->HighlightedTiles, &pos);
69 }
70 }
71}
72void EditorBrushSetHighlightedTiles(EditorBrush *b)
73{
74 bool useSimpleHighlight = true;
75 switch (b->Type)
76 {
77 case BRUSHTYPE_POINT: // fallthrough
78 case BRUSHTYPE_ROOM_PAINTER:
79 useSimpleHighlight = true;
80 break;
81 case BRUSHTYPE_LINE:
82 if (b->IsPainting)
83 {
84 useSimpleHighlight = 0;
85 // highlight a line
86 CArrayClear(&b->HighlightedTiles);
87 AlgoLineDrawData data;
88 data.Draw = EditorBrushHighlightPoint;
89 data.data = b;
90 BresenhamLineDraw(b->LastPos, b->Pos, &data);
91 }
92 break;
93 case BRUSHTYPE_BOX: // fallthrough
94 case BRUSHTYPE_ROOM: // fallthrough
95 case BRUSHTYPE_SET_EXIT:
96 if (b->IsPainting)
97 {
98 Vec2i v;
99 Vec2i d = Vec2iNew(
100 b->Pos.x > b->LastPos.x ? 1 : -1,
101 b->Pos.y > b->LastPos.y ? 1 : -1);
102 useSimpleHighlight = 0;
103 CArrayClear(&b->HighlightedTiles);
104 for (v.y = b->LastPos.y; v.y != b->Pos.y + d.y; v.y += d.y)
105 {
106 for (v.x = b->LastPos.x; v.x != b->Pos.x + d.x; v.x += d.x)
107 {
108 if (v.x == b->LastPos.x || v.x == b->Pos.x ||
109 v.y == b->LastPos.y || v.y == b->Pos.y)
110 {
111 EditorBrushHighlightPoint(b, v);
112 }
113 }
114 }
115 }
116 break;
117 case BRUSHTYPE_BOX_FILLED:
118 if (b->IsPainting)
119 {
120 Vec2i v;
121 Vec2i d = Vec2iNew(
122 b->Pos.x > b->LastPos.x ? 1 : -1,
123 b->Pos.y > b->LastPos.y ? 1 : -1);
124 useSimpleHighlight = 0;
125 CArrayClear(&b->HighlightedTiles);
126 for (v.y = b->LastPos.y; v.y != b->Pos.y + d.y; v.y += d.y)
127 {
128 for (v.x = b->LastPos.x; v.x != b->Pos.x + d.x; v.x += d.x)
129 {
130 EditorBrushHighlightPoint(b, v);
131 }
132 }
133 }
134 break;
135 case BRUSHTYPE_SELECT:
136 if (b->IsPainting)
137 {
138 if (b->IsMoving)
139 {
140 Vec2i v;
141 Vec2i offset = Vec2iNew(
142 b->Pos.x - b->DragPos.x, b->Pos.y - b->DragPos.y);
143 useSimpleHighlight = 0;
144 CArrayClear(&b->HighlightedTiles);
145 for (v.y = 0; v.y < b->SelectionSize.y; v.y++)
146 {
147 for (v.x = 0; v.x < b->SelectionSize.x; v.x++)
148 {
149 Vec2i vOffset = Vec2iAdd(
150 Vec2iAdd(v, b->SelectionStart), offset);
151 EditorBrushHighlightPoint(b, vOffset);
152 }
153 }
154 }
155 else
156 {
157 // resize the selection
158 Vec2i v;
159 Vec2i d = Vec2iNew(
160 b->Pos.x > b->LastPos.x ? 1 : -1,
161 b->Pos.y > b->LastPos.y ? 1 : -1);
162 useSimpleHighlight = 0;
163 CArrayClear(&b->HighlightedTiles);
164 for (v.y = b->LastPos.y; v.y != b->Pos.y + d.y; v.y += d.y)
165 {
166 for (v.x = b->LastPos.x; v.x != b->Pos.x + d.x; v.x += d.x)
167 {
168 EditorBrushHighlightPoint(b, v);
169 }
170 }
171 }
172 }
173 else
174 {
175 // If there's a valid selection, draw that instead
176 if (b->SelectionSize.x > 0 && b->SelectionSize.y > 0)
177 {
178 Vec2i v;
179 useSimpleHighlight = 0;
180 CArrayClear(&b->HighlightedTiles);
181 for (v.y = 0; v.y < b->SelectionSize.y; v.y++)
182 {
183 for (v.x = 0; v.x < b->SelectionSize.x; v.x++)
184 {
185 Vec2i vOffset = Vec2iAdd(v, b->SelectionStart);
186 EditorBrushHighlightPoint(b, vOffset);
187 }
188 }
189 }
190 }
191 default:
192 // do nothing
193 break;
194 }
195 if (useSimpleHighlight)
196 {
197 // Simple highlight at brush tip based on brush size
198 CArrayClear(&b->HighlightedTiles);
199 EditorBrushHighlightPoint(b, b->Pos);
200 }
201}

207 MapSetTile(&gMap, pos, tile, m);

211typedef struct
212{
213 EditorBrush *brush;
214 Mission *mission;
215} EditorBrushPaintTilesAtData;
216static void EditorBrushPaintTilesAt(void *data, Vec2i pos)
217{
218 Vec2i v;
219 EditorBrushPaintTilesAtData *paintData = data;
220 EditorBrush *b = paintData->brush;
221 Mission *m = paintData->mission;
222 for (v.y = 0; v.y < b->BrushSize; v.y++)
223 {
224 for (v.x = 0; v.x < b->BrushSize; v.x++)
225 {
226 SetTile(m, Vec2iAdd(pos, v), b->PaintType);
227 }
228 }
229}
230static void EditorBrushPaintLine(EditorBrush *b, Mission *m)
231{
232 // Draw tiles between the last point and the current point
233 EditorBrushPaintTilesAtData paintData;
234 paintData.brush = b;
235 paintData.mission = m;
236 if (b->IsPainting)
237 {
238 AlgoLineDrawData data;
239 data.Draw = EditorBrushPaintTilesAt;
240 data.data = &paintData;
241 BresenhamLineDraw(b->LastPos, b->Pos, &data);
242 }
243 EditorBrushPaintTilesAt(&paintData, b->Pos);
244 b->IsPainting = 1;
245 b->LastPos = b->Pos;
246}
247// Paint all the edge tiles as a wall, unless they are room tiles already;
248// then paint the interior as room tiles
249static void EditorBrushPaintRoom(EditorBrush *b, Mission *m)
250{
251 Vec2i v;
252 for (v.y = 0; v.y < b->BrushSize; v.y++)
253 {
254 for (v.x = 0; v.x < b->BrushSize; v.x++)
255 {
256 unsigned short tile = MAP_ROOM;
257 if (v.x == 0 || v.x == b->BrushSize - 1 ||
258 v.y == 0 || v.y == b->BrushSize - 1)
259 {
260 tile = MAP_WALL;
261 }
262 const Vec2i pos = Vec2iAdd(b->Pos, v);
263 const unsigned short tileExisting = IMapGet(&gMap, pos);
264 if (tileExisting != MAP_ROOM)
265 {
266 SetTile(m, pos, tile);
267 }
268 }
269 }
270 b->IsPainting = true;
271 b->LastPos = b->Pos;
272}

276 unsigned short fromType;
277 unsigned short toType;

281EditorResult EditorBrushStartPainting(EditorBrush *b, Mission *m, int isMain)
282{
283 if (!b->IsPainting)
284 {
285 b->LastPos = b->Pos;
286 }
287 b->PaintType = isMain ? b->MainType : b->SecondaryType;
288 switch (b->Type)
289 {
290 case BRUSHTYPE_POINT:
291 b->IsPainting = true;
292 EditorBrushPaintLine(b, m);
293 return EDITOR_RESULT_CHANGED;
294 case BRUSHTYPE_LINE: // fallthrough
295 case BRUSHTYPE_BOX: // fallthrough
296 case BRUSHTYPE_BOX_FILLED: // fallthrough
297 case BRUSHTYPE_ROOM: // fallthrough
298 case BRUSHTYPE_SET_EXIT:
299 // don't paint until the end
300 break;
301 case BRUSHTYPE_ROOM_PAINTER:
302 EditorBrushPaintRoom(b, m);
303 return EDITOR_RESULT_CHANGED;
304 case BRUSHTYPE_SELECT:
305 // Perform state changes if we've started painting
306 if (!b->IsPainting)
307 {
308 if (!b->IsMoving)
309 {
310 // check if the click was inside the selection
311 // If so, start moving
312 if (b->Pos.x >= b->SelectionStart.x &&
313 b->Pos.y >= b->SelectionStart.y &&
314 b->Pos.x < b->SelectionStart.x + b->SelectionSize.x &&
315 b->Pos.y < b->SelectionStart.y + b->SelectionSize.y)
316 {
317 b->IsMoving = 1;
318 b->DragPos = b->Pos;
319 }
320 }
321 }
322 break;
323 case BRUSHTYPE_FILL:
324 // Use flood fill to change all the tiles of the same type to
325 // another type
326 // Don't paint if target already same type
327 // Special case: don't flood-fill doors
328 if ((MissionGetTile(m, b->Pos) & MAP_MASKACCESS) != b->PaintType &&
329 b->PaintType != MAP_DOOR)
330 {
331 FloodFillData data;
332 data.Fill = MissionFillTile;
333 data.IsSame = MissionIsTileSame;
334 PaintFloodFillData pData;
335 pData.m = m;
336 pData.fromType = MissionGetTile(m, b->Pos) & MAP_MASKACCESS;
337 pData.toType = b->PaintType;
338 data.data = &pData;
339 if (CFloodFill(b->Pos, &data))
340 {
341 return EDITOR_RESULT_CHANGED;
342 }
343 }
344 return EDITOR_RESULT_NONE;
345 case BRUSHTYPE_SET_PLAYER_START:
346 if (MissionGetTile(m, b->Pos) == MAP_ROOM ||
347 MissionGetTile(m, b->Pos) == MAP_FLOOR)
348 {
349 m->u.Static.Start = b->Pos;
350 return EDITOR_RESULT_CHANGED;
351 }
352 break;
353 case BRUSHTYPE_ADD_ITEM:
354 if (isMain)
355 {
356 if (MissionStaticTryAddItem(m, b->u.MapObject, b->Pos))
357 {
358 return EDITOR_RESULT_CHANGED_AND_RELOAD;
359 }
360 }
361 else
362 {
363 if (MissionStaticTryRemoveItemAt(m, b->Pos))
364 {
365 return EDITOR_RESULT_CHANGED_AND_RELOAD;
366 }
367 }
368 break;
369 case BRUSHTYPE_ADD_CHARACTER:
370 if (isMain)
371 {
372 if (MissionStaticTryAddCharacter(m, b->u.ItemIndex, b->Pos))
373 {
374 return EDITOR_RESULT_CHANGED_AND_RELOAD;
375 }
376 }
377 else
378 {
379 if (MissionStaticTryRemoveCharacterAt(m, b->Pos))
380 {
381 return EDITOR_RESULT_CHANGED_AND_RELOAD;
382 }
383 }
384 break;
385 case BRUSHTYPE_ADD_OBJECTIVE:
386 if (isMain)
387 {
388 if (MissionStaticTryAddObjective(
389 m, b->u.ItemIndex, b->Index2, b->Pos))
390 {
391 return EDITOR_RESULT_CHANGED_AND_RELOAD;
392 }
393 }
394 else
395 {
396 if (MissionStaticTryRemoveObjectiveAt(m, b->Pos))
397 {
398 return EDITOR_RESULT_CHANGED_AND_RELOAD;
399 }
400 }
401 break;
402 case BRUSHTYPE_ADD_KEY:
403 if (isMain)
404 {
405 if (MissionStaticTryAddKey(m, b->u.ItemIndex, b->Pos))
406 {
407 return EDITOR_RESULT_CHANGED_AND_RELOAD;
408 }
409 }
410 else
411 {
412 if (MissionStaticTryRemoveKeyAt(m, b->Pos))
413 {
414 return EDITOR_RESULT_CHANGED_AND_RELOAD;
415 }
416 }
417 break;
418 case BRUSHTYPE_SET_KEY:
419 if (isMain || b->u.ItemIndex > 0)
420 {
421 if (MissionStaticTrySetKey(m, b->u.ItemIndex, b->Pos))
422 {
423 return EDITOR_RESULT_CHANGED_AND_RELOAD;
424 }
425 }
426 else
427 {
428 if (MissionStaticTryUnsetKeyAt(m, b->Pos))
429 {
430 return EDITOR_RESULT_CHANGED_AND_RELOAD;
431 }
432 }
433 break;
434 default:
435 assert(0 && "unknown brush type");
436 break;
437 }
438 b->IsPainting = 1;
439 return EDITOR_RESULT_NONE;
440}
441static void MissionFillTile(void *data, Vec2i v)
442{
443 PaintFloodFillData *pData = data;
444 SetTile(pData->m, v, pData->toType);
445}
446static bool MissionIsTileSame(void *data, Vec2i v)
447{
448 PaintFloodFillData *pData = data;
449 return (MissionGetTile(pData->m, v) & MAP_MASKACCESS) == pData->fromType;
450}
451static void EditorBrushPaintBox(
452 EditorBrush *b, Mission *m,
453 unsigned short lineType, unsigned short fillType)
454{
455 // Draw the fill first, then the line
456 // This will create the expected results when brush size is
457 // greater than one, without having the box drawing routine made
458 // aware of brush sizes
459 Vec2i v;
460 Vec2i d = Vec2iNew(
461 b->Pos.x > b->LastPos.x ? 1 : -1,
462 b->Pos.y > b->LastPos.y ? 1 : -1);
463 EditorBrushPaintTilesAtData paintData;
464 paintData.brush = b;
465 paintData.mission = m;
466 // Draw fill
467 if (fillType != MAP_UNSET)
468 {
469 b->PaintType = fillType;
470 for (v.y = b->LastPos.y; v.y != b->Pos.y + d.y; v.y += d.y)
471 {
472 for (v.x = b->LastPos.x; v.x != b->Pos.x + d.x; v.x += d.x)
473 {
474 if (v.x != b->LastPos.x && v.x != b->Pos.x &&
475 v.y != b->LastPos.y && v.y != b->Pos.y)
476 {
477 EditorBrushPaintTilesAt(&paintData, v);
478 }
479 }
480 }
481 }
482 // Draw edge
483 if (lineType != MAP_UNSET)
484 {
485 b->PaintType = lineType;
486 for (v.y = b->LastPos.y; v.y != b->Pos.y + d.y; v.y += d.y)
487 {
488 for (v.x = b->LastPos.x; v.x != b->Pos.x + d.x; v.x += d.x)
489 {
490 if (v.x == b->LastPos.x || v.x == b->Pos.x ||
491 v.y == b->LastPos.y || v.y == b->Pos.y)
492 {
493 EditorBrushPaintTilesAt(&paintData, v);
494 }
495 }
496 }
497 }

499EditorResult EditorBrushStopPainting(EditorBrush *b, Mission *m)
500{
501 EditorResult result = EDITOR_RESULT_NONE;
502 if (b->IsPainting)
503 {
504 switch (b->Type)
505 {
506 case BRUSHTYPE_LINE:
507 EditorBrushPaintLine(b, m);
508 result = EDITOR_RESULT_CHANGED;
509 break;
510 case BRUSHTYPE_BOX:
511 EditorBrushPaintBox(b, m, b->PaintType, MAP_UNSET);
512 result = EDITOR_RESULT_CHANGED;
513 break;
514 case BRUSHTYPE_BOX_FILLED:
515 EditorBrushPaintBox(b, m, b->PaintType, b->PaintType);
516 result = EDITOR_RESULT_CHANGED;
517 break;
518 case BRUSHTYPE_ROOM:
519 EditorBrushPaintBox(b, m, MAP_WALL, MAP_ROOM);
520 result = EDITOR_RESULT_CHANGED;
521 break;
522 case BRUSHTYPE_ROOM_PAINTER:
523 // Reload map to update tiles
524 result = EDITOR_RESULT_RELOAD;
525 break;
526 case BRUSHTYPE_SELECT:
527 if (b->IsMoving)
528 {
529 // Move the tiles from the source to the target
530 // Need to copy all the tiles to a temp buffer first in case
531 // we are moving to an overlapped position
532 CArray movedTiles;
533 Vec2i v;
534 int i;
535 int delta;
536 CArrayInit(&movedTiles, sizeof(unsigned short));
537 // Copy tiles to temp from selection, setting them to MAP_FLOOR
538 // in the process
539 for (v.y = 0; v.y < b->SelectionSize.y; v.y++)
540 {
541 for (v.x = 0; v.x < b->SelectionSize.x; v.x++)
542 {
543 Vec2i vOffset = Vec2iAdd(v, b->SelectionStart);
544 int idx = vOffset.y * m->Size.x + vOffset.x;
545 unsigned short *tile = CArrayGet(
546 &m->u.Static.Tiles, idx);
547 CArrayPushBack(&movedTiles, tile);
548 *tile = MAP_FLOOR;
549 }
550 }
551 // Move the selection to the new position
552 b->SelectionStart.x += b->Pos.x - b->DragPos.x;
553 b->SelectionStart.y += b->Pos.y - b->DragPos.y;
554 // Copy tiles to the new area, for parts of the new area that
555 // are valid
556 i = 0;
557 for (v.y = 0; v.y < b->SelectionSize.y; v.y++)
558 {
559 for (v.x = 0; v.x < b->SelectionSize.x; v.x++)
560 {
561 Vec2i vOffset = Vec2iAdd(v, b->SelectionStart);
562 if (vOffset.x >= 0 && vOffset.x < m->Size.x &&
563 vOffset.y >= 0 && vOffset.y < m->Size.y)
564 {
565 int idx = vOffset.y * m->Size.x + vOffset.x;
566 unsigned short *tileFrom =
567 CArrayGet(&movedTiles, i);
568 unsigned short *tileTo = CArrayGet(
569 &m->u.Static.Tiles, idx);
570 *tileTo = *tileFrom;
571 result = EDITOR_RESULT_CHANGED_AND_RELOAD;
572 }
573 i++;
574 }
575 }
576 // Update the selection to fit within map boundaries
577 delta = -b->SelectionStart.x;
578 if (delta > 0)
579 {
580 b->SelectionStart.x += delta;
581 b->SelectionSize.x -= delta;
582 }
583 delta = -b->SelectionStart.y;
584 if (delta > 0)
585 {
586 b->SelectionStart.y += delta;
587 b->SelectionSize.y -= delta;
588 }
589 delta = b->SelectionStart.x + b->SelectionSize.x - m->Size.x;
590 if (delta > 0)
591 {
592 b->SelectionSize.x -= delta;
593 }
594 delta = b->SelectionStart.y + b->SelectionSize.y - m->Size.y;
595 if (delta > 0)
596 {
597 b->SelectionSize.y -= delta;
598 }
599 // Check if the selection is still valid; if not, invalidate it
600 if (b->SelectionSize.x < 0 || b->SelectionSize.y < 0)
601 {
602 b->SelectionSize = Vec2iZero();
603 }
604
605 b->IsMoving = 0;
606 }
607 else
608 {
609 // Record the selection size
610 b->SelectionStart = Vec2iMin(b->LastPos, b->Pos);
611 b->SelectionSize.x = abs(b->LastPos.x - b->Pos.x) + 1;
612 b->SelectionSize.y = abs(b->LastPos.y - b->Pos.y) + 1;
613 // Disallow 1x1 selection sizes
614 if (b->SelectionSize.x <= 1 && b->SelectionSize.y <= 1)
615 {
616 b->SelectionSize = Vec2iZero();
617 }
618 }
619 break;
620 case BRUSHTYPE_SET_EXIT:
621 {
622 Vec2i exitStart = Vec2iMin(b->LastPos, b->Pos);
623 Vec2i exitEnd = Vec2iMax(b->LastPos, b->Pos);
624 // Clamp within map boundaries
625 exitStart = Vec2iClamp(
626 exitStart, Vec2iZero(), Vec2iMinus(m->Size, Vec2iUnit()));
627 exitEnd = Vec2iClamp(
628 exitEnd, Vec2iZero(), Vec2iMinus(m->Size, Vec2iUnit()));
629 // Check that size is big enough
630 Vec2i size =
631 Vec2iAdd(Vec2iMinus(exitEnd, exitStart), Vec2iUnit());
632 if (size.x >= 3 && size.y >= 3)
633 {
634 // Check that exit area has changed
635 if (!Vec2iEqual(exitStart, m->u.Static.Exit.Start) ||
636 !Vec2iEqual(exitEnd, m->u.Static.Exit.End))
637 {
638 m->u.Static.Exit.Start = exitStart;
639 m->u.Static.Exit.End = exitEnd;
640 result = EDITOR_RESULT_CHANGED_AND_RELOAD;
641 }
642 }
643 }
644 break;
645 default:
646 // do nothing
647 break;
648 }
649 }
650 b->IsPainting = 0;
651 CArrayClear(&b->HighlightedTiles);
652 return result;
653}
