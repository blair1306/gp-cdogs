63UIObject *CreateCaveMapObjs(Vec2i pos, CampaignOptions *co)
64{
65 const int th = FontH();
66 UIObject *c = UIObjectCreate(UITYPE_NONE, 0, Vec2iZero(), Vec2iZero());
67 UIObject *o = UIObjectCreate(
68 UITYPE_LABEL, 0, Vec2iZero(), Vec2iNew(50, th));
69 const int x = pos.x;
70 o->ChangesData = true;
71 // Check whether the map type matches, and set visibility
72 c->CheckVisible = MissionCheckTypeFunc;
73 c->Data = co;
74
75
76 UIObject *o2 = CreateCampaignSeedObj(pos, co);
77 UIObjectAddChild(c, o2);
78
79 pos.x += o2->Size.x;
80 o2 = UIObjectCopy(o);
81 o2->u.LabelFunc = MissionGetFillPercentStr;
82 o2->Data = co;
83 o2->ChangeFunc = MissionChangeFillPercent;
84 o2->Pos = pos;
85 UIObjectAddChild(c, o2);
86 pos.x += o2->Size.x;
87 o2 = UIObjectCopy(o);
88 o2->u.LabelFunc = MissionGetRepeatStr;
89 o2->Data = co;
90 o2->ChangeFunc = MissionChangeRepeat;
91 o2->Pos = pos;
92 UIObjectAddChild(c, o2);
93
94 pos.x = x;
95 pos.y += th;
96 o2 = UIObjectCopy(o);
97 o2->u.LabelFunc = MissionGetR1Str;
98 o2->Data = co;
99 o2->ChangeFunc = MissionChangeR1;
100 o2->Pos = pos;
101 UIObjectAddChild(c, o2);
102 pos.x += o2->Size.x;
103 o2 = UIObjectCopy(o);
104 o2->u.LabelFunc = MissionGetR2Str;
105 o2->Data = co;
106 o2->ChangeFunc = MissionChangeR2;
107 o2->Pos = pos;
108 UIObjectAddChild(c, o2);
109 pos.x += o2->Size.x;
110 o2 = UIObjectCopy(o);
111 o2->u.LabelFunc = MissionGetCorridorWidthStr;
112 o2->Data = co;
113 o2->ChangeFunc = MissionChangeCorridorWidth;
114 o2->Pos = pos;
115 o2->Size.x = 60;
116 UIObjectAddChild(c, o2);
117
118 pos.x = x;
119 pos.y += th;
120 o2 = UIObjectCopy(o);
121 o2->u.LabelFunc = MissionGetRoomCountStr;
122 o2->Data = co;
123 o2->ChangeFunc = MissionChangeRoomCount;
124 o2->Pos = pos;
125 UIObjectAddChild(c, o2);
126 pos.x += o2->Size.x;
127 o2 = UIObjectCopy(o);
128 o2->u.LabelFunc = MissionGetRoomMinStr;
129 o2->Data = co;
130 o2->ChangeFunc = MissionChangeRoomMin;
131 o2->Pos = pos;
132 UIObjectAddChild(c, o2);
133 pos.x += o2->Size.x;
134 o2 = UIObjectCopy(o);
135 o2->u.LabelFunc = MissionGetRoomMaxStr;
136 o2->Data = co;
137 o2->ChangeFunc = MissionChangeRoomMax;
138 o2->Pos = pos;
139 UIObjectAddChild(c, o2);
140
141 pos.x = x;
142 pos.y += th;
143 o2 = UIObjectCopy(o);
144 o2->u.LabelFunc = MissionGetRoomWallCountStr;
145 o2->Data = co;
146 o2->ChangeFunc = MissionChangeRoomWallCount;
147 o2->Pos = pos;
148 UIObjectAddChild(c, o2);
149 pos.x += o2->Size.x;
150 o2 = UIObjectCopy(o);
151 o2->u.LabelFunc = MissionGetRoomWallLenStr;
152 o2->Data = co;
153 o2->ChangeFunc = MissionChangeRoomWallLen;
154 o2->Pos = pos;
155 o2->Size.x = 60;
156 UIObjectAddChild(c, o2);
157 pos.x += o2->Size.x;
158 o2 = UIObjectCopy(o);
159 o2->u.LabelFunc = MissionGetRoomWallPadStr;
160 o2->Data = co;
161 o2->ChangeFunc = MissionChangeRoomWallPad;
162 o2->Pos = pos;
163 o2->Size.x = 60;
164 UIObjectAddChild(c, o2);
165
166 pos.x = x;
167 pos.y += th;
168 o2 = UIObjectCopy(o);
169 o2->u.LabelFunc = MissionGetSquareCountStr;
170 o2->Data = co;
171 o2->ChangeFunc = MissionChangeSquareCount;
172 o2->Pos = pos;
173 UIObjectAddChild(c, o2);
174
175 UIObjectDestroy(o);
176 return c;
177}
178static const char *MissionGetFillPercentStr(UIObject *o, void *data)
179{
180 UNUSED(o);
181 CampaignOptions *co = data;
182 const Mission *m = CampaignGetCurrentMission(co);
183 if (m == NULL) return NULL;
184 static char s[128];
185 sprintf(s, "Fill: %d%%", m->u.Cave.FillPercent);
186 return s;
187}
188static void MissionChangeFillPercent(void *data, int d)
189{
190 CampaignOptions *co = data;
191 Mission *m = CampaignGetCurrentMission(co);
192 if (gEventHandlers.keyboard.modState & KMOD_SHIFT)
193 {
194 d *= 10;
195 }
196 m->u.Cave.FillPercent = CLAMP(m->u.Cave.FillPercent + d, 0, 100);
197}

296static void MissionChangeRoomMin(void *data, int d)
297{
298 CampaignOptions *co = data;
299 CampaignGetCurrentMission(co)->u.Cave.Rooms.Min =
300 CLAMP(CampaignGetCurrentMission(co)->u.Cave.Rooms.Min + d, 5, 50);
301 CampaignGetCurrentMission(co)->u.Cave.Rooms.Max = MAX(
302 CampaignGetCurrentMission(co)->u.Cave.Rooms.Min,
303 CampaignGetCurrentMission(co)->u.Cave.Rooms.Max);
304}
