47typedef struct
48{
49 color_t Color;
50 Vec2i SwatchSize;
51 Vec2i SwatchPad;
52 void *Data;
53 ColorPickerChangeFunc ChangeFunc;
54} ColorPickerData;

59// Create a colour picker using the C-Dogs palette
60UIObject *CreateColorPicker(
61 const Vec2i pos, void *data,
62 ColorPickerGetFunc getFunc, ColorPickerChangeFunc changeFunc)
63{
64 UIObject *c = UIObjectCreate(UITYPE_CONTEXT_MENU, 0, pos, Vec2iZero());
65 c->IsDynamicData = true;
66 c->Data = data;
67
68 // Create a text input area for the colour hex code
69 UIObject *oText = UIObjectCreate(
70 UITYPE_TEXTBOX, 0, Vec2iZero(), FontStrSize("#abcdef"));
71 oText->u.Textbox.IsEditable = true;
72 oText->u.Textbox.TextLinkFunc = ColorPickerText;
73 oText->u.Textbox.MaxLen = sizeof ((ColorTextData *)0)->Text - 1;
74 CSTRDUP(oText->u.Textbox.Hint, "(enter color hex value)");
75 oText->IsDynamicData = true;
76 CMALLOC(oText->Data, sizeof(ColorTextData));
77 // Initialise colour and text
78 ((ColorTextData *)oText->Data)->Color = getFunc(data);
79 ColorStr(((ColorTextData *)oText->Data)->Text, getFunc(data));
80 ((ColorTextData *)oText->Data)->Data = data;
81 ((ColorTextData *)oText->Data)->GetFunc = getFunc;
82 ((ColorTextData *)oText->Data)->ChangeFunc = changeFunc;
83 oText->OnUnfocusFunc = ColorPickerSetFromText;
84 // Constantly check the current mission colour and update text accordingly
85 oText->CheckVisible = ColorPickerUpdateText;
86 UIObjectAddChild(c, oText);
87
88 const int yStart = oText->Size.y + 4;
89
90 // Create colour squares from the palette
91 const Vec2i swatchSize = Vec2iNew(5, 5);
92 const Vec2i swatchPad = Vec2iNew(2, 2);
93 UIObject *o = UIObjectCreate(
94 UITYPE_CUSTOM, 0, Vec2iZero(), Vec2iAdd(swatchSize, swatchPad));
95 o->ChangeFunc = ColorPickerChange;
96 // Changing colour updates the masked pics for the mission, which requires
97 // a reload
98 o->ChangesData = true;
99 o->ReloadData = true;
100 o->u.CustomDrawFunc = ColorPickerDrawSwatch;
101 const Pic *palette = PicManagerGetPic(&gPicManager, "palette");
102 Vec2i v;
103 for (v.y = 0; v.y < palette->size.y; v.y++)
104 {
105 for (v.x = 0; v.x < palette->size.x; v.x++)
106 {
107 const color_t colour = PIXEL2COLOR(
108 palette->Data[v.x + v.y * palette->size.x]);
109 if (colour.a == 0)
110 {
111 continue;
112 }
113 UIObject *o2 = UIObjectCopy(o);
114 o2->IsDynamicData = true;
115 CMALLOC(o2->Data, sizeof(ColorPickerData));
116 ((ColorPickerData *)o2->Data)->Color = colour;
117 ((ColorPickerData *)o2->Data)->SwatchSize = swatchSize;
118 ((ColorPickerData *)o2->Data)->SwatchPad = swatchPad;
119 ((ColorPickerData *)o2->Data)->Data = data;
120 ((ColorPickerData *)o2->Data)->ChangeFunc = changeFunc;
121 o2->Pos = Vec2iMult(v, o->Size);
122 o2->Pos.y += yStart;
123 UIObjectAddChild(c, o2);
124 }
125 }
126 UIObjectDestroy(o);
127
128 // Create a dummy label that can be clicked to close the context menu
129 CreateCloseLabel(c, Vec2iNew(
130 palette->size.x * (swatchSize.x + swatchPad.x) - FontStrW("Close"),
131 0));
132
133 return c;
134}
135static void ColorPickerUpdateText(UIObject *o, void *data)
136{
137 // Only update text if the text box doesn't have focus
138 if (o->Parent->Highlighted == o) return;
139 ColorTextData *cData = data;
140 cData->Color = cData->GetFunc(cData->Data);
141 ColorStr(cData->Text, cData->Color);
142}
143static char *ColorPickerText(UIObject *o, void *data)
144{
145 UNUSED(o);
146 ColorTextData *cData = data;
147 return cData->Text;
148}
149static bool ColorPickerSetFromText(void *data)
150{
151 ColorTextData *cData = data;
152 const color_t colour = StrColor(cData->Text);
153 cData->ChangeFunc(colour, cData->Data);
154 return true;
155}
156static void ColorPickerChange(void *data, int d)
157{
158 UNUSED(d);
159 ColorPickerData *mc = data;
160 mc->ChangeFunc(mc->Color, mc->Data);
161}
162static void ColorPickerDrawSwatch(
163 UIObject *o, GraphicsDevice *g, Vec2i pos, void *data)
164{
165 UNUSED(o);
166 const ColorPickerData *cpd = data;
167 DrawRectangle(
168 g,
169 Vec2iAdd(Vec2iAdd(pos, o->Pos), Vec2iScaleDiv(cpd->SwatchPad, 2)),
170 cpd->SwatchSize,
171 cpd->Color,
172 0);
173}

176typedef enum
177{
178 MISSION_COLOR_WALL,
179 MISSION_COLOR_FLOOR,
180 MISSION_COLOR_ROOM,
181 MISSION_COLOR_EXTRA,
182 MISSION_COLOR_COUNT
183} MissionColorType;
184typedef struct
185{
186 CampaignOptions *C;
187 MissionColorType Type;
188} MissionColorData;
189static const char *MissionGetColorStr(UIObject *o, void *data);
190static void MissionColorChange(const color_t c, void *data);
191static color_t CampaignGetMissionColor(void *data);
192Vec2i CreateColorObjs(CampaignOptions *co, UIObject *c, Vec2i pos)
193{
194 const int th = FontH();
195
196 UIObject *o = UIObjectCreate(
197 UITYPE_LABEL, 0, Vec2iZero(), Vec2iNew(100, th));
198 o->ChangesData = true;
199 o->ReloadData = true;
200 o->u.LabelFunc = MissionGetColorStr;
201
202 for (int i = 0; i < (int)MISSION_COLOR_COUNT; i++)
203 {
204 UIObject *o2 = UIObjectCopy(o);
205 o2->IsDynamicData = true;
206 CMALLOC(o2->Data, sizeof(MissionColorData));
207 ((MissionColorData *)o2->Data)->C = co;
208 ((MissionColorData *)o2->Data)->Type = (MissionColorType)i;
209 o2->Pos = pos;
210
211 MissionColorData *mcd;
212 CMALLOC(mcd, sizeof *mcd);
213 mcd->C = co;
214 mcd->Type = (MissionColorType)i;
215 UIObjectAddChild(o2, CreateColorPicker(
216 Vec2iZero(), mcd, CampaignGetMissionColor, MissionColorChange));
217 UIObjectAddChild(c, o2);
218 pos.y += th;
219 }
220
221 UIObjectDestroy(o);
222 return pos;
223}
224static const char *MissionGetColorStr(UIObject *o, void *data)
225{
226 static char s[128];
227 UNUSED(o);
228 MissionColorData *mc = data;
229 static const char *colourTypeNames[] =
230 {
231 "Walls", "Floors", "Rooms", "Extra"
232 };
233 char c[8];
234 ColorStr(c, CampaignGetMissionColor(mc));
235 sprintf(s, "%s: #%s", colourTypeNames[(int)mc->Type], c);
236 return s;
237}
238static void MissionColorChange(const color_t c, void *data)
239{
240 MissionColorData *mcd = data;
241 Mission *m = CampaignGetCurrentMission(mcd->C);
242 switch (mcd->Type)
243 {
244 case MISSION_COLOR_WALL: m->WallMask = c; break;
245 case MISSION_COLOR_FLOOR: m->FloorMask = c; break;
246 case MISSION_COLOR_ROOM: m->RoomMask = c; break;
247 case MISSION_COLOR_EXTRA: m->AltMask = c; break;
248 default:
249 CASSERT(false, "Unexpected mission colour");
250 break;
251 }
252}
253static color_t CampaignGetMissionColor(void *data)
254{
255 MissionColorData *mcd = data;
256 const Mission *m = CampaignGetCurrentMission(mcd->C);
257 if (m == NULL) return colorWhite;
258 switch (mcd->Type)
259 {
260 case MISSION_COLOR_WALL: return m->WallMask;
261 case MISSION_COLOR_FLOOR: return m->FloorMask;
262 case MISSION_COLOR_ROOM: return m->RoomMask;
263 case MISSION_COLOR_EXTRA: return m->AltMask;
264 default:
265 CASSERT(false, "Unexpected mission colour");
266 return colorWhite;
267 }
