56void DrawKey(UIObject *o, GraphicsDevice *g, Vec2i pos, void *vData)
57{
58 const IndexedEditorBrush *data = vData;
59 if (data->u.ItemIndex == -1)
60 {
61 // No key; don't draw
62 return;
63 }
64 const Mission *m = CampaignGetCurrentMission(&gCampaign);
65 const Pic *pic = KeyPickupClass(m->KeyStyle, data->u.ItemIndex)->Pic;
66 pos = Vec2iAdd(Vec2iAdd(pos, o->Pos), Vec2iScaleDiv(o->Size, 2));
67 pos = Vec2iMinus(pos, Vec2iScaleDiv(pic->size, 2));
68 Blit(g, pic, pos);
69}

71void InsertMission(CampaignOptions *co, Mission *mission, int idx)
72{
73 Mission m;
74 if (mission == NULL)
75 {
76 MissionInit(&m);
77 m.Size = Vec2iNew(48, 48);
78 // Set some default values for the mission
79 m.u.Classic.CorridorWidth = 1;
80 m.u.Classic.Rooms.Min = m.u.Classic.Rooms.Max = 5;
81 m.u.Classic.Rooms.WallLength = 1;
82 m.u.Classic.Rooms.WallPad = 1;
83 m.u.Classic.Doors.Min = m.u.Classic.Doors.Max = 1;
84 m.u.Classic.Pillars.Min = m.u.Classic.Pillars.Max = 1;
85 }
86 else
87 {
88 memset(&m, 0, sizeof m);
89 MissionCopy(&m, mission);
90 }
91 CArrayInsert(&co->Setting.Missions, idx, &m);
92}
93void DeleteMission(CampaignOptions *co)
94{
95 CASSERT(
96 co->MissionIndex < (int)co->Setting.Missions.size,
97 "invalid mission index");
98 MissionTerminate(CampaignGetCurrentMission(co));
99 CArrayDelete(&co->Setting.Missions, co->MissionIndex);
100 if (co->MissionIndex >= (int)co->Setting.Missions.size)
101 {
102 co->MissionIndex = MAX(0, (int)co->Setting.Missions.size - 1);
103 }
104}

106bool ConfirmScreen(const char *info, const char *msg)
107{
108 int w = gGraphicsDevice.cachedConfig.Res.x;
109 int h = gGraphicsDevice.cachedConfig.Res.y;
110 ClearScreen(&gGraphicsDevice);
111 FontStr(info, Vec2iNew((w - FontStrW(info)) / 2, (h - FontH()) / 2));
112 FontStr(msg, Vec2iNew((w - FontStrW(msg)) / 2, (h + FontH()) / 2));
113 BlitUpdateFromBuf(&gGraphicsDevice, gGraphicsDevice.screen);
114 BlitFlip(&gGraphicsDevice);
115
116 SDL_Keycode k = SDL_GetKeyFromScancode(GetKey(&gEventHandlers));
117 return k == SDLK_y;
118}

120void ClearScreen(GraphicsDevice *g)
121{
122 color_t color = { 32, 32, 60, 255 };
123 const Uint32 pixel = COLOR2PIXEL(color);
124 for (int i = 0; i < GraphicsGetScreenSize(&g->cachedConfig); i++)
125 {
126 g->buf[i] = pixel;
127 }
128}

130void DisplayFlag(
131 const Vec2i pos, const char *s, const bool isOn, const bool isHighlighted)
132{
133 color_t labelMask = isHighlighted ? colorRed : colorWhite;

139static const char *CampaignGetSeedStr(UIObject *o, void *data);
140static void CampaignChangeSeed(void *data, int d);
141UIObject *CreateCampaignSeedObj(const Vec2i pos, CampaignOptions *co)
142{
143 const int th = FontH();
144 UIObject *o = UIObjectCreate(
145 UITYPE_LABEL, 0, Vec2iZero(), Vec2iNew(50, th));
146 o->ChangesData = true;
147 o->u.LabelFunc = CampaignGetSeedStr;
148 o->Data = co;
149 o->ChangeFunc = CampaignChangeSeed;
150 CSTRDUP(o->Tooltip, "Preview with different random seed");
151 o->Pos = pos;
152 return o;
153}
154static const char *CampaignGetSeedStr(UIObject *o, void *data)
155{
156 static char s[128];
157 UNUSED(o);
158 CampaignOptions *co = data;
159 if (!CampaignGetCurrentMission(co)) return NULL;
160 sprintf(s, "Seed: %d", ConfigGetInt(&gConfig, "Game.RandomSeed"));
161 return s;
162}
163static void CampaignChangeSeed(void *data, int d)
164{
165 UNUSED(data);
166 if (gEventHandlers.keyboard.modState & KMOD_SHIFT)
167 {
168 d *= 10;
169 }
170 ConfigSetInt(
171 &gConfig, "Game.RandomSeed",
172 ConfigGetInt(&gConfig, "Game.RandomSeed") + d);
173}

175typedef struct
176{
177 bool (*ObjFunc)(UIObject *, MapObject *, void *);
178 void *Data;
179 // Data size required for map item change checking
180 size_t DataSize;
181 Vec2i GridItemSize;
182 Vec2i GridSize;
183 Vec2i PageOffset;
184} CreateAddMapItemObjsImplData;
185static UIObject *CreateAddMapItemObjsImpl(
186 Vec2i pos, CreateAddMapItemObjsImplData data);
187UIObject *CreateAddMapItemObjs(
188 const Vec2i pos, bool (*objFunc)(UIObject *, MapObject *, void *),
189 void *data, const size_t dataSize, const bool expandDown)
190{
191 CreateAddMapItemObjsImplData d;
192 d.ObjFunc = objFunc;
193 d.Data = data;
194 d.DataSize = dataSize;
195 d.GridItemSize = Vec2iNew(TILE_WIDTH + 4, TILE_HEIGHT * 2 + 4);
196 d.GridSize = Vec2iNew(8, 6);
197 if (expandDown)
198 {
199 d.PageOffset = Vec2iNew(-5, 5);
200 }
201 else
202 {
203 d.PageOffset =
204 Vec2iNew(-5, -d.GridItemSize.y * d.GridSize.y - FontH() - 5);
205 }
206 return CreateAddMapItemObjsImpl(pos, d);
207}

221static void CreateAddMapItemSubObjs(UIObject *c, void *vData);
222static UIObject *CreateAddMapItemObjsImpl(
223 Vec2i pos, CreateAddMapItemObjsImplData data)
224{
225 UIObject *c = UIObjectCreate(UITYPE_CONTEXT_MENU, 0, pos, Vec2iZero());
226 c->OnFocusFunc = CreateAddMapItemSubObjs;
227 c->IsDynamicData = true;
228 CMALLOC(c->Data, sizeof(CreateAddMapItemObjsImplData));
229 memcpy(c->Data, &data, sizeof data);
230
231 return c;
232}
233static int CountAddMapItemSubObjs(const UIObject *o);
234static void CreateAddMapItemSubObjs(UIObject *c, void *vData)
235{
236 const CreateAddMapItemObjsImplData *data = vData;
237 const int pageSize = data->GridSize.x * data->GridSize.y;
238 // Check if we need to recreate the objs
239 // TODO: this is a very heavyweight way to do it
240 int count = 0;
241 bool allChildrenSame = true;
242 for (int i = 0; i < MapObjectsCount(&gMapObjects); i++)
243 {
244 MapObject *mo = IndexMapObject(i);
245 UIObject *o2 =
246 UIObjectCreate(UITYPE_CUSTOM, 0, Vec2iZero(), data->GridItemSize);
247 o2->IsDynamicData = true;
248 CCALLOC(o2->Data, data->DataSize);
249 if (!data->ObjFunc(o2, mo, data->Data))
250 {
251 UIObjectDestroy(o2);
252 continue;
253 }
254 const int pageIdx = count / pageSize;
255 if (pageIdx >= (int)c->Children.size)
256 {
257 allChildrenSame = false;
258 break;
259 }
260 const UIObject **op = CArrayGet(&c->Children, pageIdx);
261 const UIObject **octx = CArrayGet(&(*op)->Children, 0);
262 const int idx = count % pageSize;
263 if (idx >= (int)(*octx)->Children.size)
264 {
265 allChildrenSame = false;
266 break;
267 }
268 const UIObject **oc = CArrayGet(&(*octx)->Children, idx);
269 if (memcmp(o2->Data, (*oc)->Data, data->DataSize) != 0)
270 {
271 allChildrenSame = false;
272 UIObjectDestroy(o2);
273 break;
274 }
275 count++;
276 UIObjectDestroy(o2);
277 }
278 int cCount = CountAddMapItemSubObjs(c);
279 if (cCount == count && allChildrenSame)
280 {
281 return;
282 }
283
284 // Recreate the child UI objects
285 c->Highlighted = NULL;
286 UIObject **objs = c->Children.data;
287 for (int i = 0; i < (int)c->Children.size; i++, objs++)
288 {
289 UIObjectDestroy(*objs);
290 }
291 CArrayClear(&c->Children);
292
293 // Create pagination
294 int pageNum = 1;
295 UIObject *pageLabel = NULL;
296 UIObject *page = NULL;
297 UIObject *o =
298 UIObjectCreate(UITYPE_CUSTOM, 0, Vec2iZero(), data->GridItemSize);
299 o->ChangesData = true;
300 const Vec2i gridStart = Vec2iZero();
301 Vec2i pos = Vec2iZero();
302 count = 0;
303 for (int i = 0; i < MapObjectsCount(&gMapObjects); i++)
304 {
305 // Only add normal map objects
306 MapObject *mo = IndexMapObject(i);
307 UIObject *o2 = UIObjectCopy(o);
308 o2->IsDynamicData = true;
309 CCALLOC(o2->Data, data->DataSize);
310 if (!data->ObjFunc(o2, mo, data->Data))
311 {
312 UIObjectDestroy(o2);
313 continue;
314 }
315 o2->Pos = pos;
316 if (count == 0)
317 {
318 pageLabel = UIObjectCreate(
319 UITYPE_LABEL, 0,
320 Vec2iNew((pageNum - 1) * 10, 0), Vec2iNew(10, FontH()));
321 char buf[32];
322 sprintf(buf, "%d", pageNum);
323 UIObjectSetDynamicLabel(pageLabel, buf);
324 page = UIObjectCreate(
325 UITYPE_CONTEXT_MENU, 0,
326 Vec2iAdd(data->PageOffset, pageLabel->Size), Vec2iZero());
327 UIObjectAddChild(pageLabel, page);
328 pageNum++;
329 }
330 UIObjectAddChild(page, o2);
331 pos.x += o->Size.x;
332 if (((count + 1) % data->GridSize.x) == 0)
333 {
334 pos.x = gridStart.x;
335 pos.y += o->Size.y;
336 }
337 count++;
338 if (count == pageSize)
339 {
340 count = 0;
341 pos = gridStart;
342 UIObjectAddChild(c, pageLabel);
343 }
344 }
345 if (pageLabel != NULL)
346 {
347 UIObjectAddChild(c, pageLabel);
348 }
349
350 UIObjectDestroy(o);
351}
352static int CountAddMapItemSubObjs(const UIObject *o)
353{
354 int count = 0;
355 CA_FOREACH(const UIObject *, page, o->Children)
356 const UIObject **ctx = CArrayGet(&(*page)->Children, 0);
357 count += (int)(*ctx)->Children.size;
358 CA_FOREACH_END()
359 return count;
360}

362char *MakePlacementFlagTooltip(const MapObject *mo)
363{
364 // Add a descriptive tooltip for the map object
365 char buf[512];
366 // Construct text representing the placement flags
367 char pfBuf[128];
368 if (mo->Flags == 0)
369 {
370 sprintf(pfBuf, "anywhere\n");
371 }
372 else
373 {
374 strcpy(pfBuf, "");
375 for (int i = 1; i < PLACEMENT_COUNT; i++)
376 {
377 if (mo->Flags & (1 << i))
378 {
379 if (strlen(pfBuf) > 0)
380 {
381 strcat(pfBuf, ", ");
382 }
383 strcat(pfBuf, PlacementFlagStr(i));
384 }
385 }
386 }
387 // Construct text representing explosion guns
388 char exBuf[256];
389 strcpy(exBuf, "");
390 if (mo->DestroyGuns.size > 0)
391 {
392 sprintf(exBuf, "\nExplodes: ");
393 for (int i = 0; i < (int)mo->DestroyGuns.size; i++)
394 {
395 if (i > 0)
396 {
397 strcat(exBuf, ", ");
398 }
399 const GunDescription **g = CArrayGet(&mo->DestroyGuns, i);
400 strcat(exBuf, (*g)->name);
401 }
402 }
403 sprintf(
404 buf, "%s\nHealth: %d\nPlacement: %s%s",
405 mo->Name, mo->Health, pfBuf, exBuf);
406 char *tmp;
407 CSTRDUP(tmp, buf);
408 return tmp;
409}
