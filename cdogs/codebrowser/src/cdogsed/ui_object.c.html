45UIObject *UIObjectCreate(UIType type, int id, Vec2i pos, Vec2i size)
46{
47 UIObject *o;
48 CCALLOC(o, sizeof *o);
49 o->Type = type;
50 o->Id = id;
51 o->Pos = pos;
52 o->Size = size;
53 o->IsVisible = true;
54 o->ChangeDisablesContext = true;
55 switch (type)
56 {
57 case UITYPE_TEXTBOX:
58 o->u.Textbox.IsEditable = true;
59 o->ChangesData = 1;
60 break;
61 case UITYPE_CONTEXT_MENU:
62 // Context menu always starts as invisible
63 o->IsVisible = false;
64 break;
65 default:
66 // do nothing
67 break;
68 }
69 CArrayInit(&o->Children, sizeof o);
70 return o;
71}

89UIObject *UIObjectCopy(const UIObject *o)
90{
91 UIObject *res = UIObjectCreate(o->Type, o->Id, o->Pos, o->Size);
92 res->IsVisible = o->IsVisible;
93 res->Id2 = o->Id2;
94 res->Flags = o->Flags;
95 res->Tooltip = o->Tooltip;
96 res->Parent = o->Parent;
97 res->DoNotHighlight = o->DoNotHighlight;
98 res->Label = o->Label;
99 res->IsDynamicLabel = o->IsDynamicLabel;
100 if (o->IsDynamicLabel)
101 {
102 CSTRDUP(res->Label, o->Label);
103 }
104 res->Data = o->Data;
105 CASSERT(!o->IsDynamicData, "Cannot copy unknown dynamic data size");
106 res->IsDynamicData = false;
107 res->ChangeFunc = o->ChangeFunc;
108 res->ChangeFuncAlt = o->ChangeFuncAlt;
109 res->ChangeDisablesContext = o->ChangeDisablesContext;
110 res->ChangesData = o->ChangesData;
111 res->ReloadData = o->ReloadData;
112 res->OnFocusFunc = o->OnFocusFunc;
113 res->OnUnfocusFunc = o->OnUnfocusFunc;
114 res->CheckVisible = o->CheckVisible;
115 memcpy(&res->u, &o->u, sizeof res->u);
116 return res;
117}

119void UIObjectDestroy(UIObject *o)
120{
121 CFREE(o->Tooltip);
122 if (o->IsDynamicLabel)
123 {
124 CFREE(o->Label);
125 }
126 CA_FOREACH(UIObject *, obj, o->Children)
127 UIObjectDestroy(*obj);
128 CA_FOREACH_END()
129 CArrayTerminate(&o->Children);
130 if (o->IsDynamicData)
131 {
132 CFREE(o->Data);
133 }
134 switch (o->Type)
135 {
136 case UITYPE_TEXTBOX:
137 CFREE(o->u.Textbox.Hint);
138 break;
139 default:
140 // do nothing
141 break;
142 }
143 CFREE(o);
144}

146void UIObjectAddChild(UIObject *o, UIObject *c)
147{
148 if (c == NULL)
149 {
150 return;
151 }
152 CArrayPushBack(&o->Children, &c);
153 c->Parent = o;
154 if (o->Type == UITYPE_CONTEXT_MENU)
155 {
156 // Resize context menu based on children
157 o->Size = Vec2iMax(o->Size, Vec2iAdd(c->Pos, c->Size));
158 }
159}

161void UIObjectHighlight(UIObject *o, const bool shift)
162{
163 if (o->DoNotHighlight)
164 {
165 return;
166 }
167 if (o->Parent && o->Parent->Highlighted != o)
168 {
169 o->Parent->Highlighted = o;
170 UIObjectHighlight(o->Parent, shift);
171 }
172 if (o->OnFocusFunc)
173 {
174 o->OnFocusFunc(o, o->Data);
175 }
176 // Show any context menu children
177 CA_FOREACH(UIObject *, obj, o->Children)
178 if ((*obj)->Type == UITYPE_CONTEXT_MENU && !shift)
179 {
180 (*obj)->IsVisible = true;
181 if ((*obj)->OnFocusFunc)
182 {
183 (*obj)->OnFocusFunc(*obj, (*obj)->Data);
184 }
185 }
186 CA_FOREACH_END()
187}

189int UIObjectIsHighlighted(UIObject *o)
190{
191 return o->Parent == NULL || o->Parent->Highlighted == o;
192}

194bool UIObjectUnhighlight(UIObject *o, const bool unhighlightParents)
195{
196 bool changed = false;
197 if (o->Highlighted)
198 {
199 changed =
200 UIObjectUnhighlight(o->Highlighted, unhighlightParents) || changed;
201 }
202 o->Highlighted = NULL;
203 if (o->OnUnfocusFunc)
204 {
205 changed = o->OnUnfocusFunc(o->Data) || changed;
206 }
207 // Disable any context menu children
208 if (o->Type == UITYPE_CONTEXT_MENU)
209 {
210 o->IsVisible = false;
211 }
212 CA_FOREACH(UIObject *, obj, o->Children)
213 if ((*obj)->Type == UITYPE_CONTEXT_MENU)
214 {
215 (*obj)->IsVisible = false;
216 if ((*obj)->OnUnfocusFunc)
217 {
218 changed = (*obj)->OnUnfocusFunc((*obj)->Data) || changed;
219 }
220 }
221 CA_FOREACH_END()
222 // Unhighlight all parents
223 if (unhighlightParents && o->Parent != NULL)
224 {
225 // Prevent infinite loop
226 if (o->Parent->Highlighted == o)
227 {
228 o->Parent->Highlighted = NULL;
229 }
230 changed =
231 UIObjectUnhighlight(o->Parent, unhighlightParents) || changed;
232 }
233 return changed;
234}

236static void DisableContextMenuParents(UIObject *o);
237EditorResult UIObjectChange(UIObject *o, const int d, const bool shift)
238{
239 // Activate change func if available
240 bool changed = false;
241 if (o->ChangeFuncAlt && shift)
242 {
243 o->ChangeFuncAlt(o->Data, d);
244 changed = true;
245 }
246 else if (o->ChangeFunc)
247 {
248 o->ChangeFunc(o->Data, d);
249 changed = true;
250 }
251 if (changed)
252 {
253 if (o->ChangeDisablesContext)
254 {
255 DisableContextMenuParents(o);
256 }
257 return EDITOR_RESULT_NEW(o->ChangesData, o->ReloadData);
258 }
259 return EDITOR_RESULT_NONE;
260}
261// Disable all parent context menus once the child is clicked
262static void DisableContextMenuParents(UIObject *o)
263{
264 // Don't disable if we are a textbox
265 if (o->Parent && o->Type != UITYPE_TEXTBOX)
266 {
267 if (o->Parent->Type == UITYPE_CONTEXT_MENU)
268 {
269 o->Parent->IsVisible = false;
270 }
271 DisableContextMenuParents(o->Parent);
272 }
273}

275EditorResult UIObjectAddChar(UIObject *o, char c)
276{
277 if (!o)
278 {
279 return EDITOR_RESULT_NONE;
280 }
281 const EditorResult childResult = UIObjectAddChar(o->Highlighted, c);
282 if (o->Type != UITYPE_TEXTBOX)
283 {
284 return childResult;
285 }
286 else if (childResult != EDITOR_RESULT_NONE)
287 {
288 // See if there are highlighted textbox children;
289 // if so activate them instead
290 return childResult;
291 }
292 if (o->u.Textbox.TextSourceFunc)
293 {
294 // Dynamically-allocated char buf, expand
295 char **s = o->u.Textbox.TextSourceFunc(o->Data);
296 if (!s)
297 {
298 return EDITOR_RESULT_NONE;
299 }
300 size_t l = *s ? strlen(*s) : 0;
301 CREALLOC(*s, l + 2);
302 (*s)[l + 1] = 0;
303 (*s)[l] = c;
304 }
305 else
306 {
307 // Static char buf, simply append
308 char *s = o->u.Textbox.TextLinkFunc(o, o->Data);
309 size_t l = strlen(s);
310 if ((int)l >= o->u.Textbox.MaxLen)
311 {
312 return EDITOR_RESULT_NONE;
313 }
314 s[l + 1] = 0;
315 s[l] = c;
316 }
317 if (o->ChangeFunc)
318 {
319 o->ChangeFunc(o->Data, 1);
320 }
321 return EDITOR_RESULT_NEW(o->ChangesData, o->ReloadData);
322}
323EditorResult UIObjectDelChar(UIObject *o)
324{
325 if (!o)
326 {
327 return EDITOR_RESULT_NONE;
328 }
329 const EditorResult childResult = UIObjectDelChar(o->Highlighted);
330 if (o->Type != UITYPE_TEXTBOX)
331 {
332 return childResult;
333 }
334 else if (childResult != EDITOR_RESULT_NONE)
335 {
336 // See if there are highlighted textbox children;
337 // if so activate them instead
338 return childResult;
339 }
340 char *s = o->u.Textbox.TextLinkFunc(o, o->Data);
341 if (!s || s[0] == '\0')
342 {
343 return EDITOR_RESULT_NONE;
344 }
345 s[strlen(s) - 1] = 0;
346 if (o->ChangeFunc)
347 {
348 o->ChangeFunc(o->Data, -1);
349 }
350 return EDITOR_RESULT_NEW(o->ChangesData, o->ReloadData);
351}

355static const char *LabelGetText(UIObject *o)
356{
357 assert(o->Type == UITYPE_LABEL && "invalid UIObject type");
358 if (o->Label)
359 {
360 return o->Label;
361 }
362 else if (o->u.LabelFunc)
363 {
364 return o->u.LabelFunc(o, o->Data);
365 }
366 return NULL;

368typedef struct
369{
370 UIObject *obj;
371 Vec2i pos;
372} UIObjectDrawContext;
373static void UIObjectDrawAndAddChildren(
374 UIObject *o, GraphicsDevice *g, Vec2i pos, Vec2i mouse, CArray *objs)
375{
376 if (!o)
377 {
378 return;
379 }
380 if (o->CheckVisible)
381 {
382 o->CheckVisible(o, o->Data);
383 }
384 if (!o->IsVisible)
385 {
386 return;
387 }
388 int isHighlighted = UIObjectIsHighlighted(o);
389 Vec2i oPos = Vec2iAdd(pos, o->Pos);
390 switch (o->Type)
391 {
392 case UITYPE_LABEL:
393 {
394 const char *text = LabelGetText(o);
395 if (!text)
396 {
397 break;
398 }
399 color_t textMask = isHighlighted ? colorRed : colorWhite;
400 FontStrMaskWrap(text, oPos, textMask, o->Size.x);
401 }
402 break;
403 case UITYPE_TEXTBOX:
404 {
405 int isText = !!o->u.Textbox.TextLinkFunc;
406 const char *text =
407 isText ? o->u.Textbox.TextLinkFunc(o, o->Data) : NULL;
408 int isEmptyText = !isText || !text || strlen(text) == 0;
409 color_t bracketMask = isHighlighted ? colorRed : colorWhite;
410 color_t textMask = isEmptyText ? colorGray : colorWhite;
411 int oPosX = oPos.x;
412 if (isEmptyText)
413 {
414 text = o->u.Textbox.Hint;
415 }
416 if (!o->u.Textbox.IsEditable)
417 {
418 textMask = bracketMask;
419 }
420 if (o->u.Textbox.IsEditable)
421 {
422 oPos = FontChMask('>', oPos, bracketMask);
423 }
424 oPos = FontStrMaskWrap(
425 text, oPos, textMask, o->Pos.x + o->Size.x - oPosX);
426 if (o->u.Textbox.IsEditable)
427 {
428 oPos = FontChMask('<', oPos, bracketMask);
429 }
430 oPos.x = oPosX;
431 }
432 break;
433 case UITYPE_BUTTON:
434 {
435 int isDown =
436 o->u.Button.IsDownFunc && o->u.Button.IsDownFunc(o->Data);
437 BlitMasked(
438 g, o->u.Button.Pic, oPos, isDown ? colorGray : colorWhite, 1);
439 }
440 break;
441 case UITYPE_CONTEXT_MENU:
442 {
443 // Draw background
444 DrawRectangle(
445 g,
446 Vec2iAdd(oPos, Vec2iScale(Vec2iUnit(), -TOOLTIP_PADDING)),
447 Vec2iAdd(o->Size, Vec2iScale(Vec2iUnit(), 2 * TOOLTIP_PADDING)),
448 menuBGColor,
449 0);
450 // Find if mouse over any children, and draw highlight
451 CA_FOREACH(UIObject *, child, o->Children)
452 if (IsInside(mouse, Vec2iAdd(oPos, (*child)->Pos), (*child)->Size))
453 {
454 DrawRectangle(
455 g,
456 Vec2iAdd(oPos, (*child)->Pos),
457 (*child)->Size,
458 hiliteColor,
459 0);
460 }
461 CA_FOREACH_END()
462 }
463 break;
464 case UITYPE_CUSTOM:
465 o->u.CustomDrawFunc(o, g, pos, o->Data);
466 break;
467 default:
468 // do nothing
469 break;
470 }
471
472 // add children
473 if (objs != NULL)
474 {
475 CA_FOREACH(UIObject *, obj, o->Children)
476 if (!((*obj)->Flags & UI_ENABLED_WHEN_PARENT_HIGHLIGHTED_ONLY) ||
477 isHighlighted)
478 {
479 UIObjectDrawContext c;
480 c.obj = *obj;
481 c.pos = oPos;
482 CArrayPushBack(objs, &c);
483 }
484 CA_FOREACH_END()
485 }
486}
487void UIObjectDraw(
488 UIObject *o, GraphicsDevice *g, Vec2i pos, Vec2i mouse, CArray *drawObjs)
489{
490 // Draw this UIObject and its children in BFS order
491 // Maintain a queue of UIObjects to draw
492 if (drawObjs->elemSize == 0)
493 {
494 CArrayInit(drawObjs, sizeof(UIObjectDrawContext));
495 UIObjectDrawContext c;
496 c.obj = o;
497 c.pos = pos;
498 CArrayPushBack(drawObjs, &c);
499 for (int i = 0; i < (int)drawObjs->size; i++)
500 {
501 UIObjectDrawContext *cPtr = CArrayGet(drawObjs, i);
502 UIObjectDrawAndAddChildren(
503 cPtr->obj, g, cPtr->pos, mouse, drawObjs);
504 }
505 }
506 else
507 {
508 for (int i = 0; i < (int)drawObjs->size; i++)
509 {
510 UIObjectDrawContext *cPtr = CArrayGet(drawObjs, i);
511 UIObjectDrawAndAddChildren(
512 cPtr->obj, g, cPtr->pos, mouse, NULL);
513 }
514 }
515}

517static int IsInside(Vec2i pos, Vec2i rectPos, Vec2i rectSize)
518{
519 return
520 pos.x >= rectPos.x &&
521 pos.x < rectPos.x + rectSize.x &&
522 pos.y >= rectPos.y &&
523 pos.y < rectPos.y + rectSize.y;
524}

526bool UITryGetObjectImpl(UIObject *o, const Vec2i pos, UIObject **out);
527bool UITryGetObject(UIObject *o, Vec2i pos, UIObject **out)
528{
529 if (o == NULL)
530 {
531 return false;
532 }
533 // Find the absolute coordinates of the UIObject, by recursing up to its
534 // parent
535 const UIObject *o2 = o->Parent;
536 while (o2 != NULL)
537 {
538 pos = Vec2iMinus(pos, o2->Pos);
539 o2 = o2->Parent;
540 }
541 return UITryGetObjectImpl(o, pos, out);
542}
543bool UITryGetObjectImpl(UIObject *o, const Vec2i pos, UIObject **out)
544{
545 if (!o->IsVisible)
546 {
547 return false;
548 }
549 bool isHighlighted = UIObjectIsHighlighted(o);
550 CA_FOREACH(UIObject *, obj, o->Children)
551 if ((!((*obj)->Flags & UI_ENABLED_WHEN_PARENT_HIGHLIGHTED_ONLY) ||
552 isHighlighted) &&
553 (*obj)->IsVisible &&
554 UITryGetObjectImpl(*obj, Vec2iMinus(pos, o->Pos), out))
555 {
556 return true;
557 }
558 CA_FOREACH_END()
559 if (IsInside(pos, o->Pos, o->Size) && o->Type != UITYPE_CONTEXT_MENU)
560 {
561 *out = o;
562 return true;
563 }
564 return false;
565}

567void UITooltipDraw(GraphicsDevice *device, Vec2i pos, const char *s)
568{
569 Vec2i bgSize = FontStrSize(s);
570 pos = Vec2iAdd(pos, Vec2iNew(10, 10)); // add offset
571 DrawRectangle(
572 device,
573 Vec2iAdd(pos, Vec2iScale(Vec2iUnit(), -TOOLTIP_PADDING)),
574 Vec2iAdd(bgSize, Vec2iScale(Vec2iUnit(), 2 * TOOLTIP_PADDING)),
575 bgColor,
576 0);
577 FontStr(s, pos);
578}
