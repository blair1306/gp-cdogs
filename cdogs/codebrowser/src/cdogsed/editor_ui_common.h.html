80// Macro for creating the helper function to show/hide controls for specific
81// map types
82#define MISSION_CHECK_TYPE_FUNC(_type)\
83static void MissionCheckTypeFunc(UIObject *o, void *data)\
84{\
85 CampaignOptions *co = data;\
86 const Mission *m = CampaignGetCurrentMission(co);\
87 if (!m || m->Type != (_type))\
88 {\
89 o->IsVisible = false;\
90 /* Need to unhighlight to prevent children being drawn*/\
91 UIObjectUnhighlight(o, false);\
92 return;\
93 }\
94 o->IsVisible = true;\
95}
