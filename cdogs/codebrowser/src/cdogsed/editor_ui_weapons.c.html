35typedef enum
36{
37 SELECT_ALL,
38 SELECT_NONE,
39 SELECT_INVERSE
40} SelectMode;
41typedef struct
42{
43 CampaignOptions *co;
44 SelectMode Mode;
45} MissionAllGunsData;
46static bool HasWeapon(const Mission *m, const GunDescription *g);
47static void AddAllWeapons(Mission *m);
48static void MissionSelectWeapons(void *vData, int d)
49{
50 UNUSED(d);
51 MissionAllGunsData *data = vData;
52 Mission *currentMission = CampaignGetCurrentMission(data->co);
53 if (currentMission == NULL) return;
54 switch (data->Mode)
55 {
56 case SELECT_ALL:
57 AddAllWeapons(currentMission);
58 break;
59 case SELECT_NONE:
60 CArrayClear(&currentMission->Weapons);
61 break;
62 case SELECT_INVERSE:
63 {
64 // Add all the weapons not currently available, then delete the first
65 // entries in the weapons array corresponding to the previously
66 // available weapons
67 const int availableWeapons = (int)currentMission->Weapons.size;
68 AddAllWeapons(currentMission);
69 for (int i = 0; i < availableWeapons; i++)
70 {
71 CArrayDelete(&currentMission->Weapons, 0);
72 }
73 }
74 break;
75 default:
76 CASSERT(false, "unknown case");
77 break;
78 }
79}
80static void AddAllWeapons(Mission *m)
81{
82 CA_FOREACH(const GunDescription, g, gGunDescriptions.Guns)
83 if (g->IsRealGun && !HasWeapon(m, g))
84 {
85 CArrayPushBack(&m->Weapons, &g);
86 }
87 CA_FOREACH_END()
88 CA_FOREACH(const GunDescription, g, gGunDescriptions.CustomGuns)
89 if (g->IsRealGun && !HasWeapon(m, g))
90 {
91 CArrayPushBack(&m->Weapons, &g);
92 }
93 CA_FOREACH_END()
94}

96typedef struct
97{
98 CampaignOptions *co;
99 const GunDescription *Gun;
100} MissionGunData;
101static void MissionDrawWeaponStatus(
102 UIObject *o, GraphicsDevice *g, Vec2i pos, void *vData)
103{
104 UNUSED(g);
105 const MissionGunData *data = vData;
106 const Mission *currentMission = CampaignGetCurrentMission(data->co);
107 if (currentMission == NULL) return;
108 const bool hasWeapon = HasWeapon(currentMission, data->Gun);
109 DisplayFlag(
110 Vec2iAdd(pos, o->Pos),
111 data->Gun->name,
112 hasWeapon,
113 UIObjectIsHighlighted(o));
114}
115static void MissionChangeWeapon(void *vData, int d)
116{
117 UNUSED(d);
118 MissionGunData *data = vData;
119 bool hasWeapon = false;
120 int weaponIndex = -1;
121 Mission *currentMission = CampaignGetCurrentMission(data->co);
122 CA_FOREACH(const GunDescription *, desc, currentMission->Weapons)
123 if (data->Gun == *desc)
124 {
125 hasWeapon = true;
126 weaponIndex = _ca_index;
127 break;
128 }
129 CA_FOREACH_END()
130 if (hasWeapon)
131 {
132 CArrayDelete(&currentMission->Weapons, weaponIndex);
133 }
134 else
135 {
136 CArrayPushBack(&currentMission->Weapons, &data->Gun);
137 }
138}
139// Check if the current mission has this gun available already
140static bool HasWeapon(const Mission *m, const GunDescription *g)
141{
142 CA_FOREACH(const GunDescription *, desc, m->Weapons)
143 if (g == *desc)
144 {
145 return true;
146 }
147 CA_FOREACH_END()
148 return false;
149}

151#define HEADER_COLUMN_WIDTH 75
152#define COLUMN_WIDTH 90
153
154static void CreateWeaponSpecialToggleObj(
155 CampaignOptions *co, UIObject *c, const Vec2i pos,
156 const SelectMode mode, char *label);
157static void CreateWeaponToggleObjs(
158 CampaignOptions *co, UIObject *c, const UIObject *o,
159 int *idx, const int rows, CArray *guns);
160UIObject *CreateWeaponObjs(CampaignOptions *co)
161{
162 const int th = FontH();
163 UIObject *c = UIObjectCreate(
164 UITYPE_CONTEXT_MENU, 0, Vec2iZero(), Vec2iZero());
165 c->Flags = UI_ENABLED_WHEN_PARENT_HIGHLIGHTED_ONLY;
166
167 // Add special toggle controls
168 CreateWeaponSpecialToggleObj(co, c, Vec2iZero(), SELECT_ALL, "Select All");
169 CreateWeaponSpecialToggleObj(
170 co, c, Vec2iNew(HEADER_COLUMN_WIDTH, 0), SELECT_NONE, "Select None");
171 CreateWeaponSpecialToggleObj(
172 co, c, Vec2iNew(HEADER_COLUMN_WIDTH * 2, 0), SELECT_INVERSE, "Invert");
173
174 // Create a dummy label that can be clicked to close the context menu
175 CreateCloseLabel(c, Vec2iNew(HEADER_COLUMN_WIDTH * 3, 0));
176
177 // Add a toggle entry for each gun
178 UIObject *o = UIObjectCreate(
179 UITYPE_CUSTOM, 0, Vec2iZero(), Vec2iNew(80, th));
180 o->u.CustomDrawFunc = MissionDrawWeaponStatus;
181 o->ChangeFunc = MissionChangeWeapon;
182 o->ChangeDisablesContext = false;
183 o->Flags = UI_LEAVE_YC;
184 o->ChangesData = true;
185 const int rows = 10;
186 int idx = 0;
187 CreateWeaponToggleObjs(co, c, o, &idx, rows, &gGunDescriptions.Guns);
188 CreateWeaponToggleObjs(co, c, o, &idx, rows, &gGunDescriptions.CustomGuns);
189
190 UIObjectDestroy(o);
191 return c;
192}
193static void CreateWeaponSpecialToggleObj(
194 CampaignOptions *co, UIObject *c, const Vec2i pos,
195 const SelectMode mode, char *label)
196{
197 const int th = FontH();
198 UIObject *o = UIObjectCreate(
199 UITYPE_LABEL, 0, Vec2iZero(), Vec2iNew(HEADER_COLUMN_WIDTH - 10, th));
200 o->Label = label;
201 o->ChangeFunc = MissionSelectWeapons;
202 o->ChangeDisablesContext = false;
203 o->ChangesData = true;
204 CMALLOC(o->Data, sizeof(MissionAllGunsData));
205 o->IsDynamicData = true;
206 ((MissionAllGunsData *)o->Data)->co = co;
207 ((MissionAllGunsData *)o->Data)->Mode = mode;
208 o->Pos = pos;
209 UIObjectAddChild(c, o);
210}
211static void CreateWeaponToggleObjs(
212 CampaignOptions *co, UIObject *c, const UIObject *o,
213 int *idx, const int rows, CArray *guns)
214{
215 const int th = FontH();
216 CA_FOREACH(const GunDescription, g, *guns)
217 if (!g->IsRealGun)
218 {
219 continue;
220 }
221 UIObject *o2 = UIObjectCopy(o);
222 CMALLOC(o2->Data, sizeof(MissionGunData));
223 o2->IsDynamicData = true;
224 ((MissionGunData *)o2->Data)->co = co;
225 ((MissionGunData *)o2->Data)->Gun = g;
226 o2->Pos = Vec2iNew(
227 *idx / rows * COLUMN_WIDTH, (*idx % rows) * th + th * 2);
228 UIObjectAddChild(c, o2);
229 (*idx)++;
230 CA_FOREACH_END()
231}
