38typedef struct
39{
40 CampaignOptions *Campaign;
41 int MissionObjectiveIndex;
42} MissionObjectiveData;
43static char *MissionGetObjectiveDescription(UIObject *o, void *data)
44{
45 MissionObjectiveData *mData = data;
46 Mission *m = CampaignGetCurrentMission(mData->Campaign);
47 if (!m)
48 {
49 return NULL;
50 }
51 int i = mData->MissionObjectiveIndex;
52 if ((int)m->Objectives.size <= i)
53 {
54 if (i == 0)
55 {
56 // first objective and mission has no objectives
57 o->u.Textbox.IsEditable = false;
58 return "-- mission objectives --";
59 }
60 return NULL;
61 }
62 o->u.Textbox.IsEditable = true;
63 return ((const Objective *)CArrayGet(&m->Objectives, i))->Description;
64}
65static void MissionCheckObjectiveDescription(UIObject *o, void *data)
66{
67 MissionObjectiveData *mData = data;
68 Mission *m = CampaignGetCurrentMission(mData->Campaign);
69 if (!m)
70 {
71 o->IsVisible = false;
72 return;
73 }
74 int i = mData->MissionObjectiveIndex;
75 if ((int)m->Objectives.size <= i)
76 {
77 if (i == 0)
78 {
79 // first objective and mission has no objectives
80 o->IsVisible = true;
81 return;
82 }
83 o->IsVisible = false;
84 return;
85 }
86 o->IsVisible = true;
87}
88static char **MissionGetObjectiveDescriptionSrc(void *data)
89{
90 MissionObjectiveData *mData = data;
91 Mission *m = CampaignGetCurrentMission(mData->Campaign);
92 if (!m)
93 {
94 return NULL;
95 }
96 int i = mData->MissionObjectiveIndex;
97 if ((int)m->Objectives.size <= i)
98 {
99 return NULL;
100 }
101 return &((Objective *)CArrayGet(&m->Objectives, i))->Description;
102}
103typedef struct

108static const char *MissionGetObjectiveStr(UIObject *o, void *vData)
109{
110 UNUSED(o);
111 MissionIndexData *data = vData;
112 if (!CampaignGetCurrentMission(data->co)) return NULL;
113 if ((int)CampaignGetCurrentMission(data->co)->Objectives.size <= data->index) return NULL;
114 return ObjectiveTypeStr(((const Objective *)CArrayGet(
115 &CampaignGetCurrentMission(data->co)->Objectives, data->index))->Type);
116}
117static Objective *GetMissionObjective(const Mission *m, const int idx)
118{
119 return CArrayGet(&m->Objectives, idx);
120}
121static const char *MissionGetObjectiveRequired(UIObject *o, void *vData)
122{
123 static char s[128];
124 UNUSED(o);
125 MissionIndexData *data = vData;
126 if (!CampaignGetCurrentMission(data->co)) return NULL;
127 if ((int)CampaignGetCurrentMission(data->co)->Objectives.size <=
128 data->index)
129 {
130 return NULL;
131 }
132 sprintf(s, "%d", GetMissionObjective(
133 CampaignGetCurrentMission(data->co), data->index)->Required);
134 return s;
135}
136static const char *MissionGetObjectiveTotal(UIObject *o, void *vData)
137{
138 static char s[128];
139 UNUSED(o);
140 MissionIndexData *data = vData;
141 if (!CampaignGetCurrentMission(data->co)) return NULL;
142 if ((int)CampaignGetCurrentMission(data->co)->Objectives.size <=
143 data->index)
144 {
145 return NULL;
146 }
147 sprintf(s, "out of %d", GetMissionObjective(
148 CampaignGetCurrentMission(data->co), data->index)->Count);
149 return s;
150}
151static const char *MissionGetObjectiveFlags(UIObject *o, void *vData)
152{
153 int flags;
154 static char s[128];
155 UNUSED(o);
156 MissionIndexData *data = vData;
157 if (!CampaignGetCurrentMission(data->co)) return NULL;
158 if ((int)CampaignGetCurrentMission(data->co)->Objectives.size <=
159 data->index)
160 {
161 return NULL;
162 }
163 flags = GetMissionObjective(
164 CampaignGetCurrentMission(data->co), data->index)->Flags;
165 if (!flags)
166 {
167 return "(normal)";
168 }
169 sprintf(s, "%s %s %s %s %s",
170 (flags & OBJECTIVE_HIDDEN) ? "hidden" : "",
171 (flags & OBJECTIVE_POSKNOWN) ? "pos.known" : "",
172 (flags & OBJECTIVE_HIACCESS) ? "access" : "",
173 (flags & OBJECTIVE_UNKNOWNCOUNT) ? "no-count" : "",
174 (flags & OBJECTIVE_NOACCESS) ? "no-access" : "");
175 return s;
176}

178typedef struct
179{
180 CampaignOptions *C;
181 int ObjectiveIdx;
182 ObjectiveType Type;
183} ObjectiveChangeTypeData;
184static void MissionResetObjectiveIndex(Objective *o);
185static void ObjectiveChangeType(void *vData, int d)
186{
187 UNUSED(d);
188 ObjectiveChangeTypeData *data = vData;
189 Objective *o = GetMissionObjective(
190 CampaignGetCurrentMission(data->C), data->ObjectiveIdx);
191 if (o->Type == data->Type)
192 {
193 return;
194 }
195 o->Type = data->Type;
196 MissionResetObjectiveIndex(o);
197}
198static void MissionResetObjectiveIndex(Objective *o)
199{
200 switch (o->Type)
201 {
202 case OBJECTIVE_COLLECT:
203 o->u.Pickup = IntScorePickupClass(0);
204 break;
205 case OBJECTIVE_DESTROY:
206 {
207 const char **destructibleName =
208 CArrayGet(&gMapObjects.Destructibles, 0);
209 o->u.MapObject = StrMapObject(*destructibleName);
210 }
211 CASSERT(o->u.MapObject != NULL, "cannot find map object");
212 break;
213 default:
214 o->u.Index = 0;
215 break;
216 }
217}
218static void MissionChangeObjectiveRequired(void *vData, int d)
219{
220 MissionIndexData *data = vData;
221 Objective *o = GetMissionObjective(
222 CampaignGetCurrentMission(data->co), data->index);
223 o->Required = CLAMP_OPPOSITE(o->Required + d, 0, MIN(100, o->Count));
224}
225static void MissionChangeObjectiveTotal(void *vData, int d)
226{
227 MissionIndexData *data = vData;
228 const Mission *m = CampaignGetCurrentMission(data->co);
229 Objective *o = GetMissionObjective(m, data->index);
230 o->Count = CLAMP_OPPOSITE(o->Count + d, o->Required, 100);
231 // Don't let the total reduce to less than static ones we've placed
232 if (m->Type == MAPTYPE_STATIC)
233 {
234 CA_FOREACH(const ObjectivePositions, op, m->u.Static.Objectives)
235 if (op->Index == data->index)
236 {
237 o->Count = MAX(o->Count, (int)op->Positions.size);
238 break;
239 }
240 CA_FOREACH_END()
241 }
242}
243static void MissionChangeObjectiveFlags(void *vData, int d)
244{
245 MissionIndexData *data = vData;
246 Objective *o = GetMissionObjective(
247 CampaignGetCurrentMission(data->co), data->index);
248 // Max is combination of all flags, i.e. largest flag doubled less one
249 o->Flags = CLAMP_OPPOSITE(o->Flags + d, 0, OBJECTIVE_NOACCESS * 2 - 1);
250}

253static UIObject *CreateObjectiveObjs(
254 Vec2i pos, CampaignOptions *co, const int idx);
255void CreateObjectivesObjs(CampaignOptions *co, UIObject *c, Vec2i pos)
256{
257 const int th = FontH();
258 Vec2i objectivesPos = Vec2iNew(0, 7 * th);
259 UIObject *o =
260 UIObjectCreate(UITYPE_TEXTBOX, 0, Vec2iZero(), Vec2iNew(300, th));
261 o->Flags = UI_SELECT_ONLY;
262
263 for (int i = 0; i < OBJECTIVE_MAX_OLD; i++)
264 {
265 UIObject *o2 = UIObjectCopy(o);
266 o2->Id = YC_OBJECTIVES + i;
267 o2->Type = UITYPE_TEXTBOX;
268 o2->u.Textbox.TextLinkFunc = MissionGetObjectiveDescription;
269 o2->u.Textbox.TextSourceFunc = MissionGetObjectiveDescriptionSrc;
270 o2->IsDynamicData = 1;
271 CMALLOC(o2->Data, sizeof(MissionObjectiveData));
272 ((MissionObjectiveData *)o2->Data)->Campaign = co;
273 ((MissionObjectiveData *)o2->Data)->MissionObjectiveIndex = i;
274 CSTRDUP(o2->u.Textbox.Hint, "(Objective description)");
275 o2->Pos = pos;
276 CSTRDUP(
277 o2->Tooltip,
278 "Insert/CTRL-i, Delete/CTRL+d: add/remove objective");
279 o2->CheckVisible = MissionCheckObjectiveDescription;
280 UIObjectAddChild(o2, CreateObjectiveObjs(objectivesPos, co, i));
281 UIObjectAddChild(c, o2);
282 pos.y += th;
283 }
284 UIObjectDestroy(o);
285}
286static void CreateObjectiveItemObjs(
287 UIObject *c, const Vec2i pos, CampaignOptions *co, const int idx);
288static UIObject *CreateObjectiveObjs(
289 Vec2i pos, CampaignOptions *co, const int idx)
290{
291 const int th = FontH();
292 UIObject *c;
293 UIObject *o;
294 UIObject *o2;
295 c = UIObjectCreate(UITYPE_NONE, 0, Vec2iZero(), Vec2iZero());
296 c->Flags = UI_ENABLED_WHEN_PARENT_HIGHLIGHTED_ONLY;
297
298 o = UIObjectCreate(UITYPE_LABEL, 0, Vec2iZero(), Vec2iZero());
299 o->Flags = UI_LEAVE_YC;
300 o->ChangesData = 1;
301
302 pos.y -= idx * th;
303 // Drop-down menu for objective type
304 o2 = UIObjectCopy(o);
305 o2->Size = Vec2iNew(35, th);
306 o2->u.LabelFunc = MissionGetObjectiveStr;
307 CMALLOC(o2->Data, sizeof(MissionIndexData));
308 o2->IsDynamicData = true;
309 ((MissionIndexData *)o2->Data)->co = co;
310 ((MissionIndexData *)o2->Data)->index = idx;
311 o2->Pos = pos;
312 CSTRDUP(o2->Tooltip, "Objective type");
313 UIObject *oObjType =
314 UIObjectCreate(UITYPE_CONTEXT_MENU, 0, Vec2iZero(), Vec2iZero());
315 for (int i = 0; i < (int)OBJECTIVE_MAX; i++)
316 {
317 UIObject *oObjTypeChild =
318 UIObjectCreate(UITYPE_LABEL, 0, Vec2iZero(), Vec2iNew(50, th));
319 oObjTypeChild->ChangesData = true;
320 oObjTypeChild->Pos.y = i * th;
321 UIObjectSetDynamicLabel(
322 oObjTypeChild, ObjectiveTypeStr((ObjectiveType)i));
323 oObjTypeChild->IsDynamicData = true;
324 CMALLOC(oObjTypeChild->Data, sizeof(ObjectiveChangeTypeData));
325 ObjectiveChangeTypeData *octd = oObjTypeChild->Data;
326 octd->C = co;
327 octd->ObjectiveIdx = idx;
328 octd->Type = (ObjectiveType)i;
329 oObjTypeChild->ChangeFunc = ObjectiveChangeType;
330 UIObjectAddChild(oObjType, oObjTypeChild);
331 }
332 UIObjectAddChild(o2, oObjType);
333 UIObjectAddChild(c, o2);
334
335 pos.x += 40;
336 // Choose objective object/item
337 CreateObjectiveItemObjs(c, pos, co, idx);
338
339 pos.x += 30;
340 o2 = UIObjectCopy(o);
341 o2->u.LabelFunc = MissionGetObjectiveRequired;
342 o2->ChangeFunc = MissionChangeObjectiveRequired;
343 CMALLOC(o2->Data, sizeof(MissionIndexData));
344 o2->IsDynamicData = 1;
345 ((MissionIndexData *)o2->Data)->co = co;
346 ((MissionIndexData *)o2->Data)->index = idx;
347 o2->Pos = pos;
348 o2->Size = Vec2iNew(20, th);
349 CSTRDUP(o2->Tooltip, "0: optional objective");
350 UIObjectAddChild(c, o2);
351 pos.x += 20;
352 o2 = UIObjectCopy(o);
353 o2->u.LabelFunc = MissionGetObjectiveTotal;
354 o2->ChangeFunc = MissionChangeObjectiveTotal;
355 CMALLOC(o2->Data, sizeof(MissionIndexData));
356 o2->IsDynamicData = 1;
357 ((MissionIndexData *)o2->Data)->co = co;
358 ((MissionIndexData *)o2->Data)->index = idx;
359 o2->Pos = pos;
360 o2->Size = Vec2iNew(40, th);
361 UIObjectAddChild(c, o2);
362 pos.x += 45;
363 o2 = UIObjectCopy(o);
364 o2->u.LabelFunc = MissionGetObjectiveFlags;
365 o2->ChangeFunc = MissionChangeObjectiveFlags;
366 CMALLOC(o2->Data, sizeof(MissionIndexData));
367 o2->IsDynamicData = 1;
368 ((MissionIndexData *)o2->Data)->co = co;
369 ((MissionIndexData *)o2->Data)->index = idx;
370 o2->Pos = pos;
371 o2->Size = Vec2iNew(100, th);
372 CSTRDUP(o2->Tooltip,
373 "hidden: not shown on map\n"
374 "pos.known: always shown on map\n"
375 "access: in locked room\n"
376 "no-count: don't show completed count\n"
377 "no-access: not in locked rooms");
378 UIObjectAddChild(c, o2);
379
380 UIObjectDestroy(o);
381 return c;
382}

460static void MissionDrawKillObjective(
461 UIObject *o, GraphicsDevice *g, Vec2i pos, void *vData)
462{
463 MissionIndexData *data = vData;
464 UNUSED(g);
465 const Mission *m = CampaignGetCurrentMission(data->co);
466 if (m == NULL) return;
467 if ((int)m->Objectives.size <= data->index) return;
468 // TODO: only one kill and rescue objective allowed
469 CharacterStore *store = &data->co->Setting.characters;
470 if (store->specialIds.size > 0)
471 {
472 Character *c = CArrayGet(
473 &store->OtherChars, CharacterStoreGetSpecialId(store, 0));
474 const Vec2i drawPos =
475 Vec2iAdd(Vec2iAdd(pos, o->Pos), Vec2iScaleDiv(o->Size, 2));
476 DrawCharacterSimple(c, drawPos, DIRECTION_DOWN, false, true);
477 }
478}
479static void MissionDrawCollectObjective(
480 UIObject *o, GraphicsDevice *g, Vec2i pos, void *vData)
481{
482 MissionIndexData *data = vData;
483 UNUSED(g);
484 const Mission *m = CampaignGetCurrentMission(data->co);
485 if (m == NULL) return;
486 if ((int)m->Objectives.size <= data->index) return;
487 // TODO: only one kill and rescue objective allowed
488 const Objective *obj = CArrayGet(&m->Objectives, data->index);
489 const Pic *newPic = obj->u.Pickup->Pic;
490 const Vec2i drawPos =
491 Vec2iAdd(Vec2iAdd(pos, o->Pos), Vec2iScaleDiv(o->Size, 2));
492 Blit(g, newPic, Vec2iMinus(drawPos, Vec2iScaleDiv(newPic->size, 2)));
493}
