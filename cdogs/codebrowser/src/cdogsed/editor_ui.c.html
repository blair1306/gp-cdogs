513static const char *GetWeaponCountStr(UIObject *o, void *v)
514{
515 static char s[128];
516 UNUSED(o);
517 UNUSED(v);
518 Mission *m = CampaignGetCurrentMission(&gCampaign);
519 if (!m)
520 {
521 return NULL;
522 }
523 int totalWeapons = 0;
524 CA_FOREACH(const GunDescription, g, gGunDescriptions.Guns)
525 if (g->IsRealGun)
526 {
527 totalWeapons++;
528 }
529 CA_FOREACH_END()
530 CA_FOREACH(const GunDescription, g, gGunDescriptions.CustomGuns)
531 if (g->IsRealGun)
532 {
533 totalWeapons++;
534 }
535 CA_FOREACH_END()
536 sprintf(
537 s, "Available weapons (%d/%d)",
538 (int)gMission.missionData->Weapons.size, totalWeapons);
539 return s;
540}

561static void MissionDrawEnemy(
562 UIObject *o, GraphicsDevice *g, Vec2i pos, void *vData)
563{
564 UNUSED(g);
565 MissionIndexData *data = vData;
566 if (!CampaignGetCurrentMission(data->co)) return;
567 if (data->index >= (int)CampaignGetCurrentMission(data->co)->Enemies.size)
568 {
569 return;
570 }
571 const CharacterStore *store = &data->co->Setting.characters;
572 const int charIndex = *(int *)CArrayGet(&store->baddieIds, data->index);
573 DrawCharacterSimple(
574 CArrayGet(&store->OtherChars, charIndex),
575 Vec2iAdd(Vec2iAdd(pos, o->Pos), Vec2iScaleDiv(o->Size, 2)),
576 DIRECTION_DOWN, UIObjectIsHighlighted(o), true);
577}

608 mod,
609 UIObjectIsHighlighted(o));
610}
611static void DrawStyleArea(
612 Vec2i pos,
613 const char *name,
614 const Pic *pic,
615 int idx, int count,
616 int isHighlighted)
617{
618 FontStrMask(name, pos, isHighlighted ? colorRed : colorWhite);

620 Blit(&gGraphicsDevice, pic, pos);
621 // Display style index and count, right aligned
622 char buf[16];
623 sprintf(buf, "%d/%d", idx + 1, count);
624 FontStrMask(
625 buf, Vec2iNew(pos.x + 28 - FontStrW(buf), pos.y + 17), colorGray);
626}

1018static void DrawBackground(
1019 UIObject *o, GraphicsDevice *g, Vec2i pos, void *data)
1020{
1021 UNUSED(data);
1022 // Only draw background over completely transparent pixels
1023 const color_t c = { 32, 32, 64, 196 };
1024 const Uint32 p = COLOR2PIXEL(c);
1025 Vec2i v;
1026 for (v.y = 0; v.y < o->Size.y; v.y++)
1027 {
1028 for (v.x = 0; v.x < o->Size.x; v.x++)
1029 {
1030 const Vec2i pos1 = Vec2iAdd(pos, v);
1031 if (pos1.x < g->clipping.left || pos1.x > g->clipping.right ||
1032 pos1.y < g->clipping.top || pos1.y > g->clipping.bottom)
1033 {
1034 continue;
1035 }
1036 const int idx = PixelIndex(
1037 pos1.x, pos1.y, g->cachedConfig.Res.x, g->cachedConfig.Res.y);
1038 const color_t existing = PIXEL2COLOR(g->buf[idx]);
1039 if (existing.a != 0)
1040 {
1041 continue;
1042 }
1043 g->buf[idx] = p;
1044 }
1045 }
