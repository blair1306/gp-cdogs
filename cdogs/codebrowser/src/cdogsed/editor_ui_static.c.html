46static const char *BrushGetTypeStr(EditorBrush *brush, int isMain)
47{
48 static char s[128];
49 sprintf(s, "Brush %d: %s",
50 isMain ? 1 : 2,
51 IMapTypeStr(isMain ? brush->MainType : brush->SecondaryType));
52 return s;
53}

90static void BrushChangeType(EditorBrush *b, int d, int isMain)
91{
92 unsigned short brushType = isMain ? b->MainType : b->SecondaryType;
93 brushType = (unsigned short)CLAMP_OPPOSITE(
94 (int)brushType + d, MAP_FLOOR, MAP_NOTHING);
95 if (isMain)
96 {
97 b->MainType = brushType;
98 }
99 else
100 {
101 b->SecondaryType = brushType;
102 }
103}
104static void BrushChangeMainType(void *data, int d)
105{
106 BrushChangeType(data, d, 1);
107}
108static void BrushChangeSecondaryType(void *data, int d)
109{
110 BrushChangeType(data, d, 0);
111}
112static void BrushChangeSize(void *data, int d)
113{
114 EditorBrush *b = data;
115 b->BrushSize = CLAMP(b->BrushSize + d, 1, 5);
116}
117static void BrushLoadGuideImage(void *data, int d)
118{
119 UNUSED(d);
120 EditorBrush *b = data;
121 b->IsGuideImageNew = true;
122 SDL_FreeSurface(b->GuideImageSurface);
123 SDL_Surface *s = IMG_Load(b->GuideImage);
124 if (s == NULL) return;
125 b->GuideImageSurface = SDL_ConvertSurface(s, gGraphicsDevice.Format, 0);
126 SDL_FreeSurface(s);
127}
128static void BrushChangeGuideImageAlpha(void *data, int d)
129{
130 EditorBrush *b = data;
131 b->IsGuideImageNew = true;
132 d *= 4;
133 if (gEventHandlers.keyboard.modState & KMOD_SHIFT)
134 {
135 d *= 8;
136 }
137 b->GuideImageAlpha = (Uint8)CLAMP((int)b->GuideImageAlpha + d, 0, 255);
138}

450 o2->ChangeFunc = BrushChangeGuideImageAlpha;
451 o2->Pos = pos;
452 UIObjectAddChild(c, o2);
453
454 UIObjectDestroy(o);
455 return c;
456}

458static UIObject *CreateSetKeyObjs(Vec2i pos, EditorBrush *brush)
459{
460 UIObject *o2;
461 UIObject *c = UIObjectCreate(UITYPE_CONTEXT_MENU, 0, pos, Vec2iZero());
462
463 UIObject *o = UIObjectCreate(
464 UITYPE_CUSTOM, 0,
465 Vec2iZero(), Vec2iNew(TILE_WIDTH + 4, TILE_HEIGHT + 4));
466 o->ChangeFunc = BrushSetBrushTypeSetKey;
467 o->u.CustomDrawFunc = DrawKey;
468 o->OnFocusFunc = ActivateIndexedEditorBrush;
469 o->OnUnfocusFunc = DeactivateIndexedEditorBrush;
470 pos = Vec2iZero();
471 for (int i = -1; i < KEY_COUNT; i++)
472 {
473 o2 = UIObjectCopy(o);
474 o2->IsDynamicData = 1;
475 CMALLOC(o2->Data, sizeof(IndexedEditorBrush));
476 ((IndexedEditorBrush *)o2->Data)->Brush = brush;
477 ((IndexedEditorBrush *)o2->Data)->u.ItemIndex = i;
478 o2->Pos = pos;
479 if (i == -1)
480 {
481 // -1 means no key
482 CSTRDUP(o2->Tooltip, "no key");
483 }
484 UIObjectAddChild(c, o2);
485 pos.x += o->Size.x;
486 }
487
488 UIObjectDestroy(o);
489 return c;
490}
