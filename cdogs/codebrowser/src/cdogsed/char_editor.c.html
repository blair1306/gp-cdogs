97void CharEditor(
98 GraphicsDevice *g, CampaignSetting *setting, EventHandlers *handlers,
99 bool *fileChanged)
100{
101 SDL_SetHint(SDL_HINT_VIDEO_HIGHDPI_DISABLED, "1");
102 SDL_Init(SDL_INIT_VIDEO);
103 SDL_GL_SetAttribute(SDL_GL_DOUBLEBUFFER, 1);
104 SDL_GL_SetAttribute(SDL_GL_DEPTH_SIZE, 24);
105 SDL_GL_SetAttribute(SDL_GL_STENCIL_SIZE, 8);
106 SDL_GL_SetAttribute(SDL_GL_CONTEXT_MAJOR_VERSION, 2);
107 SDL_GL_SetAttribute(SDL_GL_CONTEXT_MINOR_VERSION, 2);
108
109 SDL_Window *win = SDL_CreateWindow("Character Editor",
110 SDL_WINDOWPOS_CENTERED, SDL_WINDOWPOS_CENTERED,
111 800, 600,
112 SDL_WINDOW_OPENGL|SDL_WINDOW_SHOWN|SDL_WINDOW_RESIZABLE);
113 SDL_SetWindowMinimumSize(win, 800, 500);
114 SDL_SetWindowIcon(win, g->icon);
115
116 SDL_GLContext glContext = SDL_GL_CreateContext(win);
117 glEnable(GL_TEXTURE_2D);
118 glEnable(GL_BLEND);
119 glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
120
121 // Initialise editor context
122 EditorContext ec;
123 ec.ctx = nk_sdl_init(win);
124 ec.ctx->style.checkbox.hover.data.color = nk_rgb(96, 96, 96);
125 ec.ctx->style.checkbox.normal.data.color = nk_rgb(64, 64, 64);
126 ec.ctx->style.checkbox.cursor_hover.data.color = nk_rgb(255, 255, 255);
127 ec.ctx->style.checkbox.cursor_normal.data.color = nk_rgb(200, 200, 200);
128 ec.Char = NULL;
129 ec.Setting = setting;
130 ec.Handlers = handlers;
131 ec.FileChanged = fileChanged;
132 ec.CharacterClassNames = GetClassNames(
133 NumCharacterClasses(), IndexCharacterClassName);
134 ec.GunNames = GetClassNames(NumGuns(), IndexGunName);
135
136 CArrayInit(&ec.texidsChars, sizeof(GLuint) * BODY_PART_COUNT);
137 for (int i = 0; i < (int)setting->characters.OtherChars.size; i++)
138 {
139 AddCharacterTextures(&ec);
140 }
141 glGenTextures(BODY_PART_COUNT, ec.texidsPreview);
142 CArrayInit(&ec.texIdsCharClasses, sizeof(GLuint));
143 CArrayResize(&ec.texIdsCharClasses, NumCharacterClasses(), NULL);
144 glGenTextures(NumCharacterClasses(), (GLuint *)ec.texIdsCharClasses.data);
145 CharColors cc;
146 cc.Skin = colorSkin;
147 cc.Hair = colorRed;
148 for (int i = 0; i < NumCharacterClasses(); i++)
149 {
150 const GLuint *texid = CArrayGet(&ec.texIdsCharClasses, i);
151 const CharacterClass *c = IndexCharacterClass(i);
152 LoadMultiChannelTexFromPic(
153 *texid, GetHeadPic(c, DIRECTION_DOWN, GUNSTATE_READY), &cc);
154 }
155 CArrayInit(&ec.texIdsGuns, sizeof(GLuint));
156 CArrayResize(&ec.texIdsGuns, NumGuns(), NULL);
157 glGenTextures(NumGuns(), (GLuint *)ec.texIdsGuns.data);
158 for (int i = 0; i < NumGuns(); i++)
159 {
160 const GLuint *texid = CArrayGet(&ec.texIdsGuns, i);
161 const GunDescription *gd = IndexGunDescriptionReal(i);
162 LoadTexFromPic(*texid, gd->Icon);
163 }
164
165 ec.anim = AnimationGetActorAnimation(ACTORANIMATION_WALKING);
166 ec.previewDir = DIRECTION_DOWN;
167 ec.animSelection = AnimationGetActorAnimation(ACTORANIMATION_IDLE);
168
169 // Initialise fonts
170 struct nk_font_atlas *atlas;
171 nk_sdl_font_stash_begin(&atlas);
172 nk_sdl_font_stash_end();
173
174 Uint32 ticksNow = SDL_GetTicks();
175 Uint32 ticksElapsed = 0;
176 for (;;)
177 {
178 Uint32 ticksThen = ticksNow;
179 ticksNow = SDL_GetTicks();
180 ticksElapsed += ticksNow - ticksThen;
181 if (ticksElapsed < 1000 / FPS_FRAMELIMIT)
182 {
183 SDL_Delay(1);
184 continue;
185 }
186
187 // Note: drawing contains input processing too
188 nk_input_begin(ec.ctx);
189 if (!HandleEvents(&ec))
190 {
191 goto bail;
192 }
193 Draw(win, &ec);
194 nk_input_end(ec.ctx);
195
196 ticksElapsed = 0;
197 }
198
199bail:
200 nk_sdl_shutdown();
201 CFREE(ec.CharacterClassNames);
202 CFREE(ec.GunNames);
203 glDeleteTextures(
204 BODY_PART_COUNT * ec.texidsChars.size, ec.texidsChars.data);
205 CArrayTerminate(&ec.texidsChars);
206 glDeleteTextures(BODY_PART_COUNT, ec.texidsPreview);
207 glDeleteTextures(
208 ec.texIdsCharClasses.size, (const GLuint *)ec.texIdsCharClasses.data);
209 CArrayTerminate(&ec.texIdsCharClasses);
210 glDeleteTextures(ec.texIdsGuns.size, (const GLuint *)ec.texIdsGuns.data);
211 CArrayTerminate(&ec.texIdsGuns);
212 SDL_GL_DeleteContext(glContext);
213 SDL_DestroyWindow(win);
214}

216static char *GetClassNames(const int len, const char *(*indexNameFunc)(int))
217{
218 int classLen = 0;
219 for (int i = 0; i < (int)len; i++)
220 {
221 const char *name = indexNameFunc(i);
222 classLen += strlen(name) + 1;
223 }
224 if (classLen == 0)
225 {
226 return "";
227 }
228 char *names;
229 CMALLOC(names, classLen);
230 char *cp = names;
231 for (int i = 0; i < (int)len; i++)
232 {
233 const char *name = indexNameFunc(i);
234 CASSERT(name != NULL, "Class has no name");
235 strcpy(cp, name);
236 cp += strlen(name) + 1;
237 }
238 return names;
239}

241static const char *IndexCharacterClassName(const int i)

246static int NumCharacterClasses(void)
247{
248 return
249 gCharacterClasses.Classes.size + gCharacterClasses.CustomClasses.size;
250}
251static const char *IndexGunName(const int i)
252{
253 const GunDescription *g = IndexGunDescriptionReal(i);
254 return g ? g->name : NULL;
255}
256static int NumGuns(void)
257{
258 int totalWeapons = 0;
259 CA_FOREACH(const GunDescription, g, gGunDescriptions.Guns)
260 if (g->IsRealGun)
261 {
262 totalWeapons++;
263 }
264 CA_FOREACH_END()
265 CA_FOREACH(const GunDescription, g, gGunDescriptions.CustomGuns)
266 if (g->IsRealGun)
267 {
268 totalWeapons++;
269 }
270 CA_FOREACH_END()
271 return totalWeapons;
272}
273static int GunIndex(const GunDescription *g)
274{
275 int j = 0;
276 CA_FOREACH(const GunDescription, gg, gGunDescriptions.Guns)
277 if (!gg->IsRealGun)
278 {
279 continue;
280 }
281 if (g == gg)
282 {
283 return j;
284 }
285 j++;
286 CA_FOREACH_END()
287 CA_FOREACH(const GunDescription, gg, gGunDescriptions.CustomGuns)
288 if (!g->IsRealGun)
289 {
290 continue;
291 }
292 if (g == gg)
293 {
294 return j;
295 }
296 j++;
297 CA_FOREACH_END()
298 CASSERT(false, "cannot find gun");
299 return -1;
300}

302static bool HandleEvents(EditorContext *ec)
303{
304 SDL_Event e;
305 bool run = true;
306 while (SDL_PollEvent(&e))
307 {
308 switch (e.type)
309 {
310 case SDL_KEYDOWN:
311 if (e.key.repeat)
312 {
313 break;
314 }
315 KeyOnKeyDown(&ec->Handlers->keyboard, e.key.keysym);
316 break;
317 case SDL_KEYUP:
318 KeyOnKeyUp(&ec->Handlers->keyboard, e.key.keysym);
319 break;
320 case SDL_QUIT:
321 run = false;
322 break;
323 case SDL_WINDOWEVENT:
324 switch (e.window.event)
325 {
326 case SDL_WINDOWEVENT_CLOSE:
327 run = false;
328 break;
329 default:
330 break;
331 }
332 break;
333 default:
334 break;
335 }
336 nk_sdl_handle_event(&e);
337 }
338
339 EventPoll(ec->Handlers, 1);
340 const SDL_Scancode sc = KeyGetPressed(&gEventHandlers.keyboard);
341 if (sc == SDL_SCANCODE_ESCAPE)
342 {
343 run = false;
344 }
345
346 return run;
347}

362static void Draw(SDL_Window *win, EditorContext *ec)
363{
364 // Stretch char store with window
365 SDL_GetWindowSize(win, &ec->WindowSize.x, &ec->WindowSize.y);
366 const Vec2i charStoreSize = Vec2iNew(
367 MAX(400, ec->WindowSize.x - 100),
368 MAX(200, ec->WindowSize.y - 300));
369 const float pad = 10;
370 if (nk_begin(ec->ctx, "Character Store",
371 nk_rect(pad, pad, (float)charStoreSize.x - pad, (float)charStoreSize.y - pad),
372 NK_WINDOW_BORDER|NK_WINDOW_TITLE))
373 {
374 int selectedIndex = -1;
375 CA_FOREACH(Character, c, ec->Setting->characters.OtherChars)
376 if (ec->Char == c)
377 {
378 selectedIndex = _ca_index;
379 }
380 CA_FOREACH_END()
381
382 // TODO: keep buttons from scrolling off
383 nk_layout_row_dynamic(ec->ctx, ROW_HEIGHT, 5);
384 if (nk_button_label(ec->ctx, "Add"))
385 {
386 AddCharacter(ec, -1);
387 selectedIndex = MAX(
388 (int)ec->Setting->characters.OtherChars.size - 1, 0);
389 }
390 if (nk_button_label(ec->ctx, "Move Up"))
391 {
392 selectedIndex = MoveCharacter(ec, selectedIndex, -1);
393 }
394 if (nk_button_label(ec->ctx, "Move Down"))
395 {
396 selectedIndex = MoveCharacter(ec, selectedIndex, 1);
397 }
398 if (selectedIndex >= 0 && nk_button_label(ec->ctx, "Duplicate"))
399 {
400 AddCharacter(ec, selectedIndex);
401 selectedIndex = MAX(
402 (int)ec->Setting->characters.OtherChars.size - 1, 0);
403 }
404 if (selectedIndex >= 0 && nk_button_label(ec->ctx, "Remove"))
405 {
406 DeleteCharacter(ec, selectedIndex);
407 selectedIndex = MIN(
408 selectedIndex,
409 (int)ec->Setting->characters.OtherChars.size - 1);
410 }
411 if (selectedIndex >= 0)
412 {
413 ec->Char = CArrayGet(
414 &ec->Setting->characters.OtherChars, selectedIndex);
415 }
416 else
417 {
418 ec->Char = NULL;
419 }
420
421 // Show existing characters
422 nk_layout_row_dynamic(
423 ec->ctx, 32 * PIC_SCALE, (int)charStoreSize.x / 75);
424 CA_FOREACH(Character, c, ec->Setting->characters.OtherChars)
425 const int selected = ec->Char == c;
426 // show both label and full character
427 if (nk_select_label(ec->ctx, c->Gun->name,
428 NK_TEXT_ALIGN_BOTTOM|NK_TEXT_ALIGN_CENTERED, selected))
429 {
430 ec->Char = c;
431 }
432 DrawCharacter(
433 ec, c, CArrayGet(&ec->texidsChars, _ca_index),
434 Vec2iNew(-34, 5), &ec->animSelection, DIRECTION_DOWN);
435 CA_FOREACH_END()
436 }
437 nk_end(ec->ctx);
438
439 if (ec->Char != NULL)
440 {
441 const float previewWidth = 80;
442 if (nk_begin(ec->ctx, "Preview",
443 nk_rect(charStoreSize.x + pad, pad,
444 previewWidth, charStoreSize.y - pad),
445 NK_WINDOW_BORDER|NK_WINDOW_TITLE))
446 {
447 // Preview direction
448 nk_layout_row_dynamic(ec->ctx, ROW_HEIGHT, 2);
449 if (nk_button_label(ec->ctx, "<"))
450 {
451 ec->previewDir = (direction_e)CLAMP_OPPOSITE(
452 (int)ec->previewDir + 1, 0, DIRECTION_UPLEFT);
453 }
454 if (nk_button_label(ec->ctx, ">"))
455 {
456 ec->previewDir = (direction_e)CLAMP_OPPOSITE(
457 (int)ec->previewDir - 1, 0, DIRECTION_UPLEFT);
458 }
459 // Preview
460 nk_layout_row_dynamic(ec->ctx, 32 * PIC_SCALE, 1);
461 DrawCharacter(
462 ec, ec->Char, ec->texidsPreview, Vec2iNew(0, 5), &ec->anim,
463 ec->previewDir);
464 // Animation
465 nk_layout_row_dynamic(ec->ctx, ROW_HEIGHT, 1);
466 const int isWalking = ec->anim.Type == ACTORANIMATION_WALKING;
467 if (nk_select_label(
468 ec->ctx, "Run", NK_TEXT_ALIGN_LEFT, isWalking) &&
469 !isWalking)
470 {
471 ec->anim = AnimationGetActorAnimation(ACTORANIMATION_WALKING);
472 }
473 const int isIdle = ec->anim.Type == ACTORANIMATION_IDLE;
474 if (nk_select_label(ec->ctx, "Idle", NK_TEXT_ALIGN_LEFT, isIdle) &&
475 !isIdle)
476 {
477 ec->anim = AnimationGetActorAnimation(ACTORANIMATION_IDLE);
478 }
479 }
480 nk_end(ec->ctx);
481
482 if (nk_begin(ec->ctx, "Appearance",
483 nk_rect(pad, (float)charStoreSize.y + pad, 260, 225),
484 NK_WINDOW_BORDER|NK_WINDOW_TITLE))
485 {
486 nk_layout_row(ec->ctx, NK_DYNAMIC, ROW_HEIGHT, 2, colRatios);
487 const int selectedClass = DrawClassSelection(
488 ec, "Class:", ec->texIdsCharClasses.data,
489 ec->CharacterClassNames,
490 CharacterClassIndex(ec->Char->Class), NumCharacterClasses());
491 ec->Char->Class = IndexCharacterClass(selectedClass);
492
493 // Character colours
494 nk_layout_row(ec->ctx, NK_DYNAMIC, ROW_HEIGHT, 2, colRatios);
495 DrawCharColor(ec, "Skin:", &ec->Char->Colors.Skin);
496 DrawCharColor(ec, "Hair:", &ec->Char->Colors.Hair);
497 DrawCharColor(ec, "Arms:", &ec->Char->Colors.Arms);
498 DrawCharColor(ec, "Body:", &ec->Char->Colors.Body);
499 DrawCharColor(ec, "Legs:", &ec->Char->Colors.Legs);
500 }
501 nk_end(ec->ctx);
502
503 if (nk_begin(ec->ctx, "Attributes",
504 nk_rect(280, (float)charStoreSize.y + pad, 250, 225),
505 NK_WINDOW_BORDER|NK_WINDOW_TITLE))
506 {
507 // Speed (256 = 100%)
508 nk_layout_row_dynamic(ec->ctx, ROW_HEIGHT, 1);
509 nk_property_int(
510 ec->ctx, "Speed:", 0, &ec->Char->speed, 1024, 16, 1);
511
512 nk_layout_row(ec->ctx, NK_DYNAMIC, ROW_HEIGHT, 2, colRatios);
513 const int selectedGun = DrawClassSelection(
514 ec, "Gun:", ec->texIdsGuns.data, ec->GunNames,
515 GunIndex(ec->Char->Gun), NumGuns());
516 ec->Char->Gun = IndexGunDescriptionReal(selectedGun);
517
518 nk_layout_row_dynamic(ec->ctx, ROW_HEIGHT, 1);
519 nk_property_int(
520 ec->ctx, "Max Health:", 10, &ec->Char->maxHealth, 1000, 10, 1);
521
522 nk_layout_row_dynamic(ec->ctx, ROW_HEIGHT, 2);
523 DrawFlag(ec, "Asbestos", FLAGS_ASBESTOS, "Immune to fire");
524 DrawFlag(ec, "Immunity", FLAGS_IMMUNITY, "Immune to poison");
525 DrawFlag(ec, "See-through", FLAGS_SEETHROUGH, NULL);
526 DrawFlag(ec, "Invulnerable", FLAGS_INVULNERABLE, NULL);
527 DrawFlag(
528 ec, "Penalty", FLAGS_PENALTY, "Large score penalty when shot");
529 DrawFlag(ec, "Victim", FLAGS_VICTIM, "Takes damage from everyone");
530 }
531 nk_end(ec->ctx);
532
533 if (nk_begin(ec->ctx, "AI",
534 nk_rect(540, (float)charStoreSize.y + pad, 250, 280),
535 NK_WINDOW_BORDER|NK_WINDOW_TITLE))
536 {
537 nk_layout_row_dynamic(ec->ctx, ROW_HEIGHT, 1);
538 nk_property_int(
539 ec->ctx, "Move (%):", 0, &ec->Char->bot->probabilityToMove,
540 100, 5, 1);
541 nk_property_int(
542 ec->ctx, "Track (%):", 0, &ec->Char->bot->probabilityToTrack,
543 100, 5, 1);
544 nk_property_int(
545 ec->ctx, "Shoot (%):", 0, &ec->Char->bot->probabilityToShoot,
546 100, 5, 1);
547 nk_property_int(
548 ec->ctx, "Action delay:", 0, &ec->Char->bot->actionDelay,
549 50, 5, 1);
550
551 nk_layout_row_dynamic(ec->ctx, ROW_HEIGHT, 2);
552 DrawFlag(ec, "Runs away", FLAGS_RUNS_AWAY, "Runs away from player");
553 DrawFlag(
554 ec, "Sneaky", FLAGS_SNEAKY, "Shoots back when player shoots");
555 DrawFlag(ec, "Good guy", FLAGS_GOOD_GUY, "Same team as players");
556 DrawFlag(
557 ec, "Sleeping", FLAGS_SLEEPING, "Doesn't move unless seen");
558 DrawFlag(
559 ec, "Prisoner", FLAGS_PRISONER, "Doesn't move until touched");
560 DrawFlag(ec, "Follower", FLAGS_FOLLOWER, "Follows players");
561 DrawFlag(
562 ec, "Awake", FLAGS_AWAKEALWAYS,
563 "Don't go to sleep after players leave");
564 }
565 nk_end(ec->ctx);
566 }
567
568 AnimationUpdate(&ec->anim, 1);
569 AnimationUpdate(&ec->animSelection, 1);
570
571 int winWidth, winHeight;
572 SDL_GetWindowSize(win, &winWidth, &winHeight);
573 glViewport(0, 0, winWidth, winHeight);
574 glClear(GL_COLOR_BUFFER_BIT);
575 glClearColor(bg[0], bg[1], bg[2], bg[3]);
576
577 nk_sdl_render(NK_ANTI_ALIASING_ON);
578 SDL_GL_SwapWindow(win);
579}

581static void AddCharacter(EditorContext *ec, const int cloneIdx)
582{
583 ec->Char = CharacterStoreAddOther(&ec->Setting->characters);
584 if (cloneIdx >= 0)
585 {
586 const Character *clone =
587 CArrayGet(&ec->Setting->characters.OtherChars, cloneIdx);
588 CFREE(ec->Char->bot);
589 memcpy(ec->Char, clone, sizeof *ec->Char);
590 CMALLOC(ec->Char->bot, sizeof *ec->Char->bot);
591 memcpy(ec->Char->bot, clone->bot, sizeof *ec->Char->bot);
592 }
593 else
594 {
595 // set up character template
596 ec->Char->Class = StrCharacterClass("Ogre");
597 ec->Char->Colors.Skin = colorGreen;
598 const color_t darkGray = {64, 64, 64, 255};
599 ec->Char->Colors.Arms = darkGray;
600 ec->Char->Colors.Body = darkGray;
601 ec->Char->Colors.Legs = darkGray;
602 ec->Char->Colors.Hair = colorBlack;
603 ec->Char->speed = 256;
604 ec->Char->Gun = StrGunDescription("Machine gun");
605 ec->Char->maxHealth = 40;
606 ec->Char->flags = FLAGS_IMMUNITY;
607 ec->Char->bot->probabilityToMove = 50;
608 ec->Char->bot->probabilityToTrack = 25;
609 ec->Char->bot->probabilityToShoot = 2;
610 ec->Char->bot->actionDelay = 15;
611 }
612
613 AddCharacterTextures(ec);
614
615 *ec->FileChanged = true;
616}

625static int MoveCharacter(
626 EditorContext *ec, const int selectedIndex, const int d)
627{
628 const int moveIndex = selectedIndex + d;
629 CArray *chars = &ec->Setting->characters.OtherChars;
630 if (moveIndex < 0 || moveIndex >= (int)chars->size)
631 {
632 return selectedIndex;
633 }
634
635 Character tmp;
636 Character *selected = CArrayGet(chars, selectedIndex);
637 Character *move = CArrayGet(chars, moveIndex);
638 memcpy(&tmp, selected, chars->elemSize);
639 memcpy(selected, move, chars->elemSize);
640 memcpy(move, &tmp, chars->elemSize);
641 return moveIndex;
642}

644static void DeleteCharacter(EditorContext *ec, const int selectedIndex)
645{
646 CharacterStoreDeleteOther(&ec->Setting->characters, selectedIndex);
647 const int indexClamped = MIN(
648 selectedIndex,
649 (int)ec->Setting->characters.OtherChars.size - 1);
650 if (indexClamped >= 0)
651 {
652 ec->Char = CArrayGet(
653 &ec->Setting->characters.OtherChars, indexClamped);
654 }
655 else
656 {
657 ec->Char = NULL;
658 }
659
660 // Delete character textures
661 GLuint *texids = CArrayGet(&ec->texidsChars, selectedIndex);
662 glDeleteTextures(BODY_PART_COUNT, texids);
663 CArrayDelete(&ec->texidsChars, selectedIndex);
664
665 *ec->FileChanged = true;
666}

672static int DrawClassSelection(
673 EditorContext *ec, const char *label, const GLuint *texids,
674 const char *items, const int selected, const size_t len)
675{
676 nk_label(ec->ctx, label, NK_TEXT_LEFT);
677 const int selectedNew = nk_combo_separator_image(
678 ec->ctx, texids, items, '\0', selected, len,
679 ROW_HEIGHT, nk_vec2(nk_widget_width(ec->ctx), 8 * ROW_HEIGHT));

686static int nk_combo_separator_image(struct nk_context *ctx,
687 const GLuint *img_ids, const char *items_separated_by_separator,
688 int separator, int selected, int count, int item_height,
689 struct nk_vec2 size)
690{
691 int i;
692 int max_height;
693 struct nk_vec2 item_spacing;
694 struct nk_vec2 window_padding;
695 const char *current_item;
696 const char *iter;
697 int length = 0;
698
699 NK_ASSERT(ctx);
700 NK_ASSERT(items_separated_by_separator);
701 if (!ctx || !items_separated_by_separator)
702 return selected;
703
704 /* calculate popup window */
705 item_spacing = ctx->style.window.spacing;
706 window_padding = nk_panel_get_padding(&ctx->style, ctx->current->layout->type);
707 max_height = count * item_height + count * (int)item_spacing.y;
708 max_height += (int)item_spacing.y * 2 + (int)window_padding.y * 2;
709 size.y = NK_MIN(size.y, (float)max_height);
710
711 /* find selected item */
712 current_item = items_separated_by_separator;
713 for (i = 0; i < count; ++i) {
714 iter = current_item;
715 while (*iter && *iter != separator) iter++;
716 length = (int)(iter - current_item);
717 if (i == selected) break;
718 current_item = iter + 1;
719 }
720
721 // Get widget bounds for drawing currently selected item image later
722 struct nk_rect bounds;
723 nk_layout_widget_space(&bounds, ctx, ctx->current, nk_false);
724 bounds.x += size.x - 56;
725 bounds.y += 2;
726 bounds.w = (float)12 * PIC_SCALE;
727 bounds.h = (float)12 * PIC_SCALE;
728
729 if (nk_combo_begin_text(ctx, current_item, length, size)) {
730 current_item = items_separated_by_separator;
731 nk_layout_row_dynamic(ctx, (float)item_height, 1);
732 for (i = 0; i < count; ++i) {
733 const struct nk_image img = nk_image_id(img_ids[i]);
734 // TODO: image size
735 BeforeDrawTex(img_ids[i]);
736 iter = current_item;
737 while (*iter && *iter != separator) iter++;
738 length = (int)(iter - current_item);
739 if (nk_contextual_item_image_text(ctx, img, current_item, length, NK_TEXT_LEFT))
740 selected = i;
741 current_item = current_item + length + 1;
742 }
743 nk_combo_end(ctx);
744 }
745
746 // Also draw currently selected image
747 const struct nk_image comboImg = nk_image_id(img_ids[selected]);
748 BeforeDrawTex(img_ids[selected]);
749 nk_draw_image(&ctx->current->buffer, bounds, &comboImg, nk_white);
750
751 return selected;
752}

791static void LoadTexFromPic(const GLuint texid, const Pic *pic)
792{
793 glBindTexture(GL_TEXTURE_2D, texid);
794 glTexImage2D(
795 GL_TEXTURE_2D, 0, GL_RGBA, pic->size.x, pic->size.y, 0, GL_BGRA,
796 GL_UNSIGNED_BYTE, pic->Data);
797}
798static void LoadMultiChannelTexFromPic(
799 const GLuint texid, const Pic *pic, const CharColors *colors)
800{
801 glBindTexture(GL_TEXTURE_2D, texid);
802 Uint32 *data;
803 CMALLOC(data, pic->size.x * pic->size.y * sizeof *data);
804 for (int i = 0; i < pic->size.x * pic->size.y; i++)
805 {
806 const Uint32 pixel = pic->Data[i];
807 const color_t color = PIXEL2COLOR(pixel);
808 if (pixel == 0)
809 {
810 data[i] = 0;
811 }
812 else
813 {
814 data[i] = PixelMult(
815 pixel, COLOR2PIXEL(CharColorsGetChannelMask(colors, color.a)));
816 }
817 }
818 glTexImage2D(
819 GL_TEXTURE_2D, 0, GL_RGBA, pic->size.x, pic->size.y, 0, GL_BGRA,
820 GL_UNSIGNED_BYTE, data);
821 CFREE(data);
822}

831static void DrawCharacter(
832 EditorContext *ec, Character *c, GLuint *texids, const Vec2i pos,
833 const Animation *anim, const direction_e d)
834{
835 const int frame = AnimationGetFrame(anim);
836 ActorPics pics = GetCharacterPics(
837 c, d, anim->Type, frame,
838 c->Gun->Pic, GUNSTATE_READY, false, NULL, NULL, 0);
839 for (int i = 0; i < BODY_PART_COUNT; i++)
840 {
841 const Pic *pic = pics.OrderedPics[i];
842 if (pic == NULL)
843 {
844 continue;
845 }
846 const Vec2i drawPos = Vec2iAdd(pos, pics.OrderedOffsets[i]);
847 LoadMultiChannelTexFromPic(texids[i], pic, pics.Colors);
848 struct nk_image tex = nk_image_id((int)texids[i]);
849 glTexParameteri(
850 GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_NEAREST);
851 glTexParameteri(
852 GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_NEAREST);
853 struct nk_rect bounds;
854 nk_layout_widget_space(
855 &bounds, ec->ctx, ec->ctx->current, nk_true);
856 bounds.x += drawPos.x * PIC_SCALE + 32;
857 bounds.y += drawPos.y * PIC_SCALE + 32;
858 bounds.w = (float)pic->size.x * PIC_SCALE;
859 bounds.h = (float)pic->size.y * PIC_SCALE;
860 nk_draw_image(
861 &ec->ctx->current->buffer, bounds, &tex, nk_white);
862 }
