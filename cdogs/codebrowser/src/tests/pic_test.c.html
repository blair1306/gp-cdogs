43FEATURE(PicLoad, "Pic load")
44 SCENARIO("Load png")
45 GIVEN("a graphics context")
46 if (SDL_Init(SDL_INIT_VIDEO) != 0)
47 {
48 printf("Failed to init SDL: %s\n", SDL_GetError());
49 ASSERT(false, 1);
50 }
51 gConfig = ConfigDefault();
52 ConfigResetDefault(ConfigGet(&gConfig, "Graphics"));
53 GraphicsInit(&gGraphicsDevice, &gConfig);
54 GraphicsInitialize(&gGraphicsDevice);
55 ASSERT(gGraphicsDevice.IsInitialized, 1);
56 AND("a single pixel PNG")
57 SDL_RWops *rwops = SDL_RWFromFile("r64g128b192.png", "rb");
58 ASSERT(IMG_isPNG(rwops), 1);
59 SDL_Surface *image = IMG_Load_RW(rwops, 0);
60 SDL_LockSurface(image);
61
62 WHEN("I load the pic")
63 Pic p;
64 p.size = Vec2iNew(1, 1);
65 p.offset = Vec2iZero();
66 PicLoad(&p, p.size, Vec2iZero(), image);
67
68 THEN("the loaded pic should have values that match");
69 const color_t c = PIXEL2COLOR(p.Data[0]);
70 SHOULD_INT_EQUAL(c.r, 64);
71 SHOULD_INT_EQUAL(c.g, 128);
72 SHOULD_INT_EQUAL(c.b, 192);
73 SDL_UnlockSurface(image);
74 SDL_FreeSurface(image);
75 rwops->close(rwops);
76 SCENARIO_END
77FEATURE_END
78
