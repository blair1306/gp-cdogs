32FEATURE(json_format_string, "String format")
33 SCENARIO("Quote handling")
34 GIVEN("a JSON structure containing escaped quotes")
35 json_t *root = json_new_object();
36 AddStringPair(root, "Foo", "bar \"baz\"");
37 AddStringPair(root, "Spam", "I like \"spam\" and \"eggs\"");
38
39 WHEN("I format the structure into a string")
40 char *text;
41 json_tree_to_string(root, &text);
42 char *ftext = json_format_string(text);
43
44 THEN("the result should be valid JSON")
45 json_t *parsed = NULL;
46 SHOULD_INT_EQUAL(
47 (int)json_parse_document(&parsed, ftext), (int)JSON_OK);
48 AND("contain the same strings")
49 SHOULD_BE_TRUE(strstr(ftext, "I like \\\"spam\\\" and \\\"eggs\\\"") != NULL);
50 CFREE(text);
51 CFREE(ftext);
52 json_free_value(&root);
53 json_free_value(&parsed);
54 SCENARIO_END
55FEATURE_END
