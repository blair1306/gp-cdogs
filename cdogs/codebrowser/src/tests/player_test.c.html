91FEATURE(assign_unused, "Assign unused input device")
92 // This feature is used to assign input devices to players, before the game
93 // begins. All input devices can emit the "fire" command, at which point
94 // those input devices are assigned to the next unassigned player. This
95 // lets us easily assign different input devices to different players,
96 // regardless of the number and type of input devices and players.
97 PlayerDataInit(&gPlayerDatas);
98 NPlayerData pd = PlayerDataDefault(0);
99 PlayerDataAddOrUpdate(pd);
100 pd.UID = 1;
101 PlayerDataAddOrUpdate(pd);
102 SCENARIO("Assign device to unset player")
103 GIVEN("a player with an unset input device")
104 PlayerData *p = CArrayGet(&gPlayerDatas, 0);
105 p->inputDevice = INPUT_DEVICE_UNSET;
106 WHEN("I assign it with an input device")
107 const input_device_e d = INPUT_DEVICE_KEYBOARD;
108 const int idx = 0;
109 bool res = PlayerTrySetUnusedInputDevice(p, d, idx);
110 THEN("the assignment should succeed")
111 SHOULD_BE_TRUE(res);
112 AND("the player should have the input device set")
113 SHOULD_INT_EQUAL((int)p->inputDevice, (int)d);
114 SHOULD_INT_EQUAL(p->deviceIndex, idx);
115 SCENARIO_END
116 SCENARIO("Assign device to set player")
117 GIVEN("a player with an input device")
118 PlayerData *p = CArrayGet(&gPlayerDatas, 0);
119 const input_device_e d = INPUT_DEVICE_KEYBOARD;
120 p->inputDevice = d;
121 const int idx = 0;
122 p->deviceIndex = idx;
123 WHEN("I assign it with a different input device")
124 bool res = PlayerTrySetUnusedInputDevice(p, INPUT_DEVICE_MOUSE, 0);
125 THEN("the assignment should fail")
126 SHOULD_BE_FALSE(res);
127 AND("the player's input device should be unchanged")
128 SHOULD_INT_EQUAL((int)p->inputDevice, (int)d);
129 SHOULD_INT_EQUAL(p->deviceIndex, idx);
130 SCENARIO_END
131 SCENARIO("Assign already used device")
132 GIVEN("a player with an unset input device")
133 PlayerData *p = CArrayGet(&gPlayerDatas, 0);
134 p->inputDevice = INPUT_DEVICE_UNSET;
135 AND("a player with a keyboard device")
136 PlayerData *p2 = CArrayGet(&gPlayerDatas, 1);
137 p2->inputDevice = INPUT_DEVICE_KEYBOARD;
138 p2->deviceIndex = 0;
139 WHEN("I assign the first player with the second's input device")
140 bool res = PlayerTrySetUnusedInputDevice(
141 p, p2->inputDevice, p2->deviceIndex);
142 THEN("the assignment should fail")
143 SHOULD_BE_FALSE(res);
144 AND("the player's input device should be unset")
145 SHOULD_INT_EQUAL((int)p->inputDevice, (int)INPUT_DEVICE_UNSET);
146 SCENARIO_END
147 PlayerDataTerminate(&gPlayerDatas);
148FEATURE_END
