10FEATURE(hashmap_put, "Hashmap put")
11 SCENARIO("Put one value")
12 GIVEN("a new hashmap")
13 map_t map = hashmap_new();
14
15 WHEN("I put a new value in")
16 int value = 42;
17 int error = hashmap_put(map, "somekey", &value);
18
19 THEN("the operation should be successful");
20 SHOULD_INT_EQUAL(error, (int)MAP_OK);
21
22 hashmap_free(map);
23 SCENARIO_END
24FEATURE_END
25
26FEATURE(hashmap_get, "Hashmap get")
27 SCENARIO("Get an existing value")
28 GIVEN("a hashmap with a value")
29 map_t map = hashmap_new();
30 int value = 42;
31 hashmap_put(map, "somekey", &value);
32
33 WHEN("I get it via its key")
34 int *valueOut;
35 int error = hashmap_get(map, "somekey", (void **)&valueOut);
36
37 THEN("the operation should be successful")
38 SHOULD_INT_EQUAL(error, (int)MAP_OK);
39 AND("the value should match");
40 SHOULD_INT_EQUAL(value, *valueOut);
41
42 hashmap_free(map);
43 SCENARIO_END
44
45 SCENARIO("Get non-existing value")
46 GIVEN("a hashmap with a value")
47 map_t map = hashmap_new();
48 int value = 42;
49 hashmap_put(map, "somekey", &value);
50
51 WHEN("I get a non-existing key")
52 int *valueOut;
53 int error = hashmap_get(map, "notkey", (void **)&valueOut);
54
55 THEN("the operation should be unsuccessful")
56 SHOULD_INT_EQUAL(error, (int)MAP_MISSING);
57
58 hashmap_free(map);
59 SCENARIO_END
60FEATURE_END
61
62FEATURE(hashmap_remove, "Hashmap remove")
63 SCENARIO("Remove an existing value")
64 GIVEN("a hashmap with a value")
65 map_t map = hashmap_new();
66 int value = 42;
67 hashmap_put(map, "somekey", &value);
68
69 WHEN("I remove it via its key")
70 int error = hashmap_remove(map, "somekey");
71
72 THEN("the operation should be successful")
73 SHOULD_INT_EQUAL(error, (int)MAP_OK);
74
75 hashmap_free(map);
76 SCENARIO_END
77FEATURE_END
