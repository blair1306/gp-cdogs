13FEATURE(minkowski_hex, "Minkowski Hex")
14 SCENARIO("Same position")
15 GIVEN("two rectangles with the same position")
16 const Vec2i rectPos1 = Vec2iNew(3, 4);
17 const Vec2i rectVel1 = Vec2iNew(-4, 5);
18 const Vec2i rectSize1 = Vec2iNew(1, 3);
19 const Vec2i rectPos2 = rectPos1;
20 const Vec2i rectSize2 = Vec2iNew(2, 1);
21 const Vec2i rectVel2 = Vec2iNew(7, -8);
22
23 WHEN("I check for their collision")
24 Vec2i collide1, collide2, normal;
25 const bool result = MinkowskiHexCollide(
26 rectPos1, rectVel1, rectSize1,
27 rectPos2, rectVel2, rectSize2,
28 &collide1, &collide2, &normal);
29
30 THEN("the result should be true");
31 SHOULD_BE_TRUE(result);
32 AND("the collision points should be at the start")
33 SHOULD_INT_EQUAL(collide1.x, rectPos1.x);
34 SHOULD_INT_EQUAL(collide1.y, rectPos1.y);
35 SHOULD_INT_EQUAL(collide2.x, rectPos2.x);
36 SHOULD_INT_EQUAL(collide2.y, rectPos2.y);
37 AND("the normal should be (0, 0)")
38 SHOULD_INT_EQUAL(normal.x, 0);
39 SHOULD_INT_EQUAL(normal.y, 0);
40 SCENARIO_END
41
42 SCENARIO("No overlap, no movement")
43 GIVEN("two non-overlapping rectangles with no movement")
44 const Vec2i rectPos1 = Vec2iNew(3, 4);
45 const Vec2i rectVel1 = Vec2iZero();
46 const Vec2i rectSize1 = Vec2iNew(1, 3);
47 const Vec2i rectPos2 = Vec2iNew(8, 9);
48 const Vec2i rectSize2 = Vec2iNew(2, 1);
49 const Vec2i rectVel2 = Vec2iZero();
50
51 WHEN("I check for their collision")
52 Vec2i collide1, collide2, normal;
53 const bool result = MinkowskiHexCollide(
54 rectPos1, rectVel1, rectSize1,
55 rectPos2, rectVel2, rectSize2,
56 &collide1, &collide2, &normal);
57
58 THEN("the result should be false");
59 SHOULD_BE_FALSE(result);
60 SCENARIO_END
61
62 SCENARIO("Single axis single movement")
63 GIVEN("two rectangles, one moving into the other")
64 const Vec2i rectPos1 = Vec2iZero();
65 const Vec2i rectVel1 = Vec2iNew(10, 0);
66 const Vec2i rectSize1 = Vec2iNew(2, 2);
67 const Vec2i rectPos2 = Vec2iNew(5, 0);
68 const Vec2i rectSize2 = Vec2iNew(2, 2);
69 const Vec2i rectVel2 = Vec2iZero();
70
71 WHEN("I check for their collision")
72 Vec2i collide1, collide2, normal;
73 const bool result = MinkowskiHexCollide(
74 rectPos1, rectVel1, rectSize1,
75 rectPos2, rectVel2, rectSize2,
76 &collide1, &collide2, &normal);
77
78 THEN("the result should be true");
79 SHOULD_BE_TRUE(result);
80 AND("the collision point for the moving rectangle should not overlap "
81 "with the stationary rectangle")
82 SHOULD_INT_EQUAL(collide1.x, 3);
83 SHOULD_INT_EQUAL(collide1.y, 0);
84 AND("the collision point for the stationary rectangle should be the "
85 "same as its position")
86 SHOULD_INT_EQUAL(collide2.x, rectPos2.x);
87 SHOULD_INT_EQUAL(collide2.y, rectPos2.y);
88 AND("the normal should be (-1, 0)")
89 SHOULD_INT_EQUAL(normal.x, -1);
90 SHOULD_INT_EQUAL(normal.y, 0);
91 SCENARIO_END
92FEATURE_END
