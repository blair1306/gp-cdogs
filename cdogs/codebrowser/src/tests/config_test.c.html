36FEATURE(load_default, "Load default config")
37 SCENARIO("Load a default config")
38 GIVEN("two configs")
39 Config config1, config2;
40
41 WHEN("I load them both with defaults")
42 // Note: default loaded before loading from file
43 config1 = ConfigLoad(NULL);
44 config2 = ConfigLoad(NULL);
45
46 THEN("the two configs should have the same values")
47 SHOULD_INT_EQUAL(
48 ConfigGetBool(&config1, "Game.FriendlyFire"),
49 ConfigGetBool(&config2, "Game.FriendlyFire"));
50 SHOULD_INT_EQUAL(
51 ConfigGetInt(&config1, "Graphics.Brightness"),
52 ConfigGetInt(&config2, "Graphics.Brightness"));
53 SCENARIO_END
54FEATURE_END

56FEATURE(save_and_load, "Save and load")
57 SCENARIO("Save and load a JSON config file")
58 GIVEN("a config file with some values, and I save the config to a JSON file")
59 Config config1 = ConfigLoad(NULL);
60 ConfigGet(&config1, "Game.FriendlyFire")->u.Bool.Value = true;
61 ConfigGet(&config1, "Graphics.Brightness")->u.Int.Value = 5;
62 ConfigSave(&config1, "tmp");
63
64 WHEN("I load a second config from that file")
65 Config config2 = ConfigLoad("tmp");
66
67 THEN("the two configs should have the same values")
68 SHOULD_INT_EQUAL(
69 ConfigGetBool(&config1, "Game.FriendlyFire"),
70 ConfigGetBool(&config2, "Game.FriendlyFire"));
71 SHOULD_INT_EQUAL(
72 ConfigGetInt(&config1, "Graphics.Brightness"),
73 ConfigGetInt(&config2, "Graphics.Brightness"));
74 SCENARIO_END
75FEATURE_END

77FEATURE(detect_version, "Detect config version")
78 SCENARIO("Detect JSON config version")
79 GIVEN("a config file with some values")
80 Config config1 = ConfigLoad(NULL);
81 ConfigGet(&config1, "Game.FriendlyFire")->u.Bool.Value = true;
82 ConfigGet(&config1, "Graphics.Brightness")->u.Int.Value = 5;
83 AND("I save the config to file in the JSON format")
84 ConfigSave(&config1, "tmp");
85
86 WHEN("I detect the version")
87 FILE *file = fopen("tmp", "r");
88 int version = ConfigGetVersion(file);
89 fclose(file);
90 AND("load a second config from that file")
91 Config config2 = ConfigLoad("tmp");
92
93 THEN("the version should be " TOSTRING(CONFIG_VERSION))
94 SHOULD_INT_EQUAL(version, CONFIG_VERSION);
95 AND("the two configs should have the same values")
96 SHOULD_INT_EQUAL(
97 ConfigGetBool(&config1, "Game.FriendlyFire"),
98 ConfigGetBool(&config2, "Game.FriendlyFire"));
99 SHOULD_INT_EQUAL(
100 ConfigGetInt(&config1, "Graphics.Brightness"),
101 ConfigGetInt(&config2, "Graphics.Brightness"));
102 SCENARIO_END
103FEATURE_END
