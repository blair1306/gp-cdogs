48FEATURE(AutosaveInit, "Initialise autosave")
49 SCENARIO("Initialise autosave")
50 GIVEN("two autosaves")
51 Autosave autosave1, autosave2;
52
53 WHEN("I initialise both")
54 AutosaveInit(&autosave1);
55 AutosaveInit(&autosave2);
56
57 THEN("their campaign entries should equal each other")
58 SHOULD_MEM_EQUAL(
59 &autosave1.LastMission.Campaign,
60 &autosave2.LastMission.Campaign,
61 sizeof autosave1.LastMission.Campaign);
62 AND("their passwords should equal each other")
63 SHOULD_STR_EQUAL(
64 autosave1.LastMission.Password,
65 autosave2.LastMission.Password);
66 SCENARIO_END
67FEATURE_END

69FEATURE(save_and_load, "Save and load")
70 SCENARIO("Save and load")
71 GIVEN("an autosave with some values")
72 Autosave autosave1;
73 AutosaveInit(&autosave1);
74 MissionSave mission1;
75 memset(&mission1, 0, sizeof mission1);
76 CSTRDUP(mission1.Campaign.Path, "mission.cdogscpn");
77 strcpy(mission1.Password, "password");
78 AutosaveAddMission(&autosave1, &mission1);
79 AND("I save it to file")
80 AutosaveSave(&autosave1, "tmp");
81
82 WHEN("I initialise and load a second autosave from that file")
83 Autosave autosave2;
84 AutosaveInit(&autosave2);
85 AutosaveLoad(&autosave2, "tmp");
86
87 THEN("their last mission paths should equal")
88 SHOULD_STR_EQUAL(
89 autosave2.LastMission.Campaign.Path,
90 autosave1.LastMission.Campaign.Path);
91 AND("their last mission passwords should equal")
92 SHOULD_STR_EQUAL(
93 autosave2.LastMission.Password,
94 autosave1.LastMission.Password);
95 AND("their mission paths should equal")
96 MissionSave mission2;
97 AutosaveLoadMission(&autosave2, &mission2, mission1.Campaign.Path);
98 SHOULD_STR_EQUAL(
99 mission2.Campaign.Path, mission1.Campaign.Path);
100 AND("their mission passwords should equal")
101 SHOULD_STR_EQUAL(mission2.Password, mission1.Password);
102 SCENARIO_END
103FEATURE_END

105FEATURE(mission_autosaves, "Mission autosaves")
106 SCENARIO("Load non-existing mission autosave")
107 GIVEN("an empty autosave")
108 Autosave autosave;
109 AutosaveInit(&autosave);
110
111 WHEN("I attempt to load a non-existing mission from it")
112 MissionSave mission;
113 AutosaveLoadMission(&autosave, &mission, "mission.cdogscpn");
114
115 THEN("the mission should be empty")
116 SHOULD_BE_TRUE(mission.Campaign.Path == NULL);
117 AND("the password should be empty")
118 SHOULD_STR_EQUAL(mission.Password, "");
119 SCENARIO_END
120
121 SCENARIO("Add new mission autosave")
122 GIVEN("an autosave and a mission")
123 Autosave autosave;
124 AutosaveInit(&autosave);
125 MissionSave mission1;
126 memset(&mission1, 0, sizeof mission1);
127 CSTRDUP(mission1.Campaign.Path, "mission.cdogscpn");
128 strcpy(mission1.Password, "password");
129
130 WHEN("I add a new mission autosave to it")
131 AutosaveAddMission(&autosave, &mission1);
132
133 THEN("I should be able to find the mission in the autosave")
134 MissionSave mission2;
135 AutosaveLoadMission(&autosave, &mission2, mission1.Campaign.Path);
136 SHOULD_STR_EQUAL(mission2.Campaign.Path, mission1.Campaign.Path);
137 SHOULD_STR_EQUAL(mission2.Password, mission1.Password);
138 SCENARIO_END
139
140 SCENARIO("Add existing mission autosave")
141 GIVEN("an autosave and a mission")
142 Autosave autosave;
143 AutosaveInit(&autosave);
144 MissionSave mission1;
145 memset(&mission1, 0, sizeof mission1);
146 CSTRDUP(mission1.Campaign.Path, "mission.cdogscpn");
147 strcpy(mission1.Password, "password");
148 mission1.MissionsCompleted = 3;
149 AND("I add the mission to the autosave")
150 AutosaveAddMission(&autosave, &mission1);
151
152 WHEN("I add the same mission but with new password")
153 strcpy(mission1.Password, "new password");
154 AND("and less missions completed")
155 mission1.MissionsCompleted = 2;
156 AutosaveAddMission(&autosave, &mission1);
157
158 THEN("I should be able to find the mission in the autosave")
159 MissionSave mission2;
160 AutosaveLoadMission(&autosave, &mission2, mission1.Campaign.Path);
161 AND("with the new details")
162 SHOULD_STR_EQUAL(mission2.Campaign.Path, mission1.Campaign.Path);
163 SHOULD_STR_EQUAL(mission2.Password, mission1.Password);
164 BUT("the greatest missions completed")
165 SHOULD_INT_EQUAL(mission2.MissionsCompleted, 3);
166 SCENARIO_END
