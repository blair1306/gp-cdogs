17FEATURE(CArrayInsert, "Array insert")
18 SCENARIO("Insert in middle")
19 GIVEN("an array with some elements")
20 CArray a;
21 CArrayInit(&a, sizeof(int));
22 for (int i = 0; i < 5; i++)
23 {
24 CArrayPushBack(&a, &i);
25 }
26 size_t oldSize = a.size;
27
28 WHEN("I insert an element in the middle")
29 int middleIndex = 3;
30 int middleValue = -1;
31 CArrayInsert(&a, middleIndex, &middleValue);
32
33 THEN("the array should be one element larger")
34 SHOULD_INT_EQUAL((int)a.size, (int)oldSize + 1);
35 AND("contain the previous elements, the middle element, plus the next elements")
36 for (int i = 0; i < (int)a.size; i++)
37 {
38 int *v = CArrayGet(&a, i);
39 if (i < middleIndex)
40 {
41 SHOULD_INT_EQUAL(*v, i);
42 }
43 else if (i == middleIndex)
44 {
45 SHOULD_INT_EQUAL(*v, middleValue);
46 }
47 else
48 {
49 SHOULD_INT_EQUAL(*v, i - 1);
50 }
51 }
52 SCENARIO_END
53FEATURE_END

55FEATURE(CArrayDelete, "Array delete")
56 SCENARIO("Delete from middle")
57 GIVEN("an array with some elements")
58 CArray a;
59 CArrayInit(&a, sizeof(int));
60 for (int i = 0; i < 5; i++)
61 {
62 CArrayPushBack(&a, &i);
63 }
64 size_t oldSize = a.size;
65
66 WHEN("I delete an element from the middle")
67 int middleIndex = 3;
68 CArrayDelete(&a, middleIndex);
69
70 THEN("the array should be one element smaller")
71 SHOULD_INT_EQUAL((int)a.size, (int)oldSize - 1);
72 AND("contain all but the deleted elements in order")
73 for (int i = 0; i < (int)a.size; i++)
74 {
75 int *v = CArrayGet(&a, i);
76 if (i < middleIndex)
77 {
78 SHOULD_INT_EQUAL(*v, i);
79 }
80 else if (i >= middleIndex)
81 {
82 SHOULD_INT_EQUAL(*v, i + 1);
83 }
84 }
85 SCENARIO_END
86FEATURE_END

93FEATURE(CArrayRemoveIf, "Array remove if")
94 SCENARIO("Remove if")
95 GIVEN("an array with numbers 0-4")
96 CArray a;
97 CArrayInit(&a, sizeof(int));
98 for (int i = 0; i < 5; i++)
99 {
100 CArrayPushBack(&a, &i);
101 }
102
103 WHEN("I remove all elements less than 2")
104 CArrayRemoveIf(&a, LessThan2);
105
106 THEN("the array should contain the numbers >= 2")
107 SHOULD_INT_EQUAL((int)a.size, 3);
108 for (int i = 0; i < (int)a.size; i++)
109 {
110 int *v = CArrayGet(&a, i);
111 SHOULD_INT_GE(*v, 2);
112 }
113 SCENARIO_END
114FEATURE_END
