8FEATURE(ColorMult, "Multiply")
9 SCENARIO("Multiply two colors")
10 GIVEN("two colors")
11 color_t c1, c2;
12 c1.r = c1.g = c1.b = 123;
13 c2.r = c2.g = c2.b = 234;
14
15 WHEN("I multiply them")
16 color_t result = ColorMult(c1, c2);
17
18 THEN("the result should be a multiply blend of the two color's components");
19 SHOULD_INT_EQUAL(result.r, c1.r * c2.r / 255);
20 SHOULD_INT_EQUAL(result.g, c1.g * c2.g / 255);
21 SHOULD_INT_EQUAL(result.b, c1.b * c2.b / 255);
22 SCENARIO_END
23
24 SCENARIO("Multiply by white")
25 GIVEN("a color")
26 color_t c;
27 c.r = 123;
28 c.g = 234;
29 c.b = 45;
30
31 WHEN("I multiply it by white")
32 color_t white;
33 white.r = white.g = white.b = 255;
34 color_t result = ColorMult(c, white);
35
36 THEN("the result should be the same as the color")
37 SHOULD_INT_EQUAL(result.r, c.r);
38 SHOULD_INT_EQUAL(result.g, c.g);
39 SHOULD_INT_EQUAL(result.b, c.b);
40 SCENARIO_END

61FEATURE(ColorAlphaBlend, "Alpha blend")
62 SCENARIO("Alpha blend two colors")
63 GIVEN("two colors")
64 color_t c1, c2;
65 c1.r = c1.g = c1.b = c1.a = 123;
66 c2.r = c2.g = c2.b = c2.a = 234;
67
68 WHEN("I alpha blend them")
69 color_t result = ColorAlphaBlend(c1, c2);
70
71 THEN("the result should be an alpha blend of the two color's components")
72 SHOULD_INT_EQUAL(result.r, ((int)c1.r*(255 - c2.a) + (int)c2.r*c2.a)/255);
73 SHOULD_INT_EQUAL(result.g, ((int)c1.g*(255 - c2.a) + (int)c2.g*c2.a)/255);
74 SHOULD_INT_EQUAL(result.b, ((int)c1.r*(255 - c2.a) + (int)c2.b*c2.a)/255);
75 SHOULD_INT_EQUAL(result.a, 255);
76 SCENARIO_END
