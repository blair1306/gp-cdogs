34static void LoadFile(CArray *strings, const char *filename);
35void NameGenInit(
36 NameGen *g,
37 const char *prefixFile, const char *suffixFile, const char *suffixNameFile)
38{
39 CArrayInit(&g->prefixes, sizeof(char *));
40 CArrayInit(&g->suffixes, sizeof(char *));
41 CArrayInit(&g->suffixNames, sizeof(char *));
42 LoadFile(&g->prefixes, prefixFile);
43 LoadFile(&g->suffixes, suffixFile);
44 LoadFile(&g->suffixNames, suffixNameFile);
45}
46static void LoadFile(CArray *strings, const char *filename)
47{
48 FILE *file = fopen(filename, "r");
49 char buf[256];
50 while (fgets(buf, 256, file) != NULL)
51 {
52 char *str = malloc(strlen(buf) + 1);
53 strncpy(str, buf, strlen(buf) - 1);
54 str[strlen(buf) - 1] = '\0';
55 CArrayPushBack(strings, &str);
56 }
57 fclose(file);
58}
59static void UnloadStrings(CArray *strings);
60void NameGenTerminate(NameGen *g)
61{
62 UnloadStrings(&g->prefixes);
63 UnloadStrings(&g->suffixes);
64 UnloadStrings(&g->suffixNames);
65}
66static void UnloadStrings(CArray *strings)
67{
68 for (int i = 0; i < (int)strings->size; i++)
69 {
70 char **str = CArrayGet(strings, i);
71 free(*str);
72 }
73 CArrayTerminate(strings);
74}

76static bool IsSameWord(const char *l, const char *r);
77void NameGenMake(const NameGen *g, char *buf)
78{
79 for (;;)
80 {
81 char **prefix = CArrayGet(&g->prefixes, rand() % g->prefixes.size);
82 int suffixIndex = rand() % (g->suffixes.size + g->suffixNames.size);
83 char **suffix;
84 if (suffixIndex < (int)g->suffixes.size)
85 {
86 suffix = CArrayGet(&g->suffixes, suffixIndex);
87 }
88 else
89 {
90 suffix = CArrayGet(
91 &g->suffixNames, suffixIndex - (int)g->suffixes.size);
92 }
93 // Don't generate same prefix/suffix
94 if (IsSameWord(*prefix, *suffix))
95 {
96 continue;
97 }
98 strcpy(buf, *prefix);
99 // Add a space if we're using a suffix name
100 if (suffixIndex >= (int)g->suffixes.size)
101 {
102 strcat(buf, " ");
103 }
104 strcat(buf, *suffix);
105 break;
106 }
107}
108static char LowerCase(const char c);
109static bool IsSameWord(const char *l, const char *r)
110{
111 while (*l && *r)
112 {
113 if (LowerCase(*l) != LowerCase(*r))
114 {
115 return false;
116 }
117 l++;
118 r++;
119 }
120 if (*l || *r)
121 {
122 return false;
123 }
124 return true;
125}
126static char LowerCase(const char c)
127{
128 if (c >= 'A' && c <= 'Z')
129 {
130 return c - 32;
131 }
132 return c;
133}
