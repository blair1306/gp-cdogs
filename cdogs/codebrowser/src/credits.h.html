42/*
43credits_displayer_t is a module that loads a number of credit_t entries from a
44CREDITS file, and displays each for a fixed amount of time.
45This is used to show the credits (name + description) in the game's main menu.
46The credit entries are in a file with alternating lines of name and
47description, as so:
48
49<name1>
50<description1>
51<name2>
52<description2>
53
54Empty lines are ignored
55For more details see /doc/CREDITS
56*/
57
58typedef struct
59{
60 char *name;
61 char *message;
62} credit_t;
63
64typedef struct
65{
66 CArray credits; // of credit_t
67 time_t lastUpdateTime;
68 int creditsIndex;
69 color_t nameColor;
70 color_t textColor;
71} credits_displayer_t;
72
73void LoadCredits(
74 credits_displayer_t *displayer, color_t nameColor, color_t textColor);
75void UnloadCredits(credits_displayer_t *displayer);
76void ShowCredits(credits_displayer_t *displayer);
