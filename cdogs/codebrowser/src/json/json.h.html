44/**
45The descriptions of the json_value node type
46**/
47 enum json_value_type
48 { JSON_STRING = 0,
49 JSON_NUMBER,
50 JSON_OBJECT,
51 JSON_ARRAY,
52 JSON_TRUE,
53 JSON_FALSE,
54 JSON_NULL
55 };

57/**
58String implementation
59**/
60 struct rui_cstring
61 {
62 char *text; /*<! char c-string */
63 size_t length; /*<! put in place to avoid strlen() calls */
64 size_t max; /*<! usable memory allocated to text minus the space for the nul character */
65 };
66
67 typedef struct rui_cstring rcstring;
68
69/**
70The error messages produced by the JSON parsers
71**/
72 enum json_error
73 {
74 JSON_OK = 1, /*!< everything went smoothly */
75 JSON_INCOMPLETE_DOCUMENT, /*!< the parsed document didn't ended */
76 JSON_WAITING_FOR_EOF, /*!< A complete JSON document tree was already finished but needs to get to EOF. Other characters beyond whitespaces produce errors */
77 JSON_MALFORMED_DOCUMENT, /* the JSON document which was fed to this parser is malformed */
78 JSON_INCOMPATIBLE_TYPE, /*!< the currently parsed type does not belong here */
79 JSON_MEMORY, /*!< an error occurred when allocating memory */
80 JSON_ILLEGAL_CHARACTER, /*!< the currently parsed character does not belong here */
81 JSON_BAD_TREE_STRUCTURE, /*!< the document tree structure is malformed */
82 JSON_MAXIMUM_LENGTH, /*!< the parsed string reached the maximum allowed size */
83 JSON_UNKNOWN_PROBLEM /*!< some random, unaccounted problem occurred */
84 };
85
86
87/**
88The JSON document tree node, which is a basic JSON type
89**/
90 typedef struct json_value
91 {
92 enum json_value_type type; /*!< the type of node */
93 char *text; /*!< The text stored by the node. It stores UTF-8 strings and is used exclusively by the JSON_STRING and JSON_NUMBER node types */
94
95 /* FIFO queue data */
96 struct json_value *next; /*!< The pointer pointing to the next element in the FIFO sibling list */
97 struct json_value *previous; /*!< The pointer pointing to the previous element in the FIFO sibling list */
98 struct json_value *parent; /*!< The pointer pointing to the parent node in the document tree */
99 struct json_value *child; /*!< The pointer pointing to the first child node in the document tree */
100 struct json_value *child_end; /*!< The pointer pointing to the last child node in the document tree */
101 } json_t;
102
103
104/**
105The structure holding all information needed to resume parsing
106**/
107 struct json_parsing_info
108 {
109 unsigned int state; /*!< the state where the parsing was left on the last parser run */
110 unsigned int lex_state;
111 rcstring *lex_text;
112 const char *p;
113 int string_length_limit_reached; /*!< flag informing if the string limit length defined by JSON_MAX_STRING_LENGTH was reached */
114 size_t line; // current document line
115 json_t *cursor; /*!< pointers to nodes belonging to the document tree which aid the document parsing */
116 };
117
118
119/**
120The structure which holds the pointers to the functions that will be called by the saxy parser whenever their evens are triggered
121**/
122 struct json_saxy_functions
123 {
124 int (*open_object) (void);
125 int (*close_object) (void);
126 int (*open_array) (void);
127 int (*close_array) (void);
128 int (*new_string) (char *text);
129 int (*new_number) (char *text);
130 int (*new_true) (void);
131 int (*new_false) (void);
132 int (*new_null) (void);
133 int (*label_value_separator) (void);
134 int (*sibling_separator) (void);
135 };
136
137
138/**
139The structure holding the information needed for json_saxy_parse to resume parsing
140**/
141 struct json_saxy_parser_status
142 {
143 unsigned int state; /*!< current parser state */
144 int string_length_limit_reached; /*!< flag informing if the string limit length defined by JSON_MAX_STRING_LENGTH was reached */
145 rcstring *temp; /*!< temporary string which will be used to build up parsed strings between parser runs. */
146 };
