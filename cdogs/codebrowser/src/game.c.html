75static void PlayerSpecialCommands(TActor *actor, const int cmd)
76{
77 if ((cmd & CMD_BUTTON2) && CMD_HAS_DIRECTION(cmd))
78 {
79 if (ConfigGetEnum(&gConfig, "Game.SwitchMoveStyle") == SWITCHMOVE_SLIDE)
80 {
81 SlideActor(actor, cmd);
82 }
83 }
84 else if (
85 (actor->lastCmd & CMD_BUTTON2) &&
86 !(cmd & CMD_BUTTON2) &&
87 !actor->specialCmdDir &&
88 !actor->CanPickupSpecial &&
89 !(ConfigGetEnum(&gConfig, "Game.SwitchMoveStyle") == SWITCHMOVE_SLIDE && CMD_HAS_DIRECTION(cmd)) &&
90 ActorCanSwitchGun(actor))
91 {
92 GameEvent e = GameEventNew(GAME_EVENT_ACTOR_SWITCH_GUN);
93 e.u.ActorSwitchGun.UID = actor->uid;
94 e.u.ActorSwitchGun.GunIdx = (actor->gunIndex + 1) % actor->guns.size;
95 GameEventsEnqueue(&gGameEvents, e);
96 }
97}

100// TODO: reimplement in camera
101Vec2i GetPlayerCenter(
102 GraphicsDevice *device, const Camera *camera,
103 const PlayerData *pData, const int playerIdx)
104{
105 if (pData->ActorUID < 0)
106 {
107 // Player is dead
108 return Vec2iZero();
109 }
110 Vec2i center = Vec2iZero();
111 int w = device->cachedConfig.Res.x;
112 int h = device->cachedConfig.Res.y;
113
114 if (GetNumPlayers(PLAYER_ANY, true, true) == 1 ||
115 GetNumPlayers(PLAYER_ANY, false , true) == 1 ||
116 CameraIsSingleScreen())
117 {
118 const Vec2i pCenter = camera->lastPosition;
119 const Vec2i screenCenter =
120 Vec2iNew(w / 2, device->cachedConfig.Res.y / 2);
121 const TActor *actor = ActorGetByUID(pData->ActorUID);
122 const Vec2i p = Vec2iNew(actor->tileItem.x, actor->tileItem.y);
123 center = Vec2iAdd(Vec2iMinus(p, pCenter), screenCenter);
124 }
125 else
126 {
127 const int numLocalPlayers = GetNumPlayers(PLAYER_ANY, false, true);
128 if (numLocalPlayers == 2)
129 {
130 center.x = playerIdx == 0 ? w / 4 : w * 3 / 4;
131 center.y = h / 2;
132 }
133 else if (numLocalPlayers >= 3 && numLocalPlayers <= 4)
134 {
135 center.x = (playerIdx & 1) ? w * 3 / 4 : w / 4;
136 center.y = (playerIdx >= 2) ? h * 3 / 4 : h / 4;
137 }
138 else
139 {
140 CASSERT(false, "invalid number of players");
141 }
142 }
143 return center;
144}

170GameLoopData *RunGame(
171 const CampaignOptions *co, struct MissionOptions *m, Map *map)
172{
173 RunGameData *data;
174 CCALLOC(data, sizeof *data);
175 data->co = co;
176 data->m = m;
177 data->map = map;
178 GameLoopData *g = GameLoopDataNew(
179 data, RunGameTerminate, RunGameOnEnter, RunGameOnExit,
180 RunGameInput, RunGameUpdate, RunGameDraw);
181 g->FPS = ConfigGetInt(&gConfig, "Game.FPS");
182 g->InputEverySecondFrame = true;
183 return g;
184}

191static void RunGameOnEnter(GameLoopData *data)
192{
193 RunGameData *rData = data->Data;
194
195 // Clear the background
196 DrawRectangle(
197 &gGraphicsDevice, Vec2iZero(), gGraphicsDevice.cachedConfig.Res,
198 colorBlack, 0);
199 BlitUpdateFromBuf(&gGraphicsDevice, gGraphicsDevice.bkg);
200
201 MapLoad(rData->map, rData->m, rData->co);
202
203 // Seed random if PVP mode (otherwise players will always spawn in same
204 // position)
205 if (IsPVP(rData->co->Entry.Mode))
206 {
207 srand((unsigned int)time(NULL));
208 }
209
210 if (!rData->co->IsClient)
211 {
212 MapLoadDynamic(rData->map, rData->m, &rData->co->Setting.characters);
213
214 // For PVP modes, mark all map as explored
215 if (IsPVP(rData->co->Entry.Mode))
216 {
217 MapMarkAllAsVisited(rData->map);
218 }
219
220 // Reset players for the mission
221 CA_FOREACH(const PlayerData, p, gPlayerDatas)
222 // Only reset for local players; for remote ones wait for the
223 // client ready message
224 if (!p->IsLocal) continue;
225 GameEvent e = GameEventNew(GAME_EVENT_PLAYER_DATA);
226 e.u.PlayerData = PlayerDataMissionReset(p);
227 GameEventsEnqueue(&gGameEvents, e);
228 CA_FOREACH_END()
229 // Process the events to force add the players
230 HandleGameEvents(&gGameEvents, NULL, NULL, NULL);
231
232 // Note: place players first,
233 // as bad guys are placed away from players
234 Vec2i firstPos = Vec2iZero();
235 CA_FOREACH(const PlayerData, p, gPlayerDatas)
236 if (!p->Ready) continue;
237 firstPos = PlacePlayer(&gMap, p, firstPos, true);
238 CA_FOREACH_END()
239 if (!IsPVP(rData->co->Entry.Mode))
240 {
241 InitializeBadGuys();
242 CreateEnemies();
243 }
244 }
245
246 CameraInit(&rData->Camera);
247 // If there are no players, show the full map before starting
248 if (GetNumPlayers(PLAYER_ANY, false, true) == 0)
249 {
250 LOSSetAllVisible(&rData->map->LOS);
251 rData->Camera.lastPosition =
252 Vec2iCenterOfTile(Vec2iScaleDiv(rData->map->Size, 2));
253 rData->Camera.FollowNextPlayer = true;
254 }
255 HealthSpawnerInit(&rData->healthSpawner, rData->map);
256 CArrayInit(&rData->ammoSpawners, sizeof(PowerupSpawner));
257 for (int i = 0; i < AmmoGetNumClasses(&gAmmo); i++)
258 {
259 PowerupSpawner ps;
260 AmmoSpawnerInit(&ps, rData->map, i);
261 CArrayPushBack(&rData->ammoSpawners, &ps);
262 }
263
264 rData->m->state = MISSION_STATE_WAITING;
265 rData->m->isDone = false;
266 rData->m->DoneCounter = 0;
267 Pic *crosshair = PicManagerGetPic(&gPicManager, "crosshair");
268 crosshair->offset.x = -crosshair->size.x / 2;
269 crosshair->offset.y = -crosshair->size.y / 2;
270 EventReset(
271 &gEventHandlers, crosshair,
272 PicManagerGetPic(&gPicManager, "crosshair_trail"));
273
274 NetServerSendGameStartMessages(&gNetServer, NET_SERVER_BCAST);
275 GameEvent start = GameEventNew(GAME_EVENT_GAME_START);
276 GameEventsEnqueue(&gGameEvents, start);
277}
278static void RunGameOnExit(GameLoopData *data)
279{
280 RunGameData *rData = data->Data;
281
282 LOG(LM_MAIN, LL_INFO, "Game finished");
283
284 // Flush events
285 HandleGameEvents(&gGameEvents, NULL, NULL, NULL);
286
287 PowerupSpawnerTerminate(&rData->healthSpawner);
288 CA_FOREACH(PowerupSpawner, a, rData->ammoSpawners)
289 PowerupSpawnerTerminate(a);
290 CA_FOREACH_END()
291 CArrayTerminate(&rData->ammoSpawners);
292 CameraTerminate(&rData->Camera);
293
294 // Draw background
295 GrafxRedrawBackground(&gGraphicsDevice, rData->Camera.lastPosition);
296 // Clear other texures
297 BlitClearBuf(&gGraphicsDevice);
298 BlitUpdateFromBuf(&gGraphicsDevice, gGraphicsDevice.hud);
299
300 // Unready all the players
301 CA_FOREACH(PlayerData, p, gPlayerDatas)
302 p->Ready = false;
303 CA_FOREACH_END()
304 gNetClient.Ready = false;
305
306 // Calculate remaining health and survived
307 CA_FOREACH(PlayerData, p, gPlayerDatas)
308 p->survived = IsPlayerAlive(p);
309 if (IsPlayerAlive(p))
310 {
311 const TActor *player = ActorGetByUID(p->ActorUID);
312 p->hp = player->health;
313 }
314 CA_FOREACH_END()
315
316 MusicPlayMenu(&gSoundDevice);
317}
318static void RunGameInput(GameLoopData *data)
319{
320 RunGameData *rData = data->Data;
321
322 if (gEventHandlers.HasQuit)
323 {
324 GameEvent e = GameEventNew(GAME_EVENT_MISSION_END);
325 e.u.MissionEnd.IsQuit = true;
326 GameEventsEnqueue(&gGameEvents, e);
327 return;
328 }
329
330 int lastCmdAll = 0;
331 for (int i = 0; i < MAX_LOCAL_PLAYERS; i++)
332 {
333 rData->lastCmds[i] = rData->cmds[i];
334 lastCmdAll |= rData->lastCmds[i];
335 }
336 memset(rData->cmds, 0, sizeof rData->cmds);
337 int cmdAll = 0;
338 int idx = 0;
339 input_device_e pausingDevice = INPUT_DEVICE_UNSET;
340 input_device_e firstPausingDevice = INPUT_DEVICE_UNSET;
341 if (GetNumPlayers(PLAYER_ANY, false, true) == 0)
342 {
343 // If no players, allow default keyboard to control camera
344 rData->cmds[0] = GetKeyboardCmd(
345 &gEventHandlers.keyboard, 0, false);
346 firstPausingDevice = INPUT_DEVICE_KEYBOARD;
347 }
348 else
349 {
350 for (int i = 0; i < (int)gPlayerDatas.size; i++, idx++)
351 {
352 const PlayerData *p = CArrayGet(&gPlayerDatas, i);
353 if (!p->IsLocal)
354 {
355 idx--;
356 continue;
357 }
358 if (firstPausingDevice == INPUT_DEVICE_UNSET)
359 {
360 firstPausingDevice = p->inputDevice;
361 }
362 rData->cmds[idx] = GetGameCmd(
363 &gEventHandlers,
364 p,
365 GetPlayerCenter(&gGraphicsDevice, &rData->Camera, p, idx));
366 cmdAll |= rData->cmds[idx];
367
368 // Only allow the first player to escape
369 // Use keypress otherwise the player will quit immediately
370 if (idx == 0 &&
371 (rData->cmds[idx] & CMD_ESC) && !(rData->lastCmds[idx] & CMD_ESC))
372 {
373 pausingDevice = p->inputDevice;
374 }
375 }
376 }
377 if (KeyIsPressed(&gEventHandlers.keyboard, SDL_SCANCODE_ESCAPE))
378 {
379 pausingDevice = INPUT_DEVICE_KEYBOARD;
380 }
381
382 // Check if any controllers are unplugged
383 rData->controllerUnplugged = false;
384 CA_FOREACH(const PlayerData, p, gPlayerDatas)
385 if (p->inputDevice == INPUT_DEVICE_UNSET && p->IsLocal)
386 {
387 rData->controllerUnplugged = true;
388 break;
389 }
390 CA_FOREACH_END()
391
392 // Check if:
393 // - escape was pressed, or
394 // - window lost focus
395 // - controller unplugged
396 // If the game was not paused, enter pause mode
397 // If the game was paused and escape was pressed, exit the game
398 if (AnyButton(cmdAll))
399 {
400 rData->pausingDevice = INPUT_DEVICE_UNSET;
401 }
402 else if (rData->controllerUnplugged || gEventHandlers.HasLostFocus)
403 {
404 // Pause the game
405 rData->pausingDevice = firstPausingDevice;
406 rData->isMap = false;
407 }
408 else if (pausingDevice != INPUT_DEVICE_UNSET)
409 {
410 if (rData->pausingDevice != INPUT_DEVICE_UNSET)
411 {
412 // Already paused; exit
413 GameEvent e = GameEventNew(GAME_EVENT_MISSION_END);
414 e.u.MissionEnd.IsQuit = true;
415 GameEventsEnqueue(&gGameEvents, e);
416 // Need to unpause to process the quit
417 rData->pausingDevice = INPUT_DEVICE_UNSET;
418 rData->controllerUnplugged = false;
419 }
420 else
421 {
422 // Pause the game
423 rData->pausingDevice = pausingDevice;
424 rData->isMap = false;
425 }
426 }
427
428 const bool paused =
429 rData->pausingDevice != INPUT_DEVICE_UNSET ||
430 rData->controllerUnplugged;
431 if (!paused)
432 {
433 // Check if automap key is pressed by any player
434 // Toggle
435 if (IsAutoMapEnabled(gCampaign.Entry.Mode) &&
436 (KeyIsPressed(&gEventHandlers.keyboard, ConfigGetInt(&gConfig, "Input.PlayerCodes0.map")) ||
437 ((cmdAll & CMD_MAP) && !(lastCmdAll & CMD_MAP))))
438 {
439 rData->isMap = !rData->isMap;
440 SoundPlay(
441 &gSoundDevice,
442 StrSound(rData->isMap ? "map_open" : "map_close"));
443 }
444 }
445
446 CameraInput(&rData->Camera, rData->cmds[0], rData->lastCmds[0]);
447}
448static void NextLoop(RunGameData *rData, LoopRunner *l);

450static GameLoopResult RunGameUpdate(GameLoopData *data, LoopRunner *l)
451{
452 RunGameData *rData = data->Data;
453
454 // Detect exit
455 if (rData->m->isDone)
456 {
457 rData->m->DoneCounter--;
458 if (rData->m->DoneCounter <= 0)
459 {
460 NextLoop(rData, l);
461 return UPDATE_RESULT_OK;
462 }
463 else
464 {
465 return UPDATE_RESULT_DRAW;
466 }
467 }
468
469 // Check if game can begin
470 if (!rData->m->HasBegun && MissionCanBegin())
471 {
472 GameEvent begin = GameEventNew(GAME_EVENT_GAME_BEGIN);
473 begin.u.GameBegin.MissionTime = gMission.time;
474 GameEventsEnqueue(&gGameEvents, begin);
475 }
476
477 // Set mission complete and display exit if it is complete
478 MissionSetMessageIfComplete(rData->m);
479
480 // If we're not hosting a net game,
481 // don't update if the game has paused or has automap shown
482 // Important: don't consider paused if we are trying to quit
483 const bool paused =
484 rData->pausingDevice != INPUT_DEVICE_UNSET ||
485 rData->controllerUnplugged ||
486 rData->isMap;
487 if (!gCampaign.IsClient &&
488 !ConfigGetBool(&gConfig, "StartServer") &&
489 paused &&
490 !gEventHandlers.HasQuit)
491 {
492 return UPDATE_RESULT_DRAW;
493 }
494
495 // Update all the things in the game
496 const int ticksPerFrame = 1;
497
498 if (gPlayerDatas.size > 0)
499 {
500 LOSReset(&gMap.LOS);
501 for (int i = 0, idx = 0; i < (int)gPlayerDatas.size; i++, idx++)
502 {
503 const PlayerData *p = CArrayGet(&gPlayerDatas, i);
504 if (p->ActorUID == -1) continue;
505 TActor *player = ActorGetByUID(p->ActorUID);
506 if (player->dead > DEATH_MAX) continue;
507 // Calculate LOS for all players alive or dying
508 LOSCalcFrom(
509 &gMap,
510 Vec2iToTile(Vec2iNew(player->tileItem.x, player->tileItem.y)),
511 !gCampaign.IsClient);
512
513 if (player->dead) continue;
514
515 // Only handle inputs/commands for local players
516 if (!p->IsLocal)
517 {
518 idx--;
519 continue;
520 }
521 if (p->inputDevice == INPUT_DEVICE_AI)
522 {
523 rData->cmds[idx] = AICoopGetCmd(player, ticksPerFrame);
524 }
525 PlayerSpecialCommands(player, rData->cmds[idx]);
526 CommandActor(player, rData->cmds[idx], ticksPerFrame);
527 }
528 }
529
530 if (!gCampaign.IsClient)
531 {
532 rData->aiUpdateCounter -= ticksPerFrame;
533 if (rData->aiUpdateCounter <= 0)
534 {
535 const int enemies = AICommand(ticksPerFrame);
536 AIAddRandomEnemies(enemies, rData->m->missionData);
537 rData->aiUpdateCounter = 4;
538 }
539 else
540 {
541 AICommandLast(ticksPerFrame);
542 }
543 }
544
545 // If split screen never and players are too close to the
546 // edge of the screen, forcefully pull them towards the center
547 if (ConfigGetEnum(&gConfig, "Interface.Splitscreen") == SPLITSCREEN_NEVER &&
548 GetNumPlayers(PLAYER_ALIVE_OR_DYING, true, true) > 1 &&
549 !IsPVP(gCampaign.Entry.Mode))
550 {
551 const int w = gGraphicsDevice.cachedConfig.Res.x;
552 const int h = gGraphicsDevice.cachedConfig.Res.y;
553 const Vec2i screen = Vec2iAdd(
554 PlayersGetMidpoint(), Vec2iNew(-w / 2, -h / 2));
555 CA_FOREACH(const PlayerData, pd, gPlayerDatas)
556 if (!pd->IsLocal || !IsPlayerAlive(pd))
557 {
558 continue;
559 }
560 const TActor *p = ActorGetByUID(pd->ActorUID);
561 const int pad = CAMERA_SPLIT_PADDING;
562 Vec2i vel = Vec2iZero();
563 if (screen.x + pad > p->tileItem.x && p->tileItem.VelFull.x < 256)
564 {
565 vel.x = screen.x + pad - p->tileItem.x;
566 }
567 else if (screen.x + w - pad < p->tileItem.x &&
568 p->tileItem.VelFull.x > -256)
569 {
570 vel.x = screen.x + w - pad - p->tileItem.x;
571 }
572 if (screen.y + pad > p->tileItem.y && p->tileItem.VelFull.y < 256)
573 {
574 vel.y = screen.y + pad - p->tileItem.y;
575 }
576 else if (screen.y + h - pad < p->tileItem.y
577 && p->tileItem.VelFull.y > -256)
578 {
579 vel.y = screen.y + h - pad - p->tileItem.y;
580 }
581 if (!Vec2iIsZero(vel))
582 {
583 GameEvent ei = GameEventNew(GAME_EVENT_ACTOR_IMPULSE);
584 ei.u.ActorImpulse.UID = p->uid;
585 ei.u.ActorImpulse.Vel = Vec2i2Net(Vec2iScale(vel, 64));
586 ei.u.ActorImpulse.Pos = Vec2i2Net(Vec2iZero());
587 GameEventsEnqueue(&gGameEvents, ei);
588 LOG(LM_MAIN, LL_TRACE,
589 "playerUID(%d) pos(%d, %d) screen(%d, %d) impulse(%d, %d)",
590 p->uid, p->tileItem.x, p->tileItem.y, screen.x, screen.y,
591 ei.u.ActorImpulse.Vel.x, ei.u.ActorImpulse.Vel.y);
592 }
593 CA_FOREACH_END()
594 }
595
596 UpdateAllActors(ticksPerFrame);
597 UpdateObjects(ticksPerFrame);
598 UpdateMobileObjects(ticksPerFrame);
599 ParticlesUpdate(&gParticles, ticksPerFrame);
600
601 UpdateWatches(&rData->map->triggers, ticksPerFrame);
602
603 PowerupSpawnerUpdate(&rData->healthSpawner, ticksPerFrame);
604 CA_FOREACH(PowerupSpawner, a, rData->ammoSpawners)
605 PowerupSpawnerUpdate(a, ticksPerFrame);
606 CA_FOREACH_END()
607
608 if (!gCampaign.IsClient)
609 {
610 CheckMissionCompletion(rData->m);
611 }
612 else if (!NetClientIsConnected(&gNetClient))
613 {
614 // Check if disconnected from server; end mission
615 const NMissionEnd me = NMissionEnd_init_zero;
616 MissionDone(&gMission, me);
617 }
618
619 HandleGameEvents(
620 &gGameEvents, &rData->Camera,
621 &rData->healthSpawner, &rData->ammoSpawners);
622
623 rData->m->time += ticksPerFrame;
624
625 CameraUpdate(&rData->Camera, ticksPerFrame, 1000 / data->FPS);
626
627 return UPDATE_RESULT_DRAW;
628}
629static void NextLoop(RunGameData *rData, LoopRunner *l)
630{
631 // Find the next screen to switch to
632 const bool hasLocalPlayers = GetNumPlayers(PLAYER_ANY, false, true) > 0;
633 const int survivingPlayers =
634 GetNumPlayers(PLAYER_ALIVE, false, false);
635 const bool survivedAndCompletedObjectives =
636 survivingPlayers > 0 && MissionAllObjectivesComplete(&gMission);
637
638 // Switch to a score screen if there are local players and we haven't quit
639 const bool showScores = !gMission.IsQuit && hasLocalPlayers;
640 if (showScores)
641 {
642 switch (rData->co->Entry.Mode)
643 {
644 case GAME_MODE_DOGFIGHT:
645 LoopRunnerChange(l, ScreenDogfightScores());
646 break;
647 case GAME_MODE_DEATHMATCH:
648 LoopRunnerChange(l, ScreenDeathmatchFinalScores());
649 break;
650 default:
651 // In co-op (non-PVP) modes, at least one player must survive
652 LoopRunnerChange(l, ScreenMissionSummary(
653 rData->co, &gMission, survivedAndCompletedObjectives));
654 break;
655 }
656 }
657 else
658 {
659 LoopRunnerChange(l, HighScoresScreen(&gCampaign, &gGraphicsDevice));
660 }
661}
662static void CheckMissionCompletion(const struct MissionOptions *mo)
663{
664 // Check if we need to update explore objectives
665 CA_FOREACH(const Objective, o, mo->missionData->Objectives)
666 if (o->Type != OBJECTIVE_INVESTIGATE) continue;
667 const int update = MapGetExploredPercentage(&gMap) - o->done;
668 if (update > 0 && !gCampaign.IsClient)
669 {
670 GameEvent e = GameEventNew(GAME_EVENT_OBJECTIVE_UPDATE);
671 e.u.ObjectiveUpdate.ObjectiveId = _ca_index;
672 e.u.ObjectiveUpdate.Count = update;
673 GameEventsEnqueue(&gGameEvents, e);
674 }
675 CA_FOREACH_END()
676
677 const bool isMissionComplete =
678 GetNumPlayers(PLAYER_ALIVE_OR_DYING, false, false) > 0 && IsMissionComplete(mo);
679 if (mo->state == MISSION_STATE_PLAY && isMissionComplete)
680 {
681 GameEvent e = GameEventNew(GAME_EVENT_MISSION_PICKUP);
682 GameEventsEnqueue(&gGameEvents, e);
683 }
684 if (mo->state == MISSION_STATE_PICKUP && !isMissionComplete)
685 {
686 GameEvent e = GameEventNew(GAME_EVENT_MISSION_INCOMPLETE);
687 GameEventsEnqueue(&gGameEvents, e);
688 }
689 if (mo->state == MISSION_STATE_PICKUP &&
690 mo->pickupTime + PICKUP_LIMIT <= mo->time)
691 {
692 GameEvent e = GameEventNew(GAME_EVENT_MISSION_END);
693 GameEventsEnqueue(&gGameEvents, e);
694 }
695
696 // Check that all players have been destroyed
697 // If the server has no players at all, wait for a player to join
698 if (gPlayerDatas.size > 0)
699 {
700 // Note: there's a period of time where players are dying
701 // Wait until after this period before ending the game
702 bool allPlayersDestroyed = true;
703 CA_FOREACH(const PlayerData, p, gPlayerDatas)
704 if (p->ActorUID != -1)
705 {
706 allPlayersDestroyed = false;
707 break;
708 }
709 CA_FOREACH_END()
710 if (allPlayersDestroyed && AreAllPlayersDeadAndNoLives())
711 {
712 GameEvent e = GameEventNew(GAME_EVENT_MISSION_END);
713 e.u.MissionEnd.Delay = GAME_OVER_DELAY;
714 GameEventsEnqueue(&gGameEvents, e);
715 }
716 }
717}
