45void PrintHelp(void)
46{
47 printf("%s\n",
48 "Video Options:\n"
49 " --fullscreen Try and use a fullscreen video mode.\n"
50 " --scale=n Scale the window resolution up by a factor of n\n"
51 " Factors: 2, 3, 4\n"
52 " --screen=WxH Set virtual screen width to W x H\n"
53 );
54
55 printf("%s\n",
56 "Config:\n"
57 " --config=K,V Set arbitrary config option\n"
58 " Example: --config=Game.FriendlyFire=true\n"
59 " --config List all config options\n"
60 );
61
62 printf(
63 "Logging: logging is enabled per module and set at certain levels.\n"
64 "Log modules are: "
65 );
66 for (int i = 0; i < (int)LM_COUNT; i++)
67 {
68 printf("%s", LogModuleName((LogModule)i));
69 if (i < (int)LM_COUNT - 1) printf(", ");
70 else printf("\n");
71 }
72 printf("Log levels are: ");
73 for (int i = 0; i < (int)LL_COUNT; i++)
74 {
75 printf("%s", LogLevelName((LogLevel)i));
76 if (i < (int)LL_COUNT - 1) printf(", ");
77 else printf("\n");
78 }
79 printf(
80 " --log=M,L Enable logging for module M at level L.\n\n"
81 );
82 printf(
83 " --log=L Enable logging for all modules at level L.\n\n"
84 );
85 printf(
86 " --logfile=F Log to file by filename\n\n"
87 );
88
89 printf("%s\n",
90 "Other:\n"
91 " --connect=host (Experimental) connect to a game server\n"
92 );
93
94 printf("%s\n",
95 "The DEBUG environment variable can be set to show debug information.");
96
97 printf(
98 "The DEBUG_LEVEL environment variable can be set to between %d and %d.\n", D_NORMAL, D_MAX
99 );
100}

102void ProcessCommandLine(char *buf, const int argc, char *argv[])
103{
104 buf[0] = '\0';
105 for (int i = 0; i < argc; i++)
106 {
107 strcat(buf, " ");
108 // HACK: for OS X, blank out the -psn_XXXX argument so that it doesn't
109 // break arg parsing
110 #ifdef __APPLE__
111 if (strncmp(argv[i], "-psn", strlen("-psn")) == 0)
112 {
113 argv[i] = "";
114 }
115 #endif
116 strcat(buf, argv[i]);
117 }
118}

121bool ParseArgs(
122 const int argc, char *argv[],
123 ENetAddress *connectAddr, const char **loadCampaign)
124{
125 struct option longopts[] =
126 {
127 { "fullscreen", no_argument, NULL, 'f' },
128 { "scale", required_argument, NULL, 's' },
129 { "screen", required_argument, NULL, 'c' },
130 { "connect", required_argument, NULL, 'x' },
131 { "debug", required_argument, NULL, 'd' },
132 { "config", optional_argument, NULL, 'C' },
133 { "log", required_argument, NULL, 1000 },
134 { "logfile", required_argument, NULL, 1001 },
135 { "help", no_argument, NULL, 'h' },
136 { 0, 0, NULL, 0 }
137 };
138 int opt = 0;
139 int idx = 0;
140 while ((opt = getopt_long(argc, argv, "fs:c:x:d:C::\0:\0:h", longopts, &idx)) != -1)
141 {
142 switch (opt)
143 {
144 case 'f':
145 ConfigGet(&gConfig, "Graphics.Fullscreen")->u.Bool.Value = true;
146 break;
147 case 's':
148 ConfigSetInt(&gConfig, "Graphics.ScaleFactor", atoi(optarg));
149 break;
150 case 'c':
151 sscanf(optarg, "%dx%d",
152 &ConfigGet(&gConfig, "Graphics.ResolutionWidth")->u.Int.Value,
153 &ConfigGet(&gConfig, "Graphics.ResolutionHeight")->u.Int.Value);
154 LOG(LM_MAIN, LL_DEBUG, "Video mode %dx%d set...",
155 ConfigGetInt(&gConfig, "Graphics.ResolutionWidth"),
156 ConfigGetInt(&gConfig, "Graphics.ResolutionHeight"));
157 break;
158 case 'm':
159 {
160 ConfigGet(&gConfig, "Graphics.ShakeMultiplier")->u.Int.Value =
161 MAX(atoi(optarg), 0);
162 printf("Shake multiplier: %d\n",
163 ConfigGetInt(&gConfig, "Graphics.ShakeMultiplier"));
164 }
165 break;
166 case 'h':
167 PrintHelp();
168 return false;
169 case 'd':
170 // Set debug level
171 debug = true;
172 debug_level = CLAMP(atoi(optarg), D_NORMAL, D_MAX);
173 break;
174 case 1000:
175 {
176 char *comma = strchr(optarg, ',');
177 if (comma)
178 {
179 // Set logging level for a single module
180 // The module and level are comma separated
181 *comma = '\0';
182 const LogLevel ll = StrLogLevel(comma + 1);
183 LogModuleSetLevel(StrLogModule(optarg), ll);
184 printf("Logging %s at %s\n", optarg, LogLevelName(ll));
185 }
186 else
187 {
188 // Set logging level for all modules
189 const LogLevel ll = StrLogLevel(optarg);
190 for (int i = 0; i < (int)LM_COUNT; i++)
191 {
192 LogModuleSetLevel((LogModule)i, ll);
193 }
194 printf("Logging everything at %s\n", LogLevelName(ll));
195 }
196 }
197 break;
198 case 1001:
199 LogOpenFile(optarg);
200 break;
201 case 'x':
202 if (enet_address_set_host(connectAddr, optarg) != 0)
203 {
204 printf("Error: unknown host %s\n", optarg);
205 }
206 break;
207 case 'C':
208 if (optarg == NULL)
209 {
210 PrintConfig(&gConfig, 0);
211 return false;
212 }
213 else
214 {
215 char *comma = strchr(optarg, ',');
216 if (comma == NULL)
217 {
218 const Config *c = ConfigGet(&gConfig, optarg);
219 if (c == NULL)
220 {
221 PrintHelp();
222 }
223 else
224 {
225 PrintConfig(c, 0);
226 }
227 return false;
228 }
229 *comma = '\0';
230 const char *value = comma + 1;
231 if (!ConfigTrySetFromString(&gConfig, optarg, value))
232 {
233 PrintHelp();
234 return false;
235 }
236 }
237 default:
238 PrintHelp();
239 // Ignore unknown arguments
240 break;
241 }
242 }
243 if (optind < argc)
244 {
245 // non-option ARGV-elements
246 for (; optind < argc; optind++)
247 {
248 // Load campaign
249 *loadCampaign = argv[optind];
250 }
251 }
252
253 return true;
254}
255static void PrintConfig(const Config *c, const int indent)
256{
257 // Print this config
258 for (int i = 0; i < indent; i++)
259 {
260 printf("-");
261 }
262 printf("%s\n", c->Name);
263 if (c->Type == CONFIG_TYPE_GROUP)
264 {
265 CA_FOREACH(const Config, child, c->u.Group)
266 PrintConfig(child, indent + 1);
267 CA_FOREACH_END()
268 }
269}
