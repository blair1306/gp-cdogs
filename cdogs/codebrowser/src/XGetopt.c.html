71/* Note that the lookup routines use and modify the global variables - not
72 * nice, but better than passing everything as reference parameters.
73 */
74
75static int
76lookup_shortopt(int argc, char **argv, const char *opts)
77{
78 int c = *optarg++;
79 /* remember the Alamo - a dodgy character if ever there was one */
80 nextchar = optarg;
81 /* look at the options - like any you see? */
82 while (*opts)
83 {
84 if (c == *opts++)
85 {
86 /* OK, it's a fair (c)op, guv */
87 if (*opts == ':')
88 {
89 /* we can do arguments, but there will be nothing left of this arg */
90 nextchar = NULL;
91 if (*++opts == ':')
92 {
93 /* but it's optional, so either a 5 minute argument or nothing */
94 if (!*optarg)
95 optarg = NULL;
96 return c;
97 }
98 else
99 {
100 /* this argument's not optional, you get the full half hour */
101 if (*optarg)
102 {
103 /* We've got an argument on our hands! */
104 return c;
105 }
106 else
107 {
108 /* try the next door down */
109 if (++optind < argc)
110 {
111 /* OK, we've got it ("Oh no you haven't!") */
112 optarg = argv[optind];
113 return c;
114 }
115 else
116 {
117 /* Th-Th-Th-That's All, Folks! */
118 optopt = c;
119 if (opterr)
120 fprintf(stderr, "required parameter for -%c missing\n", c);
121 /* we've got an option, but it's dud */
122 return '?';
123 }
124 }
125 }
126 }
127 else
128 {
129 /* no arguments please, we're British */
130 nextchar = optarg;
131 optarg = NULL;
132 return c;
133 }
134 }
135 }
136 /* return to sender, address unknown */
137 optopt = c;
138 optarg = NULL;
139 return -1;
140}

142static int
143lookup_longopt(int argc, char **argv,
144 const struct option *longopts,
145 int *longindex)
146{
147 char *eqp;
148 bool ambig = false;
149 int match = -1;
150 int exact = -1;
151 int min;
152 int len;
153 int i;
154
155 /* if we have no long options, any we get are wrong */
156 if (!longopts)
157 {
158 if (opterr)
159 fprintf(stderr, "long options not supported\n");
160 return '?';
161 }
162 /* look for equality or fraternity */
163 eqp = strchr(optarg, '=');
164 min = (eqp ? eqp - optarg : (int)strlen(optarg));
165 for (i = 0; longopts[i].name; i++)
166 {
167 len = (int)strlen(longopts[i].name);
168 /* is it exact? */
169 if (len == min && strncmp(longopts[i].name, optarg, len) == 0)
170 {
171 /* an exact match - go no further, we can't get better than this */
172 exact = i;
173 break;
174 }
175 /* a match as far as the argument goes? */
176 if (strncmp(longopts[i].name, optarg, min) == 0)
177 {
178 /* if we've already found one of those, it's ambiguous */
179 if (match < 0)
180 match = i;
181 else
182 ambig = true;
183 }
184 }
185 /* we've looked at everything, now what? */
186 if (exact < 0)
187 {
188 /* no exact match, if it's ambiguous then fail othersise use what
189 * (if anything) we have */
190 if (ambig)
191 {
192 if (opterr)
193 fprintf(stderr, "option --%s is ambiguous\n", optarg);
194 return '?';
195 }
196 else
197 exact = match;
198 }
199 /* if we don't have anything, return -1 (/not/ fail, this could be a test */
200 if (exact < 0)
201 return -1;
202 switch (longopts[exact].has_arg)
203 {
204 case no_argument:
205 /* we have nothing to fight about... */
206 if (eqp)
207 {
208 /* ... except equality */
209 if (opterr)
210 fprintf(stderr, "option --%s does not take an argument\n", optarg);
211 return '?';
212 }
213 optarg = NULL;
214 break;
215 case required_argument:
216 /* you have to fight, so there! */
217 if (eqp)
218 optarg = eqp + 1;
219 else if (++optind < argc)
220 optarg = argv[optind];
221 else
222 {
223 if (opterr)
224 fprintf(stderr, "missing argument for option --%s\n", optarg);
225 return '?';
226 }
227 break;
228 case optional_argument:
229 /* you don't have to fight if you don't want to */
230 optarg = (eqp ? eqp + 1 : NULL);
231 break;
232 }
233 /* tell the caller what he's got */
234 if (longindex)
235 *longindex = exact;
236 if (longopts[exact].flag)
237 {
238 /* set a flag rather than returning the value */
239 *longopts[exact].flag = longopts[exact].val;
240 return 0;
241 }
242 return longopts[exact].val;
243}

245/*
246 * Note that this internal function drops the const on the arg list. That's
247 * because optarg etc. are defined without const.
248 */
249static int
250getnextopt(int argc, char **argv, const char *optstring,
251 const struct option *longopts, int *longindex,
252 bool longonly, enum action act)
253{
254 static int thisind = 0;
255 static int oldind = 0;
256 int c = 0;
257
258 /* if we haven't started, make sure we're not in Denmark */
259 if (!initialised)
260 {
261 initialised = true;
262 nextchar = NULL;
263 endargs = false;
264 thisind = 0;
265 oldind = 0;
266 optind = 0;
267 }
268
269 /* check for previous sequence of short options */
270 if (nextchar && *nextchar)
271 {
272 optarg = nextchar;
273 c = lookup_shortopt(argc, argv, optstring);
274 /* now shuffle the arguments if necessary */
275 if (act == E_PERMUTE && oldind != thisind && (!nextchar || !*nextchar))
276 {
277 int optlen = optind - thisind; /* the length of the option */
278 int optnum = thisind - oldind; /* number of args in between */
279 memmove(argv+oldind+optlen+1, argv+oldind, sizeof(*argv) * optnum);
280 optind = oldind + optlen;
281 }
282 return c;
283 }
284
285 /* check for end of argument list */
286 if (++optind >= argc || !argv[optind])
287 {
288 optarg = NULL;
289 return EOF; /* no arguments left */
290 }
291
292 /* get next argument from list */
293 optarg = argv[optind];
294
295 /* if we've had "--" then treat as normal argument */
296 if (endargs)
297 return (act == E_INPLACE ? '\1' : EOF);
298
299 /* if we're permuting the arguments, look for the next option */
300 if (act == E_PERMUTE)
301 {
302 int i;
303 /* save where we are, we'll need this later */
304 oldind = optind;
305 for (i = optind; i < argc; i++)
306 {
307 char *p = argv[i];
308 /* remember that "--" ends options? Ignore anything past it... */
309 if (strcmp(p, "--") == 0)
310 {
311 int optnum = i - oldind; /* number of args in between */
312 if (optnum > 0)
313 {
314 /* shuffle the arguments up over it */
315 memmove(argv+oldind+1, argv+oldind, sizeof(*argv) * optnum);
316 optind = ++oldind;
317 optarg = argv[optind];
318 }
319 break;
320 }
321 if (*p == '-' && strcmp(p, "-"))
322 {
323 /* we found a honest-to-goodness option! Set up shop here... */
324 optind = i;
325 optarg = p;
326 break;
327 }
328 }
329 }
330
331 /* test for "--" meaning "end of options" - everything after that is an
332 * argument not an option */
333 if (strcmp(optarg, "--") == 0)
334 {
335 endargs = true;
336 if (++optind < argc)
337 {
338 optarg = argv[optind];
339 return (act == E_INPLACE ? '\1' : EOF);
340 }
341 else
342 {
343 optarg = NULL;
344 return EOF;
345 }
346 }
347
348 /* an argument of "-" or not starting with '-'is not an option */
349 if (strcmp(optarg, "-") == 0 || *optarg != '-')
350 {
351 /* if we're being POSIXLY_CORRECT (ugh!) then the first non-option ends
352 * the options. I don't like it but that's what the spec. says... */
353 if (act == E_POSIX)
354 endargs = true;
355 return (act == E_INPLACE ? '\1' : EOF);
356 }
357
358 thisind = optind; /* remember this position, lookup changes it */
359
360 /* OK, it's some sort of option, junk the first '-' */
361 optarg++;
362
363 /* test the next character - another '-' means it's a long option */
364 if (longopts && *optarg == '-')
365 {
366 /* it's a long option, skip the next '-' */
367 optarg++;
368 c = lookup_longopt(argc, argv, longopts, longindex);
369 if (c < 0)
370 {
371 fprintf(stderr, "unknown option --%s\n", optarg);
372 c = '?';
373 }
374 }
375 else if (longopts && longonly)
376 {
377 int tmperr = opterr;
378 opterr = 0;
379 /* try a long option first, if that doesn't work try a short one */
380 c = lookup_longopt(argc, argv, longopts, longindex);
381 opterr = tmperr;
382 if (c < 0)
383 c = lookup_shortopt(argc, argv, optstring);
384 }
385 else
386 {
387 /* just try short options */
388 c = lookup_shortopt(argc, argv, optstring);
389 }
390
391 /* if not found give (optional) error and return '?' */
392 if (c < 0)
393 {
394 if (opterr)
395 fprintf(stderr, "unknown option -%c\n", optopt);
396 optarg = 0;
397 c = '?';
398 }
399
400 /* now shuffle the arguments if necessary - I told you we'd need to save
401 * that old index! (I wonder if there's a more elegant way to do this?) */
402 if (act == E_PERMUTE && oldind != thisind && (!nextchar || !*nextchar))
403 {
404 int optlen = optind - thisind; /* the length of the option */
405 int optnum = thisind - oldind; /* number of args in between */
406 memmove(argv+oldind+optlen+1, argv+oldind, sizeof(*argv) * optnum);
407 optind = oldind + optlen;
408 oldind = 0;
409 thisind = 0;
410 }
411 return c;
412}

415/***********************************************************************
416 * At last, the real routines we wanted to call!
417 **********************************************************************/

419/**
420 * Scan the command-line parameters for options in the form \b -x.
421 */
422int getopt(int argc, char * argv[], const char *optstring)
423{
424 return getnextopt(argc, (char**)argv, optstring, NULL, NULL,
425 false, E_PERMUTE);
426}
427
428/**
429 * Scan the command-line parameters for options, allowing both the short
430 * (single character) options and long (string) options.
431 */
432int getopt_long(int argc, char * argv[], const char *optstring,
433 const struct option *longopts, int *longindex)
434{
435 return getnextopt(argc, (char**)argv, optstring, longopts, longindex,
436 false, E_PERMUTE);
437}
438
439/**
440 * Scan the command-line parameters for long options only
441 */
442int getopt_long_only(int argc, char * argv[], const char *optstring,
443 const struct option *longopts, int *longindex)
444{
445 return getnextopt(argc, (char**)argv, optstring, longopts, longindex,
446 true, E_PERMUTE);
447}
